139  /**
140   * This tests two region merges
141   */
142  @Test
143  public void testMergeTwoRegions() throws Exception {
144    final TableName tableName = TableName.valueOf(this.name.getMethodName());
145    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
146
147    List<RegionInfo> tableRegions = createTable(tableName);
148
149    RegionInfo[] regionsToMerge = new RegionInfo[2];
150    regionsToMerge[0] = tableRegions.get(0);
151    regionsToMerge[1] = tableRegions.get(1);
152
153    // collect AM metrics before test
154    collectAssignmentManagerMetrics();
155
156    MergeTableRegionsProcedure proc =
157        new MergeTableRegionsProcedure(procExec.getEnvironment(), regionsToMerge, true);
158    long procId = procExec.submitProcedure(proc);
159    ProcedureTestingUtility.waitProcedure(procExec, procId);
160    ProcedureTestingUtility.assertProcNotFailed(procExec, procId);
161    assertRegionCount(tableName, initialRegionCount - 1);
162
163    assertEquals(mergeSubmittedCount + 1, mergeProcMetrics.getSubmittedCounter().getCount());
164    assertEquals(mergeFailedCount, mergeProcMetrics.getFailedCounter().getCount());
165    assertEquals(assignSubmittedCount + 1, assignProcMetrics.getSubmittedCounter().getCount());
166    assertEquals(assignFailedCount, assignProcMetrics.getFailedCounter().getCount());
167    assertEquals(unassignSubmittedCount + 2, unassignProcMetrics.getSubmittedCounter().getCount());
168    assertEquals(unassignFailedCount, unassignProcMetrics.getFailedCounter().getCount());
169
170    Pair<RegionInfo, RegionInfo> pair =
171      MetaTableAccessor.getRegionsFromMergeQualifier(UTIL.getConnection(),
172        proc.getMergedRegion().getRegionName());
173    assertTrue(pair.getFirst() != null && pair.getSecond() != null);
174
175    // Can I purge the merged regions from hbase:meta? Check that all went
176    // well by looking at the merged row up in hbase:meta. It should have no
177    // more mention of the merged regions; they are purged as last step in
178    // the merged regions cleanup.
179    UTIL.getHBaseCluster().getMaster().setCatalogJanitorEnabled(true);
180    UTIL.getHBaseCluster().getMaster().getCatalogJanitor().triggerNow();
181    while (pair != null && pair.getFirst() != null && pair.getSecond() != null) {
182      pair = MetaTableAccessor.getRegionsFromMergeQualifier(UTIL.getConnection(),
183          proc.getMergedRegion().getRegionName());
184    }
185  }

189   */
190  @Test
191  public void testMergeRegionsConcurrently() throws Exception {
192    final TableName tableName = TableName.valueOf("testMergeRegionsConcurrently");
193    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
194
195    List<RegionInfo> tableRegions = createTable(tableName);
196
197    RegionInfo[] regionsToMerge1 = new RegionInfo[2];
198    RegionInfo[] regionsToMerge2 = new RegionInfo[2];
199    regionsToMerge1[0] = tableRegions.get(0);
200    regionsToMerge1[1] = tableRegions.get(1);
201    regionsToMerge2[0] = tableRegions.get(2);
202    regionsToMerge2[1] = tableRegions.get(3);
203
204    // collect AM metrics before test
205    collectAssignmentManagerMetrics();
206
207    long procId1 = procExec.submitProcedure(new MergeTableRegionsProcedure(
208      procExec.getEnvironment(), regionsToMerge1, true));
209    long procId2 = procExec.submitProcedure(new MergeTableRegionsProcedure(
210      procExec.getEnvironment(), regionsToMerge2, true));
211    ProcedureTestingUtility.waitProcedure(procExec, procId1);
212    ProcedureTestingUtility.waitProcedure(procExec, procId2);
213    ProcedureTestingUtility.assertProcNotFailed(procExec, procId1);
214    ProcedureTestingUtility.assertProcNotFailed(procExec, procId2);
215    assertRegionCount(tableName, initialRegionCount - 2);
216
217    assertEquals(mergeSubmittedCount + 2, mergeProcMetrics.getSubmittedCounter().getCount());
218    assertEquals(mergeFailedCount, mergeProcMetrics.getFailedCounter().getCount());
219    assertEquals(assignSubmittedCount + 2, assignProcMetrics.getSubmittedCounter().getCount());
220    assertEquals(assignFailedCount, assignProcMetrics.getFailedCounter().getCount());
221    assertEquals(unassignSubmittedCount + 4, unassignProcMetrics.getSubmittedCounter().getCount());
222    assertEquals(unassignFailedCount, unassignProcMetrics.getFailedCounter().getCount());
223  }

249  @Test
250  public void testRollbackAndDoubleExecution() throws Exception {
251    final TableName tableName = TableName.valueOf("testRollbackAndDoubleExecution");
252    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
253
254    List<RegionInfo> tableRegions = createTable(tableName);
255
256    ProcedureTestingUtility.waitNoProcedureRunning(procExec);
257    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, true);
258
259    RegionInfo[] regionsToMerge = new RegionInfo[2];
260    regionsToMerge[0] = tableRegions.get(0);
261    regionsToMerge[1] = tableRegions.get(1);
262
263    long procId = procExec.submitProcedure(
264      new MergeTableRegionsProcedure(procExec.getEnvironment(), regionsToMerge, true));
265
266    // Failing before MERGE_TABLE_REGIONS_UPDATE_META we should trigger the rollback
267    // NOTE: the 8 (number of MERGE_TABLE_REGIONS_UPDATE_META step) is
268    // hardcoded, so you have to look at this test at least once when you add a new step.
269    int lastStep = 8;
270    MasterProcedureTestingUtility.testRollbackAndDoubleExecution(procExec, procId, lastStep, true);
271    assertEquals(initialRegionCount, UTIL.getAdmin().getRegions(tableName).size());
272    UTIL.waitUntilAllRegionsAssigned(tableName);
273    List<HRegion> regions = UTIL.getMiniHBaseCluster().getRegions(tableName);
274    assertEquals(initialRegionCount, regions.size());
275  }

306  private List<RegionInfo> createTable(final TableName tableName)
307      throws Exception {
308    HTableDescriptor desc = new HTableDescriptor(tableName);
309    desc.addFamily(new HColumnDescriptor(FAMILY));
310    byte[][] splitRows = new byte[initialRegionCount - 1][];
311    for (int i = 0; i < splitRows.length; ++i) {
312      splitRows[i] = Bytes.toBytes(String.format("%d", i));
313    }
314    admin.createTable(desc, splitRows);
315    return assertRegionCount(tableName, initialRegionCount);
316  }
317
318  public List<RegionInfo> assertRegionCount(final TableName tableName, final int nregions)
319      throws Exception {
320    UTIL.waitUntilNoRegionsInTransition();
321    List<RegionInfo> tableRegions = admin.getRegions(tableName);
322    assertEquals(nregions, tableRegions.size());
323    return tableRegions;
324  }
