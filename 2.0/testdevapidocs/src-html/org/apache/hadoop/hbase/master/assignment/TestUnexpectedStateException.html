052/**
053 * Tests for HBASE-18408 "AM consumes CPU and fills up the logs really fast when there is no RS to
054 * assign". If an {@link org.apache.hadoop.hbase.exceptions.UnexpectedStateException}, we'd spin on
055 * the ProcedureExecutor consuming CPU and filling logs. Test new back-off facility.
056 */
057@Category({MasterTests.class, MediumTests.class})
058public class TestUnexpectedStateException {
059  @ClassRule
060  public static final HBaseClassTestRule CLASS_RULE =
061      HBaseClassTestRule.forClass(TestUnexpectedStateException.class);
062  @Rule public final TestName name = new TestName();
063
064  private static final Logger LOG = LoggerFactory.getLogger(TestUnexpectedStateException.class);
065  private static final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
066  private static final byte [] FAMILY = Bytes.toBytes("family");
067  private TableName tableName;
068  private static final int REGIONS = 10;
069
070  @BeforeClass
071  public static void beforeClass() throws Exception {
072    TEST_UTIL.startMiniCluster();
073  }
074
075  @AfterClass
076  public static void afterClass() throws Exception {
077    TEST_UTIL.shutdownMiniCluster();
078  }
079
080  @Before
081  public void before() throws IOException {
082    this.tableName = TableName.valueOf(this.name.getMethodName());
083    TEST_UTIL.createMultiRegionTable(this.tableName, FAMILY, REGIONS);
084  }
085
086  private RegionInfo pickArbitraryRegion(Admin admin) throws IOException {
087    List<RegionInfo> regions = admin.getRegions(this.tableName);
088    return regions.get(3);
089  }
090
091  /**
092   * Manufacture a state that will throw UnexpectedStateException.
093   * Change an assigned region's 'state' to be OPENING. That'll mess up a subsequent unassign
094   * causing it to throw UnexpectedStateException. We can easily manufacture this infinite retry
095   * state in UnassignProcedure because it has no startTransition. AssignProcedure does where it
096   * squashes whatever the current region state is making it OFFLINE. That makes it harder to mess
097   * it up. Make do with UnassignProcedure for now.
098   */
099  @Test
100  public void testUnableToAssign() throws Exception {
101    try (Admin admin = TEST_UTIL.getAdmin()) {
102      // Pick a random region from this tests' table to play with. Get its RegionStateNode.
103      // Clone it because the original will be changed by the system. We need clone to fake out
104      // a state.
105      final RegionInfo region = pickArbitraryRegion(admin);
106      AssignmentManager am = TEST_UTIL.getHBaseCluster().getMaster().getAssignmentManager();
107      RegionStates.RegionStateNode rsn =  am.getRegionStates().getRegionStateNode(region);
108      // Now force region to be in OPENING state.
109      am.markRegionAsOpening(rsn);
110      // Now the 'region' is in an artificially bad state, try an unassign again.
111      // Run unassign in a thread because it is blocking.
112      Runnable unassign = () -> {
113        try {
114          admin.unassign(region.getRegionName(), true);
115        } catch (IOException ioe) {
116          fail("Failed assign");
117        }
118      };
119      Thread t = new Thread(unassign, "unassign");
120      t.start();
121      while(!t.isAlive()) {
122        Threads.sleep(100);
123      }
124      Threads.sleep(1000);
125      // Unassign should be running and failing. Look for incrementing timeout as evidence that
126      // Unassign is stuck and doing backoff.
127      // Now fix the condition we were waiting on so the unassign can complete.
128      JsonParser parser = new JsonParser();
129      long oldTimeout = 0;
130      int timeoutIncrements = 0;
131      while (true) {
132        long timeout = getUnassignTimeout(parser, admin.getProcedures());
133        if (timeout > oldTimeout) {
134          LOG.info("Timeout incremented, was {}, now is {}, increments={}",
135              timeout, oldTimeout, timeoutIncrements);
136          oldTimeout = timeout;
137          timeoutIncrements++;
138          if (timeoutIncrements > 3) {
139            // If we incremented at least twice, break; the backoff is working.
140            break;
141          }
142        }
143        Thread.sleep(1000);
144      }
145      TEST_UTIL.getMiniHBaseCluster().stopMaster(0).join();
146      HMaster master = TEST_UTIL.getMiniHBaseCluster().startMaster().getMaster();
147      TEST_UTIL.waitFor(30000, () -> master.isInitialized());
148      am = master.getAssignmentManager();
149      rsn = am.getRegionStates().getRegionStateNode(region);
150      am.markRegionAsOpened(rsn);
151      t.join();
152    }
153  }
154
155  /**
156   * @param proceduresAsJSON This is String returned by admin.getProcedures call... an array of
157   *                         Procedures as JSON.
158   * @return The Procedure timeout value parsed from the Unassign Procedure.
159   * @Exception Thrown if we do not find UnassignProcedure or fail to parse timeout.
160   */
161  private long getUnassignTimeout(JsonParser parser, String proceduresAsJSON) throws Exception {
162    JsonArray array = parser.parse(proceduresAsJSON).getAsJsonArray();
163    Iterator<JsonElement> iterator = array.iterator();
164    while (iterator.hasNext()) {
165      JsonElement element = iterator.next();
166      JsonObject obj = element.getAsJsonObject();
167      String className = obj.get("className").getAsString();
168      String actualClassName = UnassignProcedure.class.getName();
169      if (className.equals(actualClassName)) {
170        return obj.get("timeout").getAsLong();
171      }
172    }
173    throw new Exception("Failed to find UnassignProcedure or timeout in " + proceduresAsJSON);
174  }
