082  private static byte[][] generateSplits(int numRows, int numRegions) {
083    byte[][] splitRows = new byte[numRegions-1][];
084    for (int i = 1; i < numRegions; i++) {
085      splitRows[i-1] = Bytes.toBytes(numRows * i / numRegions);
086    }
087    return splitRows;
088  }

181  private void assertEqualTables(int expectedRows, TableName sourceTableName,
182      TableName targetTableName, boolean ignoreTimestamps) throws Exception {
183    Table sourceTable = TEST_UTIL.getConnection().getTable(sourceTableName);
184    Table targetTable = TEST_UTIL.getConnection().getTable(targetTableName);
185
186    ResultScanner sourceScanner = sourceTable.getScanner(new Scan());
187    ResultScanner targetScanner = targetTable.getScanner(new Scan());
188
189    for (int i = 0; i < expectedRows; i++) {
190      Result sourceRow = sourceScanner.next();
191      Result targetRow = targetScanner.next();
192
193      LOG.debug("SOURCE row: " + (sourceRow == null ? "null" : Bytes.toInt(sourceRow.getRow()))
194          + " cells:" + sourceRow);
195      LOG.debug("TARGET row: " + (targetRow == null ? "null" : Bytes.toInt(targetRow.getRow()))
196          + " cells:" + targetRow);
197
198      if (sourceRow == null) {
199        Assert.fail("Expected " + expectedRows
200            + " source rows but only found " + i);
201      }
202      if (targetRow == null) {
203        Assert.fail("Expected " + expectedRows
204            + " target rows but only found " + i);
205      }
206      Cell[] sourceCells = sourceRow.rawCells();
207      Cell[] targetCells = targetRow.rawCells();
208      if (sourceCells.length != targetCells.length) {
209        LOG.debug("Source cells: " + Arrays.toString(sourceCells));
210        LOG.debug("Target cells: " + Arrays.toString(targetCells));
211        Assert.fail("Row " + Bytes.toInt(sourceRow.getRow())
212            + " has " + sourceCells.length
213            + " cells in source table but " + targetCells.length
214            + " cells in target table");
215      }
216      for (int j = 0; j < sourceCells.length; j++) {
217        Cell sourceCell = sourceCells[j];
218        Cell targetCell = targetCells[j];
219        try {
220          if (!CellUtil.matchingRows(sourceCell, targetCell)) {
221            Assert.fail("Rows don't match");
222          }
223          if (!CellUtil.matchingFamily(sourceCell, targetCell)) {
224            Assert.fail("Families don't match");
225          }
226          if (!CellUtil.matchingQualifier(sourceCell, targetCell)) {
227            Assert.fail("Qualifiers don't match");
228          }
229          if (!ignoreTimestamps && !CellUtil.matchingTimestamp(sourceCell, targetCell)) {
230            Assert.fail("Timestamps don't match");
231          }
232          if (!CellUtil.matchingValue(sourceCell, targetCell)) {
233            Assert.fail("Values don't match");
234          }
235        } catch (Throwable t) {
236          LOG.debug("Source cell: " + sourceCell + " target cell: " + targetCell);
237          Throwables.propagate(t);
238        }
239      }
240    }
241    Result sourceRow = sourceScanner.next();
242    if (sourceRow != null) {
243      Assert.fail("Source table has more than " + expectedRows
244          + " rows.  Next row: " + Bytes.toInt(sourceRow.getRow()));
245    }
246    Result targetRow = targetScanner.next();
247    if (targetRow != null) {
248      Assert.fail("Target table has more than " + expectedRows
249          + " rows.  Next row: " + Bytes.toInt(targetRow.getRow()));
250    }
251    sourceScanner.close();
252    targetScanner.close();
253    sourceTable.close();
254    targetTable.close();
255  }

257  private void assertTargetDoDeletesFalse(int expectedRows, TableName sourceTableName,
258      TableName targetTableName) throws Exception {
259    Table sourceTable = TEST_UTIL.getConnection().getTable(sourceTableName);
260    Table targetTable = TEST_UTIL.getConnection().getTable(targetTableName);
261
262    ResultScanner sourceScanner = sourceTable.getScanner(new Scan());
263    ResultScanner targetScanner = targetTable.getScanner(new Scan());
264    Result targetRow = targetScanner.next();
265    Result sourceRow = sourceScanner.next();
266    int rowsCount = 0;
267    while (targetRow != null) {
268      rowsCount++;
269      //only compares values for existing rows, skipping rows existing on
270      //target only that were not deleted given --doDeletes=false
271      if (Bytes.toInt(sourceRow.getRow()) != Bytes.toInt(targetRow.getRow())) {
272        targetRow = targetScanner.next();
273        continue;
274      }
275
276      LOG.debug("SOURCE row: " + (sourceRow == null ? "null"
277          : Bytes.toInt(sourceRow.getRow()))
278          + " cells:" + sourceRow);
279      LOG.debug("TARGET row: " + (targetRow == null ? "null"
280          : Bytes.toInt(targetRow.getRow()))
281          + " cells:" + targetRow);
282
283      Cell[] sourceCells = sourceRow.rawCells();
284      Cell[] targetCells = targetRow.rawCells();
285      int targetRowKey = Bytes.toInt(targetRow.getRow());
286      if (targetRowKey >= 70 && targetRowKey < 80) {
287        if (sourceCells.length == targetCells.length) {
288          LOG.debug("Source cells: " + Arrays.toString(sourceCells));
289          LOG.debug("Target cells: " + Arrays.toString(targetCells));
290          Assert.fail("Row " + targetRowKey + " should have more cells in "
291              + "target than in source");
292        }
293
294      } else {
295        if (sourceCells.length != targetCells.length) {
296          LOG.debug("Source cells: " + Arrays.toString(sourceCells));
297          LOG.debug("Target cells: " + Arrays.toString(targetCells));
298          Assert.fail("Row " + Bytes.toInt(sourceRow.getRow())
299              + " has " + sourceCells.length
300              + " cells in source table but " + targetCells.length
301              + " cells in target table");
302        }
303      }
304      for (int j = 0; j < sourceCells.length; j++) {
305        Cell sourceCell = sourceCells[j];
306        Cell targetCell = targetCells[j];
307        try {
308          if (!CellUtil.matchingRows(sourceCell, targetCell)) {
309            Assert.fail("Rows don't match");
310          }
311          if (!CellUtil.matchingFamily(sourceCell, targetCell)) {
312            Assert.fail("Families don't match");
313          }
314          if (!CellUtil.matchingQualifier(sourceCell, targetCell)) {
315            Assert.fail("Qualifiers don't match");
316          }
317          if (targetRowKey < 80 && targetRowKey >= 90){
318            if (!CellUtil.matchingTimestamp(sourceCell, targetCell)) {
319              Assert.fail("Timestamps don't match");
320            }
321          }
322          if (!CellUtil.matchingValue(sourceCell, targetCell)) {
323            Assert.fail("Values don't match");
324          }
325        } catch (Throwable t) {
326          LOG.debug("Source cell: " + sourceCell + " target cell: "
327              + targetCell);
328          Throwables.propagate(t);
329        }
330      }
331      targetRow = targetScanner.next();
332      sourceRow = sourceScanner.next();
333    }
334    assertEquals("Target expected rows does not match.",expectedRows,
335        rowsCount);
336    sourceScanner.close();
337    targetScanner.close();
338    sourceTable.close();
339    targetTable.close();
340  }

342  private void assertTargetDoPutsFalse(int expectedRows, TableName sourceTableName,
343      TableName targetTableName) throws Exception {
344    Table sourceTable = TEST_UTIL.getConnection().getTable(sourceTableName);
345    Table targetTable = TEST_UTIL.getConnection().getTable(targetTableName);
346
347    ResultScanner sourceScanner = sourceTable.getScanner(new Scan());
348    ResultScanner targetScanner = targetTable.getScanner(new Scan());
349    Result targetRow = targetScanner.next();
350    Result sourceRow = sourceScanner.next();
351    int rowsCount = 0;
352
353    while (targetRow!=null) {
354      //only compares values for existing rows, skipping rows existing on
355      //source only that were not added to target given --doPuts=false
356      if (Bytes.toInt(sourceRow.getRow()) != Bytes.toInt(targetRow.getRow())) {
357        sourceRow = sourceScanner.next();
358        continue;
359      }
360
361      LOG.debug("SOURCE row: " + (sourceRow == null ?
362          "null" :
363          Bytes.toInt(sourceRow.getRow()))
364          + " cells:" + sourceRow);
365      LOG.debug("TARGET row: " + (targetRow == null ?
366          "null" :
367          Bytes.toInt(targetRow.getRow()))
368          + " cells:" + targetRow);
369
370      LOG.debug("rowsCount: " + rowsCount);
371
372      Cell[] sourceCells = sourceRow.rawCells();
373      Cell[] targetCells = targetRow.rawCells();
374      int targetRowKey = Bytes.toInt(targetRow.getRow());
375      if (targetRowKey >= 40 && targetRowKey < 60) {
376        LOG.debug("Source cells: " + Arrays.toString(sourceCells));
377        LOG.debug("Target cells: " + Arrays.toString(targetCells));
378        Assert.fail("There shouldn't exist any rows between 40 and 60, since "
379            + "Puts are disabled and Deletes are enabled.");
380      } else if (targetRowKey >= 60 && targetRowKey < 70) {
381        if (sourceCells.length == targetCells.length) {
382          LOG.debug("Source cells: " + Arrays.toString(sourceCells));
383          LOG.debug("Target cells: " + Arrays.toString(targetCells));
384          Assert.fail("Row " + Bytes.toInt(sourceRow.getRow())
385              + " shouldn't have same number of cells.");
386        }
387      } else if (targetRowKey >= 80 && targetRowKey < 90) {
388        LOG.debug("Source cells: " + Arrays.toString(sourceCells));
389        LOG.debug("Target cells: " + Arrays.toString(targetCells));
390        Assert.fail("There should be no rows between 80 and 90 on target, as "
391            + "these had different timestamps and should had been deleted.");
392      } else if (targetRowKey >= 90 && targetRowKey < 100) {
393        for (int j = 0; j < sourceCells.length; j++) {
394          Cell sourceCell = sourceCells[j];
395          Cell targetCell = targetCells[j];
396          if (CellUtil.matchingValue(sourceCell, targetCell)) {
397            Assert.fail("Cells values should not match for rows between "
398                + "90 and 100. Target row id: " + (Bytes.toInt(targetRow
399                .getRow())));
400          }
401        }
402      } else {
403        for (int j = 0; j < sourceCells.length; j++) {
404          Cell sourceCell = sourceCells[j];
405          Cell targetCell = targetCells[j];
406          try {
407            if (!CellUtil.matchingRows(sourceCell, targetCell)) {
408              Assert.fail("Rows don't match");
409            }
410            if (!CellUtil.matchingFamily(sourceCell, targetCell)) {
411              Assert.fail("Families don't match");
412            }
413            if (!CellUtil.matchingQualifier(sourceCell, targetCell)) {
414              Assert.fail("Qualifiers don't match");
415            }
416            if (!CellUtil.matchingTimestamp(sourceCell, targetCell)) {
417              Assert.fail("Timestamps don't match");
418            }
419            if (!CellUtil.matchingValue(sourceCell, targetCell)) {
420              Assert.fail("Values don't match");
421            }
422          } catch (Throwable t) {
423            LOG.debug(
424                "Source cell: " + sourceCell + " target cell: " + targetCell);
425            Throwables.propagate(t);
426          }
427        }
428      }
429      rowsCount++;
430      targetRow = targetScanner.next();
431      sourceRow = sourceScanner.next();
432    }
433    assertEquals("Target expected rows does not match.",expectedRows,
434        rowsCount);
435    sourceScanner.close();
436    targetScanner.close();
437    sourceTable.close();
438    targetTable.close();
439  }

446    args[options.length+1] = sourceTableName.getNameAsString();

455  private void hashSourceTable(TableName sourceTableName, Path testDir, String... options)
456      throws Exception {
457    int numHashFiles = 3;
458    long batchSize = 100;  // should be 2 batches per region
459    int scanBatch = 1;
460    HashTable hashTable = new HashTable(TEST_UTIL.getConfiguration());
461    String[] args = Arrays.copyOf(options, options.length+5);
462    args[options.length] = "--batchsize=" + batchSize;
463    args[options.length + 1] = "--numhashfiles=" + numHashFiles;
464    args[options.length + 2] = "--scanbatch=" + scanBatch;
465    args[options.length + 3] = sourceTableName.getNameAsString();
466    args[options.length + 4] = testDir.toString();
467    int code = hashTable.run(args);
468    assertEquals("hash table job failed", 0, code);
469
470    FileSystem fs = TEST_UTIL.getTestFileSystem();
471
472    HashTable.TableHash tableHash = HashTable.TableHash.read(fs.getConf(), testDir);
473    assertEquals(sourceTableName.getNameAsString(), tableHash.tableName);
474    assertEquals(batchSize, tableHash.batchSize);
475    assertEquals(numHashFiles, tableHash.numHashFiles);
476    assertEquals(numHashFiles - 1, tableHash.partitions.size());
477
478    LOG.info("Hash table completed");
479  }

481  private void writeTestData(TableName sourceTableName, TableName targetTableName,
482      long... timestamps) throws Exception {
483    final byte[] family = Bytes.toBytes("family");
484    final byte[] column1 = Bytes.toBytes("c1");
485    final byte[] column2 = Bytes.toBytes("c2");
486    final byte[] value1 = Bytes.toBytes("val1");
487    final byte[] value2 = Bytes.toBytes("val2");
488    final byte[] value3 = Bytes.toBytes("val3");
489
490    int numRows = 100;
491    int sourceRegions = 10;
492    int targetRegions = 6;
493    if (ArrayUtils.isEmpty(timestamps)) {
494      long current = EnvironmentEdgeManager.currentTime();
495      timestamps = new long[]{current,current};
496    }
497
498    Table sourceTable = TEST_UTIL.createTable(sourceTableName,
499        family, generateSplits(numRows, sourceRegions));
500
501    Table targetTable = TEST_UTIL.createTable(targetTableName,
502        family, generateSplits(numRows, targetRegions));
503
504    int rowIndex = 0;
505    // a bunch of identical rows
506    for (; rowIndex < 40; rowIndex++) {
507      Put sourcePut = new Put(Bytes.toBytes(rowIndex));
508      sourcePut.addColumn(family, column1, timestamps[0], value1);
509      sourcePut.addColumn(family, column2, timestamps[0], value2);
510      sourceTable.put(sourcePut);
511
512      Put targetPut = new Put(Bytes.toBytes(rowIndex));
513      targetPut.addColumn(family, column1, timestamps[1], value1);
514      targetPut.addColumn(family, column2, timestamps[1], value2);
515      targetTable.put(targetPut);
516    }
517    // some rows only in the source table
518    // ROWSWITHDIFFS: 10
519    // TARGETMISSINGROWS: 10
520    // TARGETMISSINGCELLS: 20
521    for (; rowIndex < 50; rowIndex++) {
522      Put put = new Put(Bytes.toBytes(rowIndex));
523      put.addColumn(family, column1, timestamps[0], value1);
524      put.addColumn(family, column2, timestamps[0], value2);
525      sourceTable.put(put);
526    }
527    // some rows only in the target table
528    // ROWSWITHDIFFS: 10
529    // SOURCEMISSINGROWS: 10
530    // SOURCEMISSINGCELLS: 20
531    for (; rowIndex < 60; rowIndex++) {
532      Put put = new Put(Bytes.toBytes(rowIndex));
533      put.addColumn(family, column1, timestamps[1], value1);
534      put.addColumn(family, column2, timestamps[1], value2);
535      targetTable.put(put);
536    }
537    // some rows with 1 missing cell in target table
538    // ROWSWITHDIFFS: 10
539    // TARGETMISSINGCELLS: 10
540    for (; rowIndex < 70; rowIndex++) {
541      Put sourcePut = new Put(Bytes.toBytes(rowIndex));
542      sourcePut.addColumn(family, column1, timestamps[0], value1);
543      sourcePut.addColumn(family, column2, timestamps[0], value2);
544      sourceTable.put(sourcePut);
545
546      Put targetPut = new Put(Bytes.toBytes(rowIndex));
547      targetPut.addColumn(family, column1, timestamps[1], value1);
548      targetTable.put(targetPut);
549    }
550    // some rows with 1 missing cell in source table
551    // ROWSWITHDIFFS: 10
552    // SOURCEMISSINGCELLS: 10
553    for (; rowIndex < 80; rowIndex++) {
554      Put sourcePut = new Put(Bytes.toBytes(rowIndex));
555      sourcePut.addColumn(family, column1, timestamps[0], value1);
556      sourceTable.put(sourcePut);
557
558      Put targetPut = new Put(Bytes.toBytes(rowIndex));
559      targetPut.addColumn(family, column1, timestamps[1], value1);
560      targetPut.addColumn(family, column2, timestamps[1], value2);
561      targetTable.put(targetPut);
562    }
563    // some rows differing only in timestamp
564    // ROWSWITHDIFFS: 10
565    // SOURCEMISSINGCELLS: 20
566    // TARGETMISSINGCELLS: 20
567    for (; rowIndex < 90; rowIndex++) {
568      Put sourcePut = new Put(Bytes.toBytes(rowIndex));
569      sourcePut.addColumn(family, column1, timestamps[0], column1);
570      sourcePut.addColumn(family, column2, timestamps[0], value2);
571      sourceTable.put(sourcePut);
572
573      Put targetPut = new Put(Bytes.toBytes(rowIndex));
574      targetPut.addColumn(family, column1, timestamps[1]+1, column1);
575      targetPut.addColumn(family, column2, timestamps[1]-1, value2);
576      targetTable.put(targetPut);
577    }
578    // some rows with different values
579    // ROWSWITHDIFFS: 10
580    // DIFFERENTCELLVALUES: 20
581    for (; rowIndex < numRows; rowIndex++) {
582      Put sourcePut = new Put(Bytes.toBytes(rowIndex));
583      sourcePut.addColumn(family, column1, timestamps[0], value1);
584      sourcePut.addColumn(family, column2, timestamps[0], value2);
585      sourceTable.put(sourcePut);
586
587      Put targetPut = new Put(Bytes.toBytes(rowIndex));
588      targetPut.addColumn(family, column1, timestamps[1], value3);
589      targetPut.addColumn(family, column2, timestamps[1], value3);
590      targetTable.put(targetPut);
591    }
592
593    sourceTable.close();
594    targetTable.close();
595  }
