039/**
040 * Test our recoverLease loop against mocked up filesystem.
041 */
042@Category({ MiscTests.class, MediumTests.class })
043public class TestFSHDFSUtils {
044
045  @ClassRule
046  public static final HBaseClassTestRule CLASS_RULE =
047      HBaseClassTestRule.forClass(TestFSHDFSUtils.class);
048
049  private static final Logger LOG = LoggerFactory.getLogger(TestFSHDFSUtils.class);
050  private static final HBaseTestingUtility HTU = new HBaseTestingUtility();
051  static {
052    Configuration conf = HTU.getConfiguration();
053    conf.setInt("hbase.lease.recovery.first.pause", 10);
054    conf.setInt("hbase.lease.recovery.pause", 10);
055  }
056
057  private static Path FILE = new Path(HTU.getDataTestDir(), "file.txt");
058
059  /**
060   * Test recover lease eventually succeeding.
061   */
062  @Test
063  public void testRecoverLease() throws IOException {
064    long startTime = EnvironmentEdgeManager.currentTime();
065    HTU.getConfiguration().setInt("hbase.lease.recovery.dfs.timeout", 1000);
066    CancelableProgressable reporter = Mockito.mock(CancelableProgressable.class);
067    Mockito.when(reporter.progress()).thenReturn(true);
068    DistributedFileSystem dfs = Mockito.mock(DistributedFileSystem.class);
069    // Fail four times and pass on the fifth.
070    Mockito.when(dfs.recoverLease(FILE)).
071      thenReturn(false).thenReturn(false).thenReturn(false).thenReturn(false).thenReturn(true);
072    FSUtils.recoverFileLease(dfs, FILE, HTU.getConfiguration(), reporter);
073    Mockito.verify(dfs, Mockito.times(5)).recoverLease(FILE);
074    // Make sure we waited at least hbase.lease.recovery.dfs.timeout * 3 (the first two
075    // invocations will happen pretty fast... the we fall into the longer wait loop).
076    assertTrue((EnvironmentEdgeManager.currentTime() - startTime) >
077      (3 * HTU.getConfiguration().getInt("hbase.lease.recovery.dfs.timeout", 61000)));
078  }
079
080  /**
081   * Test that isFileClosed makes us recover lease faster.
082   */
083  @Test
084  public void testIsFileClosed() throws IOException {
085    // Make this time long so it is plain we broke out because of the isFileClosed invocation.
086    HTU.getConfiguration().setInt("hbase.lease.recovery.dfs.timeout", 100000);
087    CancelableProgressable reporter = Mockito.mock(CancelableProgressable.class);
088    Mockito.when(reporter.progress()).thenReturn(true);
089    IsFileClosedDistributedFileSystem dfs = Mockito.mock(IsFileClosedDistributedFileSystem.class);
090    // Now make it so we fail the first two times -- the two fast invocations, then we fall into
091    // the long loop during which we will call isFileClosed.... the next invocation should
092    // therefore return true if we are to break the loop.
093    Mockito.when(dfs.recoverLease(FILE)).
094      thenReturn(false).thenReturn(false).thenReturn(true);
095    Mockito.when(dfs.isFileClosed(FILE)).thenReturn(true);
096    FSUtils.recoverFileLease(dfs, FILE, HTU.getConfiguration(), reporter);
097    Mockito.verify(dfs, Mockito.times(2)).recoverLease(FILE);
098    Mockito.verify(dfs, Mockito.times(1)).isFileClosed(FILE);
099  }

101  private void testIsSameHdfs(int nnport) throws IOException {
102    Configuration conf = HBaseConfiguration.create();
103    Path srcPath = new Path("hdfs://localhost:" + nnport + "/");
104    Path desPath = new Path("hdfs://127.0.0.1/");
105    FileSystem srcFs = srcPath.getFileSystem(conf);
106    FileSystem desFs = desPath.getFileSystem(conf);
107
108    assertTrue(FSUtils.isSameHdfs(conf, srcFs, desFs));
109
110    desPath = new Path("hdfs://127.0.0.1:8070/");
111    desFs = desPath.getFileSystem(conf);
112    assertTrue(!FSUtils.isSameHdfs(conf, srcFs, desFs));
113
114    desPath = new Path("hdfs://127.0.1.1:" + nnport + "/");
115    desFs = desPath.getFileSystem(conf);
116    assertTrue(!FSUtils.isSameHdfs(conf, srcFs, desFs));
117
118    conf.set("fs.defaultFS", "hdfs://haosong-hadoop");
119    conf.set("dfs.nameservices", "haosong-hadoop");
120    conf.set("dfs.ha.namenodes.haosong-hadoop", "nn1,nn2");
121    conf.set("dfs.client.failover.proxy.provider.haosong-hadoop",
122        "org.apache.hadoop.hdfs.server.namenode.ha.ConfiguredFailoverProxyProvider");
123
124    conf.set("dfs.namenode.rpc-address.haosong-hadoop.nn1", "127.0.0.1:"+ nnport);
125    conf.set("dfs.namenode.rpc-address.haosong-hadoop.nn2", "127.10.2.1:8000");
126    desPath = new Path("/");
127    desFs = desPath.getFileSystem(conf);
128    assertTrue(FSUtils.isSameHdfs(conf, srcFs, desFs));
129
130    conf.set("dfs.namenode.rpc-address.haosong-hadoop.nn1", "127.10.2.1:"+nnport);
131    conf.set("dfs.namenode.rpc-address.haosong-hadoop.nn2", "127.0.0.1:8000");
132    desPath = new Path("/");
133    desFs = desPath.getFileSystem(conf);
134    assertTrue(!FSUtils.isSameHdfs(conf, srcFs, desFs));
135  }
136
137  @Test
138  public void testIsSameHdfs() throws IOException {
139    String hadoopVersion = org.apache.hadoop.util.VersionInfo.getVersion();
140    LOG.info("hadoop version is: "  + hadoopVersion);
141    boolean isHadoop3_0_0 = hadoopVersion.startsWith("3.0.0");
142    if (isHadoop3_0_0) {
143      // Hadoop 3.0.0 alpha1+ ~ 3.0.0 GA changed default nn port to 9820.
144      // See HDFS-9427
145      testIsSameHdfs(9820);
146    } else {
147      // pre hadoop 3.0.0 defaults to port 8020
148      // Hadoop 3.0.1 changed it back to port 8020. See HDFS-12990
149      testIsSameHdfs(8020);
150    }
151  }
