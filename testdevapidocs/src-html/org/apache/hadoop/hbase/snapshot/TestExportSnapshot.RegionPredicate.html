171  @Test
172  public void testExportFileSystemStateWithMergeRegion() throws Exception {
173    // disable compaction
174    admin.compactionSwitch(false,
175      admin.getRegionServers().stream().map(a -> a.getServerName()).collect(Collectors.toList()));
176    // create Table
177    TableName tableName0 = TableName.valueOf("testtb-" + testName.getMethodName() + "-1");
178    String snapshotName0 = "snaptb0-" + testName.getMethodName() + "-1";
179    admin.createTable(
180      TableDescriptorBuilder.newBuilder(tableName0)
181        .setColumnFamilies(
182          Lists.newArrayList(ColumnFamilyDescriptorBuilder.newBuilder(FAMILY).build()))
183        .build(),
184      new byte[][] { Bytes.toBytes("2") });
185    // put some data
186    try (Table table = admin.getConnection().getTable(tableName0)) {
187      table.put(new Put(Bytes.toBytes("1")).addColumn(FAMILY, null, Bytes.toBytes("1")));
188      table.put(new Put(Bytes.toBytes("2")).addColumn(FAMILY, null, Bytes.toBytes("2")));
189    }
190    List<RegionInfo> regions = admin.getRegions(tableName0);
191    assertEquals(2, regions.size());
192    tableNumFiles = regions.size();
193    // merge region
194    admin.mergeRegionsAsync(new byte[][] { regions.get(0).getEncodedNameAsBytes(),
195      regions.get(1).getEncodedNameAsBytes() }, true).get();
196    // take a snapshot
197    admin.snapshot(snapshotName0, tableName0);
198    // export snapshot and verify
199    testExportFileSystemState(tableName0, snapshotName0, snapshotName0, tableNumFiles);
200    // delete table
201    TEST_UTIL.deleteTable(tableName0);
202  }

246  /**
247   * Creates destination directory, runs ExportSnapshot() tool, and runs some verifications.
248   */
249  protected static void testExportFileSystemState(final Configuration conf,
250    final TableName tableName, final String snapshotName, final String targetName,
251    final int filesExpected, final Path srcDir, Path rawTgtDir, final boolean overwrite,
252    final RegionPredicate bypassregionPredicate, boolean success) throws Exception {
253    FileSystem tgtFs = rawTgtDir.getFileSystem(conf);
254    FileSystem srcFs = srcDir.getFileSystem(conf);
255    Path tgtDir = rawTgtDir.makeQualified(tgtFs.getUri(), tgtFs.getWorkingDirectory());
256    LOG.info("tgtFsUri={}, tgtDir={}, rawTgtDir={}, srcFsUri={}, srcDir={}", tgtFs.getUri(), tgtDir,
257      rawTgtDir, srcFs.getUri(), srcDir);
258    List<String> opts = new ArrayList<>();
259    opts.add("--snapshot");
260    opts.add(snapshotName);
261    opts.add("--copy-to");
262    opts.add(tgtDir.toString());
263    if (!targetName.equals(snapshotName)) {
264      opts.add("--target");
265      opts.add(targetName);
266    }
267    if (overwrite) {
268      opts.add("--overwrite");
269    }
270
271    // Export Snapshot
272    int res = run(conf, new ExportSnapshot(), opts.toArray(new String[opts.size()]));
273    assertEquals("success " + success + ", res=" + res, success ? 0 : 1, res);
274    if (!success) {
275      final Path targetDir = new Path(HConstants.SNAPSHOT_DIR_NAME, targetName);
276      assertFalse(tgtDir.toString() + " " + targetDir.toString(),
277        tgtFs.exists(new Path(tgtDir, targetDir)));
278      return;
279    }
280    LOG.info("Exported snapshot");
281
282    // Verify File-System state
283    FileStatus[] rootFiles = tgtFs.listStatus(tgtDir);
284    assertEquals(filesExpected > 0 ? 2 : 1, rootFiles.length);
285    for (FileStatus fileStatus : rootFiles) {
286      String name = fileStatus.getPath().getName();
287      assertTrue(fileStatus.toString(), fileStatus.isDirectory());
288      assertTrue(name.toString(), name.equals(HConstants.SNAPSHOT_DIR_NAME)
289        || name.equals(HConstants.HFILE_ARCHIVE_DIRECTORY));
290    }
291    LOG.info("Verified filesystem state");
292
293    // Compare the snapshot metadata and verify the hfiles
294    final Path snapshotDir = new Path(HConstants.SNAPSHOT_DIR_NAME, snapshotName);
295    final Path targetDir = new Path(HConstants.SNAPSHOT_DIR_NAME, targetName);
296    verifySnapshotDir(srcFs, new Path(srcDir, snapshotDir), tgtFs, new Path(tgtDir, targetDir));
297    Set<String> snapshotFiles =
298      verifySnapshot(conf, tgtFs, tgtDir, tableName, targetName, bypassregionPredicate);
299    assertEquals(filesExpected, snapshotFiles.size());
300  }

305  protected static void verifySnapshotDir(final FileSystem fs1, final Path root1,

310  /*
311   * Verify if the files exists
312   */
313  protected static Set<String> verifySnapshot(final Configuration conf, final FileSystem fs,
314    final Path rootDir, final TableName tableName, final String snapshotName,
315    final RegionPredicate bypassregionPredicate) throws IOException {
316    final Path exportedSnapshot =
317      new Path(rootDir, new Path(HConstants.SNAPSHOT_DIR_NAME, snapshotName));
318    final Set<String> snapshotFiles = new HashSet<>();
319    final Path exportedArchive = new Path(rootDir, HConstants.HFILE_ARCHIVE_DIRECTORY);
320    SnapshotReferenceUtil.visitReferencedFiles(conf, fs, exportedSnapshot,
321      new SnapshotReferenceUtil.SnapshotVisitor() {
322        @Override
323        public void storeFile(final RegionInfo regionInfo, final String family,
324          final SnapshotRegionManifest.StoreFile storeFile) throws IOException {
325          if (bypassregionPredicate != null && bypassregionPredicate.evaluate(regionInfo)) {
326            return;
327          }
328
329          if (!storeFile.hasReference() && !StoreFileInfo.isReference(storeFile.getName())) {
330            String hfile = storeFile.getName();
331            snapshotFiles.add(hfile);
332            verifyNonEmptyFile(new Path(exportedArchive,
333              new Path(CommonFSUtils.getTableDir(new Path("./"), tableName),
334                new Path(regionInfo.getEncodedName(), new Path(family, hfile)))));
335          } else {
336            Pair<String, String> referredToRegionAndFile =
337              StoreFileInfo.getReferredToRegionAndFile(storeFile.getName());
338            String region = referredToRegionAndFile.getFirst();
339            String hfile = referredToRegionAndFile.getSecond();
340            snapshotFiles.add(hfile);
341            verifyNonEmptyFile(new Path(exportedArchive,
342              new Path(CommonFSUtils.getTableDir(new Path("./"), tableName),
343                new Path(region, new Path(family, hfile)))));
344          }
345        }
346
347        private void verifyNonEmptyFile(final Path path) throws IOException {
348          assertTrue(path + " should exists", fs.exists(path));
349          assertTrue(path + " should not be empty", fs.getFileStatus(path).getLen() > 0);
350        }
351      });
352
353    // Verify Snapshot description
354    SnapshotDescription desc = SnapshotDescriptionUtils.readSnapshotInfo(fs, exportedSnapshot);
355    assertTrue(desc.getName().equals(snapshotName));
356    assertTrue(desc.getTable().equals(tableName.getNameAsString()));
357    return snapshotFiles;
358  }

360  private static Set<String> listFiles(final FileSystem fs, final Path root, final Path dir)
361    throws IOException {
362    Set<String> files = new HashSet<>();
363    LOG.debug("List files in {} in root {} at {}", fs, root, dir);
364    int rootPrefix = root.makeQualified(fs.getUri(), fs.getWorkingDirectory()).toString().length();
365    FileStatus[] list = CommonFSUtils.listStatus(fs, dir);
366    if (list != null) {
367      for (FileStatus fstat : list) {
368        LOG.debug(Objects.toString(fstat.getPath()));
369        if (fstat.isDirectory()) {
370          files.addAll(listFiles(fs, root, fstat.getPath()));
371        } else {
372          files.add(fstat.getPath().makeQualified(fs).toString().substring(rootPrefix));
373        }
374      }
375    }
376    return files;
377  }

387  static Path getLocalDestinationDir(HBaseTestingUtil htu) {
388    Path path = htu.getDataTestDir("local-export-" + EnvironmentEdgeManager.currentTime());
389    try {
390      FileSystem fs = FileSystem.getLocal(htu.getConfiguration());
391      LOG.info("Local export destination path: " + path);
392      return path.makeQualified(fs.getUri(), fs.getWorkingDirectory());
393    } catch (IOException ioe) {
394      throw new RuntimeException(ioe);
395    }
396  }
