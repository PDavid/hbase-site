250  @Test
251  public void testSnapshotFailsOnNonExistantTable() throws Exception {
252    // make sure we don't fail on listing snapshots
253    SnapshotTestingUtils.assertNoSnapshots(admin);
254    TableName tableName = TableName.valueOf("_not_a_table");
255
256    // make sure the table doesn't exist
257    boolean fail = false;
258    do {
259      try {
260        admin.getDescriptor(tableName);
261        fail = true;
262        LOG.error("Table:" + tableName + " already exists, checking a new name");
263        tableName = TableName.valueOf(tableName + "!");
264      } catch (TableNotFoundException e) {
265        fail = false;
266      }
267    } while (fail);
268
269    // snapshot the non-existant table
270    try {
271      admin.snapshot("fail", tableName, SnapshotType.FLUSH);
272      fail("Snapshot succeeded even though there is not table.");
273    } catch (SnapshotCreationException e) {
274      LOG.info("Correctly failed to snapshot a non-existant table:" + e.getMessage());
275    }

278  /**
279   * Helper method for testing async snapshot operations. Just waits for the given snapshot to
280   * complete on the server by repeatedly checking the master.
281   * @param master the master running the snapshot
282   * @param snapshot the snapshot to check
283   * @param timeoutNanos the timeout in nano between checks to see if the snapshot is done
284   */
285  private static void waitForSnapshotToComplete(HMaster master,
286      SnapshotProtos.SnapshotDescription snapshot, long timeoutNanos) throws Exception {
287    final IsSnapshotDoneRequest request =
288      IsSnapshotDoneRequest.newBuilder().setSnapshot(snapshot).build();
289    long start = System.nanoTime();
290    while (System.nanoTime() - start < timeoutNanos) {
291      try {
292        IsSnapshotDoneResponse done = master.getMasterRpcServices().isSnapshotDone(null, request);
293        if (done.getDone()) {
294          return;
295        }
296      } catch (ServiceException e) {
297        // ignore UnknownSnapshotException, this is possible as for AsyncAdmin, the method will
298        // return immediately after sending out the request, no matter whether the master has
299        // processed the request or not.
300        if (!(e.getCause() instanceof UnknownSnapshotException)) {
301          throw e;
302        }
303      }
304
305      Thread.sleep(200);
306    }
307    throw new TimeoutException("Timeout waiting for snapshot " + snapshot + " to complete");
308  }

310  @Test
311  public void testAsyncFlushSnapshot() throws Exception {
312    SnapshotProtos.SnapshotDescription snapshot = SnapshotProtos.SnapshotDescription.newBuilder()
313        .setName("asyncSnapshot").setTable(TABLE_NAME.getNameAsString())
314        .setType(SnapshotProtos.SnapshotDescription.Type.FLUSH).build();
315
316    // take the snapshot async
317    admin.snapshotAsync(
318      new SnapshotDescription("asyncSnapshot", TABLE_NAME, SnapshotType.FLUSH));
319
320    // constantly loop, looking for the snapshot to complete
321    HMaster master = UTIL.getMiniHBaseCluster().getMaster();
322    waitForSnapshotToComplete(master, snapshot, TimeUnit.MINUTES.toNanos(1));
323    LOG.info(" === Async Snapshot Completed ===");
324    UTIL.getHBaseCluster().getMaster().getMasterFileSystem().logFileSystemState(LOG);
325
326    // make sure we get the snapshot
327    SnapshotTestingUtils.assertOneSnapshotThatMatches(admin, snapshot);
328  }

330  @Test
331  public void testSnapshotStateAfterMerge() throws Exception {
332    int numRows = DEFAULT_NUM_ROWS;
333    // make sure we don't fail on listing snapshots
334    SnapshotTestingUtils.assertNoSnapshots(admin);
335    // load the table so we have some data
336    SnapshotTestingUtils.loadData(UTIL, TABLE_NAME, numRows, TEST_FAM);
337
338    // Take a snapshot
339    String snapshotBeforeMergeName = "snapshotBeforeMerge";
340    admin.snapshot(snapshotBeforeMergeName, TABLE_NAME, SnapshotType.FLUSH);
341
342    // Clone the table
343    TableName cloneBeforeMergeName = TableName.valueOf("cloneBeforeMerge");
344    admin.cloneSnapshot(snapshotBeforeMergeName, cloneBeforeMergeName);
345    SnapshotTestingUtils.waitForTableToBeOnline(UTIL, cloneBeforeMergeName);
346
347    // Merge two regions
348    List<RegionInfo> regions = admin.getRegions(TABLE_NAME);
349    Collections.sort(regions, new Comparator<RegionInfo>() {
350      @Override
351      public int compare(RegionInfo r1, RegionInfo r2) {
352        return Bytes.compareTo(r1.getStartKey(), r2.getStartKey());
353      }
354    });
355
356    int numRegions = admin.getRegions(TABLE_NAME).size();
357    int numRegionsAfterMerge = numRegions - 2;
358    admin.mergeRegionsAsync(regions.get(1).getEncodedNameAsBytes(),
359        regions.get(2).getEncodedNameAsBytes(), true);
360    admin.mergeRegionsAsync(regions.get(4).getEncodedNameAsBytes(),
361        regions.get(5).getEncodedNameAsBytes(), true);
362
363    // Verify that there's one region less
364    waitRegionsAfterMerge(numRegionsAfterMerge);
365    assertEquals(numRegionsAfterMerge, admin.getRegions(TABLE_NAME).size());
366
367    // Clone the table
368    TableName cloneAfterMergeName = TableName.valueOf("cloneAfterMerge");
369    admin.cloneSnapshot(snapshotBeforeMergeName, cloneAfterMergeName);
370    SnapshotTestingUtils.waitForTableToBeOnline(UTIL, cloneAfterMergeName);
371
372    verifyRowCount(UTIL, TABLE_NAME, numRows);
373    verifyRowCount(UTIL, cloneBeforeMergeName, numRows);
374    verifyRowCount(UTIL, cloneAfterMergeName, numRows);
375
376    // test that we can delete the snapshot
377    UTIL.deleteTable(cloneAfterMergeName);
378    UTIL.deleteTable(cloneBeforeMergeName);
379  }

442  private void waitRegionsAfterMerge(final long numRegionsAfterMerge)
443      throws IOException, InterruptedException {
444    // Verify that there's one region less
445    long startTime = System.currentTimeMillis();
446    while (admin.getRegions(TABLE_NAME).size() != numRegionsAfterMerge) {
447      // This may be flaky... if after 15sec the merge is not complete give up
448      // it will fail in the assertEquals(numRegionsAfterMerge).
449      if ((System.currentTimeMillis() - startTime) > 15000)
450        break;
451      Thread.sleep(100);
452    }
453    SnapshotTestingUtils.waitForTableToBeOnline(UTIL, TABLE_NAME);
454  }
