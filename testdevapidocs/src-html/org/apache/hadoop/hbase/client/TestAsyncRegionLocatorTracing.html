018package org.apache.hadoop.hbase.client;

020import static org.junit.Assert.assertEquals;
021
022import io.opentelemetry.api.trace.StatusCode;
023import io.opentelemetry.sdk.testing.junit4.OpenTelemetryRule;
024import io.opentelemetry.sdk.trace.data.SpanData;
025import java.io.IOException;
026import java.util.List;
027import java.util.concurrent.CompletableFuture;
028import java.util.concurrent.TimeUnit;
029import org.apache.hadoop.conf.Configuration;
030import org.apache.hadoop.hbase.HBaseClassTestRule;
031import org.apache.hadoop.hbase.HBaseConfiguration;
032import org.apache.hadoop.hbase.HConstants;

036import org.apache.hadoop.hbase.TableName;
037import org.apache.hadoop.hbase.Waiter;
038import org.apache.hadoop.hbase.security.UserProvider;
039import org.apache.hadoop.hbase.testclassification.ClientTests;
040import org.apache.hadoop.hbase.testclassification.MediumTests;
041import org.apache.hadoop.hbase.trace.HBaseSemanticAttributes;
042import org.apache.hadoop.hbase.util.EnvironmentEdgeManager;
043import org.junit.After;
044import org.junit.Before;
045import org.junit.ClassRule;
046import org.junit.Rule;
047import org.junit.Test;
048import org.junit.experimental.categories.Category;
049
050import org.apache.hbase.thirdparty.com.google.common.io.Closeables;
051
052@Category({ ClientTests.class, MediumTests.class })
053public class TestAsyncRegionLocatorTracing {
054
055  @ClassRule
056  public static final HBaseClassTestRule CLASS_RULE =
057    HBaseClassTestRule.forClass(TestAsyncRegionLocatorTracing.class);
058
059  private static Configuration CONF = HBaseConfiguration.create();
060
061  private AsyncConnectionImpl conn;
062
063  private RegionLocations locs;
064
065  @Rule
066  public OpenTelemetryRule traceRule = OpenTelemetryRule.create();
067
068  @Before
069  public void setUp() throws IOException {
070    RegionInfo metaRegionInfo = RegionInfoBuilder.newBuilder(TableName.META_TABLE_NAME).build();
071    locs = new RegionLocations(
072      new HRegionLocation(metaRegionInfo,
073        ServerName.valueOf("127.0.0.1", 12345, EnvironmentEdgeManager.currentTime())),
074      new HRegionLocation(RegionReplicaUtil.getRegionInfoForReplica(metaRegionInfo, 1),
075        ServerName.valueOf("127.0.0.2", 12345, EnvironmentEdgeManager.currentTime())),
076      new HRegionLocation(RegionReplicaUtil.getRegionInfoForReplica(metaRegionInfo, 2),
077        ServerName.valueOf("127.0.0.3", 12345, EnvironmentEdgeManager.currentTime())));
078    conn = new AsyncConnectionImpl(CONF, new DoNothingConnectionRegistry(CONF) {
079
080      @Override
081      public CompletableFuture<RegionLocations> getMetaRegionLocations() {
082        return CompletableFuture.completedFuture(locs);
083      }
084    }, "test", null, UserProvider.instantiate(CONF).getCurrent());
085  }
086
087  @After
088  public void tearDown() throws IOException {
089    Closeables.close(conn, true);
090  }
091
092  private SpanData waitSpan(String name) {
093    Waiter.waitFor(CONF, 1000,
094      () -> traceRule.getSpans().stream().map(SpanData::getName).anyMatch(s -> s.equals(name)));
095    return traceRule.getSpans().stream().filter(s -> s.getName().equals(name)).findFirst().get();
096  }
097
098  @Test
099  public void testClearCache() {
100    conn.getLocator().clearCache();
101    SpanData span = waitSpan("AsyncRegionLocator.clearCache");
102    assertEquals(StatusCode.OK, span.getStatus().getStatusCode());
103  }
104
105  @Test
106  public void testClearCacheServerName() {
107    ServerName sn = ServerName.valueOf("127.0.0.1", 12345,
108      EnvironmentEdgeManager.currentTime());
109    conn.getLocator().clearCache(sn);
110    SpanData span = waitSpan("AsyncRegionLocator.clearCache");
111    assertEquals(StatusCode.OK, span.getStatus().getStatusCode());
112    assertEquals(sn.toString(), span.getAttributes().get(HBaseSemanticAttributes.SERVER_NAME_KEY));
113  }
114
115  @Test
116  public void testClearCacheTableName() {
117    conn.getLocator().clearCache(TableName.META_TABLE_NAME);
118    SpanData span = waitSpan("AsyncRegionLocator.clearCache");
119    assertEquals(StatusCode.OK, span.getStatus().getStatusCode());
120    assertEquals(TableName.META_TABLE_NAME.getNamespaceAsString(),
121      span.getAttributes().get(HBaseSemanticAttributes.NAMESPACE_KEY));
122    assertEquals(TableName.META_TABLE_NAME.getNameAsString(),
123      span.getAttributes().get(HBaseSemanticAttributes.TABLE_KEY));
124  }
125
126  @Test
127  public void testGetRegionLocation() {
128    conn.getLocator().getRegionLocation(TableName.META_TABLE_NAME, HConstants.EMPTY_START_ROW,
129      RegionLocateType.CURRENT, TimeUnit.SECONDS.toNanos(1)).join();
130    SpanData span = waitSpan("AsyncRegionLocator.getRegionLocation");
131    assertEquals(StatusCode.OK, span.getStatus().getStatusCode());
132    assertEquals(TableName.META_TABLE_NAME.getNamespaceAsString(),
133      span.getAttributes().get(HBaseSemanticAttributes.NAMESPACE_KEY));
134    assertEquals(TableName.META_TABLE_NAME.getNameAsString(),
135      span.getAttributes().get(HBaseSemanticAttributes.TABLE_KEY));
136    List<String> regionNames = span.getAttributes().get(HBaseSemanticAttributes.REGION_NAMES_KEY);
137    assertEquals(1, regionNames.size());
138    assertEquals(locs.getDefaultRegionLocation().getRegion().getRegionNameAsString(),
139      regionNames.get(0));
140  }
141
142  @Test
143  public void testGetRegionLocations() {
144    conn.getLocator().getRegionLocations(TableName.META_TABLE_NAME, HConstants.EMPTY_START_ROW,
145      RegionLocateType.CURRENT, false, TimeUnit.SECONDS.toNanos(1)).join();
146    SpanData span = waitSpan("AsyncRegionLocator.getRegionLocations");
147    assertEquals(StatusCode.OK, span.getStatus().getStatusCode());
148    assertEquals(TableName.META_TABLE_NAME.getNamespaceAsString(),
149      span.getAttributes().get(HBaseSemanticAttributes.NAMESPACE_KEY));
150    assertEquals(TableName.META_TABLE_NAME.getNameAsString(),
151      span.getAttributes().get(HBaseSemanticAttributes.TABLE_KEY));
152    List<String> regionNames = span.getAttributes().get(HBaseSemanticAttributes.REGION_NAMES_KEY);
153    assertEquals(3, regionNames.size());
154    for (int i = 0; i < 3; i++) {
155      assertEquals(locs.getRegionLocation(i).getRegion().getRegionNameAsString(),
156        regionNames.get(i));
157    }
158  }
