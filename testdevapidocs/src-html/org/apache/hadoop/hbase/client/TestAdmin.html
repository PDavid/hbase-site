095      throws IOException {
096    byte[][] splitKeys = new byte[2][];
097    splitKeys[0] = Bytes.toBytes(4);
098    splitKeys[1] = Bytes.toBytes(8);

101    Table table = TEST_UTIL.createTable(tableName, HConstants.CATALOG_FAMILY, splitKeys);
102    try {
103      TEST_UTIL.loadNumericRows(table, HConstants.CATALOG_FAMILY, 0, 10);

106      table.close();

114    try {
115      assertEquals(0, countRows(table));
116    } finally {
117      table.close();
118    }

356  private void verifyRoundRobinDistribution(RegionLocator regionLocator, int expectedRegions)
357      throws IOException {
358    int numRS = TEST_UTIL.getMiniHBaseCluster().getNumLiveRegionServers();
359    List<HRegionLocation> regions = regionLocator.getAllRegionLocations();
360    Map<ServerName, List<RegionInfo>> server2Regions = new HashMap<>();
361    for (HRegionLocation loc : regions) {
362      ServerName server = loc.getServerName();
363      List<RegionInfo> regs = server2Regions.get(server);
364      if (regs == null) {
365        regs = new ArrayList<>();
366        server2Regions.put(server, regs);
367      }
368      regs.add(loc.getRegion());
369    }
370    float average = (float) expectedRegions / numRS;
371    int min = (int) Math.floor(average);
372    int max = (int) Math.ceil(average);
373    for (List<RegionInfo> regionList : server2Regions.values()) {
374      assertTrue("numRS=" + numRS + ", min=" + min + ", max=" + max + ", size=" + regionList.size(),
375        regionList.size() == min || regionList.size() == max);
376    }
377  }

393  private void testCloneTableSchema(final TableName tableName, final TableName newTableName,
394      boolean preserveSplits) throws Exception {
395    byte[] FAMILY_0 = Bytes.toBytes("cf0");
396    byte[] FAMILY_1 = Bytes.toBytes("cf1");
397    byte[][] splitKeys = new byte[2][];
398    splitKeys[0] = Bytes.toBytes(4);
399    splitKeys[1] = Bytes.toBytes(8);
400    int NUM_FAMILYS = 2;
401    int NUM_REGIONS = 3;
402    int BLOCK_SIZE = 1024;
403    int TTL = 86400;
404    boolean BLOCK_CACHE = false;
405
406    // Create the table
407    TableDescriptor tableDesc = TableDescriptorBuilder.newBuilder(tableName)
408      .setColumnFamily(ColumnFamilyDescriptorBuilder.of(FAMILY_0))
409      .setColumnFamily(ColumnFamilyDescriptorBuilder.newBuilder(FAMILY_1).setBlocksize(BLOCK_SIZE)
410        .setBlockCacheEnabled(BLOCK_CACHE).setTimeToLive(TTL).build())
411      .build();
412    ADMIN.createTable(tableDesc, splitKeys);
413
414    assertEquals(NUM_REGIONS, TEST_UTIL.getHBaseCluster().getRegions(tableName).size());
415    assertTrue("Table should be created with splitKyes + 1 rows in META",
416      ADMIN.isTableAvailable(tableName));
417
418    // clone & Verify
419    ADMIN.cloneTableSchema(tableName, newTableName, preserveSplits);
420    TableDescriptor newTableDesc = ADMIN.getDescriptor(newTableName);
421
422    assertEquals(NUM_FAMILYS, newTableDesc.getColumnFamilyCount());
423    assertEquals(BLOCK_SIZE, newTableDesc.getColumnFamily(FAMILY_1).getBlocksize());
424    assertEquals(BLOCK_CACHE, newTableDesc.getColumnFamily(FAMILY_1).isBlockCacheEnabled());
425    assertEquals(TTL, newTableDesc.getColumnFamily(FAMILY_1).getTimeToLive());
426    TEST_UTIL.verifyTableDescriptorIgnoreTableName(tableDesc, newTableDesc);
427
428    if (preserveSplits) {
429      assertEquals(NUM_REGIONS, TEST_UTIL.getHBaseCluster().getRegions(newTableName).size());
430      assertTrue("New table should be created with splitKyes + 1 rows in META",
431        ADMIN.isTableAvailable(newTableName));
432    } else {
433      assertEquals(1, TEST_UTIL.getHBaseCluster().getRegions(newTableName).size());
434    }
435  }

487  @Test
488  public void testOnlineChangeTableSchema() throws IOException, InterruptedException {
489    final TableName tableName = TableName.valueOf(name.getMethodName());
490    List<TableDescriptor> tables = ADMIN.listTableDescriptors();
491    int numTables = tables.size();
492    TEST_UTIL.createTable(tableName, HConstants.CATALOG_FAMILY).close();
493    tables = ADMIN.listTableDescriptors();
494    assertEquals(numTables + 1, tables.size());
495    // FIRST, do htabledescriptor changes.
496    TableDescriptor htd = ADMIN.getDescriptor(tableName);
497    // Make a copy and assert copy is good.
498    TableDescriptor copy = TableDescriptorBuilder.newBuilder(htd).build();
499    assertEquals(htd, copy);
500    String key = "anyoldkey";
501    assertNull(htd.getValue(key));
502    // Now amend the copy. Introduce differences.
503    long newFlushSize = htd.getMemStoreFlushSize() / 2;
504    if (newFlushSize <= 0) {
505      newFlushSize = TableDescriptorBuilder.DEFAULT_MEMSTORE_FLUSH_SIZE / 2;
506    }
507    copy = TableDescriptorBuilder.newBuilder(copy).setMemStoreFlushSize(newFlushSize)
508      .setValue(key, key).build();
509    ADMIN.modifyTable(copy);
510    TableDescriptor modifiedHtd = ADMIN.getDescriptor(tableName);
511    assertNotEquals(htd, modifiedHtd);
512    assertEquals(copy, modifiedHtd);
513    assertEquals(newFlushSize, modifiedHtd.getMemStoreFlushSize());
514    assertEquals(key, modifiedHtd.getValue(key));
515
516    // Now work on column family changes.
517    int countOfFamilies = modifiedHtd.getColumnFamilyCount();
518    assertTrue(countOfFamilies > 0);
519    ColumnFamilyDescriptor hcd = modifiedHtd.getColumnFamilies()[0];
520    int maxversions = hcd.getMaxVersions();
521    int newMaxVersions = maxversions + 1;
522    hcd = ColumnFamilyDescriptorBuilder.newBuilder(hcd).setMaxVersions(newMaxVersions).build();
523    byte[] hcdName = hcd.getName();
524    ADMIN.modifyColumnFamily(tableName, hcd);
525    modifiedHtd = ADMIN.getDescriptor(tableName);
526    ColumnFamilyDescriptor modifiedHcd = modifiedHtd.getColumnFamily(hcdName);
527    assertEquals(newMaxVersions, modifiedHcd.getMaxVersions());
528
529    // Try adding a column
530    assertFalse(ADMIN.isTableDisabled(tableName));
531    String xtracolName = "xtracol";
532    ColumnFamilyDescriptor xtracol = ColumnFamilyDescriptorBuilder
533      .newBuilder(Bytes.toBytes(xtracolName)).setValue(xtracolName, xtracolName).build();
534    ADMIN.addColumnFamily(tableName, xtracol);
535    modifiedHtd = ADMIN.getDescriptor(tableName);
536    hcd = modifiedHtd.getColumnFamily(xtracol.getName());
537    assertNotNull(hcd);
538    assertEquals(xtracolName, Bytes.toString(hcd.getValue(Bytes.toBytes(xtracolName))));
539
540    // Delete the just-added column.
541    ADMIN.deleteColumnFamily(tableName, xtracol.getName());
542    modifiedHtd = ADMIN.getDescriptor(tableName);
543    hcd = modifiedHtd.getColumnFamily(xtracol.getName());
544    assertNull(hcd);
545
546    // Delete the table
547    ADMIN.disableTable(tableName);
548    ADMIN.deleteTable(tableName);
549    ADMIN.listTableDescriptors();
550    assertFalse(ADMIN.tableExists(tableName));
551  }
