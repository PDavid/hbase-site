189  @Test
190  public void testMaxResultSizeIsSetToDefault() throws Exception {
191    final TableName tableName = TableName.valueOf(name.getMethodName());
192    Table ht = TEST_UTIL.createTable(tableName, FAMILY);
193
194    // The max result size we expect the scan to use by default.
195    long expectedMaxResultSize =
196        TEST_UTIL.getConfiguration().getLong(HConstants.HBASE_CLIENT_SCANNER_MAX_RESULT_SIZE_KEY,
197          HConstants.DEFAULT_HBASE_CLIENT_SCANNER_MAX_RESULT_SIZE);
198
199    int numRows = 5;
200    byte[][] ROWS = HTestConst.makeNAscii(ROW, numRows);
201
202    int numQualifiers = 10;
203    byte[][] QUALIFIERS = HTestConst.makeNAscii(QUALIFIER, numQualifiers);
204
205    // Specify the cell size such that a single row will be larger than the default
206    // value of maxResultSize. This means that Scan RPCs should return at most a single
207    // result back to the client.
208    int cellSize = (int) (expectedMaxResultSize / (numQualifiers - 1));
209    byte[] cellValue = Bytes.createMaxByteArray(cellSize);
210
211    Put put;
212    List<Put> puts = new ArrayList<>();
213    for (int row = 0; row < ROWS.length; row++) {
214      put = new Put(ROWS[row]);
215      for (int qual = 0; qual < QUALIFIERS.length; qual++) {
216        KeyValue kv = new KeyValue(ROWS[row], FAMILY, QUALIFIERS[qual], cellValue);
217        put.add(kv);
218      }
219      puts.add(put);
220    }
221    ht.put(puts);
222
223    // Create a scan with the default configuration.
224    Scan scan = new Scan();
225
226    try (ResultScanner scanner = ht.getScanner(scan)) {
227      assertThat(scanner, instanceOf(AsyncTableResultScanner.class));
228      scanner.next();
229      AsyncTableResultScanner s = (AsyncTableResultScanner) scanner;
230      // The scanner should have, at most, a single result in its cache. If there more results
231      // exists
232      // in the cache it means that more than the expected max result size was fetched.
233      assertTrue("The cache contains: " + s.getCacheSize() + " results", s.getCacheSize() <= 1);
234    }
235  }

294    Scan baseScan = new Scan();

296    baseScan.setSmall(true);
297
298    Scan scan = new Scan(baseScan);
299    verifyExpectedCounts(table, scan, rows, columns);
300
301    scan = new Scan(baseScan);
302    scan.setMaxResultSize(1);
303    verifyExpectedCounts(table, scan, rows, columns);
304

306    scan.setMaxResultSize(1);
307    scan.setCaching(Integer.MAX_VALUE);
308    verifyExpectedCounts(table, scan, rows, columns);
309  }

311  private void verifyExpectedCounts(Table table, Scan scan, int expectedRowCount,
312      int expectedCellCount) throws Exception {
313    ResultScanner scanner = table.getScanner(scan);
314
315    int rowCount = 0;
316    int cellCount = 0;
317    Result r = null;
318    while ((r = scanner.next()) != null) {
319      rowCount++;
320      cellCount += r.rawCells().length;
321    }
322
323    assertTrue("Expected row count: " + expectedRowCount + " Actual row count: " + rowCount,
324        expectedRowCount == rowCount);
325    assertTrue("Expected cell count: " + expectedCellCount + " Actual cell count: " + cellCount,
326        expectedCellCount == cellCount);
327    scanner.close();
328  }

617  @Test
618  public void testScanOnReopenedRegion() throws Exception {
619    final TableName tableName = TableName.valueOf(name.getMethodName());
620    byte [][] QUALIFIERS = HTestConst.makeNAscii(QUALIFIER, 2);
621
622    Table ht = TEST_UTIL.createTable(tableName, FAMILY);
623
624    Put put;
625    Scan scan;
626    Result result;
627    ResultScanner scanner;
628    boolean toLog = false;
629    List<Cell> kvListExp;
630
631    // table: row, family, c0:0, c1:1
632    put = new Put(ROW);
633    for (int i=0; i < QUALIFIERS.length; i++) {
634      KeyValue kv = new KeyValue(ROW, FAMILY, QUALIFIERS[i], i, VALUE);
635      put.add(kv);
636    }
637    ht.put(put);
638
639    scan = new Scan().withStartRow(ROW);
640    scanner = ht.getScanner(scan);
641
642    HRegionLocation loc;
643
644    try (RegionLocator locator = TEST_UTIL.getConnection().getRegionLocator(tableName)) {
645      loc = locator.getRegionLocation(ROW);
646    }
647    RegionInfo hri = loc.getRegion();
648    MiniHBaseCluster cluster = TEST_UTIL.getMiniHBaseCluster();
649    byte[] regionName = hri.getRegionName();
650    int i = cluster.getServerWith(regionName);
651    HRegionServer rs = cluster.getRegionServer(i);
652    LOG.info("Unassigning " + hri);
653    TEST_UTIL.getAdmin().unassign(hri.getRegionName(), true);
654    long startTime = EnvironmentEdgeManager.currentTime();
655    long timeOut = 10000;
656    boolean offline = false;
657    while (true) {
658      if (rs.getOnlineRegion(regionName) == null) {
659        offline = true;
660        break;
661      }
662      assertTrue("Timed out in closing the testing region",
663        EnvironmentEdgeManager.currentTime() < startTime + timeOut);
664    }
665    assertTrue(offline);
666    LOG.info("Assigning " + hri);
667    TEST_UTIL.getAdmin().assign(hri.getRegionName());
668    startTime = EnvironmentEdgeManager.currentTime();
669    while (true) {
670      rs = cluster.getRegionServer(cluster.getServerWith(regionName));
671      if (rs != null && rs.getOnlineRegion(regionName) != null) {
672        offline = false;
673        break;
674      }
675      assertTrue("Timed out in open the testing region",
676        EnvironmentEdgeManager.currentTime() < startTime + timeOut);
677    }
678    assertFalse(offline);
679
680    // c0:0, c1:1
681    kvListExp = new ArrayList<>();
682    kvListExp.add(new KeyValue(ROW, FAMILY, QUALIFIERS[0], 0, VALUE));
683    kvListExp.add(new KeyValue(ROW, FAMILY, QUALIFIERS[1], 1, VALUE));
684    result = scanner.next();
685    verifyResult(result, kvListExp, toLog, "Testing scan on re-opened region");
686  }

688  static void verifyResult(Result result, List<Cell> expKvList, boolean toLog,
689      String msg) {
690
691    LOG.info(msg);
692    LOG.info("Expected count: " + expKvList.size());
693    LOG.info("Actual count: " + result.size());
694    if (expKvList.isEmpty())
695      return;
696
697    int i = 0;
698    for (Cell kv : result.rawCells()) {
699      if (i >= expKvList.size()) {
700        break;  // we will check the size later
701      }
702
703      Cell kvExp = expKvList.get(i++);
704      if (toLog) {
705        LOG.info("get kv is: " + kv.toString());
706        LOG.info("exp kv is: " + kvExp.toString());
707      }
708      assertTrue("Not equal", kvExp.equals(kv));
709    }
710
711    assertEquals(expKvList.size(), result.size());
712  }

795    TableName tableName = TableName.valueOf(name.getMethodName());

797    final int ROWS_TO_INSERT = 100;
798    final byte[] LARGE_VALUE = generateHugeValue(128 * 1024);
799
800    try (Table table = TEST_UTIL.createTable(tableName, FAMILY);
801        Admin admin = TEST_UTIL.getAdmin()) {
802      List<Put> putList = new ArrayList<>();
803      for (long i = 0; i < ROWS_TO_INSERT; i++) {
804        Put put = new Put(Bytes.toBytes(i));
805        put.addColumn(FAMILY, QUALIFIER, LARGE_VALUE);
806        putList.add(put);
807
808        if (putList.size() >= BATCH_SIZE) {
809          table.put(putList);
810          admin.flush(tableName);
811          putList.clear();
812        }
813      }
814
815      if (!putList.isEmpty()) {
816        table.put(putList);
817        admin.flush(tableName);
818        putList.clear();
819      }
820
821      Scan scan = new Scan();
822      scan.setReversed(true);
823      int count = 0;
824
825      try (ResultScanner results = table.getScanner(scan)) {
826        for (Result result : results) {
827          count++;
828        }
829      }
830      assertEquals("Expected " + ROWS_TO_INSERT + " rows in the table but it is " + count,
831          ROWS_TO_INSERT, count);
