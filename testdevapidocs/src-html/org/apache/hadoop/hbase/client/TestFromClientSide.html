530   * @return Scan with RowFilter that does CompareOp op on passed key.
531   */
532  private Scan createScanWithRowFilter(final byte [] key,
533      final byte [] startRow, CompareOperator op) {
534    // Make sure key is of some substance... non-null and > than first key.
535    assertTrue(key != null && key.length > 0 &&
536      Bytes.BYTES_COMPARATOR.compare(key, new byte [] {'a', 'a', 'a'}) >= 0);
537    LOG.info("Key=" + Bytes.toString(key));
538    Scan s = startRow == null? new Scan(): new Scan(startRow);
539    Filter f = new RowFilter(op, new BinaryComparator(key));
540    f = new WhileMatchFilter(f);
541    s.setFilter(f);
542    return s;
543  }

575      int originalCount = regions.size();
