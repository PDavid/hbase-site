401    LOG.info("Starting testAppend, walUsed is " + walUsed);

417      Append append_2 = new Append(row1);
418      append_2.addColumn(FAMILY, qual, Bytes.toBytes("e"));
419      if (!walUsed) {
420        append_2.setDurability(Durability.SKIP_WAL);
421      }
422      Append append_3 = new Append(row1);
423      append_3.addColumn(FAMILY, qual, Bytes.toBytes("a"));

425      s.setCaching(1);

433        t.append(append_3);

435          results.add(r);
436        }
437      }
438      TEST_UTIL.deleteTable(TABLENAME);
439      return results;
440    }
