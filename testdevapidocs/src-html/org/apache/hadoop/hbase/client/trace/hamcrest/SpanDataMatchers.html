018package org.apache.hadoop.hbase.client.trace.hamcrest;
019
020import static org.hamcrest.Matchers.equalTo;
021import static org.hamcrest.Matchers.is;
022import io.opentelemetry.api.common.Attributes;
023import io.opentelemetry.api.trace.SpanKind;
024import io.opentelemetry.api.trace.StatusCode;
025import io.opentelemetry.sdk.trace.data.EventData;
026import io.opentelemetry.sdk.trace.data.SpanData;
027import io.opentelemetry.sdk.trace.data.StatusData;
028import java.time.Duration;
029import org.hamcrest.Description;
030import org.hamcrest.FeatureMatcher;
031import org.hamcrest.Matcher;
032import org.hamcrest.TypeSafeMatcher;
033
034/**
035 * Helper methods for matching against instances of {@link SpanData}.
036 */
037public final class SpanDataMatchers {
038
039  private SpanDataMatchers() { }
040
041  public static Matcher<SpanData> hasAttributes(Matcher<Attributes> matcher) {
042    return new FeatureMatcher<SpanData, Attributes>(
043      matcher, "SpanData having attributes that ", "attributes"
044    ) {
045      @Override protected Attributes featureValueOf(SpanData item) {
046        return item.getAttributes();
047      }
048    };
049  }
050
051  public static Matcher<SpanData> hasDuration(Matcher<Duration> matcher) {
052    return new FeatureMatcher<SpanData, Duration>(
053      matcher, "SpanData having duration that ", "duration") {
054      @Override
055      protected Duration featureValueOf(SpanData item) {
056        return Duration.ofNanos(item.getEndEpochNanos() - item.getStartEpochNanos());
057      }
058    };
059  }
060
061  public static Matcher<SpanData> hasEnded() {
062    return new TypeSafeMatcher<SpanData>() {
063      @Override protected boolean matchesSafely(SpanData item) {
064        return item.hasEnded();
065      }
066      @Override public void describeTo(Description description) {
067        description.appendText("SpanData that hasEnded");
068      }
069    };
070  }
071
072  public static Matcher<SpanData> hasEvents(Matcher<Iterable<? super EventData>> matcher) {
073    return new FeatureMatcher<SpanData, Iterable<? super EventData>>(
074      matcher, "SpanData having events that", "events") {
075      @Override protected Iterable<? super EventData> featureValueOf(SpanData item) {
076        return item.getEvents();
077      }
078    };
079  }
080
081  public static Matcher<SpanData> hasKind(SpanKind kind) {
082    return new FeatureMatcher<SpanData, SpanKind>(
083      equalTo(kind), "SpanData with kind that", "SpanKind") {
084      @Override protected SpanKind featureValueOf(SpanData item) {
085        return item.getKind();
086      }
087    };
088  }
089
090  public static Matcher<SpanData> hasName(String name) {
091    return hasName(equalTo(name));
092  }
093
094  public static Matcher<SpanData> hasName(Matcher<String> matcher) {
095    return new FeatureMatcher<SpanData, String>(matcher, "SpanKind with a name that", "name") {
096      @Override protected String featureValueOf(SpanData item) {
097        return item.getName();
098      }
099    };
100  }
101
102  public static Matcher<SpanData> hasStatusWithCode(StatusCode statusCode) {
103    final Matcher<StatusCode> matcher = is(equalTo(statusCode));
104    return new TypeSafeMatcher<SpanData>() {
105      @Override protected boolean matchesSafely(SpanData item) {
106        final StatusData statusData = item.getStatus();
107        return statusData != null
108          && statusData.getStatusCode() != null
109          && matcher.matches(statusData.getStatusCode());
110      }
111      @Override public void describeTo(Description description) {
112        description.appendText("SpanData with StatusCode that ").appendDescriptionOf(matcher);
113      }
114    };
115  }
116
117  public static Matcher<SpanData> hasTraceId(String traceId) {
118    return hasTraceId(is(equalTo(traceId)));
119  }
120
121  public static Matcher<SpanData> hasTraceId(Matcher<String> matcher) {
122    return new FeatureMatcher<SpanData, String>(
123      matcher, "SpanData with a traceId that ", "traceId") {
124      @Override protected String featureValueOf(SpanData item) {
125        return item.getTraceId();
126      }
127    };
128  }
129}
