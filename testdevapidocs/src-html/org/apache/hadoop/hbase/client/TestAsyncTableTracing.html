099  @Before
100  public void setUp() throws IOException {
101    stub = mock(ClientService.Interface.class);
102    AtomicInteger scanNextCalled = new AtomicInteger(0);
103    doAnswer(new Answer<Void>() {
104
105      @Override
106      public Void answer(InvocationOnMock invocation) throws Throwable {
107        ScanRequest req = invocation.getArgument(1);
108        RpcCallback<ScanResponse> done = invocation.getArgument(2);
109        if (!req.hasScannerId()) {
110          done.run(ScanResponse.newBuilder().setScannerId(1).setTtl(800)
111            .setMoreResultsInRegion(true).setMoreResults(true).build());
112        } else {
113          if (req.hasCloseScanner() && req.getCloseScanner()) {
114            done.run(ScanResponse.getDefaultInstance());
115          } else {
116            Cell cell = CellBuilderFactory.create(CellBuilderType.SHALLOW_COPY).setType(Type.Put)
117              .setRow(Bytes.toBytes(scanNextCalled.incrementAndGet()))
118              .setFamily(Bytes.toBytes("cf")).setQualifier(Bytes.toBytes("cq"))
119              .setValue(Bytes.toBytes("v")).build();
120            Result result = Result.create(Arrays.asList(cell));
121            ScanResponse.Builder builder = ScanResponse.newBuilder().setScannerId(1).setTtl(800)
122              .addResults(ProtobufUtil.toResult(result));
123            if (req.getLimitOfRows() == 1) {
124              builder.setMoreResultsInRegion(false).setMoreResults(false);
125            } else {
126              builder.setMoreResultsInRegion(true).setMoreResults(true);
127            }
128            ForkJoinPool.commonPool().execute(() -> done.run(builder.build()));
129          }
130        }
131        return null;
132      }
133    }).when(stub).scan(any(HBaseRpcController.class), any(ScanRequest.class), any());
134    doAnswer(new Answer<Void>() {
135
136      @Override
137      public Void answer(InvocationOnMock invocation) throws Throwable {
138        ClientProtos.MultiResponse resp =
139          ClientProtos.MultiResponse.newBuilder()
140            .addRegionActionResult(RegionActionResult.newBuilder().addResultOrException(
141              ResultOrException.newBuilder().setResult(ProtobufUtil.toResult(new Result()))))
142            .build();
143        RpcCallback<ClientProtos.MultiResponse> done = invocation.getArgument(2);
144        ForkJoinPool.commonPool().execute(() -> done.run(resp));
145        return null;
146      }
147    }).when(stub).multi(any(HBaseRpcController.class), any(ClientProtos.MultiRequest.class), any());
148    doAnswer(new Answer<Void>() {
149
150      @Override
151      public Void answer(InvocationOnMock invocation) throws Throwable {
152        MutationProto req = ((MutateRequest) invocation.getArgument(1)).getMutation();
153        MutateResponse resp;
154        switch (req.getMutateType()) {
155          case INCREMENT:
156            ColumnValue value = req.getColumnValue(0);
157            QualifierValue qvalue = value.getQualifierValue(0);
158            Cell cell = CellBuilderFactory.create(CellBuilderType.SHALLOW_COPY).setType(Type.Put)
159              .setRow(req.getRow().toByteArray()).setFamily(value.getFamily().toByteArray())
160              .setQualifier(qvalue.getQualifier().toByteArray())
161              .setValue(qvalue.getValue().toByteArray()).build();
162            resp = MutateResponse.newBuilder()
163              .setResult(ProtobufUtil.toResult(Result.create(Arrays.asList(cell)))).build();
164            break;
165          default:
166            resp = MutateResponse.getDefaultInstance();
167            break;
168        }
169        RpcCallback<MutateResponse> done = invocation.getArgument(2);
170        ForkJoinPool.commonPool().execute(() -> done.run(resp));
171        return null;
172      }
173    }).when(stub).mutate(any(HBaseRpcController.class), any(MutateRequest.class), any());
174    doAnswer(new Answer<Void>() {
175
176      @Override
177      public Void answer(InvocationOnMock invocation) throws Throwable {
178        RpcCallback<GetResponse> done = invocation.getArgument(2);
179        ForkJoinPool.commonPool().execute(() -> done.run(GetResponse.getDefaultInstance()));
180        return null;
181      }
182    }).when(stub).get(any(HBaseRpcController.class), any(GetRequest.class), any());
183    conn = new AsyncConnectionImpl(CONF, new DoNothingConnectionRegistry(CONF), "test", null,
184      UserProvider.instantiate(CONF).getCurrent()) {
185
186      @Override
187      AsyncRegionLocator getLocator() {
188        AsyncRegionLocator locator = mock(AsyncRegionLocator.class);
189        Answer<CompletableFuture<HRegionLocation>> answer =
190          new Answer<CompletableFuture<HRegionLocation>>() {
191
192            @Override
193            public CompletableFuture<HRegionLocation> answer(InvocationOnMock invocation)
194              throws Throwable {
195              TableName tableName = invocation.getArgument(0);
196              RegionInfo info = RegionInfoBuilder.newBuilder(tableName).build();
197              ServerName serverName = ServerName.valueOf("rs", 16010, 12345);
198              HRegionLocation loc = new HRegionLocation(info, serverName);
199              return CompletableFuture.completedFuture(loc);
200            }
201          };
202        doAnswer(answer).when(locator).getRegionLocation(any(TableName.class), any(byte[].class),
203          any(RegionLocateType.class), anyLong());
204        doAnswer(answer).when(locator).getRegionLocation(any(TableName.class), any(byte[].class),
205          anyInt(), any(RegionLocateType.class), anyLong());
206        return locator;
207      }

209      @Override
210      ClientService.Interface getRegionServerStub(ServerName serverName) throws IOException {
211        return stub;

222  private void assertTrace(String methodName) {
223    Waiter.waitFor(CONF, 1000,
224      () -> traceRule.getSpans().stream()
225        .anyMatch(span -> span.getName().equals("AsyncTable." + methodName) &&
226          span.getKind() == SpanKind.INTERNAL && span.hasEnded()));
227    SpanData data = traceRule.getSpans().stream()
228      .filter(s -> s.getName().equals("AsyncTable." + methodName)).findFirst().get();
229    assertEquals(StatusCode.OK, data.getStatus().getStatusCode());
230    TableName tableName = table.getName();
231    assertEquals(tableName.getNamespaceAsString(), data.getAttributes().get(NAMESPACE_KEY));
232    assertEquals(tableName.getNameAsString(), data.getAttributes().get(TABLE_KEY));
233  }
