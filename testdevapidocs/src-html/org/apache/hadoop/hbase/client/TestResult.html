055  static KeyValue[] genKVs(final byte[] row, final byte[] family,
056                           final byte[] value,
057                    final long timestamp,
058                    final int cols) {
059    KeyValue [] kvs = new KeyValue[cols];
060
061    for (int i = 0; i < cols ; i++) {
062      kvs[i] = new KeyValue(
063          row, family, Bytes.toBytes(i),
064          timestamp,
065          Bytes.add(value, Bytes.toBytes(i)));
066    }
067    return kvs;
068  }

075  /**
076   * Run some tests to ensure Result acts like a proper CellScanner.
077   * @throws IOException
078   */
079  public void testResultAsCellScanner() throws IOException {
080    Cell [] cells = genKVs(row, family, value, 1, 10);
081    Arrays.sort(cells, CellComparator.getInstance());
082    Result r = Result.create(cells);
083    assertSame(r, cells);
084    // Assert I run over same result multiple times.
085    assertSame(r.cellScanner(), cells);
086    assertSame(r.cellScanner(), cells);
087    // Assert we are not creating new object when doing cellscanner
088    assertTrue(r == r.cellScanner());
089  }
090
091  private void assertSame(final CellScanner cellScanner, final Cell [] cells) throws IOException {
092    int count = 0;
093    while (cellScanner.advance()) {
094      assertTrue(cells[count].equals(cellScanner.current()));
095      count++;
096    }
097    assertEquals(cells.length, count);
098  }

190  public void testBasicLoadValue() throws Exception {
191    KeyValue [] kvs = genKVs(row, family, value, 1, 100);
192
193    Arrays.sort(kvs, CellComparator.getInstance());
194
195    Result r = Result.create(kvs);
196    ByteBuffer loadValueBuffer = ByteBuffer.allocate(1024);
197
198    for (int i = 0; i < 100; ++i) {
199      final byte[] qf = Bytes.toBytes(i);
200
201      loadValueBuffer.clear();
202      r.loadValue(family, qf, loadValueBuffer);
203      loadValueBuffer.flip();
204      assertEquals(loadValueBuffer, ByteBuffer.wrap(Bytes.add(value, Bytes.toBytes(i))));
205      assertEquals(ByteBuffer.wrap(Bytes.add(value, Bytes.toBytes(i))),
206          r.getValueAsByteBuffer(family, qf));

372  private Result getArrayBackedTagResult(Tag tag) {
373    List<Tag> tags = null;
374    if (tag != null) {
375      tags = Arrays.asList(tag);
376    }
377    KeyValue kvCell = new KeyValue(row, family, qual, 0L, KeyValue.Type.Put,
378      value, tags);
379    return Result.create(new Cell[] {kvCell});
380  }
381
382  private Result getByteBufferBackedTagResult(Tag tag) {
383    List<Tag> tags = null;
384    if (tag != null) {
385      tags = Arrays.asList(tag);
386    }
387    KeyValue kvCell = new KeyValue(row, family, qual, 0L, KeyValue.Type.Put,
388        value, tags);
389    ByteBuffer buf = ByteBuffer.allocateDirect(kvCell.getBuffer().length);
390    ByteBufferUtils.copyFromArrayToBuffer(buf, kvCell.getBuffer(), 0, kvCell.getBuffer().length);
391    ByteBufferKeyValue bbKV = new ByteBufferKeyValue(buf, 0, buf.capacity(), 0L);
392    return Result.create(new Cell[] {bbKV});
393  }

415  /**
416   * Microbenchmark that compares {@link Result#getValue} and {@link Result#loadValue} performance.
417   *
418   * @throws Exception
419   */
420  public void doReadBenchmark() throws Exception {
421
422    final int n = 5;
423    final int m = 100000000;
424
425    StringBuilder valueSB = new StringBuilder();
426    for (int i = 0; i < 100; i++) {
427      valueSB.append((byte)(Math.random() * 10));
428    }
429
430    StringBuilder rowSB = new StringBuilder();
431    for (int i = 0; i < 50; i++) {
432      rowSB.append((byte)(Math.random() * 10));
433    }
434
435    KeyValue [] kvs = genKVs(Bytes.toBytes(rowSB.toString()), family,
436        Bytes.toBytes(valueSB.toString()), 1, n);
437    Arrays.sort(kvs, CellComparator.getInstance());
438    ByteBuffer loadValueBuffer = ByteBuffer.allocate(1024);
439    Result r = Result.create(kvs);
440
441    byte[][] qfs = new byte[n][Bytes.SIZEOF_INT];
442    for (int i = 0; i < n; ++i) {
443      System.arraycopy(qfs[i], 0, Bytes.toBytes(i), 0, Bytes.SIZEOF_INT);
444    }
445
446    // warm up
447    for (int k = 0; k < 100000; k++) {
448      for (int i = 0; i < n; ++i) {
449        r.getValue(family, qfs[i]);
450        loadValueBuffer.clear();
451        r.loadValue(family, qfs[i], loadValueBuffer);
452        loadValueBuffer.flip();
453      }
454    }
455
456    System.gc();
457    long start = System.nanoTime();
458    for (int k = 0; k < m; k++) {
459      for (int i = 0; i < n; ++i) {
460        loadValueBuffer.clear();
461        r.loadValue(family, qfs[i], loadValueBuffer);
462        loadValueBuffer.flip();
463      }
464    }
465    long stop = System.nanoTime();
466    System.out.println("loadValue(): " + (stop - start));
467
468    System.gc();
469    start = System.nanoTime();
470    for (int k = 0; k < m; k++) {
471      for (int i = 0; i < n; i++) {
472        r.getValue(family, qfs[i]);
473      }
474    }
475    stop = System.nanoTime();
476    System.out.println("getValue():  " + (stop - start));
477  }
