094  public static void setUpTestBeforeClass() throws Exception {
095    Configuration conf = TEST_UTIL.getConfiguration();
096    conf.setFloat("hbase.master.balancer.stochastic.tableSkewCost", 6000);
097    if (conf.get(RSGroupUtil.RS_GROUP_ENABLED) == null) {
098      RSGroupUtil.enableRSGroup(conf);
099    }
100    if (conf.get(CoprocessorHost.MASTER_COPROCESSOR_CONF_KEY) != null) {
101      conf.set(CoprocessorHost.MASTER_COPROCESSOR_CONF_KEY,
102        conf.get(CoprocessorHost.MASTER_COPROCESSOR_CONF_KEY) + "," +
103          CPMasterObserver.class.getName());
104    } else {
105      conf.set(CoprocessorHost.MASTER_COPROCESSOR_CONF_KEY, CPMasterObserver.class.getName());
106    }
107
108    conf.setInt(ServerManager.WAIT_ON_REGIONSERVERS_MINTOSTART,
109      NUM_SLAVES_BASE - 1);
110    conf.setBoolean(SnapshotManager.HBASE_SNAPSHOT_ENABLED, true);
111    conf.setInt("hbase.rpc.timeout", 100000);
112
113    TEST_UTIL.startMiniCluster(NUM_SLAVES_BASE - 1);
114    initialize();
115  }

117  protected static void initialize() throws Exception {
118    ADMIN = new VerifyingRSGroupAdmin(TEST_UTIL.getConfiguration());
119    CLUSTER = TEST_UTIL.getHBaseCluster();
120    MASTER = TEST_UTIL.getMiniHBaseCluster().getMaster();
121
122    // wait for balancer to come online
123    TEST_UTIL.waitFor(WAIT_TIMEOUT, new Waiter.Predicate<Exception>() {
124      @Override
125      public boolean evaluate() throws Exception {
126        return MASTER.isInitialized() &&
127          ((RSGroupBasedLoadBalancer) MASTER.getLoadBalancer()).isOnline();
128      }
129    });
130    ADMIN.balancerSwitch(false, true);
131    MasterCoprocessorHost host = MASTER.getMasterCoprocessorHost();
132    OBSERVER = (CPMasterObserver) host.findCoprocessor(CPMasterObserver.class.getName());
133  }

139  public void setUpBeforeMethod() throws Exception {
140    LOG.info(name.getMethodName());
141    tableName = TableName.valueOf(TABLE_PREFIX + "_" + name.getMethodName().split("\\[")[0]);
142    if (!INIT) {
143      INIT = true;

145    }

149  public void tearDownAfterMethod() throws Exception {
150    deleteTableIfNecessary();
151    deleteNamespaceIfNecessary();
152    deleteGroups();
153
154    for (ServerName sn : ADMIN.listDecommissionedRegionServers()) {
155      ADMIN.recommissionRegionServer(sn, null);
156    }
157    assertTrue(ADMIN.listDecommissionedRegionServers().isEmpty());
158
159    int missing = NUM_SLAVES_BASE - getNumServers();
160    LOG.info("Restoring servers: " + missing);
161    for (int i = 0; i < missing; i++) {
162      ((MiniHBaseCluster) CLUSTER).startRegionServer();
163    }
164    ADMIN.addRSGroup("master");
165    ServerName masterServerName = ((MiniHBaseCluster) CLUSTER).getMaster().getServerName();
166    try {
167      ADMIN.moveServersToRSGroup(Sets.newHashSet(masterServerName.getAddress()), "master");
168    } catch (Exception ex) {
169      LOG.warn("Got this on setup, FYI", ex);
170    }
171    assertTrue(OBSERVER.preMoveServersCalled);
172    TEST_UTIL.waitFor(WAIT_TIMEOUT, new Waiter.Predicate<Exception>() {
173      @Override
174      public boolean evaluate() throws Exception {
175        LOG.info("Waiting for cleanup to finish " + ADMIN.listRSGroups());
176        // Might be greater since moving servers back to default
177        // is after starting a server
178
179        return ADMIN.getRSGroup(RSGroupInfo.DEFAULT_GROUP).getServers().size() == NUM_SLAVES_BASE;
180      }
181    });
182  }

184  protected final RSGroupInfo addGroup(String groupName, int serverCount)
185    throws IOException, InterruptedException {
186    RSGroupInfo defaultInfo = ADMIN.getRSGroup(RSGroupInfo.DEFAULT_GROUP);
187    ADMIN.addRSGroup(groupName);
188    Set<Address> set = new HashSet<>();
189    for (Address server : defaultInfo.getServers()) {
190      if (set.size() == serverCount) {
191        break;
192      }
193      set.add(server);
194    }
195    ADMIN.moveServersToRSGroup(set, groupName);
196    RSGroupInfo result = ADMIN.getRSGroup(groupName);
197    return result;
198  }
199
200  protected final void removeGroup(String groupName) throws IOException {
201    Set<TableName> tables = new HashSet<>();
202    for (TableDescriptor td : ADMIN.listTableDescriptors(true)) {
203      RSGroupInfo groupInfo = ADMIN.getRSGroup(td.getTableName());
204      if (groupInfo != null && groupInfo.getName().equals(groupName)) {
205        tables.add(td.getTableName());
206      }
207    }
208    ADMIN.setRSGroup(tables, RSGroupInfo.DEFAULT_GROUP);
209    RSGroupInfo groupInfo = ADMIN.getRSGroup(groupName);
210    ADMIN.moveServersToRSGroup(groupInfo.getServers(), RSGroupInfo.DEFAULT_GROUP);
211    ADMIN.removeRSGroup(groupName);
212  }

214  protected final void deleteTableIfNecessary() throws IOException {
215    for (TableDescriptor desc : TEST_UTIL.getAdmin()
216      .listTableDescriptors(Pattern.compile(TABLE_PREFIX + ".*"))) {
217      TEST_UTIL.deleteTable(desc.getTableName());
218    }
219  }
220
221  protected final void deleteNamespaceIfNecessary() throws IOException {
222    for (NamespaceDescriptor desc : TEST_UTIL.getAdmin().listNamespaceDescriptors()) {
223      if (desc.getName().startsWith(TABLE_PREFIX)) {
224        ADMIN.deleteNamespace(desc.getName());
225      }
226    }
227  }
228
229  protected final void deleteGroups() throws IOException {
230    for (RSGroupInfo groupInfo : ADMIN.listRSGroups()) {
231      if (!groupInfo.getName().equals(RSGroupInfo.DEFAULT_GROUP)) {
232        removeGroup(groupInfo.getName());
233      }
234    }
235  }

237  protected Map<TableName, List<String>> getTableRegionMap() throws IOException {
238    Map<TableName, List<String>> map = Maps.newTreeMap();
239    Map<TableName, Map<ServerName, List<String>>> tableServerRegionMap = getTableServerRegionMap();
240    for (TableName tableName : tableServerRegionMap.keySet()) {
241      if (!map.containsKey(tableName)) {
242        map.put(tableName, new LinkedList<>());
243      }
244      for (List<String> subset : tableServerRegionMap.get(tableName).values()) {
245        map.get(tableName).addAll(subset);
246      }
247    }
248    return map;
249  }
250
251  protected Map<TableName, Map<ServerName, List<String>>> getTableServerRegionMap()
252    throws IOException {
253    Map<TableName, Map<ServerName, List<String>>> map = Maps.newTreeMap();
254    Admin admin = TEST_UTIL.getAdmin();
255    ClusterMetrics metrics =
256      admin.getClusterMetrics(EnumSet.of(ClusterMetrics.Option.SERVERS_NAME));
257    for (ServerName serverName : metrics.getServersName()) {
258      for (RegionInfo region : admin.getRegions(serverName)) {
259        TableName tableName = region.getTable();
260        map.computeIfAbsent(tableName, k -> new TreeMap<>())
261          .computeIfAbsent(serverName, k -> new ArrayList<>()).add(region.getRegionNameAsString());
262      }
263    }
264    return map;
265  }

267  // return the real number of region servers, excluding the master embedded region server in 2.0+
268  protected int getNumServers() throws IOException {
269    ClusterMetrics status = ADMIN.getClusterMetrics(EnumSet.of(Option.MASTER, Option.LIVE_SERVERS));
270    ServerName masterName = status.getMasterName();
271    int count = 0;
272    for (ServerName sn : status.getLiveServerMetrics().keySet()) {
273      if (!sn.equals(masterName)) {
274        count++;
275      }
276    }
277    return count;
278  }

289  protected final ServerName getServerName(Address addr) {
290    return TEST_UTIL.getMiniHBaseCluster().getRegionServerThreads().stream()
291      .map(t -> t.getRegionServer().getServerName()).filter(sn -> sn.getAddress().equals(addr))
292      .findFirst().get();
