094  public static void setUpTestBeforeClass() throws Exception {
095    TEST_UTIL = new HBaseTestingUtility();
096    TEST_UTIL.getConfiguration().setFloat(
097            "hbase.master.balancer.stochastic.tableSkewCost", 6000);
098    TEST_UTIL.getConfiguration().set(
099        HConstants.HBASE_MASTER_LOADBALANCER_CLASS,
100        RSGroupBasedLoadBalancer.class.getName());
101    TEST_UTIL.getConfiguration().set(CoprocessorHost.MASTER_COPROCESSOR_CONF_KEY,
102        RSGroupAdminEndpoint.class.getName() + "," + CPMasterObserver.class.getName());
103    TEST_UTIL.getConfiguration().setInt(
104        ServerManager.WAIT_ON_REGIONSERVERS_MINTOSTART,
105        NUM_SLAVES_BASE - 1);
106    TEST_UTIL.getConfiguration().setBoolean(SnapshotManager.HBASE_SNAPSHOT_ENABLED, true);
107    TEST_UTIL.getConfiguration().setInt("hbase.rpc.timeout", 100000);
108
109    TEST_UTIL.startMiniCluster(NUM_SLAVES_BASE - 1);
110    initialize();
111  }

113  protected static void initialize() throws Exception {
114    admin = TEST_UTIL.getAdmin();
115    cluster = TEST_UTIL.getHBaseCluster();
116    master = TEST_UTIL.getMiniHBaseCluster().getMaster();
117
118    //wait for balancer to come online
119    TEST_UTIL.waitFor(WAIT_TIMEOUT, new Waiter.Predicate<Exception>() {
120      @Override
121      public boolean evaluate() throws Exception {
122        return master.isInitialized() &&
123            ((RSGroupBasedLoadBalancer) master.getLoadBalancer()).isOnline();
124      }
125    });
126    admin.balancerSwitch(false, true);
127    rsGroupAdmin = new VerifyingRSGroupAdminClient(
128        new RSGroupAdminClient(TEST_UTIL.getConnection()), TEST_UTIL.getConfiguration());
129    MasterCoprocessorHost host = master.getMasterCoprocessorHost();
130    observer = (CPMasterObserver) host.findCoprocessor(CPMasterObserver.class.getName());
131    rsGroupAdminEndpoint = (RSGroupAdminEndpoint)
132        host.findCoprocessor(RSGroupAdminEndpoint.class.getName());
133  }

139  public void setUpBeforeMethod() throws Exception {
140    LOG.info(name.getMethodName());
141    tableName = TableName.valueOf(tablePrefix + "_" + name.getMethodName());
142    if (!INIT) {
143      INIT = true;
144      tearDownAfterMethod();
145    }
146    observer.resetFlags();

149  public void tearDownAfterMethod() throws Exception {
150    deleteTableIfNecessary();
151    deleteNamespaceIfNecessary();
152    deleteGroups();
153
154    for(ServerName sn : admin.listDecommissionedRegionServers()){
155      admin.recommissionRegionServer(sn, null);
156    }
157    assertTrue(admin.listDecommissionedRegionServers().isEmpty());
158
159    int missing = NUM_SLAVES_BASE - getNumServers();
160    LOG.info("Restoring servers: "+missing);
161    for(int i=0; i<missing; i++) {
162      ((MiniHBaseCluster)cluster).startRegionServer();
163    }
164
165    rsGroupAdmin.addRSGroup("master");
166    ServerName masterServerName =
167        ((MiniHBaseCluster)cluster).getMaster().getServerName();
168
169    try {
170      rsGroupAdmin.moveServers(Sets.newHashSet(masterServerName.getAddress()), "master");
171    } catch (Exception ex) {
172      LOG.warn("Got this on setup, FYI", ex);
173    }
174    assertTrue(observer.preMoveServersCalled);
175    TEST_UTIL.waitFor(WAIT_TIMEOUT, new Waiter.Predicate<Exception>() {
176      @Override
177      public boolean evaluate() throws Exception {
178        LOG.info("Waiting for cleanup to finish " + rsGroupAdmin.listRSGroups());
179        //Might be greater since moving servers back to default
180        //is after starting a server
181
182        return rsGroupAdmin.getRSGroupInfo(RSGroupInfo.DEFAULT_GROUP).getServers().size()
183            == NUM_SLAVES_BASE;
184      }
185    });
186  }

188  public RSGroupInfo addGroup(String groupName, int serverCount)
189      throws IOException, InterruptedException {
190    RSGroupInfo defaultInfo = rsGroupAdmin.getRSGroupInfo(RSGroupInfo.DEFAULT_GROUP);
191    rsGroupAdmin.addRSGroup(groupName);
192    Set<Address> set = new HashSet<>();
193    for(Address server: defaultInfo.getServers()) {
194      if(set.size() == serverCount) {
195        break;
196      }
197      set.add(server);
198    }
199    rsGroupAdmin.moveServers(set, groupName);
200    RSGroupInfo result = rsGroupAdmin.getRSGroupInfo(groupName);
201    return result;
202  }

204  public void removeGroup(String groupName) throws IOException {
205    RSGroupInfo groupInfo = rsGroupAdmin.getRSGroupInfo(groupName);
206    rsGroupAdmin.moveTables(groupInfo.getTables(), RSGroupInfo.DEFAULT_GROUP);
207    rsGroupAdmin.moveServers(groupInfo.getServers(), RSGroupInfo.DEFAULT_GROUP);
208    rsGroupAdmin.removeRSGroup(groupName);
209  }

211  public void deleteTableIfNecessary() throws IOException {
212    for (TableDescriptor desc : TEST_UTIL.getAdmin()
213      .listTableDescriptors(Pattern.compile(tablePrefix + ".*"))) {
214      TEST_UTIL.deleteTable(desc.getTableName());
215    }
216  }

218  public void deleteNamespaceIfNecessary() throws IOException {
219    for (NamespaceDescriptor desc : TEST_UTIL.getAdmin().listNamespaceDescriptors()) {
220      if(desc.getName().startsWith(tablePrefix)) {
221        admin.deleteNamespace(desc.getName());
222      }
223    }
224  }

226  public void deleteGroups() throws IOException {
227    RSGroupAdmin groupAdmin = new RSGroupAdminClient(TEST_UTIL.getConnection());
228    for(RSGroupInfo group: groupAdmin.listRSGroups()) {
229      if(!group.getName().equals(RSGroupInfo.DEFAULT_GROUP)) {
230        groupAdmin.moveTables(group.getTables(), RSGroupInfo.DEFAULT_GROUP);
231        groupAdmin.moveServers(group.getServers(), RSGroupInfo.DEFAULT_GROUP);
232        groupAdmin.removeRSGroup(group.getName());
233      }
234    }
235  }

237  public Map<TableName, List<String>> getTableRegionMap() throws IOException {
238    Map<TableName, List<String>> map = Maps.newTreeMap();
239    Map<TableName, Map<ServerName, List<String>>> tableServerRegionMap
240        = getTableServerRegionMap();
241    for(TableName tableName : tableServerRegionMap.keySet()) {
242      if(!map.containsKey(tableName)) {
243        map.put(tableName, new LinkedList<>());
244      }
245      for(List<String> subset: tableServerRegionMap.get(tableName).values()) {
246        map.get(tableName).addAll(subset);
247      }
248    }
249    return map;
250  }
251
252  public Map<TableName, Map<ServerName, List<String>>> getTableServerRegionMap()
253      throws IOException {
254    Map<TableName, Map<ServerName, List<String>>> map = Maps.newTreeMap();
255    ClusterMetrics status = TEST_UTIL.getHBaseClusterInterface().getClusterMetrics();
256    for (Map.Entry<ServerName, ServerMetrics> entry : status.getLiveServerMetrics().entrySet()) {
257      ServerName serverName = entry.getKey();
258      for(RegionMetrics rl : entry.getValue().getRegionMetrics().values()) {
259        TableName tableName = null;
260        try {
261          tableName = RegionInfo.getTable(rl.getRegionName());
262        } catch (IllegalArgumentException e) {
263          LOG.warn("Failed parse a table name from regionname=" +
264            Bytes.toStringBinary(rl.getRegionName()));
265          continue;
266        }
267        if(!map.containsKey(tableName)) {
268          map.put(tableName, new TreeMap<>());
269        }
270        if(!map.get(tableName).containsKey(serverName)) {
271          map.get(tableName).put(serverName, new LinkedList<>());
272        }
273        map.get(tableName).get(serverName).add(rl.getNameAsString());
274      }
275    }
276    return map;
277  }

279  // return the real number of region servers, excluding the master embedded region server in 2.0+
280  public int getNumServers() throws IOException {
281    ClusterMetrics status =
282        admin.getClusterMetrics(EnumSet.of(Option.MASTER, Option.LIVE_SERVERS));
283    ServerName masterName = status.getMasterName();
284    int count = 0;
285    for (ServerName sn : status.getLiveServerMetrics().keySet()) {
286      if (!sn.equals(masterName)) {
287        count++;
288      }
289    }
290    return count;
291  }

299   * code and construct the ServerName object.
300   */
301  protected ServerName getServerName(Address addr) {
302    return TEST_UTIL.getMiniHBaseCluster().getRegionServerThreads().stream()
303      .map(t -> t.getRegionServer().getServerName()).filter(sn -> sn.getAddress().equals(addr))
304      .findFirst().get();
305  }
