041/**
042 * Manages the interactions with an already deployed distributed cluster (as opposed to
043 * a pseudo-distributed, or mini/local cluster). This is used by integration and system tests.
044 */
045@InterfaceAudience.Private
046public class DistributedHBaseCluster extends HBaseCluster {
047  private Admin admin;
048  private final Connection connection;
049
050  private ClusterManager clusterManager;
051  /**
052   * List of RegionServers killed so far. ServerName also comprises startCode of a server,
053   * so any restarted instances of the same server will have different ServerName and will not
054   * coincide with past dead ones. So there's no need to cleanup this list.
055   */
056  private Set<ServerName> killedRegionServers = new HashSet<>();
057
058  public DistributedHBaseCluster(Configuration conf, ClusterManager clusterManager)
059      throws IOException {
060    super(conf);
061    this.clusterManager = clusterManager;
062    this.connection = ConnectionFactory.createConnection(conf);
063    this.admin = this.connection.getAdmin();
064    this.initialClusterStatus = getClusterMetrics();
065  }

236  private void waitForServiceToStop(ServiceType service, ServerName serverName, long timeout)
237    throws IOException {
238    LOG.info("Waiting for service: {} to stop: {}", service, serverName.getServerName());
239    long start = EnvironmentEdgeManager.currentTime();
240
241    while ((EnvironmentEdgeManager.currentTime() - start) < timeout) {
242      if (!clusterManager.isRunning(service, serverName.getHostname(), serverName.getPort())) {
243        return;
244      }
245      Threads.sleep(100);
246    }
247    throw new IOException("Timed-out waiting for service to stop: " + serverName);
248  }
249
250  private void waitForServiceToStart(ServiceType service, ServerName serverName, long timeout)
251    throws IOException {
252    LOG.info("Waiting for service: {} to start: ", service, serverName.getServerName());
253    long start = EnvironmentEdgeManager.currentTime();
254
255    while ((EnvironmentEdgeManager.currentTime() - start) < timeout) {
256      if (clusterManager.isRunning(service, serverName.getHostname(), serverName.getPort())) {
257        return;
258      }
259      Threads.sleep(100);
260    }
261    throw new IOException("Timed-out waiting for service to start: " + serverName);
262  }

287  @Override
288  public boolean waitForActiveAndReadyMaster(long timeout) throws IOException {
289    long start = EnvironmentEdgeManager.currentTime();
290    while (EnvironmentEdgeManager.currentTime() - start < timeout) {
291      try {
292        connection.getAdmin().getClusterMetrics(EnumSet.of(ClusterMetrics.Option.HBASE_VERSION));
293        return true;
294      } catch (MasterNotRunningException m) {
295        LOG.warn("Master not started yet " + m);
296      } catch (ZooKeeperConnectionException e) {
297        LOG.warn("Failed to connect to ZK " + e);
298      }
299      Threads.sleep(1000);
300    }
301    return false;
302  }

304  @Override
305  public ServerName getServerHoldingRegion(TableName tn, byte[] regionName) throws IOException {
306    byte[] startKey = RegionInfo.getStartKey(regionName);
307    HRegionLocation regionLoc = null;
308    try (RegionLocator locator = connection.getRegionLocator(tn)) {
309      regionLoc = locator.getRegionLocation(startKey, true);
310    }
311    if (regionLoc == null) {
312      LOG.warn("Cannot find region server holding region {}", Bytes.toStringBinary(regionName));
313      return null;
314    }
315    return regionLoc.getServerName();
316  }

335  @Override
336  public boolean restoreClusterMetrics(ClusterMetrics initial) throws IOException {
337    ClusterMetrics current = getClusterMetrics();
338
339    LOG.info("Restoring cluster - started");
340
341    // do a best effort restore
342    boolean success = true;
343    success = restoreMasters(initial, current) & success;
344    success = restoreRegionServers(initial, current) & success;
345    success = restoreAdmin() & success;
346
347    LOG.info("Restoring cluster - done");
348    return success;
349  }

351  protected boolean restoreMasters(ClusterMetrics initial, ClusterMetrics current) {
352    List<IOException> deferred = new ArrayList<>();
353    //check whether current master has changed
354    final ServerName initMaster = initial.getMasterName();
355    if (!ServerName.isSameAddress(initMaster, current.getMasterName())) {
356      LOG.info("Restoring cluster - Initial active master : {} has changed to : {}",
357          initMaster.getAddress(), current.getMasterName().getAddress());
358      // If initial master is stopped, start it, before restoring the state.
359      // It will come up as a backup master, if there is already an active master.
360      try {
361        if (!clusterManager.isRunning(ServiceType.HBASE_MASTER,
362                initMaster.getHostname(), initMaster.getPort())) {
363          LOG.info("Restoring cluster - starting initial active master at:{}",
364              initMaster.getAddress());
365          startMaster(initMaster.getHostname(), initMaster.getPort());
366        }
367
368        // master has changed, we would like to undo this.
369        // 1. Kill the current backups
370        // 2. Stop current master
371        // 3. Start backup masters
372        for (ServerName currentBackup : current.getBackupMasterNames()) {
373          if (!ServerName.isSameAddress(currentBackup, initMaster)) {
374            LOG.info("Restoring cluster - stopping backup master: {}", currentBackup);
375            stopMaster(currentBackup);
376          }
377        }
378        LOG.info("Restoring cluster - stopping active master: {}", current.getMasterName());
379        stopMaster(current.getMasterName());
380        waitForActiveAndReadyMaster(); // wait so that active master takes over
381      } catch (IOException ex) {
382        // if we fail to start the initial active master, we do not want to continue stopping
383        // backup masters. Just keep what we have now
384        deferred.add(ex);
385      }
386
387      //start backup masters
388      for (ServerName backup : initial.getBackupMasterNames()) {
389        try {
390          //these are not started in backup mode, but we should already have an active master
391          if (!clusterManager.isRunning(ServiceType.HBASE_MASTER,
392                  backup.getHostname(),
393                  backup.getPort())) {
394            LOG.info("Restoring cluster - starting initial backup master: {}",
395                backup.getAddress());
396            startMaster(backup.getHostname(), backup.getPort());
397          }
398        } catch (IOException ex) {
399          deferred.add(ex);
400        }
401      }
402    } else {
403      //current master has not changed, match up backup masters
404      Set<ServerName> toStart = new TreeSet<>(new ServerNameIgnoreStartCodeComparator());
405      Set<ServerName> toKill = new TreeSet<>(new ServerNameIgnoreStartCodeComparator());
406      toStart.addAll(initial.getBackupMasterNames());
407      toKill.addAll(current.getBackupMasterNames());
408
409      for (ServerName server : current.getBackupMasterNames()) {
410        toStart.remove(server);
411      }
412      for (ServerName server: initial.getBackupMasterNames()) {
413        toKill.remove(server);
414      }
415
416      for (ServerName sn:toStart) {
417        try {
418          if(!clusterManager.isRunning(ServiceType.HBASE_MASTER, sn.getHostname(), sn.getPort())) {
419            LOG.info("Restoring cluster - starting initial backup master: {}", sn.getAddress());
420            startMaster(sn.getHostname(), sn.getPort());
421          }
422        } catch (IOException ex) {
423          deferred.add(ex);
424        }
425      }
426
427      for (ServerName sn:toKill) {
428        try {
429          if(clusterManager.isRunning(ServiceType.HBASE_MASTER, sn.getHostname(), sn.getPort())) {
430            LOG.info("Restoring cluster - stopping backup master: {}", sn.getAddress());
431            stopMaster(sn);
432          }
433        } catch (IOException ex) {
434          deferred.add(ex);
435        }
436      }
437    }
438    if (!deferred.isEmpty()) {
439      LOG.warn("Restoring cluster - restoring region servers reported {} errors:",
440          deferred.size());
441      for (int i=0; i<deferred.size() && i < 3; i++) {
442        LOG.warn(Objects.toString(deferred.get(i)));
443      }
444    }
445
446    return deferred.isEmpty();
447  }

450  private static class ServerNameIgnoreStartCodeComparator implements Comparator<ServerName> {
451    @Override
452    public int compare(ServerName o1, ServerName o2) {
453      int compare = o1.getHostname().compareToIgnoreCase(o2.getHostname());
454      if (compare != 0) return compare;
455      compare = o1.getPort() - o2.getPort();
456      if (compare != 0) return compare;
457      return 0;
458    }
459  }

461  protected boolean restoreRegionServers(ClusterMetrics initial, ClusterMetrics current) {
462    Set<ServerName> toStart = new TreeSet<>(new ServerNameIgnoreStartCodeComparator());
463    Set<ServerName> toKill = new TreeSet<>(new ServerNameIgnoreStartCodeComparator());
464    toStart.addAll(initial.getLiveServerMetrics().keySet());
465    toKill.addAll(current.getLiveServerMetrics().keySet());
466
467    ServerName master = initial.getMasterName();
468
469    for (ServerName server : current.getLiveServerMetrics().keySet()) {
470      toStart.remove(server);
471    }
472    for (ServerName server: initial.getLiveServerMetrics().keySet()) {
473      toKill.remove(server);
474    }
475
476    List<IOException> deferred = new ArrayList<>();
477
478    for(ServerName sn:toStart) {
479      try {
480        if (!clusterManager.isRunning(ServiceType.HBASE_REGIONSERVER, sn.getHostname(),
481          sn.getPort()) && master.getPort() != sn.getPort()) {
482          LOG.info("Restoring cluster - starting initial region server: {}", sn.getAddress());
483          startRegionServer(sn.getHostname(), sn.getPort());
484        }
485      } catch (IOException ex) {
486        deferred.add(ex);
487      }
488    }
489
490    for(ServerName sn:toKill) {
491      try {
492        if (clusterManager.isRunning(ServiceType.HBASE_REGIONSERVER, sn.getHostname(),
493          sn.getPort()) && master.getPort() != sn.getPort()) {
494          LOG.info("Restoring cluster - stopping initial region server: {}", sn.getAddress());
495          stopRegionServer(sn);
496        }
497      } catch (IOException ex) {
498        deferred.add(ex);
499      }
500    }
501    if (!deferred.isEmpty()) {
502      LOG.warn("Restoring cluster - restoring region servers reported {} errors:",
503          deferred.size());
504      for (int i=0; i<deferred.size() && i < 3; i++) {
505        LOG.warn(Objects.toString(deferred.get(i)));
506      }
507    }
508
509    return deferred.isEmpty();
510  }

512  protected boolean restoreAdmin() throws IOException {
513    // While restoring above, if the HBase Master which was initially the Active one, was down
514    // and the restore put the cluster back to Initial configuration, HAdmin instance will need
515    // to refresh its connections (otherwise it will return incorrect information) or we can
516    // point it to new instance.
517    admin.close();
518    this.admin = this.connection.getAdmin();
519    LOG.info("Added new HBaseAdmin");
520    return true;
521  }
