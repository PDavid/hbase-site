102  /**
103   * Sets up {@link MiniKdc} for testing security.
104   * Copied from HBaseTestingUtility#setupMiniKdc().
105   */
106  static private MiniKdc setupMiniKdc() throws Exception {
107    Properties conf = MiniKdc.createConf();
108    conf.put(MiniKdc.DEBUG, true);
109    MiniKdc kdc = null;
110    File dir = null;
111    // There is time lag between selecting a port and trying to bind with it. It's possible that
112    // another service captures the port in between which'll result in BindException.
113    boolean bindException;
114    int numTries = 0;
115    do {
116      try {
117        bindException = false;
118        dir = new File(htu.getDataTestDir("kdc").toUri().getPath());
119        kdc = new MiniKdc(conf, dir);
120        kdc.start();
121      } catch (BindException e) {
122        FileUtils.deleteDirectory(dir);  // clean directory
123        numTries++;
124        if (numTries == 3) {
125          log.error("Failed setting up MiniKDC. Tried " + numTries + " times.");
126          throw e;
127        }
128        log.error("BindException encountered when setting up MiniKdc. Trying again.");
129        bindException = true;
130      }
131    } while (bindException);
132    return kdc;
133  }

137    if (kdc != null) {
138      kdc.stop();
139    }
140

144  /**
145   * Test client command line options. Does not validate server behavior.
146   * @throws Exception if commands return unexpected results.
147   */
148  @Test(timeout=120000)
149  public void testCommandOptions() throws Exception {
150    final String className = this.getClass().getName();
151
152    assertFalse(validateCommand(new String[] {"-foo" }));
153    // fail due to insufficient number of arguments
154    assertFalse(validateCommand(new String[] {}));
155    assertFalse(validateCommand(new String[] {"-getlevel" }));
156    assertFalse(validateCommand(new String[] {"-setlevel" }));
157    assertFalse(validateCommand(new String[] {"-getlevel", "foo.bar:8080" }));
158
159    // valid command arguments
160    assertTrue(validateCommand(
161        new String[] {"-getlevel", "foo.bar:8080", className }));
162    assertTrue(validateCommand(
163        new String[] {"-setlevel", "foo.bar:8080", className, "DEBUG" }));
164    assertTrue(validateCommand(
165        new String[] {"-getlevel", "foo.bar:8080", className }));
166    assertTrue(validateCommand(
167        new String[] {"-setlevel", "foo.bar:8080", className, "DEBUG" }));
168
169    // fail due to the extra argument
170    assertFalse(validateCommand(
171        new String[] {"-getlevel", "foo.bar:8080", className, "blah" }));
172    assertFalse(validateCommand(
173        new String[] {"-setlevel", "foo.bar:8080", className, "DEBUG", "blah" }));
174    assertFalse(validateCommand(
175        new String[] {"-getlevel", "foo.bar:8080", className, "-setlevel", "foo.bar:8080",
176          className }));
177  }

179  /**
180   * Check to see if a command can be accepted.
181   *
182   * @param args a String array of arguments
183   * @return true if the command can be accepted, false if not.
184   */
185  private boolean validateCommand(String[] args) {
186    CLI cli = new CLI(clientConf);
187    try {
188      cli.parseArguments(args);
189    } catch (HadoopIllegalArgumentException e) {
190      return false;
191    } catch (Exception e) {
192      // this is used to verify the command arguments only.
193      // no HadoopIllegalArgumentException = the arguments are good.
194      return true;
195    }
196    return true;
197  }

199  /**
200   * Creates and starts a Jetty server binding at an ephemeral port to run
201   * LogLevel servlet.
202   * @param isSpnego true if SPNEGO is enabled
203   * @return a created HttpServer object
204   * @throws Exception if unable to create or start a Jetty server
205   */
206  private HttpServer createServer(boolean isSpnego)
207      throws Exception {
208    HttpServer.Builder builder = new HttpServer.Builder()
209        .setName("..")
210        .addEndpoint(new URI(PROTOCOL_HTTP + "://localhost:0"))
211        .setFindPort(true)
212        .setConf(serverConf);
213    if (isSpnego) {
214      // Set up server Kerberos credentials.
215      // Since the server may fall back to simple authentication,
216      // use ACL to make sure the connection is Kerberos/SPNEGO authenticated.
217      builder.setSecurityEnabled(true)
218          .setUsernameConfKey(PRINCIPAL)
219          .setKeytabConfKey(KEYTAB)
220          .setACL(new AccessControlList("client"));
221    }
222
223    HttpServer server = builder.build();
224    server.start();
225    return server;
226  }

233  /**
234   * Run both client and server using the given protocol.
235   *
236   * @param isSpnego true if SPNEGO is enabled
237   * @throws Exception if client can't accesss server.
238   */
239  private void testDynamicLogLevel(final boolean isSpnego, final String newLevel)
240      throws Exception {
241    Level oldLevel = log.getEffectiveLevel();
242    assertNotEquals("Get default Log Level which shouldn't be ERROR.",
243        Level.ERROR, oldLevel);
244
245    // configs needed for SPNEGO at server side
246    if (isSpnego) {
247      serverConf.set(PRINCIPAL, HTTP_PRINCIPAL);
248      serverConf.set(KEYTAB, KEYTAB_FILE.getAbsolutePath());
249      serverConf.set(CommonConfigurationKeysPublic.HADOOP_SECURITY_AUTHENTICATION, "kerberos");
250      serverConf.setBoolean(CommonConfigurationKeys.HADOOP_SECURITY_AUTHORIZATION, true);
251      UserGroupInformation.setConfiguration(serverConf);
252    } else {
253      serverConf.set(CommonConfigurationKeysPublic.HADOOP_SECURITY_AUTHENTICATION, "simple");
254      serverConf.setBoolean(CommonConfigurationKeys.HADOOP_SECURITY_AUTHORIZATION, false);
255      UserGroupInformation.setConfiguration(serverConf);
256    }
257
258    final HttpServer server = createServer(isSpnego);
259    // get server port
260    final String authority = NetUtils.getHostPortString(server.getConnectorAddress(0));
261
262    String keytabFilePath = KEYTAB_FILE.getAbsolutePath();
263
264    UserGroupInformation clientUGI = UserGroupInformation.
265        loginUserFromKeytabAndReturnUGI(clientPrincipal, keytabFilePath);
266    try {
267      clientUGI.doAs((PrivilegedExceptionAction<Void>) () -> {
268        // client command line
269        getLevel(authority);
270        setLevel(authority, newLevel);
271        return null;
272      });
273    } finally {
274      clientUGI.logoutUserFromKeytab();
275      server.stop();
276    }
277
278    // restore log level
279    GenericTestUtils.setLogLevel(log, oldLevel);
280  }

282  /**
283   * Run LogLevel command line to start a client to get log level of this test
284   * class.
285   *
286   * @param authority daemon's web UI address
287   * @throws Exception if unable to connect
288   */
289  private void getLevel(String authority) throws Exception {
290    String[] getLevelArgs = {"-getlevel", authority, logName};
291    CLI cli = new CLI(clientConf);
292    cli.run(getLevelArgs);
293  }

295  /**
296   * Run LogLevel command line to start a client to set log level of this test
297   * class to debug.
298   *
299   * @param authority daemon's web UI address
300   * @throws Exception if unable to run or log level does not change as expected
301   */
302  private void setLevel(String authority, String newLevel)
303      throws Exception {
304    String[] setLevelArgs = {"-setlevel", authority, logName, newLevel};
305    CLI cli = new CLI(clientConf);
306    cli.run(setLevelArgs);
307
308    assertEquals("new level not equal to expected: ", newLevel.toUpperCase(),
309        log.getEffectiveLevel().toString());
310  }
