198  private String getStoreFileName(TableName table, byte[] family) {
199    return Iterables
200      .getOnlyElement(Iterables.getOnlyElement(UTIL.getMiniHBaseCluster().getRegions(table))
201        .getStore(family).getStorefiles())
202      .getPath().getName();
203  }

205  @Test
206  public void testModify() throws IOException {
207    TableName tn = tableName.getTableName();
208    byte[] row = Bytes.toBytes("row");
209    byte[] family = Bytes.toBytes("family");
210    byte[] qualifier = Bytes.toBytes("qualifier");
211    byte[] value = Bytes.toBytes("value");
212    TableDescriptor td = TableDescriptorBuilder.newBuilder(tn)
213      .setColumnFamily(ColumnFamilyDescriptorBuilder.of(family)).build();
214    UTIL.getAdmin().createTable(td);
215    try (Table table = UTIL.getConnection().getTable(tn)) {
216      table.put(new Put(row).addColumn(family, qualifier, value));
217    }
218    UTIL.flush(tn);
219    String fileName = getStoreFileName(tn, family);
220
221    TableDescriptor newTd = TableDescriptorBuilder.newBuilder(td)
222      .setValue(StoreFileTrackerFactory.TRACKER_IMPL,
223        StoreFileTrackerFactory.Trackers.MIGRATION.name())
224      .setValue(MigrationStoreFileTracker.SRC_IMPL, StoreFileTrackerFactory.Trackers.DEFAULT.name())
225      .setValue(MigrationStoreFileTracker.DST_IMPL, StoreFileTrackerFactory.Trackers.FILE.name())
226      .build();
227    UTIL.getAdmin().modifyTable(newTd);
228    assertEquals(fileName, getStoreFileName(tn, family));
229    try (Table table = UTIL.getConnection().getTable(tn)) {
230      assertArrayEquals(value, table.get(new Get(row)).getValue(family, qualifier));
231    }
232
233    TableDescriptor newTd2 = TableDescriptorBuilder.newBuilder(td)
234      .setValue(StoreFileTrackerFactory.TRACKER_IMPL, StoreFileTrackerFactory.Trackers.FILE.name())
235      .build();
236    UTIL.getAdmin().modifyTable(newTd2);
237    assertEquals(fileName, getStoreFileName(tn, family));
238    try (Table table = UTIL.getConnection().getTable(tn)) {
239      assertArrayEquals(value, table.get(new Get(row)).getValue(family, qualifier));
240    }
241  }
