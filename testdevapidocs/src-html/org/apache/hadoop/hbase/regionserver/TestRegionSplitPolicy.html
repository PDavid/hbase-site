077  @Test
078  public void testForceSplitRegionWithReference() throws IOException {
079    htd.setMaxFileSize(1024L);
080    // Add a store above the requisite size. Should split.
081    HStore mockStore = Mockito.mock(HStore.class);
082    Mockito.doReturn(2000L).when(mockStore).getSize();
083    // Act as if there's a reference file or some other reason it can't split.
084    // This should prevent splitting even though it's big enough.
085    Mockito.doReturn(false).when(mockStore).canSplit();
086    stores.add(mockStore);
087
088    conf.set(HConstants.HBASE_REGION_SPLIT_POLICY_KEY,
089      ConstantSizeRegionSplitPolicy.class.getName());
090    ConstantSizeRegionSplitPolicy policy =
091        (ConstantSizeRegionSplitPolicy)RegionSplitPolicy.create(mockRegion, conf);
092    assertFalse(policy.shouldSplit());
093    Mockito.doReturn(true).when(mockRegion).shouldForceSplit();
094    assertFalse(policy.shouldSplit());
095
096    Mockito.doReturn(false).when(mockRegion).shouldForceSplit();
097    conf.set(HConstants.HBASE_REGION_SPLIT_POLICY_KEY,
098      IncreasingToUpperBoundRegionSplitPolicy.class.getName());
099    policy = (IncreasingToUpperBoundRegionSplitPolicy) RegionSplitPolicy.create(mockRegion, conf);
100    assertFalse(policy.shouldSplit());
101    Mockito.doReturn(true).when(mockRegion).shouldForceSplit();
102    assertFalse(policy.shouldSplit());
103  }

105  @Test
106  public void testIncreasingToUpperBoundRegionSplitPolicy() throws IOException {
107    // Configure IncreasingToUpperBoundRegionSplitPolicy as our split policy
108    conf.set(HConstants.HBASE_REGION_SPLIT_POLICY_KEY,
109      IncreasingToUpperBoundRegionSplitPolicy.class.getName());
110    // Now make it so the mock region has a RegionServerService that will
111    // return 'online regions'.
112    RegionServerServices rss = Mockito.mock(RegionServerServices.class);
113    final List<HRegion> regions = new ArrayList<>();
114    Mockito.doReturn(regions).when(rss).getRegions(TABLENAME);
115    Mockito.when(mockRegion.getRegionServerServices()).thenReturn(rss);
116    // Set max size for this 'table'.
117    long maxSplitSize = 1024L;
118    htd.setMaxFileSize(maxSplitSize);
119    // Set flush size to 1/8.  IncreasingToUpperBoundRegionSplitPolicy
120    // grows by the cube of the number of regions times flushsize each time.
121    long flushSize = maxSplitSize/8;
122    conf.setLong(HConstants.HREGION_MEMSTORE_FLUSH_SIZE, flushSize);
123    htd.setMemStoreFlushSize(flushSize);
124    // If RegionServerService with no regions in it -- 'online regions' == 0 --
125    // then IncreasingToUpperBoundRegionSplitPolicy should act like a
126    // ConstantSizePolicy
127    IncreasingToUpperBoundRegionSplitPolicy policy =
128      (IncreasingToUpperBoundRegionSplitPolicy)RegionSplitPolicy.create(mockRegion, conf);
129    doConstantSizePolicyTests(policy);
130
131    // Add a store in excess of split size.  Because there are "no regions"
132    // on this server -- rss.getOnlineRegions is 0 -- then we should split
133    // like a constantsizeregionsplitpolicy would
134    HStore mockStore = Mockito.mock(HStore.class);
135    Mockito.doReturn(2000L).when(mockStore).getSize();
136    Mockito.doReturn(true).when(mockStore).canSplit();
137    stores.add(mockStore);
138    // It should split
139    assertTrue(policy.shouldSplit());
140
141    // Now test that we increase our split size as online regions for a table
142    // grows. With one region, split size should be flushsize.
143    regions.add(mockRegion);
144    Mockito.doReturn(flushSize).when(mockStore).getSize();
145    // Should not split since store is flush size.
146    assertFalse(policy.shouldSplit());
147    // Set size of store to be > 2*flush size and we should split
148    Mockito.doReturn(flushSize*2 + 1).when(mockStore).getSize();
149    assertTrue(policy.shouldSplit());
150    // Add another region to the 'online regions' on this server and we should
151    // now be no longer be splittable since split size has gone up.
152    regions.add(mockRegion);
153    assertFalse(policy.shouldSplit());
154    // make sure its just over; verify it'll split
155    Mockito.doReturn((long)(maxSplitSize * 1.25 + 1)).when(mockStore).getSize();
156    assertTrue(policy.shouldSplit());
157
158    // Finally assert that even if loads of regions, we'll split at max size
159    assertWithinJitter(maxSplitSize, policy.getSizeToCheck(1000));
160    // Assert same is true if count of regions is zero.
161    assertWithinJitter(maxSplitSize, policy.getSizeToCheck(0));
162  }

164  @Test
165  public void testBusyRegionSplitPolicy() throws Exception {
166    conf.set(HConstants.HBASE_REGION_SPLIT_POLICY_KEY,
167        BusyRegionSplitPolicy.class.getName());
168    conf.setLong("hbase.busy.policy.minAge", 1000000L);
169    conf.setFloat("hbase.busy.policy.blockedRequests", 0.1f);
170
171    RegionServerServices rss  = Mockito.mock(RegionServerServices.class);
172    final List<HRegion> regions = new ArrayList<>();
173    Mockito.doReturn(regions).when(rss).getRegions(TABLENAME);
174    Mockito.when(mockRegion.getRegionServerServices()).thenReturn(rss);
175    Mockito.when(mockRegion.getBlockedRequestsCount()).thenReturn(0L);
176    Mockito.when(mockRegion.getWriteRequestsCount()).thenReturn(0L);
177
178
179    BusyRegionSplitPolicy policy =
180        (BusyRegionSplitPolicy)RegionSplitPolicy.create(mockRegion, conf);
181
182    Mockito.when(mockRegion.getBlockedRequestsCount()).thenReturn(10L);
183    Mockito.when(mockRegion.getWriteRequestsCount()).thenReturn(10L);
184    // Not enough time since region came online
185    assertFalse(policy.shouldSplit());
186
187
188    // Reset min age for split to zero
189    conf.setLong("hbase.busy.policy.minAge", 0L);
190    // Aggregate over 500 ms periods
191    conf.setLong("hbase.busy.policy.aggWindow", 500L);
192    policy =
193        (BusyRegionSplitPolicy)RegionSplitPolicy.create(mockRegion, conf);
194    long start = EnvironmentEdgeManager.currentTime();
195    Mockito.when(mockRegion.getBlockedRequestsCount()).thenReturn(10L);
196    Mockito.when(mockRegion.getWriteRequestsCount()).thenReturn(20L);
197    Thread.sleep(300);
198    assertFalse(policy.shouldSplit());
199    Mockito.when(mockRegion.getBlockedRequestsCount()).thenReturn(12L);
200    Mockito.when(mockRegion.getWriteRequestsCount()).thenReturn(30L);
201    Thread.sleep(2);
202    // Enough blocked requests since last time, but aggregate blocked request
203    // rate over last 500 ms is still low, because major portion of the window is constituted
204    // by the previous zero blocked request period which lasted at least 300 ms off last 500 ms.
205    if (EnvironmentEdgeManager.currentTime() - start < 500) {
206      assertFalse(policy.shouldSplit());
207    }
208    Mockito.when(mockRegion.getBlockedRequestsCount()).thenReturn(14L);
209    Mockito.when(mockRegion.getWriteRequestsCount()).thenReturn(40L);
210    Thread.sleep(200);
211    assertTrue(policy.shouldSplit());
212  }

214  private void assertWithinJitter(long maxSplitSize, long sizeToCheck) {
215    assertTrue("Size greater than lower bound of jitter",
216        (long)(maxSplitSize * 0.75) <= sizeToCheck);
217    assertTrue("Size less than upper bound of jitter",
218        (long)(maxSplitSize * 1.25) >= sizeToCheck);
219  }

282  /**
283   * Run through tests for a ConstantSizeRegionSplitPolicy
284   * @param policy
285   */
286  private void doConstantSizePolicyTests(final ConstantSizeRegionSplitPolicy policy) {
287    // For no stores, should not split
288    assertFalse(policy.shouldSplit());
289
290    // Add a store above the requisite size. Should split.
291    HStore mockStore = Mockito.mock(HStore.class);
292    Mockito.doReturn(2000L).when(mockStore).getSize();
293    Mockito.doReturn(true).when(mockStore).canSplit();
294    stores.add(mockStore);
295
296    assertTrue(policy.shouldSplit());
297
298    // Act as if there's a reference file or some other reason it can't split.
299    // This should prevent splitting even though it's big enough.
300    Mockito.doReturn(false).when(mockStore).canSplit();
301    assertFalse(policy.shouldSplit());
302
303    // Reset splittability after above
304    Mockito.doReturn(true).when(mockStore).canSplit();
305
306    // Set to a small size but turn on forceSplit. Should result in a split.
307    Mockito.doReturn(true).when(mockRegion).shouldForceSplit();
308    Mockito.doReturn(100L).when(mockStore).getSize();
309    assertTrue(policy.shouldSplit());
310
311    // Turn off forceSplit, should not split
312    Mockito.doReturn(false).when(mockRegion).shouldForceSplit();
313    assertFalse(policy.shouldSplit());
314
315    // Clear families we added above
316    stores.clear();
317  }

319  @Test
320  public void testGetSplitPoint() throws IOException {
321    ConstantSizeRegionSplitPolicy policy =
322      (ConstantSizeRegionSplitPolicy)RegionSplitPolicy.create(mockRegion, conf);
323
324    // For no stores, should not split
325    assertFalse(policy.shouldSplit());
326    assertNull(policy.getSplitPoint());
327
328    // Add a store above the requisite size. Should split.
329    HStore mockStore = Mockito.mock(HStore.class);
330    Mockito.doReturn(2000L).when(mockStore).getSize();
331    Mockito.doReturn(true).when(mockStore).canSplit();
332    Mockito.doReturn(Optional.of(Bytes.toBytes("store 1 split"))).when(mockStore).getSplitPoint();

334
335    assertEquals("store 1 split",
336        Bytes.toString(policy.getSplitPoint()));
337
338    // Add a bigger store. The split point should come from that one
339    HStore mockStore2 = Mockito.mock(HStore.class);
340    Mockito.doReturn(4000L).when(mockStore2).getSize();
341    Mockito.doReturn(true).when(mockStore2).canSplit();
342    Mockito.doReturn(Optional.of(Bytes.toBytes("store 2 split"))).when(mockStore2).getSplitPoint();
343    stores.add(mockStore2);
344
345    assertEquals("store 2 split",
346        Bytes.toString(policy.getSplitPoint()));
347  }

385  @Test
386  public void testConstantSizePolicyWithJitter() throws IOException {
387    conf.set(HConstants.HBASE_REGION_SPLIT_POLICY_KEY,
388      ConstantSizeRegionSplitPolicy.class.getName());
389    htd.setMaxFileSize(Long.MAX_VALUE);
390    boolean positiveJitter = false;
391    ConstantSizeRegionSplitPolicy policy = null;
392    while (!positiveJitter) {
393      policy = (ConstantSizeRegionSplitPolicy) RegionSplitPolicy.create(mockRegion, conf);
394      positiveJitter = policy.positiveJitterRate();
395    }
396    // add a store
397    HStore mockStore = Mockito.mock(HStore.class);
398    Mockito.doReturn(2000L).when(mockStore).getSize();
399    Mockito.doReturn(true).when(mockStore).canSplit();
400    stores.add(mockStore);
401    // Jitter shouldn't cause overflow when HTableDescriptor.MAX_FILESIZE set to Long.MAX_VALUE
402    assertFalse(policy.shouldSplit());
403  }
