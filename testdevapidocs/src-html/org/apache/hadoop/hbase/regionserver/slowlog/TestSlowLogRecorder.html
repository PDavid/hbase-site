087  /**
088   * confirm that for a ringbuffer of slow logs, payload on given index of buffer
089   * has expected elements
090   *
091   * @param i index of ringbuffer logs
092   * @param j data value that was put on index i
093   * @param slowLogPayloads list of payload retrieved from {@link SlowLogRecorder}
094   */
095  private void confirmPayloadParams(int i, int j, List<SlowLogPayload> slowLogPayloads) {
096
097    Assert.assertEquals(slowLogPayloads.get(i).getClientAddress(), "client_" + j);
098    Assert.assertEquals(slowLogPayloads.get(i).getUserName(), "userName_" + j);
099    Assert.assertEquals(slowLogPayloads.get(i).getServerClass(), "class_" + j);
100  }

102  @Test
103  public void testOnlieSlowLogConsumption() throws Exception {
104
105    Configuration conf = applySlowLogRecorderConf(8);
106    slowLogRecorder = new SlowLogRecorder(conf);
107    AdminProtos.SlowLogResponseRequest request =
108      AdminProtos.SlowLogResponseRequest.newBuilder().setLimit(15).build();
109
110    Assert.assertEquals(slowLogRecorder.getSlowLogPayloads(request).size(), 0);
111    LOG.debug("Initially ringbuffer of Slow Log records is empty");
112
113    int i = 0;
114
115    // add 5 records initially
116    for (; i < 5; i++) {
117      RpcLogDetails rpcLogDetails =
118        getRpcLogDetails("userName_" + (i + 1), "client_" + (i + 1), "class_" + (i + 1));
119      slowLogRecorder.addSlowLogPayload(rpcLogDetails);
120    }
121
122    Assert.assertNotEquals(-1, HBASE_TESTING_UTILITY.waitFor(3000,
123      () -> slowLogRecorder.getSlowLogPayloads(request).size() == 5));
124    List<SlowLogPayload> slowLogPayloads = slowLogRecorder.getSlowLogPayloads(request);
125    confirmPayloadParams(0, 5, slowLogPayloads);
126    confirmPayloadParams(1, 4, slowLogPayloads);
127    confirmPayloadParams(2, 3, slowLogPayloads);
128    confirmPayloadParams(3, 2, slowLogPayloads);
129    confirmPayloadParams(4, 1, slowLogPayloads);
130
131    // add 2 more records
132    for (; i < 7; i++) {
133      RpcLogDetails rpcLogDetails =
134        getRpcLogDetails("userName_" + (i + 1), "client_" + (i + 1), "class_" + (i + 1));
135      slowLogRecorder.addSlowLogPayload(rpcLogDetails);
136    }
137
138    Assert.assertNotEquals(-1, HBASE_TESTING_UTILITY.waitFor(3000,
139      () -> slowLogRecorder.getSlowLogPayloads(request).size() == 7));
140
141    slowLogPayloads = slowLogRecorder.getSlowLogPayloads(request);
142
143    Assert.assertEquals(slowLogPayloads.size(), 7);
144    confirmPayloadParams(0, 7, slowLogPayloads);
145    confirmPayloadParams(5, 2, slowLogPayloads);
146    confirmPayloadParams(6, 1, slowLogPayloads);
147
148    // add 3 more records
149    for (; i < 10; i++) {
150      RpcLogDetails rpcLogDetails =
151        getRpcLogDetails("userName_" + (i + 1), "client_" + (i + 1), "class_" + (i + 1));
152      slowLogRecorder.addSlowLogPayload(rpcLogDetails);
153    }
154
155    Assert.assertNotEquals(-1, HBASE_TESTING_UTILITY.waitFor(3000,
156      () -> slowLogRecorder.getSlowLogPayloads(request).size() == 8));
157
158    slowLogPayloads = slowLogRecorder.getSlowLogPayloads(request);
159    // confirm ringbuffer is full
160    Assert.assertEquals(slowLogPayloads.size(), 8);
161    confirmPayloadParams(7, 3, slowLogPayloads);
162    confirmPayloadParams(0, 10, slowLogPayloads);
163    confirmPayloadParams(1, 9, slowLogPayloads);
164
165    // add 4 more records
166    for (; i < 14; i++) {
167      RpcLogDetails rpcLogDetails =
168        getRpcLogDetails("userName_" + (i + 1), "client_" + (i + 1), "class_" + (i + 1));
169      slowLogRecorder.addSlowLogPayload(rpcLogDetails);
170    }
171
172    Assert.assertNotEquals(-1, HBASE_TESTING_UTILITY.waitFor(3000,
173      () -> slowLogRecorder.getSlowLogPayloads(request).size() == 8));
174
175    slowLogPayloads = slowLogRecorder.getSlowLogPayloads(request);
176    // confirm ringbuffer is full
177    Assert.assertEquals(slowLogPayloads.size(), 8);
178    confirmPayloadParams(0, 14, slowLogPayloads);
179    confirmPayloadParams(1, 13, slowLogPayloads);
180    confirmPayloadParams(2, 12, slowLogPayloads);
181    confirmPayloadParams(3, 11, slowLogPayloads);
182
183    boolean isRingBufferCleaned = slowLogRecorder.clearSlowLogPayloads();
184    Assert.assertTrue(isRingBufferCleaned);
185
186    LOG.debug("cleared the ringbuffer of Online Slow Log records");
187
188    slowLogPayloads = slowLogRecorder.getSlowLogPayloads(request);
189    // confirm ringbuffer is empty
190    Assert.assertEquals(slowLogPayloads.size(), 0);
191
192  }

194  @Test
195  public void testOnlineSlowLogWithHighRecords() throws Exception {
196
197    Configuration conf = applySlowLogRecorderConf(14);
198    slowLogRecorder = new SlowLogRecorder(conf);
199    AdminProtos.SlowLogResponseRequest request =
200      AdminProtos.SlowLogResponseRequest.newBuilder().setLimit(14 * 11).build();
201
202    Assert.assertEquals(slowLogRecorder.getSlowLogPayloads(request).size(), 0);
203    LOG.debug("Initially ringbuffer of Slow Log records is empty");
204
205    for (int i = 0; i < 14 * 11; i++) {
206      RpcLogDetails rpcLogDetails =
207        getRpcLogDetails("userName_" + (i + 1), "client_" + (i + 1), "class_" + (i + 1));
208      slowLogRecorder.addSlowLogPayload(rpcLogDetails);
209    }
210    LOG.debug("Added 14 * 11 records, ringbuffer should only provide latest 14 records");
211
212    Assert.assertNotEquals(-1, HBASE_TESTING_UTILITY.waitFor(3000,
213      () -> slowLogRecorder.getSlowLogPayloads(request).size() == 14));
214
215    List<SlowLogPayload> slowLogPayloads = slowLogRecorder.getSlowLogPayloads(request);
216    Assert.assertEquals(slowLogPayloads.size(), 14);
217
218    // confirm strict order of slow log payloads
219    confirmPayloadParams(0, 154, slowLogPayloads);
220    confirmPayloadParams(1, 153, slowLogPayloads);
221    confirmPayloadParams(2, 152, slowLogPayloads);
222    confirmPayloadParams(3, 151, slowLogPayloads);
223    confirmPayloadParams(4, 150, slowLogPayloads);
224    confirmPayloadParams(5, 149, slowLogPayloads);
225    confirmPayloadParams(6, 148, slowLogPayloads);
226    confirmPayloadParams(7, 147, slowLogPayloads);
227    confirmPayloadParams(8, 146, slowLogPayloads);
228    confirmPayloadParams(9, 145, slowLogPayloads);
229    confirmPayloadParams(10, 144, slowLogPayloads);
230    confirmPayloadParams(11, 143, slowLogPayloads);
231    confirmPayloadParams(12, 142, slowLogPayloads);
232    confirmPayloadParams(13, 141, slowLogPayloads);
233
234    boolean isRingBufferCleaned = slowLogRecorder.clearSlowLogPayloads();
235    Assert.assertTrue(isRingBufferCleaned);
236    LOG.debug("cleared the ringbuffer of Online Slow Log records");
237    slowLogPayloads = slowLogRecorder.getSlowLogPayloads(request);
238
239    // confirm ringbuffer is empty
240    Assert.assertEquals(slowLogPayloads.size(), 0);
241
242  }

338  public void testConcurrentSlowLogEvents() throws Exception {
339
340    Configuration conf = applySlowLogRecorderConf(50000);
341    slowLogRecorder = new SlowLogRecorder(conf);
342    AdminProtos.SlowLogResponseRequest request =
343      AdminProtos.SlowLogResponseRequest.newBuilder().setLimit(500000).build();

345    LOG.debug("Initially ringbuffer of Slow Log records is empty");
346
347    for (int j = 0; j < 1000; j++) {
348
349      CompletableFuture.runAsync(() -> {
350        for (int i = 0; i < 3500; i++) {
351          RpcLogDetails rpcLogDetails =
352            getRpcLogDetails("userName_" + (i + 1), "client_" + (i + 1), "class_" + (i + 1));
353          slowLogRecorder.addSlowLogPayload(rpcLogDetails);

355      });
356

358
359    Uninterruptibles.sleepUninterruptibly(500, TimeUnit.MILLISECONDS);
360
361    slowLogRecorder.clearSlowLogPayloads();
362
363    Assert.assertNotEquals(-1, HBASE_TESTING_UTILITY.waitFor(
364      4000, () -> slowLogRecorder.getSlowLogPayloads(request).size() > 10000));
365  }

520  private Message getMessage() {
521
522    i = (i + 1) % 3;
523
524    Message message = null;
525
526    switch (i) {
527
528      case 0: {
529        message = ClientProtos.ScanRequest.newBuilder()
530          .setRegion(HBaseProtos.RegionSpecifier.newBuilder()
531            .setValue(ByteString.copyFromUtf8("region1"))
532            .setType(HBaseProtos.RegionSpecifier.RegionSpecifierType.REGION_NAME)
533            .build())
534          .build();
535        break;
536      }
537      case 1: {
538        message = ClientProtos.MutateRequest.newBuilder()
539          .setRegion(HBaseProtos.RegionSpecifier.newBuilder()
540            .setValue(ByteString.copyFromUtf8("region2"))
541            .setType(HBaseProtos.RegionSpecifier.RegionSpecifierType.REGION_NAME))
542          .setMutation(ClientProtos.MutationProto.newBuilder()
543            .setRow(ByteString.copyFromUtf8("row123"))
544            .build())
545          .build();
546        break;
547      }
548      case 2: {
549        message = ClientProtos.GetRequest.newBuilder()
550          .setRegion(HBaseProtos.RegionSpecifier.newBuilder()
551            .setValue(ByteString.copyFromUtf8("region2"))
552            .setType(HBaseProtos.RegionSpecifier.RegionSpecifierType.REGION_NAME))
553          .setGet(ClientProtos.Get.newBuilder()
554            .setRow(ByteString.copyFromUtf8("row123"))
555            .build())
556          .build();
557        break;
558      }
559
560    }
561
562    return message;
563
564  }
