077  @BeforeClass
078  public static void setUp() throws Exception {
079    Configuration conf = TEST_UTIL.getConfiguration();
080    conf.setInt(HConstants.HBASE_SERVER_SCANNER_MAX_RESULT_SIZE_KEY, 4200);

086  private static void createTestData() throws IOException, InterruptedException {
087    RegionLocator locator = TEST_UTIL.getConnection().getRegionLocator(TABLE);
088    String regionName = locator.getAllRegionLocations().get(0).getRegion().getEncodedName();
089    HRegion region = TEST_UTIL.getRSForFirstRegionInTable(TABLE).getRegion(regionName);
090
091    for (int i = 1; i < 10; i++) {
092      // 5 columns per row, in 2 families
093      // Each column value is 1000 bytes, which is enough to fill a full block with row and header.
094      // So 5 blocks per row.
095      Put put = new Put(Bytes.toBytes(i));
096      for (int j = 0; j < 6; j++) {
097        put.addColumn(FAMILY1, Bytes.toBytes(j), DATA);
098      }
099
100      put.addColumn(FAMILY2, COLUMN1, DATA);
101
102      region.put(put);
103
104      if (i % 2 == 0) {
105        region.flush(true);
106      }
107    }
108
109    // we've created 10 storefiles at this point, 5 per family
110    region.flush(true);
111
112  }

114  /**
115   * Simplest test that ensures we don't count block sizes too much. These 2 requested cells are in
116   * the same block, so should be returned in 1 request. If we mis-counted blocks, it'd come in 2
117   * requests.
118   */
119  @Test
120  public void testSingleBlock() throws IOException {
121    Table table = TEST_UTIL.getConnection().getTable(TABLE);
122
123    ResultScanner scanner =
124      table.getScanner(getBaseScan().withStartRow(Bytes.toBytes(1)).withStopRow(Bytes.toBytes(2))
125        .addColumn(FAMILY1, COLUMN1).addColumn(FAMILY1, COLUMN2).setReadType(Scan.ReadType.STREAM));
126
127    ScanMetrics metrics = scanner.getScanMetrics();
128
129    scanner.next(100);
130
131    assertEquals(1, metrics.countOfRowsScanned.get());
132    assertEquals(1, metrics.countOfRPCcalls.get());
133  }

135  /**
136   * Tests that we check size limit after filterRowKey. When filterRowKey, we call nextRow to skip
137   * to next row. This should be efficient in this case, but we still need to check size limits
138   * after each row is processed. So in this test, we accumulate some block IO reading row 1, then
139   * skip row 2 and should return early at that point. The next rpc call starts with row3 blocks
140   * loaded, so can return the whole row in one rpc. If we were not checking size limits, we'd have
141   * been able to load an extra row 3 cell into the first rpc and thus split row 3 across multiple
142   * Results.
143   */
144  @Test
145  public void testCheckLimitAfterFilterRowKey() throws IOException {
146
147    Table table = TEST_UTIL.getConnection().getTable(TABLE);
148
149    ResultScanner scanner = table.getScanner(getBaseScan().addColumn(FAMILY1, COLUMN1)
150      .addColumn(FAMILY1, COLUMN2).addColumn(FAMILY1, COLUMN3).addFamily(FAMILY2)
151      .setFilter(new RowFilter(CompareOperator.NOT_EQUAL, new BinaryComparator(Bytes.toBytes(2)))));
152
153    boolean foundRow3 = false;
154    for (Result result : scanner) {
155      Set<Integer> rows = new HashSet<>();
156      for (Cell cell : result.rawCells()) {
157        rows.add(Bytes.toInt(cell.getRowArray(), cell.getRowOffset(), cell.getRowLength()));
158      }
159      if (rows.contains(3)) {
160        assertFalse("expected row3 to come all in one result, but found it in two results",
161          foundRow3);
162        assertEquals(1, rows.size());
163        foundRow3 = true;
164      }
165    }
166    ScanMetrics metrics = scanner.getScanMetrics();
167
168    // 4 blocks per row, so 36 blocks. We can scan 3 blocks per RPC, which is 12 RPCs. But we can
169    // skip 1 row, so skip 2 RPCs.
170    assertEquals(10, metrics.countOfRPCcalls.get());
171  }

173  /**
174   * After RegionScannerImpl.populateResults, row filters are run. If row is excluded due to
175   * filter.filterRow(), nextRow() is called which might accumulate more block IO. Validates that in
176   * this case we still honor block limits.
177   */
178  @Test
179  public void testCheckLimitAfterFilteringRowCellsDueToFilterRow() throws IOException {
180    Table table = TEST_UTIL.getConnection().getTable(TABLE);
181
182    ResultScanner scanner = table.getScanner(getBaseScan().withStartRow(Bytes.toBytes(1), true)
183      .addColumn(FAMILY1, COLUMN1).addColumn(FAMILY1, COLUMN2).setReadType(Scan.ReadType.STREAM)
184      .setFilter(new SkipFilter(new QualifierFilter(CompareOperator.EQUAL,
185        new BinaryComparator(Bytes.toBytes("dasfasf"))))));
186
187    // Our filter doesn't end up matching any real columns, so expect only cursors
188    for (Result result : scanner) {
189      assertTrue(result.isCursor());
190    }
191
192    ScanMetrics metrics = scanner.getScanMetrics();
193
194    // scanning over 9 rows, filtering on 2 contiguous columns each, so 9 blocks total
195    // limited to 4200 bytes per which is enough for 3 blocks (exceed limit after loading 3rd)
196    // so that's 3 RPC and the last RPC pulls the cells loaded by the last block
197    assertEquals(4, metrics.countOfRPCcalls.get());
198  }

200  /**
201   * At the end of the loop in StoreScanner, we do one more check of size limits. This is to catch
202   * block size being exceeded while filtering cells within a store. Test to ensure that we do that,
203   * otherwise we'd see no cursors below.
204   */
205  @Test
206  public void testCheckLimitAfterFilteringCell() throws IOException {
207    Table table = TEST_UTIL.getConnection().getTable(TABLE);
208
209    ResultScanner scanner = table.getScanner(getBaseScan()
210      .setFilter(new QualifierFilter(CompareOperator.EQUAL, new BinaryComparator(COLUMN2))));
211
212    int cursors = 0;
213    for (Result result : scanner) {
214      if (result.isCursor()) {
215        cursors++;
216      }
217    }
218    ScanMetrics metrics = scanner.getScanMetrics();
219
220    // We will return 9 rows, but also 2 cursors because we exceed the scan size limit partway
221    // through. So that accounts for 11 rpcs.
222    assertEquals(2, cursors);
223    assertEquals(11, metrics.countOfRPCcalls.get());
224  }

226  /**
227   * After RegionScannerImpl.populateResults, row filters are run. If row is excluded due to
228   * filter.filterRowCells(), we fall through to a final results.isEmpty() check near the end of the
229   * method. If results are empty at this point (which they are), nextRow() is called which might
230   * accumulate more block IO. Validates that in this case we still honor block limits.
231   */
232  @Test
233  public void testCheckLimitAfterFilteringRowCells() throws IOException {
234    Table table = TEST_UTIL.getConnection().getTable(TABLE);
235
236    ResultScanner scanner = table
237      .getScanner(getBaseScan().withStartRow(Bytes.toBytes(1), true).addColumn(FAMILY1, COLUMN1)
238        .setReadType(Scan.ReadType.STREAM).setFilter(new SingleColumnValueExcludeFilter(FAMILY1,
239          COLUMN1, CompareOperator.EQUAL, new BinaryComparator(DATA))));
240
241    // Since we use SingleColumnValueExcludeFilter and dont include any other columns, the column
242    // we load to test ends up being excluded from the result. So we only expect cursors here.
243    for (Result result : scanner) {
244      assertTrue(result.isCursor());
245    }
246
247    ScanMetrics metrics = scanner.getScanMetrics();
248
249    // Our filter causes us to read the first column of each row, then INCLUDE_AND_SEEK_NEXT_ROW.
250    // So we load 1 block per row, and there are 9 rows. Our max scan size is large enough to
251    // return 2 full blocks, with some overflow. So we are able to squeeze this all into 4 RPCs.
252    assertEquals(4, metrics.countOfRPCcalls.get());
253  }

255  /**
256   * Tests that when we seek over blocks we dont include them in the block size of the request
257   */
258  @Test
259  public void testSeekNextUsingHint() throws IOException {
260    Table table = TEST_UTIL.getConnection().getTable(TABLE);
261
262    ResultScanner scanner = table.getScanner(
263      getBaseScan().addFamily(FAMILY1).setFilter(new ColumnPaginationFilter(1, COLUMN5)));
264
265    scanner.next(100);
266    ScanMetrics metrics = scanner.getScanMetrics();
267
268    // We have to read the first cell/block of each row, then can skip to the last block. So that's
269    // 2 blocks per row to read (18 total). Our max scan size is enough to read 3 blocks per RPC,
270    // plus one final RPC to finish region.
271    assertEquals(7, metrics.countOfRPCcalls.get());
272  }

274  /**
275   * We enable cursors and partial results to give us more granularity over counting of results, and
276   * we enable STREAM so that no auto switching from pread to stream occurs -- this throws off the
277   * rpc counts.
278   */
279  private Scan getBaseScan() {
280    return new Scan().setScanMetricsEnabled(true).setNeedCursorResult(true)
281      .setAllowPartialResults(true).setReadType(Scan.ReadType.STREAM);
282  }
