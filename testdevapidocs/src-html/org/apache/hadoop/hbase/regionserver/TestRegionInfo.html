100  public void testIsOverlap() {
101    byte [] a = Bytes.toBytes("a");
102    byte [] b = Bytes.toBytes("b");
103    byte [] c = Bytes.toBytes("c");
104    byte [] d = Bytes.toBytes("d");
105    org.apache.hadoop.hbase.client.RegionInfo all =
106        RegionInfoBuilder.FIRST_META_REGIONINFO;
107    org.apache.hadoop.hbase.client.RegionInfo ari =
108        org.apache.hadoop.hbase.client.RegionInfoBuilder.newBuilder(TableName.META_TABLE_NAME).
109            setEndKey(a).build();
110    org.apache.hadoop.hbase.client.RegionInfo abri =
111        org.apache.hadoop.hbase.client.RegionInfoBuilder.newBuilder(TableName.META_TABLE_NAME).
112            setStartKey(a).setEndKey(b).build();
113    org.apache.hadoop.hbase.client.RegionInfo adri =
114        org.apache.hadoop.hbase.client.RegionInfoBuilder.newBuilder(TableName.META_TABLE_NAME).
115            setStartKey(a).setEndKey(d).build();
116    org.apache.hadoop.hbase.client.RegionInfo cdri =
117        org.apache.hadoop.hbase.client.RegionInfoBuilder.newBuilder(TableName.META_TABLE_NAME).
118            setStartKey(c).setEndKey(d).build();
119    org.apache.hadoop.hbase.client.RegionInfo dri =
120        org.apache.hadoop.hbase.client.RegionInfoBuilder.newBuilder(TableName.META_TABLE_NAME).
121            setStartKey(d).build();
122    assertTrue(all.isOverlap(all));
123    assertTrue(all.isOverlap(abri));
124    assertFalse(abri.isOverlap(cdri));
125    assertTrue(all.isOverlap(ari));
126    assertFalse(ari.isOverlap(abri));
127    assertFalse(ari.isOverlap(abri));
128    assertTrue(ari.isOverlap(all));
129    assertTrue(dri.isOverlap(all));
130    assertTrue(abri.isOverlap(adri));
131    assertFalse(dri.isOverlap(ari));
132    assertTrue(abri.isOverlap(adri));
133    assertTrue(adri.isOverlap(abri));
134  }

176  @Test
177  public void testReadAndWriteHRegionInfoFile() throws IOException, InterruptedException {
178    HBaseTestingUtility htu = new HBaseTestingUtility();
179    RegionInfo hri = RegionInfoBuilder.FIRST_META_REGIONINFO;
180    Path basedir = htu.getDataTestDir();
181    // Create a region.  That'll write the .regioninfo file.
182    FSTableDescriptors fsTableDescriptors = new FSTableDescriptors(htu.getConfiguration());
183    FSTableDescriptors.tryUpdateMetaTableDescriptor(htu.getConfiguration());
184    HRegion r = HBaseTestingUtility.createRegionAndWAL(hri, basedir, htu.getConfiguration(),
185        fsTableDescriptors.get(TableName.META_TABLE_NAME));
186    // Get modtime on the file.
187    long modtime = getModTime(r);
188    HBaseTestingUtility.closeRegionAndWAL(r);
189    Thread.sleep(1001);
190    r = HRegion.openHRegion(basedir, hri, fsTableDescriptors.get(TableName.META_TABLE_NAME),
191        null, htu.getConfiguration());
192    // Ensure the file is not written for a second time.
193    long modtime2 = getModTime(r);
194    assertEquals(modtime, modtime2);
195    // Now load the file.
196    org.apache.hadoop.hbase.client.RegionInfo deserializedHri =
197      HRegionFileSystem.loadRegionInfoFileContent(
198        r.getRegionFileSystem().getFileSystem(), r.getRegionFileSystem().getRegionDir());
199    assertEquals(0,
200      org.apache.hadoop.hbase.client.RegionInfo.COMPARATOR.compare(hri, deserializedHri));
201    HBaseTestingUtility.closeRegionAndWAL(r);
202  }

205    FileStatus[] statuses = r.getRegionFileSystem().getFileSystem().listStatus(

235  @Test
236  public void testContainsRange() {
237    TableDescriptor tableDesc =
238      TableDescriptorBuilder.newBuilder(TableName.valueOf(name.getMethodName())).build();
239    RegionInfo hri = RegionInfoBuilder.newBuilder(tableDesc.getTableName())
240      .setStartKey(Bytes.toBytes("a")).setEndKey(Bytes.toBytes("g")).build();
241    // Single row range at start of region
242    assertTrue(hri.containsRange(Bytes.toBytes("a"), Bytes.toBytes("a")));
243    // Fully contained range
244    assertTrue(hri.containsRange(Bytes.toBytes("b"), Bytes.toBytes("c")));
245    // Range overlapping start of region
246    assertTrue(hri.containsRange(Bytes.toBytes("a"), Bytes.toBytes("c")));
247    // Fully contained single-row range
248    assertTrue(hri.containsRange(Bytes.toBytes("c"), Bytes.toBytes("c")));
249    // Range that overlaps end key and hence doesn't fit
250    assertFalse(hri.containsRange(Bytes.toBytes("a"), Bytes.toBytes("g")));
251    // Single row range on end key
252    assertFalse(hri.containsRange(Bytes.toBytes("g"), Bytes.toBytes("g")));
253    // Single row range entirely outside
254    assertFalse(hri.containsRange(Bytes.toBytes("z"), Bytes.toBytes("z")));
255
256    // Degenerate range
257    try {
258      hri.containsRange(Bytes.toBytes("z"), Bytes.toBytes("a"));
259      fail("Invalid range did not throw IAE");
260    } catch (IllegalArgumentException iae) {
261    }
262  }

409  @Test
410  public void testRegionDetailsForDisplay() throws IOException {
411    byte[] startKey = new byte[] {0x01, 0x01, 0x02, 0x03};
412    byte[] endKey = new byte[] {0x01, 0x01, 0x02, 0x04};
413    Configuration conf = new Configuration();
414    conf.setBoolean("hbase.display.keys", false);
415    RegionInfo h = RegionInfoBuilder.newBuilder(TableName.valueOf(name.getMethodName()))
416      .setStartKey(startKey).setEndKey(endKey).build();
417    checkEquality(h, conf);
418    // check HRIs with non-default replicaId
419    h = RegionInfoBuilder.newBuilder(TableName.valueOf(name.getMethodName())).setStartKey(startKey)
420      .setEndKey(endKey).setRegionId(System.currentTimeMillis()).setReplicaId(1).build();
421    checkEquality(h, conf);
422    assertArrayEquals(RegionInfoDisplay.HIDDEN_END_KEY,
423      RegionInfoDisplay.getEndKeyForDisplay(h, conf));
424    assertArrayEquals(RegionInfoDisplay.HIDDEN_START_KEY,
425      RegionInfoDisplay.getStartKeyForDisplay(h, conf));
426
427    RegionState state = RegionState.createForTesting(h, RegionState.State.OPEN);
428    String descriptiveNameForDisplay =
429      RegionInfoDisplay.getDescriptiveNameFromRegionStateForDisplay(state, conf);
430    checkDescriptiveNameEquality(descriptiveNameForDisplay, state.toDescriptiveString(), startKey);
431
432    conf.setBoolean("hbase.display.keys", true);
433    assertArrayEquals(endKey, RegionInfoDisplay.getEndKeyForDisplay(h, conf));
434    assertArrayEquals(startKey, RegionInfoDisplay.getStartKeyForDisplay(h, conf));
435    assertEquals(state.toDescriptiveString(),
436      RegionInfoDisplay.getDescriptiveNameFromRegionStateForDisplay(state, conf));
437  }

439  private void checkDescriptiveNameEquality(String descriptiveNameForDisplay, String origDesc,
440    byte[] startKey) {
441    // except for the "hidden-start-key" substring everything else should exactly match
442    String firstPart = descriptiveNameForDisplay.substring(0,
443      descriptiveNameForDisplay.indexOf(new String(RegionInfoDisplay.HIDDEN_START_KEY)));
444    String secondPart = descriptiveNameForDisplay
445      .substring(descriptiveNameForDisplay.indexOf(new String(RegionInfoDisplay.HIDDEN_START_KEY)) +
446        RegionInfoDisplay.HIDDEN_START_KEY.length);
447    String firstPartOrig = origDesc.substring(0, origDesc.indexOf(Bytes.toStringBinary(startKey)));
448    String secondPartOrig = origDesc.substring(
449      origDesc.indexOf(Bytes.toStringBinary(startKey)) + Bytes.toStringBinary(startKey).length());
450    assert (firstPart.equals(firstPartOrig));
451    assert (secondPart.equals(secondPartOrig));
452  }
453
454  private void checkEquality(RegionInfo h, Configuration conf) throws IOException {
455    byte[] modifiedRegionName = RegionInfoDisplay.getRegionNameForDisplay(h, conf);
456    byte[][] modifiedRegionNameParts = RegionInfo.parseRegionName(modifiedRegionName);
457    byte[][] regionNameParts = RegionInfo.parseRegionName(h.getRegionName());
458
459    // same number of parts
460    assert (modifiedRegionNameParts.length == regionNameParts.length);
461
462    for (int i = 0; i < regionNameParts.length; i++) {
463      // all parts should match except for [1] where in the modified one,
464      // we should have "hidden_start_key"
465      if (i != 1) {
466        Assert.assertArrayEquals(regionNameParts[i], modifiedRegionNameParts[i]);
467      } else {
468        assertNotEquals(regionNameParts[i][0], modifiedRegionNameParts[i][0]);
469        Assert.assertArrayEquals(modifiedRegionNameParts[1],
470          RegionInfoDisplay.getStartKeyForDisplay(h, conf));
471      }
472    }
473  }
