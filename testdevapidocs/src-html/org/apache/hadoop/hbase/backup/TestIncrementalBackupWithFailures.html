060      HBaseClassTestRule.forClass(TestIncrementalBackupWithFailures.class);

069    params.add(new Object[] { Boolean.TRUE });

073  public TestIncrementalBackupWithFailures(Boolean b) {
074  }

076  // implement all test cases in 1 test since incremental backup/restore has dependencies
077  @Test
078  public void testIncBackupRestore() throws Exception {
079
080    int ADD_ROWS = 99;
081    // #1 - create full backup for all tables
082    LOG.info("create full backup image for all tables");
083
084    List<TableName> tables = Lists.newArrayList(table1, table2);
085    final byte[] fam3Name = Bytes.toBytes("f3");
086    table1Desc.addFamily(new HColumnDescriptor(fam3Name));
087    HBaseTestingUtility.modifyTableSync(TEST_UTIL.getAdmin(), table1Desc);
088
089    Connection conn = ConnectionFactory.createConnection(conf1);
090    int NB_ROWS_FAM3 = 6;
091    insertIntoTable(conn, table1, fam3Name, 3, NB_ROWS_FAM3).close();
092
093    Admin admin = conn.getAdmin();
094    BackupAdminImpl client = new BackupAdminImpl(conn);
095
096    BackupRequest request = createBackupRequest(BackupType.FULL, tables, BACKUP_ROOT_DIR);
097    String backupIdFull = client.backupTables(request);
098
099    assertTrue(checkSucceeded(backupIdFull));
100
101    // #2 - insert some data to table
102    Table t1 = insertIntoTable(conn, table1, famName, 1, ADD_ROWS);
103    LOG.debug("writing " + ADD_ROWS + " rows to " + table1);
104
105    Assert.assertEquals(TEST_UTIL.countRows(t1), NB_ROWS_IN_BATCH + ADD_ROWS + NB_ROWS_FAM3);
106    t1.close();
107    LOG.debug("written " + ADD_ROWS + " rows to " + table1);
108
109    Table t2 = conn.getTable(table2);
110    Put p2;
111    for (int i = 0; i < 5; i++) {
112      p2 = new Put(Bytes.toBytes("row-t2" + i));
113      p2.addColumn(famName, qualName, Bytes.toBytes("val" + i));
114      t2.put(p2);
115    }
116
117    Assert.assertEquals(TEST_UTIL.countRows(t2), NB_ROWS_IN_BATCH + 5);
118    t2.close();
119    LOG.debug("written " + 5 + " rows to " + table2);
120
121    // #3 - incremental backup for multiple tables
122    incrementalBackupWithFailures();
123
124    admin.close();
125    conn.close();
126
127  }
128
129
130  private void incrementalBackupWithFailures() throws Exception {
131    conf1.set(TableBackupClient.BACKUP_CLIENT_IMPL_CLASS,
132      IncrementalTableBackupClientForTest.class.getName());
133    int maxStage = Stage.values().length -1;
134    // Fail stages between 0 and 4 inclusive
135    for (int stage = 0; stage <= maxStage; stage++) {
136      LOG.info("Running stage " + stage);
137      runBackupAndFailAtStage(stage);
138    }
139  }
140
141  private void runBackupAndFailAtStage(int stage) throws Exception {
142
143    conf1.setInt(FullTableBackupClientForTest.BACKUP_TEST_MODE_STAGE, stage);
144    try (BackupSystemTable table = new BackupSystemTable(TEST_UTIL.getConnection())) {
145      int before = table.getBackupHistory().size();
146      String[] args =
147          new String[] { "create", "incremental", BACKUP_ROOT_DIR, "-t",
148              table1.getNameAsString() + "," + table2.getNameAsString() };
149      // Run backup
150      int ret = ToolRunner.run(conf1, new BackupDriver(), args);
151      assertFalse(ret == 0);
152      List<BackupInfo> backups = table.getBackupHistory();
153      int after = table.getBackupHistory().size();
154
155      assertTrue(after ==  before +1);
156      for (BackupInfo data : backups) {
157        if(data.getType() == BackupType.FULL) {
158          assertTrue(data.getState() == BackupState.COMPLETE);
159        } else {
160          assertTrue(data.getState() == BackupState.FAILED);
161        }
162      }
163    }
164  }
