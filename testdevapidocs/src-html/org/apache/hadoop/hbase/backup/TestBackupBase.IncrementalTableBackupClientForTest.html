110  static class IncrementalTableBackupClientForTest extends IncrementalTableBackupClient {
111    public IncrementalTableBackupClientForTest() {
112    }
113
114    public IncrementalTableBackupClientForTest(Connection conn,
115        String backupId, BackupRequest request) throws IOException {
116      super(conn, backupId, request);
117    }
118
119    @Override
120    public void execute() throws IOException {
121      // case INCREMENTAL_COPY:
122      try {
123        // case PREPARE_INCREMENTAL:
124        failStageIf(Stage.stage_0);
125        beginBackup(backupManager, backupInfo);
126
127        failStageIf(Stage.stage_1);
128        backupInfo.setPhase(BackupPhase.PREPARE_INCREMENTAL);
129        LOG.debug("For incremental backup, current table set is "
130            + backupManager.getIncrementalBackupTableSet());
131        newTimestamps = ((IncrementalBackupManager) backupManager).getIncrBackupLogFileMap();
132        // copy out the table and region info files for each table
133        BackupUtils.copyTableRegionInfo(conn, backupInfo, conf);
134        // convert WAL to HFiles and copy them to .tmp under BACKUP_ROOT
135        convertWALsToHFiles();
136        incrementalCopyHFiles(new String[] {getBulkOutputDir().toString()},
137          backupInfo.getBackupRootDir());
138        failStageIf(Stage.stage_2);
139        // Save list of WAL files copied
140        backupManager.recordWALFiles(backupInfo.getIncrBackupFileList());
141
142        // case INCR_BACKUP_COMPLETE:
143        // set overall backup status: complete. Here we make sure to complete the backup.
144        // After this checkpoint, even if entering cancel process, will let the backup finished
145        // Set the previousTimestampMap which is before this current log roll to the manifest.
146        HashMap<TableName, HashMap<String, Long>> previousTimestampMap =
147            backupManager.readLogTimestampMap();
148        backupInfo.setIncrTimestampMap(previousTimestampMap);
149
150        // The table list in backupInfo is good for both full backup and incremental backup.
151        // For incremental backup, it contains the incremental backup table set.
152        backupManager.writeRegionServerLogTimestamp(backupInfo.getTables(), newTimestamps);
153        failStageIf(Stage.stage_3);
154
155        HashMap<TableName, HashMap<String, Long>> newTableSetTimestampMap =
156            backupManager.readLogTimestampMap();
157
158        Long newStartCode =
159            BackupUtils.getMinValue(BackupUtils.getRSLogTimestampMins(newTableSetTimestampMap));
160        backupManager.writeBackupStartCode(newStartCode);
161
162        handleBulkLoad(backupInfo.getTableNames());
163        failStageIf(Stage.stage_4);
164
165        // backup complete
166        completeBackup(conn, backupInfo, backupManager, BackupType.INCREMENTAL, conf);
167
168      } catch (Exception e) {
169        failBackup(conn, backupInfo, backupManager, e, "Unexpected Exception : ",
170          BackupType.INCREMENTAL, conf);
171        throw new IOException(e);
172      }
173    }
174  }

176  static class FullTableBackupClientForTest extends FullTableBackupClient {
177    public FullTableBackupClientForTest() {
178    }
179
180    public FullTableBackupClientForTest(Connection conn, String backupId, BackupRequest request)
181        throws IOException {
182      super(conn, backupId, request);
183    }
184
185    @Override
186    public void execute() throws IOException {
187      // Get the stage ID to fail on
188      try (Admin admin = conn.getAdmin()) {
189        // Begin BACKUP
190        beginBackup(backupManager, backupInfo);
191        failStageIf(Stage.stage_0);
192        String savedStartCode;
193        boolean firstBackup;
194        // do snapshot for full table backup
195        savedStartCode = backupManager.readBackupStartCode();
196        firstBackup = savedStartCode == null || Long.parseLong(savedStartCode) == 0L;
197        if (firstBackup) {
198          // This is our first backup. Let's put some marker to system table so that we can hold the
199          // logs while we do the backup.
200          backupManager.writeBackupStartCode(0L);
201        }
202        failStageIf(Stage.stage_1);
203        // We roll log here before we do the snapshot. It is possible there is duplicate data
204        // in the log that is already in the snapshot. But if we do it after the snapshot, we
205        // could have data loss.
206        // A better approach is to do the roll log on each RS in the same global procedure as
207        // the snapshot.
208        LOG.info("Execute roll log procedure for full backup ...");
209
210        Map<String, String> props = new HashMap<>();
211        props.put("backupRoot", backupInfo.getBackupRootDir());
212        admin.execProcedure(LogRollMasterProcedureManager.ROLLLOG_PROCEDURE_SIGNATURE,
213          LogRollMasterProcedureManager.ROLLLOG_PROCEDURE_NAME, props);
214        failStageIf(Stage.stage_2);
215        newTimestamps = backupManager.readRegionServerLastLogRollResult();
216        if (firstBackup) {
217          // Updates registered log files
218          // We record ALL old WAL files as registered, because
219          // this is a first full backup in the system and these
220          // files are not needed for next incremental backup
221          List<String> logFiles = BackupUtils.getWALFilesOlderThan(conf, newTimestamps);
222          backupManager.recordWALFiles(logFiles);
223        }
224
225        // SNAPSHOT_TABLES:
226        backupInfo.setPhase(BackupPhase.SNAPSHOT);
227        for (TableName tableName : tableList) {
228          String snapshotName =
229              "snapshot_" + Long.toString(EnvironmentEdgeManager.currentTime()) + "_"
230                  + tableName.getNamespaceAsString() + "_" + tableName.getQualifierAsString();
231
232          snapshotTable(admin, tableName, snapshotName);
233          backupInfo.setSnapshotName(tableName, snapshotName);
234        }
235        failStageIf(Stage.stage_3);
236        // SNAPSHOT_COPY:
237        // do snapshot copy
238        LOG.debug("snapshot copy for " + backupId);
239        snapshotCopy(backupInfo);
240        // Updates incremental backup table set
241        backupManager.addIncrementalBackupTableSet(backupInfo.getTables());
242
243        // BACKUP_COMPLETE:
244        // set overall backup status: complete. Here we make sure to complete the backup.
245        // After this checkpoint, even if entering cancel process, will let the backup finished
246        backupInfo.setState(BackupState.COMPLETE);
247        // The table list in backupInfo is good for both full backup and incremental backup.
248        // For incremental backup, it contains the incremental backup table set.
249        backupManager.writeRegionServerLogTimestamp(backupInfo.getTables(), newTimestamps);
250
251        HashMap<TableName, HashMap<String, Long>> newTableSetTimestampMap =
252            backupManager.readLogTimestampMap();
253
254        Long newStartCode =
255            BackupUtils.getMinValue(BackupUtils
256                .getRSLogTimestampMins(newTableSetTimestampMap));
257        backupManager.writeBackupStartCode(newStartCode);
258        failStageIf(Stage.stage_4);
259        // backup complete
260        completeBackup(conn, backupInfo, backupManager, BackupType.FULL, conf);
261
262      } catch (Exception e) {
263
264        if(autoRestoreOnFailure) {
265          failBackup(conn, backupInfo, backupManager, e, "Unexpected BackupException : ",
266            BackupType.FULL, conf);
267        }
268        throw new IOException(e);
269      }
270    }

276  @Before
277  public void setUp() throws Exception {
278    if (setupIsDone) {
279      return;
280    }
281    if (secure) {
282      // set the always on security provider
283      UserProvider.setUserProviderForTesting(TEST_UTIL.getConfiguration(),
284          HadoopSecurityEnabledUserProviderForTesting.class);
285      // setup configuration
286      SecureTestUtil.enableSecurity(TEST_UTIL.getConfiguration());
287    }
288    conf1.setBoolean(BackupRestoreConstants.BACKUP_ENABLE_KEY, true);
289    BackupManager.decorateMasterConfiguration(conf1);
290    BackupManager.decorateRegionServerConfiguration(conf1);
291    conf1.set(HConstants.ZOOKEEPER_ZNODE_PARENT, "/1");
292    // Set TTL for old WALs to 1 sec to enforce fast cleaning of an archived
293    // WAL files
294    conf1.setLong(TimeToLiveLogCleaner.TTL_CONF_KEY, 1000);
295    conf1.setLong(LogCleaner.OLD_WALS_CLEANER_THREAD_TIMEOUT_MSEC, 1000);
296
297    // Set MultiWAL (with 2 default WAL files per RS)
298    conf1.set(WALFactory.WAL_PROVIDER, provider);
299    TEST_UTIL.startMiniCluster();
300
301    if (useSecondCluster) {
302      conf2 = HBaseConfiguration.create(conf1);
303      conf2.set(HConstants.ZOOKEEPER_ZNODE_PARENT, "/2");
304      TEST_UTIL2 = new HBaseTestingUtility(conf2);
305      TEST_UTIL2.setZkCluster(TEST_UTIL.getZkCluster());
306      TEST_UTIL2.startMiniDFSCluster(3);
307      String root2 = TEST_UTIL2.getConfiguration().get("fs.defaultFS");
308      Path p = new Path(new Path(root2), "/tmp/wal");
309      CommonFSUtils.setWALRootDir(TEST_UTIL2.getConfiguration(), p);
310      TEST_UTIL2.startMiniCluster();
311    }
312    conf1 = TEST_UTIL.getConfiguration();
313
314    TEST_UTIL.startMiniMapReduceCluster();
315    BACKUP_ROOT_DIR =
316        new Path(new Path(TEST_UTIL.getConfiguration().get("fs.defaultFS")),
317          BACKUP_ROOT_DIR).toString();
318    LOG.info("ROOTDIR " + BACKUP_ROOT_DIR);
319    if (useSecondCluster) {
320      BACKUP_REMOTE_ROOT_DIR =
321          new Path(new Path(TEST_UTIL2.getConfiguration().get("fs.defaultFS"))
322          + BACKUP_REMOTE_ROOT_DIR).toString();
323      LOG.info("REMOTE ROOTDIR " + BACKUP_REMOTE_ROOT_DIR);
324    }
325    createTables();
326    populateFromMasterConfig(TEST_UTIL.getHBaseCluster().getMaster().getConfiguration(), conf1);
327    setupIsDone = true;
328  }

330  private static void populateFromMasterConfig(Configuration masterConf, Configuration conf) {
331    Iterator<Entry<String, String>> it = masterConf.iterator();
332    while (it.hasNext()) {
333      Entry<String, String> e = it.next();
334      conf.set(e.getKey(), e.getValue());
335    }
336  }

355  Table insertIntoTable(Connection conn, TableName table, byte[] family, int id, int numRows)
356      throws IOException {
357    Table t = conn.getTable(table);
358    Put p1;
359    for (int i = 0; i < numRows; i++) {
360      p1 = new Put(Bytes.toBytes("row-" + table + "-" + id + "-" + i));
361      p1.addColumn(family, qualName, Bytes.toBytes("val" + i));
362      t.put(p1);
363    }
364    return t;
365  }

367  protected BackupRequest createBackupRequest(BackupType type,
368      List<TableName> tables, String path) {
369    BackupRequest.Builder builder = new BackupRequest.Builder();
370    BackupRequest request = builder.withBackupType(type)
371                                    .withTableList(tables)
372                                    .withTargetRootDir(path).build();
373    return request;
374  }

376  protected String backupTables(BackupType type, List<TableName> tables, String path)
377      throws IOException {
378    Connection conn = null;
379    BackupAdmin badmin = null;
380    String backupId;
381    try {
382      conn = ConnectionFactory.createConnection(conf1);
383      badmin = new BackupAdminImpl(conn);
384      BackupRequest request = createBackupRequest(type, tables, path);
385      backupId = badmin.backupTables(request);
386    } finally {
387      if (badmin != null) {
388        badmin.close();
389      }
390      if (conn != null) {
391        conn.close();
392      }
393    }
394    return backupId;
395  }

405  protected static void loadTable(Table table) throws Exception {
406    Put p; // 100 + 1 row to t1_syncup
407    for (int i = 0; i < NB_ROWS_IN_BATCH; i++) {
408      p = new Put(Bytes.toBytes("row" + i));
409      p.setDurability(Durability.SKIP_WAL);
410      p.addColumn(famName, qualName, Bytes.toBytes("val" + i));
411      table.put(p);
412    }
413  }

415  protected static void createTables() throws Exception {
416    long tid = System.currentTimeMillis();
417    table1 = TableName.valueOf("test-" + tid);
418    Admin ha = TEST_UTIL.getAdmin();
419
420    // Create namespaces
421    NamespaceDescriptor desc1 = NamespaceDescriptor.create("ns1").build();
422    NamespaceDescriptor desc2 = NamespaceDescriptor.create("ns2").build();
423    NamespaceDescriptor desc3 = NamespaceDescriptor.create("ns3").build();
424    NamespaceDescriptor desc4 = NamespaceDescriptor.create("ns4").build();
425
426    ha.createNamespace(desc1);
427    ha.createNamespace(desc2);
428    ha.createNamespace(desc3);
429    ha.createNamespace(desc4);
430
431    TableDescriptor desc = TableDescriptorBuilder.newBuilder(table1)
432      .setColumnFamily(ColumnFamilyDescriptorBuilder.of(famName)).build();
433    ha.createTable(desc);
434    table1Desc = desc;
435    Connection conn = ConnectionFactory.createConnection(conf1);
436    Table table = conn.getTable(table1);
437    loadTable(table);
438    table.close();
439    table2 = TableName.valueOf("ns2:test-" + tid + 1);
440    desc = TableDescriptorBuilder.newBuilder(table2)
441      .setColumnFamily(ColumnFamilyDescriptorBuilder.of(famName)).build();
442    ha.createTable(desc);
443    table = conn.getTable(table2);
444    loadTable(table);
445    table.close();
446    table3 = TableName.valueOf("ns3:test-" + tid + 2);
447    table = TEST_UTIL.createTable(table3, famName);
448    table.close();
449    table4 = TableName.valueOf("ns4:test-" + tid + 3);
450    table = TEST_UTIL.createTable(table4, famName);
451    table.close();
452    ha.close();
453    conn.close();
454  }

456  protected boolean checkSucceeded(String backupId) throws IOException {
457    BackupInfo status = getBackupInfo(backupId);
458
459    if (status == null) {
460      return false;
461    }
462
463    return status.getState() == BackupState.COMPLETE;
464  }
465
466  protected boolean checkFailed(String backupId) throws IOException {
467    BackupInfo status = getBackupInfo(backupId);
468
469    if (status == null) {
470      return false;
471    }
472
473    return status.getState() == BackupState.FAILED;
474  }
