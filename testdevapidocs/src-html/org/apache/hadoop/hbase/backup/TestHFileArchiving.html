188  /**
189   * Test that the region directory is removed when we archive a region without store files, but
190   * still has hidden files.
191   * @throws IOException throws an IOException if there's problem creating a table
192   *   or if there's an issue with accessing FileSystem.
193   */
194  @Test
195  public void testDeleteRegionWithNoStoreFiles() throws IOException {
196    final TableName tableName = TableName.valueOf(name.getMethodName());
197    UTIL.createTable(tableName, TEST_FAM);
198
199    // get the current store files for the region
200    List<HRegion> servingRegions = UTIL.getHBaseCluster().getRegions(tableName);
201    // make sure we only have 1 region serving this table
202    assertEquals(1, servingRegions.size());
203    HRegion region = servingRegions.get(0);
204
205    FileSystem fs = region.getRegionFileSystem().getFileSystem();
206
207    // make sure there are some files in the regiondir
208    Path rootDir = FSUtils.getRootDir(fs.getConf());
209    Path regionDir = FSUtils.getRegionDirFromRootDir(rootDir, region.getRegionInfo());
210    FileStatus[] regionFiles = FSUtils.listStatus(fs, regionDir, null);
211    Assert.assertNotNull("No files in the region directory", regionFiles);
212    if (LOG.isDebugEnabled()) {
213      List<Path> files = new ArrayList<>();
214      for (FileStatus file : regionFiles) {
215        files.add(file.getPath());
216      }
217      LOG.debug("Current files:" + files);
218    }
219    // delete the visible folders so we just have hidden files/folders
220    final PathFilter dirFilter = new FSUtils.DirFilter(fs);
221    PathFilter nonHidden = new PathFilter() {
222      @Override
223      public boolean accept(Path file) {
224        return dirFilter.accept(file) && !file.getName().startsWith(".");
225      }
226    };
227    FileStatus[] storeDirs = FSUtils.listStatus(fs, regionDir, nonHidden);
228    for (FileStatus store : storeDirs) {
229      LOG.debug("Deleting store for test");
230      fs.delete(store.getPath(), true);
231    }
232
233    // then archive the region
234    HFileArchiver.archiveRegion(UTIL.getConfiguration(), fs, region.getRegionInfo());
235
236    // and check to make sure the region directoy got deleted
237    assertFalse("Region directory (" + regionDir + "), still exists.", fs.exists(regionDir));
238
239    UTIL.deleteTable(tableName);
240  }

266  public void testArchiveRegions() throws Exception {
267    final TableName tableName = TableName.valueOf(name.getMethodName());
268    List<HRegion> regions = initTableForArchivingRegions(tableName);
269
270    FileSystem fs = UTIL.getTestFileSystem();
271
272    // now attempt to depose the regions

274    Path tableDir = FSUtils.getTableDir(rootDir, regions.get(0).getRegionInfo().getTable());

291
292      // make sure we archived the store files

298    for (Path regionDir: regionDirList) {

319      ugi = UserGroupInformation.createUserForTesting("foo1234", new String[]{"group1"});

338    final TableName tableName = TableName.valueOf(name.getMethodName());
339    UTIL.createTable(tableName, TEST_FAM);
340
341    List<HRegion> servingRegions = UTIL.getHBaseCluster().getRegions(tableName);
342    // make sure we only have 1 region serving this table
343    assertEquals(1, servingRegions.size());
344    HRegion region = servingRegions.get(0);
345
346    // get the parent RS and monitor
347    HRegionServer hrs = UTIL.getRSForFirstRegionInTable(tableName);
348    FileSystem fs = hrs.getFileSystem();
349
350    // put some data on the region
351    LOG.debug("-------Loading table");
352    UTIL.loadRegion(region, TEST_FAM);
353
354    // get the hfiles in the region
355    List<HRegion> regions = hrs.getRegions(tableName);
356    assertEquals("More that 1 region for test table.", 1, regions.size());
357
358    region = regions.get(0);
359    // wait for all the compactions to complete
360    region.waitForFlushesAndCompactions();
361
362    // disable table to prevent new updates
363    UTIL.getAdmin().disableTable(tableName);
364    LOG.debug("Disabled table");
365
366    // remove all the files from the archive to get a fair comparison
367    clearArchiveDirectory();
368
369    // then get the current store files
370    byte[][]columns = region.getTableDescriptor().getColumnFamilyNames().toArray(new byte[0][]);
371    List<String> storeFiles = region.getStoreFileList(columns);
372
373    // then delete the table so the hfiles get archived
374    UTIL.deleteTable(tableName);
375    LOG.debug("Deleted table");
376
377    assertArchiveFiles(fs, storeFiles, 30000);
378  }

380  private void assertArchiveFiles(FileSystem fs, List<String> storeFiles, long timeout)
381          throws IOException {
382    long end = System.currentTimeMillis() + timeout;
383    Path archiveDir = HFileArchiveUtil.getArchivePath(UTIL.getConfiguration());
384    List<String> archivedFiles = new ArrayList<>();
385
386    // We have to ensure that the DeleteTableHandler is finished. HBaseAdmin.deleteXXX()
387    // can return before all files
388    // are archived. We should fix HBASE-5487 and fix synchronous operations from admin.
389    while (System.currentTimeMillis() < end) {
390      archivedFiles = getAllFileNames(fs, archiveDir);
391      if (archivedFiles.size() >= storeFiles.size()) {
392        break;
393      }
394    }
395
396    Collections.sort(storeFiles);
397    Collections.sort(archivedFiles);
398
399    LOG.debug("Store files:");
400    for (int i = 0; i < storeFiles.size(); i++) {
401      LOG.debug(i + " - " + storeFiles.get(i));
402    }
403    LOG.debug("Archive files:");
404    for (int i = 0; i < archivedFiles.size(); i++) {
405      LOG.debug(i + " - " + archivedFiles.get(i));
406    }
407
408    assertTrue("Archived files are missing some of the store files!",
409      archivedFiles.containsAll(storeFiles));
410  }

463  /**
464   * Test HFileArchiver.resolveAndArchive() race condition HBASE-7643
465   */
466  @Test
467  public void testCleaningRace() throws Exception {
468    final long TEST_TIME = 20 * 1000;
469    final ChoreService choreService = new ChoreService("TEST_SERVER_NAME");
470
471    Configuration conf = UTIL.getMiniHBaseCluster().getMaster().getConfiguration();
472    Path rootDir = UTIL.getDataTestDirOnTestFS("testCleaningRace");
473    FileSystem fs = UTIL.getTestFileSystem();
474
475    Path archiveDir = new Path(rootDir, HConstants.HFILE_ARCHIVE_DIRECTORY);
476    Path regionDir = new Path(FSUtils.getTableDir(new Path("./"),
477        TableName.valueOf(name.getMethodName())), "abcdef");
478    Path familyDir = new Path(regionDir, "cf");
479
480    Path sourceRegionDir = new Path(rootDir, regionDir);
481    fs.mkdirs(sourceRegionDir);
482
483    Stoppable stoppable = new StoppableImplementation();
484
485    // The cleaner should be looping without long pauses to reproduce the race condition.
486    HFileCleaner cleaner = getHFileCleaner(stoppable, conf, fs, archiveDir);
487    assertNotNull("cleaner should not be null", cleaner);
488    try {
489      choreService.scheduleChore(cleaner);
490      // Keep creating/archiving new files while the cleaner is running in the other thread
491      long startTime = System.currentTimeMillis();
492      for (long fid = 0; (System.currentTimeMillis() - startTime) < TEST_TIME; ++fid) {
493        Path file = new Path(familyDir,  String.valueOf(fid));
494        Path sourceFile = new Path(rootDir, file);
495        Path archiveFile = new Path(archiveDir, file);
496
497        fs.createNewFile(sourceFile);
498
499        try {
500          // Try to archive the file
501          HFileArchiver.archiveRegion(fs, rootDir,
502              sourceRegionDir.getParent(), sourceRegionDir);
503
504          // The archiver succeded, the file is no longer in the original location
505          // but it's in the archive location.
506          LOG.debug("hfile=" + fid + " should be in the archive");
507          assertTrue(fs.exists(archiveFile));
508          assertFalse(fs.exists(sourceFile));
509        } catch (IOException e) {
510          // The archiver is unable to archive the file. Probably HBASE-7643 race condition.
511          // in this case, the file should not be archived, and we should have the file
512          // in the original location.
513          LOG.debug("hfile=" + fid + " should be in the source location");
514          assertFalse(fs.exists(archiveFile));
515          assertTrue(fs.exists(sourceFile));
516
517          // Avoid to have this file in the next run
518          fs.delete(sourceFile, false);
519        }
520      }
521    } finally {
522      stoppable.stop("test end");
523      cleaner.cancel(true);
524      choreService.shutdown();
525      fs.delete(rootDir, true);
526    }
527  }

529  // Avoid passing a null master to CleanerChore, see HBASE-21175
530  private HFileCleaner getHFileCleaner(Stoppable stoppable, Configuration conf,
531        FileSystem fs, Path archiveDir) throws IOException {
532    Map<String, Object> params = new HashMap<>();
533    params.put(HMaster.MASTER, UTIL.getMiniHBaseCluster().getMaster());
534    HFileCleaner cleaner = new HFileCleaner(1, stoppable, conf, fs, archiveDir);
535    cleaner.init(params);
536    return Objects.requireNonNull(cleaner);
537  }

544  /**
545   * Get the names of all the files below the given directory
546   * @param fs the file system to inspect
547   * @param archiveDir the directory in which to look
548   * @return a list of all files in the directory and sub-directories
549   * @throws java.io.IOException throws IOException in case FS is unavailable
550   */
551  private List<String> getAllFileNames(final FileSystem fs, Path archiveDir) throws IOException  {
552    FileStatus[] files = FSUtils.listStatus(fs, archiveDir, new PathFilter() {
553      @Override
554      public boolean accept(Path p) {
555        if (p.getName().contains(HConstants.RECOVERED_EDITS_DIR)) {
556          return false;
557        }
558        return true;
559      }
560    });
561    return recurseOnFiles(fs, files, new ArrayList<>());
562  }
563
564  /** Recursively lookup all the file names under the file[] array **/
565  private List<String> recurseOnFiles(FileSystem fs, FileStatus[] files, List<String> fileNames)
566      throws IOException {
567    if (files == null || files.length == 0) {
568      return fileNames;
569    }
570
571    for (FileStatus file : files) {
572      if (file.isDirectory()) {
573        recurseOnFiles(fs, FSUtils.listStatus(fs, file.getPath(), null), fileNames);
574      } else {
575        fileNames.add(file.getPath().getName());
576      }
577    }
578    return fileNames;
579  }
