047/**
048 * 1. Create table t1
049 * 2. Load data to t1
050 * 3 Full backup t1
051 * 4 Load data to t1
052 * 5 bulk load into t1
053 * 6 Incremental backup t1
054 */
055@Category(LargeTests.class)
056public class TestIncrementalBackupWithBulkLoad extends TestBackupBase {
057
058  @ClassRule
059  public static final HBaseClassTestRule CLASS_RULE =
060      HBaseClassTestRule.forClass(TestIncrementalBackupWithBulkLoad.class);
061
062  private static final Logger LOG = LoggerFactory.getLogger(TestIncrementalBackupDeleteTable.class);
063
064  // implement all test cases in 1 test since incremental backup/restore has dependencies
065  @Test
066  public void TestIncBackupDeleteTable() throws Exception {
067    String testName = "TestIncBackupDeleteTable";
068    // #1 - create full backup for all tables
069    LOG.info("create full backup image for all tables");
070
071    List<TableName> tables = Lists.newArrayList(table1);
072    Connection conn = ConnectionFactory.createConnection(conf1);
073    Admin admin = conn.getAdmin();
074    BackupAdminImpl client = new BackupAdminImpl(conn);
075
076    BackupRequest request = createBackupRequest(BackupType.FULL, tables, BACKUP_ROOT_DIR);
077    String backupIdFull = client.backupTables(request);
078
079    assertTrue(checkSucceeded(backupIdFull));
080
081    // #2 - insert some data to table table1
082    Table t1 = conn.getTable(table1);
083    Put p1;
084    for (int i = 0; i < NB_ROWS_IN_BATCH; i++) {
085      p1 = new Put(Bytes.toBytes("row-t1" + i));
086      p1.addColumn(famName, qualName, Bytes.toBytes("val" + i));
087      t1.put(p1);
088    }
089
090    Assert.assertEquals(TEST_UTIL.countRows(t1), NB_ROWS_IN_BATCH * 2);
091    t1.close();
092
093    int NB_ROWS2 = 20;
094    LOG.debug("bulk loading into " + testName);
095    int actual = TestBulkLoadHFiles.loadHFiles(testName, table1Desc, TEST_UTIL, famName,
096        qualName, false, null, new byte[][][] {
097          new byte[][]{ Bytes.toBytes("aaaa"), Bytes.toBytes("cccc") },
098          new byte[][]{ Bytes.toBytes("ddd"), Bytes.toBytes("ooo") },
099        }, true, false, true, NB_ROWS_IN_BATCH*2, NB_ROWS2);
100
101    // #3 - incremental backup for table1
102    tables = Lists.newArrayList(table1);
103    request = createBackupRequest(BackupType.INCREMENTAL, tables, BACKUP_ROOT_DIR);
104    String backupIdIncMultiple = client.backupTables(request);
105    assertTrue(checkSucceeded(backupIdIncMultiple));
106    // #4 bulk load again
107    LOG.debug("bulk loading into " + testName);
108    int actual1 = TestBulkLoadHFiles.loadHFiles(testName, table1Desc, TEST_UTIL, famName,
109      qualName, false, null,
110      new byte[][][] { new byte[][] { Bytes.toBytes("ppp"), Bytes.toBytes("qqq") },
111        new byte[][] { Bytes.toBytes("rrr"), Bytes.toBytes("sss") }, },
112      true, false, true, NB_ROWS_IN_BATCH * 2 + actual, NB_ROWS2);
113
114    // #5 - incremental backup for table1
115    tables = Lists.newArrayList(table1);
116    request = createBackupRequest(BackupType.INCREMENTAL, tables, BACKUP_ROOT_DIR);
117    String backupIdIncMultiple1 = client.backupTables(request);
118    assertTrue(checkSucceeded(backupIdIncMultiple1));
119    // Delete all data in table1
120    TEST_UTIL.deleteTableData(table1);
121    // #5.1 - check tables for full restore */
122    Admin hAdmin = TEST_UTIL.getAdmin();
123
124    // #6 - restore incremental backup for table1
125    TableName[] tablesRestoreIncMultiple = new TableName[] { table1 };
126    //TableName[] tablesMapIncMultiple = new TableName[] { table1_restore };
127    client.restore(BackupUtils.createRestoreRequest(BACKUP_ROOT_DIR, backupIdIncMultiple1,
128      false, tablesRestoreIncMultiple, tablesRestoreIncMultiple, true));
129
130    Table hTable = conn.getTable(table1);
131    Assert.assertEquals(TEST_UTIL.countRows(hTable), NB_ROWS_IN_BATCH * 2 + actual + actual1);
132    request = createBackupRequest(BackupType.FULL, tables, BACKUP_ROOT_DIR);
133
134    backupIdFull = client.backupTables(request);
135    try (final BackupSystemTable table = new BackupSystemTable(conn)) {
136      Pair<Map<TableName, Map<String, Map<String, List<Pair<String, Boolean>>>>>, List<byte[]>> pair
137        = table.readBulkloadRows(tables);
138      assertTrue("map still has " + pair.getSecond().size() + " entries",
139          pair.getSecond().isEmpty());
140    }
141    assertTrue(checkSucceeded(backupIdFull));
142
143    hTable.close();
144    admin.close();
145    conn.close();
146  }
