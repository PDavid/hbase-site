052@Category(LargeTests.class)
053@RunWith(Parameterized.class)
054public class TestIncrementalBackup extends TestBackupBase {
055
056  @ClassRule
057  public static final HBaseClassTestRule CLASS_RULE =
058      HBaseClassTestRule.forClass(TestIncrementalBackup.class);
059
060  private static final Logger LOG = LoggerFactory.getLogger(TestIncrementalBackup.class);
061
062  @Parameterized.Parameters
063  public static Collection<Object[]> data() {
064    provider = "multiwal";
065    List<Object[]> params = new ArrayList<>();
066    params.add(new Object[] { Boolean.TRUE });
067    return params;
068  }

070  public TestIncrementalBackup(Boolean b) {
071  }
072
073  // implement all test cases in 1 test since incremental
074  // backup/restore has dependencies
075  @Test
076  public void TestIncBackupRestore() throws Exception {
077    int ADD_ROWS = 99;
078
079    // #1 - create full backup for all tables
080    LOG.info("create full backup image for all tables");
081    List<TableName> tables = Lists.newArrayList(table1, table2);
082    final byte[] fam3Name = Bytes.toBytes("f3");
083    final byte[] mobName = Bytes.toBytes("mob");
084
085    table1Desc.addFamily(new HColumnDescriptor(fam3Name));
086    HColumnDescriptor mobHcd = new HColumnDescriptor(mobName);
087    mobHcd.setMobEnabled(true);
088    mobHcd.setMobThreshold(5L);
089    table1Desc.addFamily(mobHcd);
090    HBaseTestingUtility.modifyTableSync(TEST_UTIL.getAdmin(), table1Desc);
091
092    try (Connection conn = ConnectionFactory.createConnection(conf1)) {
093      int NB_ROWS_FAM3 = 6;
094      insertIntoTable(conn, table1, fam3Name, 3, NB_ROWS_FAM3).close();
095      insertIntoTable(conn, table1, mobName, 3, NB_ROWS_FAM3).close();
096      Admin admin = conn.getAdmin();
097      BackupAdminImpl client = new BackupAdminImpl(conn);
098      BackupRequest request = createBackupRequest(BackupType.FULL, tables, BACKUP_ROOT_DIR);
099      String backupIdFull = client.backupTables(request);
100      assertTrue(checkSucceeded(backupIdFull));
101
102      // #2 - insert some data to table
103      Table t1 = insertIntoTable(conn, table1, famName, 1, ADD_ROWS);
104      LOG.debug("writing " + ADD_ROWS + " rows to " + table1);
105      Assert.assertEquals(HBaseTestingUtility.countRows(t1),
106              NB_ROWS_IN_BATCH + ADD_ROWS + NB_ROWS_FAM3);
107      LOG.debug("written " + ADD_ROWS + " rows to " + table1);
108
109      // additionally, insert rows to MOB cf
110      int NB_ROWS_MOB = 111;
111      insertIntoTable(conn, table1, mobName, 3, NB_ROWS_MOB);
112      LOG.debug("written " + NB_ROWS_MOB + " rows to " + table1 + " to Mob enabled CF");
113      t1.close();
114      Assert.assertEquals(HBaseTestingUtility.countRows(t1),
115              NB_ROWS_IN_BATCH + ADD_ROWS + NB_ROWS_MOB);
116
117      Table t2 = conn.getTable(table2);
118      Put p2;
119      for (int i = 0; i < 5; i++) {
120        p2 = new Put(Bytes.toBytes("row-t2" + i));
121        p2.addColumn(famName, qualName, Bytes.toBytes("val" + i));
122        t2.put(p2);
123      }
124      Assert.assertEquals(NB_ROWS_IN_BATCH + 5, HBaseTestingUtility.countRows(t2));
125      t2.close();
126      LOG.debug("written " + 5 + " rows to " + table2);
127
128      // split table1
129      MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
130      List<HRegion> regions = cluster.getRegions(table1);
131      byte[] name = regions.get(0).getRegionInfo().getRegionName();
132      long startSplitTime = EnvironmentEdgeManager.currentTime();
133
134      try {
135        admin.splitRegionAsync(name).get();
136      } catch (Exception e) {
137        // although split fail, this may not affect following check in current API,
138        // exception will be thrown.
139        LOG.debug("region is not splittable, because " + e);
140      }
141      while (!admin.isTableAvailable(table1)) {
142        Thread.sleep(100);
143      }
144
145      long endSplitTime = EnvironmentEdgeManager.currentTime();
146      // split finished
147      LOG.debug("split finished in =" + (endSplitTime - startSplitTime));
148
149      // #3 - incremental backup for multiple tables
150      tables = Lists.newArrayList(table1, table2);
151      request = createBackupRequest(BackupType.INCREMENTAL, tables, BACKUP_ROOT_DIR);
152      String backupIdIncMultiple = client.backupTables(request);
153      assertTrue(checkSucceeded(backupIdIncMultiple));
154
155      // add column family f2 to table1
156      final byte[] fam2Name = Bytes.toBytes("f2");
157      table1Desc.addFamily(new HColumnDescriptor(fam2Name));
158
159      // drop column family f3
160      table1Desc.removeFamily(fam3Name);
161      HBaseTestingUtility.modifyTableSync(TEST_UTIL.getAdmin(), table1Desc);
162
163      int NB_ROWS_FAM2 = 7;
164      Table t3 = insertIntoTable(conn, table1, fam2Name, 2, NB_ROWS_FAM2);
165      t3.close();
166
167      // Wait for 5 sec to make sure that old WALs were deleted
168      Thread.sleep(5000);
169
170      // #4 - additional incremental backup for multiple tables
171      request = createBackupRequest(BackupType.INCREMENTAL, tables, BACKUP_ROOT_DIR);
172      String backupIdIncMultiple2 = client.backupTables(request);
173      assertTrue(checkSucceeded(backupIdIncMultiple2));
174
175      // #5 - restore full backup for all tables
176      TableName[] tablesRestoreFull = new TableName[] { table1, table2 };
177      TableName[] tablesMapFull = new TableName[] { table1_restore, table2_restore };
178
179      LOG.debug("Restoring full " + backupIdFull);
180      client.restore(BackupUtils.createRestoreRequest(BACKUP_ROOT_DIR, backupIdFull, false,
181                tablesRestoreFull, tablesMapFull, true));
182
183      // #6.1 - check tables for full restore
184      Admin hAdmin = TEST_UTIL.getAdmin();
185      assertTrue(hAdmin.tableExists(table1_restore));
186      assertTrue(hAdmin.tableExists(table2_restore));
187      hAdmin.close();
188
189      // #6.2 - checking row count of tables for full restore
190      Table hTable = conn.getTable(table1_restore);
191      Assert.assertEquals(HBaseTestingUtility.countRows(hTable), NB_ROWS_IN_BATCH + NB_ROWS_FAM3);
192      hTable.close();
193
194      hTable = conn.getTable(table2_restore);
195      Assert.assertEquals(NB_ROWS_IN_BATCH, HBaseTestingUtility.countRows(hTable));
196      hTable.close();
197
198      // #7 - restore incremental backup for multiple tables, with overwrite
199      TableName[] tablesRestoreIncMultiple = new TableName[] { table1, table2 };
200      TableName[] tablesMapIncMultiple = new TableName[] { table1_restore, table2_restore };
201      client.restore(BackupUtils.createRestoreRequest(BACKUP_ROOT_DIR, backupIdIncMultiple2,
202              false, tablesRestoreIncMultiple, tablesMapIncMultiple, true));
203      hTable = conn.getTable(table1_restore);
204
205      LOG.debug("After incremental restore: " + hTable.getDescriptor());
206      int countFamName = TEST_UTIL.countRows(hTable, famName);
207      LOG.debug("f1 has " + countFamName + " rows");
208      Assert.assertEquals(countFamName, NB_ROWS_IN_BATCH + ADD_ROWS);
209
210      int countFam2Name = TEST_UTIL.countRows(hTable, fam2Name);
211      LOG.debug("f2 has " + countFam2Name + " rows");
212      Assert.assertEquals(countFam2Name, NB_ROWS_FAM2);
213
214      int countMobName = TEST_UTIL.countRows(hTable, mobName);
215      LOG.debug("mob has " + countMobName + " rows");
216      Assert.assertEquals(countMobName, NB_ROWS_MOB);
217      hTable.close();
218
219      hTable = conn.getTable(table2_restore);
220      Assert.assertEquals(NB_ROWS_IN_BATCH + 5, HBaseTestingUtility.countRows(hTable));
221      hTable.close();
222      admin.close();
223    }
224  }
