097   * tests same in presence of deletes.  Test cores are written so can be

108        .setColumnFamily(ColumnFamilyDescriptorBuilder.newBuilder(TimestampTestBase.FAMILY_NAME)
109          .setMaxVersions(3).build())

111    this.admin.createTable(tableDescriptor);

134  /**
135   * Verifies versions across a cluster restart.
136   * <p/>
137   * Port of old TestGetRowVersions test to here so can better utilize the spun
138   * up cluster running more than a single test per spin up.  Keep old tests'
139   * crazyness.
140   */
141  @Test
142  public void testGetRowVersions() throws Exception {
143    final byte [] contents = Bytes.toBytes("contents");
144    final byte [] row = Bytes.toBytes("row");
145    final byte [] value1 = Bytes.toBytes("value1");
146    final byte [] value2 = Bytes.toBytes("value2");
147    final long timestamp1 = 100L;
148    final long timestamp2 = 200L;
149    TableDescriptor tableDescriptor =
150      TableDescriptorBuilder.newBuilder(TableName.valueOf(name.getMethodName()))
151        .setColumnFamily(ColumnFamilyDescriptorBuilder.newBuilder(contents)
152          .setMaxVersions(3).build())
153        .build();
154    this.admin.createTable(tableDescriptor);
155    Put put = new Put(row, timestamp1);
156    put.addColumn(contents, contents, value1);
157    Table table = UTIL.getConnection().getTable(tableDescriptor.getTableName());
158    table.put(put);
159    // Shut down and restart the HBase cluster
160    table.close();
161    UTIL.shutdownMiniHBaseCluster();
162    LOG.debug("HBase cluster shut down -- restarting");
163    StartTestingClusterOption option = StartTestingClusterOption.builder()
164        .numRegionServers(NUM_SLAVES).build();
165    UTIL.startMiniHBaseCluster(option);
166    // Make a new connection.
167    table = UTIL.getConnection().getTable(tableDescriptor.getTableName());
168    // Overwrite previous value
169    put = new Put(row, timestamp2);
170    put.addColumn(contents, contents, value2);
171    table.put(put);
172    // Now verify that getRow(row, column, latest) works
173    Get get = new Get(row);
174    // Should get one version by default
175    Result r = table.get(get);
176    assertNotNull(r);
177    assertFalse(r.isEmpty());
178    assertEquals(1, r.size());
179    byte [] value = r.getValue(contents, contents);
180    assertNotEquals(0, value.length);
181    assertTrue(Bytes.equals(value, value2));
182    // Now check getRow with multiple versions
183    get = new Get(row);
184    get.readAllVersions();
185    r = table.get(get);
186    assertEquals(2, r.size());
187    value = r.getValue(contents, contents);
188    assertNotEquals(0, value.length);
189    assertArrayEquals(value, value2);
190    NavigableMap<byte[], NavigableMap<byte[], NavigableMap<Long, byte[]>>> map =
191      r.getMap();
192    NavigableMap<byte[], NavigableMap<Long, byte[]>> familyMap =
193      map.get(contents);
194    NavigableMap<Long, byte[]> versionMap = familyMap.get(contents);
195    assertEquals(2, versionMap.size());
196    assertArrayEquals(value1, versionMap.get(timestamp1));
197    assertArrayEquals(value2, versionMap.get(timestamp2));
198    table.close();
199  }

201  /**
202   * Port of old TestScanMultipleVersions test here so can better utilize the
203   * spun up cluster running more than just a single test.  Keep old tests
204   * crazyness.
205   *
206   * <p>Tests five cases of scans and timestamps.
207   */
208  @Test
209  public void testScanMultipleVersions() throws Exception {
210    final TableName tableName = TableName.valueOf(name.getMethodName());
211    TableDescriptor tableDescriptor = TableDescriptorBuilder.newBuilder(tableName)
212      .setColumnFamily(ColumnFamilyDescriptorBuilder.of(HConstants.CATALOG_FAMILY)).build();
213
214    final byte[][] rows = new byte[][] { Bytes.toBytes("row_0200"), Bytes.toBytes("row_0800") };
215    final byte [][] splitRows = new byte[][] {Bytes.toBytes("row_0500")};
216    final long [] timestamp = new long[] {100L, 1000L};
217    this.admin.createTable(tableDescriptor, splitRows);
218    Table table = UTIL.getConnection().getTable(tableName);
219    // Assert we got the region layout wanted.
220    Pair<byte[][], byte[][]> keys = UTIL.getConnection()
221        .getRegionLocator(tableName).getStartEndKeys();
222    assertEquals(2, keys.getFirst().length);
223    byte[][] startKeys = keys.getFirst();
224    byte[][] endKeys = keys.getSecond();
225
226    for (int i = 0; i < startKeys.length; i++) {
227      if (i == 0) {
228        assertArrayEquals(HConstants.EMPTY_START_ROW, startKeys[i]);
229        assertArrayEquals(endKeys[i], splitRows[0]);
230      } else if (i == 1) {
231        assertArrayEquals(splitRows[0], startKeys[i]);
232        assertArrayEquals(endKeys[i], HConstants.EMPTY_END_ROW);
233      }
234    }
235    // Insert data
236    List<Put> puts = new ArrayList<>();
237    for (int i = 0; i < startKeys.length; i++) {
238      for (int j = 0; j < timestamp.length; j++) {
239        Put put = new Put(rows[i], timestamp[j]);
240        put.addColumn(HConstants.CATALOG_FAMILY, null, timestamp[j], Bytes.toBytes(timestamp[j]));
241        puts.add(put);
242      }
243    }
244    table.put(puts);
245    // There are 5 cases we have to test. Each is described below.
246    for (int i = 0; i < rows.length; i++) {
247      for (int j = 0; j < timestamp.length; j++) {
248        Get get = new Get(rows[i]);
249        get.addFamily(HConstants.CATALOG_FAMILY);
250        get.setTimestamp(timestamp[j]);
251        Result result = table.get(get);
252        int cellCount = result.rawCells().length;
253        assertEquals(1, cellCount);
254      }
255    }
256
257    // Case 1: scan with LATEST_TIMESTAMP. Should get two rows
258    int count;
259    Scan scan = new Scan();
260    scan.addFamily(HConstants.CATALOG_FAMILY);
261    try (ResultScanner s = table.getScanner(scan)) {
262      count = Iterables.size(s);
263    }
264    assertEquals("Number of rows should be 2", 2, count);
265
266    // Case 2: Scan with a timestamp greater than most recent timestamp
267    // (in this case > 1000 and < LATEST_TIMESTAMP. Should get 2 rows.
268    scan = new Scan();
269    scan.setTimeRange(1000L, Long.MAX_VALUE);
270    scan.addFamily(HConstants.CATALOG_FAMILY);
271    try (ResultScanner s = table.getScanner(scan)) {
272      count = Iterables.size(s);
273    }
274    assertEquals("Number of rows should be 2", 2, count);
275
276    // Case 3: scan with timestamp equal to most recent timestamp
277    // (in this case == 1000. Should get 2 rows.
278    scan = new Scan();
279    scan.setTimestamp(1000L);
280    scan.addFamily(HConstants.CATALOG_FAMILY);
281    try (ResultScanner s = table.getScanner(scan)) {
282      count = Iterables.size(s);
283    }
284    assertEquals("Number of rows should be 2", 2, count);
285
286    // Case 4: scan with timestamp greater than first timestamp but less than
287    // second timestamp (100 < timestamp < 1000). Should get 2 rows.
288    scan = new Scan();
289    scan.setTimeRange(100L, 1000L);
290    scan.addFamily(HConstants.CATALOG_FAMILY);
291    try (ResultScanner s = table.getScanner(scan)) {
292      count = Iterables.size(s);
293    }
294    assertEquals("Number of rows should be 2", 2, count);
295
296    // Case 5: scan with timestamp equal to first timestamp (100)
297    // Should get 2 rows.
298    scan = new Scan();
299    scan.setTimestamp(100L);
300    scan.addFamily(HConstants.CATALOG_FAMILY);
301    try (ResultScanner s = table.getScanner(scan)) {
302      count = Iterables.size(s);
303    }
304    assertEquals("Number of rows should be 2", 2, count);
305  }
