134  private TableDescriptor createTableDesc(TableName name, int cfs) {
135    TableDescriptorBuilder builder = TableDescriptorBuilder.newBuilder(name);
136    IntStream.range(0, cfs).mapToObj(i -> ColumnFamilyDescriptorBuilder.of(family(i)))
137      .forEachOrdered(builder::setColumnFamily);
138    return builder.build();

192  /**
193   * Split the known table in half. (this is hard coded for this test suite)
194   */
195  private void forceSplit(TableName table) {
196    try {
197      // need to call regions server to by synchronous but isn't visible.
198      HRegionServer hrs = util.getRSForFirstRegionInTable(table);
199
200      for (RegionInfo hri : ProtobufUtil.getOnlineRegions(hrs.getRSRpcServices())) {
201        if (hri.getTable().equals(table)) {
202          util.getAdmin().splitRegionAsync(hri.getRegionName(), rowkey(ROWCOUNT / 2));
203          // ProtobufUtil.split(null, hrs.getRSRpcServices(), hri, rowkey(ROWCOUNT / 2));
204        }
205      }
206
207      // verify that split completed.
208      int regions;
209      do {
210        regions = 0;
211        for (RegionInfo hri : ProtobufUtil.getOnlineRegions(hrs.getRSRpcServices())) {
212          if (hri.getTable().equals(table)) {
213            regions++;
214          }
215        }
216        if (regions != 2) {
217          LOG.info("Taking some time to complete split...");
218          Thread.sleep(250);
219        }
220      } while (regions != 2);
221    } catch (IOException e) {
222      e.printStackTrace();
223    } catch (InterruptedException e) {
224      e.printStackTrace();
225    }
226  }

262    CompletableFuture<T> future = new CompletableFuture<>();
263    future.completeExceptionally(error);
264    return future;
265  }

299  /**
300   * Test that shows that exception thrown from the RS side will result in the expected number of
301   * retries set by ${@link HConstants#HBASE_CLIENT_RETRIES_NUMBER} when
302   * ${@link BulkLoadHFiles#RETRY_ON_IO_EXCEPTION} is set
303   */
304  @Test
305  public void testRetryOnIOException() throws Exception {
306    TableName table = TableName.valueOf(name.getMethodName());
307    AtomicInteger calls = new AtomicInteger(0);
308    setupTable(util.getConnection(), table, 10);
309    Configuration conf = new Configuration(util.getConfiguration());
310    conf.setInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER, 2);
311    conf.setBoolean(BulkLoadHFiles.RETRY_ON_IO_EXCEPTION, true);
312    BulkLoadHFilesTool loader = new BulkLoadHFilesTool(conf) {
313
314      @Override
315      protected void bulkLoadPhase(AsyncClusterConnection conn, TableName tableName,
316          Deque<LoadQueueItem> queue, Multimap<ByteBuffer, LoadQueueItem> regionGroups,
317          boolean copyFiles, Map<LoadQueueItem, ByteBuffer> item2RegionMap) throws IOException {
318        if (calls.get() < conf.getInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER,
319          HConstants.DEFAULT_HBASE_CLIENT_RETRIES_NUMBER)) {
320          calls.incrementAndGet();
321          super.bulkLoadPhase(mockAndInjectError(conn), tableName, queue, regionGroups, copyFiles,
322            item2RegionMap);
323        } else {
324          super.bulkLoadPhase(conn, tableName, queue, regionGroups, copyFiles, item2RegionMap);
325        }
326      }
327    };
328    Path dir = buildBulkFiles(table, 1);
329    loader.bulkLoad(table, dir);
330    assertEquals(calls.get(), 2);
331  }

333  /**
334   * This test exercises the path where there is a split after initial validation but before the
335   * atomic bulk load call. We cannot use presplitting to test this path, so we actually inject a
336   * split just before the atomic region load.
337   */
338  @Test
339  public void testSplitWhileBulkLoadPhase() throws Exception {
340    final TableName table = TableName.valueOf(name.getMethodName());
341    setupTable(util.getConnection(), table, 10);
342    populateTable(util.getConnection(), table, 1);
343    assertExpectedTable(table, ROWCOUNT, 1);
344
345    // Now let's cause trouble. This will occur after checks and cause bulk
346    // files to fail when attempt to atomically import. This is recoverable.
347    final AtomicInteger attemptedCalls = new AtomicInteger();
348    BulkLoadHFilesTool loader = new BulkLoadHFilesTool(util.getConfiguration()) {
349
350      @Override
351      protected void bulkLoadPhase(AsyncClusterConnection conn, TableName tableName,
352          Deque<LoadQueueItem> queue, Multimap<ByteBuffer, LoadQueueItem> regionGroups,
353          boolean copyFiles, Map<LoadQueueItem, ByteBuffer> item2RegionMap) throws IOException {
354        int i = attemptedCalls.incrementAndGet();
355        if (i == 1) {
356          // On first attempt force a split.
357          forceSplit(table);
358        }
359        super.bulkLoadPhase(conn, tableName, queue, regionGroups, copyFiles, item2RegionMap);
360      }
361    };
362
363    // create HFiles for different column families
364    Path dir = buildBulkFiles(table, 2);
365    loader.bulkLoad(table, dir);
366
367    // check that data was loaded
368    // The three expected attempts are 1) failure because need to split, 2)
369    // load of split top 3) load of split bottom
370    assertEquals(3, attemptedCalls.get());
371    assertExpectedTable(table, ROWCOUNT, 2);
372  }

374  /**
375   * This test splits a table and attempts to bulk load. The bulk import files should be split
376   * before atomically importing.
377   */
378  @Test
379  public void testGroupOrSplitPresplit() throws Exception {
380    final TableName table = TableName.valueOf(name.getMethodName());
381    setupTable(util.getConnection(), table, 10);
382    populateTable(util.getConnection(), table, 1);
383    assertExpectedTable(util.getConnection(), table, ROWCOUNT, 1);
384    forceSplit(table);
385
386    final AtomicInteger countedLqis = new AtomicInteger();
387    BulkLoadHFilesTool loader = new BulkLoadHFilesTool(util.getConfiguration()) {
388
389      @Override
390      protected Pair<List<LoadQueueItem>, String> groupOrSplit(AsyncClusterConnection conn,
391          TableName tableName, Multimap<ByteBuffer, LoadQueueItem> regionGroups, LoadQueueItem item,
392          List<Pair<byte[], byte[]>> startEndKeys) throws IOException {
393        Pair<List<LoadQueueItem>, String> lqis =
394          super.groupOrSplit(conn, tableName, regionGroups, item, startEndKeys);
395        if (lqis != null && lqis.getFirst() != null) {
396          countedLqis.addAndGet(lqis.getFirst().size());
397        }
398        return lqis;
399      }
400    };
401
402    // create HFiles for different column families
403    Path dir = buildBulkFiles(table, 2);
404    loader.bulkLoad(table, dir);
405    assertExpectedTable(util.getConnection(), table, ROWCOUNT, 2);
406    assertEquals(20, countedLqis.get());
407  }

409  /**
410   * This test creates a table with many small regions. The bulk load files would be splitted
411   * multiple times before all of them can be loaded successfully.
412   */
413  @Test
414  public void testSplitTmpFileCleanUp() throws Exception {
415    final TableName table = TableName.valueOf(name.getMethodName());
416    byte[][] SPLIT_KEYS = new byte[][] { Bytes.toBytes("row_00000010"),
417      Bytes.toBytes("row_00000020"), Bytes.toBytes("row_00000030"), Bytes.toBytes("row_00000040"),
418      Bytes.toBytes("row_00000050") };
419    setupTableWithSplitkeys(table, 10, SPLIT_KEYS);
420
421    BulkLoadHFiles loader = BulkLoadHFiles.create(util.getConfiguration());
422
423    // create HFiles
424    Path dir = buildBulkFiles(table, 2);
425    loader.bulkLoad(table, dir);
426    // family path
427    Path tmpPath = new Path(dir, family(0));
428    // TMP_DIR under family path
429    tmpPath = new Path(tmpPath, BulkLoadHFilesTool.TMP_DIR);
430    FileSystem fs = dir.getFileSystem(util.getConfiguration());
431    // HFiles have been splitted, there is TMP_DIR
432    assertTrue(fs.exists(tmpPath));
433    // TMP_DIR should have been cleaned-up
434    assertNull(BulkLoadHFilesTool.TMP_DIR + " should be empty.", FSUtils.listStatus(fs, tmpPath));
435    assertExpectedTable(util.getConnection(), table, ROWCOUNT, 2);
436  }
