098  static int insertData(Configuration conf, TableName tableName, String column, double prob)
099      throws IOException {
100    Random rng = ThreadLocalRandom.current();
101    byte[] k = new byte[3];
102    byte [][] famAndQf = CellUtil.parseColumn(Bytes.toBytes(column));
103    List<Put> puts = new ArrayList<>();
104    for (byte b1 = 'a'; b1 < 'z'; b1++) {
105      for (byte b2 = 'a'; b2 < 'z'; b2++) {
106        for (byte b3 = 'a'; b3 < 'z'; b3++) {
107          if (rng.nextDouble() < prob) {
108            k[0] = b1;
109            k[1] = b2;
110            k[2] = b3;
111            Put put = new Put(k);
112            put.setDurability(Durability.SKIP_WAL);
113            put.addColumn(famAndQf[0], famAndQf[1], k);
114            puts.add(put);
115          }
116        }
117      }
118    }
119    try (Connection conn = ConnectionFactory.createConnection(conf);
120        Table table = conn.getTable(tableName)) {
121      table.put(puts);
122    }
123    return puts.size();
124  }

126  static int countCellSet(CellSetModel model) {
127    int count = 0;
128    Iterator<RowModel> rows = model.getRows().iterator();
129    while (rows.hasNext()) {
130      RowModel row = rows.next();
131      Iterator<CellModel> cells = row.getCells().iterator();
132      while (cells.hasNext()) {
133        cells.next();
134        count++;
135      }
136    }
137    return count;
138  }

140  private static int fullTableScan(ScannerModel model) throws IOException {
141    model.setBatch(100);
142    Response response = client.put("/" + TABLE + "/scanner",
143      Constants.MIMETYPE_PROTOBUF, model.createProtobufOutput());
144    assertEquals(201, response.getCode());
145    String scannerURI = response.getLocation();
146    assertNotNull(scannerURI);
147    int count = 0;
148    while (true) {
149      response = client.get(scannerURI, Constants.MIMETYPE_PROTOBUF);
150      assertTrue(response.getCode() == 200 || response.getCode() == 204);
151      if (response.getCode() == 200) {
152        assertEquals(Constants.MIMETYPE_PROTOBUF, response.getHeader("content-type"));
153        CellSetModel cellSet = new CellSetModel();
154        cellSet.getObjectFromMessage(response.getBody());
155        Iterator<RowModel> rows = cellSet.getRows().iterator();
156        while (rows.hasNext()) {
157          RowModel row = rows.next();
158          Iterator<CellModel> cells = row.getCells().iterator();
159          while (cells.hasNext()) {
160            cells.next();
161            count++;
162          }
163        }
164      } else {
165        break;
166      }
167    }
168    // delete the scanner
169    response = client.delete(scannerURI);
170    assertEquals(200, response.getCode());
171    return count;
172  }

222  @Test
223  public void testSimpleScannerXML() throws IOException, JAXBException {
224    final int BATCH_SIZE = 5;
225    // new scanner
226    ScannerModel model = new ScannerModel();
227    model.setBatch(BATCH_SIZE);
228    model.addColumn(Bytes.toBytes(COLUMN_1));
229    StringWriter writer = new StringWriter();
230    marshaller.marshal(model, writer);
231    byte[] body = Bytes.toBytes(writer.toString());
232
233    // test put operation is forbidden in read-only mode
234    conf.set("hbase.rest.readonly", "true");
235    Response response = client.put("/" + TABLE + "/scanner",
236      Constants.MIMETYPE_XML, body);
237    assertEquals(403, response.getCode());
238    String scannerURI = response.getLocation();
239    assertNull(scannerURI);
240
241    // recall previous put operation with read-only off
242    conf.set("hbase.rest.readonly", "false");
243    response = client.put("/" + TABLE + "/scanner", Constants.MIMETYPE_XML,
244      body);
245    assertEquals(201, response.getCode());
246    scannerURI = response.getLocation();
247    assertNotNull(scannerURI);
248
249    // get a cell set
250    response = client.get(scannerURI, Constants.MIMETYPE_XML);
251    assertEquals(200, response.getCode());
252    assertEquals(Constants.MIMETYPE_XML, response.getHeader("content-type"));
253    CellSetModel cellSet = (CellSetModel)
254      unmarshaller.unmarshal(new ByteArrayInputStream(response.getBody()));
255    // confirm batch size conformance
256    assertEquals(BATCH_SIZE, countCellSet(cellSet));
257
258    // test delete scanner operation is forbidden in read-only mode
259    conf.set("hbase.rest.readonly", "true");
260    response = client.delete(scannerURI);
261    assertEquals(403, response.getCode());
262
263    // recall previous delete scanner operation with read-only off
264    conf.set("hbase.rest.readonly", "false");
265    response = client.delete(scannerURI);
266    assertEquals(200, response.getCode());
267  }

331      Constants.MIMETYPE_PROTOBUF, model.createProtobufOutput());
332    assertEquals(201, response.getCode());
333    scannerURI = response.getLocation();
334    assertNotNull(scannerURI);
335
336    // get a cell
337    response = client.get(scannerURI, Constants.MIMETYPE_BINARY);
338    assertEquals(200, response.getCode());
339    assertEquals(Constants.MIMETYPE_BINARY, response.getHeader("content-type"));
340    // verify that data was returned
341    assertTrue(response.getBody().length > 0);
342    // verify that the expected X-headers are present
343    boolean foundRowHeader = false, foundColumnHeader = false,
344      foundTimestampHeader = false;
345    for (Header header: response.getHeaders()) {
346      if (header.getName().equals("X-Row")) {
347        foundRowHeader = true;
348      } else if (header.getName().equals("X-Column")) {
349        foundColumnHeader = true;
350      } else if (header.getName().equals("X-Timestamp")) {
351        foundTimestampHeader = true;
352      }
353    }
354    assertTrue(foundRowHeader);
355    assertTrue(foundColumnHeader);
356    assertTrue(foundTimestampHeader);
357
358    // test delete scanner operation is forbidden in read-only mode
359    conf.set("hbase.rest.readonly", "true");
360    response = client.delete(scannerURI);
361    assertEquals(403, response.getCode());
362
363    // recall previous delete scanner operation with read-only off

365    response = client.delete(scannerURI);
