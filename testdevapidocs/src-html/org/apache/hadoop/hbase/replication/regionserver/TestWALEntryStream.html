198  @Test
199  public void testAppendsWithRolls() throws Exception {
200    appendToLogAndSync();
201    long oldPos;
202    try (WALEntryStream entryStream =
203        new WALEntryStream(logQueue, CONF, 0, log, null, new MetricsSource("1"), fakeWalGroupId)) {
204      // There's one edit in the log, read it. Reading past it needs to throw exception
205      assertTrue(entryStream.hasNext());
206      WAL.Entry entry = entryStream.peek();
207      assertSame(entry, entryStream.next());
208      assertNotNull(entry);
209      assertFalse(entryStream.hasNext());
210      assertNull(entryStream.peek());
211      assertNull(entryStream.next());
212      oldPos = entryStream.getPosition();
213    }
214
215    appendToLogAndSync();
216
217    try (WALEntryStream entryStream = new WALEntryStream(logQueue, CONF, oldPos,
218        log, null, new MetricsSource("1"), fakeWalGroupId)) {
219      // Read the newly added entry, make sure we made progress
220      WAL.Entry entry = entryStream.next();
221      assertNotEquals(oldPos, entryStream.getPosition());
222      assertNotNull(entry);
223      oldPos = entryStream.getPosition();
224    }
225
226    // We rolled but we still should see the end of the first log and get that item
227    appendToLogAndSync();
228    log.rollWriter();
229    appendToLogAndSync();
230
231    try (WALEntryStream entryStream = new WALEntryStream(logQueue, CONF, oldPos,
232        log, null, new MetricsSource("1"), fakeWalGroupId)) {
233      WAL.Entry entry = entryStream.next();
234      assertNotEquals(oldPos, entryStream.getPosition());
235      assertNotNull(entry);
236
237      // next item should come from the new log
238      entry = entryStream.next();
239      assertNotEquals(oldPos, entryStream.getPosition());
240      assertNotNull(entry);
241
242      // no more entries to read
243      assertFalse(entryStream.hasNext());
244      oldPos = entryStream.getPosition();
245    }
246  }

249   * Tests that if after a stream is opened, more entries come in and then the log is rolled, we
250   * don't mistakenly dequeue the current log thinking we're done with it
251   */
252  @Test
253  public void testLogrollWhileStreaming() throws Exception {
254    appendToLog("1");
255    appendToLog("2");// 2
256    try (WALEntryStream entryStream =
257        new WALEntryStream(logQueue, CONF, 0, log, null,
258          new MetricsSource("1"), fakeWalGroupId)) {
259      assertEquals("1", getRow(entryStream.next()));
260
261      appendToLog("3"); // 3 - comes in after reader opened
262      log.rollWriter(); // log roll happening while we're reading
263      appendToLog("4"); // 4 - this append is in the rolled log
264
265      assertEquals("2", getRow(entryStream.next()));
266      assertEquals(2, getQueue().size()); // we should not have dequeued yet since there's still an
267                                        // entry in first log
268      assertEquals("3", getRow(entryStream.next())); // if implemented improperly, this would be 4
269                                                     // and 3 would be skipped
270      assertEquals("4", getRow(entryStream.next())); // 4
271      assertEquals(1, getQueue().size()); // now we've dequeued and moved on to next log properly
272      assertFalse(entryStream.hasNext());
273    }
274  }

617          System.currentTimeMillis(), qualifier));

654    long size = log.getLogFileSizeIfBeingWritten(getQueue().peek()).getAsLong();
655    AtomicLong fileLength = new AtomicLong(size - 1);
656    try (WALEntryStream entryStream = new WALEntryStream(logQueue,  CONF, 0,
657      p -> OptionalLong.of(fileLength.get()), null, new MetricsSource("1"), fakeWalGroupId)) {

659      assertNotNull(entryStream.next());

661      assertFalse(entryStream.hasNext());
662      Thread.sleep(1000);
663      entryStream.reset();
664      // still can not get log 2
665      assertFalse(entryStream.hasNext());
666
667      // can get log 2 now
668      fileLength.set(size);
669      entryStream.reset();
670      assertTrue(entryStream.hasNext());
671      assertNotNull(entryStream.next());
672
673      assertFalse(entryStream.hasNext());
674    }

757  private void appendEntries(WALProvider.Writer writer, int numEntries) throws IOException {
758    for (int i = 0; i < numEntries; i++) {
759      byte[] b = Bytes.toBytes(Integer.toString(i));
760      KeyValue kv = new KeyValue(b,b,b);
761      WALEdit edit = new WALEdit();
762      edit.add(kv);
763      WALKeyImpl key = new WALKeyImpl(b, TableName.valueOf(b), 0, 0,
764        HConstants.DEFAULT_CLUSTER_ID);
765      NavigableMap<byte[], Integer> scopes = new TreeMap<byte[], Integer>(Bytes.BYTES_COMPARATOR);
766      scopes.put(b, HConstants.REPLICATION_SCOPE_GLOBAL);
767      writer.append(new WAL.Entry(key, edit));
768      writer.sync(false);
769    }
770    writer.close();
771  }
