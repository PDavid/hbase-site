082/**
083 * Simple test of sink-side wal entry filter facility.
084 */
085@Category({ReplicationTests.class, SmallTests.class})
086public class TestWALEntrySinkFilter {
087
088  @ClassRule
089  public static final HBaseClassTestRule CLASS_RULE =
090      HBaseClassTestRule.forClass(TestWALEntrySinkFilter.class);
091
092  private static final Logger LOG = LoggerFactory.getLogger(TestReplicationSink.class);
093  @Rule public TestName name = new TestName();
094  static final int BOUNDARY = 5;
095  static final AtomicInteger UNFILTERED = new AtomicInteger();
096  static final AtomicInteger FILTERED = new AtomicInteger();
097
098  /**
099   * Implemetentation of Stoppable to pass into ReplicationSink.
100   */
101  private static Stoppable STOPPABLE = new Stoppable() {
102    private final AtomicBoolean stop = new AtomicBoolean(false);
103
104    @Override
105    public boolean isStopped() {
106      return this.stop.get();
107    }
108
109    @Override
110    public void stop(String why) {
111      LOG.info("STOPPING BECAUSE: " + why);
112      this.stop.set(true);
113    }
114  };

116  /**
117   * Test filter.
118   * Filter will filter out any write time that is <= 5 (BOUNDARY). We count how many items we
119   * filter out and we count how many cells make it through for distribution way down below in the
120   * Table#batch implementation. Puts in place a custom DevNullConnection so we can insert our
121   * counting Table.
122   * @throws IOException
123   */
124  @Test
125  public void testWALEntryFilter() throws IOException {
126    Configuration conf = HBaseConfiguration.create();
127    // Make it so our filter is instantiated on construction of ReplicationSink.
128    conf.setClass(WALEntrySinkFilter.WAL_ENTRY_FILTER_KEY,
129        IfTimeIsGreaterThanBOUNDARYWALEntrySinkFilterImpl.class, WALEntrySinkFilter.class);
130    conf.setClass("hbase.client.connection.impl", DevNullConnection.class,
131      Connection.class);
132    ReplicationSink sink = new ReplicationSink(conf, STOPPABLE);
133    // Create some dumb walentries.
134    List< org.apache.hadoop.hbase.shaded.protobuf.generated.AdminProtos.WALEntry > entries =
135        new ArrayList<>();
136    AdminProtos.WALEntry.Builder entryBuilder = AdminProtos.WALEntry.newBuilder();
137    // Need a tablename.
138    ByteString tableName =
139        ByteString.copyFromUtf8(TableName.valueOf(this.name.getMethodName()).toString());
140    // Add WALEdit Cells to Cells List. The way edits arrive at the sink is with protos
141    // describing the edit with all Cells from all edits aggregated in a single CellScanner.
142    final List<Cell> cells = new ArrayList<>();
143    int count = BOUNDARY * 2;
144    for(int i = 0; i < count; i++) {
145      byte [] bytes = Bytes.toBytes(i);
146      // Create a wal entry. Everything is set to the current index as bytes or int/long.
147      entryBuilder.clear();
148      entryBuilder.setKey(entryBuilder.getKeyBuilder().
149          setLogSequenceNumber(i).
150          setEncodedRegionName(ByteString.copyFrom(bytes)).
151          setWriteTime(i).
152          setTableName(tableName).build());
153      // Lets have one Cell associated with each WALEdit.
154      entryBuilder.setAssociatedCellCount(1);
155      entries.add(entryBuilder.build());
156      // We need to add a Cell per WALEdit to the cells array.
157      CellBuilder cellBuilder = CellBuilderFactory.create(CellBuilderType.DEEP_COPY);
158      // Make cells whose row, family, cell, value, and ts are == 'i'.
159      Cell cell = cellBuilder.
160          setRow(bytes).
161          setFamily(bytes).
162          setQualifier(bytes).
163          setType(Cell.Type.Put).
164          setTimestamp(i).
165          setValue(bytes).build();
166      cells.add(cell);
167    }
168    // Now wrap our cells array in a CellScanner that we can pass in to replicateEntries. It has
169    // all Cells from all the WALEntries made above.
170    CellScanner cellScanner = new CellScanner() {
171      // Set to -1 because advance gets called before current.
172      int index = -1;
173
174      @Override
175      public Cell current() {
176        return cells.get(index);
177      }
178
179      @Override
180      public boolean advance() throws IOException {
181        index++;
182        return index < cells.size();
183      }
184    };
185    // Call our sink.
186    sink.replicateEntries(entries, cellScanner, null, null, null);
187    // Check what made it through and what was filtered.
188    assertTrue(FILTERED.get() > 0);
189    assertTrue(UNFILTERED.get() > 0);
190    assertEquals(count, FILTERED.get() + UNFILTERED.get());
191  }

193  /**
194   * Simple filter that will filter out any entry wholse writeTime is <= 5.
195   */
196  public static class IfTimeIsGreaterThanBOUNDARYWALEntrySinkFilterImpl implements WALEntrySinkFilter {
197    public IfTimeIsGreaterThanBOUNDARYWALEntrySinkFilterImpl() {}
198
199    @Override
200    public void init(Connection connection) {
201      // Do nothing.
202    }
203
204    @Override
205    public boolean filter(TableName table, long writeTime) {
206      boolean b = writeTime <= BOUNDARY;
207      if (b) {
208        FILTERED.incrementAndGet();
209      }
210      return b;
211    }
212  }
