085  protected static void initTestingUtility(HBaseTestingUtility util, String zkParent) {
086    util.setZkCluster(ZK_UTIL.getZkCluster());
087    Configuration conf = util.getConfiguration();
088    conf.set(HConstants.ZOOKEEPER_ZNODE_PARENT, zkParent);
089    conf.setInt("replication.source.size.capacity", 102400);
090    conf.setLong("replication.source.sleepforretries", 100);
091    conf.setInt("hbase.regionserver.maxlogs", 10);
092    conf.setLong("hbase.master.logcleaner.ttl", 10);
093    conf.setInt("zookeeper.recovery.retry", 1);
094    conf.setInt("zookeeper.recovery.retry.intervalmill", 10);
095    conf.setLong(HConstants.THREAD_WAKE_FREQUENCY, 100);
096    conf.setInt("replication.stats.thread.period.seconds", 5);
097    conf.setBoolean("hbase.tests.use.shortcircuit.reads", false);
098    conf.setLong("replication.sleep.before.failover", 2000);
099    conf.setInt("replication.source.maxretriesmultiplier", 10);
100    conf.setFloat("replication.source.ratio", 1.0f);
101    conf.setBoolean("replication.source.eof.autorecovery", true);
102  }

104  @BeforeClass
105  public static void setUp() throws Exception {
106    ZK_UTIL.startMiniZKCluster();

108    initTestingUtility(UTIL2, "/cluster2");

110      StartMiniClusterOption.builder().numMasters(2).numRegionServers(3).numDataNodes(3).build();
111    UTIL1.startMiniCluster(option);
112    UTIL2.startMiniCluster(option);
113    TableDescriptor td =
114      TableDescriptorBuilder.newBuilder(TABLE_NAME).setColumnFamily(ColumnFamilyDescriptorBuilder
115        .newBuilder(CF).setScope(HConstants.REPLICATION_SCOPE_GLOBAL).build()).build();
116    UTIL1.getAdmin().createTable(td);
117    UTIL2.getAdmin().createTable(td);
118    FileSystem fs1 = UTIL1.getTestFileSystem();
119    FileSystem fs2 = UTIL2.getTestFileSystem();
120    REMOTE_WAL_DIR1 =
121      new Path(UTIL1.getMiniHBaseCluster().getMaster().getMasterFileSystem().getWALRootDir(),

123    REMOTE_WAL_DIR2 =

131    UTIL2.getAdmin().addReplicationPeer(PEER_ID,

138  private static void shutdown(HBaseTestingUtility util) throws Exception {
139    if (util.getHBaseCluster() == null) {
140      return;
141    }
142    Admin admin = util.getAdmin();
143    if (!admin.listReplicationPeers(Pattern.compile(PEER_ID)).isEmpty()) {
144      if (admin
145        .getReplicationPeerSyncReplicationState(PEER_ID) != SyncReplicationState.DOWNGRADE_ACTIVE) {
146        admin.transitReplicationPeerSyncReplicationState(PEER_ID,
147          SyncReplicationState.DOWNGRADE_ACTIVE);
148      }
149      admin.removeReplicationPeer(PEER_ID);
150    }
151    util.shutdownMiniCluster();
152  }
153
154  @AfterClass
155  public static void tearDown() throws Exception {
156    shutdown(UTIL1);
157    shutdown(UTIL2);
158    ZK_UTIL.shutdownMiniZKCluster();
159  }

161  protected final void write(HBaseTestingUtility util, int start, int end) throws IOException {
162    try (Table table = util.getConnection().getTable(TABLE_NAME)) {
163      for (int i = start; i < end; i++) {
164        table.put(new Put(Bytes.toBytes(i)).addColumn(CF, CQ, Bytes.toBytes(i)));
165      }
166    }
167  }
168
169  protected final void verify(HBaseTestingUtility util, int start, int end) throws IOException {
170    try (Table table = util.getConnection().getTable(TABLE_NAME)) {
171      for (int i = start; i < end; i++) {
172        assertEquals(i, Bytes.toInt(table.get(new Get(Bytes.toBytes(i))).getValue(CF, CQ)));
173      }
174    }
175  }

177  protected final void verifyThroughRegion(HBaseTestingUtility util, int start, int end)
178      throws IOException {
179    HRegion region = util.getMiniHBaseCluster().getRegions(TABLE_NAME).get(0);
180    for (int i = start; i < end; i++) {
181      assertEquals(i, Bytes.toInt(region.get(new Get(Bytes.toBytes(i))).getValue(CF, CQ)));
182    }
183  }
184
185  protected final void verifyNotReplicatedThroughRegion(HBaseTestingUtility util, int start,
186      int end) throws IOException {
187    HRegion region = util.getMiniHBaseCluster().getRegions(TABLE_NAME).get(0);
188    for (int i = start; i < end; i++) {
189      assertTrue(region.get(new Get(Bytes.toBytes(i))).isEmpty());
190    }
191  }

193  protected final void waitUntilReplicationDone(HBaseTestingUtility util, int end)
194      throws Exception {
195    // The reject check is in RSRpcService so we can still read through HRegion
196    HRegion region = util.getMiniHBaseCluster().getRegions(TABLE_NAME).get(0);
197    util.waitFor(30000, new ExplainingPredicate<Exception>() {
198
199      @Override
200      public boolean evaluate() throws Exception {
201        return !region.get(new Get(Bytes.toBytes(end - 1))).isEmpty();
202      }
203
204      @Override
205      public String explainFailure() throws Exception {
206        return "Replication has not been catched up yet";
207      }
208    });
209  }

211  protected final void writeAndVerifyReplication(HBaseTestingUtility util1,
212      HBaseTestingUtility util2, int start, int end) throws Exception {
213    write(util1, start, end);
214    waitUntilReplicationDone(util2, end);
215    verifyThroughRegion(util2, start, end);

231  protected final void verifyRemovedPeer(String peerId, Path remoteWALDir,
232      HBaseTestingUtility utility) throws Exception {
233    ReplicationPeerStorage rps = ReplicationStorageFactory
234      .getReplicationPeerStorage(utility.getZooKeeperWatcher(), utility.getConfiguration());
235    try {
236      rps.getPeerSyncReplicationState(peerId);
237      fail("Should throw exception when get the sync replication state of a removed peer.");
238    } catch (ReplicationException e) {
239      // ignore.
240    }
241    try {
242      rps.getPeerNewSyncReplicationState(peerId);
243      fail("Should throw exception when get the new sync replication state of a removed peer");
244    } catch (ReplicationException e) {
245      // ignore.
246    }
247    try (FileSystem fs = utility.getTestFileSystem()) {
248      assertFalse(fs.exists(getRemoteWALDir(remoteWALDir, peerId)));
249      assertFalse(fs.exists(getReplayRemoteWALs(remoteWALDir, peerId)));
250    }
251  }

253  private void assertRejection(Throwable error) {
254    assertThat(error, instanceOf(DoNotRetryIOException.class));
255    assertTrue(error.getMessage().contains("Reject to apply to sink cluster"));
256    assertTrue(error.getMessage().contains(TABLE_NAME.toString()));
257  }

259  protected final void verifyReplicationRequestRejection(HBaseTestingUtility utility,
260      boolean expectedRejection) throws Exception {
261    HRegionServer regionServer = utility.getRSForFirstRegionInTable(TABLE_NAME);
262    AsyncClusterConnection connection = regionServer.getAsyncClusterConnection();
263    Entry[] entries = new Entry[10];
264    for (int i = 0; i < entries.length; i++) {
265      entries[i] =
266        new Entry(new WALKeyImpl(HConstants.EMPTY_BYTE_ARRAY, TABLE_NAME, 0), new WALEdit());
267    }
268    if (!expectedRejection) {
269      ReplicationProtbufUtil.replicateWALEntry(
270        connection.getRegionServerAdmin(regionServer.getServerName()), entries, null, null, null,
271        HConstants.REPLICATION_SOURCE_SHIPEDITS_TIMEOUT_DFAULT);
272    } else {
273      try {
274        ReplicationProtbufUtil.replicateWALEntry(
275          connection.getRegionServerAdmin(regionServer.getServerName()), entries, null, null, null,
276          HConstants.REPLICATION_SOURCE_SHIPEDITS_TIMEOUT_DFAULT);
277        fail("Should throw IOException when sync-replication state is in A or DA");
278      } catch (RemoteException e) {
279        assertRejection(e.unwrapRemoteException());
280      } catch (DoNotRetryIOException e) {
281        assertRejection(e);
282      }
283    }
284  }
