154    final String cf3 = "cf3";

164
165    long procId = ProcedureTestingUtility.submitAndWait(

167    ProcedureTestingUtility.assertProcNotFailed(procExec.getResult(procId));
168
169    currentHtd = UTIL.getAdmin().getDescriptor(tableName);
170    assertEquals(2, currentHtd.getColumnFamilyNames().size());
171    assertFalse(currentHtd.hasColumnFamily(Bytes.toBytes(cf2)));
172
173    // Test 2: Modify the table descriptor offline
174    UTIL.getAdmin().disableTable(tableName);
175    ProcedureTestingUtility.waitNoProcedureRunning(procExec);
176
177    TableDescriptor htd2 = UTIL.getAdmin().getDescriptor(tableName);
178    // Disable Sanity check
179    htd2 = TableDescriptorBuilder.newBuilder(htd2).removeColumnFamily(Bytes.toBytes(cf3))
180      .setValue(TableDescriptorChecker.TABLE_SANITY_CHECKS, Boolean.FALSE.toString())
181      .build();
182
183    long procId2 =
184        ProcedureTestingUtility.submitAndWait(procExec,
185          new ModifyTableProcedure(procExec.getEnvironment(), htd2));
186    ProcedureTestingUtility.assertProcNotFailed(procExec.getResult(procId2));
187
188    currentHtd = UTIL.getAdmin().getDescriptor(tableName);
189    assertEquals(1, currentHtd.getColumnFamilyNames().size());
190    assertFalse(currentHtd.hasColumnFamily(Bytes.toBytes(cf3)));
191
192    // Removing the last family will fail
193    TableDescriptor htd3 = UTIL.getAdmin().getDescriptor(tableName);
194    htd3 = TableDescriptorBuilder.newBuilder(htd3).removeColumnFamily(Bytes.toBytes(cf1)).build();
195    long procId3 =
196        ProcedureTestingUtility.submitAndWait(procExec,
197            new ModifyTableProcedure(procExec.getEnvironment(), htd3));
198    final Procedure<?> result = procExec.getResult(procId3);
199    assertEquals(true, result.isFailed());
200    Throwable cause = ProcedureTestingUtility.getExceptionCause(result);
201    assertTrue("expected DoNotRetryIOException, got " + cause,
202        cause instanceof DoNotRetryIOException);
203    assertEquals(1, currentHtd.getColumnFamilyNames().size());
204    assertTrue(currentHtd.hasColumnFamily(Bytes.toBytes(cf1)));
205  }

290  @Test
291  public void testColumnFamilyAdditionTwiceWithNonce() throws Exception {
292    final TableName tableName = TableName.valueOf(name.getMethodName());
293    final String cf2 = "cf2";
294    final String cf3 = "cf3";
295    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
296
297    // create the table
298    RegionInfo[] regions =
299        MasterProcedureTestingUtility.createTable(procExec, tableName, null, "cf1", cf3);
300
301    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, true);
302    // Modify multiple properties of the table.
303    TableDescriptor td = UTIL.getAdmin().getDescriptor(tableName);
304    TableDescriptor newTd =
305        TableDescriptorBuilder.newBuilder(td).setCompactionEnabled(!td.isCompactionEnabled())
306          .setColumnFamily(ColumnFamilyDescriptorBuilder.of(cf2)).build();
307
308    PerClientRandomNonceGenerator nonceGenerator = PerClientRandomNonceGenerator.get();
309    long nonceGroup = nonceGenerator.getNonceGroup();
310    long newNonce = nonceGenerator.newNonce();
311    NonceKey nonceKey = new NonceKey(nonceGroup, newNonce);
312    procExec.registerNonce(nonceKey);
313
314    // Start the Modify procedure && kill the executor
315    final long procId = procExec
316        .submitProcedure(new ModifyTableProcedure(procExec.getEnvironment(), newTd), nonceKey);
317
318    // Restart the executor after MODIFY_TABLE_UPDATE_TABLE_DESCRIPTOR and try to add column family
319    // as nonce are there , we should not fail
320    MasterProcedureTestingUtility.testRecoveryAndDoubleExecution(procExec, procId, new StepHook() {
321      @Override
322      public boolean execute(int step) throws IOException {
323        if (step == 3) {
324          return procId == UTIL.getHBaseCluster().getMaster().addColumn(tableName,
325            ColumnFamilyDescriptorBuilder.newBuilder(Bytes.toBytes(cf2)).build(), nonceGroup,
326            newNonce);
327        }
328        return true;
329      }
330    });
331
332    //Try with different nonce, now it should fail the checks
333    try {
334      UTIL.getHBaseCluster().getMaster().addColumn(tableName,
335        ColumnFamilyDescriptorBuilder.newBuilder(Bytes.toBytes(cf2)).build(), nonceGroup,
336        nonceGenerator.newNonce());
337      Assert.fail();
338    } catch (InvalidFamilyOperationException e) {
339    }
340
341    // Validate descriptor
342    TableDescriptor currentHtd = UTIL.getAdmin().getDescriptor(tableName);
343    assertEquals(!td.isCompactionEnabled(), currentHtd.isCompactionEnabled());
344    assertEquals(3, currentHtd.getColumnFamilyCount());
345    assertTrue(currentHtd.hasColumnFamily(Bytes.toBytes(cf2)));
346    assertTrue(currentHtd.hasColumnFamily(Bytes.toBytes(cf3)));
347
348    // cf2 should be added
349    MasterProcedureTestingUtility.validateTableCreation(UTIL.getHBaseCluster().getMaster(),
350      tableName, regions, "cf1", cf2, cf3);
351  }

417    final TableName tableName = TableName.valueOf(name.getMethodName());
418    UTIL.createTable(tableName, column_Family1);
419
420    class ConcurrentAddColumnFamily extends Thread {
421      TableName tableName = null;
422      ColumnFamilyDescriptor columnFamilyDescriptor;
423      boolean exception;
424
425      public ConcurrentAddColumnFamily(TableName tableName,
426          ColumnFamilyDescriptor columnFamilyDescriptor) {
427        this.tableName = tableName;
428        this.columnFamilyDescriptor = columnFamilyDescriptor;
429        this.exception = false;
430      }
431
432      public void run() {
433        try {
434          UTIL.getAdmin().addColumnFamily(tableName, columnFamilyDescriptor);

436          if (e.getClass().equals(ConcurrentTableModificationException.class)) {
437            this.exception = true;
438          }
439        }
440      }
441    }
442    ColumnFamilyDescriptor columnFamilyDescriptor =
443      ColumnFamilyDescriptorBuilder.newBuilder(Bytes.toBytes(column_Family2)).build();

446    columnFamilyDescriptor =
447      ColumnFamilyDescriptorBuilder.newBuilder(Bytes.toBytes(column_Family3)).build();
448    ConcurrentAddColumnFamily t2 =
449      new ConcurrentAddColumnFamily(tableName, columnFamilyDescriptor);
450
451    t1.start();
452    t2.start();
453
454    t1.join();
455    t2.join();
456    int noOfColumnFamilies = UTIL.getAdmin().getDescriptor(tableName).getColumnFamilies().length;
457    assertTrue("Expected ConcurrentTableModificationException.",
458      ((t1.exception || t2.exception) && noOfColumnFamilies == 2) || noOfColumnFamilies == 3);
