125  @Test
126  public void testMasterRestart() throws Exception {
127    ProcedureExecutor<MasterProcedureEnv> procExec = master.getMasterProcedureExecutor();
128    MasterProcedureEnv env = procExec.getEnvironment();
129    SnapshotProcedure sp = new SnapshotProcedure(env, snapshotProto);
130    SnapshotProcedure spySp = getDelayedOnSpecificStateSnapshotProcedure(sp,
131      procExec.getEnvironment(), SnapshotState.SNAPSHOT_SNAPSHOT_ONLINE_REGIONS);
132
133    long procId = procExec.submitProcedure(spySp);
134
135    TEST_UTIL.waitFor(2000, () -> env.getMasterServices().getProcedures()
136      .stream().map(Procedure::getProcId).collect(Collectors.toList()).contains(procId));
137    TEST_UTIL.getHBaseCluster().killMaster(master.getServerName());
138    TEST_UTIL.getHBaseCluster().waitForMasterToStop(master.getServerName(), 30000);
139    TEST_UTIL.getHBaseCluster().startMaster();
140    TEST_UTIL.getHBaseCluster().waitForActiveAndReadyMaster();
141
142    master = TEST_UTIL.getHBaseCluster().getMaster();
143    assertTrue(master.getSnapshotManager().isTakingAnySnapshot());
144    assertTrue(master.getSnapshotManager().isTableTakingAnySnapshot(TABLE_NAME));
145
146    List<SnapshotProcedure> unfinishedProcedures = master
147      .getMasterProcedureExecutor().getProcedures().stream()
148      .filter(p -> p instanceof SnapshotProcedure)
149      .filter(p -> !p.isFinished()).map(p -> (SnapshotProcedure) p)
150      .collect(Collectors.toList());
151    assertEquals(unfinishedProcedures.size(), 1);
152    long newProcId = unfinishedProcedures.get(0).getProcId();
153    assertEquals(procId, newProcId);
154
155    ProcedureTestingUtility.waitProcedure(master.getMasterProcedureExecutor(), newProcId);
156    assertFalse(master.getSnapshotManager().isTableTakingAnySnapshot(TABLE_NAME));
157
158    List<SnapshotProtos.SnapshotDescription> snapshots
159      = master.getSnapshotManager().getCompletedSnapshots();
160    assertEquals(1, snapshots.size());
161    assertEquals(SNAPSHOT_NAME, snapshots.get(0).getName());
162    assertEquals(TABLE_NAME, TableName.valueOf(snapshots.get(0).getTable()));
163    SnapshotTestingUtils.confirmSnapshotValid(TEST_UTIL, snapshotProto, TABLE_NAME, CF);
164  }

208  public <T extends Procedure<MasterProcedureEnv>> T waitProcedureRunnableAndGetFirst(
209    Class<T> clazz, long timeout) throws IOException {
210    TEST_UTIL.waitFor(timeout, () -> master.getProcedures().stream()
211      .anyMatch(clazz::isInstance));
212    Optional<T> procOpt =  master.getMasterProcedureExecutor().getProcedures().stream()
213      .filter(clazz::isInstance).map(clazz::cast).findFirst();
214    assertTrue(procOpt.isPresent());
215    return procOpt.get();
216  }

303  @Test
304  public void testRunningTwoSnapshotProcedureOnSameTable() throws Exception {
305    String newSnapshotName = SNAPSHOT_NAME + "_2";
306    SnapshotProtos.SnapshotDescription snapshotProto2 = SnapshotProtos.SnapshotDescription
307      .newBuilder(snapshotProto).setName(newSnapshotName).build();
308
309    ProcedureExecutor<MasterProcedureEnv> procExec = master.getMasterProcedureExecutor();
310    MasterProcedureEnv env = procExec.getEnvironment();
311
312    SnapshotProcedure sp1 = new SnapshotProcedure(env, snapshotProto);
313    SnapshotProcedure sp2 = new SnapshotProcedure(env, snapshotProto2);
314    SnapshotProcedure spySp1 = getDelayedOnSpecificStateSnapshotProcedure(sp1,
315      procExec.getEnvironment(), SnapshotState.SNAPSHOT_SNAPSHOT_ONLINE_REGIONS);
316    SnapshotProcedure spySp2 = getDelayedOnSpecificStateSnapshotProcedure(sp2,
317      procExec.getEnvironment(), SnapshotState.SNAPSHOT_SNAPSHOT_ONLINE_REGIONS);
318
319    long procId1 = procExec.submitProcedure(spySp1);
320    long procId2 = procExec.submitProcedure(spySp2);
321    TEST_UTIL.waitFor(2000, () -> env.getMasterServices().getProcedures()
322      .stream().map(Procedure::getProcId).collect(Collectors.toList())
323      .containsAll(Arrays.asList(procId1, procId2)));
324
325    assertFalse(procExec.isFinished(procId1));
326    assertFalse(procExec.isFinished(procId2));
327
328    ProcedureTestingUtility.waitProcedure(master.getMasterProcedureExecutor(), procId1);
329    ProcedureTestingUtility.waitProcedure(master.getMasterProcedureExecutor(), procId2);
330
331    List<SnapshotProtos.SnapshotDescription> snapshots =
332      master.getSnapshotManager().getCompletedSnapshots();
333    assertEquals(2, snapshots.size());
334    snapshots.sort(Comparator.comparing(SnapshotProtos.SnapshotDescription::getName));
335    assertEquals(SNAPSHOT_NAME, snapshots.get(0).getName());
336    assertEquals(newSnapshotName, snapshots.get(1).getName());
337    SnapshotTestingUtils.confirmSnapshotValid(TEST_UTIL, snapshotProto, TABLE_NAME, CF);
338    SnapshotTestingUtils.confirmSnapshotValid(TEST_UTIL, snapshotProto2, TABLE_NAME, CF);
339  }

360  @Test
361  public void testSnapshotCorruptedAndRollback() throws Exception {
362    ProcedureExecutor<MasterProcedureEnv> procExec = master.getMasterProcedureExecutor();
363    SnapshotProcedure sp = new SnapshotProcedure(procExec.getEnvironment(), snapshotProto);
364    procExec.submitProcedure(sp);
365    TEST_UTIL.waitFor(60000, 500, () -> sp.getCurrentStateId() >
366      SnapshotState.SNAPSHOT_CONSOLIDATE_SNAPSHOT_VALUE);
367    DistributedFileSystem dfs = TEST_UTIL.getDFSCluster().getFileSystem();
368    Optional<HRegion> region = TEST_UTIL.getHBaseCluster().getRegions(TABLE_NAME).stream()
369      .filter(r -> !r.getStoreFileList(new byte[][] { CF }).isEmpty())
370      .findFirst();
371    assertTrue(region.isPresent());
372    region.get().getStoreFileList(new byte[][] { CF }).forEach(s -> {
373        try {
374          // delete real data files to trigger the CorruptedSnapshotException
375          dfs.delete(new Path(s), true);
376          LOG.info("delete {} to make snapshot corrupt", s);
377        } catch (Exception e) {
378          LOG.warn("Failed delete {} to make snapshot corrupt", s, e);
379        }
380      }
381    );
382    TEST_UTIL.waitFor(60000, () -> sp.isFailed() && sp.isFinished());
383    Configuration conf = master.getConfiguration();
384    Path workingDir = SnapshotDescriptionUtils.getWorkingSnapshotDir(
385      snapshotProto, CommonFSUtils.getRootDir(conf), conf);
386    assertFalse(dfs.exists(workingDir));
387    assertFalse(master.getSnapshotManager().isTakingSnapshot(TABLE_NAME));
388    assertFalse(master.getSnapshotManager().isTakingAnySnapshot());
389  }

392  private SnapshotProcedure getDelayedOnSpecificStateSnapshotProcedure(
393    SnapshotProcedure sp, MasterProcedureEnv env, SnapshotState state)
394    throws ProcedureSuspendedException, ProcedureYieldException, InterruptedException {
395    SnapshotProcedure spySp = Mockito.spy(sp);
396    Mockito.doAnswer(new AnswersWithDelay(60000, new Answer<Object>() {
397      @Override
398      public Object answer(InvocationOnMock invocation) throws Throwable {
399        return invocation.callRealMethod();
400      }
401    })).when(spySp).executeFromState(env, state);
402    return spySp;
403  }
