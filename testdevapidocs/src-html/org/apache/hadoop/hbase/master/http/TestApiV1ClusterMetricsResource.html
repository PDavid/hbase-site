058/**
059 * Tests for the master api_v1 {@link ClusterMetricsResource}.
060 */
061@Category({ MasterTests.class, LargeTests.class})
062public class TestApiV1ClusterMetricsResource {
063
064  @ClassRule
065  public static final HBaseClassTestRule CLASS_RULE =
066    HBaseClassTestRule.forClass(TestApiV1ClusterMetricsResource.class);
067
068  private static final MiniClusterRule miniClusterRule = MiniClusterRule.newBuilder()
069    .setMiniClusterOption(StartTestingClusterOption.builder()
070      .numZkServers(3)
071      .numMasters(3)
072      .numDataNodes(3)
073      .build())
074    .setConfiguration(() -> {
075      // enable Master InfoServer and random port selection
076      final Configuration conf = HBaseConfiguration.create();
077      conf.setInt(HConstants.MASTER_INFO_PORT, 0);
078      conf.set("hbase.http.jersey.tracing.type", "ON_DEMAND");
079      return conf;
080    })
081    .build();
082  private static final ConnectionRule connectionRule =
083    new ConnectionRule(miniClusterRule::createConnection);
084  private static final ClassSetup classRule = new ClassSetup(connectionRule::getConnection);
085
086  private static final class ClassSetup extends ExternalResource {
087
088    private final Supplier<AsyncConnection> connectionSupplier;
089    private final TableName tableName;
090    private AsyncAdmin admin;
091    private WebTarget target;
092
093    public ClassSetup(final Supplier<AsyncConnection> connectionSupplier) {
094      this.connectionSupplier = connectionSupplier;
095      tableName = TableName.valueOf(TestApiV1ClusterMetricsResource.class.getSimpleName());
096    }
097
098    public WebTarget getTarget() {
099      return target;
100    }
101
102    @Override
103    protected void before() throws Throwable {
104      final AsyncConnection conn = connectionSupplier.get();
105      admin = conn.getAdmin();
106      final TableDescriptor tableDescriptor = TableDescriptorBuilder
107        .newBuilder(tableName)
108        .setColumnFamily(
109          ColumnFamilyDescriptorBuilder.newBuilder(Bytes.toBytes("c")).build())
110        .setDurability(Durability.SKIP_WAL)
111        .build();
112      admin.createTable(tableDescriptor).get();
113
114      final String baseUrl = admin.getMaster()
115        .thenApply(ServerName::getHostname)
116        .thenCombine(
117          admin.getMasterInfoPort(),
118          (hostName, infoPort) -> "http://" + hostName + ":" + infoPort)
119        .get();
120      final Client client = ClientBuilder.newClient();
121      target = client.target(baseUrl).path("api/v1/cluster_metrics");
122    }
123
124    @Override
125    protected void after() {
126      final TableName tableName = TableName.valueOf("test");
127      try {
128        admin.tableExists(tableName)
129          .thenCompose(val -> {
130            if (val) {
131              return admin.disableTable(tableName)
132                .thenCompose(ignored -> admin.deleteTable(tableName));
133            } else {
134              return CompletableFuture.completedFuture(null);
135            }
136          })
137          .get();
138      } catch (Exception e) {
139        throw new RuntimeException(e);
140      }
141    }
142  }
143
144  @ClassRule
145  public static RuleChain ruleChain = RuleChain.outerRule(miniClusterRule)
146    .around(connectionRule)
147    .around(classRule);
148
149  @Test
150  public void testGetRoot() {
151    final String response = classRule.getTarget()
152      .request(MediaType.APPLICATION_JSON_TYPE)
153      .header("X-Jersey-Tracing-Accept", true)
154      .get(String.class);
155    assertThat(response, allOf(
156      containsString("\"hbase_version\":"),
157      containsString("\"cluster_id\":"),
158      containsString("\"master_name\":"),
159      containsString("\"backup_master_names\":")));
160  }
161

165      .request(MediaType.TEXT_HTML_TYPE)
166      .header("X-Jersey-Tracing-Accept", true)
167      .get(String.class));
168  }
