075  /**
076   * Test the load balancing algorithm.
077   *
078   * Invariant is that all servers should be hosting either floor(average) or
079   * ceiling(average) at both table level and cluster level
080   */
081  @Test
082  public void testBalanceClusterOverall() throws Exception {
083    Map<TableName, Map<ServerName, List<RegionInfo>>> clusterLoad = new TreeMap<>();
084    for (int[] mockCluster : clusterStateMocks) {
085      Map<ServerName, List<RegionInfo>> clusterServers = mockClusterServers(mockCluster, 30);
086      List<ServerAndLoad> clusterList = convertToList(clusterServers);
087      clusterLoad.put(TableName.valueOf(name.getMethodName()), clusterServers);
088      HashMap<TableName, TreeMap<ServerName, List<RegionInfo>>> result =
089          mockClusterServersWithTables(clusterServers);
090      loadBalancer.setClusterLoad(clusterLoad);
091      List<RegionPlan> clusterplans = new ArrayList<>();
092      for (Map.Entry<TableName, TreeMap<ServerName, List<RegionInfo>>> mapEntry : result
093          .entrySet()) {
094        TableName tableName = mapEntry.getKey();
095        TreeMap<ServerName, List<RegionInfo>> servers = mapEntry.getValue();
096        List<ServerAndLoad> list = convertToList(servers);
097        LOG.info("Mock Cluster : " + printMock(list) + " " + printStats(list));
098        List<RegionPlan> partialplans = loadBalancer.balanceTable(tableName, servers);
099        if(partialplans != null) clusterplans.addAll(partialplans);
100        List<ServerAndLoad> balancedClusterPerTable = reconcile(list, partialplans, servers);
101        LOG.info("Mock Balance : " + printMock(balancedClusterPerTable));
102        assertClusterAsBalanced(balancedClusterPerTable);
103        for (Map.Entry<ServerName, List<RegionInfo>> entry : servers.entrySet()) {
104          returnRegions(entry.getValue());
105          returnServer(entry.getKey());
106        }
107      }
108      List<ServerAndLoad> balancedCluster = reconcile(clusterList, clusterplans, clusterServers);
109      assertTrue(assertClusterOverallAsBalanced(balancedCluster, result.keySet().size()));
110    }
111  }

131  private void testImpactOfBalanceClusterOverall(boolean useLoadOfAllTable) throws Exception {
132    Map<TableName, Map<ServerName, List<RegionInfo>>> clusterLoad = new TreeMap<>();
133    Map<ServerName, List<RegionInfo>> clusterServers =
134        mockUniformClusterServers(mockUniformCluster);
135    List<ServerAndLoad> clusterList = convertToList(clusterServers);
136    clusterLoad.put(TableName.valueOf(name.getMethodName()), clusterServers);
137    // use overall can achieve both table and cluster level balance
138    HashMap<TableName, TreeMap<ServerName, List<RegionInfo>>> LoadOfAllTable =
139        mockClusterServersWithTables(clusterServers);
140    if (useLoadOfAllTable) {
141      loadBalancer.setClusterLoad((Map) LoadOfAllTable);
142    } else {
143      loadBalancer.setClusterLoad(clusterLoad);
144    }
145    List<RegionPlan> clusterplans1 = new ArrayList<RegionPlan>();
146    for (Map.Entry<TableName, TreeMap<ServerName, List<RegionInfo>>> mapEntry : LoadOfAllTable
147        .entrySet()) {
148      TableName tableName = mapEntry.getKey();
149      TreeMap<ServerName, List<RegionInfo>> servers = mapEntry.getValue();
150      List<ServerAndLoad> list = convertToList(servers);
151      LOG.info("Mock Cluster : " + printMock(list) + " " + printStats(list));
152      List<RegionPlan> partialplans = loadBalancer.balanceTable(tableName, servers);
153      if (partialplans != null) clusterplans1.addAll(partialplans);
154      List<ServerAndLoad> balancedClusterPerTable = reconcile(list, partialplans, servers);
155      LOG.info("Mock Balance : " + printMock(balancedClusterPerTable));
156      assertClusterAsBalanced(balancedClusterPerTable);
157      for (Map.Entry<ServerName, List<RegionInfo>> entry : servers.entrySet()) {
158        returnRegions(entry.getValue());
159        returnServer(entry.getKey());
160      }
161    }
162    List<ServerAndLoad> balancedCluster1 = reconcile(clusterList, clusterplans1, clusterServers);
163    assertTrue(assertClusterOverallAsBalanced(balancedCluster1, LoadOfAllTable.keySet().size()));
164  }

166  @Test
167  public void testBalanceClusterOverallStrictly() throws Exception {
168    int[] regionNumOfTable1PerServer = { 3, 3, 4, 4, 4, 4, 5, 5, 5 };
169    int[] regionNumOfTable2PerServer = { 2, 2, 2, 2, 2, 2, 2, 2, 1 };
170    TreeMap<ServerName, List<RegionInfo>> serverRegionInfo = new TreeMap<>();
171    List<ServerAndLoad> serverAndLoads = new ArrayList<>();
172    for (int i = 0; i < regionNumOfTable1PerServer.length; i++) {
173      ServerName serverName = ServerName.valueOf("server" + i, 1000, -1);
174      List<RegionInfo> regions1 =
175          createRegions(regionNumOfTable1PerServer[i], TableName.valueOf("table1"));
176      List<RegionInfo> regions2 =
177          createRegions(regionNumOfTable2PerServer[i], TableName.valueOf("table2"));
178      regions1.addAll(regions2);
179      serverRegionInfo.put(serverName, regions1);
180      ServerAndLoad serverAndLoad = new ServerAndLoad(serverName,
181          regionNumOfTable1PerServer[i] + regionNumOfTable2PerServer[i]);
182      serverAndLoads.add(serverAndLoad);
183    }
184    HashMap<TableName, TreeMap<ServerName, List<RegionInfo>>> LoadOfAllTable =
185        mockClusterServersWithTables(serverRegionInfo);
186    loadBalancer.setClusterLoad((Map) LoadOfAllTable);
187    List<RegionPlan> partialplans = loadBalancer.balanceTable(TableName.valueOf("table1"),
188      LoadOfAllTable.get(TableName.valueOf("table1")));
189    List<ServerAndLoad> balancedServerLoads =
190        reconcile(serverAndLoads, partialplans, serverRegionInfo);
191    for (ServerAndLoad serverAndLoad : balancedServerLoads) {
192      assertEquals(6, serverAndLoad.getLoad());
193    }
194  }
