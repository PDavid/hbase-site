160  /**
161   * Invariant is that all servers have between floor(avg) and ceiling(avg)
162   * number of regions.
163   */
164  public void assertClusterAsBalanced(List<ServerAndLoad> servers) {
165    int numServers = servers.size();
166    int numRegions = 0;
167    int maxRegions = 0;
168    int minRegions = Integer.MAX_VALUE;
169    for (ServerAndLoad server : servers) {
170      int nr = server.getLoad();
171      if (nr > maxRegions) {
172        maxRegions = nr;
173      }
174      if (nr < minRegions) {
175        minRegions = nr;
176      }
177      numRegions += nr;
178    }
179    if (maxRegions - minRegions < 2) {
180      // less than 2 between max and min, can't balance
181      return;
182    }
183    int min = numRegions / numServers;
184    int max = numRegions % numServers == 0 ? min : min + 1;
185
186    for (ServerAndLoad server : servers) {
187      assertTrue("All servers should have a positive load. " + server, server.getLoad() >= 0);
188      assertTrue("All servers should have load no more than " + max + ". " + server,
189          server.getLoad() <= max);
190      assertTrue("All servers should have load no less than " + min + ". " + server,
191          server.getLoad() >= min);
192    }
193  }

195  /**
196   * Invariant is that all servers have between acceptable range
197   * number of regions.
198   */
199  public boolean assertClusterOverallAsBalanced(List<ServerAndLoad> servers, int tablenum) {
200    int numServers = servers.size();
201    int numRegions = 0;
202    int maxRegions = 0;
203    int minRegions = Integer.MAX_VALUE;
204    for (ServerAndLoad server : servers) {
205      int nr = server.getLoad();
206      if (nr > maxRegions) {
207        maxRegions = nr;
208      }
209      if (nr < minRegions) {
210        minRegions = nr;
211      }
212      numRegions += nr;
213    }
214    if (maxRegions - minRegions < 2) {
215      // less than 2 between max and min, can't balance
216      return true;
217    }
218    int min = numRegions / numServers;
219    int max = numRegions % numServers == 0 ? min : min + 1;
220
221    for (ServerAndLoad server : servers) {
222      // The '5' in below is arbitrary.
223      if (server.getLoad() < 0 || server.getLoad() > max + (tablenum/2 + 5)  ||
224          server.getLoad() < (min - tablenum/2 - 5)) {
225        LOG.warn("server={}, load={}, max={}, tablenum={}, min={}",
226          server.getServerName(), server.getLoad(), max, tablenum, min);
227        return false;
228      }
229    }
230    return true;
231  }

233  /**
234   * Checks whether region replicas are not hosted on the same host.
235   */
236  public void assertRegionReplicaPlacement(Map<ServerName, List<RegionInfo>> serverMap, RackManager rackManager) {
237    TreeMap<String, Set<RegionInfo>> regionsPerHost = new TreeMap<>();
238    TreeMap<String, Set<RegionInfo>> regionsPerRack = new TreeMap<>();
239
240    for (Map.Entry<ServerName, List<RegionInfo>> entry : serverMap.entrySet()) {
241      String hostname = entry.getKey().getHostname();
242      Set<RegionInfo> infos = regionsPerHost.get(hostname);
243      if (infos == null) {
244        infos = new HashSet<>();
245        regionsPerHost.put(hostname, infos);
246      }
247
248      for (RegionInfo info : entry.getValue()) {
249        RegionInfo primaryInfo = RegionReplicaUtil.getRegionInfoForDefaultReplica(info);
250        if (!infos.add(primaryInfo)) {
251          Assert.fail("Two or more region replicas are hosted on the same host after balance");
252        }
253      }
254    }
255
256    if (rackManager == null) {
257      return;
258    }
259
260    for (Map.Entry<ServerName, List<RegionInfo>> entry : serverMap.entrySet()) {
261      String rack = rackManager.getRack(entry.getKey());
262      Set<RegionInfo> infos = regionsPerRack.get(rack);
263      if (infos == null) {
264        infos = new HashSet<>();
265        regionsPerRack.put(rack, infos);
266      }
267
268      for (RegionInfo info : entry.getValue()) {
269        RegionInfo primaryInfo = RegionReplicaUtil.getRegionInfoForDefaultReplica(info);
270        if (!infos.add(primaryInfo)) {
271          Assert.fail("Two or more region replicas are hosted on the same rack after balance");
272        }
273      }
274    }
275  }

277  protected String printStats(List<ServerAndLoad> servers) {
278    int numServers = servers.size();
279    int totalRegions = 0;
280    for (ServerAndLoad server : servers) {
281      totalRegions += server.getLoad();
282    }
283    float average = (float) totalRegions / numServers;
284    int max = (int) Math.ceil(average);
285    int min = (int) Math.floor(average);
286    return "[srvr=" + numServers + " rgns=" + totalRegions + " avg=" + average + " max=" + max
287        + " min=" + min + "]";
288  }

315  /**
316   * This assumes the RegionPlan HSI instances are the same ones in the map, so
317   * actually no need to even pass in the map, but I think it's clearer.
318   * @return a list of all added {@link ServerAndLoad} values.
319   */
320  protected List<ServerAndLoad> reconcile(List<ServerAndLoad> list, List<RegionPlan> plans,
321    Map<ServerName, List<RegionInfo>> servers) {
322    List<ServerAndLoad> result = new ArrayList<>(list.size());
323
324    Map<ServerName, ServerAndLoad> map = new HashMap<>(list.size());
325    for (ServerAndLoad sl : list) {
326      map.put(sl.getServerName(), sl);
327    }
328    if (plans != null) {
329      for (RegionPlan plan : plans) {
330        ServerName source = plan.getSource();
331
332        updateLoad(map, source, -1);
333        ServerName destination = plan.getDestination();
334        updateLoad(map, destination, +1);
335
336        servers.get(source).remove(plan.getRegionInfo());
337        servers.get(destination).add(plan.getRegionInfo());
338      }
339    }
340    result.clear();
341    result.addAll(map.values());
342    return result;
343  }

345  protected void updateLoad(final Map<ServerName, ServerAndLoad> map,
346                            final ServerName sn,
347                            final int diff) {
348    ServerAndLoad sal = map.get(sn);
349    if (sal == null) sal = new ServerAndLoad(sn, 0);
350    sal = new ServerAndLoad(sn, sal.getLoad() + diff);
351    map.put(sn, sal);
352  }

388    HashMap<TableName, TreeMap<ServerName, List<RegionInfo>>> result = new HashMap<>();
389    for (Map.Entry<ServerName, List<RegionInfo>> entry : clusterServers.entrySet()) {
390      ServerName sal = entry.getKey();
391      List<RegionInfo> regions = entry.getValue();
392      for (RegionInfo hri : regions){
393        TreeMap<ServerName, List<RegionInfo>> servers = result.get(hri.getTable());
394        if (servers == null) {
395          servers = new TreeMap<>();
396          result.put(hri.getTable(), servers);
397        }
398        List<RegionInfo> hrilist = servers.get(sal);
399        if (hrilist == null) {
400          hrilist = new ArrayList<>();
401          servers.put(sal, hrilist);
402        }
403        hrilist.add(hri);
404      }
405    }
406    for(Map.Entry<TableName, TreeMap<ServerName, List<RegionInfo>>> entry : result.entrySet()){
407      for(ServerName srn : clusterServers.keySet()){
408        if (!entry.getValue().containsKey(srn)) entry.getValue().put(srn, new ArrayList<>());
409      }
410    }
411    return result;
412  }

420  protected List<RegionInfo> createRegions(int numRegions, TableName tableName) {
421    List<RegionInfo> regions = new ArrayList<>(numRegions);
422    byte[] start = new byte[16];
423    Bytes.random(start);
424    byte[] end = new byte[16];
425    Bytes.random(end);
426    for (int i = 0; i < numRegions; i++) {
427      Bytes.putInt(start, 0, numRegions << 1);
428      Bytes.putInt(end, 0, (numRegions << 1) + 1);
429      RegionInfo hri = RegionInfoBuilder.newBuilder(tableName)
430        .setStartKey(start)
431        .setEndKey(end)
432        .setSplit(false)
433        .build();
434      regions.add(hri);
435    }
436    return regions;
437  }

439  protected List<RegionInfo> randomRegions(int numRegions, int numTables) {
440    List<RegionInfo> regions = new ArrayList<>(numRegions);
441    byte[] start = new byte[16];
442    Bytes.random(start);
443    byte[] end = new byte[16];
444    Bytes.random(end);
445    for (int i = 0; i < numRegions; i++) {
446      if (!regionQueue.isEmpty()) {
447        regions.add(regionQueue.poll());
448        continue;
449      }
450      Bytes.putInt(start, 0, numRegions << 1);
451      Bytes.putInt(end, 0, (numRegions << 1) + 1);
452      TableName tableName = TableName.valueOf("table" +
453        (numTables > 0 ? ThreadLocalRandom.current().nextInt(numTables) : i));
454      RegionInfo hri = RegionInfoBuilder.newBuilder(tableName)
455          .setStartKey(start)
456          .setEndKey(end)
457          .setSplit(false)
458          .setRegionId(regionId++)
459          .build();
460      regions.add(hri);
461    }
462    return regions;
463  }

491  protected ServerAndLoad randomServer(final int numRegionsPerServer) {
492    if (!this.serverQueue.isEmpty()) {
493      ServerName sn = this.serverQueue.poll();
494      return new ServerAndLoad(sn, numRegionsPerServer);
495    }
496    Random rand = ThreadLocalRandom.current();
497    String host = "srv" + rand.nextInt(Integer.MAX_VALUE);
498    int port = rand.nextInt(60000);
499    long startCode = rand.nextLong();
500    ServerName sn = ServerName.valueOf(host, port, startCode);
501    return new ServerAndLoad(sn, numRegionsPerServer);
502  }

520  protected Map<ServerName, List<RegionInfo>> createServerMap(int numNodes, int numRegions,
521    int numRegionsPerServer, int replication, int numTables) {
522    // construct a cluster of numNodes, having a total of numRegions. Each RS will hold
523    // numRegionsPerServer many regions except for the last one, which will host all the
524    // remaining regions
525    int[] cluster = new int[numNodes];
526    for (int i = 0; i < numNodes; i++) {
527      cluster[i] = numRegionsPerServer;
528    }
529    cluster[cluster.length - 1] = numRegions - ((cluster.length - 1) * numRegionsPerServer);
530    Map<ServerName, List<RegionInfo>> clusterState = mockClusterServers(cluster, numTables);
531    if (replication > 0) {
532      // replicate the regions to the same servers
533      for (List<RegionInfo> regions : clusterState.values()) {
534        int length = regions.size();
535        for (int i = 0; i < length; i++) {
536          for (int r = 1; r < replication; r++) {
537            regions.add(RegionReplicaUtil.getRegionInfoForReplica(regions.get(i), r));
538          }
539        }
540      }
541    }
542
543    return clusterState;
544  }
