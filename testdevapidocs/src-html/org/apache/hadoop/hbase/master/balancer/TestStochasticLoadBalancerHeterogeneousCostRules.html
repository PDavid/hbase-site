015package org.apache.hadoop.hbase.master.balancer;
016
017import java.io.File;
018import java.io.IOException;
019import java.nio.charset.Charset;
020import java.nio.file.Files;
021import java.nio.file.Path;
022import java.nio.file.Paths;
023import java.util.ArrayList;
024import java.util.Arrays;
025import java.util.Collections;
026import java.util.List;
027
028import org.apache.hadoop.conf.Configuration;
029import org.apache.hadoop.fs.FSDataOutputStream;
030import org.apache.hadoop.hbase.HBaseClassTestRule;
031import org.apache.hadoop.hbase.HBaseTestingUtility;
032import org.apache.hadoop.hbase.testclassification.MasterTests;
033import org.apache.hadoop.hbase.testclassification.MediumTests;
034import org.apache.hadoop.hdfs.DistributedFileSystem;
035import org.apache.hadoop.hdfs.MiniDFSCluster;
036import org.junit.AfterClass;
037import org.junit.Assert;
038import org.junit.BeforeClass;
039import org.junit.ClassRule;
040import org.junit.Test;
041import org.junit.experimental.categories.Category;
042
043@Category({ MasterTests.class, MediumTests.class })
044public class TestStochasticLoadBalancerHeterogeneousCostRules extends BalancerTestBase {
045
046  @ClassRule
047  public static final HBaseClassTestRule CLASS_RULE =
048      HBaseClassTestRule.forClass(TestStochasticLoadBalancerHeterogeneousCostRules.class);
049
050  static final String DEFAULT_RULES_TMP_LOCATION = "/tmp/hbase-balancer.rules";
051  static Configuration conf;
052  private HeterogeneousRegionCountCostFunction costFunction;
053
054  @BeforeClass
055  public static void beforeAllTests() throws Exception {
056    createSimpleRulesFile(new ArrayList<>());
057    conf = new Configuration();
058    conf.set(HeterogeneousRegionCountCostFunction.HBASE_MASTER_BALANCER_HETEROGENEOUS_RULES_FILE,
059      DEFAULT_RULES_TMP_LOCATION);
060  }
061
062  static void createSimpleRulesFile(final List<String> lines) throws IOException {
063    cleanup();
064    final Path file = Paths.get(DEFAULT_RULES_TMP_LOCATION);
065    Files.write(file, lines, Charset.forName("UTF-8"));
066  }
067
068  protected static void cleanup() {
069    final File file = new File(DEFAULT_RULES_TMP_LOCATION);
070    file.delete();
071  }
072
073  @AfterClass
074  public static void afterAllTests() {
075    cleanup();
076  }
077
078  @Test
079  public void testNoRules() {
080    cleanup();
081    this.costFunction = new HeterogeneousRegionCountCostFunction(conf);
082    this.costFunction.loadRules();
083    Assert.assertEquals(0, this.costFunction.getNumberOfRulesLoaded());
084  }
085
086  @Test
087  public void testBadFormatInRules() throws IOException {
088    createSimpleRulesFile(new ArrayList<>());
089    this.costFunction = new HeterogeneousRegionCountCostFunction(conf);
090    this.costFunction.loadRules();
091    Assert.assertEquals(0, this.costFunction.getNumberOfRulesLoaded());
092
093    createSimpleRulesFile(Collections.singletonList("bad rules format"));
094    this.costFunction = new HeterogeneousRegionCountCostFunction(conf);
095    this.costFunction.loadRules();
096    Assert.assertEquals(0, this.costFunction.getNumberOfRulesLoaded());
097
098    createSimpleRulesFile(Arrays.asList("srv[1-2] 10", "bad_rules format", "a"));
099    this.costFunction = new HeterogeneousRegionCountCostFunction(conf);
100    this.costFunction.loadRules();
101    Assert.assertEquals(1, this.costFunction.getNumberOfRulesLoaded());
102  }
103
104  @Test
105  public void testTwoRules() throws IOException {
106    createSimpleRulesFile(Arrays.asList("^server1$ 10", "^server2 21"));
107    this.costFunction = new HeterogeneousRegionCountCostFunction(conf);
108    this.costFunction.loadRules();
109    Assert.assertEquals(2, this.costFunction.getNumberOfRulesLoaded());
110  }
111
112  @Test
113  public void testBadRegexp() throws IOException {
114    createSimpleRulesFile(Collections.singletonList("server[ 1"));
115    this.costFunction = new HeterogeneousRegionCountCostFunction(conf);
116    this.costFunction.loadRules();
117    Assert.assertEquals(0, this.costFunction.getNumberOfRulesLoaded());
118  }
119
120  @Test
121  public void testNoOverride() throws IOException {
122    createSimpleRulesFile(Arrays.asList("^server1$ 10", "^server2 21"));
123    this.costFunction = new HeterogeneousRegionCountCostFunction(conf);
124    this.costFunction.loadRules();
125    Assert.assertEquals(2, this.costFunction.getNumberOfRulesLoaded());
126
127    // loading malformed configuration does not overload current
128    cleanup();
129    this.costFunction.loadRules();
130    Assert.assertEquals(2, this.costFunction.getNumberOfRulesLoaded());
131  }
132
133  @Test
134  public void testLoadingFomHDFS() throws Exception {
135
136    HBaseTestingUtility hBaseTestingUtility = new HBaseTestingUtility();
137    hBaseTestingUtility.startMiniDFSCluster(3);
138
139    MiniDFSCluster cluster = hBaseTestingUtility.getDFSCluster();
140    DistributedFileSystem fs = cluster.getFileSystem();
141
142    String path = cluster.getURI() + DEFAULT_RULES_TMP_LOCATION;
143
144    // writing file
145    FSDataOutputStream stream = fs.create(new org.apache.hadoop.fs.Path(path));
146    stream.write("server1 10".getBytes());
147    stream.flush();
148    stream.close();
149
150    Configuration configuration = hBaseTestingUtility.getConfiguration();
151
152    // start costFunction
153    configuration.set(
154      HeterogeneousRegionCountCostFunction.HBASE_MASTER_BALANCER_HETEROGENEOUS_RULES_FILE, path);
155    this.costFunction = new HeterogeneousRegionCountCostFunction(configuration);
156    this.costFunction.loadRules();
157    Assert.assertEquals(1, this.costFunction.getNumberOfRulesLoaded());
158
159    hBaseTestingUtility.shutdownMiniCluster();
160  }
161}
