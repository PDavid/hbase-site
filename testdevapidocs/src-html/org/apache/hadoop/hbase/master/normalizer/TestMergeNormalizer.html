140  @Test
141  public void testMergeOfSmallRegions() throws HBaseIOException {
142    TableName testTable = TableName.valueOf("testMergeOfSmallRegions");
143    List<RegionInfo> hris = new ArrayList<>();
144    Map<byte[], Integer> regionSizes = new HashMap<>();
145
146    Timestamp currentTime = new Timestamp(System.currentTimeMillis());
147    Timestamp threedaysBefore = new Timestamp(currentTime.getTime() - TimeUnit.DAYS.toMillis(3));
148
149    RegionInfo hri1 = RegionInfoBuilder.newBuilder(testTable).setStartKey(Bytes.toBytes("aaa"))
150        .setEndKey(Bytes.toBytes("bbb")).setRegionId(threedaysBefore.getTime()).build();
151    hris.add(hri1);
152    regionSizes.put(hri1.getRegionName(), 15);
153
154    RegionInfo hri2 = RegionInfoBuilder.newBuilder(testTable).setStartKey(Bytes.toBytes("bbb"))
155        .setEndKey(Bytes.toBytes("ccc")).setRegionId(threedaysBefore.getTime()).build();
156    hris.add(hri2);
157    regionSizes.put(hri2.getRegionName(), 5);
158
159    RegionInfo hri3 = RegionInfoBuilder.newBuilder(testTable).setStartKey(Bytes.toBytes("ccc"))
160        .setEndKey(Bytes.toBytes("ddd")).setRegionId(threedaysBefore.getTime()).build();
161    hris.add(hri3);
162    regionSizes.put(hri3.getRegionName(), 5);
163
164    RegionInfo hri4 = RegionInfoBuilder.newBuilder(testTable).setStartKey(Bytes.toBytes("ddd"))
165        .setEndKey(Bytes.toBytes("eee")).setRegionId(threedaysBefore.getTime()).build();
166    hris.add(hri4);
167    regionSizes.put(hri4.getRegionName(), 15);
168
169    RegionInfo hri5 = RegionInfoBuilder.newBuilder(testTable).setStartKey(Bytes.toBytes("eee"))
170        .setEndKey(Bytes.toBytes("fff")).build();
171    hris.add(hri5);
172    regionSizes.put(hri5.getRegionName(), 16);
173
174    RegionInfo hri6 = RegionInfoBuilder.newBuilder(testTable).setStartKey(Bytes.toBytes("fff"))
175        .setEndKey(Bytes.toBytes("ggg")).setRegionId(threedaysBefore.getTime()).build();
176    hris.add(hri6);
177    regionSizes.put(hri6.getRegionName(), 0);
178
179    RegionInfo hri7 = RegionInfoBuilder.newBuilder(testTable).setStartKey(Bytes.toBytes("ggg"))
180        .setEndKey(Bytes.toBytes("hhh")).build();
181    hris.add(hri7);
182    regionSizes.put(hri7.getRegionName(), 0);
183
184    setupMocksForNormalizer(regionSizes, hris);
185    List<NormalizationPlan> plans = normalizer.computePlanForTable(testTable);
186
187    NormalizationPlan plan = plans.get(0);
188    assertTrue(plan instanceof MergeNormalizationPlan);
189    assertEquals(hri2, ((MergeNormalizationPlan) plan).getFirstRegion());
190    assertEquals(hri3, ((MergeNormalizationPlan) plan).getSecondRegion());
191
192    // to check last 0 sized regions are merged
193    plan = plans.get(1);
194    assertEquals(hri6, ((MergeNormalizationPlan) plan).getFirstRegion());
195    assertEquals(hri7, ((MergeNormalizationPlan) plan).getSecondRegion());
196  }

235  @SuppressWarnings("MockitoCast")
236  protected void setupMocksForNormalizer(Map<byte[], Integer> regionSizes,
237      List<RegionInfo> RegionInfo) {
238    masterServices = Mockito.mock(MasterServices.class, RETURNS_DEEP_STUBS);
239    masterRpcServices = Mockito.mock(MasterRpcServices.class, RETURNS_DEEP_STUBS);
240
241    // for simplicity all regions are assumed to be on one server; doesn't matter to us
242    ServerName sn = ServerName.valueOf("localhost", 0, 1L);
243    when(masterServices.getAssignmentManager().getRegionStates().getRegionsOfTable(any()))
244        .thenReturn(RegionInfo);
245    when(masterServices.getAssignmentManager().getRegionStates().getRegionServerOfRegion(any()))
246        .thenReturn(sn);
247
248    for (Map.Entry<byte[], Integer> region : regionSizes.entrySet()) {
249      RegionMetrics regionLoad = Mockito.mock(RegionMetrics.class);
250      when(regionLoad.getRegionName()).thenReturn(region.getKey());
251      when(regionLoad.getStoreFileSize())
252          .thenReturn(new Size(region.getValue(), Size.Unit.MEGABYTE));
253
254      // this is possibly broken with jdk9, unclear if false positive or not
255      // suppress it for now, fix it when we get to running tests on 9
256      // see: http://errorprone.info/bugpattern/MockitoCast
257      when((Object) masterServices.getServerManager().getLoad(sn).getRegionMetrics()
258          .get(region.getKey())).thenReturn(regionLoad);
259    }
260    try {
261      when(masterRpcServices.isSplitOrMergeEnabled(any(), any())).thenReturn(
262        MasterProtos.IsSplitOrMergeEnabledResponse.newBuilder().setEnabled(true).build());
263    } catch (ServiceException se) {
264      LOG.debug("error setting isSplitOrMergeEnabled switch", se);
265    }
266
267    normalizer.setMasterServices(masterServices);
268    normalizer.setMasterRpcServices(masterRpcServices);
269  }
