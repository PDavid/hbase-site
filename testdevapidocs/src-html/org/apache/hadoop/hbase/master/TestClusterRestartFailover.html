018package org.apache.hadoop.hbase.master;
019
020import java.util.List;
021import java.util.stream.Collectors;
022import org.apache.hadoop.hbase.HBaseClassTestRule;
023import org.apache.hadoop.hbase.ServerName;
024import org.apache.hadoop.hbase.TableName;
025import org.apache.hadoop.hbase.client.Table;
026import org.apache.hadoop.hbase.master.assignment.ServerState;
027import org.apache.hadoop.hbase.master.assignment.ServerStateNode;
028import org.apache.hadoop.hbase.master.procedure.ServerCrashProcedure;
029import org.apache.hadoop.hbase.procedure2.Procedure;
030import org.apache.hadoop.hbase.testclassification.MasterTests;
031import org.apache.hadoop.hbase.testclassification.MediumTests;
032import org.junit.Assert;
033import org.junit.ClassRule;
034import org.junit.Test;
035import org.junit.experimental.categories.Category;
036import org.slf4j.Logger;
037import org.slf4j.LoggerFactory;
038
039@Category({ MasterTests.class, MediumTests.class })
040public class TestClusterRestartFailover extends AbstractTestRestartCluster {
041
042  @ClassRule
043  public static final HBaseClassTestRule CLASS_RULE =
044    HBaseClassTestRule.forClass(TestClusterRestartFailover.class);
045
046  private static final Logger LOG = LoggerFactory.getLogger(TestClusterRestartFailover.class);
047
048  @Override
049  protected boolean splitWALCoordinatedByZk() {
050    return true;
051  }
052
053  private ServerStateNode getServerStateNode(ServerName serverName) {
054    return UTIL.getHBaseCluster().getMaster().getAssignmentManager().getRegionStates()
055      .getServerNode(serverName);
056  }
057
058  @Test
059  public void test() throws Exception {
060    UTIL.startMiniCluster(3);
061    UTIL.waitFor(60000, () -> UTIL.getMiniHBaseCluster().getMaster().isInitialized());
062    // wait for all SCPs finished
063    UTIL.waitFor(60000, () -> UTIL.getHBaseCluster().getMaster().getProcedures().stream()
064      .noneMatch(p -> p instanceof ServerCrashProcedure));
065    TableName tableName = TABLES[0];
066    ServerName testServer = UTIL.getHBaseCluster().getRegionServer(0).getServerName();
067    UTIL.waitFor(10000, () -> getServerStateNode(testServer) != null);
068    ServerStateNode serverNode = getServerStateNode(testServer);
069    Assert.assertNotNull(serverNode);
070    Assert.assertTrue("serverNode should be ONLINE when cluster runs normally",
071      serverNode.isInState(ServerState.ONLINE));
072    UTIL.createMultiRegionTable(tableName, FAMILY);
073    UTIL.waitTableEnabled(tableName);
074    Table table = UTIL.getConnection().getTable(tableName);
075    for (int i = 0; i < 100; i++) {
076      UTIL.loadTable(table, FAMILY);
077    }
078    List<Integer> ports =
079      UTIL.getHBaseCluster().getMaster().getServerManager().getOnlineServersList().stream()
080        .map(serverName -> serverName.getPort()).collect(Collectors.toList());
081    LOG.info("Shutting down cluster");
082    UTIL.getHBaseCluster().killAll();
083    UTIL.getHBaseCluster().waitUntilShutDown();
084    LOG.info("Starting cluster the second time");
085    UTIL.restartHBaseCluster(3, ports);
086    UTIL.waitFor(10000, () -> UTIL.getHBaseCluster().getMaster().isInitialized());
087    serverNode = UTIL.getHBaseCluster().getMaster().getAssignmentManager().getRegionStates()
088      .getServerNode(testServer);
089    Assert.assertNotNull("serverNode should not be null when restart whole cluster", serverNode);
090    Assert.assertFalse(serverNode.isInState(ServerState.ONLINE));
091    LOG.info("start to find the procedure of SCP for the severName we choose");
092    UTIL.waitFor(60000,
093      () -> UTIL.getHBaseCluster().getMaster().getProcedures().stream()
094        .anyMatch(procedure -> (procedure instanceof ServerCrashProcedure) &&
095          ((ServerCrashProcedure) procedure).getServerName().equals(testServer)));
096    Assert.assertFalse("serverNode should not be ONLINE during SCP processing",
097      serverNode.isInState(ServerState.ONLINE));
098    LOG.info("start to submit the SCP for the same serverName {} which should fail", testServer);
099    Assert
100      .assertFalse(UTIL.getHBaseCluster().getMaster().getServerManager().expireServer(testServer));
101    Procedure<?> procedure = UTIL.getHBaseCluster().getMaster().getProcedures().stream()
102      .filter(p -> (p instanceof ServerCrashProcedure) &&
103        ((ServerCrashProcedure) p).getServerName().equals(testServer))
104      .findAny().get();
105    UTIL.waitFor(60000, () -> procedure.isFinished());
106    LOG.info("even when the SCP is finished, the duplicate SCP should not be scheduled for {}",
107      testServer);
108    Assert
109      .assertFalse(UTIL.getHBaseCluster().getMaster().getServerManager().expireServer(testServer));
110    serverNode = UTIL.getHBaseCluster().getMaster().getAssignmentManager().getRegionStates()
111      .getServerNode(testServer);
112    Assert.assertNull("serverNode should be deleted after SCP finished", serverNode);
113  }
114}
