022import org.apache.hadoop.hbase.HBaseClassTestRule;

037
038@Category({ MasterTests.class, LargeTests.class })
039public class TestTransitPeerSyncReplicationStateProcedureBackoff {
040
041  @ClassRule
042  public static final HBaseClassTestRule CLASS_RULE =
043    HBaseClassTestRule.forClass(TestTransitPeerSyncReplicationStateProcedureBackoff.class);
044
045  private static final HBaseTestingUtil UTIL = new HBaseTestingUtil();
046
047  private static boolean FAIL = true;
048
049  public static class TestTransitPeerSyncReplicationStateProcedure
050      extends TransitPeerSyncReplicationStateProcedure {
051
052    public TestTransitPeerSyncReplicationStateProcedure() {
053    }
054
055    public TestTransitPeerSyncReplicationStateProcedure(String peerId, SyncReplicationState state) {
056      super(peerId, state);
057    }
058
059    private void tryFail() throws ReplicationException {
060      synchronized (TestTransitPeerSyncReplicationStateProcedureBackoff.class) {
061        if (FAIL) {
062          throw new ReplicationException("Inject error");
063        }
064        FAIL = true;
065      }
066    }
067
068    @Override
069    protected <T extends Procedure<MasterProcedureEnv>> void addChildProcedure(
070        @SuppressWarnings("unchecked") T... subProcedure) {
071      // Make it a no-op
072    }
073
074    @Override
075    protected void preTransit(MasterProcedureEnv env) throws IOException {
076      fromState = SyncReplicationState.DOWNGRADE_ACTIVE;
077    }
078
079    @Override
080    protected void setPeerNewSyncReplicationState(MasterProcedureEnv env)
081        throws ReplicationException {
082      tryFail();
083    }
084
085    @Override
086    protected void removeAllReplicationQueues(MasterProcedureEnv env) throws ReplicationException {
087      tryFail();
088    }
089
090    @Override
091    protected void reopenRegions(MasterProcedureEnv env) {
092      // do nothing;
093    }
094
095    @Override
096    protected void transitPeerSyncReplicationState(MasterProcedureEnv env)
097        throws ReplicationException {
098      tryFail();
099    }
100
101    @Override
102    protected void createDirForRemoteWAL(MasterProcedureEnv env) throws IOException {
103      try {
104        tryFail();
105      } catch (ReplicationException e) {
106        throw new IOException(e);
107      }
108    }
109  }
110
111  @BeforeClass
112  public static void setUp() throws Exception {
113    UTIL.startMiniCluster(1);
114  }
115
116  @AfterClass
117  public static void tearDown() throws Exception {
118    UTIL.shutdownMiniCluster();
119  }
120
121  private void assertBackoffIncrease() throws IOException, InterruptedException {
122    ProcedureTestUtil.waitUntilProcedureWaitingTimeout(UTIL,
123      TestTransitPeerSyncReplicationStateProcedure.class, 30000);
124    ProcedureTestUtil.waitUntilProcedureTimeoutIncrease(UTIL,
125      TestTransitPeerSyncReplicationStateProcedure.class, 2);
126    synchronized (TestTransitPeerSyncReplicationStateProcedure.class) {
127      FAIL = false;
128    }
129    UTIL.waitFor(30000, () -> FAIL);
130  }
131
132  @Test
133  public void testDowngradeActiveToActive() throws IOException, InterruptedException {
134    ProcedureExecutor<MasterProcedureEnv> procExec =
135        UTIL.getMiniHBaseCluster().getMaster().getMasterProcedureExecutor();
136    // Test procedure: DOWNGRADE_ACTIVE ==> ACTIVE
137    long procId = procExec.submitProcedure(
138      new TestTransitPeerSyncReplicationStateProcedure("1", SyncReplicationState.ACTIVE));
139    // No retry for PRE_PEER_SYNC_REPLICATION_STATE_TRANSITION
140    // SET_PEER_NEW_SYNC_REPLICATION_STATE
141    assertBackoffIncrease();
142    // No retry for REFRESH_PEER_SYNC_REPLICATION_STATE_ON_RS_BEGIN
143    // No retry for REOPEN_ALL_REGIONS_IN_PEER
144    // TRANSIT_PEER_NEW_SYNC_REPLICATION_STATE
145    assertBackoffIncrease();
146    // No retry for REFRESH_PEER_SYNC_REPLICATION_STATE_ON_RS_END
147    // No retry for POST_PEER_SYNC_REPLICATION_STATE_TRANSITION
148    UTIL.waitFor(30000, () -> procExec.isFinished(procId));
149  }
150
151  @Test
152  public void testDowngradeActiveToStandby() throws IOException, InterruptedException {
153    ProcedureExecutor<MasterProcedureEnv> procExec =
154        UTIL.getMiniHBaseCluster().getMaster().getMasterProcedureExecutor();
155    // Test procedure: DOWNGRADE_ACTIVE ==> ACTIVE
156    long procId = procExec.submitProcedure(
157      new TestTransitPeerSyncReplicationStateProcedure("2", SyncReplicationState.STANDBY));
158    // No retry for PRE_PEER_SYNC_REPLICATION_STATE_TRANSITION
159    // SET_PEER_NEW_SYNC_REPLICATION_STATE
160    assertBackoffIncrease();
161    // No retry for REFRESH_PEER_SYNC_REPLICATION_STATE_ON_RS_BEGIN
162    // REMOVE_ALL_REPLICATION_QUEUES_IN_PEER
163    assertBackoffIncrease();
164    // TRANSIT_PEER_NEW_SYNC_REPLICATION_STATE
165    assertBackoffIncrease();
166    // No retry for REFRESH_PEER_SYNC_REPLICATION_STATE_ON_RS_END
167    // CREATE_DIR_FOR_REMOTE_WAL
168    assertBackoffIncrease();
169    // No retry for POST_PEER_SYNC_REPLICATION_STATE_TRANSITION
170    UTIL.waitFor(30000, () -> procExec.isFinished(procId));
171  }
