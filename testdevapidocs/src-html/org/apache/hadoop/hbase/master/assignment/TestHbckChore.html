045@Category({ MasterTests.class, MediumTests.class })
046public class TestHbckChore extends TestAssignmentManagerBase {
047  private static final Logger LOG = LoggerFactory.getLogger(TestHbckChore.class);
048
049  @ClassRule
050  public static final HBaseClassTestRule CLASS_RULE =
051      HBaseClassTestRule.forClass(TestHbckChore.class);
052
053  private HbckChore hbckChore;
054
055  @Before
056  public void setUp() throws Exception {
057    super.setUp();
058    hbckChore = new HbckChore(master);
059  }
060
061  @Test
062  public void testForMeta() {
063    byte[] metaRegionNameAsBytes = RegionInfoBuilder.FIRST_META_REGIONINFO.getRegionName();
064    String metaRegionName = RegionInfoBuilder.FIRST_META_REGIONINFO.getEncodedName();
065    List<ServerName> serverNames = master.getServerManager().getOnlineServersList();
066    assertEquals(NSERVERS, serverNames.size());
067
068    hbckChore.choreForTesting();
069    Map<String, Pair<ServerName, List<ServerName>>> inconsistentRegions =
070        hbckChore.getInconsistentRegions();
071
072    // Test for case1: Master thought this region opened, but no regionserver reported it.
073    assertTrue(inconsistentRegions.containsKey(metaRegionName));
074    Pair<ServerName, List<ServerName>> pair = inconsistentRegions.get(metaRegionName);
075    ServerName locationInMeta = pair.getFirst();
076    List<ServerName> reportedRegionServers = pair.getSecond();
077    assertTrue(serverNames.contains(locationInMeta));
078    assertEquals(0, reportedRegionServers.size());
079
080    // Reported right region location. Then not in problematic regions.
081    am.reportOnlineRegions(locationInMeta, Collections.singleton(metaRegionNameAsBytes));
082    hbckChore.choreForTesting();
083    inconsistentRegions = hbckChore.getInconsistentRegions();
084    assertFalse(inconsistentRegions.containsKey(metaRegionName));
085  }
086
087  @Test
088  public void testForUserTable() throws Exception {
089    TableName tableName = TableName.valueOf("testForUserTable");
090    RegionInfo hri = createRegionInfo(tableName, 1);
091    String regionName = hri.getEncodedName();
092    rsDispatcher.setMockRsExecutor(new GoodRsExecutor());
093    Future<byte[]> future = submitProcedure(createAssignProcedure(hri));
094    waitOnFuture(future);
095
096    List<ServerName> serverNames = master.getServerManager().getOnlineServersList();
097    assertEquals(NSERVERS, serverNames.size());
098
099    // Test for case1: Master thought this region opened, but no regionserver reported it.
100    hbckChore.choreForTesting();
101    Map<String, Pair<ServerName, List<ServerName>>> inconsistentRegions =
102        hbckChore.getInconsistentRegions();
103    assertTrue(inconsistentRegions.containsKey(regionName));
104    Pair<ServerName, List<ServerName>> pair = inconsistentRegions.get(regionName);
105    ServerName locationInMeta = pair.getFirst();
106    List<ServerName> reportedRegionServers = pair.getSecond();
107    assertTrue(serverNames.contains(locationInMeta));
108    assertEquals(0, reportedRegionServers.size());
109
110    // Test for case2: Master thought this region opened on Server1, but regionserver reported
111    // Server2
112    final ServerName tempLocationInMeta = locationInMeta;
113    final ServerName anotherServer =
114        serverNames.stream().filter(s -> !s.equals(tempLocationInMeta)).findFirst().get();
115    am.reportOnlineRegions(anotherServer, Collections.singleton(hri.getRegionName()));
116    hbckChore.choreForTesting();
117    inconsistentRegions = hbckChore.getInconsistentRegions();
118    assertTrue(inconsistentRegions.containsKey(regionName));
119    pair = inconsistentRegions.get(regionName);
120    locationInMeta = pair.getFirst();
121    reportedRegionServers = pair.getSecond();
122    assertEquals(1, reportedRegionServers.size());
123    assertFalse(reportedRegionServers.contains(locationInMeta));
124    assertTrue(reportedRegionServers.contains(anotherServer));
125
126    // Test for case3: More than one regionservers reported opened this region.
127    am.reportOnlineRegions(locationInMeta, Collections.singleton(hri.getRegionName()));
128    hbckChore.choreForTesting();
129    inconsistentRegions = hbckChore.getInconsistentRegions();
130    assertTrue(inconsistentRegions.containsKey(regionName));
131    pair = inconsistentRegions.get(regionName);
132    locationInMeta = pair.getFirst();
133    reportedRegionServers = pair.getSecond();
134    assertEquals(2, reportedRegionServers.size());
135    assertTrue(reportedRegionServers.contains(locationInMeta));
136    assertTrue(reportedRegionServers.contains(anotherServer));
137
138    // Reported right region location. Then not in problematic regions.
139    am.reportOnlineRegions(anotherServer, Collections.EMPTY_SET);
140    hbckChore.choreForTesting();
141    inconsistentRegions = hbckChore.getInconsistentRegions();
142    assertFalse(inconsistentRegions.containsKey(regionName));
143  }
