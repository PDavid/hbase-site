262  @Test
263  public void testMastersSystemTimeIsUsedInMergeRegions() throws IOException {
264    long regionId = EnvironmentEdgeManager.currentTime();
265    TableName tableName = name.getTableName();
266
267    RegionInfo regionInfoA = RegionInfoBuilder.newBuilder(tableName)
268      .setStartKey(HConstants.EMPTY_START_ROW).setEndKey(new byte[] { 'a' }).setSplit(false)
269      .setRegionId(regionId).setReplicaId(0).build();
270
271    RegionInfo regionInfoB = RegionInfoBuilder.newBuilder(tableName).setStartKey(new byte[] { 'a' })
272      .setEndKey(HConstants.EMPTY_END_ROW).setSplit(false).setRegionId(regionId).setReplicaId(0)
273      .build();
274    RegionInfo mergedRegionInfo = RegionInfoBuilder.newBuilder(tableName)
275      .setStartKey(HConstants.EMPTY_START_ROW).setEndKey(HConstants.EMPTY_END_ROW).setSplit(false)
276      .setRegionId(regionId).setReplicaId(0).build();
277
278    ServerName sn = ServerName.valueOf("bar", 0, 0);
279    try (Table meta = MetaTableAccessor.getMetaHTable(UTIL.getConnection())) {
280      List<RegionInfo> regionInfos = Lists.newArrayList(regionInfoA, regionInfoB);
281      MetaTableAccessor.addRegionsToMeta(UTIL.getConnection(), regionInfos, 1);
282
283      // write the serverName column with a big current time, but set the masters time as even
284      // bigger. When region merge deletes the rows for regionA and regionB, the serverName columns
285      // should not be seen by the following get
286      long serverNameTime = EnvironmentEdgeManager.currentTime() + 100000000;
287      long masterSystemTime = EnvironmentEdgeManager.currentTime() + 123456789;
288
289      // write the serverName columns
290      MetaTableAccessor.updateRegionLocation(UTIL.getConnection(), regionInfoA, sn, 1,
291        serverNameTime);
292
293      // assert that we have the serverName column with expected ts
294      Get get = new Get(mergedRegionInfo.getRegionName());
295      Result result = meta.get(get);
296      Cell serverCell = result.getColumnLatestCell(HConstants.CATALOG_FAMILY,
297        CatalogFamilyFormat.getServerColumn(0));
298      assertNotNull(serverCell);
299      assertEquals(serverNameTime, serverCell.getTimestamp());
300
301      final RegionStateStore regionStateStore =
302        UTIL.getHBaseCluster().getMaster().getAssignmentManager().getRegionStateStore();
303
304      ManualEnvironmentEdge edge = new ManualEnvironmentEdge();
305      edge.setValue(masterSystemTime);
306      EnvironmentEdgeManager.injectEdge(edge);
307      try {
308        // now merge the regions, effectively deleting the rows for region a and b.
309        regionStateStore.mergeRegions(mergedRegionInfo,
310          new RegionInfo[] { regionInfoA, regionInfoB }, sn,
311          TableDescriptorBuilder.newBuilder(tableName).build());
312      } finally {
313        EnvironmentEdgeManager.reset();
314      }
315
316      result = meta.get(get);
317      serverCell = result.getColumnLatestCell(HConstants.CATALOG_FAMILY,
318        CatalogFamilyFormat.getServerColumn(0));
319      Cell startCodeCell = result.getColumnLatestCell(HConstants.CATALOG_FAMILY,
320        CatalogFamilyFormat.getStartCodeColumn(0));
321      Cell seqNumCell = result.getColumnLatestCell(HConstants.CATALOG_FAMILY,
322        CatalogFamilyFormat.getSeqNumColumn(0));
323      assertNull(serverCell);
324      assertNull(startCodeCell);
325      assertNull(seqNumCell);
326    }
327  }

332  @Test
333  public void testGetMergeRegions() throws Exception {
334    TableName tn = name.getTableName();
335    UTIL.createMultiRegionTable(tn, Bytes.toBytes("CF"), 4);
336    UTIL.waitTableAvailable(tn);
337    Admin admin = UTIL.getAdmin();
338    List<RegionInfo> regions = admin.getRegions(tn);
339    assertEquals(4, regions.size());
340    admin
341      .mergeRegionsAsync(
342        new byte[][] { regions.get(0).getRegionName(), regions.get(1).getRegionName() }, false)
343      .get(60, TimeUnit.SECONDS);
344    admin
345      .mergeRegionsAsync(
346        new byte[][] { regions.get(2).getRegionName(), regions.get(3).getRegionName() }, false)
347      .get(60, TimeUnit.SECONDS);
348
349    List<RegionInfo> mergedRegions = admin.getRegions(tn);
350    assertEquals(2, mergedRegions.size());
351    RegionInfo mergedRegion0 = mergedRegions.get(0);
352    RegionInfo mergedRegion1 = mergedRegions.get(1);
353
354    final RegionStateStore regionStateStore =
355      UTIL.getHBaseCluster().getMaster().getAssignmentManager().getRegionStateStore();
356
357    List<RegionInfo> mergeParents = regionStateStore.getMergeRegions(mergedRegion0);
358    assertTrue(mergeParents.contains(regions.get(0)));
359    assertTrue(mergeParents.contains(regions.get(1)));
360    mergeParents = regionStateStore.getMergeRegions(mergedRegion1);
361    assertTrue(mergeParents.contains(regions.get(2)));
362    assertTrue(mergeParents.contains(regions.get(3)));
363
364    // Delete merge qualifiers for mergedRegion0, then cannot getMergeRegions again
365    regionStateStore.deleteMergeQualifiers(mergedRegion0);
366    mergeParents = regionStateStore.getMergeRegions(mergedRegion0);
367    assertNull(mergeParents);
368
369    mergeParents = regionStateStore.getMergeRegions(mergedRegion1);
370    assertTrue(mergeParents.contains(regions.get(2)));
371    assertTrue(mergeParents.contains(regions.get(3)));
372  }

374  @Test
375  public void testAddMergeRegions() throws IOException {
376    TableName tn = name.getTableName();
377    Put put = new Put(Bytes.toBytes(name.getTableName().getNameAsString()));
378    List<RegionInfo> ris = new ArrayList<>();
379    int limit = 10;
380    byte[] previous = HConstants.EMPTY_START_ROW;
381    for (int i = 0; i < limit; i++) {
382      RegionInfo ri =
383        RegionInfoBuilder.newBuilder(tn).setStartKey(previous).setEndKey(Bytes.toBytes(i)).build();
384      ris.add(ri);
385    }
386    put = RegionStateStore.addMergeRegions(put, ris);
387    List<Cell> cells = put.getFamilyCellMap().get(HConstants.CATALOG_FAMILY);
388    String previousQualifier = null;
389    assertEquals(limit, cells.size());
390    for (Cell cell : cells) {
391      String qualifier = Bytes.toString(cell.getQualifierArray());
392      assertTrue(qualifier.startsWith(HConstants.MERGE_QUALIFIER_PREFIX_STR));
393      assertNotEquals(qualifier, previousQualifier);
394      previousQualifier = qualifier;
395    }
396  }

398  @Test
399  public void testMetaLocationForRegionReplicasIsRemovedAtTableDeletion() throws IOException {
400    long regionId = EnvironmentEdgeManager.currentTime();
401    TableName tableName = name.getTableName();
402    RegionInfo primary = RegionInfoBuilder.newBuilder(tableName)
403      .setStartKey(HConstants.EMPTY_START_ROW).setEndKey(HConstants.EMPTY_END_ROW).setSplit(false)
404      .setRegionId(regionId).setReplicaId(0).build();
405
406    try (Table meta = MetaTableAccessor.getMetaHTable(UTIL.getConnection())) {
407      List<RegionInfo> regionInfos = Lists.newArrayList(primary);
408      MetaTableAccessor.addRegionsToMeta(UTIL.getConnection(), regionInfos, 3);
409      final RegionStateStore regionStateStore =
410        UTIL.getHBaseCluster().getMaster().getAssignmentManager().getRegionStateStore();
411      regionStateStore.removeRegionReplicas(tableName, 3, 1);
412      Get get = new Get(primary.getRegionName());
413      Result result = meta.get(get);
414      for (int replicaId = 0; replicaId < 3; replicaId++) {
415        Cell serverCell = result.getColumnLatestCell(HConstants.CATALOG_FAMILY,
416          CatalogFamilyFormat.getServerColumn(replicaId));
417        Cell startCodeCell = result.getColumnLatestCell(HConstants.CATALOG_FAMILY,
418          CatalogFamilyFormat.getStartCodeColumn(replicaId));
419        Cell stateCell = result.getColumnLatestCell(HConstants.CATALOG_FAMILY,
420          CatalogFamilyFormat.getRegionStateColumn(replicaId));
421        Cell snCell = result.getColumnLatestCell(HConstants.CATALOG_FAMILY,
422          CatalogFamilyFormat.getServerNameColumn(replicaId));
423        if (replicaId == 0) {
424          assertNotNull(stateCell);
425        } else {
426          assertNull(serverCell);
427          assertNull(startCodeCell);
428          assertNull(stateCell);
429          assertNull(snCell);
430        }
431      }
432    }
