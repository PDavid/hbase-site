071/**
072 * This class tests operations in MasterRpcServices which require ADMIN access.
073 * It doesn't test all operations which require ADMIN access, only those which get vetted within
074 * MasterRpcServices at the point of entry itself (unlike old approach of using
075 * hooks in AccessController).
076 *
077 * Sidenote:
078 * There is one big difference between how security tests for AccessController hooks work, and how
079 * the tests in this class for security in MasterRpcServices work.
080 * The difference arises because of the way AC & MasterRpcServices get the user.
081 *
082 * In AccessController, it first checks if there is an active rpc user in ObserverContext. If not,
083 * it uses UserProvider for current user. This *might* make sense in the context of coprocessors,
084 * because they can be called outside the context of RPCs.
085 * But in the context of MasterRpcServices, only one way makes sense - RPCServer.getRequestUser().
086 *
087 * In AC tests, when we do FooUser.runAs on AccessController instance directly, it bypasses
088 * the rpc framework completely, but works because UserProvider provides the correct user, i.e.
089 * FooUser in this case.
090 *
091 * But this doesn't work for the tests here, so we go around by doing complete RPCs.
092 */

167  interface Action {
168    void run(Admin admin) throws Exception;
169  }
170
171  private void verifyAllowed(User user, Action action) throws Exception {
172    user.runAs((PrivilegedExceptionAction<?>) () -> {
173      try (Connection conn = ConnectionFactory.createConnection(conf);
174          Admin admin = conn.getAdmin()) {
175        action.run(admin);
176      } catch (IOException e) {
177        fail(e.toString());
178      }
179      return null;
180    });
181  }
182
183  private void verifyDenied(User user, Action action) throws Exception {
184    user.runAs((PrivilegedExceptionAction<?>) () -> {
185      boolean accessDenied = false;
186      try (Connection conn = ConnectionFactory.createConnection(conf);
187          Admin admin = conn.getAdmin()) {
188        action.run(admin);
189      } catch (AccessDeniedException e) {
190        accessDenied = true;
191      }
192      assertTrue("Expected access to be denied", accessDenied);
193      return null;
194    });
195  }
196
197  private void verifiedDeniedServiceException(User user, Action action) throws Exception {
198    user.runAs((PrivilegedExceptionAction<?>) () -> {
199      boolean accessDenied = false;
200      try (Connection conn = ConnectionFactory.createConnection(conf);
201          Admin admin = conn.getAdmin()) {
202        action.run(admin);
203      } catch (ServiceException e) {
204        // For MasterRpcServices.execService.
205        if (e.getCause() instanceof AccessDeniedException) {
206          accessDenied = true;
207        }
208      }
209      assertTrue("Expected access to be denied", accessDenied);
210      return null;
211    });
212
213  }

215  private void verifyAdminCheckForAction(Action action) throws Exception {
216    verifyAllowed(USER_ADMIN, action);
217    verifyAllowed(USER_GROUP_ADMIN, action);
218    verifyDenied(USER_NON_ADMIN, action);
219  }

298  @Test
299  public void testTableFlush() throws Exception {
300    TableName tn = TableName.valueOf(TEST_NAME.getMethodName());
301    TableDescriptor desc = TableDescriptorBuilder.newBuilder(tn)
302        .setColumnFamily(ColumnFamilyDescriptorBuilder.of("f1")).build();
303    Action adminAction = (admin) -> {
304      admin.createTable(desc);
305      // Avoid giving a global permission which may screw up other tests
306      SecureTestUtil.grantOnTable(
307          TEST_UTIL, USER_NON_ADMIN.getShortName(), tn, null, null, Permission.Action.READ,
308          Permission.Action.WRITE, Permission.Action.CREATE);
309    };
310    verifyAllowed(USER_ADMIN, adminAction);
311
312    Action userAction = (admin) -> {
313      Connection conn = admin.getConnection();
314      final byte[] rowKey = Bytes.toBytes("row1");
315      final byte[] col = Bytes.toBytes("q1");
316      final byte[] val = Bytes.toBytes("v1");
317      try (Table table = conn.getTable(tn)) {
318        // Write a value
319        Put p = new Put(rowKey);
320        p.addColumn(Bytes.toBytes("f1"), col, val);
321        table.put(p);
322        // Flush should not require ADMIN permission
323        admin.flush(tn);
324        // Nb: ideally, we would verify snapshot permission too (as that was fixed in the
325        //   regression HBASE-20185) but taking a snapshot requires ADMIN permission which
326        //   masks the root issue.
327        // Make sure we read the value
328        Result result = table.get(new Get(rowKey));
329        assertFalse(result.isEmpty());
330        Cell c = result.getColumnLatestCell(Bytes.toBytes("f1"), col);
331        assertArrayEquals(val, CellUtil.cloneValue(c));
332      }
333    };
334    verifyAllowed(USER_NON_ADMIN, userAction);
335  }

337  @Test
338  public void testTableFlushAndSnapshot() throws Exception {
339    TableName tn = TableName.valueOf(TEST_NAME.getMethodName());
340    TableDescriptor desc = TableDescriptorBuilder.newBuilder(tn)
341        .setColumnFamily(ColumnFamilyDescriptorBuilder.of("f1")).build();
342    Action adminAction = (admin) -> {
343      admin.createTable(desc);
344      // Giving ADMIN here, but only on this table, *not* globally
345      SecureTestUtil.grantOnTable(
346          TEST_UTIL, USER_NON_ADMIN.getShortName(), tn, null, null, Permission.Action.READ,
347          Permission.Action.WRITE, Permission.Action.CREATE, Permission.Action.ADMIN);
348    };
349    verifyAllowed(USER_ADMIN, adminAction);
350
351    Action userAction = (admin) -> {
352      Connection conn = admin.getConnection();
353      final byte[] rowKey = Bytes.toBytes("row1");
354      final byte[] col = Bytes.toBytes("q1");
355      final byte[] val = Bytes.toBytes("v1");
356      try (Table table = conn.getTable(tn)) {
357        // Write a value
358        Put p = new Put(rowKey);
359        p.addColumn(Bytes.toBytes("f1"), col, val);
360        table.put(p);
361        // Flush should not require ADMIN permission
362        admin.flush(tn);
363        // Table admin should be sufficient to snapshot this table
364        admin.snapshot(tn.getNameAsString() + "_snapshot1", tn);
365        // Read the value just because
366        Result result = table.get(new Get(rowKey));
367        assertFalse(result.isEmpty());
368        Cell c = result.getColumnLatestCell(Bytes.toBytes("f1"), col);
369        assertArrayEquals(val, CellUtil.cloneValue(c));
370      }
371    };
372    verifyAllowed(USER_NON_ADMIN, userAction);
373  }

375  @Test
376  public void testGrantDeniedOnSuperUsersGroups() {
377    /** User */
378    try {
379      // Global
380      SecureTestUtil.grantGlobal(USER_ADMIN_NOT_SUPER, TEST_UTIL, USER_ADMIN.getShortName(),
381        Permission.Action.ADMIN, Permission.Action.CREATE);
382      fail("Granting superuser's global permissions is not allowed.");
383    } catch (Exception e) {
384    }
385    try {
386      // Namespace
387      SecureTestUtil.grantOnNamespace(USER_ADMIN_NOT_SUPER, TEST_UTIL, USER_ADMIN.getShortName(),
388        TEST_NAME.getMethodName(),
389        Permission.Action.ADMIN, Permission.Action.CREATE);
390      fail("Granting superuser's namespace permissions is not allowed.");
391    } catch (Exception e) {
392    }
393    try {
394      // Table
395      SecureTestUtil.grantOnTable(USER_ADMIN_NOT_SUPER, TEST_UTIL, USER_ADMIN.getName(),
396        TableName.valueOf(TEST_NAME.getMethodName()), null, null,
397        Permission.Action.ADMIN, Permission.Action.CREATE);
398      fail("Granting superuser's table permissions is not allowed.");
399    } catch (Exception e) {
400    }
401
402    /** Group */
403    try {
404      SecureTestUtil.grantGlobal(USER_ADMIN_NOT_SUPER, TEST_UTIL,
405        USER_IN_SUPERGROUPS.getShortName(), Permission.Action.ADMIN, Permission.Action.CREATE);
406      fail("Granting superuser's global permissions is not allowed.");
407    } catch (Exception e) {
408    }
409  }
