019package org.apache.hadoop.hbase.mob;
020import java.io.IOException;
021
022import org.apache.hadoop.hbase.HBaseClassTestRule;
023import org.apache.hadoop.hbase.HColumnDescriptor;
024import org.apache.hadoop.hbase.HTableDescriptor;
025import org.apache.hadoop.hbase.client.Admin;
026import org.apache.hadoop.hbase.testclassification.LargeTests;
027import org.junit.Before;
028import org.junit.ClassRule;
029import org.junit.Test;
030import org.junit.experimental.categories.Category;
031import org.slf4j.Logger;
032import org.slf4j.LoggerFactory;
033
034/**
035  * Mob file compaction chore in a generational batch mode test.
036  * 1. Enables batch mode for regular MOB compaction,
037  *    Sets batch size to 7 regions. Enables generational mode.
038  * 2. Disables periodic MOB compactions, sets minimum age to archive to 10 sec
039  * 3. Creates MOB table with 20 regions
040  * 4. Loads MOB data (randomized keys, 1000 rows), flushes data.
041  * 5. Repeats 4. two more times
042  * 6. Verifies that we have 20 *3 = 60 mob files (equals to number of regions x 3)
043  * 7. Runs major MOB compaction.
044  * 8. Verifies that number of MOB files in a mob directory is 20 x4 = 80
045  * 9. Waits for a period of time larger than minimum age to archive
046  * 10. Runs Mob cleaner chore
047  * 11 Verifies that number of MOB files in a mob directory is 20.
048  * 12 Runs scanner and checks all 3 * 1000 rows.
049 */
050@SuppressWarnings("deprecation")
051@Category(LargeTests.class)
052public class TestMobCompactionOptRegionBatchMode extends TestMobCompactionBase{
053  private static final Logger LOG =
054      LoggerFactory.getLogger(TestMobCompactionOptRegionBatchMode.class);
055  @ClassRule
056  public static final HBaseClassTestRule CLASS_RULE =
057      HBaseClassTestRule.forClass(TestMobCompactionOptRegionBatchMode.class);
058
059  private int batchSize = 7;
060  private MobFileCompactionChore compactionChore;
061
062  public TestMobCompactionOptRegionBatchMode() {
063  }
064
065  @Before
066  public void setUp() throws Exception {
067    super.setUp();
068    compactionChore = new MobFileCompactionChore(conf, batchSize);
069  }
070
071  protected void initConf() {
072    super.initConf();
073    conf.setInt(MobConstants.MOB_MAJOR_COMPACTION_REGION_BATCH_SIZE, batchSize);
074    conf.set(MobConstants.MOB_COMPACTION_TYPE_KEY,
075      MobConstants.OPTIMIZED_MOB_COMPACTION_TYPE);
076    conf.setLong(MobConstants.MOB_COMPACTION_MAX_FILE_SIZE_KEY, 1000000);
077  }
078
079  @Test
080  public void testMobFileCompactionBatchMode() throws InterruptedException, IOException {
081    LOG.info("MOB compaction chore generational batch mode started");
082    baseTestMobFileCompaction();
083    LOG.info("MOB compaction chore generational batch mode finished OK");
084
085  }
086
087  @Override
088  protected void mobCompact(Admin admin, HTableDescriptor hdt, HColumnDescriptor hcd)
089      throws IOException, InterruptedException {
090    // Major compact with batch mode enabled
091    compactionChore.performMajorCompactionInBatches(admin, hdt, hcd);
092  }
093
094}
