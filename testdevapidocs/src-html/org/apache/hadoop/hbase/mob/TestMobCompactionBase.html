049/**
050  * Mob file compaction base test.
051  * 1. Enables batch mode for regular MOB compaction,
052  *    Sets batch size to 7 regions. (Optional)
053  * 2. Disables periodic MOB compactions, sets minimum age to archive to 10 sec
054  * 3. Creates MOB table with 20 regions
055  * 4. Loads MOB data (randomized keys, 1000 rows), flushes data.
056  * 5. Repeats 4. two more times
057  * 6. Verifies that we have 20 *3 = 60 mob files (equals to number of regions x 3)
058  * 7. Runs major MOB compaction.
059  * 8. Verifies that number of MOB files in a mob directory is 20 x4 = 80
060  * 9. Waits for a period of time larger than minimum age to archive
061  * 10. Runs Mob cleaner chore
062  * 11 Verifies that number of MOB files in a mob directory is 20.
063  * 12 Runs scanner and checks all 3 * 1000 rows.
064 */
065@SuppressWarnings("deprecation")
066public abstract class TestMobCompactionBase {
067  private static final Logger LOG =
068      LoggerFactory.getLogger(TestMobCompactionBase.class);
069
070  protected HBaseTestingUtility HTU;
071
072  protected final static String famStr = "f1";
073  protected final static byte[] fam = Bytes.toBytes(famStr);
074  protected final static byte[] qualifier = Bytes.toBytes("q1");
075  protected final static long mobLen = 10;
076  protected final static byte[] mobVal = Bytes
077      .toBytes("01234567890123456789012345678901234567890123456789012345678901234567890123456789");
078
079  protected Configuration conf;
080  protected HTableDescriptor hdt;
081  private HColumnDescriptor hcd;
082  protected Admin admin;
083  protected Table table = null;
084  protected long minAgeToArchive = 10000;
085  protected int numRegions = 20;
086  protected int rows = 1000;
087
088  protected MobFileCleanerChore cleanerChore;
089
090  public TestMobCompactionBase() {
091  }
092
093
094  @Before
095  public void setUp() throws Exception {
096    HTU = new HBaseTestingUtility();
097    hdt = HTU.createTableDescriptor(TableName.valueOf(getClass().getName()));
098    conf = HTU.getConfiguration();
099
100    initConf();
101
102    HTU.startMiniCluster();
103    admin = HTU.getAdmin();
104    cleanerChore = new MobFileCleanerChore();
105    hcd = new HColumnDescriptor(fam);
106    hcd.setMobEnabled(true);
107    hcd.setMobThreshold(mobLen);
108    hcd.setMaxVersions(1);
109    hdt.addFamily(hcd);
110    RegionSplitter.UniformSplit splitAlgo = new RegionSplitter.UniformSplit();
111    byte[][] splitKeys = splitAlgo.split(numRegions);
112    table = HTU.createTable(hdt, splitKeys);
113
114  }
115
116  protected void initConf() {
117
118    conf.setInt("hfile.format.version", 3);
119    // Disable automatic MOB compaction
120    conf.setLong(MobConstants.MOB_COMPACTION_CHORE_PERIOD, 0);
121    // Disable automatic MOB file cleaner chore
122    conf.setLong(MobConstants.MOB_CLEANER_PERIOD, 0);
123    // Set minimum age to archive to 10 sec
124    conf.setLong(MobConstants.MIN_AGE_TO_ARCHIVE_KEY, minAgeToArchive);
125    // Set compacted file discharger interval to a half minAgeToArchive
126    conf.setLong("hbase.hfile.compaction.discharger.interval", minAgeToArchive/2);
127  }
128
129  private void loadData(int num) {
130
131    Random r = new Random();
132    try {
133      LOG.info("Started loading {} rows", num);
134      for (int i = 0; i < num; i++) {
135        byte[] key = new byte[32];
136        r.nextBytes(key);
137        Put p = new Put(key);
138        p.addColumn(fam, qualifier, mobVal);
139        table.put(p);
140      }
141      admin.flush(table.getName());
142      LOG.info("Finished loading {} rows", num);
143    } catch (Exception e) {
144      LOG.error("MOB file compaction chore test FAILED", e);
145      fail("MOB file compaction chore test FAILED");
146    }
147  }
148
149  @After
150  public void tearDown() throws Exception {
151    admin.disableTable(hdt.getTableName());
152    admin.deleteTable(hdt.getTableName());
153    HTU.shutdownMiniCluster();
154  }
155
156
157  public void baseTestMobFileCompaction() throws InterruptedException, IOException {
158
159    // Load and flush data 3 times
160    loadData(rows);
161    loadData(rows);
162    loadData(rows);
163    long num = getNumberOfMobFiles(conf, table.getName(), new String(fam));
164    assertEquals(numRegions * 3, num);
165    // Major MOB compact
166    mobCompact(admin, hdt, hcd);
167    // wait until compaction is complete
168    while (admin.getCompactionState(hdt.getTableName()) != CompactionState.NONE) {
169      Thread.sleep(100);
170    }
171
172    num = getNumberOfMobFiles(conf, table.getName(), new String(fam));
173    assertEquals(numRegions * 4, num);
174    // We have guarantee, that compacted file discharger will run during this pause
175    // because it has interval less than this wait time
176    LOG.info("Waiting for {}ms", minAgeToArchive + 1000);
177
178    Thread.sleep(minAgeToArchive + 1000);
179    LOG.info("Cleaning up MOB files");
180    // Cleanup again
181    cleanerChore.cleanupObsoleteMobFiles(conf, table.getName());
182
183    num = getNumberOfMobFiles(conf, table.getName(), new String(fam));
184    assertEquals(numRegions, num);
185
186    long scanned = scanTable();
187    assertEquals(3 * rows, scanned);
188
189  }
190
191  protected abstract void mobCompact(Admin admin2, HTableDescriptor hdt2, HColumnDescriptor hcd2)
192      throws IOException, InterruptedException;
193
194
195  protected  long getNumberOfMobFiles(Configuration conf, TableName tableName, String family)
196      throws IOException {
197    FileSystem fs = FileSystem.get(conf);
198    Path dir = MobUtils.getMobFamilyPath(conf, tableName, family);
199    FileStatus[] stat = fs.listStatus(dir);
200    for (FileStatus st : stat) {
201      LOG.debug("MOB Directory content: {}", st.getPath());
202    }
203    LOG.debug("MOB Directory content total files: {}", stat.length);
204
205    return stat.length;
206  }
207
208
209  protected long scanTable() {
210    try {
211
212      Result result;
213      ResultScanner scanner = table.getScanner(fam);
214      long counter = 0;
215      while ((result = scanner.next()) != null) {
216        assertTrue(Arrays.equals(result.getValue(fam, qualifier), mobVal));
217        counter++;
218      }
219      return counter;
220    } catch (Exception e) {
221      LOG.error("MOB file compaction test FAILED", e);
222      if (HTU != null) {
223        fail(e.getMessage());
224      } else {
225        System.exit(-1);
226      }
227    }
228    return 0;
229  }
