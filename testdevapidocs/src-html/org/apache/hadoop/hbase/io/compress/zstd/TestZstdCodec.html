017package org.apache.hadoop.hbase.io.compress.zstd;
018
019import static org.junit.Assert.assertNotNull;
020import static org.junit.Assert.assertTrue;
021import java.io.File;
022import java.io.FileOutputStream;
023import java.io.IOException;
024import java.util.Random;
025import org.apache.hadoop.conf.Configuration;
026import org.apache.hadoop.fs.CommonConfigurationKeys;
027import org.apache.hadoop.hbase.HBaseClassTestRule;
028import org.apache.hadoop.hbase.HBaseConfiguration;
029import org.apache.hadoop.hbase.io.compress.CompressionTestBase;
030import org.apache.hadoop.hbase.io.compress.DictionaryCache;
031import org.apache.hadoop.hbase.testclassification.SmallTests;
032import org.apache.hadoop.hbase.util.RandomDistribution;
033import org.junit.ClassRule;
034import org.junit.Test;
035import org.junit.experimental.categories.Category;
036
037@Category(SmallTests.class)
038public class TestZstdCodec extends CompressionTestBase {
039
040  @ClassRule
041  public static final HBaseClassTestRule CLASS_RULE =
042      HBaseClassTestRule.forClass(TestZstdCodec.class);
043
044  @Test
045  public void testZstdCodecSmall() throws Exception {
046    codecSmallTest(new ZstdCodec());
047  }
048
049  @Test
050  public void testZstdCodecLarge() throws Exception {
051    codecLargeTest(new ZstdCodec(), 1.1); // poor compressability
052    codecLargeTest(new ZstdCodec(),   2);
053    codecLargeTest(new ZstdCodec(),  10); // very high compressability
054  }
055
056  @Test
057  public void testZstdCodecVeryLarge() throws Exception {
058    Configuration conf = HBaseConfiguration.create();
059    // ZStandard levels range from 1 to 22.
060    // Level 22 might take up to a minute to complete. 3 is the Hadoop default, and will be fast.
061    conf.setInt(CommonConfigurationKeys.IO_COMPRESSION_CODEC_ZSTD_LEVEL_KEY, 3);
062    ZstdCodec codec = new ZstdCodec();
063    codec.setConf(conf);
064    codecVeryLargeTest(codec, 3); // like text
065  }
066
067  @Test
068  public void testZstdCodecWithDictionary() throws Exception {
069    // zstd.test.data compressed with zstd.test.dict at level 3 will produce a result of
070    // 365533 bytes
071    final int expectedCompressedSize = 365533;
072    Configuration conf = HBaseConfiguration.create();
073    conf.setInt(CommonConfigurationKeys.IO_COMPRESSION_CODEC_ZSTD_LEVEL_KEY, 3);
074    // Configure for dictionary available in test resources
075    final String dictionaryPath = DictionaryCache.RESOURCE_SCHEME + "zstd.test.dict";
076    conf.set(ZstdCodec.ZSTD_DICTIONARY_KEY, dictionaryPath);
077    // Load test data from test resources
078    // This will throw an IOException if the test data cannot be loaded
079    final byte[] testData = DictionaryCache.loadFromResource(conf,
080      DictionaryCache.RESOURCE_SCHEME + "zstd.test.data", /* maxSize */ 1024*1024);
081    assertNotNull("Failed to load test data", testData);
082    // Run the test
083    // This will throw an IOException of some kind if there is a problem loading or using the
084    // dictionary.
085    ZstdCodec codec = new ZstdCodec();
086    codec.setConf(conf);
087    codecTest(codec, new byte[][] { testData }, expectedCompressedSize);
088    // Assert that the dictionary was actually loaded
089    assertTrue("Dictionary was not loaded by codec", DictionaryCache.contains(dictionaryPath));
090  }
091
092  //
093  // For generating the test data in src/test/resources/
094  //
095
096  public static void main(String[] args) throws IOException {
097    // Write 1000 1k blocks for training to the specified file
098    // Train with:
099    //   zstd --train-fastcover=k=32,b=8 -B1024 -o <dictionary_file> <input_file>
100    if (args.length < 1) {
101      System.err.println("Usage: TestZstdCodec <outFile>");
102      System.exit(-1);
103    }
104    final RandomDistribution.DiscreteRNG rng =
105      new RandomDistribution.Zipf(new Random(), 0, Byte.MAX_VALUE, 2);
106    final File outFile = new File(args[0]);
107    final byte[] buffer = new byte[1024];
108    System.out.println("Generating " + outFile);
109    try (FileOutputStream os = new FileOutputStream(outFile)) {
110      for (int i = 0; i < 1000; i++) {
111        fill(rng, buffer);
112        os.write(buffer);
113      }
114    }
115    System.out.println("Done");
116  }
