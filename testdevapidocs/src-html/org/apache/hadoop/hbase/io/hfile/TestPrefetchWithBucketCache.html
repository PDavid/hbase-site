059@Category({ IOTests.class, MediumTests.class })
060public class TestPrefetchWithBucketCache {
061
062  private static final Logger LOG = LoggerFactory.getLogger(TestPrefetchWithBucketCache.class);
063
064  @ClassRule
065  public static final HBaseClassTestRule CLASS_RULE =
066    HBaseClassTestRule.forClass(TestPrefetchWithBucketCache.class);

071  private static final HBaseTestingUtil TEST_UTIL = new HBaseTestingUtil();

073  private static final int NUM_VALID_KEY_TYPES = KeyValue.Type.values().length - 2;
074  private static final int DATA_BLOCK_SIZE = 2048;
075  private static final int NUM_KV = 100;

077  private Configuration conf;
078  private CacheConfig cacheConf;
079  private FileSystem fs;
080  private BlockCache blockCache;

082  @Before
083  public void setUp() throws IOException {
084    conf = TEST_UTIL.getConfiguration();
085    conf.setBoolean(CacheConfig.PREFETCH_BLOCKS_ON_OPEN_KEY, true);
086    fs = HFileSystem.get(conf);
087    File testDir = new File(name.getMethodName());
088    testDir.mkdir();
089    conf.set(BUCKET_CACHE_IOENGINE_KEY, "file:/" + testDir.getAbsolutePath() + "/bucket.cache");
090    conf.setLong(BUCKET_CACHE_SIZE_KEY, 200);
091    blockCache = BlockCacheFactory.createBlockCache(conf);
092    cacheConf = new CacheConfig(conf, blockCache);
093  }
094
095  @After
096  public void tearDown() {
097    File cacheFile = new File(name.getMethodName() + "/bucket.cache");
098    File dir = new File(name.getMethodName());
099    cacheFile.delete();
100    dir.delete();
101  }

103  @Test
104  public void testPrefetchDoesntOverwork() throws Exception {
105    Path storeFile = writeStoreFile("TestPrefetchDoesntOverwork");
106    // Prefetches the file blocks
107    LOG.debug("First read should prefetch the blocks.");
108    readStoreFile(storeFile);
109    BucketCache bc = BucketCache.getBuckedCacheFromCacheConfig(cacheConf).get();
110    // Our file should have 6 DATA blocks. We should wait for all of them to be cached
111    Waiter.waitFor(conf, 300, () -> bc.getBackingMap().size() == 6);
112    Map<BlockCacheKey, BucketEntry> snapshot = ImmutableMap.copyOf(bc.getBackingMap());
113    // Reads file again and check we are not prefetching it again
114    LOG.debug("Second read, no prefetch should happen here.");
115    readStoreFile(storeFile);
116    // Makes sure the cache hasn't changed
117    snapshot.entrySet().forEach(e -> {
118      BucketEntry entry = bc.getBackingMap().get(e.getKey());
119      assertNotNull(entry);
120      assertEquals(e.getValue().getCachedTime(), entry.getCachedTime());
121    });
122    // forcibly removes first block from the bc backing map, in order to cause it to be cached again
123    BlockCacheKey key = snapshot.keySet().stream().findFirst().get();
124    LOG.debug("removing block {}", key);
125    bc.getBackingMap().remove(key);
126    bc.getFullyCachedFiles().remove(storeFile.getName());
127    assertTrue(snapshot.size() > bc.getBackingMap().size());
128    LOG.debug("Third read should prefetch again, as we removed one block for the file.");
129    readStoreFile(storeFile);
130    Waiter.waitFor(conf, 300, () -> snapshot.size() == bc.getBackingMap().size());
131    assertTrue(snapshot.get(key).getCachedTime() < bc.getBackingMap().get(key).getCachedTime());
132  }
133
134  private void readStoreFile(Path storeFilePath) throws Exception {
135    readStoreFile(storeFilePath, (r, o) -> {
136      HFileBlock block = null;
137      try {
138        block = r.readBlock(o, -1, false, true, false, true, null, null);
139      } catch (IOException e) {
140        fail(e.getMessage());
141      }
142      return block;
143    }, (key, block) -> {
144      boolean isCached = blockCache.getBlock(key, true, false, true) != null;
145      if (
146        block.getBlockType() == BlockType.DATA || block.getBlockType() == BlockType.ROOT_INDEX
147          || block.getBlockType() == BlockType.INTERMEDIATE_INDEX
148      ) {
149        assertTrue(isCached);
150      }
151    });
152  }
153
154  private void readStoreFile(Path storeFilePath,
155    BiFunction<HFile.Reader, Long, HFileBlock> readFunction,
156    BiConsumer<BlockCacheKey, HFileBlock> validationFunction) throws Exception {
157    // Open the file
158    HFile.Reader reader = HFile.createReader(fs, storeFilePath, cacheConf, true, conf);
159
160    while (!reader.prefetchComplete()) {
161      // Sleep for a bit
162      Thread.sleep(1000);
163    }
164    long offset = 0;
165    while (offset < reader.getTrailer().getLoadOnOpenDataOffset()) {
166      HFileBlock block = readFunction.apply(reader, offset);
167      BlockCacheKey blockCacheKey = new BlockCacheKey(reader.getName(), offset);
168      validationFunction.accept(blockCacheKey, block);
169      offset += block.getOnDiskSizeWithHeader();
170    }
171  }
172
173  private Path writeStoreFile(String fname) throws IOException {
174    HFileContext meta = new HFileContextBuilder().withBlockSize(DATA_BLOCK_SIZE).build();
175    return writeStoreFile(fname, meta);
176  }
177
178  private Path writeStoreFile(String fname, HFileContext context) throws IOException {
179    Path storeFileParentDir = new Path(TEST_UTIL.getDataTestDir(), fname);
180    StoreFileWriter sfw = new StoreFileWriter.Builder(conf, cacheConf, fs)
181      .withOutputDir(storeFileParentDir).withFileContext(context).build();
182    Random rand = ThreadLocalRandom.current();
183    final int rowLen = 32;
184    for (int i = 0; i < NUM_KV; ++i) {
185      byte[] k = RandomKeyValueUtil.randomOrderedKey(rand, i);
186      byte[] v = RandomKeyValueUtil.randomValue(rand);
187      int cfLen = rand.nextInt(k.length - rowLen + 1);
188      KeyValue kv = new KeyValue(k, 0, rowLen, k, rowLen, cfLen, k, rowLen + cfLen,
189        k.length - rowLen - cfLen, rand.nextLong(), generateKeyType(rand), v, 0, v.length);
190      sfw.append(kv);
191    }
192
193    sfw.close();
194    return sfw.getPath();
195  }
196
197  public static KeyValue.Type generateKeyType(Random rand) {
198    if (rand.nextBoolean()) {
199      // Let's make half of KVs puts.
200      return KeyValue.Type.Put;
201    } else {
202      KeyValue.Type keyType = KeyValue.Type.values()[1 + rand.nextInt(NUM_VALID_KEY_TYPES)];
203      if (keyType == KeyValue.Type.Minimum || keyType == KeyValue.Type.Maximum) {
204        throw new RuntimeException("Generated an invalid key type: " + keyType + ". "
205          + "Probably the layout of KeyValue.Type has changed.");
206      }
207      return keyType;
208    }
209  }
