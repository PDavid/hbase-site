112  @Test
113  public void testPrefetchDoesntOverwork() throws Exception {
114    conf.setLong(BUCKET_CACHE_SIZE_KEY, 200);
115    blockCache = BlockCacheFactory.createBlockCache(conf);
116    cacheConf = new CacheConfig(conf, blockCache);
117    Path storeFile = writeStoreFile("TestPrefetchDoesntOverwork", 100);
118    // Prefetches the file blocks
119    LOG.debug("First read should prefetch the blocks.");
120    readStoreFile(storeFile);
121    BucketCache bc = BucketCache.getBucketCacheFromCacheConfig(cacheConf).get();
122    // Our file should have 6 DATA blocks. We should wait for all of them to be cached
123    Waiter.waitFor(conf, 300, () -> bc.getBackingMap().size() == 6);
124    Map<BlockCacheKey, BucketEntry> snapshot = ImmutableMap.copyOf(bc.getBackingMap());
125    // Reads file again and check we are not prefetching it again
126    LOG.debug("Second read, no prefetch should happen here.");
127    readStoreFile(storeFile);
128    // Makes sure the cache hasn't changed
129    snapshot.entrySet().forEach(e -> {
130      BucketEntry entry = bc.getBackingMap().get(e.getKey());
131      assertNotNull(entry);
132      assertEquals(e.getValue().getCachedTime(), entry.getCachedTime());
133    });
134    // forcibly removes first block from the bc backing map, in order to cause it to be cached again
135    BlockCacheKey key = snapshot.keySet().stream().findFirst().get();
136    LOG.debug("removing block {}", key);
137    bc.getBackingMap().remove(key);
138    bc.getFullyCachedFiles().get().remove(storeFile.getName());
139    assertTrue(snapshot.size() > bc.getBackingMap().size());
140    LOG.debug("Third read should prefetch again, as we removed one block for the file.");
141    readStoreFile(storeFile);
142    Waiter.waitFor(conf, 300, () -> snapshot.size() == bc.getBackingMap().size());
143    assertTrue(snapshot.get(key).getCachedTime() < bc.getBackingMap().get(key).getCachedTime());
144  }

146  @Test
147  public void testPrefetchRefsAfterSplit() throws Exception {
148    conf.setLong(BUCKET_CACHE_SIZE_KEY, 200);
149    blockCache = BlockCacheFactory.createBlockCache(conf);
150    cacheConf = new CacheConfig(conf, blockCache);
151
152    Path tableDir = new Path(TEST_UTIL.getDataTestDir(), "testPrefetchRefsAfterSplit");
153    RegionInfo region = RegionInfoBuilder.newBuilder(TableName.valueOf(tableDir.getName())).build();
154    Path regionDir = new Path(tableDir, region.getEncodedName());
155    Path cfDir = new Path(regionDir, "cf");
156    HRegionFileSystem regionFS =
157      HRegionFileSystem.createRegionOnFileSystem(conf, fs, tableDir, region);
158    Path storeFile = writeStoreFile(100, cfDir);
159
160    // Prefetches the file blocks
161    LOG.debug("First read should prefetch the blocks.");
162    readStoreFile(storeFile);
163    BucketCache bc = BucketCache.getBucketCacheFromCacheConfig(cacheConf).get();
164    // Our file should have 6 DATA blocks. We should wait for all of them to be cached
165    Waiter.waitFor(conf, 300, () -> bc.getBackingMap().size() == 6);
166
167    // split the file and return references to the original file
168    Random rand = ThreadLocalRandom.current();
169    byte[] splitPoint = RandomKeyValueUtil.randomOrderedKey(rand, 50);
170    HStoreFile file = new HStoreFile(fs, storeFile, conf, cacheConf, BloomType.NONE, true);
171    Path ref = regionFS.splitStoreFile(region, "cf", file, splitPoint, false,
172      new ConstantSizeRegionSplitPolicy());
173    HStoreFile refHsf = new HStoreFile(this.fs, ref, conf, cacheConf, BloomType.NONE, true);
174    // starts reader for the ref. The ref should resolve to the original file blocks
175    // and not duplicate blocks in the cache.
176    refHsf.initReader();
177    HFile.Reader reader = refHsf.getReader().getHFileReader();
178    while (!reader.prefetchComplete()) {
179      // Sleep for a bit
180      Thread.sleep(1000);
181    }
182    // the ref file blocks keys should actually resolve to the referred file blocks,
183    // so we should not see additional blocks in the cache.
184    Waiter.waitFor(conf, 300, () -> bc.getBackingMap().size() == 6);
185
186    BlockCacheKey refCacheKey = new BlockCacheKey(ref.getName(), 0);
187    Cacheable result = bc.getBlock(refCacheKey, true, false, true);
188    assertNotNull(result);
189    BlockCacheKey fileCacheKey = new BlockCacheKey(file.getPath().getName(), 0);
190    assertEquals(result, bc.getBlock(fileCacheKey, true, false, true));
191    assertNull(bc.getBackingMap().get(refCacheKey));
192    assertNotNull(bc.getBlockForReference(refCacheKey));
193  }

195  @Test
196  public void testPrefetchInterruptOnCapacity() throws Exception {
197    conf.setLong(BUCKET_CACHE_SIZE_KEY, 1);
198    conf.set(BUCKET_CACHE_BUCKETS_KEY, "3072");
199    conf.setDouble("hbase.bucketcache.acceptfactor", 0.98);
200    conf.setDouble("hbase.bucketcache.minfactor", 0.95);
201    conf.setDouble("hbase.bucketcache.extrafreefactor", 0.01);
202    blockCache = BlockCacheFactory.createBlockCache(conf);
203    cacheConf = new CacheConfig(conf, blockCache);
204    Path storeFile = writeStoreFile("testPrefetchInterruptOnCapacity", 10000);
205    // Prefetches the file blocks
206    LOG.debug("First read should prefetch the blocks.");
207    createReaderAndWaitForPrefetchInterruption(storeFile);
208    BucketCache bc = BucketCache.getBucketCacheFromCacheConfig(cacheConf).get();
209    long evictionsFirstPrefetch = bc.getStats().getEvictionCount();
210    LOG.debug("evictions after first prefetch: {}", bc.getStats().getEvictionCount());
211    HFile.Reader reader = createReaderAndWaitForPrefetchInterruption(storeFile);
212    LOG.debug("evictions after second prefetch: {}", bc.getStats().getEvictionCount());
213    assertTrue((bc.getStats().getEvictionCount() - evictionsFirstPrefetch) < 10);
214    HFileScanner scanner = reader.getScanner(conf, true, true);
215    scanner.seekTo();
216    while (scanner.next()) {
217      // do a full scan to force some evictions
218      LOG.trace("Iterating the full scan to evict some blocks");
219    }
220    scanner.close();
221    LOG.debug("evictions after scanner: {}", bc.getStats().getEvictionCount());
222    // The scanner should had triggered at least 3x evictions from the prefetch,
223    // as we try cache each block without interruption.
224    assertTrue(bc.getStats().getEvictionCount() > evictionsFirstPrefetch);
225  }

246  @Test
247  public void testPrefetchMetricProgress() throws Exception {
248    conf.setLong(BUCKET_CACHE_SIZE_KEY, 200);
249    blockCache = BlockCacheFactory.createBlockCache(conf);
250    cacheConf = new CacheConfig(conf, blockCache);
251    Path storeFile = writeStoreFile("testPrefetchMetricsProgress", 100);
252    // Prefetches the file blocks
253    LOG.debug("First read should prefetch the blocks.");
254    readStoreFile(storeFile);
255    String regionName = storeFile.getParent().getParent().getName();
256    BucketCache bc = BucketCache.getBucketCacheFromCacheConfig(cacheConf).get();
257    MutableLong regionCachedSize = new MutableLong(0);
258    // Our file should have 6 DATA blocks. We should wait for all of them to be cached
259    long waitedTime = Waiter.waitFor(conf, 300, () -> {
260      if (bc.getBackingMap().size() > 0) {
261        long currentSize = bc.getRegionCachedInfo().get().get(regionName);
262        assertTrue(regionCachedSize.getValue() <= currentSize);
263        LOG.debug("Logging progress of region caching: {}", currentSize);
264        regionCachedSize.setValue(currentSize);
265      }
266      return bc.getBackingMap().size() == 6;
267    });
268  }

270  private void readStoreFile(Path storeFilePath) throws Exception {
271    readStoreFile(storeFilePath, (r, o) -> {
272      HFileBlock block = null;
273      try {
274        block = r.readBlock(o, -1, false, true, false, true, null, null);
275      } catch (IOException e) {
276        fail(e.getMessage());
277      }
278      return block;
279    }, (key, block) -> {
280      boolean isCached = blockCache.getBlock(key, true, false, true) != null;
281      if (
282        block.getBlockType() == BlockType.DATA || block.getBlockType() == BlockType.ROOT_INDEX
283          || block.getBlockType() == BlockType.INTERMEDIATE_INDEX
284      ) {
285        assertTrue(isCached);
286      }
287    });
288  }

290  private void readStoreFile(Path storeFilePath,
291    BiFunction<HFile.Reader, Long, HFileBlock> readFunction,
292    BiConsumer<BlockCacheKey, HFileBlock> validationFunction) throws Exception {
293    // Open the file
294    HFile.Reader reader = HFile.createReader(fs, storeFilePath, cacheConf, true, conf);
295
296    while (!reader.prefetchComplete()) {
297      // Sleep for a bit
298      Thread.sleep(1000);
299    }
300    long offset = 0;
301    long sizeForDataBlocks = 0;
302    while (offset < reader.getTrailer().getLoadOnOpenDataOffset()) {
303      HFileBlock block = readFunction.apply(reader, offset);
304      BlockCacheKey blockCacheKey = new BlockCacheKey(reader.getName(), offset);
305      validationFunction.accept(blockCacheKey, block);
306      offset += block.getOnDiskSizeWithHeader();
307    }
308  }

310  private HFile.Reader createReaderAndWaitForPrefetchInterruption(Path storeFilePath)
311    throws Exception {
312    // Open the file
313    HFile.Reader reader = HFile.createReader(fs, storeFilePath, cacheConf, true, conf);
314
315    while (!reader.prefetchComplete()) {
316      // Sleep for a bit
317      Thread.sleep(1000);
318    }
319    assertEquals(0, BucketCache.getBucketCacheFromCacheConfig(cacheConf).get().getFullyCachedFiles()
320      .get().size());
321
322    return reader;
323  }

338
339  private Path writeStoreFile(HFileContext context, int numKVs, Path regionCFDir)
340    throws IOException {
341    StoreFileWriter sfw = new StoreFileWriter.Builder(conf, cacheConf, fs)
342      .withOutputDir(regionCFDir).withFileContext(context).build();
343    Random rand = ThreadLocalRandom.current();
344    final int rowLen = 32;
345    for (int i = 0; i < numKVs; ++i) {
346      byte[] k = RandomKeyValueUtil.randomOrderedKey(rand, i);
347      byte[] v = RandomKeyValueUtil.randomValue(rand);
348      int cfLen = rand.nextInt(k.length - rowLen + 1);
349      KeyValue kv = new KeyValue(k, 0, rowLen, k, rowLen, cfLen, k, rowLen + cfLen,
350        k.length - rowLen - cfLen, rand.nextLong(), generateKeyType(rand), v, 0, v.length);
351      sfw.append(kv);
352    }
353
354    sfw.close();
355    return sfw.getPath();
356  }
357
358  public static KeyValue.Type generateKeyType(Random rand) {
359    if (rand.nextBoolean()) {
360      // Let's make half of KVs puts.
361      return KeyValue.Type.Put;
362    } else {
363      KeyValue.Type keyType = KeyValue.Type.values()[1 + rand.nextInt(NUM_VALID_KEY_TYPES)];
364      if (keyType == KeyValue.Type.Minimum || keyType == KeyValue.Type.Maximum) {
365        throw new RuntimeException("Generated an invalid key type: " + keyType + ". "
366          + "Probably the layout of KeyValue.Type has changed.");
367      }
368      return keyType;
369    }
