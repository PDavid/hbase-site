021import static org.junit.Assert.assertTrue;
022

024import java.util.Random;
025import org.apache.hadoop.conf.Configuration;
026import org.apache.hadoop.fs.FSDataInputStream;
027import org.apache.hadoop.fs.FSDataOutputStream;
028import org.apache.hadoop.fs.FileSystem;

030import org.apache.hadoop.hbase.HBaseClassTestRule;
031import org.apache.hadoop.hbase.HBaseConfiguration;
032import org.apache.hadoop.hbase.HBaseTestingUtil;
033import org.apache.hadoop.hbase.KeyValue;
034import org.apache.hadoop.hbase.fs.HFileSystem;
035import org.apache.hadoop.hbase.io.ByteBuffAllocator;
036import org.apache.hadoop.hbase.io.FSDataInputStreamWrapper;
037import org.apache.hadoop.hbase.io.compress.Compression;
038import org.apache.hadoop.hbase.testclassification.IOTests;
039import org.apache.hadoop.hbase.testclassification.MediumTests;
040import org.apache.hadoop.hbase.util.Bytes;
041import org.junit.Before;
042import org.junit.ClassRule;
043import org.junit.Rule;
044import org.junit.Test;
045import org.junit.experimental.categories.Category;
046import org.junit.rules.TestName;
047
048@Category({ IOTests.class, MediumTests.class })
049public class TestHFileBlockUnpack {
050
051  @ClassRule
052  public static final HBaseClassTestRule CLASS_RULE =
053    HBaseClassTestRule.forClass(TestHFileBlockUnpack.class);
054
055  private static final HBaseTestingUtil TEST_UTIL = new HBaseTestingUtil();
056
057  // repetition gives us some chance to get a good compression ratio
058  private static float CHANCE_TO_REPEAT = 0.6f;
059
060  private static final int MIN_ALLOCATION_SIZE = 10 * 1024;
061
062  ByteBuffAllocator allocator;
063
064  @Rule
065  public TestName name = new TestName();
066  private FileSystem fs;
067
068  @Before
069  public void setUp() throws Exception {
070    fs = HFileSystem.get(TEST_UTIL.getConfiguration());
071    Configuration conf = HBaseConfiguration.create(TEST_UTIL.getConfiguration());
072    conf.setInt(ByteBuffAllocator.MIN_ALLOCATE_SIZE_KEY, MIN_ALLOCATION_SIZE);
073    allocator = ByteBuffAllocator.create(conf, true);
074
075  }
076
077  /**
078   * If the block on disk size is less than {@link ByteBuffAllocator}'s min allocation size, that
079   * block will be allocated to heap regardless of desire for off-heap. After de-compressing the
080   * block, the new size may now exceed the min allocation size. This test ensures that those
081   * de-compressed blocks, which will be allocated off-heap, are properly marked as
082   * {@link HFileBlock#isSharedMem()} == true See https://issues.apache.org/jira/browse/HBASE-27170
083   */
084  @Test
085  public void itUsesSharedMemoryIfUnpackedBlockExceedsMinAllocationSize() throws IOException {
086    Configuration conf = TEST_UTIL.getConfiguration();
087    HFileContext meta =
088      new HFileContextBuilder().withCompression(Compression.Algorithm.GZ).withIncludesMvcc(false)
089        .withIncludesTags(false).withBytesPerCheckSum(HFile.DEFAULT_BYTES_PER_CHECKSUM).build();
090
091    Path path = new Path(TEST_UTIL.getDataTestDir(), name.getMethodName());
092    int totalSize;
093    try (FSDataOutputStream os = fs.create(path)) {
094      HFileBlock.Writer hbw = new HFileBlock.Writer(conf, NoOpDataBlockEncoder.INSTANCE, meta);
095      hbw.startWriting(BlockType.DATA);
096      writeTestKeyValues(hbw, MIN_ALLOCATION_SIZE - 1);
097      hbw.writeHeaderAndData(os);
098      totalSize = hbw.getOnDiskSizeWithHeader();
099      assertTrue(
100        "expected generated block size " + totalSize + " to be less than " + MIN_ALLOCATION_SIZE,
101        totalSize < MIN_ALLOCATION_SIZE);
102    }
103
104    try (FSDataInputStream is = fs.open(path)) {
105      meta =
106        new HFileContextBuilder().withHBaseCheckSum(true).withCompression(Compression.Algorithm.GZ)
107          .withIncludesMvcc(false).withIncludesTags(false).build();
108      ReaderContext context =
109        new ReaderContextBuilder().withInputStreamWrapper(new FSDataInputStreamWrapper(is))
110          .withFileSize(totalSize).withFilePath(path).withFileSystem(fs).build();
111      HFileBlock.FSReaderImpl hbr = new HFileBlock.FSReaderImpl(context, meta, allocator, conf);
112      hbr.setDataBlockEncoder(NoOpDataBlockEncoder.INSTANCE, conf);
113      hbr.setIncludesMemStoreTS(false);
114      HFileBlock blockFromHFile = hbr.readBlockData(0, -1, false, false, false);
115      blockFromHFile.sanityCheck();
116      assertFalse("expected hfile block to NOT be unpacked", blockFromHFile.isUnpacked());
117      assertFalse("expected hfile block to NOT use shared memory", blockFromHFile.isSharedMem());
118
119      assertTrue(
120        "expected generated block size " + blockFromHFile.getOnDiskSizeWithHeader()
121          + " to be less than " + MIN_ALLOCATION_SIZE,
122        blockFromHFile.getOnDiskSizeWithHeader() < MIN_ALLOCATION_SIZE);
123      assertTrue(
124        "expected generated block uncompressed size "
125          + blockFromHFile.getUncompressedSizeWithoutHeader() + " to be more than "
126          + MIN_ALLOCATION_SIZE,
127        blockFromHFile.getUncompressedSizeWithoutHeader() > MIN_ALLOCATION_SIZE);
128
129      HFileBlock blockUnpacked = blockFromHFile.unpack(meta, hbr);
130      assertTrue("expected unpacked block to be unpacked", blockUnpacked.isUnpacked());
131      assertTrue("expected unpacked block to use shared memory", blockUnpacked.isSharedMem());
132    }
133  }
134
135  static int writeTestKeyValues(HFileBlock.Writer hbw, int desiredSize) throws IOException {
136    Random random = new Random(42);
137
138    byte[] family = new byte[] { 1 };
139    int rowKey = 0;
140    int qualifier = 0;
141    int value = 0;
142    long timestamp = 0;
143
144    int totalSize = 0;
145
146    // go until just up to the limit. compression should bring the total on-disk size under
147    while (totalSize < desiredSize) {
148      rowKey = maybeIncrement(random, rowKey);
149      qualifier = maybeIncrement(random, qualifier);
150      value = maybeIncrement(random, value);
151      timestamp = maybeIncrement(random, (int) timestamp);
152
153      KeyValue keyValue = new KeyValue(Bytes.toBytes(rowKey), family, Bytes.toBytes(qualifier),
154        timestamp, Bytes.toBytes(value));
155      hbw.write(keyValue);
156      totalSize += keyValue.getLength();
157    }
158
159    return totalSize;
160  }
161
162  private static int maybeIncrement(Random random, int value) {
163    if (random.nextFloat() < CHANCE_TO_REPEAT) {
164      return value;
165    }
166    return value + 1;
167  }
