048/**
049 * The base class for the remote procedures used to open/close a region.
050 * <p/>
051 * Notice that here we do not care about the result of the remote call, if the remote call is
052 * finished, either succeeded or not, we will always finish the procedure. The parent procedure
053 * should take care of the result and try to reschedule if the result is not good.
054 */
055@InterfaceAudience.Private
056public abstract class RegionRemoteProcedureBase extends Procedure<MasterProcedureEnv>
057    implements TableProcedureInterface, RemoteProcedure<MasterProcedureEnv, ServerName> {
058
059  private static final Logger LOG = LoggerFactory.getLogger(RegionRemoteProcedureBase.class);
060
061  protected RegionInfo region;
062
063  protected ServerName targetServer;
064
065  private RegionRemoteProcedureBaseState state =
066    RegionRemoteProcedureBaseState.REGION_REMOTE_PROCEDURE_DISPATCH;
067
068  private TransitionCode transitionCode;
069
070  private long seqId;
071
072  private RetryCounter retryCounter;
073
074  protected RegionRemoteProcedureBase() {
075  }
076
077  protected RegionRemoteProcedureBase(TransitRegionStateProcedure parent, RegionInfo region,
078      ServerName targetServer) {
079    this.region = region;
080    this.targetServer = targetServer;
081    parent.attachRemoteProc(this);
082  }
083
084  @Override
085  public void remoteOperationCompleted(MasterProcedureEnv env) {
086    // should not be called since we use reportRegionStateTransition to report the result
087    throw new UnsupportedOperationException();
088  }
089
090  @Override
091  public void remoteOperationFailed(MasterProcedureEnv env, RemoteProcedureException error) {
092    // should not be called since we use reportRegionStateTransition to report the result
093    throw new UnsupportedOperationException();
094  }
095
096  private RegionStateNode getRegionNode(MasterProcedureEnv env) {
097    return env.getAssignmentManager().getRegionStates().getRegionStateNode(region);
098  }
099
100  @Override
101  public void remoteCallFailed(MasterProcedureEnv env, ServerName remote, IOException exception) {
102    RegionStateNode regionNode = getRegionNode(env);
103    regionNode.lock();
104    try {
105      if (!env.getMasterServices().getServerManager().isServerOnline(remote)) {
106        // the SCP will interrupt us, give up
107        LOG.debug("{} for region {}, targetServer {} is dead, SCP will interrupt us, give up", this,
108          regionNode, remote);
109        return;
110      }
111      if (state != RegionRemoteProcedureBaseState.REGION_REMOTE_PROCEDURE_DISPATCH) {
112        // not sure how can this happen but anyway let's add a check here to avoid waking the wrong
113        // procedure...
114        LOG.warn("{} for region {}, targetServer={} has already been woken up, ignore", this,
115          regionNode, remote);
116        return;
117      }
118      LOG.warn("The remote operation {} for region {} to server {} failed", this, regionNode,
119        remote, exception);
120      // It is OK to not persist the state here, as we do not need to change the region state if the
121      // remote call is failed. If the master crashed before we actually execute the procedure and
122      // persist the new state, it is fine to retry on the same target server again.
123      state = RegionRemoteProcedureBaseState.REGION_REMOTE_PROCEDURE_DISPATCH_FAIL;
124      regionNode.getProcedureEvent().wake(env.getProcedureScheduler());
125    } finally {
126      regionNode.unlock();
127    }
128  }
129
130  @Override
131  public TableName getTableName() {
132    return region.getTable();
133  }
134
135  @Override
136  protected boolean waitInitialized(MasterProcedureEnv env) {
137    if (TableName.isMetaTableName(getTableName())) {
138      return false;
139    }
140    // First we need meta to be loaded, and second, if meta is not online then we will likely to
141    // fail when updating meta so we wait until it is assigned.
142    AssignmentManager am = env.getAssignmentManager();
143    return am.waitMetaLoaded(this) || am.waitMetaAssigned(this, region);
144  }
145
146  @Override
147  protected void rollback(MasterProcedureEnv env) throws IOException, InterruptedException {
148    throw new UnsupportedOperationException();
149  }
150
151  @Override
152  protected boolean abort(MasterProcedureEnv env) {
153    return false;
154  }
155
156  // do some checks to see if the report is valid, without actually updating meta.
157  protected abstract void reportTransition(RegionStateNode regionNode,
158      TransitionCode transitionCode, long seqId) throws IOException;
159
160  // A bit strange but the procedure store will throw RuntimeException if we can not persist the
161  // state, so upper layer should take care of this...
162  private void persistAndWake(MasterProcedureEnv env, RegionStateNode regionNode) {
163    env.getMasterServices().getMasterProcedureExecutor().getStore().update(this);
164    regionNode.getProcedureEvent().wake(env.getProcedureScheduler());
165  }
166
167  // should be called with RegionStateNode locked, to avoid race with the execute method below
168  void reportTransition(MasterProcedureEnv env, RegionStateNode regionNode, ServerName serverName,
169      TransitionCode transitionCode, long seqId) throws IOException {
170    if (state != RegionRemoteProcedureBaseState.REGION_REMOTE_PROCEDURE_DISPATCH) {
171      // should be a retry
172      return;
173    }
174    if (!targetServer.equals(serverName)) {
175      throw new UnexpectedStateException("Received report from " + serverName + ", expected " +
176        targetServer + ", " + regionNode + ", proc=" + this);
177    }
178    reportTransition(regionNode, transitionCode, seqId);
179    // this state means we have received the report from RS, does not mean the result is fine, as we
180    // may received a FAILED_OPEN.
181    this.state = RegionRemoteProcedureBaseState.REGION_REMOTE_PROCEDURE_REPORT_SUCCEED;
182    this.transitionCode = transitionCode;
183    this.seqId = seqId;
184    // Persist the transition code and openSeqNum(if provided).
185    // We should not update the hbase:meta directly as this may cause races when master restarts,
186    // as the old active master may incorrectly report back to RS and cause the new master to hang
187    // on a OpenRegionProcedure forever. See HBASE-22060 and HBASE-22074 for more details.
188    boolean succ = false;
189    try {
190      persistAndWake(env, regionNode);
191      succ = true;
192    } finally {
193      if (!succ) {
194        this.state = RegionRemoteProcedureBaseState.REGION_REMOTE_PROCEDURE_DISPATCH;
195        this.transitionCode = null;
196        this.seqId = HConstants.NO_SEQNUM;
197      }
198    }
199  }
200
201  void serverCrashed(MasterProcedureEnv env, RegionStateNode regionNode, ServerName serverName) {
202    if (state != RegionRemoteProcedureBaseState.REGION_REMOTE_PROCEDURE_DISPATCH) {
203      // should be a retry
204      return;
205    }
206    this.state = RegionRemoteProcedureBaseState.REGION_REMOTE_PROCEDURE_SERVER_CRASH;
207    boolean succ = false;
208    try {
209      persistAndWake(env, regionNode);
210      succ = true;
211    } finally {
212      if (!succ) {
213        this.state = RegionRemoteProcedureBaseState.REGION_REMOTE_PROCEDURE_DISPATCH;
214      }
215    }
216  }

218  private TransitRegionStateProcedure getParent(MasterProcedureEnv env) {
219    return (TransitRegionStateProcedure) env.getMasterServices().getMasterProcedureExecutor()
220      .getProcedure(getParentProcId());
221  }
222
223  private void unattach(MasterProcedureEnv env) {
224    getParent(env).unattachRemoteProc(this);
225  }
226
227  // actually update the state to meta
228  protected abstract void updateTransition(MasterProcedureEnv env, RegionStateNode regionNode,
229      TransitionCode transitionCode, long seqId) throws IOException;
230
231  @Override
232  protected Procedure<MasterProcedureEnv>[] execute(MasterProcedureEnv env)
233      throws ProcedureYieldException, ProcedureSuspendedException, InterruptedException {
234    RegionStateNode regionNode = getRegionNode(env);
235    regionNode.lock();
236    try {
237      switch (state) {
238        case REGION_REMOTE_PROCEDURE_DISPATCH: {
239          // The code which wakes us up also needs to lock the RSN so here we do not need to
240          // synchronize
241          // on the event.
242          ProcedureEvent<?> event = regionNode.getProcedureEvent();
243          try {
244            env.getRemoteDispatcher().addOperationToNode(targetServer, this);
245          } catch (FailedRemoteDispatchException e) {
246            LOG.warn("Can not add remote operation {} for region {} to server {}, this usually " +
247              "because the server is alread dead, give up and mark the procedure as complete, " +
248              "the parent procedure will take care of this.", this, region, targetServer, e);
249            unattach(env);
250            return null;
251          }
252          event.suspend();
253          event.suspendIfNotReady(this);
254          throw new ProcedureSuspendedException();
255        }
256        case REGION_REMOTE_PROCEDURE_REPORT_SUCCEED:
257          updateTransition(env, regionNode, transitionCode, seqId);
258          unattach(env);
259          return null;
260        case REGION_REMOTE_PROCEDURE_DISPATCH_FAIL:
261          // the remote call is failed so we do not need to change the region state, just return.
262          unattach(env);
263          return null;
264        case REGION_REMOTE_PROCEDURE_SERVER_CRASH:
265          env.getAssignmentManager().regionClosed(regionNode, false);
266          unattach(env);
267          return null;
268        default:
269          throw new IllegalStateException("Unknown state: " + state);
270      }
271    } catch (IOException e) {
272      if (retryCounter == null) {
273        retryCounter = ProcedureUtil.createRetryCounter(env.getMasterConfiguration());
274      }
275      long backoff = retryCounter.getBackoffTimeAndIncrementAttempts();
276      LOG.warn("Failed updating meta, suspend {}secs {}; {};", backoff / 1000, this, regionNode, e);
277      setTimeout(Math.toIntExact(backoff));
278      setState(ProcedureProtos.ProcedureState.WAITING_TIMEOUT);
279      skipPersistence();
280      throw new ProcedureSuspendedException();
281    } finally {
282      regionNode.unlock();
283    }
284  }

286  @Override
287  public boolean storeInDispatchedQueue() {
288    return false;
289  }
290
291  @Override
292  protected void serializeStateData(ProcedureStateSerializer serializer) throws IOException {
293    RegionRemoteProcedureBaseStateData.Builder builder =
294      RegionRemoteProcedureBaseStateData.newBuilder().setRegion(ProtobufUtil.toRegionInfo(region))
295        .setTargetServer(ProtobufUtil.toServerName(targetServer)).setState(state);
296    if (transitionCode != null) {
297      builder.setTransitionCode(transitionCode);
298      builder.setSeqId(seqId);
299    }
300    serializer.serialize(builder.build());
301  }
302
303  @Override
304  protected void deserializeStateData(ProcedureStateSerializer serializer) throws IOException {
305    RegionRemoteProcedureBaseStateData data =
306      serializer.deserialize(RegionRemoteProcedureBaseStateData.class);
307    region = ProtobufUtil.toRegionInfo(data.getRegion());
308    targetServer = ProtobufUtil.toServerName(data.getTargetServer());
309    state = data.getState();
310    if (data.hasTransitionCode()) {
311      transitionCode = data.getTransitionCode();
312      seqId = data.getSeqId();
313    }
314  }
315
316  @Override
317  protected void afterReplay(MasterProcedureEnv env) {
318    getParent(env).attachRemoteProc(this);
319  }
