107  public void visitMeta(final RegionStateVisitor visitor) throws IOException {
108    MetaTableAccessor.fullScanRegions(master.getConnection(),
109      new ClientMetaTableAccessor.Visitor() {
110        final boolean isDebugEnabled = LOG.isDebugEnabled();
111
112        @Override
113        public boolean visit(final Result r) throws IOException {
114          if (r != null && !r.isEmpty()) {
115            long st = 0;
116            if (LOG.isTraceEnabled()) {
117              st = EnvironmentEdgeManager.currentTime();
118            }
119            visitMetaEntry(visitor, r);
120            if (LOG.isTraceEnabled()) {
121              long et = EnvironmentEdgeManager.currentTime();
122              LOG.trace("[T] LOAD META PERF " + StringUtils.humanTimeDiff(et - st));
123            }
124          } else if (isDebugEnabled) {
125            LOG.debug("NULL result from meta - ignoring but this is strange.");
126          }
127          return true;
128        }
129      });
130  }

132  /**
133   * Queries META table for the passed region encoded name, delegating action upon results to the
134   * <code>RegionStateVisitor</code> passed as second parameter.
135   * @param regionEncodedName encoded name for the Region we want to query META for.
136   * @param visitor The <code>RegionStateVisitor</code> instance to react over the query results.
137   * @throws IOException If some error occurs while querying META or parsing results.
138   */
139  public void visitMetaForRegion(final String regionEncodedName, final RegionStateVisitor visitor)
140    throws IOException {
141    Result result =
142      MetaTableAccessor.scanByRegionEncodedName(master.getConnection(), regionEncodedName);
143    if (result != null) {
144      visitMetaEntry(visitor, result);
145    }
146  }

148  public static void visitMetaEntry(final RegionStateVisitor visitor, final Result result)
149      throws IOException {
150    final RegionLocations rl = CatalogFamilyFormat.getRegionLocations(result);
151    if (rl == null) return;
152
153    final HRegionLocation[] locations = rl.getRegionLocations();
154    if (locations == null) return;
155
156    for (int i = 0; i < locations.length; ++i) {
157      final HRegionLocation hrl = locations[i];
158      if (hrl == null) continue;
159
160      final RegionInfo regionInfo = hrl.getRegion();
161      if (regionInfo == null) continue;
162
163      final int replicaId = regionInfo.getReplicaId();
164      final State state = getRegionState(result, regionInfo);
165
166      final ServerName lastHost = hrl.getServerName();
167      ServerName regionLocation = MetaTableAccessor.getTargetServerName(result, replicaId);
168      final long openSeqNum = hrl.getSeqNum();
169
170      LOG.debug(
171        "Load hbase:meta entry region={}, regionState={}, lastHost={}, " +
172          "regionLocation={}, openSeqNum={}",
173        regionInfo.getEncodedName(), state, lastHost, regionLocation, openSeqNum);
174      visitor.visitRegionState(result, regionInfo, state, regionLocation, lastHost, openSeqNum);
175    }
176  }

178  void updateRegionLocation(RegionStateNode regionStateNode) throws IOException {
179    long time = EnvironmentEdgeManager.currentTime();
180    long openSeqNum = regionStateNode.getState() == State.OPEN ? regionStateNode.getOpenSeqNum() :
181      HConstants.NO_SEQNUM;
182    RegionInfo regionInfo = regionStateNode.getRegionInfo();
183    State state = regionStateNode.getState();
184    ServerName regionLocation = regionStateNode.getRegionLocation();
185    TransitRegionStateProcedure rit = regionStateNode.getProcedure();
186    long pid = rit != null ? rit.getProcId() : Procedure.NO_PROC_ID;
187    final int replicaId = regionInfo.getReplicaId();
188    final Put put = new Put(CatalogFamilyFormat.getMetaKeyForRegion(regionInfo), time);
189    MetaTableAccessor.addRegionInfo(put, regionInfo);
190    final StringBuilder info =
191      new StringBuilder("pid=").append(pid).append(" updating hbase:meta row=")
192        .append(regionInfo.getEncodedName()).append(", regionState=").append(state);
193    if (openSeqNum >= 0) {
194      Preconditions.checkArgument(state == State.OPEN && regionLocation != null,
195        "Open region should be on a server");
196      MetaTableAccessor.addLocation(put, regionLocation, openSeqNum, replicaId);
197      // only update replication barrier for default replica
198      if (regionInfo.getReplicaId() == RegionInfo.DEFAULT_REPLICA_ID &&
199        hasGlobalReplicationScope(regionInfo.getTable())) {
200        ReplicationBarrierFamilyFormat.addReplicationBarrier(put, openSeqNum);
201        info.append(", repBarrier=").append(openSeqNum);
202      }
203      info.append(", openSeqNum=").append(openSeqNum);
204      info.append(", regionLocation=").append(regionLocation);
205    } else if (regionLocation != null) {
206      // Ideally, if no regionLocation, write null to the hbase:meta but this will confuse clients
207      // currently; they want a server to hit. TODO: Make clients wait if no location.
208      put.add(CellBuilderFactory.create(CellBuilderType.SHALLOW_COPY).setRow(put.getRow())
209        .setFamily(HConstants.CATALOG_FAMILY)
210        .setQualifier(CatalogFamilyFormat.getServerNameColumn(replicaId))
211        .setTimestamp(put.getTimestamp()).setType(Cell.Type.Put)
212        .setValue(Bytes.toBytes(regionLocation.getServerName())).build());
213      info.append(", regionLocation=").append(regionLocation);
214    }
215    put.add(CellBuilderFactory.create(CellBuilderType.SHALLOW_COPY).setRow(put.getRow())
216      .setFamily(HConstants.CATALOG_FAMILY).setQualifier(getStateColumn(replicaId))
217      .setTimestamp(put.getTimestamp()).setType(Cell.Type.Put).setValue(Bytes.toBytes(state.name()))
218      .build());
219    LOG.info(info.toString());
220    updateRegionLocation(regionInfo, state, put);
221    if (regionInfo.isMetaRegion() && regionInfo.isFirst()) {
222      // mirror the meta location to zookeeper
223      mirrorMetaLocation(regionInfo, regionLocation, state);
224    }
225  }

227  private void mirrorMetaLocation(RegionInfo regionInfo, ServerName serverName, State state)
228    throws IOException {
229    try {
230      MetaTableLocator.setMetaLocation(master.getZooKeeper(), serverName, regionInfo.getReplicaId(),
231        state);
232    } catch (KeeperException e) {
233      throw new IOException(e);
234    }
235  }

237  private void removeMirrorMetaLocation(int oldReplicaCount, int newReplicaCount)
238    throws IOException {
239    try {
240      for (int i = newReplicaCount; i < oldReplicaCount; i++) {
241        MetaTableLocator.deleteMetaLocation(master.getZooKeeper(), i);
242      }
243    } catch (KeeperException e) {
244      throw new IOException(e);
245    }
246  }

248  private void updateRegionLocation(RegionInfo regionInfo, State state, Put put)
249    throws IOException {
250    try {
251      if (regionInfo.isMetaRegion()) {
252        masterRegion.update(r -> r.put(put));
253      } else {
254        try (Table table = master.getConnection().getTable(TableName.META_TABLE_NAME)) {
255          table.put(put);
256        }
257      }
258    } catch (IOException e) {
259      // TODO: Revist!!!! Means that if a server is loaded, then we will abort our host!
260      // In tests we abort the Master!
261      String msg = String.format("FAILED persisting region=%s state=%s",
262        regionInfo.getShortNameToLog(), state);
263      LOG.error(msg, e);
264      master.abort(msg, e);
265      throw e;
266    }
267  }

269  private long getOpenSeqNumForParentRegion(RegionInfo region) throws IOException {
270    MasterFileSystem fs = master.getMasterFileSystem();
271    long maxSeqId = WALSplitUtil.getMaxRegionSequenceId(master.getConfiguration(), region,
272      fs::getFileSystem, fs::getWALFileSystem);
273    return maxSeqId > 0 ? maxSeqId + 1 : HConstants.NO_SEQNUM;

276  /**
277   * Performs an atomic multi-mutate operation against the given table. Used by the likes of merge
278   * and split as these want to make atomic mutations across multiple rows.
279   */
280  private void multiMutate(RegionInfo ri, List<Mutation> mutations) throws IOException {
281    debugLogMutations(mutations);
282    byte[] row =
283      Bytes.toBytes(RegionReplicaUtil.getRegionInfoForDefaultReplica(ri).getRegionNameAsString() +
284        HConstants.DELIMITER);
285    MutateRowsRequest.Builder builder = MutateRowsRequest.newBuilder();
286    for (Mutation mutation : mutations) {
287      if (mutation instanceof Put) {
288        builder.addMutationRequest(
289          ProtobufUtil.toMutation(ClientProtos.MutationProto.MutationType.PUT, mutation));
290      } else if (mutation instanceof Delete) {
291        builder.addMutationRequest(
292          ProtobufUtil.toMutation(ClientProtos.MutationProto.MutationType.DELETE, mutation));
293      } else {
294        throw new DoNotRetryIOException(
295          "multi in MetaEditor doesn't support " + mutation.getClass().getName());
296      }
297    }
298    MutateRowsRequest request = builder.build();
299    AsyncTable<?> table =
300      master.getConnection().toAsyncConnection().getTable(TableName.META_TABLE_NAME);
301    CompletableFuture<MutateRowsResponse> future =
302      table.<MultiRowMutationService, MutateRowsResponse> coprocessorService(
303        MultiRowMutationService::newStub,
304        (stub, controller, done) -> stub.mutateRows(controller, request, done), row);
305    FutureUtils.get(future);
306  }

330  /**
331   * Splits the region into two in an atomic operation. Offlines the parent region with the
332   * information that it is split into two, and also adds the daughter regions. Does not add the
333   * location information to the daughter regions since they are not open yet.
334   */
335  public void splitRegion(RegionInfo parent, RegionInfo splitA, RegionInfo splitB,
336    ServerName serverName, TableDescriptor htd) throws IOException {
337    long parentOpenSeqNum = HConstants.NO_SEQNUM;
338    if (htd.hasGlobalReplicationScope()) {
339      parentOpenSeqNum = getOpenSeqNumForParentRegion(parent);
340    }
341    long time = EnvironmentEdgeManager.currentTime();
342    // Put for parent
343    Put putParent = MetaTableAccessor.makePutFromRegionInfo(
344      RegionInfoBuilder.newBuilder(parent).setOffline(true).setSplit(true).build(), time);
345    MetaTableAccessor.addDaughtersToPut(putParent, splitA, splitB);
346
347    // Puts for daughters
348    Put putA = MetaTableAccessor.makePutFromRegionInfo(splitA, time);
349    Put putB = MetaTableAccessor.makePutFromRegionInfo(splitB, time);
350    if (parentOpenSeqNum > 0) {
351      ReplicationBarrierFamilyFormat.addReplicationBarrier(putParent, parentOpenSeqNum);
352      ReplicationBarrierFamilyFormat.addReplicationParent(putA, Collections.singletonList(parent));
353      ReplicationBarrierFamilyFormat.addReplicationParent(putB, Collections.singletonList(parent));
354    }
355    // Set initial state to CLOSED
356    // NOTE: If initial state is not set to CLOSED then daughter regions get added with the
357    // default OFFLINE state. If Master gets restarted after this step, start up sequence of
358    // master tries to assign these offline regions. This is followed by re-assignments of the
359    // daughter regions from resumed {@link SplitTableRegionProcedure}
360    MetaTableAccessor.addRegionStateToPut(putA, RegionState.State.CLOSED);
361    MetaTableAccessor.addRegionStateToPut(putB, RegionState.State.CLOSED);
362
363    // new regions, openSeqNum = 1 is fine.
364    addSequenceNum(putA, 1, splitA.getReplicaId());
365    addSequenceNum(putB, 1, splitB.getReplicaId());
366
367    // Add empty locations for region replicas of daughters so that number of replicas can be
368    // cached whenever the primary region is looked up from meta
369    int regionReplication = getRegionReplication(htd);
370    for (int i = 1; i < regionReplication; i++) {
371      MetaTableAccessor.addEmptyLocation(putA, i);
372      MetaTableAccessor.addEmptyLocation(putB, i);
373    }
374
375    multiMutate(parent, Arrays.asList(putParent, putA, putB));
376  }

381  public void mergeRegions(RegionInfo child, RegionInfo[] parents, ServerName serverName,
382    TableDescriptor htd) throws IOException {
383    boolean globalScope = htd.hasGlobalReplicationScope();
384    long time = HConstants.LATEST_TIMESTAMP;
385    List<Mutation> mutations = new ArrayList<>();
386    List<RegionInfo> replicationParents = new ArrayList<>();
387    for (RegionInfo ri : parents) {
388      long seqNum = globalScope ? getOpenSeqNumForParentRegion(ri) : -1;
389      // Deletes for merging regions
390      mutations.add(MetaTableAccessor.makeDeleteFromRegionInfo(ri, time));
391      if (seqNum > 0) {
392        mutations
393          .add(ReplicationBarrierFamilyFormat.makePutForReplicationBarrier(ri, seqNum, time));
394        replicationParents.add(ri);
395      }
396    }
397    // Put for parent
398    Put putOfMerged = MetaTableAccessor.makePutFromRegionInfo(child, time);
399    putOfMerged = addMergeRegions(putOfMerged, Arrays.asList(parents));
400    // Set initial state to CLOSED.
401    // NOTE: If initial state is not set to CLOSED then merged region gets added with the
402    // default OFFLINE state. If Master gets restarted after this step, start up sequence of
403    // master tries to assign this offline region. This is followed by re-assignments of the
404    // merged region from resumed {@link MergeTableRegionsProcedure}
405    MetaTableAccessor.addRegionStateToPut(putOfMerged, RegionState.State.CLOSED);
406    mutations.add(putOfMerged);
407    // The merged is a new region, openSeqNum = 1 is fine. ServerName may be null
408    // if crash after merge happened but before we got to here.. means in-memory
409    // locations of offlined merged, now-closed, regions is lost. Should be ok. We
410    // assign the merged region later.
411    if (serverName != null) {
412      MetaTableAccessor.addLocation(putOfMerged, serverName, 1, child.getReplicaId());
413    }
414
415    // Add empty locations for region replicas of the merged region so that number of replicas
416    // can be cached whenever the primary region is looked up from meta
417    int regionReplication = getRegionReplication(htd);
418    for (int i = 1; i < regionReplication; i++) {
419      MetaTableAccessor.addEmptyLocation(putOfMerged, i);
420    }
421    // add parent reference for serial replication
422    if (!replicationParents.isEmpty()) {
423      ReplicationBarrierFamilyFormat.addReplicationParent(putOfMerged, replicationParents);
424    }
425    multiMutate(child, mutations);
426  }

442  /**
443   * Deletes merge qualifiers for the specified merge region.
444   * @param connection connection we're using
445   * @param mergeRegion the merged region
446   */
447  public void deleteMergeQualifiers(RegionInfo mergeRegion) throws IOException {
448    // NOTE: We are doing a new hbase:meta read here.
449    Cell[] cells = getRegionCatalogResult(mergeRegion).rawCells();
450    if (cells == null || cells.length == 0) {
451      return;
452    }
453    Delete delete = new Delete(mergeRegion.getRegionName());
454    List<byte[]> qualifiers = new ArrayList<>();
455    for (Cell cell : cells) {
456      if (!CatalogFamilyFormat.isMergeQualifierPrefix(cell)) {
457        continue;
458      }
459      byte[] qualifier = CellUtil.cloneQualifier(cell);
460      qualifiers.add(qualifier);
461      delete.addColumns(HConstants.CATALOG_FAMILY, qualifier, HConstants.LATEST_TIMESTAMP);
462    }
463
464    // There will be race condition that a GCMultipleMergedRegionsProcedure is scheduled while
465    // the previous GCMultipleMergedRegionsProcedure is still going on, in this case, the second
466    // GCMultipleMergedRegionsProcedure could delete the merged region by accident!
467    if (qualifiers.isEmpty()) {
468      LOG.info("No merged qualifiers for region " + mergeRegion.getRegionNameAsString() +
469        " in meta table, they are cleaned up already, Skip.");
470      return;
471    }
472    try (Table table = master.getConnection().getTable(TableName.META_TABLE_NAME)) {
473      table.delete(delete);
474    }
475    LOG.info("Deleted merge references in " + mergeRegion.getRegionNameAsString() +
476      ", deleted qualifiers " +
477      qualifiers.stream().map(Bytes::toStringBinary).collect(Collectors.joining(", ")));
478  }

480  static Put addMergeRegions(Put put, Collection<RegionInfo> mergeRegions) throws IOException {
481    int limit = 10000; // Arbitrary limit. No room in our formatted 'task0000' below for more.
482    int max = mergeRegions.size();
483    if (max > limit) {
484      // Should never happen!!!!! But just in case.
485      throw new RuntimeException(
486        "Can't merge " + max + " regions in one go; " + limit + " is upper-limit.");
487    }
488    int counter = 0;
489    for (RegionInfo ri : mergeRegions) {
490      String qualifier = String.format(HConstants.MERGE_QUALIFIER_PREFIX_STR + "%04d", counter++);
491      put.add(CellBuilderFactory.create(CellBuilderType.SHALLOW_COPY).setRow(put.getRow())
492        .setFamily(HConstants.CATALOG_FAMILY).setQualifier(Bytes.toBytes(qualifier))
493        .setTimestamp(put.getTimestamp()).setType(Type.Put).setValue(RegionInfo.toByteArray(ri))
494        .build());
495    }
496    return put;
497  }

516  private void deleteRegions(List<RegionInfo> regions, long ts) throws IOException {
517    List<Delete> deletes = new ArrayList<>(regions.size());
518    for (RegionInfo hri : regions) {
519      Delete e = new Delete(hri.getRegionName());
520      e.addFamily(HConstants.CATALOG_FAMILY, ts);
521      deletes.add(e);
522    }
523    try (Table table = getMetaTable()) {
524      debugLogMutations(deletes);
525      table.delete(deletes);
526    }
527    LOG.info("Deleted {} regions from META", regions.size());
528    LOG.debug("Deleted regions: {}", regions);
529  }

531  /**
532   * Overwrites the specified regions from hbase:meta. Deletes old rows for the given regions and
533   * adds new ones. Regions added back have state CLOSED.
534   * @param connection connection we're using
535   * @param regionInfos list of regions to be added to META
536   */
537  public void overwriteRegions(List<RegionInfo> regionInfos, int regionReplication)
538    throws IOException {
539    // use master time for delete marker and the Put
540    long now = EnvironmentEdgeManager.currentTime();
541    deleteRegions(regionInfos, now);
542    // Why sleep? This is the easiest way to ensure that the previous deletes does not
543    // eclipse the following puts, that might happen in the same ts from the server.
544    // See HBASE-9906, and HBASE-9879. Once either HBASE-9879, HBASE-8770 is fixed,
545    // or HBASE-9905 is fixed and meta uses seqIds, we do not need the sleep.
546    //
547    // HBASE-13875 uses master timestamp for the mutations. The 20ms sleep is not needed
548    MetaTableAccessor.addRegionsToMeta(master.getConnection(), regionInfos, regionReplication,
549      now + 1);
550    LOG.info("Overwritten " + regionInfos.size() + " regions to Meta");
551    LOG.debug("Overwritten regions: {} ", regionInfos);
552  }

554  private Scan getScanForUpdateRegionReplicas(TableName tableName) {
555    Scan scan;
556    if (TableName.isMetaTableName(tableName)) {
557      // Notice that, we do not use MetaCellComparator for master local region, so we can not use
558      // the same logic to set start key and end key for scanning meta table when locating entries
559      // in master local region. And since there is only one table in master local region(the record
560      // for meta table), so we do not need set start key and end key.
561      scan = new Scan();
562    } else {
563      scan = MetaTableAccessor.getScanForTableName(master.getConfiguration(), tableName);
564    }
565    return scan.addColumn(HConstants.CATALOG_FAMILY, HConstants.REGIONINFO_QUALIFIER);
566  }
567
568  private List<Delete> deleteRegionReplicas(ResultScanner scanner, int oldReplicaCount,
569    int newReplicaCount, long now) throws IOException {
570    List<Delete> deletes = new ArrayList<>();
571    for (;;) {
572      Result result = scanner.next();
573      if (result == null) {
574        break;
575      }
576      RegionInfo primaryRegionInfo = CatalogFamilyFormat.getRegionInfo(result);
577      if (primaryRegionInfo == null || primaryRegionInfo.isSplit()) {
578        continue;
579      }
580      Delete delete = new Delete(result.getRow());
581      for (int i = newReplicaCount; i < oldReplicaCount; i++) {
582        delete.addColumns(HConstants.CATALOG_FAMILY, CatalogFamilyFormat.getServerColumn(i), now);
583        delete.addColumns(HConstants.CATALOG_FAMILY, CatalogFamilyFormat.getSeqNumColumn(i), now);
584        delete.addColumns(HConstants.CATALOG_FAMILY, CatalogFamilyFormat.getStartCodeColumn(i),
585          now);
586        delete.addColumns(HConstants.CATALOG_FAMILY, CatalogFamilyFormat.getServerNameColumn(i),
587          now);
588        delete.addColumns(HConstants.CATALOG_FAMILY, CatalogFamilyFormat.getRegionStateColumn(i),
589          now);
590      }
591      deletes.add(delete);
592    }
593    return deletes;
594  }
595
596
597  public void removeRegionReplicas(TableName tableName, int oldReplicaCount, int newReplicaCount)
598    throws IOException {
599    Scan scan = getScanForUpdateRegionReplicas(tableName);
600    long now = EnvironmentEdgeManager.currentTime();
601    if (TableName.isMetaTableName(tableName)) {
602      List<Delete> deletes;
603      try (ResultScanner scanner = masterRegion.getScanner(scan)) {
604        deletes = deleteRegionReplicas(scanner, oldReplicaCount, newReplicaCount, now);
605      }
606      debugLogMutations(deletes);
607      masterRegion.update(r -> {
608        for (Delete d : deletes) {
609          r.delete(d);
610        }
611      });
612      // also delete the mirrored location on zk
613      removeMirrorMetaLocation(oldReplicaCount, newReplicaCount);
614    } else {
615      try (Table metaTable = getMetaTable(); ResultScanner scanner = metaTable.getScanner(scan)) {
616        List<Delete> deletes = deleteRegionReplicas(scanner, oldReplicaCount, newReplicaCount, now);
617        debugLogMutations(deletes);
618        metaTable.delete(deletes);
619      }
620    }
621  }

646  /**
647   * Pull the region state from a catalog table {@link Result}.
648   * @return the region state, or null if unknown.
649   */
650  public static State getRegionState(final Result r, RegionInfo regionInfo) {
651    Cell cell =
652      r.getColumnLatestCell(HConstants.CATALOG_FAMILY, getStateColumn(regionInfo.getReplicaId()));
653    if (cell == null || cell.getValueLength() == 0) {
654      return null;
655    }
656
657    String state =
658      Bytes.toString(cell.getValueArray(), cell.getValueOffset(), cell.getValueLength());
659    try {
660      return State.valueOf(state);
661    } catch (IllegalArgumentException e) {
662      LOG.warn(
663        "BAD value {} in hbase:meta info:state column for region {} , " +
664          "Consider using HBCK2 setRegionState ENCODED_REGION_NAME STATE",
665        state, regionInfo.getEncodedName());
666      return null;
667    }
668  }
669
670  public static byte[] getStateColumn(int replicaId) {
671    return replicaId == 0 ? HConstants.STATE_QUALIFIER :
672      Bytes.toBytes(HConstants.STATE_QUALIFIER_STR + META_REPLICA_ID_DELIMITER +
673        String.format(RegionInfo.REPLICA_ID_FORMAT, replicaId));
674  }

676  private static void debugLogMutations(List<? extends Mutation> mutations) throws IOException {
677    if (!METALOG.isDebugEnabled()) {
678      return;
679    }
680    // Logging each mutation in separate line makes it easier to see diff between them visually
681    // because of common starting indentation.
682    for (Mutation mutation : mutations) {
683      debugLogMutation(mutation);
684    }
685  }
686
687  private static void debugLogMutation(Mutation p) throws IOException {
688    METALOG.debug("{} {}", p.getClass().getSimpleName(), p.toJSON());
689  }
