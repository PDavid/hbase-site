101  public void visitMeta(final RegionStateVisitor visitor) throws IOException {
102    MetaTableAccessor.fullScanRegions(master.getConnection(),
103      new ClientMetaTableAccessor.Visitor() {
104        final boolean isDebugEnabled = LOG.isDebugEnabled();
105
106        @Override
107        public boolean visit(final Result r) throws IOException {
108          if (r != null && !r.isEmpty()) {
109            long st = 0;
110            if (LOG.isTraceEnabled()) {
111              st = System.currentTimeMillis();
112            }
113            visitMetaEntry(visitor, r);
114            if (LOG.isTraceEnabled()) {
115              long et = System.currentTimeMillis();
116              LOG.trace("[T] LOAD META PERF " + StringUtils.humanTimeDiff(et - st));
117            }
118          } else if (isDebugEnabled) {
119            LOG.debug("NULL result from meta - ignoring but this is strange.");
120          }
121          return true;
122        }
123      });
124  }

126  /**
127   * Queries META table for the passed region encoded name,
128   * delegating action upon results to the <code>RegionStateVisitor</code>
129   * passed as second parameter.
130   * @param regionEncodedName encoded name for the Region we want to query META for.
131   * @param visitor The <code>RegionStateVisitor</code> instance to react over the query results.
132   * @throws IOException If some error occurs while querying META or parsing results.
133   */
134  public void visitMetaForRegion(final String regionEncodedName, final RegionStateVisitor visitor)
135      throws IOException {
136    Result result = MetaTableAccessor.
137      scanByRegionEncodedName(master.getConnection(), regionEncodedName);
138    if (result != null) {
139      visitMetaEntry(visitor, result);
140    }
141  }
142
143  private void visitMetaEntry(final RegionStateVisitor visitor, final Result result)
144      throws IOException {
145    final RegionLocations rl = CatalogFamilyFormat.getRegionLocations(result);
146    if (rl == null) return;
147
148    final HRegionLocation[] locations = rl.getRegionLocations();
149    if (locations == null) return;
150
151    for (int i = 0; i < locations.length; ++i) {
152      final HRegionLocation hrl = locations[i];
153      if (hrl == null) continue;
154
155      final RegionInfo regionInfo = hrl.getRegion();
156      if (regionInfo == null) continue;
157
158      final int replicaId = regionInfo.getReplicaId();
159      final State state = getRegionState(result, regionInfo);
160
161      final ServerName lastHost = hrl.getServerName();
162      ServerName regionLocation = MetaTableAccessor.getTargetServerName(result, replicaId);
163      final long openSeqNum = hrl.getSeqNum();
164
165      // TODO: move under trace, now is visible for debugging
166      LOG.info(
167        "Load hbase:meta entry region={}, regionState={}, lastHost={}, " +
168          "regionLocation={}, openSeqNum={}",
169        regionInfo.getEncodedName(), state, lastHost, regionLocation, openSeqNum);
170      visitor.visitRegionState(result, regionInfo, state, regionLocation, lastHost, openSeqNum);
171    }
172  }

174  void updateRegionLocation(RegionStateNode regionStateNode) throws IOException {
175    if (regionStateNode.getRegionInfo().isMetaRegion()) {
176      updateMetaLocation(regionStateNode.getRegionInfo(), regionStateNode.getRegionLocation(),
177        regionStateNode.getState());
178    } else {
179      long openSeqNum = regionStateNode.getState() == State.OPEN ? regionStateNode.getOpenSeqNum()
180        : HConstants.NO_SEQNUM;
181      updateUserRegionLocation(regionStateNode.getRegionInfo(), regionStateNode.getState(),
182        regionStateNode.getRegionLocation(), openSeqNum,
183        // The regionStateNode may have no procedure in a test scenario; allow for this.
184        regionStateNode.getProcedure() != null ? regionStateNode.getProcedure().getProcId()
185          : Procedure.NO_PROC_ID);
186    }
187  }

189  private void updateMetaLocation(RegionInfo regionInfo, ServerName serverName, State state)
190      throws IOException {
191    try {
192      MetaTableLocator.setMetaLocation(master.getZooKeeper(), serverName, regionInfo.getReplicaId(),
193        state);
194    } catch (KeeperException e) {
195      throw new IOException(e);
196    }
197  }

199  private void updateUserRegionLocation(RegionInfo regionInfo, State state,
200      ServerName regionLocation, long openSeqNum,
201       long pid) throws IOException {
202    long time = EnvironmentEdgeManager.currentTime();
203    final int replicaId = regionInfo.getReplicaId();
204    final Put put = new Put(CatalogFamilyFormat.getMetaKeyForRegion(regionInfo), time);
205    MetaTableAccessor.addRegionInfo(put, regionInfo);
206    final StringBuilder info =
207      new StringBuilder("pid=").append(pid).append(" updating hbase:meta row=")
208        .append(regionInfo.getEncodedName()).append(", regionState=").append(state);
209    if (openSeqNum >= 0) {
210      Preconditions.checkArgument(state == State.OPEN && regionLocation != null,
211          "Open region should be on a server");
212      MetaTableAccessor.addLocation(put, regionLocation, openSeqNum, replicaId);
213      // only update replication barrier for default replica
214      if (regionInfo.getReplicaId() == RegionInfo.DEFAULT_REPLICA_ID &&
215        hasGlobalReplicationScope(regionInfo.getTable())) {
216        ReplicationBarrierFamilyFormat.addReplicationBarrier(put, openSeqNum);
217        info.append(", repBarrier=").append(openSeqNum);
218      }
219      info.append(", openSeqNum=").append(openSeqNum);
220      info.append(", regionLocation=").append(regionLocation);
221    } else if (regionLocation != null) {
222      // Ideally, if no regionLocation, write null to the hbase:meta but this will confuse clients
223      // currently; they want a server to hit. TODO: Make clients wait if no location.
224      put.add(CellBuilderFactory.create(CellBuilderType.SHALLOW_COPY)
225          .setRow(put.getRow())
226          .setFamily(HConstants.CATALOG_FAMILY)
227          .setQualifier(CatalogFamilyFormat.getServerNameColumn(replicaId))
228          .setTimestamp(put.getTimestamp())
229          .setType(Cell.Type.Put)
230          .setValue(Bytes.toBytes(regionLocation.getServerName()))
231          .build());
232      info.append(", regionLocation=").append(regionLocation);
233    }
234    put.add(CellBuilderFactory.create(CellBuilderType.SHALLOW_COPY)
235        .setRow(put.getRow())
236        .setFamily(HConstants.CATALOG_FAMILY)
237        .setQualifier(getStateColumn(replicaId))
238        .setTimestamp(put.getTimestamp())
239        .setType(Cell.Type.Put)
240        .setValue(Bytes.toBytes(state.name()))
241        .build());
242    LOG.info(info.toString());
243    updateRegionLocation(regionInfo, state, put);
244  }
245
246  private void updateRegionLocation(RegionInfo regionInfo, State state, Put put)
247      throws IOException {
248    try (Table table = master.getConnection().getTable(TableName.META_TABLE_NAME)) {
249      table.put(put);
250    } catch (IOException e) {
251      // TODO: Revist!!!! Means that if a server is loaded, then we will abort our host!
252      // In tests we abort the Master!
253      String msg = String.format("FAILED persisting region=%s state=%s",
254        regionInfo.getShortNameToLog(), state);
255      LOG.error(msg, e);
256      master.abort(msg, e);
257      throw e;
258    }
259  }

261  private long getOpenSeqNumForParentRegion(RegionInfo region) throws IOException {
262    MasterFileSystem fs = master.getMasterFileSystem();
263    long maxSeqId = WALSplitUtil.getMaxRegionSequenceId(master.getConfiguration(), region,
264      fs::getFileSystem, fs::getWALFileSystem);
265    return maxSeqId > 0 ? maxSeqId + 1 : HConstants.NO_SEQNUM;
266  }

268  /**
269   * Performs an atomic multi-mutate operation against the given table. Used by the likes of merge
270   * and split as these want to make atomic mutations across multiple rows.
271   */
272  private void multiMutate(RegionInfo ri, List<Mutation> mutations) throws IOException {
273    debugLogMutations(mutations);
274    byte[] row =
275      Bytes.toBytes(RegionReplicaUtil.getRegionInfoForDefaultReplica(ri).getRegionNameAsString() +
276        HConstants.DELIMITER);
277    MutateRowsRequest.Builder builder = MutateRowsRequest.newBuilder();
278    for (Mutation mutation : mutations) {
279      if (mutation instanceof Put) {
280        builder.addMutationRequest(
281          ProtobufUtil.toMutation(ClientProtos.MutationProto.MutationType.PUT, mutation));
282      } else if (mutation instanceof Delete) {
283        builder.addMutationRequest(
284          ProtobufUtil.toMutation(ClientProtos.MutationProto.MutationType.DELETE, mutation));
285      } else {
286        throw new DoNotRetryIOException(
287          "multi in MetaEditor doesn't support " + mutation.getClass().getName());
288      }
289    }
290    MutateRowsRequest request = builder.build();
291    AsyncTable<?> table =
292      master.getConnection().toAsyncConnection().getTable(TableName.META_TABLE_NAME);
293    CompletableFuture<MutateRowsResponse> future =
294      table.<MultiRowMutationService, MutateRowsResponse> coprocessorService(
295        MultiRowMutationService::newStub,
296        (stub, controller, done) -> stub.mutateRows(controller, request, done), row);
297    FutureUtils.get(future);
298  }

300  private Result getRegionCatalogResult(RegionInfo region) throws IOException {
301    Get get =
302      new Get(CatalogFamilyFormat.getMetaKeyForRegion(region)).addFamily(HConstants.CATALOG_FAMILY);
303    try (Table table = master.getConnection().getTable(TableName.META_TABLE_NAME)) {
304      return table.get(get);
305    }

309  private static Put addSequenceNum(Put p, long openSeqNum, int replicaId) throws IOException {
310    return p.add(CellBuilderFactory.create(CellBuilderType.SHALLOW_COPY).setRow(p.getRow())
311      .setFamily(HConstants.CATALOG_FAMILY)

319  /**
320   * Splits the region into two in an atomic operation. Offlines the parent region with the
321   * information that it is split into two, and also adds the daughter regions. Does not add the
322   * location information to the daughter regions since they are not open yet.
323   */
324  public void splitRegion(RegionInfo parent, RegionInfo splitA, RegionInfo splitB,
325    ServerName serverName, TableDescriptor htd) throws IOException {
326    long parentOpenSeqNum = HConstants.NO_SEQNUM;
327    if (htd.hasGlobalReplicationScope()) {
328      parentOpenSeqNum = getOpenSeqNumForParentRegion(parent);
329    }
330    long time = EnvironmentEdgeManager.currentTime();
331    // Put for parent
332    Put putParent = MetaTableAccessor.makePutFromRegionInfo(
333      RegionInfoBuilder.newBuilder(parent).setOffline(true).setSplit(true).build(), time);
334    MetaTableAccessor.addDaughtersToPut(putParent, splitA, splitB);
335
336    // Puts for daughters
337    Put putA = MetaTableAccessor.makePutFromRegionInfo(splitA, time);
338    Put putB = MetaTableAccessor.makePutFromRegionInfo(splitB, time);
339    if (parentOpenSeqNum > 0) {
340      ReplicationBarrierFamilyFormat.addReplicationBarrier(putParent, parentOpenSeqNum);
341      ReplicationBarrierFamilyFormat.addReplicationParent(putA, Collections.singletonList(parent));
342      ReplicationBarrierFamilyFormat.addReplicationParent(putB, Collections.singletonList(parent));
343    }
344    // Set initial state to CLOSED
345    // NOTE: If initial state is not set to CLOSED then daughter regions get added with the
346    // default OFFLINE state. If Master gets restarted after this step, start up sequence of
347    // master tries to assign these offline regions. This is followed by re-assignments of the
348    // daughter regions from resumed {@link SplitTableRegionProcedure}
349    MetaTableAccessor.addRegionStateToPut(putA, RegionState.State.CLOSED);
350    MetaTableAccessor.addRegionStateToPut(putB, RegionState.State.CLOSED);
351
352    // new regions, openSeqNum = 1 is fine.
353    addSequenceNum(putA, 1, splitA.getReplicaId());
354    addSequenceNum(putB, 1, splitB.getReplicaId());
355
356    // Add empty locations for region replicas of daughters so that number of replicas can be
357    // cached whenever the primary region is looked up from meta
358    int regionReplication = getRegionReplication(htd);
359    for (int i = 1; i < regionReplication; i++) {
360      MetaTableAccessor.addEmptyLocation(putA, i);
361      MetaTableAccessor.addEmptyLocation(putB, i);
362    }
363
364    multiMutate(parent, Arrays.asList(putParent, putA, putB));
365  }

370  public void mergeRegions(RegionInfo child, RegionInfo[] parents, ServerName serverName,
371    TableDescriptor htd) throws IOException {
372    boolean globalScope = htd.hasGlobalReplicationScope();
373    long time = HConstants.LATEST_TIMESTAMP;
374    List<Mutation> mutations = new ArrayList<>();
375    List<RegionInfo> replicationParents = new ArrayList<>();
376    for (RegionInfo ri: parents) {
377      long seqNum = globalScope ? getOpenSeqNumForParentRegion(ri) : -1;
378      // Deletes for merging regions
379      mutations.add(MetaTableAccessor.makeDeleteFromRegionInfo(ri, time));
380      if (seqNum > 0) {
381        mutations
382          .add(ReplicationBarrierFamilyFormat.makePutForReplicationBarrier(ri, seqNum, time));
383        replicationParents.add(ri);
384      }
385    }
386    // Put for parent
387    Put putOfMerged = MetaTableAccessor.makePutFromRegionInfo(child, time);
388    putOfMerged = addMergeRegions(putOfMerged, Arrays.asList(parents));
389    // Set initial state to CLOSED.
390    // NOTE: If initial state is not set to CLOSED then merged region gets added with the
391    // default OFFLINE state. If Master gets restarted after this step, start up sequence of
392    // master tries to assign this offline region. This is followed by re-assignments of the
393    // merged region from resumed {@link MergeTableRegionsProcedure}
394    MetaTableAccessor.addRegionStateToPut(putOfMerged, RegionState.State.CLOSED);
395    mutations.add(putOfMerged);
396    // The merged is a new region, openSeqNum = 1 is fine. ServerName may be null
397    // if crash after merge happened but before we got to here.. means in-memory
398    // locations of offlined merged, now-closed, regions is lost. Should be ok. We
399    // assign the merged region later.
400    if (serverName != null) {
401      MetaTableAccessor.addLocation(putOfMerged, serverName, 1, child.getReplicaId());
402    }
403
404    // Add empty locations for region replicas of the merged region so that number of replicas
405    // can be cached whenever the primary region is looked up from meta
406    int regionReplication = getRegionReplication(htd);
407    for (int i = 1; i < regionReplication; i++) {
408      MetaTableAccessor.addEmptyLocation(putOfMerged, i);
409    }
410    // add parent reference for serial replication
411    if (!replicationParents.isEmpty()) {
412      ReplicationBarrierFamilyFormat.addReplicationParent(putOfMerged, replicationParents);
413    }
414    multiMutate(child, mutations);
415  }

431  /**
432   * Deletes merge qualifiers for the specified merge region.
433   * @param connection connection we're using
434   * @param mergeRegion the merged region
435   */
436  public void deleteMergeQualifiers(RegionInfo mergeRegion)
437    throws IOException {
438    // NOTE: We are doing a new hbase:meta read here.
439    Cell[] cells = getRegionCatalogResult(mergeRegion).rawCells();
440    if (cells == null || cells.length == 0) {
441      return;
442    }
443    Delete delete = new Delete(mergeRegion.getRegionName());
444    List<byte[]> qualifiers = new ArrayList<>();
445    for (Cell cell : cells) {
446      if (!CatalogFamilyFormat.isMergeQualifierPrefix(cell)) {
447        continue;
448      }
449      byte[] qualifier = CellUtil.cloneQualifier(cell);
450      qualifiers.add(qualifier);
451      delete.addColumns(HConstants.CATALOG_FAMILY, qualifier, HConstants.LATEST_TIMESTAMP);
452    }
453
454    // There will be race condition that a GCMultipleMergedRegionsProcedure is scheduled while
455    // the previous GCMultipleMergedRegionsProcedure is still going on, in this case, the second
456    // GCMultipleMergedRegionsProcedure could delete the merged region by accident!
457    if (qualifiers.isEmpty()) {
458      LOG.info("No merged qualifiers for region " + mergeRegion.getRegionNameAsString() +
459        " in meta table, they are cleaned up already, Skip.");
460      return;
461    }
462    try (Table table = master.getConnection().getTable(TableName.META_TABLE_NAME)) {
463      table.delete(delete);
464    }
465    LOG.info("Deleted merge references in " + mergeRegion.getRegionNameAsString() +
466      ", deleted qualifiers " +
467      qualifiers.stream().map(Bytes::toStringBinary).collect(Collectors.joining(", ")));
468  }

470  @VisibleForTesting
471  static Put addMergeRegions(Put put, Collection<RegionInfo> mergeRegions) throws IOException {
472    int limit = 10000; // Arbitrary limit. No room in our formatted 'task0000' below for more.
473    int max = mergeRegions.size();
474    if (max > limit) {
475      // Should never happen!!!!! But just in case.
476      throw new RuntimeException(
477        "Can't merge " + max + " regions in one go; " + limit + " is upper-limit.");
478    }
479    int counter = 0;
480    for (RegionInfo ri : mergeRegions) {
481      String qualifier = String.format(HConstants.MERGE_QUALIFIER_PREFIX_STR + "%04d", counter++);
482      put.add(CellBuilderFactory.create(CellBuilderType.SHALLOW_COPY).setRow(put.getRow())
483        .setFamily(HConstants.CATALOG_FAMILY).setQualifier(Bytes.toBytes(qualifier))
484        .setTimestamp(put.getTimestamp()).setType(Type.Put).setValue(RegionInfo.toByteArray(ri))
485        .build());
486    }
487    return put;
488  }

524  /**
525   * Pull the region state from a catalog table {@link Result}.
526   * @return the region state, or null if unknown.
527   */
528  public static State getRegionState(final Result r, RegionInfo regionInfo) {
529    Cell cell = r.getColumnLatestCell(HConstants.CATALOG_FAMILY,
530        getStateColumn(regionInfo.getReplicaId()));
531    if (cell == null || cell.getValueLength() == 0) {
532      return null;
533    }
534
535    String state = Bytes.toString(cell.getValueArray(), cell.getValueOffset(),
536        cell.getValueLength());
537    try {
538      return State.valueOf(state);
539    } catch (IllegalArgumentException e) {
540      LOG.warn("BAD value {} in hbase:meta info:state column for region {} , " +
541              "Consider using HBCK2 setRegionState ENCODED_REGION_NAME STATE",
542          state, regionInfo.getEncodedName());
543      return null;
544    }
545  }

547  private static byte[] getStateColumn(int replicaId) {
548    return replicaId == 0
549        ? HConstants.STATE_QUALIFIER
550        : Bytes.toBytes(HConstants.STATE_QUALIFIER_STR + META_REPLICA_ID_DELIMITER
551          + String.format(RegionInfo.REPLICA_ID_FORMAT, replicaId));
552  }

554  private static void debugLogMutations(List<? extends Mutation> mutations) throws IOException {
555    if (!METALOG.isDebugEnabled()) {
556      return;
557    }
558    // Logging each mutation in separate line makes it easier to see diff between them visually
559    // because of common starting indentation.
560    for (Mutation mutation : mutations) {
561      debugLogMutation(mutation);
562    }
563  }
564
565  private static void debugLogMutation(Mutation p) throws IOException {
566    METALOG.debug("{} {}", p.getClass().getSimpleName(), p.toJSON());
567  }
