053/**
054 * This will find where data for a region is located in HDFS. It ranks {@link ServerName}'s by the
055 * size of the store files they are holding for a given region.
056 */
057@InterfaceAudience.Private
058class RegionHDFSBlockLocationFinder extends Configured {
059  private static final Logger LOG = LoggerFactory.getLogger(RegionHDFSBlockLocationFinder.class);
060  private static final long CACHE_TIME = 240 * 60 * 1000;
061  private static final HDFSBlocksDistribution EMPTY_BLOCK_DISTRIBUTION =
062    new HDFSBlocksDistribution();
063  private volatile ClusterMetrics status;
064  private volatile ClusterInfoProvider provider;
065  private final ListeningExecutorService executor;
066  // Do not scheduleFullRefresh at master startup
067  private long lastFullRefresh = EnvironmentEdgeManager.currentTime();
068
069  private CacheLoader<RegionInfo, HDFSBlocksDistribution> loader =
070    new CacheLoader<RegionInfo, HDFSBlocksDistribution>() {
071
072      @Override
073      public ListenableFuture<HDFSBlocksDistribution> reload(final RegionInfo hri,
074        HDFSBlocksDistribution oldValue) throws Exception {
075        return executor.submit(new Callable<HDFSBlocksDistribution>() {
076          @Override
077          public HDFSBlocksDistribution call() throws Exception {
078            return internalGetTopBlockLocation(hri);
079          }
080        });
081      }
082
083      @Override
084      public HDFSBlocksDistribution load(RegionInfo key) throws Exception {
085        return internalGetTopBlockLocation(key);
086      }
087    };
088
089  // The cache for where regions are located.
090  private LoadingCache<RegionInfo, HDFSBlocksDistribution> cache = null;
091
092  RegionHDFSBlockLocationFinder() {
093    this.cache = createCache();
094    executor = MoreExecutors.listeningDecorator(Executors.newScheduledThreadPool(5,
095      new ThreadFactoryBuilder().setDaemon(true).setNameFormat("region-location-%d").build()));
096  }
097
098  /**
099   * Create a cache for region to list of servers
100   * @return A new Cache.
101   */
102  private LoadingCache<RegionInfo, HDFSBlocksDistribution> createCache() {
103    return CacheBuilder.newBuilder().expireAfterWrite(CACHE_TIME, TimeUnit.MILLISECONDS)
104      .build(loader);
105  }
106
107  void setClusterInfoProvider(ClusterInfoProvider provider) {
108    this.provider = provider;
109  }
110
111  void setClusterMetrics(ClusterMetrics status) {
112    long currentTime = EnvironmentEdgeManager.currentTime();
113    this.status = status;
114    if (currentTime > lastFullRefresh + (CACHE_TIME / 2)) {
115      // Only count the refresh if it includes user tables ( eg more than meta and namespace ).
116      lastFullRefresh = scheduleFullRefresh() ? currentTime : lastFullRefresh;
117    }
118
119  }
120
121  /**
122   * Refresh all the region locations.
123   * @return true if user created regions got refreshed.
124   */
125  private boolean scheduleFullRefresh() {
126    ClusterInfoProvider service = this.provider;
127    // Protect from anything being null while starting up.
128    if (service == null) {
129      return false;
130    }
131
132    // TODO: Should this refresh all the regions or only the ones assigned?
133    boolean includesUserTables = false;
134    for (final RegionInfo hri : service.getAssignedRegions()) {
135      cache.refresh(hri);
136      includesUserTables |= !hri.getTable().isSystemTable();
137    }
138    return includesUserTables;
139  }
140
141  List<ServerName> getTopBlockLocations(RegionInfo region) {
142    List<String> topHosts = getBlockDistribution(region).getTopHosts();
143    return mapHostNameToServerName(topHosts);
144  }
145
146  /**
147   * Returns an ordered list of hosts that are hosting the blocks for this region. The weight of
148   * each host is the sum of the block lengths of all files on that host, so the first host in the
149   * list is the server which holds the most bytes of the given region's HFiles.
150   * @param region region
151   * @return ordered list of hosts holding blocks of the specified region
152   */
153  private HDFSBlocksDistribution internalGetTopBlockLocation(RegionInfo region) {
154    try {
155      TableDescriptor tableDescriptor = getDescriptor(region.getTable());
156      if (tableDescriptor != null) {
157        HDFSBlocksDistribution blocksDistribution =
158          provider.computeHDFSBlocksDistribution(getConf(), tableDescriptor, region);
159        return blocksDistribution;
160      }
161    } catch (IOException ioe) {
162      LOG.warn("IOException during HDFSBlocksDistribution computation. for " + "region = " +
163        region.getEncodedName(), ioe);
164    }
165
166    return EMPTY_BLOCK_DISTRIBUTION;
167  }
168
169  /**
170   * return TableDescriptor for a given tableName
171   * @param tableName the table name
172   */
173  private TableDescriptor getDescriptor(TableName tableName) throws IOException {
174    ClusterInfoProvider service = this.provider;
175    if (service == null) {
176      return null;
177    }
178    return service.getTableDescriptor(tableName);
179  }
180
181  /**
182   * Map hostname to ServerName, The output ServerName list will have the same order as input hosts.
183   * @param hosts the list of hosts
184   * @return ServerName list
185   */
186  @RestrictedApi(explanation = "Should only be called in tests", link = "",
187    allowedOnPath = ".*/src/test/.*|.*/RegionHDFSBlockLocationFinder.java")
188  List<ServerName> mapHostNameToServerName(List<String> hosts) {
189    if (hosts == null || status == null) {
190      if (hosts == null) {
191        LOG.warn("RegionLocationFinder top hosts is null");
192      }
193      return Collections.emptyList();
194    }
195
196    List<ServerName> topServerNames = new ArrayList<>();
197    Collection<ServerName> regionServers = status.getLiveServerMetrics().keySet();
198
199    // create a mapping from hostname to ServerName for fast lookup
200    Map<String, List<ServerName>> hostToServerName = new HashMap<>();
201    for (ServerName sn : regionServers) {
202      String host = sn.getHostname();
203      if (!hostToServerName.containsKey(host)) {
204        hostToServerName.put(host, new ArrayList<>());
205      }
206      hostToServerName.get(host).add(sn);
207    }
208
209    for (String host : hosts) {
210      if (!hostToServerName.containsKey(host)) {
211        continue;
212      }
213      for (ServerName sn : hostToServerName.get(host)) {
214        // it is possible that HDFS is up ( thus host is valid ),
215        // but RS is down ( thus sn is null )
216        if (sn != null) {
217          topServerNames.add(sn);
218        }
219      }
220    }
221    return topServerNames;
222  }
223
224  HDFSBlocksDistribution getBlockDistribution(RegionInfo hri) {
225    HDFSBlocksDistribution blockDistbn = null;
226    try {
227      if (cache.asMap().containsKey(hri)) {
228        blockDistbn = cache.get(hri);
229        return blockDistbn;
230      } else {
231        LOG.trace("HDFSBlocksDistribution not found in cache for {}", hri.getRegionNameAsString());
232        blockDistbn = internalGetTopBlockLocation(hri);
233        cache.put(hri, blockDistbn);
234        return blockDistbn;
235      }
236    } catch (ExecutionException e) {
237      LOG.warn("Error while fetching cache entry ", e);
238      blockDistbn = internalGetTopBlockLocation(hri);
239      cache.put(hri, blockDistbn);
240      return blockDistbn;
241    }
242  }
243
244  private ListenableFuture<HDFSBlocksDistribution> asyncGetBlockDistribution(RegionInfo hri) {
245    try {
246      return loader.reload(hri, EMPTY_BLOCK_DISTRIBUTION);
247    } catch (Exception e) {
248      return Futures.immediateFuture(EMPTY_BLOCK_DISTRIBUTION);
249    }
250  }
251
252  void refreshAndWait(Collection<RegionInfo> hris) {
253    ArrayList<ListenableFuture<HDFSBlocksDistribution>> regionLocationFutures =
254      new ArrayList<>(hris.size());
255    for (RegionInfo hregionInfo : hris) {
256      regionLocationFutures.add(asyncGetBlockDistribution(hregionInfo));
257    }
258    int index = 0;
259    for (RegionInfo hregionInfo : hris) {
260      ListenableFuture<HDFSBlocksDistribution> future = regionLocationFutures.get(index);
261      try {
262        cache.put(hregionInfo, future.get());
263      } catch (InterruptedException ite) {
264        Thread.currentThread().interrupt();
265      } catch (ExecutionException ee) {
266        LOG.debug("ExecutionException during HDFSBlocksDistribution computation. for region = " +
267          hregionInfo.getEncodedName(), ee);
268      }
269      index++;
270    }
271  }
272
273  @RestrictedApi(explanation = "Should only be called in tests", link = "",
274    allowedOnPath = ".*/src/test/.*")
275  LoadingCache<RegionInfo, HDFSBlocksDistribution> getCache() {
276    return cache;
277  }
