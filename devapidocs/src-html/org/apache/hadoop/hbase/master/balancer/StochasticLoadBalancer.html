052/**
053 * <p>This is a best effort load balancer. Given a Cost function F(C) =&gt; x It will
054 * randomly try and mutate the cluster to Cprime. If F(Cprime) &lt; F(C) then the
055 * new cluster state becomes the plan. It includes costs functions to compute the cost of:</p>
056 * <ul>
057 * <li>Region Load</li>
058 * <li>Table Load</li>
059 * <li>Data Locality</li>
060 * <li>Memstore Sizes</li>
061 * <li>Storefile Sizes</li>
062 * </ul>
063 *
064 *
065 * <p>Every cost function returns a number between 0 and 1 inclusive; where 0 is the lowest cost
066 * best solution, and 1 is the highest possible cost and the worst solution.  The computed costs are
067 * scaled by their respective multipliers:</p>
068 *
069 * <ul>
070 *   <li>hbase.master.balancer.stochastic.regionLoadCost</li>
071 *   <li>hbase.master.balancer.stochastic.moveCost</li>
072 *   <li>hbase.master.balancer.stochastic.tableLoadCost</li>
073 *   <li>hbase.master.balancer.stochastic.localityCost</li>
074 *   <li>hbase.master.balancer.stochastic.memstoreSizeCost</li>
075 *   <li>hbase.master.balancer.stochastic.storefileSizeCost</li>
076 * </ul>
077 *
078 * <p>You can also add custom Cost function by setting the the following configuration value:</p>
079 * <ul>
080 *     <li>hbase.master.balancer.stochastic.additionalCostFunctions</li>
081 * </ul>
082 *
083 * <p>All custom Cost Functions needs to extends {@link CostFunction}</p>
084 *
085 * <p>In addition to the above configurations, the balancer can be tuned by the following
086 * configuration values:</p>
087 * <ul>
088 *   <li>hbase.master.balancer.stochastic.maxMoveRegions which
089 *   controls what the max number of regions that can be moved in a single invocation of this
090 *   balancer.</li>
091 *   <li>hbase.master.balancer.stochastic.stepsPerRegion is the coefficient by which the number of
092 *   regions is multiplied to try and get the number of times the balancer will
093 *   mutate all servers.</li>
094 *   <li>hbase.master.balancer.stochastic.maxSteps which controls the maximum number of times that
095 *   the balancer will try and mutate all the servers. The balancer will use the minimum of this
096 *   value and the above computation.</li>
097 * </ul>
098 *
099 * <p>This balancer is best used with hbase.master.loadbalance.bytable set to false
100 * so that the balancer gets the full picture of all loads on the cluster.</p>
101 */

241  private static CostFunction createCostFunction(Class<? extends CostFunction> clazz,
242    Configuration conf) {
243    try {
244      Constructor<? extends CostFunction> ctor = clazz.getDeclaredConstructor(Configuration.class);
245      return ReflectionUtils.instantiate(clazz.getName(), ctor, conf);
246    } catch (NoSuchMethodException e) {
247      // will try construct with no parameter
248    }
249    return ReflectionUtils.newInstance(clazz);
250  }

252  private void loadCustomCostFunctions(Configuration conf) {
253    String[] functionsNames = conf.getStrings(COST_FUNCTIONS_COST_FUNCTIONS_KEY);
254
255    if (null == functionsNames) {
256      return;
257    }
258    for (String className : functionsNames) {
259      Class<? extends CostFunction> clazz;
260      try {
261        clazz = Class.forName(className).asSubclass(CostFunction.class);
262      } catch (ClassNotFoundException e) {
263        LOG.warn("Cannot load class '{}': {}", className, e.getMessage());
264        continue;
265      }
266      CostFunction func = createCostFunction(clazz, conf);
267      LOG.info("Successfully loaded custom CostFunction '{}'", func.getClass().getSimpleName());
268      costFunctions.add(func);
269    }
270  }

331  @Override
332  protected boolean needsBalance(TableName tableName, BalancerClusterState cluster) {
333    ClusterLoadState cs = new ClusterLoadState(cluster.clusterState);
334    if (cs.getNumServers() < MIN_SERVER_BALANCE) {
335      if (LOG.isDebugEnabled()) {
336        LOG.debug("Not running balancer because only " + cs.getNumServers()
337            + " active regionserver(s)");
338      }
339      if (this.isBalancerRejectionRecording) {
340        sendRejectionReasonToRingBuffer("The number of RegionServers " +
341          cs.getNumServers() + " < MIN_SERVER_BALANCE(" + MIN_SERVER_BALANCE + ")", null);
342      }
343      return false;
344    }
345    if (areSomeRegionReplicasColocated(cluster)) {
346      return true;
347    }
348
349    if (idleRegionServerExist(cluster)){
350      return true;
351    }
352
353    double total = 0.0;
354    float sumMultiplier = 0.0f;
355    for (CostFunction c : costFunctions) {
356      float multiplier = c.getMultiplier();
357      if (multiplier <= 0) {
358        LOG.trace("{} not needed because multiplier is <= 0", c.getClass().getSimpleName());
359        continue;
360      }
361      if (!c.isNeeded()) {
362        LOG.trace("{} not needed", c.getClass().getSimpleName());
363        continue;
364      }
365      sumMultiplier += multiplier;
366      total += c.cost() * multiplier;
367    }
368
369    boolean balanced = total <= 0 || sumMultiplier <= 0 ||
370        (sumMultiplier > 0 && (total / sumMultiplier) < minCostNeedBalance);
371    if(balanced && isBalancerRejectionRecording){
372      String reason = "";
373      if (total <= 0) {
374        reason = "(cost1*multiplier1)+(cost2*multiplier2)+...+(costn*multipliern) = " + total + " <= 0";
375      } else if (sumMultiplier <= 0) {
376        reason = "sumMultiplier = " + sumMultiplier + " <= 0";
377      } else if ((total / sumMultiplier) < minCostNeedBalance) {
378        reason =
379          "[(cost1*multiplier1)+(cost2*multiplier2)+...+(costn*multipliern)]/sumMultiplier = " + (total
380            / sumMultiplier) + " <= minCostNeedBalance(" + minCostNeedBalance + ")";
381      }
382      sendRejectionReasonToRingBuffer(reason, costFunctions);
383    }
384    if (LOG.isDebugEnabled()) {
385      LOG.debug("{} {}; total cost={}, sum multiplier={}; cost/multiplier to need a balance is {}",
386          balanced ? "Skipping load balancing because balanced" : "We need to load balance",
387          isByTable ? String.format("table (%s)", tableName) : "cluster",
388          total, sumMultiplier, minCostNeedBalance);
389      if (LOG.isTraceEnabled()) {
390        LOG.trace("Balance decision detailed function costs={}", functionCost());
391      }
392    }
393    return !balanced;
394  }

403  /**
404   * Given the cluster state this will try and approach an optimal balance. This
405   * should always approach the optimal state given enough steps.
406   */
407  @Override
408  public synchronized List<RegionPlan> balanceTable(TableName tableName, Map<ServerName,
409    List<RegionInfo>> loadOfOneTable) {
410    // On clusters with lots of HFileLinks or lots of reference files,
411    // instantiating the storefile infos can be quite expensive.
412    // Allow turning this feature off if the locality cost is not going to
413    // be used in any computations.
414    RegionHDFSBlockLocationFinder finder = null;
415    if ((this.localityCost != null && this.localityCost.getMultiplier() > 0)
416        || (this.rackLocalityCost != null && this.rackLocalityCost.getMultiplier() > 0)) {
417      finder = this.regionFinder;
418    }
419
420    //The clusterState that is given to this method contains the state
421    //of all the regions in the table(s) (that's true today)
422    // Keep track of servers to iterate through them.
423    BalancerClusterState cluster =
424      new BalancerClusterState(loadOfOneTable, loads, finder, rackManager);
425
426    long startTime = EnvironmentEdgeManager.currentTime();
427
428    initCosts(cluster);
429
430    if (!needsBalance(tableName, cluster)) {
431      return null;
432    }
433
434    double currentCost = computeCost(cluster, Double.MAX_VALUE);
435    curOverallCost = currentCost;
436    System.arraycopy(tempFunctionCosts, 0, curFunctionCosts, 0, curFunctionCosts.length);
437    double initCost = currentCost;
438    double newCost;
439
440    long computedMaxSteps;
441    if (runMaxSteps) {
442      computedMaxSteps = Math.max(this.maxSteps,
443          ((long)cluster.numRegions * (long)this.stepsPerRegion * (long)cluster.numServers));
444    } else {
445      long calculatedMaxSteps = (long)cluster.numRegions * (long)this.stepsPerRegion *
446          (long)cluster.numServers;
447      computedMaxSteps = Math.min(this.maxSteps, calculatedMaxSteps);
448      if (calculatedMaxSteps > maxSteps) {
449        LOG.warn("calculatedMaxSteps:{} for loadbalancer's stochastic walk is larger than "
450            + "maxSteps:{}. Hence load balancing may not work well. Setting parameter "
451            + "\"hbase.master.balancer.stochastic.runMaxSteps\" to true can overcome this issue."
452            + "(This config change does not require service restart)", calculatedMaxSteps,
453            maxSteps);
454      }
455    }
456    LOG.info("start StochasticLoadBalancer.balancer, initCost=" + currentCost + ", functionCost="
457        + functionCost() + " computedMaxSteps: " + computedMaxSteps);
458
459    final String initFunctionTotalCosts = totalCostsPerFunc();
460    // Perform a stochastic walk to see if we can get a good fit.
461    long step;
462
463    for (step = 0; step < computedMaxSteps; step++) {
464      BalanceAction action = nextAction(cluster);
465
466      if (action.getType() == BalanceAction.Type.NULL) {
467        continue;
468      }
469
470      cluster.doAction(action);
471      updateCostsWithAction(cluster, action);
472
473      newCost = computeCost(cluster, currentCost);
474
475      // Should this be kept?
476      if (newCost < currentCost) {
477        currentCost = newCost;
478
479        // save for JMX
480        curOverallCost = currentCost;
481        System.arraycopy(tempFunctionCosts, 0, curFunctionCosts, 0, curFunctionCosts.length);
482      } else {
483        // Put things back the way they were before.
484        // TODO: undo by remembering old values
485        BalanceAction undoAction = action.undoAction();
486        cluster.doAction(undoAction);
487        updateCostsWithAction(cluster, undoAction);
488      }
489
490      if (EnvironmentEdgeManager.currentTime() - startTime >
491          maxRunningTime) {
492        break;
493      }
494    }
495    long endTime = EnvironmentEdgeManager.currentTime();
496
497    metricsBalancer.balanceCluster(endTime - startTime);
498
499    // update costs metrics
500    updateStochasticCosts(tableName, curOverallCost, curFunctionCosts);
501    if (initCost > currentCost) {
502      List<RegionPlan> plans = createRegionPlans(cluster);
503      LOG.info("Finished computing new load balance plan. Computation took {}" +
504        " to try {} different iterations.  Found a solution that moves " +
505        "{} regions; Going from a computed cost of {}" +
506        " to a new cost of {}", java.time.Duration.ofMillis(endTime - startTime),
507        step, plans.size(), initCost, currentCost);
508      sendRegionPlansToRingBuffer(plans, currentCost, initCost, initFunctionTotalCosts, step);
509      return plans;
510    }
511    LOG.info("Could not find a better load balance plan.  Tried {} different configurations in " +
512      "{}, and did not find anything with a computed cost less than {}", step,
513      java.time.Duration.ofMillis(endTime - startTime), initCost);
514    return null;
515  }

517  private void sendRejectionReasonToRingBuffer(String reason, List<CostFunction> costFunctions){
518    if (this.isBalancerRejectionRecording){
519      BalancerRejection.Builder builder =
520        new BalancerRejection.Builder()
521        .setReason(reason);
522      if (costFunctions != null) {
523        for (CostFunction c : costFunctions) {
524          float multiplier = c.getMultiplier();
525          if (multiplier <= 0 || !c.isNeeded()) {
526            continue;
527          }
528          builder.addCostFuncInfo(c.getClass().getName(), c.cost(), c.getMultiplier());
529        }
530      }
531      namedQueueRecorder.addRecord(new BalancerRejectionDetails(builder.build()));
532    }
533  }
534
535  private void sendRegionPlansToRingBuffer(List<RegionPlan> plans, double currentCost,
536      double initCost, String initFunctionTotalCosts, long step) {
537    if (this.isBalancerDecisionRecording) {
538      List<String> regionPlans = new ArrayList<>();
539      for (RegionPlan plan : plans) {
540        regionPlans.add(
541          "table: " + plan.getRegionInfo().getTable() + " , region: " + plan.getRegionName()
542            + " , source: " + plan.getSource() + " , destination: " + plan.getDestination());
543      }
544      BalancerDecision balancerDecision =
545        new BalancerDecision.Builder()
546          .setInitTotalCost(initCost)
547          .setInitialFunctionCosts(initFunctionTotalCosts)
548          .setComputedTotalCost(currentCost)
549          .setFinalFunctionCosts(totalCostsPerFunc())
550          .setComputedSteps(step)
551          .setRegionPlans(regionPlans).build();
552      namedQueueRecorder.addRecord(new BalancerDecisionDetails(balancerDecision));
553    }
554  }

556  /**
557   * update costs to JMX
558   */
559  private void updateStochasticCosts(TableName tableName, double overall, double[] subCosts) {
560    if (tableName == null) {
561      return;
562    }
563
564    // check if the metricsBalancer is MetricsStochasticBalancer before casting
565    if (metricsBalancer instanceof MetricsStochasticBalancer) {
566      MetricsStochasticBalancer balancer = (MetricsStochasticBalancer) metricsBalancer;
567      // overall cost
568      balancer.updateStochasticCost(tableName.getNameAsString(),
569        "Overall", "Overall cost", overall);
570
571      // each cost function
572      for (int i = 0; i < costFunctions.size(); i++) {
573        CostFunction costFunction = costFunctions.get(i);
574        String costFunctionName = costFunction.getClass().getSimpleName();
575        double costPercent = (overall == 0) ? 0 : (subCosts[i] / overall);
576        // TODO: cost function may need a specific description
577        balancer.updateStochasticCost(tableName.getNameAsString(), costFunctionName,
578          "The percent of " + costFunctionName, costPercent);
579      }
580    }
581  }

589  private String functionCost() {
590    StringBuilder builder = new StringBuilder();
591    for (CostFunction c:costFunctions) {
592      builder.append(c.getClass().getSimpleName());
593      builder.append(" : (");
594      builder.append(c.getMultiplier());
595      builder.append(", ");
596      builder.append(c.cost());
597      builder.append("); ");
598    }
599    return builder.toString();
600  }

602  private String totalCostsPerFunc() {
603    StringBuilder builder = new StringBuilder();
604    for (CostFunction c : costFunctions) {
605      if (c.getMultiplier() * c.cost() > 0.0) {
606        builder.append(" ");
607        builder.append(c.getClass().getSimpleName());
608        builder.append(" : ");
609        builder.append(c.getMultiplier() * c.cost());
610        builder.append(";");
611      }
612    }
613    if (builder.length() > 0) {
614      builder.deleteCharAt(builder.length() - 1);
615    }
616    return builder.toString();
617  }

619  /**
620   * Create all of the RegionPlan's needed to move from the initial cluster state to the desired
621   * state.
622   *
623   * @param cluster The state of the cluster
624   * @return List of RegionPlan's that represent the moves needed to get to desired final state.
625   */
626  private List<RegionPlan> createRegionPlans(BalancerClusterState cluster) {
627    List<RegionPlan> plans = new ArrayList<>();
628    for (int regionIndex = 0;
629         regionIndex < cluster.regionIndexToServerIndex.length; regionIndex++) {
630      int initialServerIndex = cluster.initialRegionIndexToServerIndex[regionIndex];
631      int newServerIndex = cluster.regionIndexToServerIndex[regionIndex];
632
633      if (initialServerIndex != newServerIndex) {
634        RegionInfo region = cluster.regions[regionIndex];
635        ServerName initialServer = cluster.servers[initialServerIndex];
636        ServerName newServer = cluster.servers[newServerIndex];
637
638        if (LOG.isTraceEnabled()) {
639          LOG.trace("Moving Region " + region.getEncodedName() + " from server "
640              + initialServer.getHostname() + " to " + newServer.getHostname());
641        }
642        RegionPlan rp = new RegionPlan(region, initialServer, newServer);
643        plans.add(rp);
644      }
645    }
646    return plans;
647  }

649  /**
650   * Store the current region loads.
651   */
652  private synchronized void updateRegionLoad() {
653    // We create a new hashmap so that regions that are no longer there are removed.
654    // However we temporarily need the old loads so we can use them to keep the rolling average.
655    Map<String, Deque<BalancerRegionLoad>> oldLoads = loads;
656    loads = new HashMap<>();
657
658    clusterStatus.getLiveServerMetrics().forEach((ServerName sn, ServerMetrics sm) -> {
659      sm.getRegionMetrics().forEach((byte[] regionName, RegionMetrics rm) -> {
660        String regionNameAsString = RegionInfo.getRegionNameAsString(regionName);
661        Deque<BalancerRegionLoad> rLoads = oldLoads.get(regionNameAsString);
662        if (rLoads == null) {
663          rLoads = new ArrayDeque<>(numRegionLoadsToRemember + 1);
664        } else if (rLoads.size() >= numRegionLoadsToRemember) {
665          rLoads.remove();
666        }
667        rLoads.add(new BalancerRegionLoad(rm));
668        loads.put(regionNameAsString, rLoads);
669      });
670    });
671
672    for(CostFromRegionLoadFunction cost : regionLoadFunctions) {
673      cost.setLoads(loads);
674    }
675  }

711  /**
712   * This is the main cost function.  It will compute a cost associated with a proposed cluster
713   * state.  All different costs will be combined with their multipliers to produce a double cost.
714   *
715   * @param cluster The state of the cluster
716   * @param previousCost the previous cost. This is used as an early out.
717   * @return a double of a cost associated with the proposed cluster state.  This cost is an
718   *         aggregate of all individual cost functions.
719   */
720  @RestrictedApi(explanation = "Should only be called in tests", link = "",
721    allowedOnPath = ".*(/src/test/.*|StochasticLoadBalancer).java")
722  double computeCost(BalancerClusterState cluster, double previousCost) {
723    double total = 0;
724
725    for (int i = 0; i < costFunctions.size(); i++) {
726      CostFunction c = costFunctions.get(i);
727      this.tempFunctionCosts[i] = 0.0;
728
729      if (c.getMultiplier() <= 0) {
730        continue;
731      }
732
733      Float multiplier = c.getMultiplier();
734      double cost = c.cost();
735
736      this.tempFunctionCosts[i] = multiplier*cost;
737      total += this.tempFunctionCosts[i];
738
739      if (total > previousCost) {
740        break;
741      }
742    }
743
744    return total;
745  }
