050/**
051 * <p>This is a best effort load balancer. Given a Cost function F(C) =&gt; x It will
052 * randomly try and mutate the cluster to Cprime. If F(Cprime) &lt; F(C) then the
053 * new cluster state becomes the plan. It includes costs functions to compute the cost of:</p>
054 * <ul>
055 * <li>Region Load</li>
056 * <li>Table Load</li>
057 * <li>Data Locality</li>
058 * <li>Memstore Sizes</li>
059 * <li>Storefile Sizes</li>
060 * </ul>
061 *
062 *
063 * <p>Every cost function returns a number between 0 and 1 inclusive; where 0 is the lowest cost
064 * best solution, and 1 is the highest possible cost and the worst solution.  The computed costs are
065 * scaled by their respective multipliers:</p>
066 *
067 * <ul>
068 *   <li>hbase.master.balancer.stochastic.regionLoadCost</li>
069 *   <li>hbase.master.balancer.stochastic.moveCost</li>
070 *   <li>hbase.master.balancer.stochastic.tableLoadCost</li>
071 *   <li>hbase.master.balancer.stochastic.localityCost</li>
072 *   <li>hbase.master.balancer.stochastic.memstoreSizeCost</li>
073 *   <li>hbase.master.balancer.stochastic.storefileSizeCost</li>
074 * </ul>
075 *
076 * <p>You can also add custom Cost function by setting the the following configuration value:</p>
077 * <ul>
078 *     <li>hbase.master.balancer.stochastic.additionalCostFunctions</li>
079 * </ul>
080 *
081 * <p>All custom Cost Functions needs to extends {@link CostFunction}</p>
082 *
083 * <p>In addition to the above configurations, the balancer can be tuned by the following
084 * configuration values:</p>
085 * <ul>
086 *   <li>hbase.master.balancer.stochastic.maxMoveRegions which
087 *   controls what the max number of regions that can be moved in a single invocation of this
088 *   balancer.</li>
089 *   <li>hbase.master.balancer.stochastic.stepsPerRegion is the coefficient by which the number of
090 *   regions is multiplied to try and get the number of times the balancer will
091 *   mutate all servers.</li>
092 *   <li>hbase.master.balancer.stochastic.maxSteps which controls the maximum number of times that
093 *   the balancer will try and mutate all the servers. The balancer will use the minimum of this
094 *   value and the above computation.</li>
095 * </ul>
096 *
097 * <p>This balancer is best used with hbase.master.loadbalance.bytable set to false
098 * so that the balancer gets the full picture of all loads on the cluster.</p>
099 */

163  private static CostFunction createCostFunction(Class<? extends CostFunction> clazz,
164    Configuration conf) {
165    try {
166      Constructor<? extends CostFunction> ctor = clazz.getDeclaredConstructor(Configuration.class);
167      return ReflectionUtils.instantiate(clazz.getName(), ctor, conf);
168    } catch (NoSuchMethodException e) {
169      // will try construct with no parameter
170    }
171    return ReflectionUtils.newInstance(clazz);
172  }

174  private void loadCustomCostFunctions(Configuration conf) {
175    String[] functionsNames = conf.getStrings(COST_FUNCTIONS_COST_FUNCTIONS_KEY);
176
177    if (null == functionsNames) {
178      return;
179    }
180    for (String className : functionsNames) {
181      Class<? extends CostFunction> clazz;
182      try {
183        clazz = Class.forName(className).asSubclass(CostFunction.class);
184      } catch (ClassNotFoundException e) {
185        LOG.warn("Cannot load class '{}': {}", className, e.getMessage());
186        continue;
187      }
188      CostFunction func = createCostFunction(clazz, conf);
189      LOG.info("Successfully loaded custom CostFunction '{}'", func.getClass().getSimpleName());
190      costFunctions.add(func);
191    }
192  }

216    super.loadConf(conf);

218    stepsPerRegion = conf.getInt(STEPS_PER_REGION_KEY, stepsPerRegion);
219    maxRunningTime = conf.getLong(MAX_RUNNING_TIME_KEY, maxRunningTime);

222    numRegionLoadsToRemember = conf.getInt(KEEP_REGION_LOADS, numRegionLoadsToRemember);
223    minCostNeedBalance = conf.getFloat(MIN_COST_NEED_BALANCE_KEY, minCostNeedBalance);
224    localityCandidateGenerator = new LocalityBasedCandidateGenerator();
225    localityCost = new ServerLocalityCostFunction(conf);
226    rackLocalityCost = new RackLocalityCostFunction(conf);
227
228    this.candidateGenerators = createCandidateGenerators();
229
230    regionReplicaHostCostFunction = new RegionReplicaHostCostFunction(conf);
231    regionReplicaRackCostFunction = new RegionReplicaRackCostFunction(conf);
232    costFunctions = new ArrayList<>();
233    addCostFunction(new RegionCountSkewCostFunction(conf));
234    addCostFunction(new PrimaryRegionCountSkewCostFunction(conf));
235    addCostFunction(new MoveCostFunction(conf, provider));
236    addCostFunction(localityCost);
237    addCostFunction(rackLocalityCost);
238    addCostFunction(new TableSkewCostFunction(conf));
239    addCostFunction(regionReplicaHostCostFunction);
240    addCostFunction(regionReplicaRackCostFunction);
241    addCostFunction(new ReadRequestCostFunction(conf));
242    addCostFunction(new CPRequestCostFunction(conf));
243    addCostFunction(new WriteRequestCostFunction(conf));
244    addCostFunction(new MemStoreSizeCostFunction(conf));
245    addCostFunction(new StoreFileCostFunction(conf));
246    loadCustomCostFunctions(conf);
247
248    curFunctionCosts = new double[costFunctions.size()];
249    tempFunctionCosts = new double[costFunctions.size()];
250
251    LOG.info(
252      "Loaded config; maxSteps=" + maxSteps + ", runMaxSteps=" + runMaxSteps +
253        ", stepsPerRegion=" + stepsPerRegion +
254        ", maxRunningTime=" + maxRunningTime + ", isByTable=" + isByTable +
255        ", CostFunctions=" + Arrays.toString(getCostFunctionNames()) +
256        " , sum of multiplier of cost functions = " + sumMultiplier + " etc.");  }

317  private String getBalanceReason(double total, double sumMultiplier) {
318    if (total <= 0) {
319      return "(cost1*multiplier1)+(cost2*multiplier2)+...+(costn*multipliern) = " + total + " <= 0";
320    } else if (sumMultiplier <= 0) {
321      return "sumMultiplier = " + sumMultiplier + " <= 0";
322    } else if ((total / sumMultiplier) < minCostNeedBalance) {
323      return "[(cost1*multiplier1)+(cost2*multiplier2)+...+(costn*multipliern)]/sumMultiplier = " +
324        (total / sumMultiplier) + " <= minCostNeedBalance(" + minCostNeedBalance + ")";
325    } else {
326      return "";
327    }
328  }

330  @RestrictedApi(explanation = "Should only be called in tests", link = "",
331    allowedOnPath = ".*(/src/test/.*|StochasticLoadBalancer).java")
332  boolean needsBalance(TableName tableName, BalancerClusterState cluster) {
333    ClusterLoadState cs = new ClusterLoadState(cluster.clusterState);
334    if (cs.getNumServers() < MIN_SERVER_BALANCE) {
335      LOG.info("Not running balancer because only " + cs.getNumServers()
336            + " active regionserver(s)");
337      sendRejectionReasonToRingBuffer(() -> "The number of RegionServers " + cs.getNumServers() +
338        " < MIN_SERVER_BALANCE(" + MIN_SERVER_BALANCE + ")", null);
339      return false;
340    }
341    if (areSomeRegionReplicasColocated(cluster)) {
342      LOG.info("Running balancer because at least one server hosts replicas of the same region." +
343        " function cost={}", functionCost());
344      return true;
345    }
346
347    if (idleRegionServerExist(cluster)){
348      LOG.info("Running balancer because cluster has idle server(s)."+
349        " function cost={}", functionCost());
350      return true;
351    }
352
353    double total = 0.0;
354    for (CostFunction c : costFunctions) {
355      if (!c.isNeeded()) {
356        LOG.trace("{} not needed", c.getClass().getSimpleName());
357        continue;
358      }
359      total += c.cost() * c.getMultiplier();
360    }
361    boolean balanced = (total / sumMultiplier < minCostNeedBalance);
362
363    if (balanced) {
364      final double calculatedTotal = total;
365      sendRejectionReasonToRingBuffer(() ->
366        getBalanceReason(calculatedTotal, sumMultiplier), costFunctions);
367      LOG.info("{} - skipping load balancing because weighted average imbalance={} <= "
368          + "threshold({}). If you want more aggressive balancing, either lower "
369          + "hbase.master.balancer.stochastic.minCostNeedBalance from {} or increase the relative "
370          + "multiplier(s) of the specific cost function(s). functionCost={}",
371        isByTable ? "Table specific ("+tableName+")" : "Cluster wide", total / sumMultiplier,
372        minCostNeedBalance, minCostNeedBalance, functionCost());
373    } else {
374      LOG.info("{} - Calculating plan. may take up to {}ms to complete.",
375        isByTable ? "Table specific ("+tableName+")" : "Cluster wide", maxRunningTime);
376    }
377    return !balanced;
378  }

393  private long calculateMaxSteps(BalancerClusterState cluster) {
394    return (long) cluster.numRegions * (long) this.stepsPerRegion * (long) cluster.numServers;
395  }

397  /**
398   * Given the cluster state this will try and approach an optimal balance. This
399   * should always approach the optimal state given enough steps.
400   */
401  @Override
402  protected List<RegionPlan> balanceTable(TableName tableName, Map<ServerName,
403    List<RegionInfo>> loadOfOneTable) {
404    // On clusters with lots of HFileLinks or lots of reference files,
405    // instantiating the storefile infos can be quite expensive.
406    // Allow turning this feature off if the locality cost is not going to
407    // be used in any computations.
408    RegionHDFSBlockLocationFinder finder = null;
409    if ((this.localityCost != null)
410        || (this.rackLocalityCost != null)) {
411      finder = this.regionFinder;
412    }
413
414    //The clusterState that is given to this method contains the state
415    //of all the regions in the table(s) (that's true today)
416    // Keep track of servers to iterate through them.
417    BalancerClusterState cluster =
418      new BalancerClusterState(loadOfOneTable, loads, finder, rackManager);
419
420    long startTime = EnvironmentEdgeManager.currentTime();
421
422    initCosts(cluster);
423
424    sumMultiplier = 0;
425    for (CostFunction c : costFunctions) {
426      if(c.isNeeded()) {
427        sumMultiplier += c.getMultiplier();
428      }
429    }
430    if (sumMultiplier <= 0) {
431      LOG.error("At least one cost function needs a multiplier > 0. For example, set "
432        + "hbase.master.balancer.stochastic.regionCountCost to a positive value or default");
433      return null;
434    }
435
436    double currentCost = computeCost(cluster, Double.MAX_VALUE);
437    curOverallCost = currentCost;
438    System.arraycopy(tempFunctionCosts, 0, curFunctionCosts, 0, curFunctionCosts.length);
439    updateStochasticCosts(tableName, curOverallCost, curFunctionCosts);
440    double initCost = currentCost;
441    double newCost;
442
443    if (!needsBalance(tableName, cluster)) {
444      return null;
445    }
446
447    long computedMaxSteps;
448    if (runMaxSteps) {
449      computedMaxSteps = Math.max(this.maxSteps, calculateMaxSteps(cluster));
450    } else {
451      long calculatedMaxSteps = calculateMaxSteps(cluster);
452      computedMaxSteps = Math.min(this.maxSteps, calculatedMaxSteps);
453      if (calculatedMaxSteps > maxSteps) {
454        LOG.warn("calculatedMaxSteps:{} for loadbalancer's stochastic walk is larger than "
455            + "maxSteps:{}. Hence load balancing may not work well. Setting parameter "
456            + "\"hbase.master.balancer.stochastic.runMaxSteps\" to true can overcome this issue."
457            + "(This config change does not require service restart)", calculatedMaxSteps,
458            maxSteps);
459      }
460    }
461    LOG.info("Start StochasticLoadBalancer.balancer, initial weighted average imbalance={}, "
462      + "functionCost={} computedMaxSteps={}",
463      currentCost / sumMultiplier, functionCost(), computedMaxSteps);
464
465    final String initFunctionTotalCosts = totalCostsPerFunc();
466    // Perform a stochastic walk to see if we can get a good fit.
467    long step;
468
469    for (step = 0; step < computedMaxSteps; step++) {
470      BalanceAction action = nextAction(cluster);
471
472      if (action.getType() == BalanceAction.Type.NULL) {
473        continue;
474      }
475
476      cluster.doAction(action);
477      updateCostsWithAction(cluster, action);
478
479      newCost = computeCost(cluster, currentCost);
480
481      // Should this be kept?
482      if (newCost < currentCost) {
483        currentCost = newCost;
484
485        // save for JMX
486        curOverallCost = currentCost;
487        System.arraycopy(tempFunctionCosts, 0, curFunctionCosts, 0, curFunctionCosts.length);
488      } else {
489        // Put things back the way they were before.
490        // TODO: undo by remembering old values
491        BalanceAction undoAction = action.undoAction();
492        cluster.doAction(undoAction);
493        updateCostsWithAction(cluster, undoAction);
494      }
495
496      if (EnvironmentEdgeManager.currentTime() - startTime >
497          maxRunningTime) {
498        break;
499      }
500    }
501    long endTime = EnvironmentEdgeManager.currentTime();
502
503    metricsBalancer.balanceCluster(endTime - startTime);
504
505    if (initCost > currentCost) {
506      updateStochasticCosts(tableName, curOverallCost, curFunctionCosts);
507      List<RegionPlan> plans = createRegionPlans(cluster);
508      LOG.info("Finished computing new moving plan. Computation took {} ms" +
509          " to try {} different iterations.  Found a solution that moves " +
510          "{} regions; Going from a computed imbalance of {}" +
511          " to a new imbalance of {}. funtionCost={}",
512        endTime - startTime, step, plans.size(),
513        initCost / sumMultiplier, currentCost / sumMultiplier, functionCost());
514      sendRegionPlansToRingBuffer(plans, currentCost, initCost, initFunctionTotalCosts, step);
515      return plans;
516    }
517    LOG.info("Could not find a better moving plan.  Tried {} different configurations in " +
518        "{} ms, and did not find anything with an imbalance score less than {}", step,
519      endTime - startTime, initCost / sumMultiplier);
520    return null;
521  }

523  private void sendRejectionReasonToRingBuffer(Supplier<String> reason,
524    List<CostFunction> costFunctions) {
525    provider.recordBalancerRejection(() -> {
526      BalancerRejection.Builder builder = new BalancerRejection.Builder().setReason(reason.get());
527      if (costFunctions != null) {
528        for (CostFunction c : costFunctions) {
529          if (!c.isNeeded()) {
530            continue;
531          }
532          builder.addCostFuncInfo(c.getClass().getName(), c.cost(), c.getMultiplier());
533        }
534      }
535      return builder.build();
536    });
537  }

539  private void sendRegionPlansToRingBuffer(List<RegionPlan> plans, double currentCost,
540    double initCost, String initFunctionTotalCosts, long step) {
541    provider.recordBalancerDecision(() -> {
542      List<String> regionPlans = new ArrayList<>();
543      for (RegionPlan plan : plans) {
544        regionPlans
545          .add("table: " + plan.getRegionInfo().getTable() + " , region: " + plan.getRegionName() +
546            " , source: " + plan.getSource() + " , destination: " + plan.getDestination());
547      }
548      return new BalancerDecision.Builder().setInitTotalCost(initCost)
549        .setInitialFunctionCosts(initFunctionTotalCosts).setComputedTotalCost(currentCost)
550        .setFinalFunctionCosts(totalCostsPerFunc()).setComputedSteps(step)
551        .setRegionPlans(regionPlans).build();
552    });
553  }

555  /**
556   * update costs to JMX
557   */
558  private void updateStochasticCosts(TableName tableName, double overall, double[] subCosts) {
559    if (tableName == null) {
560      return;
561    }
562
563    // check if the metricsBalancer is MetricsStochasticBalancer before casting
564    if (metricsBalancer instanceof MetricsStochasticBalancer) {
565      MetricsStochasticBalancer balancer = (MetricsStochasticBalancer) metricsBalancer;
566      // overall cost
567      balancer.updateStochasticCost(tableName.getNameAsString(),
568        OVERALL_COST_FUNCTION_NAME, "Overall cost", overall);
569
570      // each cost function
571      for (int i = 0; i < costFunctions.size(); i++) {
572        CostFunction costFunction = costFunctions.get(i);
573        String costFunctionName = costFunction.getClass().getSimpleName();
574        double costPercent = (overall == 0) ? 0 : (subCosts[i] / overall);
575        // TODO: cost function may need a specific description
576        balancer.updateStochasticCost(tableName.getNameAsString(), costFunctionName,
577          "The percent of " + costFunctionName, costPercent);
578      }
579    }
580  }

582  private void addCostFunction(CostFunction costFunction) {
583    float multiplier = costFunction.getMultiplier();
584    if (multiplier > 0) {
585      costFunctions.add(costFunction);
586    }
587  }

589  private String functionCost() {
590    StringBuilder builder = new StringBuilder();
591    for (CostFunction c : costFunctions) {
592      builder.append(c.getClass().getSimpleName());
593      builder.append(" : (");
594      if (c.isNeeded()) {
595        builder.append("multiplier=" + c.getMultiplier());
596        builder.append(", ");
597        double cost = c.cost();
598        builder.append("imbalance=" + cost);
599        if (cost >= minCostNeedBalance) {
600          builder.append(", need balance");
601        }
602      } else {
603        builder.append("not needed");
604      }
605      builder.append("); ");
606    }
607    return builder.toString();
608  }

610  private String totalCostsPerFunc() {
611    StringBuilder builder = new StringBuilder();
612    for (CostFunction c : costFunctions) {
613      if (!c.isNeeded()) {
614        continue;
615      }
616      double cost = c.getMultiplier() * c.cost();
617      if (cost > 0.0) {
618        builder.append(" ");
619        builder.append(c.getClass().getSimpleName());
620        builder.append(" : ");
621        builder.append(cost);
622        builder.append(";");
623      }
624    }
625    if (builder.length() > 0) {
626      builder.deleteCharAt(builder.length() - 1);
627    }
628    return builder.toString();
629  }

631  /**
632   * Create all of the RegionPlan's needed to move from the initial cluster state to the desired
633   * state.
634   *
635   * @param cluster The state of the cluster
636   * @return List of RegionPlan's that represent the moves needed to get to desired final state.
637   */
638  private List<RegionPlan> createRegionPlans(BalancerClusterState cluster) {
639    List<RegionPlan> plans = new ArrayList<>();
640    for (int regionIndex = 0;
641         regionIndex < cluster.regionIndexToServerIndex.length; regionIndex++) {
642      int initialServerIndex = cluster.initialRegionIndexToServerIndex[regionIndex];
643      int newServerIndex = cluster.regionIndexToServerIndex[regionIndex];
644
645      if (initialServerIndex != newServerIndex) {
646        RegionInfo region = cluster.regions[regionIndex];
647        ServerName initialServer = cluster.servers[initialServerIndex];
648        ServerName newServer = cluster.servers[newServerIndex];
649
650        if (LOG.isTraceEnabled()) {
651          LOG.trace("Moving Region " + region.getEncodedName() + " from server "
652              + initialServer.getHostname() + " to " + newServer.getHostname());
653        }
654        RegionPlan rp = new RegionPlan(region, initialServer, newServer);
655        plans.add(rp);
656      }
657    }
658    return plans;
659  }

661  /**
662   * Store the current region loads.
663   */
664  private void updateRegionLoad() {
665    // We create a new hashmap so that regions that are no longer there are removed.
666    // However we temporarily need the old loads so we can use them to keep the rolling average.
667    Map<String, Deque<BalancerRegionLoad>> oldLoads = loads;
668    loads = new HashMap<>();
669
670    clusterStatus.getLiveServerMetrics().forEach((ServerName sn, ServerMetrics sm) -> {
671      sm.getRegionMetrics().forEach((byte[] regionName, RegionMetrics rm) -> {
672        String regionNameAsString = RegionInfo.getRegionNameAsString(regionName);
673        Deque<BalancerRegionLoad> rLoads = oldLoads.get(regionNameAsString);
674        if (rLoads == null) {
675          rLoads = new ArrayDeque<>(numRegionLoadsToRemember + 1);
676        } else if (rLoads.size() >= numRegionLoadsToRemember) {
677          rLoads.remove();
678        }
679        rLoads.add(new BalancerRegionLoad(rm));
680        loads.put(regionNameAsString, rLoads);
681      });
682    });
683  }

718  /**
719   * This is the main cost function.  It will compute a cost associated with a proposed cluster
720   * state.  All different costs will be combined with their multipliers to produce a double cost.
721   *
722   * @param cluster The state of the cluster
723   * @param previousCost the previous cost. This is used as an early out.
724   * @return a double of a cost associated with the proposed cluster state.  This cost is an
725   *         aggregate of all individual cost functions.
726   */
727  @RestrictedApi(explanation = "Should only be called in tests", link = "",
728    allowedOnPath = ".*(/src/test/.*|StochasticLoadBalancer).java")
729  double computeCost(BalancerClusterState cluster, double previousCost) {
730    double total = 0;
731
732    for (int i = 0; i < costFunctions.size(); i++) {
733      CostFunction c = costFunctions.get(i);
734      this.tempFunctionCosts[i] = 0.0;
735
736      if (!c.isNeeded()) {
737        continue;
738      }
739
740      Float multiplier = c.getMultiplier();
741      double cost = c.cost();
742
743      this.tempFunctionCosts[i] = multiplier * cost;
744      total += this.tempFunctionCosts[i];
745
746      if (total > previousCost) {
747        break;
748      }
749    }
750
751    return total;
752  }
