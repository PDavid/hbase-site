039/**
040 * An efficient array based implementation similar to ClusterState for keeping the status of the
041 * cluster in terms of region assignment and distribution. LoadBalancers, such as
042 * StochasticLoadBalancer uses this Cluster object because of hundreds of thousands of hashmap
043 * manipulations are very costly, which is why this class uses mostly indexes and arrays.
044 * <p/>
045 * BalancerClusterState tracks a list of unassigned regions, region assignments, and the server
046 * topology in terms of server names, hostnames and racks.
047 */
048@InterfaceAudience.Private
049class BalancerClusterState {
050
051  private static final Logger LOG = LoggerFactory.getLogger(BalancerClusterState.class);
052
053  ServerName[] servers;
054  // ServerName uniquely identifies a region server. multiple RS can run on the same host
055  String[] hosts;
056  String[] racks;
057  boolean multiServersPerHost = false; // whether or not any host has more than one server
058
059  ArrayList<String> tables;
060  RegionInfo[] regions;
061  Deque<BalancerRegionLoad>[] regionLoads;
062  private RegionHDFSBlockLocationFinder regionFinder;
063
064  int[][] regionLocations; // regionIndex -> list of serverIndex sorted by locality
065
066  int[] serverIndexToHostIndex; // serverIndex -> host index
067  int[] serverIndexToRackIndex; // serverIndex -> rack index
068
069  int[][] regionsPerServer; // serverIndex -> region list
070  int[] serverIndexToRegionsOffset; // serverIndex -> offset of region list
071  int[][] regionsPerHost; // hostIndex -> list of regions
072  int[][] regionsPerRack; // rackIndex -> region list
073  int[][] primariesOfRegionsPerServer; // serverIndex -> sorted list of regions by primary region
074                                       // index
075  int[][] primariesOfRegionsPerHost; // hostIndex -> sorted list of regions by primary region index
076  int[][] primariesOfRegionsPerRack; // rackIndex -> sorted list of regions by primary region index
077
078  int[][] serversPerHost; // hostIndex -> list of server indexes
079  int[][] serversPerRack; // rackIndex -> list of server indexes
080  int[] regionIndexToServerIndex; // regionIndex -> serverIndex
081  int[] initialRegionIndexToServerIndex; // regionIndex -> serverIndex (initial cluster state)
082  int[] regionIndexToTableIndex; // regionIndex -> tableIndex
083  int[][] numRegionsPerServerPerTable; // serverIndex -> tableIndex -> # regions
084  int[] numRegionsPerTable; // tableIndex -> region count
085  double[] meanRegionsPerTable; // mean region count per table
086  double[] regionSkewByTable;       // skew on RS per by table
087  double[] minRegionSkewByTable;       // min skew on RS per by table
088  double[] maxRegionSkewByTable;       // max skew on RS per by table
089  int[] regionIndexToPrimaryIndex; // regionIndex -> regionIndex of the primary
090  boolean hasRegionReplicas = false; // whether there is regions with replicas
091
092  Integer[] serverIndicesSortedByRegionCount;
093  Integer[] serverIndicesSortedByLocality;
094
095  Map<Address, Integer> serversToIndex;
096  Map<String, Integer> hostsToIndex;
097  Map<String, Integer> racksToIndex;
098  Map<String, Integer> tablesToIndex;
099  Map<RegionInfo, Integer> regionsToIndex;
100  float[] localityPerServer;
101
102  int numServers;
103  int numHosts;
104  int numRacks;
105  int numTables;
106  int numRegions;
107
108  int numMovedRegions = 0; // num moved regions from the initial configuration
109  Map<ServerName, List<RegionInfo>> clusterState;
110
111  private final RackManager rackManager;
112  // Maps region -> rackIndex -> locality of region on rack
113  private float[][] rackLocalities;
114  // Maps localityType -> region -> [server|rack]Index with highest locality
115  private int[][] regionsToMostLocalEntities;

125    Map<String, Deque<BalancerRegionLoad>> loads, RegionHDFSBlockLocationFinder regionFinder,
126    RackManager rackManager) {

130  @SuppressWarnings("unchecked")
131  BalancerClusterState(Collection<RegionInfo> unassignedRegions,
132    Map<ServerName, List<RegionInfo>> clusterState, Map<String, Deque<BalancerRegionLoad>> loads,
133    RegionHDFSBlockLocationFinder regionFinder, RackManager rackManager) {
134    if (unassignedRegions == null) {
135      unassignedRegions = Collections.emptyList();
136    }
137
138    serversToIndex = new HashMap<>();
139    hostsToIndex = new HashMap<>();
140    racksToIndex = new HashMap<>();
141    tablesToIndex = new HashMap<>();
142
143    // TODO: We should get the list of tables from master
144    tables = new ArrayList<>();
145    this.rackManager = rackManager != null ? rackManager : new DefaultRackManager();
146
147    numRegions = 0;
148
149    List<List<Integer>> serversPerHostList = new ArrayList<>();
150    List<List<Integer>> serversPerRackList = new ArrayList<>();
151    this.clusterState = clusterState;
152    this.regionFinder = regionFinder;
153
154    // Use servername and port as there can be dead servers in this list. We want everything with
155    // a matching hostname and port to have the same index.
156    for (ServerName sn : clusterState.keySet()) {
157      if (sn == null) {
158        LOG.warn("TODO: Enable TRACE on BaseLoadBalancer. Empty servername); " +
159          "skipping; unassigned regions?");
160        if (LOG.isTraceEnabled()) {
161          LOG.trace("EMPTY SERVERNAME " + clusterState.toString());
162        }
163        continue;
164      }
165      if (serversToIndex.get(sn.getAddress()) == null) {
166        serversToIndex.put(sn.getAddress(), numServers++);
167      }
168      if (!hostsToIndex.containsKey(sn.getHostname())) {
169        hostsToIndex.put(sn.getHostname(), numHosts++);
170        serversPerHostList.add(new ArrayList<>(1));
171      }
172
173      int serverIndex = serversToIndex.get(sn.getAddress());
174      int hostIndex = hostsToIndex.get(sn.getHostname());
175      serversPerHostList.get(hostIndex).add(serverIndex);
176
177      String rack = this.rackManager.getRack(sn);
178      if (!racksToIndex.containsKey(rack)) {
179        racksToIndex.put(rack, numRacks++);
180        serversPerRackList.add(new ArrayList<>());
181      }
182      int rackIndex = racksToIndex.get(rack);
183      serversPerRackList.get(rackIndex).add(serverIndex);
184    }
185
186    // Count how many regions there are.
187    for (Map.Entry<ServerName, List<RegionInfo>> entry : clusterState.entrySet()) {
188      numRegions += entry.getValue().size();
189    }
190    numRegions += unassignedRegions.size();
191
192    regionsToIndex = new HashMap<>(numRegions);
193    servers = new ServerName[numServers];
194    serversPerHost = new int[numHosts][];
195    serversPerRack = new int[numRacks][];
196    regions = new RegionInfo[numRegions];
197    regionIndexToServerIndex = new int[numRegions];
198    initialRegionIndexToServerIndex = new int[numRegions];
199    regionIndexToTableIndex = new int[numRegions];
200    regionIndexToPrimaryIndex = new int[numRegions];
201    regionLoads = new Deque[numRegions];
202
203    regionLocations = new int[numRegions][];
204    serverIndicesSortedByRegionCount = new Integer[numServers];
205    serverIndicesSortedByLocality = new Integer[numServers];
206    localityPerServer = new float[numServers];
207
208    serverIndexToHostIndex = new int[numServers];
209    serverIndexToRackIndex = new int[numServers];
210    regionsPerServer = new int[numServers][];
211    serverIndexToRegionsOffset = new int[numServers];
212    regionsPerHost = new int[numHosts][];
213    regionsPerRack = new int[numRacks][];
214    primariesOfRegionsPerServer = new int[numServers][];
215    primariesOfRegionsPerHost = new int[numHosts][];
216    primariesOfRegionsPerRack = new int[numRacks][];
217
218    int tableIndex = 0, regionIndex = 0, regionPerServerIndex = 0;
219
220    for (Map.Entry<ServerName, List<RegionInfo>> entry : clusterState.entrySet()) {
221      if (entry.getKey() == null) {
222        LOG.warn("SERVERNAME IS NULL, skipping " + entry.getValue());
223        continue;
224      }
225      int serverIndex = serversToIndex.get(entry.getKey().getAddress());
226
227      // keep the servername if this is the first server name for this hostname
228      // or this servername has the newest startcode.
229      if (servers[serverIndex] == null ||
230        servers[serverIndex].getStartcode() < entry.getKey().getStartcode()) {
231        servers[serverIndex] = entry.getKey();
232      }
233
234      if (regionsPerServer[serverIndex] != null) {
235        // there is another server with the same hostAndPort in ClusterState.
236        // allocate the array for the total size
237        regionsPerServer[serverIndex] =
238          new int[entry.getValue().size() + regionsPerServer[serverIndex].length];
239      } else {
240        regionsPerServer[serverIndex] = new int[entry.getValue().size()];
241      }
242      primariesOfRegionsPerServer[serverIndex] = new int[regionsPerServer[serverIndex].length];
243      serverIndicesSortedByRegionCount[serverIndex] = serverIndex;
244      serverIndicesSortedByLocality[serverIndex] = serverIndex;
245    }
246
247    hosts = new String[numHosts];
248    for (Map.Entry<String, Integer> entry : hostsToIndex.entrySet()) {
249      hosts[entry.getValue()] = entry.getKey();
250    }
251    racks = new String[numRacks];
252    for (Map.Entry<String, Integer> entry : racksToIndex.entrySet()) {
253      racks[entry.getValue()] = entry.getKey();
254    }
255
256    for (Map.Entry<ServerName, List<RegionInfo>> entry : clusterState.entrySet()) {
257      int serverIndex = serversToIndex.get(entry.getKey().getAddress());
258      regionPerServerIndex = serverIndexToRegionsOffset[serverIndex];
259
260      int hostIndex = hostsToIndex.get(entry.getKey().getHostname());
261      serverIndexToHostIndex[serverIndex] = hostIndex;
262
263      int rackIndex = racksToIndex.get(this.rackManager.getRack(entry.getKey()));
264      serverIndexToRackIndex[serverIndex] = rackIndex;
265
266      for (RegionInfo region : entry.getValue()) {
267        registerRegion(region, regionIndex, serverIndex, loads, regionFinder);
268        regionsPerServer[serverIndex][regionPerServerIndex++] = regionIndex;
269        regionIndex++;
270      }
271      serverIndexToRegionsOffset[serverIndex] = regionPerServerIndex;
272    }
273
274    for (RegionInfo region : unassignedRegions) {
275      registerRegion(region, regionIndex, -1, loads, regionFinder);
276      regionIndex++;
277    }
278
279    for (int i = 0; i < serversPerHostList.size(); i++) {
280      serversPerHost[i] = new int[serversPerHostList.get(i).size()];
281      for (int j = 0; j < serversPerHost[i].length; j++) {
282        serversPerHost[i][j] = serversPerHostList.get(i).get(j);
283      }
284      if (serversPerHost[i].length > 1) {
285        multiServersPerHost = true;
286      }
287    }
288
289    for (int i = 0; i < serversPerRackList.size(); i++) {
290      serversPerRack[i] = new int[serversPerRackList.get(i).size()];
291      for (int j = 0; j < serversPerRack[i].length; j++) {
292        serversPerRack[i][j] = serversPerRackList.get(i).get(j);
293      }
294    }
295
296    numTables = tables.size();
297    LOG.debug("Number of tables={}", numTables);
298    numRegionsPerServerPerTable = new int[numServers][numTables];
299    numRegionsPerTable = new int[numTables];
300
301    for (int i = 0; i < numServers; i++) {
302      for (int j = 0; j < numTables; j++) {
303        numRegionsPerServerPerTable[i][j] = 0;
304      }
305    }
306
307    for (int i = 0; i < regionIndexToServerIndex.length; i++) {
308      if (regionIndexToServerIndex[i] >= 0) {
309        numRegionsPerServerPerTable[regionIndexToServerIndex[i]][regionIndexToTableIndex[i]]++;
310        numRegionsPerTable[regionIndexToTableIndex[i]]++;
311      }
312    }
313
314    // Avoid repeated computation for planning
315    meanRegionsPerTable = new double[numTables];
316    regionSkewByTable = new double[numTables];
317    maxRegionSkewByTable  = new double[numTables];
318    minRegionSkewByTable = new double[numTables];
319
320    for (int i = 0; i < numTables; i++) {
321      meanRegionsPerTable[i] = Double.valueOf(numRegionsPerTable[i]) / numServers;
322      minRegionSkewByTable[i] += DoubleArrayCost.getMinSkew(numRegionsPerTable[i], numServers);
323      maxRegionSkewByTable[i] += DoubleArrayCost.getMaxSkew(numRegionsPerTable[i], numServers);
324    }
325
326    for (int[] aNumRegionsPerServerPerTable : numRegionsPerServerPerTable) {
327      for (int tableIdx = 0; tableIdx < aNumRegionsPerServerPerTable.length; tableIdx++) {
328        regionSkewByTable[tableIdx] += Math.abs(aNumRegionsPerServerPerTable[tableIdx] -
329          meanRegionsPerTable[tableIdx]);
330      }
331    }
332
333    for (int i = 0; i < regions.length; i++) {
334      RegionInfo info = regions[i];
335      if (RegionReplicaUtil.isDefaultReplica(info)) {
336        regionIndexToPrimaryIndex[i] = i;
337      } else {
338        hasRegionReplicas = true;
339        RegionInfo primaryInfo = RegionReplicaUtil.getRegionInfoForDefaultReplica(info);
340        regionIndexToPrimaryIndex[i] = regionsToIndex.getOrDefault(primaryInfo, -1);
341      }
342    }
343
344    for (int i = 0; i < regionsPerServer.length; i++) {
345      primariesOfRegionsPerServer[i] = new int[regionsPerServer[i].length];
346      for (int j = 0; j < regionsPerServer[i].length; j++) {
347        int primaryIndex = regionIndexToPrimaryIndex[regionsPerServer[i][j]];
348        primariesOfRegionsPerServer[i][j] = primaryIndex;
349      }
350      // sort the regions by primaries.
351      Arrays.sort(primariesOfRegionsPerServer[i]);
352    }
353
354    // compute regionsPerHost
355    if (multiServersPerHost) {
356      for (int i = 0; i < serversPerHost.length; i++) {
357        int numRegionsPerHost = 0;
358        for (int j = 0; j < serversPerHost[i].length; j++) {
359          numRegionsPerHost += regionsPerServer[serversPerHost[i][j]].length;
360        }
361        regionsPerHost[i] = new int[numRegionsPerHost];
362        primariesOfRegionsPerHost[i] = new int[numRegionsPerHost];
363      }
364      for (int i = 0; i < serversPerHost.length; i++) {
365        int numRegionPerHostIndex = 0;
366        for (int j = 0; j < serversPerHost[i].length; j++) {
367          for (int k = 0; k < regionsPerServer[serversPerHost[i][j]].length; k++) {
368            int region = regionsPerServer[serversPerHost[i][j]][k];
369            regionsPerHost[i][numRegionPerHostIndex] = region;
370            int primaryIndex = regionIndexToPrimaryIndex[region];
371            primariesOfRegionsPerHost[i][numRegionPerHostIndex] = primaryIndex;
372            numRegionPerHostIndex++;
373          }
374        }
375        // sort the regions by primaries.
376        Arrays.sort(primariesOfRegionsPerHost[i]);
377      }
378    }
379
380    // compute regionsPerRack
381    if (numRacks > 1) {
382      for (int i = 0; i < serversPerRack.length; i++) {
383        int numRegionsPerRack = 0;
384        for (int j = 0; j < serversPerRack[i].length; j++) {
385          numRegionsPerRack += regionsPerServer[serversPerRack[i][j]].length;
386        }
387        regionsPerRack[i] = new int[numRegionsPerRack];
388        primariesOfRegionsPerRack[i] = new int[numRegionsPerRack];
389      }
390
391      for (int i = 0; i < serversPerRack.length; i++) {
392        int numRegionPerRackIndex = 0;
393        for (int j = 0; j < serversPerRack[i].length; j++) {
394          for (int k = 0; k < regionsPerServer[serversPerRack[i][j]].length; k++) {
395            int region = regionsPerServer[serversPerRack[i][j]][k];
396            regionsPerRack[i][numRegionPerRackIndex] = region;
397            int primaryIndex = regionIndexToPrimaryIndex[region];
398            primariesOfRegionsPerRack[i][numRegionPerRackIndex] = primaryIndex;
399            numRegionPerRackIndex++;
400          }
401        }
402        // sort the regions by primaries.
403        Arrays.sort(primariesOfRegionsPerRack[i]);
404      }
405    }
406  }

408  /** Helper for Cluster constructor to handle a region */
409  private void registerRegion(RegionInfo region, int regionIndex, int serverIndex,
410    Map<String, Deque<BalancerRegionLoad>> loads, RegionHDFSBlockLocationFinder regionFinder) {
411    String tableName = region.getTable().getNameAsString();
412    if (!tablesToIndex.containsKey(tableName)) {
413      tables.add(tableName);
414      tablesToIndex.put(tableName, tablesToIndex.size());
415    }
416    int tableIndex = tablesToIndex.get(tableName);
417
418    regionsToIndex.put(region, regionIndex);
419    regions[regionIndex] = region;
420    regionIndexToServerIndex[regionIndex] = serverIndex;
421    initialRegionIndexToServerIndex[regionIndex] = serverIndex;
422    regionIndexToTableIndex[regionIndex] = tableIndex;
423
424    // region load
425    if (loads != null) {
426      Deque<BalancerRegionLoad> rl = loads.get(region.getRegionNameAsString());
427      // That could have failed if the RegionLoad is using the other regionName
428      if (rl == null) {
429        // Try getting the region load using encoded name.
430        rl = loads.get(region.getEncodedName());
431      }
432      regionLoads[regionIndex] = rl;
433    }
434
435    if (regionFinder != null) {
436      // region location
437      List<ServerName> loc = regionFinder.getTopBlockLocations(region);
438      regionLocations[regionIndex] = new int[loc.size()];
439      for (int i = 0; i < loc.size(); i++) {
440        regionLocations[regionIndex][i] = loc.get(i) == null ? -1 :
441          (serversToIndex.get(loc.get(i).getAddress()) == null ? -1 :
442            serversToIndex.get(loc.get(i).getAddress()));
443      }
444    }
445  }

447  /**
448   * Returns true iff a given server has less regions than the balanced amount
449   */
450  public boolean serverHasTooFewRegions(int server) {
451    int minLoad = this.numRegions / numServers;
452    int numRegions = getNumRegions(server);
453    return numRegions < minLoad;
454  }

456  /**
457   * Retrieves and lazily initializes a field storing the locality of every region/server
458   * combination
459   */
460  public float[][] getOrComputeRackLocalities() {
461    if (rackLocalities == null || regionsToMostLocalEntities == null) {
462      computeCachedLocalities();
463    }
464    return rackLocalities;
465  }
466
467  /**
468   * Lazily initializes and retrieves a mapping of region -> server for which region has the highest
469   * the locality
470   */
471  public int[] getOrComputeRegionsToMostLocalEntities(BalancerClusterState.LocalityType type) {
472    if (rackLocalities == null || regionsToMostLocalEntities == null) {
473      computeCachedLocalities();
474    }
475    return regionsToMostLocalEntities[type.ordinal()];
476  }

478  /**
479   * Looks up locality from cache of localities. Will create cache if it does not already exist.
480   */
481  public float getOrComputeLocality(int region, int entity,
482    BalancerClusterState.LocalityType type) {
483    switch (type) {
484      case SERVER:
485        return getLocalityOfRegion(region, entity);
486      case RACK:
487        return getOrComputeRackLocalities()[region][entity];
488      default:
489        throw new IllegalArgumentException("Unsupported LocalityType: " + type);
490    }
491  }
492
493  /**
494   * Returns locality weighted by region size in MB. Will create locality cache if it does not
495   * already exist.
496   */
497  public double getOrComputeWeightedLocality(int region, int server,
498    BalancerClusterState.LocalityType type) {
499    return getRegionSizeMB(region) * getOrComputeLocality(region, server, type);
500  }

502  /**
503   * Returns the size in MB from the most recent RegionLoad for region
504   */
505  public int getRegionSizeMB(int region) {
506    Deque<BalancerRegionLoad> load = regionLoads[region];
507    // This means regions have no actual data on disk
508    if (load == null) {
509      return 0;
510    }
511    return regionLoads[region].getLast().getStorefileSizeMB();
512  }

514  /**
515   * Computes and caches the locality for each region/rack combinations, as well as storing a
516   * mapping of region -> server and region -> rack such that server and rack have the highest
517   * locality for region
518   */
519  private void computeCachedLocalities() {
520    rackLocalities = new float[numRegions][numRacks];
521    regionsToMostLocalEntities = new int[LocalityType.values().length][numRegions];
522
523    // Compute localities and find most local server per region
524    for (int region = 0; region < numRegions; region++) {
525      int serverWithBestLocality = 0;
526      float bestLocalityForRegion = 0;
527      for (int server = 0; server < numServers; server++) {
528        // Aggregate per-rack locality
529        float locality = getLocalityOfRegion(region, server);
530        int rack = serverIndexToRackIndex[server];
531        int numServersInRack = serversPerRack[rack].length;
532        rackLocalities[region][rack] += locality / numServersInRack;
533
534        if (locality > bestLocalityForRegion) {
535          serverWithBestLocality = server;
536          bestLocalityForRegion = locality;
537        }
538      }
539      regionsToMostLocalEntities[LocalityType.SERVER.ordinal()][region] = serverWithBestLocality;
540
541      // Find most local rack per region
542      int rackWithBestLocality = 0;
543      float bestRackLocalityForRegion = 0.0f;
544      for (int rack = 0; rack < numRacks; rack++) {
545        float rackLocality = rackLocalities[region][rack];
546        if (rackLocality > bestRackLocalityForRegion) {
547          bestRackLocalityForRegion = rackLocality;
548          rackWithBestLocality = rack;
549        }
550      }
551      regionsToMostLocalEntities[LocalityType.RACK.ordinal()][region] = rackWithBestLocality;
552    }
553
554  }

556  /**
557   * Maps region index to rack index

559  public int getRackForRegion(int region) {
560    return serverIndexToRackIndex[regionIndexToServerIndex[region]];

567  public void doAction(BalanceAction action) {
568    switch (action.getType()) {
569      case NULL:
570        break;
571      case ASSIGN_REGION:
572        // FindBugs: Having the assert quietens FB BC_UNCONFIRMED_CAST warnings
573        assert action instanceof AssignRegionAction : action.getClass();
574        AssignRegionAction ar = (AssignRegionAction) action;
575        regionsPerServer[ar.getServer()] =
576          addRegion(regionsPerServer[ar.getServer()], ar.getRegion());
577        regionMoved(ar.getRegion(), -1, ar.getServer());
578        break;
579      case MOVE_REGION:
580        assert action instanceof MoveRegionAction : action.getClass();
581        MoveRegionAction mra = (MoveRegionAction) action;
582        regionsPerServer[mra.getFromServer()] =
583          removeRegion(regionsPerServer[mra.getFromServer()], mra.getRegion());
584        regionsPerServer[mra.getToServer()] =
585          addRegion(regionsPerServer[mra.getToServer()], mra.getRegion());
586        regionMoved(mra.getRegion(), mra.getFromServer(), mra.getToServer());
587        break;
588      case SWAP_REGIONS:
589        assert action instanceof SwapRegionsAction : action.getClass();
590        SwapRegionsAction a = (SwapRegionsAction) action;
591        regionsPerServer[a.getFromServer()] =
592          replaceRegion(regionsPerServer[a.getFromServer()], a.getFromRegion(), a.getToRegion());
593        regionsPerServer[a.getToServer()] =
594          replaceRegion(regionsPerServer[a.getToServer()], a.getToRegion(), a.getFromRegion());
595        regionMoved(a.getFromRegion(), a.getFromServer(), a.getToServer());
596        regionMoved(a.getToRegion(), a.getToServer(), a.getFromServer());
597        break;
598      default:
599        throw new RuntimeException("Uknown action:" + action.getType());
600    }
601  }

603  /**
604   * Return true if the placement of region on server would lower the availability of the region in
605   * question
606   * @return true or false
607   */
608  boolean wouldLowerAvailability(RegionInfo regionInfo, ServerName serverName) {
609    if (!serversToIndex.containsKey(serverName.getAddress())) {
610      return false; // safeguard against race between cluster.servers and servers from LB method
611                    // args
612    }
613    int server = serversToIndex.get(serverName.getAddress());
614    int region = regionsToIndex.get(regionInfo);
615
616    // Region replicas for same region should better assign to different servers
617    for (int i : regionsPerServer[server]) {
618      RegionInfo otherRegionInfo = regions[i];
619      if (RegionReplicaUtil.isReplicasForSameRegion(regionInfo, otherRegionInfo)) {
620        return true;
621      }
622    }
623
624    int primary = regionIndexToPrimaryIndex[region];
625    if (primary == -1) {
626      return false;
627    }
628    // there is a subset relation for server < host < rack
629    // check server first
630    if (contains(primariesOfRegionsPerServer[server], primary)) {
631      // check for whether there are other servers that we can place this region
632      for (int i = 0; i < primariesOfRegionsPerServer.length; i++) {
633        if (i != server && !contains(primariesOfRegionsPerServer[i], primary)) {
634          return true; // meaning there is a better server
635        }
636      }
637      return false; // there is not a better server to place this
638    }
639
640    // check host
641    if (multiServersPerHost) {
642      // these arrays would only be allocated if we have more than one server per host
643      int host = serverIndexToHostIndex[server];
644      if (contains(primariesOfRegionsPerHost[host], primary)) {
645        // check for whether there are other hosts that we can place this region
646        for (int i = 0; i < primariesOfRegionsPerHost.length; i++) {
647          if (i != host && !contains(primariesOfRegionsPerHost[i], primary)) {
648            return true; // meaning there is a better host
649          }
650        }
651        return false; // there is not a better host to place this
652      }
653    }
654
655    // check rack
656    if (numRacks > 1) {
657      int rack = serverIndexToRackIndex[server];
658      if (contains(primariesOfRegionsPerRack[rack], primary)) {
659        // check for whether there are other racks that we can place this region
660        for (int i = 0; i < primariesOfRegionsPerRack.length; i++) {
661          if (i != rack && !contains(primariesOfRegionsPerRack[i], primary)) {
662            return true; // meaning there is a better rack
663          }
664        }
665        return false; // there is not a better rack to place this
666      }
667    }
668
669    return false;
670  }

677    int region = regionsToIndex.get(regionInfo);
678    doAction(new AssignRegionAction(region, server));

681  void regionMoved(int region, int oldServer, int newServer) {
682    regionIndexToServerIndex[region] = newServer;
683    if (initialRegionIndexToServerIndex[region] == newServer) {
684      numMovedRegions--; // region moved back to original location
685    } else if (oldServer >= 0 && initialRegionIndexToServerIndex[region] == oldServer) {
686      numMovedRegions++; // region moved from original location
687    }
688    int tableIndex = regionIndexToTableIndex[region];
689    if (oldServer >= 0) {
690      numRegionsPerServerPerTable[oldServer][tableIndex]--;
691      // update regionSkewPerTable for the move from old server
692      regionSkewByTable[tableIndex] += getSkewChangeFor(oldServer, tableIndex, -1);
693    }
694    numRegionsPerServerPerTable[newServer][tableIndex]++;
695
696    // update regionSkewPerTable for the move to new server
697    regionSkewByTable[tableIndex] += getSkewChangeFor(newServer, tableIndex, 1);
698
699    // update for servers
700    int primary = regionIndexToPrimaryIndex[region];
701    if (oldServer >= 0) {
702      primariesOfRegionsPerServer[oldServer] =
703        removeRegion(primariesOfRegionsPerServer[oldServer], primary);
704    }
705    primariesOfRegionsPerServer[newServer] =
706      addRegionSorted(primariesOfRegionsPerServer[newServer], primary);
707
708    // update for hosts
709    if (multiServersPerHost) {
710      int oldHost = oldServer >= 0 ? serverIndexToHostIndex[oldServer] : -1;
711      int newHost = serverIndexToHostIndex[newServer];
712      if (newHost != oldHost) {
713        regionsPerHost[newHost] = addRegion(regionsPerHost[newHost], region);
714        primariesOfRegionsPerHost[newHost] =
715          addRegionSorted(primariesOfRegionsPerHost[newHost], primary);
716        if (oldHost >= 0) {
717          regionsPerHost[oldHost] = removeRegion(regionsPerHost[oldHost], region);
718          primariesOfRegionsPerHost[oldHost] =
719            removeRegion(primariesOfRegionsPerHost[oldHost], primary); // will still be sorted
720        }
721      }
722    }
723
724    // update for racks
725    if (numRacks > 1) {
726      int oldRack = oldServer >= 0 ? serverIndexToRackIndex[oldServer] : -1;
727      int newRack = serverIndexToRackIndex[newServer];
728      if (newRack != oldRack) {
729        regionsPerRack[newRack] = addRegion(regionsPerRack[newRack], region);
730        primariesOfRegionsPerRack[newRack] =
731          addRegionSorted(primariesOfRegionsPerRack[newRack], primary);
732        if (oldRack >= 0) {
733          regionsPerRack[oldRack] = removeRegion(regionsPerRack[oldRack], region);
734          primariesOfRegionsPerRack[oldRack] =
735            removeRegion(primariesOfRegionsPerRack[oldRack], primary); // will still be sorted
736        }
737      }
738    }
739  }

741  int[] removeRegion(int[] regions, int regionIndex) {
742    // TODO: this maybe costly. Consider using linked lists
743    int[] newRegions = new int[regions.length - 1];
744    int i = 0;
745    for (i = 0; i < regions.length; i++) {
746      if (regions[i] == regionIndex) {
747        break;
748      }
749      newRegions[i] = regions[i];
750    }
751    System.arraycopy(regions, i + 1, newRegions, i, newRegions.length - i);
752    return newRegions;
753  }
754
755  int[] addRegion(int[] regions, int regionIndex) {
756    int[] newRegions = new int[regions.length + 1];
757    System.arraycopy(regions, 0, newRegions, 0, regions.length);
758    newRegions[newRegions.length - 1] = regionIndex;
759    return newRegions;
760  }
761
762  int[] addRegionSorted(int[] regions, int regionIndex) {
763    int[] newRegions = new int[regions.length + 1];
764    int i = 0;
765    for (i = 0; i < regions.length; i++) { // find the index to insert
766      if (regions[i] > regionIndex) {
767        break;
768      }
769    }
770    System.arraycopy(regions, 0, newRegions, 0, i); // copy first half
771    System.arraycopy(regions, i, newRegions, i + 1, regions.length - i); // copy second half
772    newRegions[i] = regionIndex;
773
774    return newRegions;
775  }

777  int[] replaceRegion(int[] regions, int regionIndex, int newRegionIndex) {
778    int i = 0;
779    for (i = 0; i < regions.length; i++) {
780      if (regions[i] == regionIndex) {
781        regions[i] = newRegionIndex;
782        break;
783      }
784    }
785    return regions;
786  }

802  int getLowestLocalityRegionOnServer(int serverIndex) {
803    if (regionFinder != null) {
804      float lowestLocality = 1.0f;
805      int lowestLocalityRegionIndex = -1;
806      if (regionsPerServer[serverIndex].length == 0) {
807        // No regions on that region server
808        return -1;
809      }
810      for (int j = 0; j < regionsPerServer[serverIndex].length; j++) {
811        int regionIndex = regionsPerServer[serverIndex][j];
812        HDFSBlocksDistribution distribution =
813          regionFinder.getBlockDistribution(regions[regionIndex]);
814        float locality = distribution.getBlockLocalityIndex(servers[serverIndex].getHostname());
815        // skip empty region
816        if (distribution.getUniqueBlocksTotalWeight() == 0) {
817          continue;
818        }
819        if (locality < lowestLocality) {
820          lowestLocality = locality;
821          lowestLocalityRegionIndex = j;
822        }
823      }
824      if (lowestLocalityRegionIndex == -1) {
825        return -1;
826      }
827      if (LOG.isTraceEnabled()) {
828        LOG.trace("Lowest locality region is " +
829          regions[regionsPerServer[serverIndex][lowestLocalityRegionIndex]]
830            .getRegionNameAsString() +
831          " with locality " + lowestLocality + " and its region server contains " +
832          regionsPerServer[serverIndex].length + " regions");
833      }
834      return regionsPerServer[serverIndex][lowestLocalityRegionIndex];
835    } else {
836      return -1;
837    }
838  }

874  private double getSkewChangeFor(int serverIndex, int tableIndex, int regionCountChange) {
875    double curSkew = Math.abs(numRegionsPerServerPerTable[serverIndex][tableIndex] -
876      meanRegionsPerTable[tableIndex]);
877    double oldSkew = Math.abs(numRegionsPerServerPerTable[serverIndex][tableIndex] -
878      regionCountChange - meanRegionsPerTable[tableIndex]);
879    return curSkew - oldSkew;
880  }
