189  protected boolean needsBalance(TableName tableName, BalancerClusterState c) {
190    ClusterLoadState cs = new ClusterLoadState(c.clusterState);
191    if (cs.getNumServers() < MIN_SERVER_BALANCE) {
192      if (LOG.isDebugEnabled()) {
193        LOG.debug("Not running balancer because only " + cs.getNumServers()
194            + " active regionserver(s)");
195      }
196      return false;
197    }
198    if(areSomeRegionReplicasColocated(c)) return true;
199    if(idleRegionServerExist(c)) {
200      return true;
201    }
202
203    // Check if we even need to do any load balancing
204    // HBASE-3681 check sloppiness first
205    float average = cs.getLoadAverage(); // for logging
206    int floor = (int) Math.floor(average * (1 - slop));
207    int ceiling = (int) Math.ceil(average * (1 + slop));
208    if (!(cs.getMaxLoad() > ceiling || cs.getMinLoad() < floor)) {
209      NavigableMap<ServerAndLoad, List<RegionInfo>> serversByLoad = cs.getServersByLoad();
210      if (LOG.isTraceEnabled()) {
211        // If nothing to balance, then don't say anything unless trace-level logging.
212        LOG.trace("Skipping load balancing because balanced cluster; " +
213          "servers=" + cs.getNumServers() +
214          " regions=" + cs.getNumRegions() + " average=" + average +
215          " mostloaded=" + serversByLoad.lastKey().getLoad() +
216          " leastloaded=" + serversByLoad.firstKey().getLoad());
217      }
218      return false;
219    }
220    return true;
221  }

234  protected final boolean idleRegionServerExist(BalancerClusterState c){
235    boolean isServerExistsWithMoreRegions = false;
236    boolean isServerExistsWithZeroRegions = false;
237    for (int[] serverList: c.regionsPerServer){
238      if (serverList.length > 1) {
239        isServerExistsWithMoreRegions = true;
240      }
241      if (serverList.length == 0) {
242        isServerExistsWithZeroRegions = true;
243      }
244    }
245    return isServerExistsWithMoreRegions && isServerExistsWithZeroRegions;
246  }

248  /**
249   * Generates a bulk assignment plan to be used on cluster startup using a
250   * simple round-robin assignment.
251   * <p>
252   * Takes a list of all the regions and all the servers in the cluster and
253   * returns a map of each server to the regions that it should be assigned.
254   * <p>
255   * Currently implemented as a round-robin assignment. Same invariant as load
256   * balancing, all servers holding floor(avg) or ceiling(avg).
257   *
258   * TODO: Use block locations from HDFS to place regions with their blocks
259   *
260   * @param regions all regions
261   * @param servers all servers
262   * @return map of server to the regions it should take, or emptyMap if no
263   *         assignment is possible (ie. no servers)
264   */
265  @Override
266  @NonNull
267  public Map<ServerName, List<RegionInfo>> roundRobinAssignment(List<RegionInfo> regions,
268    List<ServerName> servers) throws HBaseIOException {
269    metricsBalancer.incrMiscInvocations();
270    int numServers = servers == null ? 0 : servers.size();
271    if (numServers == 0) {
272      LOG.warn("Wanted to do round robin assignment but no servers to assign to");
273      return Collections.emptyMap();
274    }
275
276    // TODO: instead of retainAssignment() and roundRobinAssignment(), we should just run the
277    // normal LB.balancerCluster() with unassignedRegions. We only need to have a candidate
278    // generator for AssignRegionAction. The LB will ensure the regions are mostly local
279    // and balanced. This should also run fast with fewer number of iterations.
280
281    if (numServers == 1) { // Only one server, nothing fancy we can do here
282      return Collections.singletonMap(servers.get(0), new ArrayList<>(regions));
283    }
284
285    BalancerClusterState cluster = createCluster(servers, regions);
286    Map<ServerName, List<RegionInfo>> assignments = new HashMap<>();
287    roundRobinAssignment(cluster, regions, servers, assignments);
288    return assignments;
289  }

291  protected BalancerClusterState createCluster(List<ServerName> servers,
292    Collection<RegionInfo> regions) throws HBaseIOException {
293    boolean hasRegionReplica = false;
294    try {
295      if (services != null && services.getTableDescriptors() != null) {
296        Map<String, TableDescriptor> tds = services.getTableDescriptors().getAll();
297        for (RegionInfo regionInfo : regions) {
298          TableDescriptor td = tds.get(regionInfo.getTable().getNameWithNamespaceInclAsString());
299          if (td != null && td.getRegionReplication() > 1) {
300            hasRegionReplica = true;
301            break;
302          }
303        }
304      }
305    } catch (IOException ioe) {
306      throw new HBaseIOException(ioe);
307    }
308
309    // Get the snapshot of the current assignments for the regions in question, and then create
310    // a cluster out of it. Note that we might have replicas already assigned to some servers
311    // earlier. So we want to get the snapshot to see those assignments, but this will only contain
312    // replicas of the regions that are passed (for performance).
313    Map<ServerName, List<RegionInfo>> clusterState = null;
314    if (!hasRegionReplica) {
315      clusterState = getRegionAssignmentsByServer(regions);
316    } else {
317      // for the case where we have region replica it is better we get the entire cluster's snapshot
318      clusterState = getRegionAssignmentsByServer(null);
319    }
320
321    for (ServerName server : servers) {
322      if (!clusterState.containsKey(server)) {
323        clusterState.put(server, EMPTY_REGION_LIST);
324      }
325    }
326    return new BalancerClusterState(regions, clusterState, null, this.regionFinder,
327        rackManager);
328  }

335  /**
336   * Used to assign a single region to a random server.
337   */
338  @Override
339  public ServerName randomAssignment(RegionInfo regionInfo, List<ServerName> servers)
340      throws HBaseIOException {
341    metricsBalancer.incrMiscInvocations();
342    int numServers = servers == null ? 0 : servers.size();
343    if (numServers == 0) {
344      LOG.warn("Wanted to retain assignment but no servers to assign to");
345      return null;
346    }
347    if (numServers == 1) { // Only one server, nothing fancy we can do here
348      return servers.get(0);
349    }
350    List<ServerName> idleServers = findIdleServers(servers);
351    if (idleServers.size() == 1) {
352      return idleServers.get(0);
353    }
354    final List<ServerName> finalServers = idleServers.isEmpty() ?
355            servers : idleServers;
356    List<RegionInfo> regions = Lists.newArrayList(regionInfo);
357    BalancerClusterState cluster = createCluster(finalServers, regions);
358    return randomAssignment(cluster, regionInfo, finalServers);
359  }

361  /**
362   * Generates a bulk assignment startup plan, attempting to reuse the existing
363   * assignment information from META, but adjusting for the specified list of
364   * available/online servers available for assignment.
365   * <p>
366   * Takes a map of all regions to their existing assignment from META. Also
367   * takes a list of online servers for regions to be assigned to. Attempts to
368   * retain all assignment, so in some instances initial assignment will not be
369   * completely balanced.
370   * <p>
371   * Any leftover regions without an existing server to be assigned to will be
372   * assigned randomly to available servers.
373   *
374   * @param regions regions and existing assignment from meta
375   * @param servers available servers
376   * @return map of servers and regions to be assigned to them, or emptyMap if no
377   *           assignment is possible (ie. no servers)
378   */
379  @Override
380  @NonNull
381  public Map<ServerName, List<RegionInfo>> retainAssignment(Map<RegionInfo, ServerName> regions,
382      List<ServerName> servers) throws HBaseIOException {
383    // Update metrics
384    metricsBalancer.incrMiscInvocations();
385    int numServers = servers == null ? 0 : servers.size();
386    if (numServers == 0) {
387      LOG.warn("Wanted to do retain assignment but no servers to assign to");
388      return Collections.emptyMap();
389    }
390
391    if (numServers == 1) { // Only one server, nothing fancy we can do here
392      return Collections.singletonMap(servers.get(0), new ArrayList<>(regions.keySet()));
393    }
394
395    // Group all of the old assignments by their hostname.
396    // We can't group directly by ServerName since the servers all have
397    // new start-codes.
398
399    // Group the servers by their hostname. It's possible we have multiple
400    // servers on the same host on different ports.
401    Map<ServerName, List<RegionInfo>> assignments = new HashMap<>();
402    ArrayListMultimap<String, ServerName> serversByHostname = ArrayListMultimap.create();
403    for (ServerName server : servers) {
404      assignments.put(server, new ArrayList<>());
405      serversByHostname.put(server.getHostnameLowerCase(), server);
406    }
407
408    // Collection of the hostnames that used to have regions
409    // assigned, but for which we no longer have any RS running
410    // after the cluster restart.
411    Set<String> oldHostsNoLongerPresent = Sets.newTreeSet();
412
413    // If the old servers aren't present, lets assign those regions later.
414    List<RegionInfo> randomAssignRegions = Lists.newArrayList();
415
416    int numRandomAssignments = 0;
417    int numRetainedAssigments = 0;
418    for (Map.Entry<RegionInfo, ServerName> entry : regions.entrySet()) {
419      RegionInfo region = entry.getKey();
420      ServerName oldServerName = entry.getValue();
421      List<ServerName> localServers = new ArrayList<>();
422      if (oldServerName != null) {
423        localServers = serversByHostname.get(oldServerName.getHostnameLowerCase());
424      }
425      if (localServers.isEmpty()) {
426        // No servers on the new cluster match up with this hostname, assign randomly, later.
427        randomAssignRegions.add(region);
428        if (oldServerName != null) {
429          oldHostsNoLongerPresent.add(oldServerName.getHostnameLowerCase());
430        }
431      } else if (localServers.size() == 1) {
432        // the usual case - one new server on same host
433        ServerName target = localServers.get(0);
434        assignments.get(target).add(region);
435        numRetainedAssigments++;
436      } else {
437        // multiple new servers in the cluster on this same host
438        if (localServers.contains(oldServerName)) {
439          assignments.get(oldServerName).add(region);
440          numRetainedAssigments++;
441        } else {
442          ServerName target = null;
443          for (ServerName tmp : localServers) {
444            if (tmp.getPort() == oldServerName.getPort()) {
445              target = tmp;
446              assignments.get(tmp).add(region);
447              numRetainedAssigments++;
448              break;
449            }
450          }
451          if (target == null) {
452            randomAssignRegions.add(region);
453          }
454        }
455      }
456    }
457
458    // If servers from prior assignment aren't present, then lets do randomAssignment on regions.
459    if (randomAssignRegions.size() > 0) {
460      BalancerClusterState cluster = createCluster(servers, regions.keySet());
461      for (Map.Entry<ServerName, List<RegionInfo>> entry : assignments.entrySet()) {
462        ServerName sn = entry.getKey();
463        for (RegionInfo region : entry.getValue()) {
464          cluster.doAssignRegion(region, sn);
465        }
466      }
467      for (RegionInfo region : randomAssignRegions) {
468        ServerName target = randomAssignment(cluster, region, servers);
469        assignments.get(target).add(region);
470        numRandomAssignments++;
471      }
472    }
473
474    String randomAssignMsg = "";
475    if (numRandomAssignments > 0) {
476      randomAssignMsg =
477          numRandomAssignments + " regions were assigned "
478              + "to random hosts, since the old hosts for these regions are no "
479              + "longer present in the cluster. These hosts were:\n  "
480              + Joiner.on("\n  ").join(oldHostsNoLongerPresent);
481    }
482
483    LOG.info("Reassigned " + regions.size() + " regions. " + numRetainedAssigments
484        + " retained the pre-restart assignment. " + randomAssignMsg);
485    return assignments;
486  }

518  /**
519   * Used to assign a single region to a random server.
520   */
521  private ServerName randomAssignment(BalancerClusterState cluster, RegionInfo regionInfo,
522      List<ServerName> servers) {
523    int numServers = servers.size(); // servers is not null, numServers > 1
524    ServerName sn = null;
525    final int maxIterations = numServers * 4;
526    int iterations = 0;
527    List<ServerName> usedSNs = new ArrayList<>(servers.size());
528    Random rand = ThreadLocalRandom.current();
529    do {
530      int i = rand.nextInt(numServers);
531      sn = servers.get(i);
532      if (!usedSNs.contains(sn)) {
533        usedSNs.add(sn);
534      }
535    } while (cluster.wouldLowerAvailability(regionInfo, sn)
536        && iterations++ < maxIterations);
537    if (iterations >= maxIterations) {
538      // We have reached the max. Means the servers that we collected is still lowering the
539      // availability
540      for (ServerName unusedServer : servers) {
541        if (!usedSNs.contains(unusedServer)) {
542          // check if any other unused server is there for us to use.
543          // If so use it. Else we have not other go but to go with one of them
544          if (!cluster.wouldLowerAvailability(regionInfo, unusedServer)) {
545            sn = unusedServer;
546            break;
547          }
548        }
549      }
550    }
551    cluster.doAssignRegion(regionInfo, sn);
552    return sn;
553  }

555  /**
556   * Round robin a list of regions to a list of servers
557   */
558  private void roundRobinAssignment(BalancerClusterState cluster, List<RegionInfo> regions,
559    List<ServerName> servers, Map<ServerName, List<RegionInfo>> assignments) {
560    Random rand = ThreadLocalRandom.current();
561    List<RegionInfo> unassignedRegions = new ArrayList<>();
562    int numServers = servers.size();
563    int numRegions = regions.size();
564    int max = (int) Math.ceil((float) numRegions / numServers);
565    int serverIdx = 0;
566    if (numServers > 1) {
567      serverIdx = rand.nextInt(numServers);
568    }
569    int regionIdx = 0;
570    for (int j = 0; j < numServers; j++) {
571      ServerName server = servers.get((j + serverIdx) % numServers);
572      List<RegionInfo> serverRegions = new ArrayList<>(max);
573      for (int i = regionIdx; i < numRegions; i += numServers) {
574        RegionInfo region = regions.get(i % numRegions);
575        if (cluster.wouldLowerAvailability(region, server)) {
576          unassignedRegions.add(region);
577        } else {
578          serverRegions.add(region);
579          cluster.doAssignRegion(region, server);
580        }
581      }
582      assignments.put(server, serverRegions);
583      regionIdx++;
584    }
585
586
587    List<RegionInfo> lastFewRegions = new ArrayList<>();
588    // assign the remaining by going through the list and try to assign to servers one-by-one
589    serverIdx = rand.nextInt(numServers);
590    for (RegionInfo region : unassignedRegions) {
591      boolean assigned = false;
592      for (int j = 0; j < numServers; j++) { // try all servers one by one
593        ServerName server = servers.get((j + serverIdx) % numServers);
594        if (cluster.wouldLowerAvailability(region, server)) {
595          continue;
596        } else {
597          assignments.computeIfAbsent(server, k -> new ArrayList<>()).add(region);
598          cluster.doAssignRegion(region, server);
599          serverIdx = (j + serverIdx + 1) % numServers; // remain from next server
600          assigned = true;
601          break;
602        }
603      }
604      if (!assigned) {
605        lastFewRegions.add(region);
606      }
607    }
608    // just sprinkle the rest of the regions on random regionservers. The balanceCluster will
609    // make it optimal later. we can end up with this if numReplicas > numServers.
610    for (RegionInfo region : lastFewRegions) {
611      int i = rand.nextInt(numServers);
612      ServerName server = servers.get(i);
613      assignments.computeIfAbsent(server, k -> new ArrayList<>()).add(region);
614      cluster.doAssignRegion(region, server);
615    }
616  }

627  private Map<ServerName, List<RegionInfo>> toEnsumbleTableLoad(
628      Map<TableName, Map<ServerName, List<RegionInfo>>> LoadOfAllTable) {
629    Map<ServerName, List<RegionInfo>> returnMap = new TreeMap<>();
630    for (Map<ServerName, List<RegionInfo>> serverNameListMap : LoadOfAllTable.values()) {
631      serverNameListMap.forEach((serverName, regionInfoList) -> {
632        List<RegionInfo> regionInfos =
633            returnMap.computeIfAbsent(serverName, k -> new ArrayList<>());
634        regionInfos.addAll(regionInfoList);
635      });
636    }
637    return returnMap;
638  }

644  @Override
645  public List<RegionPlan>
646      balanceCluster(Map<TableName, Map<ServerName, List<RegionInfo>>> loadOfAllTable) {
647    if (isByTable) {
648      List<RegionPlan> result = new ArrayList<>();
649      loadOfAllTable.forEach((tableName, loadOfOneTable) -> {
650        LOG.info("Start Generate Balance plan for table: " + tableName);
651        List<RegionPlan> partialPlans = balanceTable(tableName, loadOfOneTable);
652        if (partialPlans != null) {
653          result.addAll(partialPlans);
654        }
655      });
656      return result;
657    } else {
658      LOG.info("Start Generate Balance plan for cluster.");
659      return balanceTable(HConstants.ENSEMBLE_TABLE_NAME, toEnsumbleTableLoad(loadOfAllTable));
660    }
661  }
