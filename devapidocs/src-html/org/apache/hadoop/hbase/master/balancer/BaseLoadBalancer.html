128  protected final boolean idleRegionServerExist(BalancerClusterState c){
129    boolean isServerExistsWithMoreRegions = false;
130    boolean isServerExistsWithZeroRegions = false;
131    for (int[] serverList: c.regionsPerServer){
132      if (serverList.length > 1) {
133        isServerExistsWithMoreRegions = true;
134      }
135      if (serverList.length == 0) {
136        isServerExistsWithZeroRegions = true;
137      }
138    }
139    return isServerExistsWithMoreRegions && isServerExistsWithZeroRegions;
140  }

142  /**
143   * Generates a bulk assignment plan to be used on cluster startup using a
144   * simple round-robin assignment.
145   * <p/>
146   * Takes a list of all the regions and all the servers in the cluster and
147   * returns a map of each server to the regions that it should be assigned.
148   * <p/>
149   * Currently implemented as a round-robin assignment. Same invariant as load
150   * balancing, all servers holding floor(avg) or ceiling(avg).
151   *
152   * TODO: Use block locations from HDFS to place regions with their blocks
153   *
154   * @param regions all regions
155   * @param servers all servers
156   * @return map of server to the regions it should take, or emptyMap if no
157   *         assignment is possible (ie. no servers)
158   */
159  @Override
160  @NonNull
161  public Map<ServerName, List<RegionInfo>> roundRobinAssignment(List<RegionInfo> regions,
162    List<ServerName> servers) throws HBaseIOException {
163    metricsBalancer.incrMiscInvocations();
164    int numServers = servers == null ? 0 : servers.size();
165    if (numServers == 0) {
166      LOG.warn("Wanted to do round robin assignment but no servers to assign to");
167      return Collections.singletonMap(BOGUS_SERVER_NAME, new ArrayList<>(regions));
168    }
169
170    // TODO: instead of retainAssignment() and roundRobinAssignment(), we should just run the
171    // normal LB.balancerCluster() with unassignedRegions. We only need to have a candidate
172    // generator for AssignRegionAction. The LB will ensure the regions are mostly local
173    // and balanced. This should also run fast with fewer number of iterations.
174
175    if (numServers == 1) { // Only one server, nothing fancy we can do here
176      return Collections.singletonMap(servers.get(0), new ArrayList<>(regions));
177    }
178
179    BalancerClusterState cluster = createCluster(servers, regions);
180    Map<ServerName, List<RegionInfo>> assignments = new HashMap<>();
181    roundRobinAssignment(cluster, regions, servers, assignments);
182    return Collections.unmodifiableMap(assignments);
183  }

185  private BalancerClusterState createCluster(List<ServerName> servers,
186    Collection<RegionInfo> regions) throws HBaseIOException {
187    boolean hasRegionReplica= false;
188    try {
189      if (provider != null) {
190        hasRegionReplica = provider.hasRegionReplica(regions);
191      }
192    } catch (IOException ioe) {
193      throw new HBaseIOException(ioe);
194    }
195
196    // Get the snapshot of the current assignments for the regions in question, and then create
197    // a cluster out of it. Note that we might have replicas already assigned to some servers
198    // earlier. So we want to get the snapshot to see those assignments, but this will only contain
199    // replicas of the regions that are passed (for performance).
200    Map<ServerName, List<RegionInfo>> clusterState = null;
201    if (!hasRegionReplica) {
202      clusterState = getRegionAssignmentsByServer(regions);
203    } else {
204      // for the case where we have region replica it is better we get the entire cluster's snapshot
205      clusterState = getRegionAssignmentsByServer(null);
206    }
207
208    for (ServerName server : servers) {
209      if (!clusterState.containsKey(server)) {
210        clusterState.put(server, Collections.emptyList());
211      }
212    }
213    return new BalancerClusterState(regions, clusterState, null, this.regionFinder,
214        rackManager);
215  }

222  /**
223   * Used to assign a single region to a random server.
224   */
225  @Override
226  public ServerName randomAssignment(RegionInfo regionInfo, List<ServerName> servers)
227      throws HBaseIOException {
228    metricsBalancer.incrMiscInvocations();
229    int numServers = servers == null ? 0 : servers.size();
230    if (numServers == 0) {
231      LOG.warn("Wanted to retain assignment but no servers to assign to");
232      return null;
233    }
234    if (numServers == 1) { // Only one server, nothing fancy we can do here
235      return servers.get(0);
236    }
237    List<ServerName> idleServers = findIdleServers(servers);
238    if (idleServers.size() == 1) {
239      return idleServers.get(0);
240    }
241    final List<ServerName> finalServers = idleServers.isEmpty() ?
242            servers : idleServers;
243    List<RegionInfo> regions = Lists.newArrayList(regionInfo);
244    BalancerClusterState cluster = createCluster(finalServers, regions);
245    return randomAssignment(cluster, regionInfo, finalServers);
246  }

248  /**
249   * Generates a bulk assignment startup plan, attempting to reuse the existing
250   * assignment information from META, but adjusting for the specified list of
251   * available/online servers available for assignment.
252   * <p>
253   * Takes a map of all regions to their existing assignment from META. Also
254   * takes a list of online servers for regions to be assigned to. Attempts to
255   * retain all assignment, so in some instances initial assignment will not be
256   * completely balanced.
257   * <p>
258   * Any leftover regions without an existing server to be assigned to will be
259   * assigned randomly to available servers.
260   *
261   * @param regions regions and existing assignment from meta
262   * @param servers available servers
263   * @return map of servers and regions to be assigned to them, or emptyMap if no
264   *           assignment is possible (ie. no servers)
265   */
266  @Override
267  @NonNull
268  public Map<ServerName, List<RegionInfo>> retainAssignment(Map<RegionInfo, ServerName> regions,
269      List<ServerName> servers) throws HBaseIOException {
270    // Update metrics
271    metricsBalancer.incrMiscInvocations();
272    int numServers = servers == null ? 0 : servers.size();
273    if (numServers == 0) {
274      LOG.warn("Wanted to do retain assignment but no servers to assign to");
275      return Collections.singletonMap(BOGUS_SERVER_NAME, new ArrayList<>(regions.keySet()));
276    }
277
278    if (numServers == 1) { // Only one server, nothing fancy we can do here
279      return Collections.singletonMap(servers.get(0), new ArrayList<>(regions.keySet()));
280    }
281
282    // Group all of the old assignments by their hostname.
283    // We can't group directly by ServerName since the servers all have
284    // new start-codes.
285
286    // Group the servers by their hostname. It's possible we have multiple
287    // servers on the same host on different ports.
288    Map<ServerName, List<RegionInfo>> assignments = new HashMap<>();
289    ArrayListMultimap<String, ServerName> serversByHostname = ArrayListMultimap.create();
290    for (ServerName server : servers) {
291      assignments.put(server, new ArrayList<>());
292      serversByHostname.put(server.getHostnameLowerCase(), server);
293    }
294
295    // Collection of the hostnames that used to have regions
296    // assigned, but for which we no longer have any RS running
297    // after the cluster restart.
298    Set<String> oldHostsNoLongerPresent = Sets.newTreeSet();
299
300    // If the old servers aren't present, lets assign those regions later.
301    List<RegionInfo> randomAssignRegions = Lists.newArrayList();
302
303    int numRandomAssignments = 0;
304    int numRetainedAssigments = 0;
305    for (Map.Entry<RegionInfo, ServerName> entry : regions.entrySet()) {
306      RegionInfo region = entry.getKey();
307      ServerName oldServerName = entry.getValue();
308      List<ServerName> localServers = new ArrayList<>();
309      if (oldServerName != null) {
310        localServers = serversByHostname.get(oldServerName.getHostnameLowerCase());
311      }
312      if (localServers.isEmpty()) {
313        // No servers on the new cluster match up with this hostname, assign randomly, later.
314        randomAssignRegions.add(region);
315        if (oldServerName != null) {
316          oldHostsNoLongerPresent.add(oldServerName.getHostnameLowerCase());
317        }
318      } else if (localServers.size() == 1) {
319        // the usual case - one new server on same host
320        ServerName target = localServers.get(0);
321        assignments.get(target).add(region);
322        numRetainedAssigments++;
323      } else {
324        // multiple new servers in the cluster on this same host
325        if (localServers.contains(oldServerName)) {
326          assignments.get(oldServerName).add(region);
327          numRetainedAssigments++;
328        } else {
329          ServerName target = null;
330          for (ServerName tmp : localServers) {
331            if (tmp.getPort() == oldServerName.getPort()) {
332              target = tmp;
333              assignments.get(tmp).add(region);
334              numRetainedAssigments++;
335              break;
336            }
337          }
338          if (target == null) {
339            randomAssignRegions.add(region);
340          }
341        }
342      }
343    }
344
345    // If servers from prior assignment aren't present, then lets do randomAssignment on regions.
346    if (randomAssignRegions.size() > 0) {
347      BalancerClusterState cluster = createCluster(servers, regions.keySet());
348      for (Map.Entry<ServerName, List<RegionInfo>> entry : assignments.entrySet()) {
349        ServerName sn = entry.getKey();
350        for (RegionInfo region : entry.getValue()) {
351          cluster.doAssignRegion(region, sn);
352        }
353      }
354      for (RegionInfo region : randomAssignRegions) {
355        ServerName target = randomAssignment(cluster, region, servers);
356        assignments.get(target).add(region);
357        numRandomAssignments++;
358      }
359    }
360
361    String randomAssignMsg = "";
362    if (numRandomAssignments > 0) {
363      randomAssignMsg =
364          numRandomAssignments + " regions were assigned "
365              + "to random hosts, since the old hosts for these regions are no "
366              + "longer present in the cluster. These hosts were:\n  "
367              + Joiner.on("\n  ").join(oldHostsNoLongerPresent);
368    }
369
370    LOG.info("Reassigned " + regions.size() + " regions. " + numRetainedAssigments
371        + " retained the pre-restart assignment. " + randomAssignMsg);
372    return Collections.unmodifiableMap(assignments);
373  }

375  protected final float normalizeSlop(float slop) {
376    if (slop < 0) {
377      return 0;
378    }
379    if (slop > 1) {
380      return 1;
381    }
382    return slop;
383  }

396  protected void loadConf(Configuration conf) {
397    this.slop = normalizeSlop(conf.getFloat("hbase.regions.slop", getDefaultSlop()));
398    this.rackManager = new RackManager(conf);
399    useRegionFinder = conf.getBoolean("hbase.master.balancer.uselocality", true);
400    if (useRegionFinder) {
401      regionFinder = createRegionLocationFinder(conf);
402    } else {
403      regionFinder = null;
404    }
405    this.isByTable = conf.getBoolean(HConstants.HBASE_MASTER_LOADBALANCE_BYTABLE, isByTable);
406    // Print out base configs. Don't print overallSlop since it for simple balancer exclusively.
407    LOG.info("slop={}", this.slop);
408  }

442  /**
443   * Used to assign a single region to a random server.
444   */
445  private ServerName randomAssignment(BalancerClusterState cluster, RegionInfo regionInfo,
446      List<ServerName> servers) {
447    int numServers = servers.size(); // servers is not null, numServers > 1
448    ServerName sn = null;
449    final int maxIterations = numServers * 4;
450    int iterations = 0;
451    List<ServerName> usedSNs = new ArrayList<>(servers.size());
452    Random rand = ThreadLocalRandom.current();
453    do {
454      int i = rand.nextInt(numServers);
455      sn = servers.get(i);
456      if (!usedSNs.contains(sn)) {
457        usedSNs.add(sn);
458      }
459    } while (cluster.wouldLowerAvailability(regionInfo, sn)
460        && iterations++ < maxIterations);
461    if (iterations >= maxIterations) {
462      // We have reached the max. Means the servers that we collected is still lowering the
463      // availability
464      for (ServerName unusedServer : servers) {
465        if (!usedSNs.contains(unusedServer)) {
466          // check if any other unused server is there for us to use.
467          // If so use it. Else we have not other go but to go with one of them
468          if (!cluster.wouldLowerAvailability(regionInfo, unusedServer)) {
469            sn = unusedServer;
470            break;
471          }
472        }
473      }
474    }
475    cluster.doAssignRegion(regionInfo, sn);
476    return sn;
477  }

479  /**
480   * Round robin a list of regions to a list of servers
481   */
482  private void roundRobinAssignment(BalancerClusterState cluster, List<RegionInfo> regions,
483    List<ServerName> servers, Map<ServerName, List<RegionInfo>> assignments) {
484    Random rand = ThreadLocalRandom.current();
485    List<RegionInfo> unassignedRegions = new ArrayList<>();
486    int numServers = servers.size();
487    int numRegions = regions.size();
488    int max = (int) Math.ceil((float) numRegions / numServers);
489    int serverIdx = 0;
490    if (numServers > 1) {
491      serverIdx = rand.nextInt(numServers);
492    }
493    int regionIdx = 0;
494    for (int j = 0; j < numServers; j++) {
495      ServerName server = servers.get((j + serverIdx) % numServers);
496      List<RegionInfo> serverRegions = new ArrayList<>(max);
497      for (int i = regionIdx; i < numRegions; i += numServers) {
498        RegionInfo region = regions.get(i % numRegions);
499        if (cluster.wouldLowerAvailability(region, server)) {
500          unassignedRegions.add(region);
501        } else {
502          serverRegions.add(region);
503          cluster.doAssignRegion(region, server);
504        }
505      }
506      assignments.put(server, serverRegions);
507      regionIdx++;
508    }
509
510
511    List<RegionInfo> lastFewRegions = new ArrayList<>();
512    // assign the remaining by going through the list and try to assign to servers one-by-one
513    serverIdx = rand.nextInt(numServers);
514    for (RegionInfo region : unassignedRegions) {
515      boolean assigned = false;
516      for (int j = 0; j < numServers; j++) { // try all servers one by one
517        ServerName server = servers.get((j + serverIdx) % numServers);
518        if (cluster.wouldLowerAvailability(region, server)) {
519          continue;
520        } else {
521          assignments.computeIfAbsent(server, k -> new ArrayList<>()).add(region);
522          cluster.doAssignRegion(region, server);
523          serverIdx = (j + serverIdx + 1) % numServers; // remain from next server
524          assigned = true;
525          break;
526        }
527      }
528      if (!assigned) {
529        lastFewRegions.add(region);
530      }
531    }
532    // just sprinkle the rest of the regions on random regionservers. The balanceCluster will
533    // make it optimal later. we can end up with this if numReplicas > numServers.
534    for (RegionInfo region : lastFewRegions) {
535      int i = rand.nextInt(numServers);
536      ServerName server = servers.get(i);
537      assignments.computeIfAbsent(server, k -> new ArrayList<>()).add(region);
538      cluster.doAssignRegion(region, server);
539    }
540  }

549  protected final Map<ServerName, List<RegionInfo>> toEnsumbleTableLoad(
550      Map<TableName, Map<ServerName, List<RegionInfo>>> LoadOfAllTable) {
551    Map<ServerName, List<RegionInfo>> returnMap = new TreeMap<>();
552    for (Map<ServerName, List<RegionInfo>> serverNameListMap : LoadOfAllTable.values()) {
553      serverNameListMap.forEach((serverName, regionInfoList) -> {
554        List<RegionInfo> regionInfos =
555            returnMap.computeIfAbsent(serverName, k -> new ArrayList<>());
556        regionInfos.addAll(regionInfoList);
557      });
558    }
559    return returnMap;
560  }

562  /**
563   * Perform the major balance operation for table, all sub classes should override this method.
564   * <p/>
565   * Will be invoked by {@link #balanceCluster(Map)}. If
566   * {@link HConstants#HBASE_MASTER_LOADBALANCE_BYTABLE} is enabled, we will call this method
567   * multiple times, one table a time, where we will only pass in the regions for a single table
568   * each time. If not, we will pass in all the regions at once, and the {@code tableName} will be
569   * {@link HConstants#ENSEMBLE_TABLE_NAME}.
570   * @param tableName the table to be balanced
571   * @param loadOfOneTable region load of servers for the specific one table
572   * @return List of plans
573   */
574  protected abstract List<RegionPlan> balanceTable(TableName tableName,
575    Map<ServerName, List<RegionInfo>> loadOfOneTable);

585  /**
586   * Perform the major balance operation for cluster, will invoke
587   * {@link #balanceTable(TableName, Map)} to do actual balance.
588   * <p/>
589   * THIs method is marked as final which means you should not override this method. See the javadoc
590   * for {@link #balanceTable(TableName, Map)} for more details.
591   * @param loadOfAllTable region load of servers for all table
592   * @return a list of regions to be moved, including source and destination, or null if cluster is
593   *         already balanced
594   * @see #balanceTable(TableName, Map)
595   */
596  @Override
597  public final List<RegionPlan>
598    balanceCluster(Map<TableName, Map<ServerName, List<RegionInfo>>> loadOfAllTable) {
599    preBalanceCluster(loadOfAllTable);
600    if (isByTable) {
601      List<RegionPlan> result = new ArrayList<>();
602      loadOfAllTable.forEach((tableName, loadOfOneTable) -> {
603        LOG.info("Start Generate Balance plan for table: " + tableName);
604        List<RegionPlan> partialPlans = balanceTable(tableName, loadOfOneTable);
605        if (partialPlans != null) {
606          result.addAll(partialPlans);
607        }
608      });
609      return result;
610    } else {
611      LOG.debug("Start Generate Balance plan for cluster.");
612      return balanceTable(HConstants.ENSEMBLE_TABLE_NAME, toEnsumbleTableLoad(loadOfAllTable));
613    }
614  }
