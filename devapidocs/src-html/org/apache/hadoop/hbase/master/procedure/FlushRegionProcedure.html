055@InterfaceAudience.Private
056public class FlushRegionProcedure extends Procedure<MasterProcedureEnv>
057  implements TableProcedureInterface, RemoteProcedure<MasterProcedureEnv, ServerName> {
058  private static final Logger LOG = LoggerFactory.getLogger(FlushRegionProcedure.class);
059
060  private RegionInfo region;
061  private List<byte[]> columnFamilies;
062  private ProcedureEvent<?> event;
063  private boolean dispatched;
064  private boolean succ;
065  private RetryCounter retryCounter;
066
067  public FlushRegionProcedure() {
068  }
069
070  public FlushRegionProcedure(RegionInfo region) {
071    this(region, null);
072  }
073
074  public FlushRegionProcedure(RegionInfo region, List<byte[]> columnFamilies) {
075    this.region = region;
076    this.columnFamilies = columnFamilies;
077  }
078
079  @Override
080  protected Procedure<MasterProcedureEnv>[] execute(MasterProcedureEnv env)
081    throws ProcedureYieldException, ProcedureSuspendedException, InterruptedException {
082    if (dispatched) {
083      if (succ) {
084        return null;
085      }
086      dispatched = false;
087    }
088
089    RegionStates regionStates = env.getAssignmentManager().getRegionStates();
090    RegionStateNode regionNode = regionStates.getRegionStateNode(region);
091    regionNode.lock();
092    try {
093      if (!regionNode.isInState(State.OPEN) || regionNode.isInTransition()) {
094        LOG.info("State of region {} is not OPEN or in transition. Skip {} ...", region, this);
095        return null;
096      }
097      ServerName targetServer = regionNode.getRegionLocation();
098      if (targetServer == null) {
099        setTimeoutForSuspend(env,
100          String.format("target server of region %s is null", region.getRegionNameAsString()));
101        throw new ProcedureSuspendedException();
102      }
103      ServerState serverState = regionStates.getServerNode(targetServer).getState();
104      if (serverState != ServerState.ONLINE) {
105        setTimeoutForSuspend(env, String.format("target server of region %s %s is in state %s",
106          region.getRegionNameAsString(), targetServer, serverState));
107        throw new ProcedureSuspendedException();
108      }
109      try {
110        env.getRemoteDispatcher().addOperationToNode(targetServer, this);
111        dispatched = true;
112        event = new ProcedureEvent<>(this);
113        event.suspendIfNotReady(this);
114        throw new ProcedureSuspendedException();
115      } catch (FailedRemoteDispatchException e) {
116        setTimeoutForSuspend(env, "Failed send request to " + targetServer);
117        throw new ProcedureSuspendedException();
118      }
119    } finally {
120      regionNode.unlock();
121    }
122  }
123
124  @Override
125  protected synchronized boolean setTimeoutFailure(MasterProcedureEnv env) {
126    setState(ProcedureProtos.ProcedureState.RUNNABLE);
127    env.getProcedureScheduler().addFront(this);
128    return false;
129  }
130
131  @Override
132  protected void rollback(MasterProcedureEnv env) throws IOException, InterruptedException {
133    throw new UnsupportedOperationException();
134  }
135
136  @Override
137  protected boolean abort(MasterProcedureEnv env) {
138    return false;
139  }
140
141  @Override
142  public void remoteCallFailed(MasterProcedureEnv env, ServerName serverName, IOException e) {
143    complete(env, e);
144  }
145
146  @Override
147  public void remoteOperationCompleted(MasterProcedureEnv env) {
148    complete(env, null);
149  }
150
151  @Override
152  public void remoteOperationFailed(MasterProcedureEnv env, RemoteProcedureException error) {
153    complete(env, error);
154  }
155
156  private void complete(MasterProcedureEnv env, Throwable error) {
157    if (isFinished()) {
158      LOG.info("This procedure {} is already finished, skip the rest processes", this.getProcId());
159      return;
160    }
161    if (event == null) {
162      LOG.warn("procedure event for {} is null, maybe the procedure is created when recovery",
163        getProcId());
164      return;
165    }
166    if (error == null) {
167      succ = true;
168    }
169    event.wake(env.getProcedureScheduler());
170    event = null;
171  }
172
173  private void setTimeoutForSuspend(MasterProcedureEnv env, String reason) {
174    if (retryCounter == null) {
175      retryCounter = ProcedureUtil.createRetryCounter(env.getMasterConfiguration());
176    }
177    long backoff = retryCounter.getBackoffTimeAndIncrementAttempts();
178    LOG.warn("{} can not run currently because {}, wait {} ms to retry", this, reason, backoff);
179    setTimeout(Math.toIntExact(backoff));
180    setState(ProcedureProtos.ProcedureState.WAITING_TIMEOUT);
181    skipPersistence();
182  }
183
184  @Override
185  protected void serializeStateData(ProcedureStateSerializer serializer) throws IOException {
186    FlushRegionProcedureStateData.Builder builder = FlushRegionProcedureStateData.newBuilder();
187    builder.setRegion(ProtobufUtil.toRegionInfo(region));
188    if (columnFamilies != null) {
189      for (byte[] columnFamily : columnFamilies) {
190        if (columnFamily != null && columnFamily.length > 0) {
191          builder.addColumnFamily(UnsafeByteOperations.unsafeWrap(columnFamily));
192        }
193      }
194    }
195    serializer.serialize(builder.build());
196  }
197
198  @Override
199  protected void deserializeStateData(ProcedureStateSerializer serializer) throws IOException {
200    FlushRegionProcedureStateData data =
201      serializer.deserialize(FlushRegionProcedureStateData.class);
202    this.region = ProtobufUtil.toRegionInfo(data.getRegion());
203    if (data.getColumnFamilyCount() > 0) {
204      this.columnFamilies = data.getColumnFamilyList().stream().filter(cf -> !cf.isEmpty())
205        .map(ByteString::toByteArray).collect(Collectors.toList());
206    }
207  }
208
209  @Override
210  public Optional<RemoteOperation> remoteCallBuild(MasterProcedureEnv env, ServerName serverName) {
211    FlushRegionParameter.Builder builder = FlushRegionParameter.newBuilder();
212    builder.setRegion(ProtobufUtil.toRegionInfo(region));
213    if (columnFamilies != null) {
214      for (byte[] columnFamily : columnFamilies) {
215        if (columnFamily != null && columnFamily.length > 0) {
216          builder.addColumnFamily(UnsafeByteOperations.unsafeWrap(columnFamily));
217        }
218      }
219    }
220    return Optional.of(new RSProcedureDispatcher.ServerOperation(this, getProcId(),
221      FlushRegionCallable.class, builder.build().toByteArray()));
222  }
223
224  @Override
225  public TableOperationType getTableOperationType() {
226    return TableOperationType.FLUSH;
227  }
228
229  @Override
230  protected boolean waitInitialized(MasterProcedureEnv env) {
231    return env.waitInitialized(this);
232  }
233
234  @Override
235  public TableName getTableName() {
236    return region.getTable();
237  }
