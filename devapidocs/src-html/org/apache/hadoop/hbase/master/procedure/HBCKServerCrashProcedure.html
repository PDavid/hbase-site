018package org.apache.hadoop.hbase.master.procedure;
019
020import java.io.IOException;
021import java.util.ArrayList;
022import java.util.List;
023
024import org.apache.hadoop.hbase.MetaTableAccessor;
025import org.apache.hadoop.hbase.ServerName;
026import org.apache.hadoop.hbase.client.RegionInfo;
027import org.apache.hadoop.hbase.util.Pair;
028import org.apache.yetus.audience.InterfaceAudience;
029import org.slf4j.Logger;
030import org.slf4j.LoggerFactory;
031
032/**
033 * A SCP that differs from default only in how it gets the list of
034 * Regions hosted on the crashed-server; it also reads hbase:meta directly rather
035 * than rely solely on Master memory for list of Regions that were on crashed server.
036 * This version of SCP is for external invocation as part of fix-up (e.g. HBCK2's
037 * scheduleRecoveries). It is for the case where meta has references to 'Unknown Servers',
038 * servers that are in hbase:meta but not in live-server or dead-server lists; i.e. Master
039 * and hbase:meta content have deviated. It should never happen in normal running
040 * cluster but if we do drop accounting of servers, we need a means of fix-up.
041 * Eventually, as part of normal CatalogJanitor task, rather than just identify
042 * these 'Unknown Servers', it would make repair, queuing something like this
043 * HBCKSCP to do cleanup, reassigning them so Master and hbase:meta are aligned again.
044 *
045 * <p>NOTE that this SCP is costly to run; does a full scan of hbase:meta.</p>
046 */
047@InterfaceAudience.Private
048public class HBCKServerCrashProcedure extends ServerCrashProcedure {
049  private static final Logger LOG = LoggerFactory.getLogger(HBCKServerCrashProcedure.class);
050
051  /**
052   * @param serverName Name of the crashed server.
053   * @param shouldSplitWal True if we should split WALs as part of crashed server processing.
054   * @param carryingMeta True if carrying hbase:meta table region.
055   */
056  public HBCKServerCrashProcedure(final MasterProcedureEnv env, final ServerName serverName,
057                              final boolean shouldSplitWal, final boolean carryingMeta) {
058    super(env, serverName, shouldSplitWal, carryingMeta);
059  }
060
061  /**
062   * Used when deserializing from a procedure store; we'll construct one of these then call
063   * #deserializeStateData(InputStream). Do not use directly.
064   */
065  public HBCKServerCrashProcedure() {}
066
067  /**
068   * Adds Regions found by super method any found scanning hbase:meta.
069   */
070  @Override
071  @edu.umd.cs.findbugs.annotations.SuppressWarnings(value="NP_NULL_ON_SOME_PATH_EXCEPTION",
072    justification="FindBugs seems confused on ps in below.")
073  List<RegionInfo> getRegionsOnCrashedServer(MasterProcedureEnv env) {
074    // Super can return immutable emptyList.
075    List<RegionInfo> ris = super.getRegionsOnCrashedServer(env);
076    List<Pair<RegionInfo, ServerName>> ps = null;
077    try {
078      ps = MetaTableAccessor.getTableRegionsAndLocations(env.getMasterServices().getConnection(),
079              null, false);
080    } catch (IOException ioe) {
081      LOG.warn("Failed get of all regions; continuing", ioe);
082    }
083    if (ps == null || ps.isEmpty()) {
084      return ris;
085    }
086    List<RegionInfo> aggregate = ris == null || ris.isEmpty()?
087        new ArrayList<>(): new ArrayList<>(ris);
088    ps.stream().filter(p -> p.getSecond() != null && p.getSecond().equals(getServerName())).
089        forEach(p -> aggregate.add(p.getFirst()));
090    return aggregate;
091  }
092}
