232/**
233 * HMaster is the "master server" for HBase. An HBase cluster has one active master. If many masters
234 * are started, all compete. Whichever wins goes on to run the cluster. All others park themselves
235 * in their constructor until master or cluster shutdown or until the active master loses its lease
236 * in zookeeper. Thereafter, all running master jostle to take over master role.
237 * <p/>
238 * The Master can be asked shutdown the cluster. See {@link #shutdown()}. In this case it will tell
239 * all regionservers to go down and then wait on them all reporting in that they are down. This
240 * master will then shut itself down.
241 * <p/>
242 * You can also shutdown just this master. Call {@link #stopMaster()}.
243 * @see org.apache.zookeeper.Watcher
244 */

406  /**
407   * Initializes the HMaster. The steps are as follows:
408   * <p>
409   * <ol>
410   * <li>Initialize the local HRegionServer
411   * <li>Start the ActiveMasterManager.
412   * </ol>
413   * <p>
414   * Remaining steps of initialization occur in
415   * {@link #finishActiveMasterInitialization(MonitoredTask)} after the master becomes the
416   * active one.
417   */
418  public HMaster(final Configuration conf) throws IOException {
419    super(conf);
420    try {
421      if (conf.getBoolean(MAINTENANCE_MODE, false)) {
422        LOG.info("Detected {}=true via configuration.", MAINTENANCE_MODE);
423        maintenanceMode = true;
424      } else if (Boolean.getBoolean(MAINTENANCE_MODE)) {
425        LOG.info("Detected {}=true via environment variables.", MAINTENANCE_MODE);
426        maintenanceMode = true;
427      } else {
428        maintenanceMode = false;
429      }
430      this.rsFatals = new MemoryBoundedLogMessageBuffer(
431          conf.getLong("hbase.master.buffer.for.rs.fatals", 1 * 1024 * 1024));
432      LOG.info("hbase.rootdir={}, hbase.cluster.distributed={}", getDataRootDir(),
433          this.conf.getBoolean(HConstants.CLUSTER_DISTRIBUTED, false));
434
435      // Disable usage of meta replicas in the master
436      this.conf.setBoolean(HConstants.USE_META_REPLICAS, false);
437
438      decorateMasterConfiguration(this.conf);
439
440      // Hack! Maps DFSClient => Master for logs.  HDFS made this
441      // config param for task trackers, but we can piggyback off of it.
442      if (this.conf.get("mapreduce.task.attempt.id") == null) {
443        this.conf.set("mapreduce.task.attempt.id", "hb_m_" + this.serverName.toString());
444      }
445
446      this.metricsMaster = new MetricsMaster(new MetricsMasterWrapperImpl(this));
447
448      // preload table descriptor at startup
449      this.preLoadTableDescriptors = conf.getBoolean("hbase.master.preload.tabledescriptors", true);
450
451      this.maxBalancingTime = getMaxBalancingTime();
452      this.maxRitPercent = conf.getDouble(HConstants.HBASE_MASTER_BALANCER_MAX_RIT_PERCENT,
453          HConstants.DEFAULT_HBASE_MASTER_BALANCER_MAX_RIT_PERCENT);
454
455      // Do we publish the status?
456
457      boolean shouldPublish = conf.getBoolean(HConstants.STATUS_PUBLISHED,
458          HConstants.STATUS_PUBLISHED_DEFAULT);
459      Class<? extends ClusterStatusPublisher.Publisher> publisherClass =
460          conf.getClass(ClusterStatusPublisher.STATUS_PUBLISHER_CLASS,
461              ClusterStatusPublisher.DEFAULT_STATUS_PUBLISHER_CLASS,
462              ClusterStatusPublisher.Publisher.class);
463
464      if (shouldPublish) {
465        if (publisherClass == null) {
466          LOG.warn(HConstants.STATUS_PUBLISHED + " is true, but " +
467              ClusterStatusPublisher.DEFAULT_STATUS_PUBLISHER_CLASS +
468              " is not set - not publishing status");
469        } else {
470          clusterStatusPublisherChore = new ClusterStatusPublisher(this, conf, publisherClass);
471          LOG.debug("Created {}", this.clusterStatusPublisherChore);
472          getChoreService().scheduleChore(clusterStatusPublisherChore);
473        }
474      }
475
476      this.metaRegionLocationCache = new MetaRegionLocationCache(this.zooKeeper);
477      this.activeMasterManager = createActiveMasterManager(zooKeeper, serverName, this);
478
479      cachedClusterId = new CachedClusterId(this, conf);
480    } catch (Throwable t) {
481      // Make sure we log the exception. HMaster is often started via reflection and the
482      // cause of failed startup is lost.
483      LOG.error("Failed construction of Master", t);
484      throw t;
485    }
486  }

502  // Main run loop. Calls through to the regionserver run loop AFTER becoming active Master; will
503  // block in here until then.
504  @Override
505  public void run() {
506    try {
507      Threads.setDaemonThreadRunning(new Thread(() -> {
508        try {
509          int infoPort = putUpJettyServer();
510          startActiveMasterManager(infoPort);
511        } catch (Throwable t) {
512          // Make sure we log the exception.
513          String error = "Failed to become Active Master";
514          LOG.error(error, t);
515          // Abort should have been called already.
516          if (!isAborted()) {
517            abort(error, t);
518          }
519        }
520      }), getName() + ":becomeActiveMaster");
521      // Fall in here even if we have been aborted. Need to run the shutdown services and
522      // the super run call will do this for us.
523      super.run();
524    } finally {
525      if (this.clusterSchemaService != null) {
526        // If on way out, then we are no longer active master.
527        this.clusterSchemaService.stopAsync();
528        try {
529          this.clusterSchemaService.awaitTerminated(
530              getConfiguration().getInt(HBASE_MASTER_WAIT_ON_SERVICE_IN_SECONDS,
531              DEFAULT_HBASE_MASTER_WAIT_ON_SERVICE_IN_SECONDS), TimeUnit.SECONDS);
532        } catch (TimeoutException te) {
533          LOG.warn("Failed shutdown of clusterSchemaService", te);
534        }
535      }
536      this.activeMaster = false;
537    }
538  }

540  // return the actual infoPort, -1 means disable info server.
541  private int putUpJettyServer() throws IOException {
542    if (!conf.getBoolean("hbase.master.infoserver.redirect", true)) {
543      return -1;
544    }
545    final int infoPort = conf.getInt("hbase.master.info.port.orig",
546      HConstants.DEFAULT_MASTER_INFOPORT);
547    // -1 is for disabling info server, so no redirecting
548    if (infoPort < 0 || infoServer == null) {
549      return -1;
550    }
551    if (infoPort == infoServer.getPort()) {
552      // server is already running
553      return infoPort;
554    }
555    final String addr = conf.get("hbase.master.info.bindAddress", "0.0.0.0");
556    if (!Addressing.isLocalAddress(InetAddress.getByName(addr))) {
557      String msg =
558          "Failed to start redirecting jetty server. Address " + addr
559              + " does not belong to this host. Correct configuration parameter: "
560              + "hbase.master.info.bindAddress";
561      LOG.error(msg);
562      throw new IOException(msg);
563    }
564
565    // TODO I'm pretty sure we could just add another binding to the InfoServer run by
566    // the RegionServer and have it run the RedirectServlet instead of standing up
567    // a second entire stack here.
568    masterJettyServer = new Server();
569    final ServerConnector connector = new ServerConnector(masterJettyServer);
570    connector.setHost(addr);
571    connector.setPort(infoPort);
572    masterJettyServer.addConnector(connector);
573    masterJettyServer.setStopAtShutdown(true);
574    masterJettyServer.setHandler(HttpServer.buildGzipHandler(masterJettyServer.getHandler()));
575
576    final String redirectHostname =
577        StringUtils.isBlank(useThisHostnameInstead) ? null : useThisHostnameInstead;
578
579    final MasterRedirectServlet redirect = new MasterRedirectServlet(infoServer, redirectHostname);
580    final WebAppContext context = new WebAppContext(null, "/", null, null, null, null, WebAppContext.NO_SESSIONS);
581    context.addServlet(new ServletHolder(redirect), "/*");
582    context.setServer(masterJettyServer);
583
584    try {
585      masterJettyServer.start();
586    } catch (Exception e) {
587      throw new IOException("Failed to start redirecting jetty server", e);
588    }
589    return connector.getLocalPort();
590  }

759  /**
760   * Finish initialization of HMaster after becoming the primary master.
761   * <p/>
762   * The startup order is a bit complicated but very important, do not change it unless you know
763   * what you are doing.
764   * <ol>
765   * <li>Initialize file system based components - file system manager, wal manager, table
766   * descriptors, etc</li>
767   * <li>Publish cluster id</li>
768   * <li>Here comes the most complicated part - initialize server manager, assignment manager and
769   * region server tracker
770   * <ol type='i'>
771   * <li>Create server manager</li>
772   * <li>Create procedure executor, load the procedures, but do not start workers. We will start it
773   * later after we finish scheduling SCPs to avoid scheduling duplicated SCPs for the same
774   * server</li>
775   * <li>Create assignment manager and start it, load the meta region state, but do not load data
776   * from meta region</li>
777   * <li>Start region server tracker, construct the online servers set and find out dead servers and
778   * schedule SCP for them. The online servers will be constructed by scanning zk, and we will also
779   * scan the wal directory to find out possible live region servers, and the differences between
780   * these two sets are the dead servers</li>
781   * </ol>
782   * </li>
783   * <li>If this is a new deploy, schedule a InitMetaProcedure to initialize meta</li>
784   * <li>Start necessary service threads - balancer, catalog janitor, executor services, and also
785   * the procedure executor, etc. Notice that the balancer must be created first as assignment
786   * manager may use it when assigning regions.</li>
787   * <li>Wait for meta to be initialized if necessary, start table state manager.</li>
788   * <li>Wait for enough region servers to check-in</li>
789   * <li>Let assignment manager load data from meta and construct region states</li>
790   * <li>Start all other things such as chore services, etc</li>
791   * </ol>
792   * <p/>
793   * Notice that now we will not schedule a special procedure to make meta online(unless the first
794   * time where meta has not been created yet), we will rely on SCP to bring meta online.
795   */
796  private void finishActiveMasterInitialization(MonitoredTask status) throws IOException,
797          InterruptedException, KeeperException, ReplicationException {
798    /*
799     * We are active master now... go initialize components we need to run.
800     */
801    status.setStatus("Initializing Master file system");
802
803    this.masterActiveTime = System.currentTimeMillis();
804    // TODO: Do this using Dependency Injection, using PicoContainer, Guice or Spring.
805
806    // always initialize the MemStoreLAB as we use a region to store data in master now, see
807    // localStore.
808    initializeMemStoreChunkCreator();
809    this.fileSystemManager = new MasterFileSystem(conf);
810    this.walManager = new MasterWalManager(this);
811
812    // warm-up HTDs cache on master initialization
813    if (preLoadTableDescriptors) {
814      status.setStatus("Pre-loading table descriptors");
815      this.tableDescriptors.getAll();
816    }
817
818    // Publish cluster ID; set it in Master too. The superclass RegionServer does this later but
819    // only after it has checked in with the Master. At least a few tests ask Master for clusterId
820    // before it has called its run method and before RegionServer has done the reportForDuty.
821    ClusterId clusterId = fileSystemManager.getClusterId();
822    status.setStatus("Publishing Cluster ID " + clusterId + " in ZooKeeper");
823    ZKClusterId.setClusterId(this.zooKeeper, fileSystemManager.getClusterId());
824    this.clusterId = clusterId.toString();
825
826    // Precaution. Put in place the old hbck1 lock file to fence out old hbase1s running their
827    // hbck1s against an hbase2 cluster; it could do damage. To skip this behavior, set
828    // hbase.write.hbck1.lock.file to false.
829    if (this.conf.getBoolean("hbase.write.hbck1.lock.file", true)) {
830      Pair<Path, FSDataOutputStream> result = null;
831      try {
832        result = HBaseFsck.checkAndMarkRunningHbck(this.conf,
833            HBaseFsck.createLockRetryCounterFactory(this.conf).create());
834      } finally {
835        if (result != null) {
836          Closeables.close(result.getSecond(), true);
837        }
838      }
839    }
840
841    status.setStatus("Initialize ServerManager and schedule SCP for crash servers");
842    // The below two managers must be created before loading procedures, as they will be used during
843    // loading.
844    this.serverManager = createServerManager(this);
845    this.syncReplicationReplayWALManager = new SyncReplicationReplayWALManager(this);
846    if (!conf.getBoolean(HBASE_SPLIT_WAL_COORDINATED_BY_ZK,
847      DEFAULT_HBASE_SPLIT_COORDINATED_BY_ZK)) {
848      this.splitWALManager = new SplitWALManager(this);
849    }
850
851    // initialize master local region
852    masterRegion = MasterRegionFactory.create(this);
853    createProcedureExecutor();
854    Map<Class<?>, List<Procedure<MasterProcedureEnv>>> procsByType =
855      procedureExecutor.getActiveProceduresNoCopy().stream()
856        .collect(Collectors.groupingBy(p -> p.getClass()));
857
858    // Create Assignment Manager
859    this.assignmentManager = createAssignmentManager(this);
860    this.assignmentManager.start();
861    // TODO: TRSP can perform as the sub procedure for other procedures, so even if it is marked as
862    // completed, it could still be in the procedure list. This is a bit strange but is another
863    // story, need to verify the implementation for ProcedureExecutor and ProcedureStore.
864    List<TransitRegionStateProcedure> ritList =
865      procsByType.getOrDefault(TransitRegionStateProcedure.class, Collections.emptyList()).stream()
866        .filter(p -> !p.isFinished()).map(p -> (TransitRegionStateProcedure) p)
867        .collect(Collectors.toList());
868    this.assignmentManager.setupRIT(ritList);
869
870    // Start RegionServerTracker with listing of servers found with exiting SCPs -- these should
871    // be registered in the deadServers set -- and with the list of servernames out on the
872    // filesystem that COULD BE 'alive' (we'll schedule SCPs for each and let SCP figure it out).
873    // We also pass dirs that are already 'splitting'... so we can do some checks down in tracker.
874    // TODO: Generate the splitting and live Set in one pass instead of two as we currently do.
875    this.regionServerTracker = new RegionServerTracker(zooKeeper, this, this.serverManager);
876    this.regionServerTracker.start(
877      procsByType.getOrDefault(ServerCrashProcedure.class, Collections.emptyList()).stream()
878        .map(p -> (ServerCrashProcedure) p).map(p -> p.getServerName()).collect(Collectors.toSet()),
879      walManager.getLiveServersFromWALDir(), walManager.getSplittingServersFromWALDir());
880    // This manager must be accessed AFTER hbase:meta is confirmed on line..
881    this.tableStateManager = new TableStateManager(this);
882
883    status.setStatus("Initializing ZK system trackers");
884    initializeZKBasedSystemTrackers();
885    status.setStatus("Loading last flushed sequence id of regions");
886    try {
887      this.serverManager.loadLastFlushedSequenceIds();
888    } catch (IOException e) {
889      LOG.info("Failed to load last flushed sequence id of regions"
890          + " from file system", e);
891    }
892    // Set ourselves as active Master now our claim has succeeded up in zk.
893    this.activeMaster = true;
894
895    // Start the Zombie master detector after setting master as active, see HBASE-21535
896    Thread zombieDetector = new Thread(new MasterInitializationMonitor(this),
897        "ActiveMasterInitializationMonitor-" + System.currentTimeMillis());
898    zombieDetector.setDaemon(true);
899    zombieDetector.start();
900
901    if (!maintenanceMode) {
902      // Add the Observer to delete quotas on table deletion before starting all CPs by
903      // default with quota support, avoiding if user specifically asks to not load this Observer.
904      if (QuotaUtil.isQuotaEnabled(conf)) {
905        updateConfigurationForQuotasObserver(conf);
906      }
907      // initialize master side coprocessors before we start handling requests
908      status.setStatus("Initializing master coprocessors");
909      this.cpHost = new MasterCoprocessorHost(this, this.conf);
910    }
911
912    // Checking if meta needs initializing.
913    status.setStatus("Initializing meta table if this is a new deploy");
914    InitMetaProcedure initMetaProc = null;
915    // Print out state of hbase:meta on startup; helps debugging.
916    if (!this.assignmentManager.getRegionStates().hasTableRegionStates(TableName.META_TABLE_NAME)) {
917      Optional<InitMetaProcedure> optProc = procedureExecutor.getProcedures().stream()
918        .filter(p -> p instanceof InitMetaProcedure).map(o -> (InitMetaProcedure) o).findAny();
919      initMetaProc = optProc.orElseGet(() -> {
920        // schedule an init meta procedure if meta has not been deployed yet
921        InitMetaProcedure temp = new InitMetaProcedure();
922        procedureExecutor.submitProcedure(temp);
923        return temp;
924      });
925    }
926
927    // initialize load balancer
928    this.balancer.setMasterServices(this);
929    this.balancer.setClusterMetrics(getClusterMetricsWithoutCoprocessor());
930    this.balancer.initialize();
931
932    // start up all service threads.
933    status.setStatus("Initializing master service threads");
934    startServiceThreads();
935    // wait meta to be initialized after we start procedure executor
936    if (initMetaProc != null) {
937      initMetaProc.await();
938    }
939    // Wake up this server to check in
940    sleeper.skipSleepCycle();
941
942    // Wait for region servers to report in.
943    // With this as part of master initialization, it precludes our being able to start a single
944    // server that is both Master and RegionServer. Needs more thought. TODO.
945    String statusStr = "Wait for region servers to report in";
946    status.setStatus(statusStr);
947    LOG.info(Objects.toString(status));
948    waitForRegionServers(status);
949
950    // Check if master is shutting down because issue initializing regionservers or balancer.
951    if (isStopped()) {
952      return;
953    }
954
955    status.setStatus("Starting assignment manager");
956    // FIRST HBASE:META READ!!!!
957    // The below cannot make progress w/o hbase:meta being online.
958    // This is the FIRST attempt at going to hbase:meta. Meta on-lining is going on in background
959    // as procedures run -- in particular SCPs for crashed servers... One should put up hbase:meta
960    // if it is down. It may take a while to come online. So, wait here until meta if for sure
961    // available. That's what waitForMetaOnline does.
962    if (!waitForMetaOnline()) {
963      return;
964    }
965    this.assignmentManager.joinCluster();
966    // The below depends on hbase:meta being online.
967    this.assignmentManager.processOfflineRegions();
968    // this must be called after the above processOfflineRegions to prevent race
969    this.assignmentManager.wakeMetaLoadedEvent();
970
971    // for migrating from a version without HBASE-25099, and also for honoring the configuration
972    // first.
973    if (conf.get(HConstants.META_REPLICAS_NUM) != null) {
974      int replicasNumInConf =
975        conf.getInt(HConstants.META_REPLICAS_NUM, HConstants.DEFAULT_META_REPLICA_NUM);
976      TableDescriptor metaDesc = tableDescriptors.get(TableName.META_TABLE_NAME);
977      if (metaDesc.getRegionReplication() != replicasNumInConf) {
978        // it is possible that we already have some replicas before upgrading, so we must set the
979        // region replication number in meta TableDescriptor directly first, without creating a
980        // ModifyTableProcedure, otherwise it may cause a double assign for the meta replicas.
981        int existingReplicasCount =
982          assignmentManager.getRegionStates().getRegionsOfTable(TableName.META_TABLE_NAME).size();
983        if (existingReplicasCount > metaDesc.getRegionReplication()) {
984          LOG.info("Update replica count of hbase:meta from {}(in TableDescriptor)" +
985            " to {}(existing ZNodes)", metaDesc.getRegionReplication(), existingReplicasCount);
986          metaDesc = TableDescriptorBuilder.newBuilder(metaDesc)
987            .setRegionReplication(existingReplicasCount).build();
988          tableDescriptors.update(metaDesc);
989        }
990        // check again, and issue a ModifyTableProcedure if needed
991        if (metaDesc.getRegionReplication() != replicasNumInConf) {
992          LOG.info(
993            "The {} config is {} while the replica count in TableDescriptor is {}" +
994              " for hbase:meta, altering...",
995            HConstants.META_REPLICAS_NUM, replicasNumInConf, metaDesc.getRegionReplication());
996          procedureExecutor.submitProcedure(new ModifyTableProcedure(
997            procedureExecutor.getEnvironment(), TableDescriptorBuilder.newBuilder(metaDesc)
998              .setRegionReplication(replicasNumInConf).build(),
999            null, metaDesc, false));
1000        }
1001      }
1002    }
1003    // Initialize after meta is up as below scans meta
1004    if (getFavoredNodesManager() != null && !maintenanceMode) {
1005      SnapshotOfRegionAssignmentFromMeta snapshotOfRegionAssignment =
1006          new SnapshotOfRegionAssignmentFromMeta(getConnection());
1007      snapshotOfRegionAssignment.initialize();
1008      getFavoredNodesManager().initialize(snapshotOfRegionAssignment);
1009    }
1010
1011    // set cluster status again after user regions are assigned
1012    this.balancer.setClusterMetrics(getClusterMetricsWithoutCoprocessor());
1013
1014    // Start balancer and meta catalog janitor after meta and regions have been assigned.
1015    status.setStatus("Starting balancer and catalog janitor");
1016    this.clusterStatusChore = new ClusterStatusChore(this, balancer);
1017    getChoreService().scheduleChore(clusterStatusChore);
1018    this.balancerChore = new BalancerChore(this);
1019    getChoreService().scheduleChore(balancerChore);
1020    if (regionNormalizerManager != null) {
1021      getChoreService().scheduleChore(regionNormalizerManager.getRegionNormalizerChore());
1022    }
1023    this.catalogJanitorChore = new CatalogJanitor(this);
1024    getChoreService().scheduleChore(catalogJanitorChore);
1025    this.hbckChore = new HbckChore(this);
1026    getChoreService().scheduleChore(hbckChore);
1027    this.serverManager.startChore();
1028
1029    // Only for rolling upgrade, where we need to migrate the data in namespace table to meta table.
1030    if (!waitForNamespaceOnline()) {
1031      return;
1032    }
1033    status.setStatus("Starting cluster schema service");
1034    initClusterSchemaService();
1035
1036    if (this.cpHost != null) {
1037      try {
1038        this.cpHost.preMasterInitialization();
1039      } catch (IOException e) {
1040        LOG.error("Coprocessor preMasterInitialization() hook failed", e);
1041      }
1042    }
1043
1044    status.markComplete("Initialization successful");
1045    LOG.info(String.format("Master has completed initialization %.3fsec",
1046       (System.currentTimeMillis() - masterActiveTime) / 1000.0f));
1047    this.masterFinishedInitializationTime = System.currentTimeMillis();
1048    configurationManager.registerObserver(this.balancer);
1049    configurationManager.registerObserver(this.cleanerPool);
1050    configurationManager.registerObserver(this.hfileCleaner);
1051    configurationManager.registerObserver(this.logCleaner);
1052    configurationManager.registerObserver(this.regionsRecoveryConfigManager);
1053    // Set master as 'initialized'.
1054    setInitialized(true);
1055
1056    if (maintenanceMode) {
1057      LOG.info("Detected repair mode, skipping final initialization steps.");
1058      return;
1059    }
1060
1061    assignmentManager.checkIfShouldMoveSystemRegionAsync();
1062    status.setStatus("Starting quota manager");
1063    initQuotaManager();
1064    if (QuotaUtil.isQuotaEnabled(conf)) {
1065      // Create the quota snapshot notifier
1066      spaceQuotaSnapshotNotifier = createQuotaSnapshotNotifier();
1067      spaceQuotaSnapshotNotifier.initialize(getConnection());
1068      this.quotaObserverChore = new QuotaObserverChore(this, getMasterMetrics());
1069      // Start the chore to read the region FS space reports and act on them
1070      getChoreService().scheduleChore(quotaObserverChore);
1071
1072      this.snapshotQuotaChore = new SnapshotQuotaObserverChore(this, getMasterMetrics());
1073      // Start the chore to read snapshots and add their usage to table/NS quotas
1074      getChoreService().scheduleChore(snapshotQuotaChore);
1075    }
1076    final SlowLogMasterService slowLogMasterService = new SlowLogMasterService(conf, this);
1077    slowLogMasterService.init();
1078
1079    // clear the dead servers with same host name and port of online server because we are not
1080    // removing dead server with same hostname and port of rs which is trying to check in before
1081    // master initialization. See HBASE-5916.

1125  /**
1126   * @return True if region is online and scannable else false if an error or shutdown (Otherwise
1127   *   we just block in here holding up all forward-progess).
1128   */
1129  private boolean isRegionOnline(RegionInfo ri) {
1130    RetryCounter rc = null;
1131    while (!isStopped()) {
1132      RegionState rs = this.assignmentManager.getRegionStates().getRegionState(ri);
1133      if (rs.isOpened()) {
1134        if (this.getServerManager().isServerOnline(rs.getServerName())) {
1135          return true;
1136        }
1137      }
1138      // Region is not OPEN.
1139      Optional<Procedure<MasterProcedureEnv>> optProc = this.procedureExecutor.getProcedures().
1140          stream().filter(p -> p instanceof ServerCrashProcedure).findAny();
1141      // TODO: Add a page to refguide on how to do repair. Have this log message point to it.
1142      // Page will talk about loss of edits, how to schedule at least the meta WAL recovery, and
1143      // then how to assign including how to break region lock if one held.
1144      LOG.warn("{} is NOT online; state={}; ServerCrashProcedures={}. Master startup cannot " +
1145          "progress, in holding-pattern until region onlined.",
1146          ri.getRegionNameAsString(), rs, optProc.isPresent());
1147      // Check once-a-minute.
1148      if (rc == null) {
1149        rc = new RetryCounterFactory(Integer.MAX_VALUE, 1000, 60_000).create();
1150      }
1151      Threads.sleep(rc.getBackoffTimeAndIncrementAttempts());
1152    }
1153    return false;
1154  }

1158   * <p/>
1159   * This is for rolling upgrading, later we will migrate the data in ns table to the ns family of
1160   * meta table. And if this is a new cluster, this method will return immediately as there will be

1168      // this means we have already migrated the data and disabled or deleted the namespace table,

1170      return true;
1171    }

1607  /**
1608   * It first sleep to the next balance plan start time. Meanwhile, throttling by the max
1609   * number regions in transition to protect availability.
1610   * @param nextBalanceStartTime The next balance plan start time
1611   * @param maxRegionsInTransition max number of regions in transition
1612   * @param cutoffTime when to exit balancer
1613   */
1614  private void balanceThrottling(long nextBalanceStartTime, int maxRegionsInTransition,
1615      long cutoffTime) {
1616    boolean interrupted = false;
1617
1618    // Sleep to next balance plan start time
1619    // But if there are zero regions in transition, it can skip sleep to speed up.
1620    while (!interrupted && System.currentTimeMillis() < nextBalanceStartTime
1621        && this.assignmentManager.getRegionStates().hasRegionsInTransition()) {
1622      try {
1623        Thread.sleep(100);
1624      } catch (InterruptedException ie) {
1625        interrupted = true;
1626      }
1627    }
1628
1629    // Throttling by max number regions in transition
1630    while (!interrupted
1631        && maxRegionsInTransition > 0
1632        && this.assignmentManager.getRegionStates().getRegionsInTransitionCount()
1633        >= maxRegionsInTransition && System.currentTimeMillis() <= cutoffTime) {
1634      try {
1635        // sleep if the number of regions in transition exceeds the limit
1636        Thread.sleep(100);
1637      } catch (InterruptedException ie) {
1638        interrupted = true;
1639      }
1640    }
1641
1642    if (interrupted) Thread.currentThread().interrupt();
1643  }

1672  public boolean balance(boolean force) throws IOException {
1673    if (loadBalancerTracker == null || !loadBalancerTracker.isBalancerOn()) {
1674      return false;
1675    }
1676    if (skipRegionManagementAction("balancer")) {
1677      return false;
1678    }
1679
1680    synchronized (this.balancer) {
1681        // Only allow one balance run at at time.
1682      if (this.assignmentManager.hasRegionsInTransition()) {
1683        List<RegionStateNode> regionsInTransition = assignmentManager.getRegionsInTransition();
1684        // if hbase:meta region is in transition, result of assignment cannot be recorded
1685        // ignore the force flag in that case
1686        boolean metaInTransition = assignmentManager.isMetaRegionInTransition();
1687        String prefix = force && !metaInTransition ? "R" : "Not r";
1688        List<RegionStateNode> toPrint = regionsInTransition;
1689        int max = 5;
1690        boolean truncated = false;
1691        if (regionsInTransition.size() > max) {
1692          toPrint = regionsInTransition.subList(0, max);
1693          truncated = true;
1694        }
1695        LOG.info(prefix + " not running balancer because " + regionsInTransition.size() +
1696          " region(s) in transition: " + toPrint + (truncated? "(truncated list)": ""));
1697        if (!force || metaInTransition) return false;
1698      }
1699      if (this.serverManager.areDeadServersInProgress()) {
1700        LOG.info("Not running balancer because processing dead regionserver(s): " +
1701          this.serverManager.getDeadServers());
1702        return false;
1703      }
1704
1705      if (this.cpHost != null) {
1706        try {
1707          if (this.cpHost.preBalance()) {
1708            LOG.debug("Coprocessor bypassing balancer request");
1709            return false;
1710          }
1711        } catch (IOException ioe) {
1712          LOG.error("Error invoking master coprocessor preBalance()", ioe);
1713          return false;
1714        }
1715      }
1716
1717      Map<TableName, Map<ServerName, List<RegionInfo>>> assignments =
1718        this.assignmentManager.getRegionStates()
1719          .getAssignmentsForBalancer(tableStateManager, this.serverManager.getOnlineServersList());
1720      for (Map<ServerName, List<RegionInfo>> serverMap : assignments.values()) {
1721        serverMap.keySet().removeAll(this.serverManager.getDrainingServersList());
1722      }
1723
1724      //Give the balancer the current cluster state.
1725      this.balancer.setClusterMetrics(getClusterMetricsWithoutCoprocessor());
1726
1727      List<RegionPlan> plans = this.balancer.balanceCluster(assignments);
1728
1729      if (skipRegionManagementAction("balancer")) {
1730        // make one last check that the cluster isn't shutting down before proceeding.
1731        return false;
1732      }
1733
1734      List<RegionPlan> sucRPs = executeRegionPlansWithThrottling(plans);
1735
1736      if (this.cpHost != null) {
1737        try {
1738          this.cpHost.postBalance(sucRPs);
1739        } catch (IOException ioe) {
1740          // balancing already succeeded so don't change the result
1741          LOG.error("Error invoking master coprocessor postBalance()", ioe);
1742        }
1743      }
1744    }
1745    // If LoadBalancer did not generate any plans, it means the cluster is already balanced.
1746    // Return true indicating a success.
1747    return true;
1748  }

1750  /**
1751   * Execute region plans with throttling
1752   * @param plans to execute
1753   * @return succeeded plans
1754   */
1755  public List<RegionPlan> executeRegionPlansWithThrottling(List<RegionPlan> plans) {
1756    List<RegionPlan> successRegionPlans = new ArrayList<>();
1757    int maxRegionsInTransition = getMaxRegionsInTransition();
1758    long balanceStartTime = System.currentTimeMillis();
1759    long cutoffTime = balanceStartTime + this.maxBalancingTime;
1760    int rpCount = 0;  // number of RegionPlans balanced so far
1761    if (plans != null && !plans.isEmpty()) {
1762      int balanceInterval = this.maxBalancingTime / plans.size();
1763      LOG.info("Balancer plans size is " + plans.size() + ", the balance interval is "
1764          + balanceInterval + " ms, and the max number regions in transition is "
1765          + maxRegionsInTransition);
1766
1767      for (RegionPlan plan: plans) {
1768        LOG.info("balance " + plan);
1769        //TODO: bulk assign
1770        try {
1771          this.assignmentManager.balance(plan);
1772        } catch (HBaseIOException hioe) {
1773          //should ignore failed plans here, avoiding the whole balance plans be aborted
1774          //later calls of balance() can fetch up the failed and skipped plans
1775          LOG.warn("Failed balance plan {}, skipping...", plan, hioe);
1776        }
1777        //rpCount records balance plans processed, does not care if a plan succeeds
1778        rpCount++;
1779        successRegionPlans.add(plan);
1780
1781        if (this.maxBalancingTime > 0) {
1782          balanceThrottling(balanceStartTime + rpCount * balanceInterval, maxRegionsInTransition,
1783            cutoffTime);
1784        }
1785
1786        // if performing next balance exceeds cutoff time, exit the loop
1787        if (this.maxBalancingTime > 0 && rpCount < plans.size()
1788          && System.currentTimeMillis() > cutoffTime) {
1789          // TODO: After balance, there should not be a cutoff time (keeping it as
1790          // a security net for now)
1791          LOG.debug("No more balancing till next balance run; maxBalanceTime="
1792              + this.maxBalancingTime);
1793          break;
1794        }
1795      }
1796    }
1797    return successRegionPlans;
1798  }

1931  public void move(final byte[] encodedRegionName, byte[] destServerName) throws IOException {
1932    RegionState regionState = assignmentManager.getRegionStates().
1933      getRegionState(Bytes.toString(encodedRegionName));
1934
1935    RegionInfo hri;
1936    if (regionState != null) {
1937      hri = regionState.getRegion();
1938    } else {
1939      throw new UnknownRegionException(Bytes.toStringBinary(encodedRegionName));
1940    }
1941
1942    ServerName dest;
1943    List<ServerName> exclude = hri.getTable().isSystemTable() ? assignmentManager.getExcludedServersForSystemTable()
1944        : new ArrayList<>(1);
1945    if (destServerName != null && exclude.contains(ServerName.valueOf(Bytes.toString(destServerName)))) {
1946      LOG.info(
1947          Bytes.toString(encodedRegionName) + " can not move to " + Bytes.toString(destServerName)
1948              + " because the server is in exclude list");
1949      destServerName = null;
1950    }
1951    if (destServerName == null || destServerName.length == 0) {
1952      LOG.info("Passed destination servername is null/empty so " +
1953        "choosing a server at random");
1954      exclude.add(regionState.getServerName());
1955      final List<ServerName> destServers = this.serverManager.createDestinationServersList(exclude);
1956      dest = balancer.randomAssignment(hri, destServers);
1957      if (dest == null) {
1958        LOG.debug("Unable to determine a plan to assign " + hri);
1959        return;
1960      }
1961    } else {
1962      ServerName candidate = ServerName.valueOf(Bytes.toString(destServerName));
1963      dest = balancer.randomAssignment(hri, Lists.newArrayList(candidate));
1964      if (dest == null) {
1965        LOG.debug("Unable to determine a plan to assign " + hri);
1966        return;
1967      }
1968      // TODO: deal with table on master for rs group.
1969      if (dest.equals(serverName)) {
1970        // To avoid unnecessary region moving later by balancer. Don't put user
1971        // regions on master.
1972        LOG.debug("Skipping move of region " + hri.getRegionNameAsString() +
1973          " to avoid unnecessary region moving later by load balancer," +
1974          " because it should not be on master");
1975        return;
1976      }
1977    }
1978
1979    if (dest.equals(regionState.getServerName())) {
1980      LOG.debug("Skipping move of region " + hri.getRegionNameAsString()
1981        + " because region already assigned to the same server " + dest + ".");
1982      return;
1983    }
1984
1985    // Now we can do the move
1986    RegionPlan rp = new RegionPlan(hri, regionState.getServerName(), dest);
1987    assert rp.getDestination() != null: rp.toString() + " " + dest;
1988
1989    try {
1990      checkInitialized();
1991      if (this.cpHost != null) {
1992        this.cpHost.preMove(hri, rp.getSource(), rp.getDestination());
1993      }
1994
1995      TransitRegionStateProcedure proc =
1996        this.assignmentManager.createMoveRegionProcedure(rp.getRegionInfo(), rp.getDestination());
1997      // Warmup the region on the destination before initiating the move.
1998      // A region server could reject the close request because it either does not
1999      // have the specified region or the region is being split.
2000      warmUpRegion(rp.getDestination(), hri);
2001
2002      LOG.info(getClientIdAuditPrefix() + " move " + rp + ", running balancer");
2003      Future<byte[]> future = ProcedureSyncWait.submitProcedure(this.procedureExecutor, proc);
2004      try {
2005        // Is this going to work? Will we throw exception on error?
2006        // TODO: CompletableFuture rather than this stunted Future.
2007        future.get();
2008      } catch (InterruptedException | ExecutionException e) {
2009        throw new HBaseIOException(e);
2010      }
2011      if (this.cpHost != null) {
2012        this.cpHost.postMove(hri, rp.getSource(), rp.getDestination());
2013      }
2014    } catch (IOException ioe) {
2015      if (ioe instanceof HBaseIOException) {
2016        throw (HBaseIOException)ioe;
2017      }
2018      throw new HBaseIOException(ioe);
2019    }

2088  private void startActiveMasterManager(int infoPort) throws KeeperException {
2089    String backupZNode = ZNodePaths.joinZNode(
2090      zooKeeper.getZNodePaths().backupMasterAddressesZNode, serverName.toString());
2091    /*
2092    * Add a ZNode for ourselves in the backup master directory since we
2093    * may not become the active master. If so, we want the actual active
2094    * master to know we are backup masters, so that it won't assign
2095    * regions to us if so configured.
2096    *
2097    * If we become the active master later, ActiveMasterManager will delete
2098    * this node explicitly.  If we crash before then, ZooKeeper will delete
2099    * this node for us since it is ephemeral.
2100    */
2101    LOG.info("Adding backup master ZNode " + backupZNode);
2102    if (!MasterAddressTracker.setMasterAddress(zooKeeper, backupZNode, serverName, infoPort)) {
2103      LOG.warn("Failed create of " + backupZNode + " by " + serverName);
2104    }
2105    this.activeMasterManager.setInfoPort(infoPort);
2106    int timeout = conf.getInt(HConstants.ZK_SESSION_TIMEOUT, HConstants.DEFAULT_ZK_SESSION_TIMEOUT);
2107    // If we're a backup master, stall until a primary to write this address
2108    if (conf.getBoolean(HConstants.MASTER_TYPE_BACKUP, HConstants.DEFAULT_MASTER_TYPE_BACKUP)) {
2109      LOG.debug("HMaster started in backup mode. Stalling until master znode is written.");
2110      // This will only be a minute or so while the cluster starts up,
2111      // so don't worry about setting watches on the parent znode
2112      while (!activeMasterManager.hasActiveMaster()) {
2113        LOG.debug("Waiting for master address and cluster state znode to be written.");
2114        Threads.sleep(timeout);
2115      }
2116    }
2117    MonitoredTask status = TaskMonitor.get().createStatus("Master startup");
2118    status.setDescription("Master startup");
2119    try {
2120      if (activeMasterManager.blockUntilBecomingActiveMaster(timeout, status)) {
2121        finishActiveMasterInitialization(status);
2122      }
2123    } catch (Throwable t) {
2124      status.setStatus("Failed to become active: " + t.getMessage());
2125      LOG.error(HBaseMarkers.FATAL, "Failed to become active master", t);
2126      // HBASE-5680: Likely hadoop23 vs hadoop 20.x/1.x incompatibility
2127      if (t instanceof NoClassDefFoundError && t.getMessage().
2128          contains("org/apache/hadoop/hdfs/protocol/HdfsConstants$SafeModeAction")) {
2129        // improved error message for this special case
2130        abort("HBase is having a problem with its Hadoop jars.  You may need to recompile " +
2131          "HBase against Hadoop version " + org.apache.hadoop.util.VersionInfo.getVersion() +
2132          " or change your hadoop jars to start properly", t);
2133      } else {
2134        abort("Unhandled exception. Starting shutdown.", t);
2135      }
2136    } finally {
2137      status.cleanup();
2138    }
2139  }

2704  /**
2705   * Shutdown the cluster.
2706   * Master runs a coordinated stop of all RegionServers and then itself.
2707   */
2708  public void shutdown() throws IOException {
2709    if (cpHost != null) {
2710      cpHost.preShutdown();
2711    }
2712
2713    // Tell the servermanager cluster shutdown has been called. This makes it so when Master is
2714    // last running server, it'll stop itself. Next, we broadcast the cluster shutdown by setting
2715    // the cluster status as down. RegionServers will notice this change in state and will start
2716    // shutting themselves down. When last has exited, Master can go down.
2717    if (this.serverManager != null) {
2718      this.serverManager.shutdownCluster();
2719    }
2720    if (this.clusterStatusTracker != null) {
2721      try {
2722        this.clusterStatusTracker.setClusterDown();
2723      } catch (KeeperException e) {
2724        LOG.error("ZooKeeper exception trying to set cluster as down in ZK", e);
2725      }
2726    }
2727    // Stop the procedure executor. Will stop any ongoing assign, unassign, server crash etc.,
2728    // processing so we can go down.
2729    if (this.procedureExecutor != null) {
2730      this.procedureExecutor.stop();
2731    }
2732    // Shutdown our cluster connection. This will kill any hosted RPCs that might be going on;
2733    // this is what we want especially if the Master is in startup phase doing call outs to
2734    // hbase:meta, etc. when cluster is down. Without ths connection close, we'd have to wait on
2735    // the rpc to timeout.
2736    if (this.asyncClusterConnection != null) {
2737      this.asyncClusterConnection.close();
2738    }
2739  }

3645        new HashMap<>(peerList.size());
3646    peerList.stream()
3647        .forEach(peer -> replicationLoadSourceMap.put(peer.getPeerId(), new ArrayList<>()));
3648    for (ServerName serverName : serverNames) {

3650          getServerManager().getLoad(serverName).getReplicationLoadSourceList();

3652        List<Pair<ServerName, ReplicationLoadSource>> replicationLoadSourceList =
3653          replicationLoadSourceMap.get(replicationLoadSource.getPeerID());
3654        if (replicationLoadSourceList == null) {
3655          LOG.debug("{} does not exist, but it exists "
3656            + "in znode(/hbase/replication/rs). when the rs restarts, peerId is deleted, so "
3657            + "we just need to ignore it", replicationLoadSource.getPeerID());
3658          continue;
3659        }
3660        replicationLoadSourceList.add(new Pair<>(serverName, replicationLoadSource));
3661      }
3662    }
3663    for (List<Pair<ServerName, ReplicationLoadSource>> loads : replicationLoadSourceMap.values()) {
3664      if (loads.size() > 0) {
3665        loads.sort(Comparator.comparingLong(load -> (-1) * load.getSecond().getReplicationLag()));
3666      }
3667    }
3668    return replicationLoadSourceMap;

3743   * @param tableName The table name
3744   * @return CompactionState Compaction state of the table
3745   */
3746  public CompactionState getCompactionState(final TableName tableName) {
3747    CompactionState compactionState = CompactionState.NONE;
3748    try {
3749      List<RegionInfo> regions =
3750        assignmentManager.getRegionStates().getRegionsOfTable(tableName);
3751      for (RegionInfo regionInfo : regions) {
3752        ServerName serverName =
3753          assignmentManager.getRegionStates().getRegionServerOfRegion(regionInfo);
3754        if (serverName == null) {
3755          continue;
3756        }
3757        ServerMetrics sl = serverManager.getLoad(serverName);
3758        if (sl == null) {
3759          continue;
3760        }
3761        RegionMetrics regionMetrics = sl.getRegionMetrics().get(regionInfo.getRegionName());
3762        if (regionMetrics.getCompactionState() == CompactionState.MAJOR) {
3763          if (compactionState == CompactionState.MINOR) {
3764            compactionState = CompactionState.MAJOR_AND_MINOR;
3765          } else {
3766            compactionState = CompactionState.MAJOR;
3767          }
3768        } else if (regionMetrics.getCompactionState() == CompactionState.MINOR) {
3769          if (compactionState == CompactionState.MAJOR) {
3770            compactionState = CompactionState.MAJOR_AND_MINOR;
3771          } else {
3772            compactionState = CompactionState.MINOR;
3773          }
3774        }
3775      }
3776    } catch (Exception e) {
3777      compactionState = null;
3778      LOG.error("Exception when get compaction state for " + tableName.getNameAsString(), e);
3779    }
3780    return compactionState;
3781  }
