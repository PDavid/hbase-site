061/**
062 * A procedure for migrating replication queue data from zookeeper to hbase:replication table.

087  private ProcedureSuspendedException suspend(Configuration conf, LongConsumer backoffConsumer)
088    throws ProcedureSuspendedException {
089    if (retryCounter == null) {
090      retryCounter = ProcedureUtil.createRetryCounter(conf);
091    }
092    long backoff = retryCounter.getBackoffTimeAndIncrementAttempts();
093    backoffConsumer.accept(backoff);
094    throw suspend(Math.toIntExact(backoff), true);
095  }
096
097  private void resetRetry() {
098    retryCounter = null;
099  }
100
101  private ExecutorService getExecutorService() {
102    if (executor == null) {
103      executor = Executors.newCachedThreadPool(new ThreadFactoryBuilder()
104        .setNameFormat(getClass().getSimpleName() + "-%d").setDaemon(true).build());
105    }
106    return executor;
107  }
108
109  private void shutdownExecutorService() {
110    if (executor != null) {
111      executor.shutdown();
112      executor = null;
113    }
114  }
115
116  private void disableReplicationLogCleaner(MasterProcedureEnv env)
117    throws ProcedureSuspendedException {
118    if (!env.getMasterServices().getReplicationLogCleanerBarrier().disable()) {
119      // it is not likely that we can reach here as we will schedule this procedure immediately
120      // after master restarting, where ReplicationLogCleaner should have not started its first run
121      // yet. But anyway, let's make the code more robust. And it is safe to wait a bit here since
122      // there will be no data in the new replication queue storage before we execute this procedure
123      // so ReplicationLogCleaner will quit immediately without doing anything.
124      throw suspend(env.getMasterConfiguration(),
125        backoff -> LOG.info(
126          "Can not disable replication log cleaner, sleep {} secs and retry later",
127          backoff / 1000));
128    }
129    resetRetry();
130  }
131
132  private void enableReplicationLogCleaner(MasterProcedureEnv env) {
133    env.getMasterServices().getReplicationLogCleanerBarrier().enable();
134  }
135
136  private void waitUntilNoPeerProcedure(MasterProcedureEnv env) throws ProcedureSuspendedException {
137    long peerProcCount;
138    try {
139      peerProcCount = env.getMasterServices().getProcedures().stream()
140        .filter(p -> p instanceof PeerProcedureInterface).filter(p -> !p.isFinished()).count();
141    } catch (IOException e) {
142      throw suspend(env.getMasterConfiguration(),
143        backoff -> LOG.warn("failed to check peer procedure status, sleep {} secs and retry later",
144          backoff / 1000, e));
145    }
146    if (peerProcCount > 0) {
147      throw suspend(env.getMasterConfiguration(),
148        backoff -> LOG.info(
149          "There are still {} pending peer procedures, sleep {} secs and retry later",
150          peerProcCount, backoff / 1000));
151    }
152    resetRetry();
153    LOG.info("No pending peer procedures found, continue...");
154  }

156  @Override
157  protected Flow executeFromState(MasterProcedureEnv env,
158    MigrateReplicationQueueFromZkToTableState state)
159    throws ProcedureSuspendedException, ProcedureYieldException, InterruptedException {
160    switch (state) {
161      case MIGRATE_REPLICATION_QUEUE_FROM_ZK_TO_TABLE_DISABLE_CLEANER:
162        disableReplicationLogCleaner(env);
163        setNextState(MIGRATE_REPLICATION_QUEUE_FROM_ZK_TO_TABLE_PREPARE);
164        return Flow.HAS_MORE_STATE;
165      case MIGRATE_REPLICATION_QUEUE_FROM_ZK_TO_TABLE_PREPARE:
166        waitUntilNoPeerProcedure(env);
167        List<ReplicationPeerDescription> peers = env.getReplicationPeerManager().listPeers(null);
168        if (peers.isEmpty()) {
169          LOG.info("No active replication peer found, delete old replication queue data and quit");
170          ZKReplicationQueueStorageForMigration oldStorage =
171            new ZKReplicationQueueStorageForMigration(env.getMasterServices().getZooKeeper(),
172              env.getMasterConfiguration());
173          try {
174            oldStorage.deleteAllData();
175          } catch (KeeperException e) {
176            throw suspend(env.getMasterConfiguration(),
177              backoff -> LOG.warn(
178                "failed to delete old replication queue data, sleep {} secs and retry later",
179                backoff / 1000, e));
180          }
181          setNextState(MIGRATE_REPLICATION_QUEUE_FROM_ZK_TO_TABLE_ENABLE_CLEANER);
182          return Flow.HAS_MORE_STATE;
183        }
184        // here we do not care the peers which have already been disabled, as later we do not need
185        // to enable them
186        disabledPeerIds = peers.stream().filter(ReplicationPeerDescription::isEnabled)
187          .map(ReplicationPeerDescription::getPeerId).collect(Collectors.toList());
188        setNextState(MIGRATE_REPLICATION_QUEUE_FROM_ZK_TO_TABLE_DISABLE_PEER);
189        resetRetry();
190        return Flow.HAS_MORE_STATE;
191      case MIGRATE_REPLICATION_QUEUE_FROM_ZK_TO_TABLE_DISABLE_PEER:
192        for (String peerId : disabledPeerIds) {
193          addChildProcedure(new DisablePeerProcedure(peerId));
194        }
195        setNextState(MIGRATE_REPLICATION_QUEUE_FROM_ZK_TO_TABLE_MIGRATE);
196        return Flow.HAS_MORE_STATE;
197      case MIGRATE_REPLICATION_QUEUE_FROM_ZK_TO_TABLE_MIGRATE:
198        if (future != null) {
199          // should have finished when we arrive here
200          assert future.isDone();
201          try {
202            future.get();
203          } catch (Exception e) {
204            future = null;
205            throw suspend(env.getMasterConfiguration(),
206              backoff -> LOG.warn("failed to migrate queue data, sleep {} secs and retry later",
207                backoff / 1000, e));
208          }
209          shutdownExecutorService();
210          setNextState(MIGRATE_REPLICATION_QUEUE_FROM_ZK_TO_TABLE_WAIT_UPGRADING);
211          resetRetry();
212          return Flow.HAS_MORE_STATE;
213        }
214        future = env.getReplicationPeerManager()
215          .migrateQueuesFromZk(env.getMasterServices().getZooKeeper(), getExecutorService());
216        FutureUtils.addListener(future, (r, e) -> {
217          // should acquire procedure execution lock to make sure that the procedure executor has
218          // finished putting this procedure to the WAITING_TIMEOUT state, otherwise there could be
219          // race and cause unexpected result
220          IdLock procLock =
221            env.getMasterServices().getMasterProcedureExecutor().getProcExecutionLock();
222          IdLock.Entry lockEntry;
223          try {
224            lockEntry = procLock.getLockEntry(getProcId());
225          } catch (IOException ioe) {
226            LOG.error("Error while acquiring execution lock for procedure {}"
227              + " when trying to wake it up, aborting...", this, ioe);
228            env.getMasterServices().abort("Can not acquire procedure execution lock", e);
229            return;
230          }
231          try {
232            setTimeoutFailure(env);
233          } finally {
234            procLock.releaseLockEntry(lockEntry);
235          }
236        });
237        // here we set timeout to -1 so the ProcedureExecutor will not schedule a Timer for us
238        setTimeout(-1);
239        setState(ProcedureProtos.ProcedureState.WAITING_TIMEOUT);
240        // skip persistence is a must now since when restarting, if the procedure is in
241        // WAITING_TIMEOUT state and has -1 as timeout, it will block there forever...
242        skipPersistence();
243        throw new ProcedureSuspendedException();
244      case MIGRATE_REPLICATION_QUEUE_FROM_ZK_TO_TABLE_WAIT_UPGRADING:
245        long rsWithLowerVersion =
246          env.getMasterServices().getServerManager().getOnlineServers().values().stream()
247            .filter(sm -> VersionInfo.getMajorVersion(sm.getVersion()) < MIN_MAJOR_VERSION).count();
248        if (rsWithLowerVersion == 0) {
249          setNextState(MIGRATE_REPLICATION_QUEUE_FROM_ZK_TO_TABLE_ENABLE_PEER);
250          return Flow.HAS_MORE_STATE;
251        } else {
252          throw suspend(env.getMasterConfiguration(),
253            backoff -> LOG.warn(
254              "There are still {} region servers which have a major version"
255                + " less than {}, sleep {} secs and check later",
256              rsWithLowerVersion, MIN_MAJOR_VERSION, backoff / 1000));
257        }
258      case MIGRATE_REPLICATION_QUEUE_FROM_ZK_TO_TABLE_ENABLE_PEER:
259        for (String peerId : disabledPeerIds) {
260          addChildProcedure(new EnablePeerProcedure(peerId));
261        }
262        setNextState(MIGRATE_REPLICATION_QUEUE_FROM_ZK_TO_TABLE_ENABLE_CLEANER);
263        return Flow.HAS_MORE_STATE;
264      case MIGRATE_REPLICATION_QUEUE_FROM_ZK_TO_TABLE_ENABLE_CLEANER:
265        enableReplicationLogCleaner(env);
266        return Flow.NO_MORE_STATE;
267      default:
268        throw new UnsupportedOperationException("unhandled state=" + state);
269    }
270  }

272  @Override

274    setState(ProcedureProtos.ProcedureState.RUNNABLE);
275    env.getProcedureScheduler().addFront(this);
276    return false;
277  }

300  @Override
301  protected void afterReplay(MasterProcedureEnv env) {
302    if (getCurrentState() == getInitialState()) {
303      // do not need to disable log cleaner or acquire lock if we are in the initial state, later
304      // when executing the procedure we will try to disable and acquire.
305      return;
306    }
307    if (!env.getMasterServices().getReplicationLogCleanerBarrier().disable()) {
308      throw new IllegalStateException("can not disable log cleaner, this should not happen");
309    }
310  }
