107  private void checkQueuesDeleted(String peerId)
108    throws ReplicationException, DoNotRetryIOException {
109    for (ServerName replicator : queueStorage.getListOfReplicators()) {
110      List<String> queueIds = queueStorage.getAllQueues(replicator);
111      for (String queueId : queueIds) {
112        ReplicationQueueInfo queueInfo = new ReplicationQueueInfo(queueId);
113        if (queueInfo.getPeerId().equals(peerId)) {
114          throw new DoNotRetryIOException("undeleted queue for peerId: " + peerId + ", replicator: "
115            + replicator + ", queueId: " + queueId);
116        }
117      }
118    }
119    if (queueStorage.getAllPeersFromHFileRefsQueue().contains(peerId)) {
120      throw new DoNotRetryIOException("Undeleted queue for peer " + peerId + " in hfile-refs");
121    }
122  }

124  void preAddPeer(String peerId, ReplicationPeerConfig peerConfig)
125    throws DoNotRetryIOException, ReplicationException {
126    if (peerId.contains("-")) {
127      throw new DoNotRetryIOException("Found invalid peer name: " + peerId);
128    }
129    checkPeerConfig(peerConfig);
130    if (peerConfig.isSyncReplication()) {
131      checkSyncReplicationPeerConfigConflict(peerConfig);
132    }
133    if (peers.containsKey(peerId)) {
134      throw new DoNotRetryIOException("Replication peer " + peerId + " already exists");
135    }
136    // make sure that there is no queues with the same peer id. This may happen when we create a
137    // peer with the same id with a old deleted peer. If the replication queues for the old peer
138    // have not been cleaned up yet then we should not create the new peer, otherwise the old wal
139    // file may also be replicated.
140    checkQueuesDeleted(peerId);
141  }

143  private ReplicationPeerDescription checkPeerExists(String peerId) throws DoNotRetryIOException {
144    ReplicationPeerDescription desc = peers.get(peerId);
145    if (desc == null) {
146      throw new ReplicationPeerNotFoundException(peerId);
147    }
148    return desc;
149  }

151  private void checkPeerInDAStateIfSyncReplication(String peerId) throws DoNotRetryIOException {
152    ReplicationPeerDescription desc = peers.get(peerId);
153    if (
154      desc != null && desc.getPeerConfig().isSyncReplication()
155        && !SyncReplicationState.DOWNGRADE_ACTIVE.equals(desc.getSyncReplicationState())
156    ) {
157      throw new DoNotRetryIOException(
158        "Couldn't remove synchronous replication peer with state=" + desc.getSyncReplicationState()
159          + ", Transit the synchronous replication state to be DOWNGRADE_ACTIVE firstly.");
160    }
161  }

183  /**
184   * Return the old peer description. Can never be null.
185   */
186  ReplicationPeerDescription preUpdatePeerConfig(String peerId, ReplicationPeerConfig peerConfig)
187    throws DoNotRetryIOException {
188    checkPeerConfig(peerConfig);
189    ReplicationPeerDescription desc = checkPeerExists(peerId);
190    ReplicationPeerConfig oldPeerConfig = desc.getPeerConfig();
191    if (!isStringEquals(peerConfig.getClusterKey(), oldPeerConfig.getClusterKey())) {
192      throw new DoNotRetryIOException(
193        "Changing the cluster key on an existing peer is not allowed. Existing key '"
194          + oldPeerConfig.getClusterKey() + "' for peer " + peerId + " does not match new key '"
195          + peerConfig.getClusterKey() + "'");
196    }
197
198    if (
199      !isStringEquals(peerConfig.getReplicationEndpointImpl(),
200        oldPeerConfig.getReplicationEndpointImpl())
201    ) {
202      throw new DoNotRetryIOException("Changing the replication endpoint implementation class "
203        + "on an existing peer is not allowed. Existing class '"
204        + oldPeerConfig.getReplicationEndpointImpl() + "' for peer " + peerId
205        + " does not match new class '" + peerConfig.getReplicationEndpointImpl() + "'");
206    }
207
208    if (!isStringEquals(peerConfig.getRemoteWALDir(), oldPeerConfig.getRemoteWALDir())) {
209      throw new DoNotRetryIOException(
210        "Changing the remote wal dir on an existing peer is not allowed. Existing remote wal "
211          + "dir '" + oldPeerConfig.getRemoteWALDir() + "' for peer " + peerId
212          + " does not match new remote wal dir '" + peerConfig.getRemoteWALDir() + "'");
213    }
214
215    if (oldPeerConfig.isSyncReplication()) {
216      if (!ReplicationUtils.isNamespacesAndTableCFsEqual(oldPeerConfig, peerConfig)) {
217        throw new DoNotRetryIOException(
218          "Changing the replicated namespace/table config on a synchronous replication "
219            + "peer(peerId: " + peerId + ") is not allowed.");
220      }
221    }
222    return desc;
223  }

225  /** Returns the old desciption of the peer */
226  ReplicationPeerDescription preTransitPeerSyncReplicationState(String peerId,
227    SyncReplicationState state) throws DoNotRetryIOException {
228    ReplicationPeerDescription desc = checkPeerExists(peerId);
229    SyncReplicationState fromState = desc.getSyncReplicationState();
230    EnumSet<SyncReplicationState> allowedToStates = allowedTransition.get(fromState);
231    if (allowedToStates == null || !allowedToStates.contains(state)) {
232      throw new DoNotRetryIOException("Can not transit current cluster state from " + fromState
233        + " to " + state + " for peer id=" + peerId);
234    }
235    return desc;
236  }

238  public void addPeer(String peerId, ReplicationPeerConfig peerConfig, boolean enabled)
239    throws ReplicationException {
240    if (peers.containsKey(peerId)) {
241      // this should be a retry, just return
242      return;
243    }
244    peerConfig = ReplicationPeerConfigUtil.updateReplicationBasePeerConfigs(conf, peerConfig);
245    ReplicationPeerConfig copiedPeerConfig = ReplicationPeerConfig.newBuilder(peerConfig).build();
246    SyncReplicationState syncReplicationState = copiedPeerConfig.isSyncReplication()
247      ? SyncReplicationState.DOWNGRADE_ACTIVE
248      : SyncReplicationState.NONE;
249    peerStorage.addPeer(peerId, copiedPeerConfig, enabled, syncReplicationState);
250    peers.put(peerId,
251      new ReplicationPeerDescription(peerId, enabled, copiedPeerConfig, syncReplicationState));
252  }
253
254  public void removePeer(String peerId) throws ReplicationException {
255    if (!peers.containsKey(peerId)) {
256      // this should be a retry, just return
257      return;
258    }
259    peerStorage.removePeer(peerId);
260    peers.remove(peerId);
261  }

263  private void setPeerState(String peerId, boolean enabled) throws ReplicationException {
264    ReplicationPeerDescription desc = peers.get(peerId);
265    if (desc.isEnabled() == enabled) {
266      // this should be a retry, just return

268    }
269    peerStorage.setPeerState(peerId, enabled);
270    peers.put(peerId, new ReplicationPeerDescription(peerId, enabled, desc.getPeerConfig(),
271      desc.getSyncReplicationState()));
272  }

282  public void updatePeerConfig(String peerId, ReplicationPeerConfig peerConfig)
283    throws ReplicationException {
284    // the checking rules are too complicated here so we give up checking whether this is a retry.
285    ReplicationPeerDescription desc = peers.get(peerId);
286    ReplicationPeerConfig oldPeerConfig = desc.getPeerConfig();
287    ReplicationPeerConfigBuilder newPeerConfigBuilder =
288      ReplicationPeerConfig.newBuilder(peerConfig);
289    // we need to use the new conf to overwrite the old one.
290    newPeerConfigBuilder.putAllConfiguration(oldPeerConfig.getConfiguration());
291    newPeerConfigBuilder.putAllConfiguration(peerConfig.getConfiguration());
292    ReplicationPeerConfig newPeerConfig = newPeerConfigBuilder.build();
293    peerStorage.updatePeerConfig(peerId, newPeerConfig);
294    peers.put(peerId, new ReplicationPeerDescription(peerId, desc.isEnabled(), newPeerConfig,
295      desc.getSyncReplicationState()));
296  }

320  public void transitPeerSyncReplicationState(String peerId, SyncReplicationState newState)
321    throws ReplicationException {
322    if (peerStorage.getPeerNewSyncReplicationState(peerId) != SyncReplicationState.NONE) {
323      // Only transit if this is not a retry
324      peerStorage.transitPeerSyncReplicationState(peerId);
325    }
326    ReplicationPeerDescription desc = peers.get(peerId);
327    if (desc.getSyncReplicationState() != newState) {
328      // Only recreate the desc if this is not a retry
329      peers.put(peerId,
330        new ReplicationPeerDescription(peerId, desc.isEnabled(), desc.getPeerConfig(), newState));
331    }
332  }

334  public void removeAllQueues(String peerId) throws ReplicationException {
335    // Here we need two passes to address the problem of claimQueue. Maybe a claimQueue is still
336    // on-going when the refresh peer config procedure is done, if a RS which has already been
337    // scanned claims the queue of a RS which has not been scanned yet, we will miss that queue in
338    // the scan here, and if the RS who has claimed the queue crashed before creating recovered
339    // source, then the queue will leave there until the another RS detects the crash and helps
340    // removing the queue.
341    // A two pass scan can solve the problem. Anyway, the queue will not disappear during the
342    // claiming, it will either under the old RS or under the new RS, and a queue can only be
343    // claimed once after the refresh peer procedure done(as the next claim queue will just delete
344    // it), so we can make sure that a two pass scan will finally find the queue and remove it,
345    // unless it has already been removed by others.
346    ReplicationUtils.removeAllQueues(queueStorage, peerId);
347    ReplicationUtils.removeAllQueues(queueStorage, peerId);
348  }

355  private void checkPeerConfig(ReplicationPeerConfig peerConfig) throws DoNotRetryIOException {
356    String replicationEndpointImpl = peerConfig.getReplicationEndpointImpl();
357    ReplicationEndpoint endpoint = null;
358    if (!StringUtils.isBlank(replicationEndpointImpl)) {
359      try {
360        // try creating a instance
361        endpoint = Class.forName(replicationEndpointImpl).asSubclass(ReplicationEndpoint.class)
362          .getDeclaredConstructor().newInstance();
363      } catch (Throwable e) {
364        throw new DoNotRetryIOException(
365          "Can not instantiate configured replication endpoint class=" + replicationEndpointImpl,
366          e);
367      }
368    }
369    // Endpoints implementing HBaseReplicationEndpoint need to check cluster key
370    if (endpoint == null || endpoint instanceof HBaseReplicationEndpoint) {
371      checkClusterKey(peerConfig.getClusterKey());
372      // Check if endpoint can replicate to the same cluster
373      if (endpoint == null || !endpoint.canReplicateToSameCluster()) {
374        checkSameClusterKey(peerConfig.getClusterKey());
375      }
376    }
377
378    if (peerConfig.replicateAllUserTables()) {
379      // If replicate_all flag is true, it means all user tables will be replicated to peer cluster.
380      // Then allow config exclude namespaces or exclude table-cfs which can't be replicated to peer
381      // cluster.
382      if (
383        (peerConfig.getNamespaces() != null && !peerConfig.getNamespaces().isEmpty())
384          || (peerConfig.getTableCFsMap() != null && !peerConfig.getTableCFsMap().isEmpty())
385      ) {
386        throw new DoNotRetryIOException("Need clean namespaces or table-cfs config firstly "
387          + "when you want replicate all cluster");
388      }
389      checkNamespacesAndTableCfsConfigConflict(peerConfig.getExcludeNamespaces(),
390        peerConfig.getExcludeTableCFsMap());
391    } else {
392      // If replicate_all flag is false, it means all user tables can't be replicated to peer
393      // cluster. Then allow to config namespaces or table-cfs which will be replicated to peer
394      // cluster.
395      if (
396        (peerConfig.getExcludeNamespaces() != null && !peerConfig.getExcludeNamespaces().isEmpty())
397          || (peerConfig.getExcludeTableCFsMap() != null
398            && !peerConfig.getExcludeTableCFsMap().isEmpty())
399      ) {
400        throw new DoNotRetryIOException(
401          "Need clean exclude-namespaces or exclude-table-cfs config firstly"
402            + " when replicate_all flag is false");
403      }
404      checkNamespacesAndTableCfsConfigConflict(peerConfig.getNamespaces(),
405        peerConfig.getTableCFsMap());
406    }
407
408    if (peerConfig.isSyncReplication()) {
409      checkPeerConfigForSyncReplication(peerConfig);
410    }
411
412    checkConfiguredWALEntryFilters(peerConfig);
413  }

415  private void checkPeerConfigForSyncReplication(ReplicationPeerConfig peerConfig)
416    throws DoNotRetryIOException {
417    // This is used to reduce the difficulty for implementing the sync replication state transition
418    // as we need to reopen all the related regions.
419    // TODO: Add namespace, replicat_all flag back
420    if (peerConfig.replicateAllUserTables()) {
421      throw new DoNotRetryIOException(
422        "Only support replicated table config for sync replication peer");
423    }
424    if (peerConfig.getNamespaces() != null && !peerConfig.getNamespaces().isEmpty()) {
425      throw new DoNotRetryIOException(
426        "Only support replicated table config for sync replication peer");
427    }
428    if (peerConfig.getTableCFsMap() == null || peerConfig.getTableCFsMap().isEmpty()) {
429      throw new DoNotRetryIOException("Need config replicated tables for sync replication peer");
430    }
431    for (List<String> cfs : peerConfig.getTableCFsMap().values()) {
432      if (cfs != null && !cfs.isEmpty()) {
433        throw new DoNotRetryIOException(
434          "Only support replicated table config for sync replication peer");
435      }
436    }
437
438    Path remoteWALDir = new Path(peerConfig.getRemoteWALDir());
439    if (!remoteWALDir.isAbsolute()) {
440      throw new DoNotRetryIOException(
441        "The remote WAL directory " + peerConfig.getRemoteWALDir() + " is not absolute");
442    }
443    URI remoteWALDirUri = remoteWALDir.toUri();
444    if (remoteWALDirUri.getScheme() == null || remoteWALDirUri.getAuthority() == null) {
445      throw new DoNotRetryIOException("The remote WAL directory " + peerConfig.getRemoteWALDir()
446        + " is not qualified, you must provide scheme and authority");
447    }
448  }

450  private void checkSyncReplicationPeerConfigConflict(ReplicationPeerConfig peerConfig)
451    throws DoNotRetryIOException {
452    for (TableName tableName : peerConfig.getTableCFsMap().keySet()) {
453      for (Map.Entry<String, ReplicationPeerDescription> entry : peers.entrySet()) {
454        ReplicationPeerConfig rpc = entry.getValue().getPeerConfig();
455        if (rpc.isSyncReplication() && rpc.getTableCFsMap().containsKey(tableName)) {
456          throw new DoNotRetryIOException(
457            "Table " + tableName + " has been replicated by peer " + entry.getKey());
458        }
459      }
460    }
461  }

463  /**
464   * Set a namespace in the peer config means that all tables in this namespace will be replicated
465   * to the peer cluster.
466   * <ol>
467   * <li>If peer config already has a namespace, then not allow set any table of this namespace to
468   * the peer config.</li>
469   * <li>If peer config already has a table, then not allow set this table's namespace to the peer
470   * config.</li>
471   * </ol>
472   * <p>
473   * Set a exclude namespace in the peer config means that all tables in this namespace can't be
474   * replicated to the peer cluster.
475   * <ol>
476   * <li>If peer config already has a exclude namespace, then not allow set any exclude table of
477   * this namespace to the peer config.</li>
478   * <li>If peer config already has a exclude table, then not allow set this table's namespace as a
479   * exclude namespace.</li>
480   * </ol>
481   */
482  private void checkNamespacesAndTableCfsConfigConflict(Set<String> namespaces,
483    Map<TableName, ? extends Collection<String>> tableCfs) throws DoNotRetryIOException {
484    if (namespaces == null || namespaces.isEmpty()) {
485      return;
486    }
487    if (tableCfs == null || tableCfs.isEmpty()) {
488      return;
489    }
490    for (Map.Entry<TableName, ? extends Collection<String>> entry : tableCfs.entrySet()) {
491      TableName table = entry.getKey();
492      if (namespaces.contains(table.getNamespaceAsString())) {
493        throw new DoNotRetryIOException("Table-cfs " + table + " is conflict with namespaces "
494          + table.getNamespaceAsString() + " in peer config");
495      }
496    }
497  }

499  private void checkConfiguredWALEntryFilters(ReplicationPeerConfig peerConfig)
500    throws DoNotRetryIOException {
501    String filterCSV = peerConfig.getConfiguration()
502      .get(BaseReplicationEndpoint.REPLICATION_WALENTRYFILTER_CONFIG_KEY);
503    if (filterCSV != null && !filterCSV.isEmpty()) {
504      String[] filters = filterCSV.split(",");
505      for (String filter : filters) {
506        try {
507          Class.forName(filter).getDeclaredConstructor().newInstance();
508        } catch (Exception e) {
509          throw new DoNotRetryIOException("Configured WALEntryFilter " + filter
510            + " could not be created. Failing add/update peer operation.", e);
511        }
512      }
513    }
514  }

524  private void checkSameClusterKey(String clusterKey) throws DoNotRetryIOException {
525    String peerClusterId = "";
526    try {
527      // Create the peer cluster config for get peer cluster id
528      Configuration peerConf = HBaseConfiguration.createClusterConf(conf, clusterKey);
529      try (ZKWatcher zkWatcher = new ZKWatcher(peerConf, this + "check-peer-cluster-id", null)) {
530        peerClusterId = ZKClusterId.readClusterIdZNode(zkWatcher);
531      }
532    } catch (IOException | KeeperException e) {
533      throw new DoNotRetryIOException("Can't get peerClusterId for clusterKey=" + clusterKey, e);
534    }
535    // In rare case, zookeeper setting may be messed up. That leads to the incorrect
536    // peerClusterId value, which is the same as the source clusterId
537    if (clusterId.equals(peerClusterId)) {
538      throw new DoNotRetryIOException("Invalid cluster key: " + clusterKey
539        + ", should not replicate to itself for HBaseInterClusterReplicationEndpoint");
540    }
541  }

553  public static ReplicationPeerManager create(ZKWatcher zk, Configuration conf, String clusterId)
554    throws ReplicationException {
555    ReplicationPeerStorage peerStorage =
556      ReplicationStorageFactory.getReplicationPeerStorage(zk, conf);
557    ConcurrentMap<String, ReplicationPeerDescription> peers = new ConcurrentHashMap<>();
558    for (String peerId : peerStorage.listPeerIds()) {
559      ReplicationPeerConfig peerConfig = peerStorage.getPeerConfig(peerId);
560      if (
561        ReplicationUtils.LEGACY_REGION_REPLICATION_ENDPOINT_NAME
562          .equals(peerConfig.getReplicationEndpointImpl())
563      ) {
564        // we do not use this endpoint for region replication any more, see HBASE-26233
565        LOG.info("Legacy region replication peer found, removing: {}", peerConfig);
566        peerStorage.removePeer(peerId);
567        continue;
568      }
569      peerConfig = ReplicationPeerConfigUtil.updateReplicationBasePeerConfigs(conf, peerConfig);
570      peerStorage.updatePeerConfig(peerId, peerConfig);
571      boolean enabled = peerStorage.isPeerEnabled(peerId);
572      SyncReplicationState state = peerStorage.getPeerSyncReplicationState(peerId);
573      peers.put(peerId, new ReplicationPeerDescription(peerId, enabled, peerConfig, state));
574    }
575    return new ReplicationPeerManager(peerStorage,
576      ReplicationStorageFactory.getReplicationQueueStorage(zk, conf), peers, conf, clusterId);
577  }

580   * For replication peer cluster key or endpoint class, null and empty string is same. So here
581   * don't use {@link StringUtils#equals(CharSequence, CharSequence)} directly.
582   */
583  private boolean isStringEquals(String s1, String s2) {
584    if (StringUtils.isBlank(s1)) {
585      return StringUtils.isBlank(s2);
586    }
587    return s1.equals(s2);
588  }
