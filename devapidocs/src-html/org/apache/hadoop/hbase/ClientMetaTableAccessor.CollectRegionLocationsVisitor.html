077  public static CompletableFuture<Boolean> tableExists(AsyncTable<?> metaTable,
078    TableName tableName) {
079    return getTableState(metaTable, tableName).thenApply(Optional::isPresent);
080  }

082  public static CompletableFuture<Optional<TableState>> getTableState(AsyncTable<?> metaTable,
083    TableName tableName) {
084    CompletableFuture<Optional<TableState>> future = new CompletableFuture<>();
085    Get get = new Get(tableName.getName()).addColumn(HConstants.TABLE_FAMILY,
086      HConstants.TABLE_STATE_QUALIFIER);
087    addListener(metaTable.get(get), (result, error) -> {
088      if (error != null) {
089        future.completeExceptionally(error);
090        return;
091      }
092      try {
093        future.complete(getTableState(result));
094      } catch (IOException e) {
095        future.completeExceptionally(e);
096      }
097    });
098    return future;
099  }

101  /**
102   * Returns the HRegionLocation from meta for the given region n * @param regionName region we're
103   * looking for
104   * @return HRegionLocation for the given region
105   */
106  public static CompletableFuture<Optional<HRegionLocation>>
107    getRegionLocation(AsyncTable<?> metaTable, byte[] regionName) {
108    CompletableFuture<Optional<HRegionLocation>> future = new CompletableFuture<>();
109    try {
110      RegionInfo parsedRegionInfo = CatalogFamilyFormat.parseRegionInfoFromRegionName(regionName);
111      addListener(metaTable.get(new Get(CatalogFamilyFormat.getMetaKeyForRegion(parsedRegionInfo))
112        .addFamily(HConstants.CATALOG_FAMILY)), (r, err) -> {
113          if (err != null) {
114            future.completeExceptionally(err);
115            return;
116          }
117          future.complete(getRegionLocations(r)
118            .map(locations -> locations.getRegionLocation(parsedRegionInfo.getReplicaId())));
119        });
120    } catch (IOException parseEx) {
121      LOG.warn("Failed to parse the passed region name: " + Bytes.toStringBinary(regionName));
122      future.completeExceptionally(parseEx);
123    }
124    return future;
125  }

127  /**
128   * Returns the HRegionLocation from meta for the given encoded region name n * @param
129   * encodedRegionName region we're looking for
130   * @return HRegionLocation for the given region
131   */
132  public static CompletableFuture<Optional<HRegionLocation>>
133    getRegionLocationWithEncodedName(AsyncTable<?> metaTable, byte[] encodedRegionName) {
134    CompletableFuture<Optional<HRegionLocation>> future = new CompletableFuture<>();
135    addListener(
136      metaTable
137        .scanAll(new Scan().setReadType(ReadType.PREAD).addFamily(HConstants.CATALOG_FAMILY)),
138      (results, err) -> {
139        if (err != null) {
140          future.completeExceptionally(err);
141          return;
142        }
143        String encodedRegionNameStr = Bytes.toString(encodedRegionName);
144        results.stream().filter(result -> !result.isEmpty())
145          .filter(result -> CatalogFamilyFormat.getRegionInfo(result) != null).forEach(result -> {
146            getRegionLocations(result).ifPresent(locations -> {
147              for (HRegionLocation location : locations.getRegionLocations()) {
148                if (
149                  location != null
150                    && encodedRegionNameStr.equals(location.getRegion().getEncodedName())
151                ) {
152                  future.complete(Optional.of(location));
153                  return;
154                }
155              }
156            });
157          });
158        future.complete(Optional.empty());
159      });
160    return future;
161  }

167  /**
168   * Used to get all region locations for the specific table. n * @param tableName table we're
169   * looking for, can be null for getting all regions
170   * @return the list of region locations. The return value will be wrapped by a
171   *         {@link CompletableFuture}.
172   */
173  public static CompletableFuture<List<HRegionLocation>> getTableHRegionLocations(
174    AsyncTable<AdvancedScanResultConsumer> metaTable, TableName tableName) {
175    CompletableFuture<List<HRegionLocation>> future = new CompletableFuture<>();
176    addListener(getTableRegionsAndLocations(metaTable, tableName, true), (locations, err) -> {
177      if (err != null) {
178        future.completeExceptionally(err);
179      } else if (locations == null || locations.isEmpty()) {
180        future.complete(Collections.emptyList());
181      } else {
182        List<HRegionLocation> regionLocations =
183          locations.stream().map(loc -> new HRegionLocation(loc.getFirst(), loc.getSecond()))
184            .collect(Collectors.toList());
185        future.complete(regionLocations);
186      }
187    });
188    return future;
189  }

191  /**
192   * Used to get table regions' info and server. n * @param tableName table we're looking for, can
193   * be null for getting all regions
194   * @param excludeOfflinedSplitParents don't return split parents
195   * @return the list of regioninfos and server. The return value will be wrapped by a
196   *         {@link CompletableFuture}.
197   */
198  private static CompletableFuture<List<Pair<RegionInfo, ServerName>>> getTableRegionsAndLocations(
199    final AsyncTable<AdvancedScanResultConsumer> metaTable, final TableName tableName,
200    final boolean excludeOfflinedSplitParents) {
201    CompletableFuture<List<Pair<RegionInfo, ServerName>>> future = new CompletableFuture<>();
202    if (TableName.META_TABLE_NAME.equals(tableName)) {
203      future.completeExceptionally(new IOException(
204        "This method can't be used to locate meta regions;" + " use MetaTableLocator instead"));
205    }
206
207    // Make a version of CollectingVisitor that collects RegionInfo and ServerAddress
208    CollectRegionLocationsVisitor visitor =
209      new CollectRegionLocationsVisitor(excludeOfflinedSplitParents);
210
211    addListener(scanMeta(metaTable, tableName, QueryType.REGION, visitor), (v, error) -> {
212      if (error != null) {
213        future.completeExceptionally(error);
214        return;
215      }
216      future.complete(visitor.getResults());
217    });
218    return future;
219  }

221  /**
222   * Performs a scan of META table for given table. n * @param tableName table withing we scan
223   * @param type    scanned part of meta
224   * @param visitor Visitor invoked against each row
225   */
226  private static CompletableFuture<Void> scanMeta(AsyncTable<AdvancedScanResultConsumer> metaTable,
227    TableName tableName, QueryType type, final Visitor visitor) {
228    return scanMeta(metaTable, getTableStartRowForMeta(tableName, type),
229      getTableStopRowForMeta(tableName, type), type, Integer.MAX_VALUE, visitor);
230  }
231
232  /**
233   * Performs a scan of META table for given table. n * @param startRow Where to start the scan
234   * @param stopRow Where to stop the scan
235   * @param type    scanned part of meta
236   * @param maxRows maximum rows to return
237   * @param visitor Visitor invoked against each row
238   */
239  private static CompletableFuture<Void> scanMeta(AsyncTable<AdvancedScanResultConsumer> metaTable,
240    byte[] startRow, byte[] stopRow, QueryType type, int maxRows, final Visitor visitor) {
241    int rowUpperLimit = maxRows > 0 ? maxRows : Integer.MAX_VALUE;
242    Scan scan = getMetaScan(metaTable, rowUpperLimit);
243    for (byte[] family : type.getFamilies()) {
244      scan.addFamily(family);
245    }
246    if (startRow != null) {
247      scan.withStartRow(startRow);
248    }
249    if (stopRow != null) {
250      scan.withStopRow(stopRow);
251    }
252
253    if (LOG.isDebugEnabled()) {
254      LOG.debug("Scanning META" + " starting at row=" + Bytes.toStringBinary(scan.getStartRow())
255        + " stopping at row=" + Bytes.toStringBinary(scan.getStopRow()) + " for max="
256        + rowUpperLimit + " with caching=" + scan.getCaching());
257    }
258
259    CompletableFuture<Void> future = new CompletableFuture<Void>();
260    metaTable.scan(scan, new MetaTableScanResultConsumer(rowUpperLimit, visitor, future));
261    return future;
262  }

264  private static final class MetaTableScanResultConsumer implements AdvancedScanResultConsumer {
265
266    private int currentRowCount;
267
268    private final int rowUpperLimit;
269
270    private final Visitor visitor;
271
272    private final CompletableFuture<Void> future;
273
274    MetaTableScanResultConsumer(int rowUpperLimit, Visitor visitor,
275      CompletableFuture<Void> future) {
276      this.rowUpperLimit = rowUpperLimit;
277      this.visitor = visitor;
278      this.future = future;
279      this.currentRowCount = 0;
280    }
281
282    @Override
283    public void onError(Throwable error) {
284      future.completeExceptionally(error);
285    }
286
287    @Override
288    @edu.umd.cs.findbugs.annotations.SuppressWarnings(value = "NP_NONNULL_PARAM_VIOLATION",
289        justification = "https://github.com/findbugsproject/findbugs/issues/79")
290    public void onComplete() {
291      future.complete(null);
292    }
293
294    @Override
295    public void onNext(Result[] results, ScanController controller) {
296      boolean terminateScan = false;
297      for (Result result : results) {
298        try {
299          if (!visitor.visit(result)) {
300            terminateScan = true;
301            break;
302          }
303        } catch (Exception e) {
304          future.completeExceptionally(e);
305          terminateScan = true;
306          break;
307        }
308        if (++currentRowCount >= rowUpperLimit) {
309          terminateScan = true;
310          break;
311        }
312      }
313      if (terminateScan) {
314        controller.terminate();
315      }
316    }
317  }

337  /**
338   * A {@link Visitor} that collects content out of passed {@link Result}.
339   */
340  private static abstract class CollectingVisitor<T> implements Visitor {
341    final List<T> results = new ArrayList<>();
342
343    @Override
344    public boolean visit(Result r) throws IOException {
345      if (r != null && !r.isEmpty()) {
346        add(r);
347      }
348      return true;
349    }
350
351    abstract void add(Result r);
352
353    /**
354     * @return Collected results; wait till visits complete to collect all possible results
355     */
356    List<T> getResults() {
357      return this.results;
358    }
359  }
360
361  static class CollectRegionLocationsVisitor
362    extends CollectingVisitor<Pair<RegionInfo, ServerName>> {
363
364    private final boolean excludeOfflinedSplitParents;
365
366    private RegionLocations current = null;
367
368    CollectRegionLocationsVisitor(boolean excludeOfflinedSplitParents) {
369      this.excludeOfflinedSplitParents = excludeOfflinedSplitParents;
370    }
371
372    @Override
373    public boolean visit(Result r) throws IOException {
374      Optional<RegionLocations> currentRegionLocations = getRegionLocations(r);
375      current = currentRegionLocations.orElse(null);
376      if (current == null || current.getRegionLocation().getRegion() == null) {
377        LOG.warn("No serialized RegionInfo in " + r);
378        return true;
379      }
380      RegionInfo hri = current.getRegionLocation().getRegion();
381      if (excludeOfflinedSplitParents && hri.isSplitParent()) {
382        return true;
383      }
384      // Else call super and add this Result to the collection.
385      return super.visit(r);
386    }
387
388    @Override
389    void add(Result r) {
390      if (current == null) {
391        return;
392      }
393      for (HRegionLocation loc : current.getRegionLocations()) {
394        if (loc != null) {
395          this.results.add(new Pair<RegionInfo, ServerName>(loc.getRegion(), loc.getServerName()));
396        }
397      }
398    }
399  }
400
401  /**
402   * Collects all returned.
403   */
404  static class CollectAllVisitor extends CollectingVisitor<Result> {
405    @Override
406    void add(Result r) {
407      this.results.add(r);
408    }
409  }

411  private static Scan getMetaScan(AsyncTable<?> metaTable, int rowUpperLimit) {
412    Scan scan = new Scan();
413    int scannerCaching = metaTable.getConfiguration().getInt(HConstants.HBASE_META_SCANNER_CACHING,
414      HConstants.DEFAULT_HBASE_META_SCANNER_CACHING);
415    if (
416      metaTable.getConfiguration().getBoolean(HConstants.USE_META_REPLICAS,
417        HConstants.DEFAULT_USE_META_REPLICAS)
418    ) {
419      scan.setConsistency(Consistency.TIMELINE);
420    }
421    if (rowUpperLimit <= scannerCaching) {
422      scan.setLimit(rowUpperLimit);
423    }
424    int rows = Math.min(rowUpperLimit, scannerCaching);
425    scan.setCaching(rows);
426    return scan;
427  }

438  /**
439   * @param tableName table we're working with
440   * @return start row for scanning META according to query type
441   */
442  public static byte[] getTableStartRowForMeta(TableName tableName, QueryType type) {
443    if (tableName == null) {
444      return null;
445    }
446    switch (type) {
447      case REGION:
448      case REPLICATION: {
449        byte[] startRow = new byte[tableName.getName().length + 2];
450        System.arraycopy(tableName.getName(), 0, startRow, 0, tableName.getName().length);
451        startRow[startRow.length - 2] = HConstants.DELIMITER;
452        startRow[startRow.length - 1] = HConstants.DELIMITER;
453        return startRow;
454      }
455      case ALL:
456      case TABLE:
457      default: {
458        return tableName.getName();
459      }
460    }
461  }
462
463  /**
464   * @param tableName table we're working with
465   * @return stop row for scanning META according to query type
466   */
467  public static byte[] getTableStopRowForMeta(TableName tableName, QueryType type) {
468    if (tableName == null) {
469      return null;
470    }
471    final byte[] stopRow;
472    switch (type) {
473      case REGION:
474      case REPLICATION: {
475        stopRow = new byte[tableName.getName().length + 3];
476        System.arraycopy(tableName.getName(), 0, stopRow, 0, tableName.getName().length);
477        stopRow[stopRow.length - 3] = ' ';
478        stopRow[stopRow.length - 2] = HConstants.DELIMITER;
479        stopRow[stopRow.length - 1] = HConstants.DELIMITER;
480        break;
481      }
482      case ALL:
483      case TABLE:
484      default: {
485        stopRow = new byte[tableName.getName().length + 1];
486        System.arraycopy(tableName.getName(), 0, stopRow, 0, tableName.getName().length);
487        stopRow[stopRow.length - 1] = ' ';
488        break;
489      }
490    }
491    return stopRow;
492  }
