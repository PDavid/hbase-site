079  public static CompletableFuture<Boolean> tableExists(AsyncTable<?> metaTable,
080    TableName tableName) {
081    return getTableState(metaTable, tableName).thenApply(Optional::isPresent);
082  }

084  public static CompletableFuture<Optional<TableState>> getTableState(AsyncTable<?> metaTable,
085    TableName tableName) {
086    CompletableFuture<Optional<TableState>> future = new CompletableFuture<>();
087    Get get = new Get(tableName.getName()).addColumn(HConstants.TABLE_FAMILY,
088      HConstants.TABLE_STATE_QUALIFIER);
089    addListener(metaTable.get(get), (result, error) -> {
090      if (error != null) {
091        future.completeExceptionally(error);
092        return;
093      }
094      try {
095        future.complete(getTableState(result));
096      } catch (IOException e) {
097        future.completeExceptionally(e);
098      }
099    });
100    return future;
101  }

103  /**
104   * Returns the HRegionLocation from meta for the given region n * @param regionName region we're
105   * looking for
106   * @return HRegionLocation for the given region
107   */
108  public static CompletableFuture<Optional<HRegionLocation>>
109    getRegionLocation(AsyncTable<?> metaTable, byte[] regionName) {
110    CompletableFuture<Optional<HRegionLocation>> future = new CompletableFuture<>();
111    try {
112      RegionInfo parsedRegionInfo = CatalogFamilyFormat.parseRegionInfoFromRegionName(regionName);
113      addListener(metaTable.get(new Get(CatalogFamilyFormat.getMetaKeyForRegion(parsedRegionInfo))
114        .addFamily(HConstants.CATALOG_FAMILY)), (r, err) -> {
115          if (err != null) {
116            future.completeExceptionally(err);
117            return;
118          }
119          future.complete(getRegionLocations(r)
120            .map(locations -> locations.getRegionLocation(parsedRegionInfo.getReplicaId())));
121        });
122    } catch (IOException parseEx) {
123      LOG.warn("Failed to parse the passed region name: " + Bytes.toStringBinary(regionName));
124      future.completeExceptionally(parseEx);
125    }
126    return future;
127  }

129  /**
130   * Returns the HRegionLocation from meta for the given encoded region name n * @param
131   * encodedRegionName region we're looking for
132   * @return HRegionLocation for the given region
133   */
134  public static CompletableFuture<Optional<HRegionLocation>>
135    getRegionLocationWithEncodedName(AsyncTable<?> metaTable, byte[] encodedRegionName) {
136    CompletableFuture<Optional<HRegionLocation>> future = new CompletableFuture<>();
137    addListener(
138      metaTable
139        .scanAll(new Scan().setReadType(ReadType.PREAD).addFamily(HConstants.CATALOG_FAMILY)),
140      (results, err) -> {
141        if (err != null) {
142          future.completeExceptionally(err);
143          return;
144        }
145        String encodedRegionNameStr = Bytes.toString(encodedRegionName);
146        results.stream().filter(result -> !result.isEmpty())
147          .filter(result -> CatalogFamilyFormat.getRegionInfo(result) != null).forEach(result -> {
148            getRegionLocations(result).ifPresent(locations -> {
149              for (HRegionLocation location : locations.getRegionLocations()) {
150                if (
151                  location != null
152                    && encodedRegionNameStr.equals(location.getRegion().getEncodedName())
153                ) {
154                  future.complete(Optional.of(location));
155                  return;
156                }
157              }
158            });
159          });
160        future.complete(Optional.empty());
161      });
162    return future;
163  }

169  /**
170   * Used to get all region locations for the specific table. n * @param tableName table we're
171   * looking for, can be null for getting all regions
172   * @return the list of region locations. The return value will be wrapped by a
173   *         {@link CompletableFuture}.
174   */
175  public static CompletableFuture<List<HRegionLocation>> getTableHRegionLocations(
176    AsyncTable<AdvancedScanResultConsumer> metaTable, TableName tableName) {
177    CompletableFuture<List<HRegionLocation>> future = new CompletableFuture<>();
178    addListener(getTableRegionsAndLocations(metaTable, tableName, true), (locations, err) -> {
179      if (err != null) {
180        future.completeExceptionally(err);
181      } else if (locations == null || locations.isEmpty()) {
182        future.complete(Collections.emptyList());
183      } else {
184        List<HRegionLocation> regionLocations =
185          locations.stream().map(loc -> new HRegionLocation(loc.getFirst(), loc.getSecond()))
186            .collect(Collectors.toList());
187        future.complete(regionLocations);
188      }
189    });
190    return future;
191  }

193  /**
194   * Used to get table regions' info and server. n * @param tableName table we're looking for, can
195   * be null for getting all regions
196   * @param excludeOfflinedSplitParents don't return split parents
197   * @return the list of regioninfos and server. The return value will be wrapped by a
198   *         {@link CompletableFuture}.
199   */
200  private static CompletableFuture<List<Pair<RegionInfo, ServerName>>> getTableRegionsAndLocations(
201    final AsyncTable<AdvancedScanResultConsumer> metaTable, final TableName tableName,
202    final boolean excludeOfflinedSplitParents) {
203    CompletableFuture<List<Pair<RegionInfo, ServerName>>> future = new CompletableFuture<>();
204    if (TableName.META_TABLE_NAME.equals(tableName)) {
205      future.completeExceptionally(new IOException(
206        "This method can't be used to locate meta regions;" + " use MetaTableLocator instead"));
207    }
208
209    // Make a version of CollectingVisitor that collects RegionInfo and ServerAddress
210    CollectRegionLocationsVisitor visitor =
211      new CollectRegionLocationsVisitor(excludeOfflinedSplitParents);
212
213    addListener(scanMeta(metaTable, tableName, QueryType.REGION, visitor), (v, error) -> {
214      if (error != null) {
215        future.completeExceptionally(error);
216        return;
217      }
218      future.complete(visitor.getResults());
219    });
220    return future;
221  }

223  /**
224   * Performs a scan of META table for given table. n * @param tableName table withing we scan
225   * @param type    scanned part of meta
226   * @param visitor Visitor invoked against each row
227   */
228  private static CompletableFuture<Void> scanMeta(AsyncTable<AdvancedScanResultConsumer> metaTable,
229    TableName tableName, QueryType type, final Visitor visitor) {
230    return scanMeta(metaTable, getTableStartRowForMeta(tableName, type),
231      getTableStopRowForMeta(tableName, type), type, Integer.MAX_VALUE, visitor);
232  }
233
234  /**
235   * Performs a scan of META table for given table. n * @param startRow Where to start the scan
236   * @param stopRow Where to stop the scan
237   * @param type    scanned part of meta
238   * @param maxRows maximum rows to return
239   * @param visitor Visitor invoked against each row
240   */
241  private static CompletableFuture<Void> scanMeta(AsyncTable<AdvancedScanResultConsumer> metaTable,
242    byte[] startRow, byte[] stopRow, QueryType type, int maxRows, final Visitor visitor) {
243    int rowUpperLimit = maxRows > 0 ? maxRows : Integer.MAX_VALUE;
244    Scan scan = getMetaScan(metaTable, rowUpperLimit);
245    for (byte[] family : type.getFamilies()) {
246      scan.addFamily(family);
247    }
248    if (startRow != null) {
249      scan.withStartRow(startRow);
250    }
251    if (stopRow != null) {
252      scan.withStopRow(stopRow);
253    }
254
255    if (LOG.isDebugEnabled()) {
256      LOG.debug("Scanning META" + " starting at row=" + Bytes.toStringBinary(scan.getStartRow())
257        + " stopping at row=" + Bytes.toStringBinary(scan.getStopRow()) + " for max="
258        + rowUpperLimit + " with caching=" + scan.getCaching());
259    }
260
261    CompletableFuture<Void> future = new CompletableFuture<Void>();
262    // Get the region locator's meta replica mode.
263    CatalogReplicaMode metaReplicaMode = CatalogReplicaMode.fromString(metaTable.getConfiguration()
264      .get(LOCATOR_META_REPLICAS_MODE, CatalogReplicaMode.NONE.toString()));
265
266    if (metaReplicaMode == CatalogReplicaMode.LOAD_BALANCE) {
267      addListener(metaTable.getDescriptor(), (desc, error) -> {
268        if (error != null) {
269          LOG.error("Failed to get meta table descriptor, error: ", error);
270          future.completeExceptionally(error);
271          return;
272        }
273
274        int numOfReplicas = desc.getRegionReplication();
275        if (numOfReplicas > 1) {
276          int replicaId = ThreadLocalRandom.current().nextInt(numOfReplicas);
277
278          // When the replicaId is 0, do not set to Consistency.TIMELINE
279          if (replicaId > 0) {
280            scan.setReplicaId(replicaId);
281            scan.setConsistency(Consistency.TIMELINE);
282          }
283        }
284        metaTable.scan(scan, new MetaTableScanResultConsumer(rowUpperLimit, visitor, future));
285      });
286    } else {
287      if (metaReplicaMode == CatalogReplicaMode.HEDGED_READ) {
288        scan.setConsistency(Consistency.TIMELINE);
289      }
290      metaTable.scan(scan, new MetaTableScanResultConsumer(rowUpperLimit, visitor, future));
291    }
292
293    return future;
294  }

296  private static final class MetaTableScanResultConsumer implements AdvancedScanResultConsumer {
297
298    private int currentRowCount;
299
300    private final int rowUpperLimit;
301
302    private final Visitor visitor;
303
304    private final CompletableFuture<Void> future;
305
306    MetaTableScanResultConsumer(int rowUpperLimit, Visitor visitor,
307      CompletableFuture<Void> future) {
308      this.rowUpperLimit = rowUpperLimit;
309      this.visitor = visitor;
310      this.future = future;
311      this.currentRowCount = 0;
312    }
313
314    @Override
315    public void onError(Throwable error) {
316      future.completeExceptionally(error);
317    }
318
319    @Override
320    @edu.umd.cs.findbugs.annotations.SuppressWarnings(value = "NP_NONNULL_PARAM_VIOLATION",
321        justification = "https://github.com/findbugsproject/findbugs/issues/79")
322    public void onComplete() {
323      future.complete(null);
324    }
325
326    @Override
327    public void onNext(Result[] results, ScanController controller) {
328      boolean terminateScan = false;
329      for (Result result : results) {
330        try {
331          if (!visitor.visit(result)) {
332            terminateScan = true;
333            break;
334          }
335        } catch (Exception e) {
336          future.completeExceptionally(e);
337          terminateScan = true;
338          break;
339        }
340        if (++currentRowCount >= rowUpperLimit) {
341          terminateScan = true;
342          break;
343        }
344      }
345      if (terminateScan) {
346        controller.terminate();
347      }
348    }
349  }

369  /**
370   * A {@link Visitor} that collects content out of passed {@link Result}.
371   */
372  private static abstract class CollectingVisitor<T> implements Visitor {
373    final List<T> results = new ArrayList<>();
374
375    @Override
376    public boolean visit(Result r) throws IOException {
377      if (r != null && !r.isEmpty()) {
378        add(r);
379      }
380      return true;
381    }
382
383    abstract void add(Result r);
384
385    /** Returns Collected results; wait till visits complete to collect all possible results */
386    List<T> getResults() {
387      return this.results;
388    }
389  }
390
391  static class CollectRegionLocationsVisitor
392    extends CollectingVisitor<Pair<RegionInfo, ServerName>> {
393
394    private final boolean excludeOfflinedSplitParents;
395
396    private RegionLocations current = null;
397
398    CollectRegionLocationsVisitor(boolean excludeOfflinedSplitParents) {
399      this.excludeOfflinedSplitParents = excludeOfflinedSplitParents;
400    }
401
402    @Override
403    public boolean visit(Result r) throws IOException {
404      Optional<RegionLocations> currentRegionLocations = getRegionLocations(r);
405      current = currentRegionLocations.orElse(null);
406      if (current == null || current.getRegionLocation().getRegion() == null) {
407        LOG.warn("No serialized RegionInfo in " + r);
408        return true;
409      }
410      RegionInfo hri = current.getRegionLocation().getRegion();
411      if (excludeOfflinedSplitParents && hri.isSplitParent()) {
412        return true;
413      }
414      // Else call super and add this Result to the collection.
415      return super.visit(r);
416    }
417
418    @Override
419    void add(Result r) {
420      if (current == null) {
421        return;
422      }
423      for (HRegionLocation loc : current.getRegionLocations()) {
424        if (loc != null) {
425          this.results.add(new Pair<RegionInfo, ServerName>(loc.getRegion(), loc.getServerName()));
426        }
427      }
428    }
429  }
430
431  /**
432   * Collects all returned.
433   */
434  static class CollectAllVisitor extends CollectingVisitor<Result> {
435    @Override
436    void add(Result r) {
437      this.results.add(r);
438    }
439  }

441  private static Scan getMetaScan(AsyncTable<?> metaTable, int rowUpperLimit) {
442    Scan scan = new Scan();
443    int scannerCaching = metaTable.getConfiguration().getInt(HConstants.HBASE_META_SCANNER_CACHING,
444      HConstants.DEFAULT_HBASE_META_SCANNER_CACHING);
445    if (
446      metaTable.getConfiguration().getBoolean(HConstants.USE_META_REPLICAS,
447        HConstants.DEFAULT_USE_META_REPLICAS)
448    ) {
449      scan.setConsistency(Consistency.TIMELINE);
450    }
451    if (rowUpperLimit <= scannerCaching) {
452      scan.setLimit(rowUpperLimit);
453    }
454    int rows = Math.min(rowUpperLimit, scannerCaching);
455    scan.setCaching(rows);
456    return scan;
457  }

468  /**
469   * @param tableName table we're working with
470   * @return start row for scanning META according to query type
471   */
472  public static byte[] getTableStartRowForMeta(TableName tableName, QueryType type) {
473    if (tableName == null) {
474      return null;
475    }
476    switch (type) {
477      case REGION:
478      case REPLICATION: {
479        byte[] startRow = new byte[tableName.getName().length + 2];
480        System.arraycopy(tableName.getName(), 0, startRow, 0, tableName.getName().length);
481        startRow[startRow.length - 2] = HConstants.DELIMITER;
482        startRow[startRow.length - 1] = HConstants.DELIMITER;
483        return startRow;
484      }
485      case ALL:
486      case TABLE:
487      default: {
488        return tableName.getName();
489      }
490    }
491  }
492
493  /**
494   * @param tableName table we're working with
495   * @return stop row for scanning META according to query type
496   */
497  public static byte[] getTableStopRowForMeta(TableName tableName, QueryType type) {
498    if (tableName == null) {
499      return null;
500    }
501    final byte[] stopRow;
502    switch (type) {
503      case REGION:
504      case REPLICATION: {
505        stopRow = new byte[tableName.getName().length + 3];
506        System.arraycopy(tableName.getName(), 0, stopRow, 0, tableName.getName().length);
507        stopRow[stopRow.length - 3] = ' ';
508        stopRow[stopRow.length - 2] = HConstants.DELIMITER;
509        stopRow[stopRow.length - 1] = HConstants.DELIMITER;
510        break;
511      }
512      case ALL:
513      case TABLE:
514      default: {
515        stopRow = new byte[tableName.getName().length + 1];
516        System.arraycopy(tableName.getName(), 0, stopRow, 0, tableName.getName().length);
517        stopRow[stopRow.length - 1] = ' ';
518        break;
519      }
520    }
521    return stopRow;
522  }
