040/**
041 * A filter for adding inter-column timestamp matching Only cells with a correspondingly timestamped
042 * entry in the target column will be retained Not compatible with Scan.setBatch as operations need
043 * full rows for correct filtering
044 */
045@InterfaceAudience.Public
046public class DependentColumnFilter extends CompareFilter {
047
048  protected byte[] columnFamily;
049  protected byte[] columnQualifier;
050  protected boolean dropDependentColumn;
051
052  protected Set<Long> stampSet = new HashSet<>();
053
054  /**
055   * Build a dependent column filter with value checking dependent column varies will be compared
056   * using the supplied compareOp and comparator, for usage of which refer to {@link CompareFilter}
057   * @param family              dependent column family
058   * @param qualifier           dependent column qualifier
059   * @param dropDependentColumn whether the column should be discarded after
060   * @param op                  Value comparison op
061   * @param valueComparator     comparator
062   */
063  public DependentColumnFilter(final byte[] family, final byte[] qualifier,
064    final boolean dropDependentColumn, final CompareOperator op,
065    final ByteArrayComparable valueComparator) {
066    // set up the comparator
067    super(op, valueComparator);
068    this.columnFamily = family;
069    this.columnQualifier = qualifier;
070    this.dropDependentColumn = dropDependentColumn;
071  }
072
073  /**
074   * Constructor for DependentColumn filter. Cells where a Cell from target column with the same
075   * timestamp do not exist will be dropped.
076   * @param family    name of target column family
077   * @param qualifier name of column qualifier
078   */
079  public DependentColumnFilter(final byte[] family, final byte[] qualifier) {
080    this(family, qualifier, false);
081  }
082
083  /**
084   * Constructor for DependentColumn filter. Cells where a Cell from target column with the same
085   * timestamp do not exist will be dropped.
086   * @param family              name of dependent column family
087   * @param qualifier           name of dependent qualifier
088   * @param dropDependentColumn whether the dependent columns Cells should be discarded
089   */
090  public DependentColumnFilter(final byte[] family, final byte[] qualifier,
091    final boolean dropDependentColumn) {
092    this(family, qualifier, dropDependentColumn, CompareOperator.NO_OP, null);
093  }
094
095  /**
096   * @return the column family
097   */
098  public byte[] getFamily() {
099    return this.columnFamily;
100  }
101
102  /**
103   * @return the column qualifier
104   */
105  public byte[] getQualifier() {
106    return this.columnQualifier;
107  }
108
109  /**
110   * @return true if we should drop the dependent column, false otherwise
111   */
112  public boolean dropDependentColumn() {
113    return this.dropDependentColumn;
114  }
115
116  public boolean getDropDependentColumn() {
117    return this.dropDependentColumn;
118  }
119
120  @Override
121  public boolean filterAllRemaining() {
122    return false;
123  }
124
125  @Override
126  public ReturnCode filterCell(final Cell c) {
127    // Check if the column and qualifier match
128    if (!CellUtil.matchingColumn(c, this.columnFamily, this.columnQualifier)) {
129      // include non-matches for the time being, they'll be discarded afterwards
130      return ReturnCode.INCLUDE;
131    }
132    // If it doesn't pass the op, skip it
133    if (comparator != null && compareValue(getCompareOperator(), comparator, c))
134      return ReturnCode.SKIP;
135
136    stampSet.add(c.getTimestamp());
137    if (dropDependentColumn) {
138      return ReturnCode.SKIP;
139    }
140    return ReturnCode.INCLUDE;
141  }
142
143  @Override
144  public void filterRowCells(List<Cell> kvs) {
145    kvs.removeIf(kv -> !stampSet.contains(kv.getTimestamp()));
146  }
147
148  @Override
149  public boolean hasFilterRow() {
150    return true;
151  }
152
153  @Override
154  public boolean filterRow() {
155    return false;
156  }
157
158  @Override
159  public void reset() {
160    stampSet.clear();
161  }
162
163  public static Filter createFilterFromArguments(ArrayList<byte[]> filterArguments) {
164    Preconditions.checkArgument(
165      filterArguments.size() == 2 || filterArguments.size() == 3 || filterArguments.size() == 5,
166      "Expected 2, 3 or 5 but got: %s", filterArguments.size());
167    if (filterArguments.size() == 2) {
168      byte[] family = ParseFilter.removeQuotesFromByteArray(filterArguments.get(0));
169      byte[] qualifier = ParseFilter.removeQuotesFromByteArray(filterArguments.get(1));
170      return new DependentColumnFilter(family, qualifier);
171
172    } else if (filterArguments.size() == 3) {
173      byte[] family = ParseFilter.removeQuotesFromByteArray(filterArguments.get(0));
174      byte[] qualifier = ParseFilter.removeQuotesFromByteArray(filterArguments.get(1));
175      boolean dropDependentColumn = ParseFilter.convertByteArrayToBoolean(filterArguments.get(2));
176      return new DependentColumnFilter(family, qualifier, dropDependentColumn);
177
178    } else if (filterArguments.size() == 5) {
179      byte[] family = ParseFilter.removeQuotesFromByteArray(filterArguments.get(0));
180      byte[] qualifier = ParseFilter.removeQuotesFromByteArray(filterArguments.get(1));
181      boolean dropDependentColumn = ParseFilter.convertByteArrayToBoolean(filterArguments.get(2));
182      CompareOperator op = ParseFilter.createCompareOperator(filterArguments.get(3));
183      ByteArrayComparable comparator =
184        ParseFilter.createComparator(ParseFilter.removeQuotesFromByteArray(filterArguments.get(4)));
185      return new DependentColumnFilter(family, qualifier, dropDependentColumn, op, comparator);
186    } else {
187      throw new IllegalArgumentException("Expected 2, 3 or 5 but got: " + filterArguments.size());
188    }
189  }
190
191  /**
192   * @return The filter serialized using pb
193   */
194  @Override
195  public byte[] toByteArray() {
196    FilterProtos.DependentColumnFilter.Builder builder =
197      FilterProtos.DependentColumnFilter.newBuilder();
198    builder.setCompareFilter(super.convert());
199    if (this.columnFamily != null) {
200      builder.setColumnFamily(UnsafeByteOperations.unsafeWrap(this.columnFamily));
201    }
202    if (this.columnQualifier != null) {
203      builder.setColumnQualifier(UnsafeByteOperations.unsafeWrap(this.columnQualifier));
204    }
205    builder.setDropDependentColumn(this.dropDependentColumn);
206    return builder.build().toByteArray();
207  }
208
209  /**
210   * @param pbBytes A pb serialized {@link DependentColumnFilter} instance
211   * @return An instance of {@link DependentColumnFilter} made from <code>bytes</code> n * @see
212   *         #toByteArray
213   */
214  public static DependentColumnFilter parseFrom(final byte[] pbBytes)
215    throws DeserializationException {
216    FilterProtos.DependentColumnFilter proto;
217    try {
218      proto = FilterProtos.DependentColumnFilter.parseFrom(pbBytes);
219    } catch (InvalidProtocolBufferException e) {
220      throw new DeserializationException(e);
221    }
222    final CompareOperator valueCompareOp =
223      CompareOperator.valueOf(proto.getCompareFilter().getCompareOp().name());
224    ByteArrayComparable valueComparator = null;
225    try {
226      if (proto.getCompareFilter().hasComparator()) {
227        valueComparator = ProtobufUtil.toComparator(proto.getCompareFilter().getComparator());
228      }
229    } catch (IOException ioe) {
230      throw new DeserializationException(ioe);
231    }
232    return new DependentColumnFilter(
233      proto.hasColumnFamily() ? proto.getColumnFamily().toByteArray() : null,
234      proto.hasColumnQualifier() ? proto.getColumnQualifier().toByteArray() : null,
235      proto.getDropDependentColumn(), valueCompareOp, valueComparator);
236  }

240   * corresponding fields in other. Used for testing.

243      value = "RCN_REDUNDANT_NULLCHECK_OF_NONNULL_VALUE")
244  @Override

246    if (o == this) return true;
247    if (!(o instanceof DependentColumnFilter)) return false;

250    return other != null && super.areSerializedFieldsEqual(other)
251      && Bytes.equals(this.getFamily(), other.getFamily())
252      && Bytes.equals(this.getQualifier(), other.getQualifier())
253      && this.dropDependentColumn() == other.dropDependentColumn();
254  }

273  }
