018package org.apache.hadoop.hbase.io.hfile;
019
020import java.io.IOException;
021import java.util.Optional;
022import org.apache.commons.lang3.mutable.MutableBoolean;
023import org.apache.hadoop.conf.Configuration;
024import org.apache.hadoop.fs.Path;
025import org.apache.hadoop.hbase.io.FSDataInputStreamWrapper;
026import org.apache.hadoop.hbase.io.hfile.bucket.BucketCache;
027import org.apache.hadoop.hbase.io.hfile.bucket.BucketEntry;
028import org.apache.yetus.audience.InterfaceAudience;
029import org.slf4j.Logger;
030import org.slf4j.LoggerFactory;
031
032/**
033 * Implementation of {@link HFile.Reader} to deal with pread.
034 */
035@InterfaceAudience.Private
036public class HFilePreadReader extends HFileReaderImpl {
037  private static final Logger LOG = LoggerFactory.getLogger(HFileReaderImpl.class);
038
039  public HFilePreadReader(ReaderContext context, HFileInfo fileInfo, CacheConfig cacheConf,
040    Configuration conf) throws IOException {
041    super(context, fileInfo, cacheConf, conf);
042    final MutableBoolean fileAlreadyCached = new MutableBoolean(false);
043    BucketCache.getBuckedCacheFromCacheConfig(cacheConf).ifPresent(bc -> fileAlreadyCached
044      .setValue(bc.getFullyCachedFiles().get(path.getName()) == null ? false : true));
045    // Prefetch file blocks upon open if requested
046    if (
047      cacheConf.shouldPrefetchOnOpen() && cacheIfCompactionsOff()
048        && !fileAlreadyCached.booleanValue()
049    ) {
050      PrefetchExecutor.request(path, new Runnable() {
051        @Override
052        public void run() {
053          long offset = 0;
054          long end = 0;
055          HFile.Reader prefetchStreamReader = null;
056          try {
057            ReaderContext streamReaderContext = ReaderContextBuilder.newBuilder(context)
058              .withReaderType(ReaderContext.ReaderType.STREAM)
059              .withInputStreamWrapper(new FSDataInputStreamWrapper(context.getFileSystem(),
060                context.getInputStreamWrapper().getReaderPath()))
061              .build();
062            prefetchStreamReader =
063              new HFileStreamReader(streamReaderContext, fileInfo, cacheConf, conf);
064            end = getTrailer().getLoadOnOpenDataOffset();
065            if (LOG.isTraceEnabled()) {
066              LOG.trace("Prefetch start " + getPathOffsetEndStr(path, offset, end));
067            }
068            Optional<BucketCache> bucketCacheOptional =
069              BucketCache.getBuckedCacheFromCacheConfig(cacheConf);
070            // Don't use BlockIterator here, because it's designed to read load-on-open section.
071            long onDiskSizeOfNextBlock = -1;
072            while (offset < end) {
073              if (Thread.interrupted()) {
074                break;
075              }
076              // BucketCache can be persistent and resilient to restarts, so we check first if the
077              // block exists on its in-memory index, if so, we just update the offset and move on
078              // to the next block without actually going read all the way to the cache.
079              if (bucketCacheOptional.isPresent()) {
080                BucketCache cache = bucketCacheOptional.get();
081                BlockCacheKey cacheKey = new BlockCacheKey(name, offset);
082                BucketEntry entry = cache.getBackingMap().get(cacheKey);
083                if (entry != null) {
084                  cacheKey = new BlockCacheKey(name, offset);
085                  entry = cache.getBackingMap().get(cacheKey);
086                  if (entry == null) {
087                    LOG.debug("No cache key {}, we'll read and cache it", cacheKey);
088                  } else {
089                    offset += entry.getOnDiskSizeWithHeader();
090                    LOG.debug("Found cache key {}. Skipping prefetch, the block is already cached.",
091                      cacheKey);
092                    continue;
093                  }
094                } else {
095                  LOG.debug("No entry in the backing map for cache key {}", cacheKey);
096                }
097              }
098              // Perhaps we got our block from cache? Unlikely as this may be, if it happens, then
099              // the internal-to-hfileblock thread local which holds the overread that gets the
100              // next header, will not have happened...so, pass in the onDiskSize gotten from the
101              // cached block. This 'optimization' triggers extremely rarely I'd say.
102              HFileBlock block = prefetchStreamReader.readBlock(offset, onDiskSizeOfNextBlock,
103                /* cacheBlock= */true, /* pread= */false, false, false, null, null, true);
104              try {
105                onDiskSizeOfNextBlock = block.getNextBlockOnDiskSize();
106                offset += block.getOnDiskSizeWithHeader();
107              } finally {
108                // Ideally here the readBlock won't find the block in cache. We call this
109                // readBlock so that block data is read from FS and cached in BC. we must call
110                // returnBlock here to decrease the reference count of block.
111                block.release();
112              }
113            }
114            BucketCache.getBuckedCacheFromCacheConfig(cacheConf)
115              .ifPresent(bc -> bc.fileCacheCompleted(path.getName()));
116
117          } catch (IOException e) {
118            // IOExceptions are probably due to region closes (relocation, etc.)
119            if (LOG.isTraceEnabled()) {
120              LOG.trace("Prefetch " + getPathOffsetEndStr(path, offset, end), e);
121            }
122          } catch (Throwable e) {
123            // Other exceptions are interesting
124            LOG.warn("Prefetch " + getPathOffsetEndStr(path, offset, end), e);
125          } finally {
126            if (prefetchStreamReader != null) {
127              try {
128                prefetchStreamReader.close(false);
129              } catch (IOException e) {
130                LOG.warn("Close prefetch stream reader failed, path: " + path, e);
131              }
132            }
133            PrefetchExecutor.complete(path);
134          }
135        }
136      });
137    }
138  }
139
140  private static String getPathOffsetEndStr(final Path path, final long offset, final long end) {
141    return "path=" + path.toString() + ", offset=" + offset + ", end=" + end;
142  }
143
144  public void close(boolean evictOnClose) throws IOException {
145    PrefetchExecutor.cancel(path);
146    // Deallocate blocks in load-on-open section
147    this.fileInfo.close();
148    // Deallocate data blocks
149    cacheConf.getBlockCache().ifPresent(cache -> {
150      if (evictOnClose) {
151        int numEvicted = cache.evictBlocksByHfileName(name);
152        if (LOG.isTraceEnabled()) {
153          LOG.trace("On close, file= {} evicted= {} block(s)", name, numEvicted);
154        }
155      }
156    });
157    fsBlockReader.closeStreams();
158  }
159}
