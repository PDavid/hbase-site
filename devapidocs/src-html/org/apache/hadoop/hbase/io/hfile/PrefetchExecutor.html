047@InterfaceAudience.Private
048public final class PrefetchExecutor {
049
050  private static final Logger LOG = LoggerFactory.getLogger(PrefetchExecutor.class);
051
052  /** Futures for tracking block prefetch activity */
053  private static final Map<Path, Future<?>> prefetchFutures = new ConcurrentSkipListMap<>();
054  /** Set of files for which prefetch is completed */
055  @edu.umd.cs.findbugs.annotations.SuppressWarnings(value = "MS_SHOULD_BE_FINAL")
056  private static HashMap<String, Boolean> prefetchCompleted = new HashMap<>();
057  /** Executor pool shared among all HFiles for block prefetch */
058  private static final ScheduledExecutorService prefetchExecutorPool;
059  /** Delay before beginning prefetch */
060  private static final int prefetchDelayMillis;
061  /** Variation in prefetch delay times, to mitigate stampedes */
062  private static final float prefetchDelayVariation;
063  static String prefetchedFileListPath;
064  static {
065    // Consider doing this on demand with a configuration passed in rather
066    // than in a static initializer.
067    Configuration conf = HBaseConfiguration.create();
068    // 1s here for tests, consider 30s in hbase-default.xml
069    // Set to 0 for no delay
070    prefetchDelayMillis = conf.getInt("hbase.hfile.prefetch.delay", 1000);
071    prefetchDelayVariation = conf.getFloat("hbase.hfile.prefetch.delay.variation", 0.2f);
072    int prefetchThreads = conf.getInt("hbase.hfile.thread.prefetch", 4);
073    prefetchExecutorPool = new ScheduledThreadPoolExecutor(prefetchThreads, new ThreadFactory() {
074      @Override
075      public Thread newThread(Runnable r) {
076        String name = "hfile-prefetch-" + EnvironmentEdgeManager.currentTime();
077        Thread t = new Thread(r, name);
078        t.setDaemon(true);
079        return t;
080      }
081    });
082  }
083
084  // TODO: We want HFile, which is where the blockcache lives, to handle
085  // prefetching of file blocks but the Store level is where path convention
086  // knowledge should be contained
087  private static final Pattern prefetchPathExclude =
088    Pattern.compile("(" + Path.SEPARATOR_CHAR + HConstants.HBASE_TEMP_DIRECTORY.replace(".", "\\.")
089      + Path.SEPARATOR_CHAR + ")|(" + Path.SEPARATOR_CHAR
090      + HConstants.HREGION_COMPACTIONDIR_NAME.replace(".", "\\.") + Path.SEPARATOR_CHAR + ")");
091
092  public static void request(Path path, Runnable runnable) {
093    if (prefetchCompleted != null) {
094      if (isFilePrefetched(path.getName())) {
095        LOG.info(
096          "File has already been prefetched before the restart, so skipping prefetch : " + path);
097        return;
098      }
099    }
100    if (!prefetchPathExclude.matcher(path.toString()).find()) {
101      long delay;
102      if (prefetchDelayMillis > 0) {
103        delay = (long) ((prefetchDelayMillis * (1.0f - (prefetchDelayVariation / 2)))
104          + (prefetchDelayMillis * (prefetchDelayVariation / 2)
105            * ThreadLocalRandom.current().nextFloat()));
106      } else {
107        delay = 0;
108      }
109      try {
110        LOG.debug("Prefetch requested for {}, delay={} ms", path, delay);
111        final Runnable tracedRunnable =
112          TraceUtil.tracedRunnable(runnable, "PrefetchExecutor.request");
113        final Future<?> future =
114          prefetchExecutorPool.schedule(tracedRunnable, delay, TimeUnit.MILLISECONDS);
115        prefetchFutures.put(path, future);
116      } catch (RejectedExecutionException e) {
117        prefetchFutures.remove(path);
118        LOG.warn("Prefetch request rejected for {}", path);
119      }
120    }
121  }
122
123  public static void complete(Path path) {
124    prefetchFutures.remove(path);
125    prefetchCompleted.put(path.getName(), true);
126    LOG.debug("Prefetch completed for {}", path);
127  }
128
129  public static void cancel(Path path) {
130    Future<?> future = prefetchFutures.get(path);
131    if (future != null) {
132      // ok to race with other cancellation attempts
133      future.cancel(true);
134      prefetchFutures.remove(path);
135      LOG.debug("Prefetch cancelled for {}", path);
136    }
137    prefetchCompleted.remove(path.getName());
138  }
139
140  public static boolean isCompleted(Path path) {
141    Future<?> future = prefetchFutures.get(path);
142    if (future != null) {
143      return future.isDone();
144    }
145    return true;
146  }
147
148  @edu.umd.cs.findbugs.annotations.SuppressWarnings(value = "OBL_UNSATISFIED_OBLIGATION",
149      justification = "false positive, try-with-resources ensures close is called.")
150  public static void persistToFile(String path) throws IOException {
151    prefetchedFileListPath = path;
152    if (prefetchedFileListPath == null) {
153      LOG.info("Exception while persisting prefetch!");
154      throw new IOException("Error persisting prefetched HFiles set!");
155    }
156    if (!prefetchCompleted.isEmpty()) {
157      try (FileOutputStream fos = new FileOutputStream(prefetchedFileListPath, true)) {
158        PrefetchProtoUtils.toPB(prefetchCompleted).writeDelimitedTo(fos);
159      }
160    }
161  }
162
163  public static void retrieveFromFile(String path) throws IOException {
164    prefetchedFileListPath = path;
165    File prefetchPersistenceFile = new File(prefetchedFileListPath);
166    if (!prefetchPersistenceFile.exists()) {
167      LOG.warn("Prefetch persistence file does not exist!");
168      return;
169    }
170    LOG.info("Retrieving from prefetch persistence file " + path);
171    assert (prefetchedFileListPath != null);
172    try (FileInputStream fis = deleteFileOnClose(prefetchPersistenceFile)) {
173      PersistentPrefetchProtos.PrefetchedHfileName proto =
174        PersistentPrefetchProtos.PrefetchedHfileName.parseDelimitedFrom(fis);
175      Map<String, Boolean> protoPrefetchedFilesMap = proto.getPrefetchedFilesMap();
176      prefetchCompleted.putAll(protoPrefetchedFilesMap);
177    }
178  }
179
180  private static FileInputStream deleteFileOnClose(final File file) throws IOException {
181    return new FileInputStream(file) {
182      private File myFile;
183
184      private FileInputStream init(File file) {
185        myFile = file;
186        return this;
187      }
188
189      @Override
190      public void close() throws IOException {
191        if (myFile == null) {
192          return;
193        }
194
195        super.close();
196        if (!myFile.delete()) {
197          throw new IOException("Failed deleting persistence file " + myFile.getAbsolutePath());
198        }
199        myFile = null;
200      }
201    }.init(file);
202  }
203
204  public static void removePrefetchedFileWhileEvict(String hfileName) {
205    prefetchCompleted.remove(hfileName);
206  }
207
208  public static boolean isFilePrefetched(String hfileName) {
209    return prefetchCompleted.containsKey(hfileName);
210  }
211
212  private PrefetchExecutor() {
213  }
