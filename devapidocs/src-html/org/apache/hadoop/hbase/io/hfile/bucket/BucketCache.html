096/**
097 * BucketCache uses {@link BucketAllocator} to allocate/free blocks, and uses BucketCache#ramCache
098 * and BucketCache#backingMap in order to determine if a given element is in the cache. The bucket
099 * cache can use off-heap memory {@link ByteBufferIOEngine} or mmap
100 * {@link ExclusiveMemoryMmapIOEngine} or pmem {@link SharedMemoryMmapIOEngine} or local files
101 * {@link FileIOEngine} to store/read the block data.
102 * <p>
103 * Eviction is via a similar algorithm as used in
104 * {@link org.apache.hadoop.hbase.io.hfile.LruBlockCache}
105 * <p>
106 * BucketCache can be used as mainly a block cache (see
107 * {@link org.apache.hadoop.hbase.io.hfile.CombinedBlockCache}), combined with a BlockCache to
108 * decrease CMS GC and heap fragmentation.
109 * <p>
110 * It also can be used as a secondary cache (e.g. using a file on ssd/fusionio to store blocks) to
111 * enlarge cache space via a victim cache.
112 */

283  public BucketCache(String ioEngineName, long capacity, int blockSize, int[] bucketSizes,
284    int writerThreadNum, int writerQLen, String persistencePath, int ioErrorsTolerationDuration,
285    Configuration conf) throws IOException {
286    Preconditions.checkArgument(blockSize > 0,
287      "BucketCache capacity is set to " + blockSize + ", can not be less than 0");
288    boolean useStrongRef = conf.getBoolean(STRONG_REF_KEY, STRONG_REF_DEFAULT);
289    if (useStrongRef) {
290      this.offsetLock = new IdReadWriteLockStrongRef<>();
291    } else {
292      this.offsetLock = new IdReadWriteLockWithObjectPool<>(ReferenceType.SOFT);
293    }
294    this.algorithm = conf.get(FILE_VERIFY_ALGORITHM, DEFAULT_FILE_VERIFY_ALGORITHM);
295    this.ioEngine = getIOEngineFromName(ioEngineName, capacity, persistencePath);
296    this.writerThreads = new WriterThread[writerThreadNum];
297    long blockNumCapacity = capacity / blockSize;
298    if (blockNumCapacity >= Integer.MAX_VALUE) {
299      // Enough for about 32TB of cache!
300      throw new IllegalArgumentException("Cache capacity is too large, only support 32TB now");
301    }
302
303    this.acceptableFactor = conf.getFloat(ACCEPT_FACTOR_CONFIG_NAME, DEFAULT_ACCEPT_FACTOR);
304    this.minFactor = conf.getFloat(MIN_FACTOR_CONFIG_NAME, DEFAULT_MIN_FACTOR);
305    this.extraFreeFactor = conf.getFloat(EXTRA_FREE_FACTOR_CONFIG_NAME, DEFAULT_EXTRA_FREE_FACTOR);
306    this.singleFactor = conf.getFloat(SINGLE_FACTOR_CONFIG_NAME, DEFAULT_SINGLE_FACTOR);
307    this.multiFactor = conf.getFloat(MULTI_FACTOR_CONFIG_NAME, DEFAULT_MULTI_FACTOR);
308    this.memoryFactor = conf.getFloat(MEMORY_FACTOR_CONFIG_NAME, DEFAULT_MEMORY_FACTOR);
309    this.queueAdditionWaitTime =
310      conf.getLong(QUEUE_ADDITION_WAIT_TIME, DEFAULT_QUEUE_ADDITION_WAIT_TIME);
311    this.bucketcachePersistInterval = conf.getLong(BUCKETCACHE_PERSIST_INTERVAL_KEY, 1000);
312
313    sanityCheckConfigs();
314
315    LOG.info("Instantiating BucketCache with acceptableFactor: " + acceptableFactor
316      + ", minFactor: " + minFactor + ", extraFreeFactor: " + extraFreeFactor + ", singleFactor: "
317      + singleFactor + ", multiFactor: " + multiFactor + ", memoryFactor: " + memoryFactor
318      + ", useStrongRef: " + useStrongRef);
319
320    this.cacheCapacity = capacity;
321    this.persistencePath = persistencePath;
322    this.blockSize = blockSize;
323    this.ioErrorsTolerationDuration = ioErrorsTolerationDuration;
324
325    this.allocFailLogPrevTs = 0;
326
327    for (int i = 0; i < writerThreads.length; ++i) {
328      writerQueues.add(new ArrayBlockingQueue<>(writerQLen));
329    }
330
331    assert writerQueues.size() == writerThreads.length;
332    this.ramCache = new RAMCache();
333
334    this.backingMap = new ConcurrentHashMap<>((int) blockNumCapacity);
335
336    if (isCachePersistent()) {
337      if (ioEngine instanceof FileIOEngine) {
338        startBucketCachePersisterThread();
339      }
340      try {
341        retrieveFromFile(bucketSizes);
342      } catch (IOException ioex) {
343        LOG.error("Can't restore from file[{}] because of ", persistencePath, ioex);
344        backingMap.clear();
345        fullyCachedFiles.clear();
346        backingMapValidated.set(true);
347        bucketAllocator = new BucketAllocator(capacity, bucketSizes);
348        regionCachedSize.clear();
349      }
350    } else {
351      bucketAllocator = new BucketAllocator(capacity, bucketSizes);
352    }
353    final String threadName = Thread.currentThread().getName();
354    this.cacheEnabled = true;
355    for (int i = 0; i < writerThreads.length; ++i) {
356      writerThreads[i] = new WriterThread(writerQueues.get(i));
357      writerThreads[i].setName(threadName + "-BucketCacheWriter-" + i);
358      writerThreads[i].setDaemon(true);
359    }
360    startWriterThreads();
361
362    // Run the statistics thread periodically to print the cache statistics log
363    // TODO: Add means of turning this off. Bit obnoxious running thread just to make a log
364    // every five minutes.
365    this.scheduleThreadPool.scheduleAtFixedRate(new StatisticsThread(this), statThreadPeriod,
366      statThreadPeriod, TimeUnit.SECONDS);
367    LOG.info("Started bucket cache; ioengine=" + ioEngineName + ", capacity="
368      + StringUtils.byteDesc(capacity) + ", blockSize=" + StringUtils.byteDesc(blockSize)
369      + ", writerThreadNum=" + writerThreadNum + ", writerQLen=" + writerQLen + ", persistencePath="
370      + persistencePath + ", bucketAllocator=" + this.bucketAllocator.getClass().getName());
371  }

423  /**
424   * Get the IOEngine from the IO engine name
425   * @return the IOEngine
426   */
427  private IOEngine getIOEngineFromName(String ioEngineName, long capacity, String persistencePath)
428    throws IOException {
429    if (ioEngineName.startsWith("file:") || ioEngineName.startsWith("files:")) {
430      // In order to make the usage simple, we only need the prefix 'files:' in
431      // document whether one or multiple file(s), but also support 'file:' for
432      // the compatibility
433      String[] filePaths =
434        ioEngineName.substring(ioEngineName.indexOf(":") + 1).split(FileIOEngine.FILE_DELIMITER);
435      return new FileIOEngine(capacity, persistencePath != null, filePaths);
436    } else if (ioEngineName.startsWith("offheap")) {
437      return new ByteBufferIOEngine(capacity);
438    } else if (ioEngineName.startsWith("mmap:")) {
439      return new ExclusiveMemoryMmapIOEngine(ioEngineName.substring(5), capacity);
440    } else if (ioEngineName.startsWith("pmem:")) {
441      // This mode of bucket cache creates an IOEngine over a file on the persistent memory
442      // device. Since the persistent memory device has its own address space the contents
443      // mapped to this address space does not get swapped out like in the case of mmapping
444      // on to DRAM. Hence the cells created out of the hfile blocks in the pmem bucket cache
445      // can be directly referred to without having to copy them onheap. Once the RPC is done,
446      // the blocks can be returned back as in case of ByteBufferIOEngine.
447      return new SharedMemoryMmapIOEngine(ioEngineName.substring(5), capacity);
448    } else {
449      throw new IllegalArgumentException(
450        "Don't understand io engine name for cache- prefix with file:, files:, mmap: or offheap");
451    }
452  }

491  /**
492   * Cache the block to ramCache
493   * @param cacheKey   block's cache key
494   * @param cachedItem block buffer
495   * @param inMemory   if block is in-memory
496   * @param wait       if true, blocking wait when queue is full
497   */
498  public void cacheBlockWithWait(BlockCacheKey cacheKey, Cacheable cachedItem, boolean inMemory,
499    boolean wait) {
500    if (cacheEnabled) {
501      if (backingMap.containsKey(cacheKey) || ramCache.containsKey(cacheKey)) {
502        if (shouldReplaceExistingCacheBlock(cacheKey, cachedItem)) {
503          BucketEntry bucketEntry = backingMap.get(cacheKey);
504          if (bucketEntry != null && bucketEntry.isRpcRef()) {
505            // avoid replace when there are RPC refs for the bucket entry in bucket cache
506            return;
507          }
508          cacheBlockWithWaitInternal(cacheKey, cachedItem, inMemory, wait);
509        }
510      } else {
511        cacheBlockWithWaitInternal(cacheKey, cachedItem, inMemory, wait);
512      }
513    }
514  }
515
516  protected boolean shouldReplaceExistingCacheBlock(BlockCacheKey cacheKey, Cacheable newBlock) {
517    return BlockCacheUtil.shouldReplaceExistingCacheBlock(this, cacheKey, newBlock);
518  }
519
520  protected void cacheBlockWithWaitInternal(BlockCacheKey cacheKey, Cacheable cachedItem,
521    boolean inMemory, boolean wait) {
522    if (!cacheEnabled) {
523      return;
524    }
525    if (cacheKey.getBlockType() == null && cachedItem.getBlockType() != null) {
526      cacheKey.setBlockType(cachedItem.getBlockType());
527    }
528    LOG.trace("Caching key={}, item={}", cacheKey, cachedItem);
529    // Stuff the entry into the RAM cache so it can get drained to the persistent store
530    RAMQueueEntry re = new RAMQueueEntry(cacheKey, cachedItem, accessCount.incrementAndGet(),
531      inMemory, isCachePersistent() && ioEngine instanceof FileIOEngine);
532    /**
533     * Don't use ramCache.put(cacheKey, re) here. because there may be a existing entry with same
534     * key in ramCache, the heap size of bucket cache need to update if replacing entry from
535     * ramCache. But WriterThread will also remove entry from ramCache and update heap size, if
536     * using ramCache.put(), It's possible that the removed entry in WriterThread is not the correct
537     * one, then the heap size will mess up (HBASE-20789)
538     */
539    if (ramCache.putIfAbsent(cacheKey, re) != null) {
540      return;
541    }
542    int queueNum = (cacheKey.hashCode() & 0x7FFFFFFF) % writerQueues.size();
543    BlockingQueue<RAMQueueEntry> bq = writerQueues.get(queueNum);
544    boolean successfulAddition = false;
545    if (wait) {
546      try {
547        successfulAddition = bq.offer(re, queueAdditionWaitTime, TimeUnit.MILLISECONDS);
548      } catch (InterruptedException e) {
549        Thread.currentThread().interrupt();
550      }
551    } else {
552      successfulAddition = bq.offer(re);
553    }
554    if (!successfulAddition) {
555      ramCache.remove(cacheKey);
556      cacheStats.failInsert();
557    } else {
558      this.blockNumber.increment();
559      this.heapSize.add(cachedItem.heapSize());
560    }
561  }

567   * @param repeat             Whether this is a repeat lookup for the same block
568   * @param updateCacheMetrics Whether we should update cache metrics or not
569   * @return buffer of specified cache key, or null if not in cache
570   */
571  @Override
572  public Cacheable getBlock(BlockCacheKey key, boolean caching, boolean repeat,
573    boolean updateCacheMetrics) {
574    if (!cacheEnabled) {
575      return null;
576    }
577    RAMQueueEntry re = ramCache.get(key);
578    if (re != null) {
579      if (updateCacheMetrics) {
580        cacheStats.hit(caching, key.isPrimary(), key.getBlockType());
581      }
582      re.access(accessCount.incrementAndGet());
583      return re.getData();
584    }
585    BucketEntry bucketEntry = backingMap.get(key);
586    if (bucketEntry != null) {
587      long start = System.nanoTime();
588      ReentrantReadWriteLock lock = offsetLock.getLock(bucketEntry.offset());
589      try {
590        lock.readLock().lock();
591        // We can not read here even if backingMap does contain the given key because its offset
592        // maybe changed. If we lock BlockCacheKey instead of offset, then we can only check
593        // existence here.
594        if (bucketEntry.equals(backingMap.get(key))) {
595          // Read the block from IOEngine based on the bucketEntry's offset and length, NOTICE: the
596          // block will use the refCnt of bucketEntry, which means if two HFileBlock mapping to
597          // the same BucketEntry, then all of the three will share the same refCnt.
598          Cacheable cachedBlock = ioEngine.read(bucketEntry);
599          if (ioEngine.usesSharedMemory()) {
600            // If IOEngine use shared memory, cachedBlock and BucketEntry will share the
601            // same RefCnt, do retain here, in order to count the number of RPC references
602            cachedBlock.retain();
603          }
604          // Update the cache statistics.
605          if (updateCacheMetrics) {
606            cacheStats.hit(caching, key.isPrimary(), key.getBlockType());
607            cacheStats.ioHit(System.nanoTime() - start);
608          }
609          bucketEntry.access(accessCount.incrementAndGet());
610          if (this.ioErrorStartTime > 0) {
611            ioErrorStartTime = -1;
612          }
613          return cachedBlock;
614        }
615      } catch (HBaseIOException hioex) {
616        // When using file io engine persistent cache,
617        // the cache map state might differ from the actual cache. If we reach this block,
618        // we should remove the cache key entry from the backing map
619        backingMap.remove(key);
620        fullyCachedFiles.remove(key.getHfileName());
621        LOG.debug("Failed to fetch block for cache key: {}.", key, hioex);
622      } catch (IOException ioex) {
623        LOG.error("Failed reading block " + key + " from bucket cache", ioex);
624        checkIOErrorIsTolerated();
625      } finally {
626        lock.readLock().unlock();
627      }
628    }
629    if (!repeat && updateCacheMetrics) {
630      cacheStats.miss(caching, key.isPrimary(), key.getBlockType());
631    }
632    return null;
633  }

636   * This method is invoked after the bucketEntry is removed from {@link BucketCache#backingMap}
637   */
638  void blockEvicted(BlockCacheKey cacheKey, BucketEntry bucketEntry, boolean decrementBlockNumber,
639    boolean evictedByEvictionProcess) {
640    bucketEntry.markAsEvicted();
641    blocksByHFile.remove(cacheKey);
642    if (decrementBlockNumber) {
643      this.blockNumber.decrement();
644      if (ioEngine.isPersistent()) {
645        fileNotFullyCached(cacheKey.getHfileName());
646      }
647    }
648    if (evictedByEvictionProcess) {
649      cacheStats.evicted(bucketEntry.getCachedTime(), cacheKey.isPrimary());
650    }
651    if (ioEngine.isPersistent()) {
652      setCacheInconsistent(true);
653    }
654  }

657    // Update the regionPrefetchedSizeMap before removing the file from prefetchCompleted
658    if (fullyCachedFiles.containsKey(hfileName)) {

660      String regionEncodedName = regionEntry.getFirst();
661      long filePrefetchSize = regionEntry.getSecond();
662      LOG.debug("Removing file {} for region {}", hfileName, regionEncodedName);
663      regionCachedSize.computeIfPresent(regionEncodedName, (rn, pf) -> pf - filePrefetchSize);
664      // If all the blocks for a region are evicted from the cache, remove the entry for that region
665      if (
666        regionCachedSize.containsKey(regionEncodedName)
667          && regionCachedSize.get(regionEncodedName) == 0
668      ) {
669        regionCachedSize.remove(regionEncodedName);
670      }
671    }
672    fullyCachedFiles.remove(hfileName);
673  }

702   * Try to evict the block from {@link BlockCache} by force. We'll call this in few cases:<br>
703   * 1. Close an HFile, and clear all cached blocks. <br>
704   * 2. Call {@link Admin#clearBlockCache(TableName)} to clear all blocks for a given table.<br>
705   * <p>
706   * Firstly, we'll try to remove the block from RAMCache,and then try to evict from backingMap.
707   * Here we evict the block from backingMap immediately, but only free the reference from bucket
708   * cache by calling {@link BucketEntry#markedAsEvicted}. If there're still some RPC referring this
709   * block, block can only be de-allocated when all of them release the block.
710   * <p>
711   * NOTICE: we need to grab the write offset lock firstly before releasing the reference from
712   * bucket cache. if we don't, we may read an {@link BucketEntry} with refCnt = 0 when
713   * {@link BucketCache#getBlock(BlockCacheKey, boolean, boolean, boolean)}, it's a memory leak.
714   * @param cacheKey Block to evict
715   * @return true to indicate whether we've evicted successfully or not.
716   */

722  /**
723   * Evict the {@link BlockCacheKey} and {@link BucketEntry} from {@link BucketCache#backingMap} and
724   * {@link BucketCache#ramCache}. <br/>
725   * NOTE:When Evict from {@link BucketCache#backingMap},only the matched {@link BlockCacheKey} and
726   * {@link BucketEntry} could be removed.
727   * @param cacheKey    {@link BlockCacheKey} to evict.
728   * @param bucketEntry {@link BucketEntry} matched {@link BlockCacheKey} to evict.
729   * @return true to indicate whether we've evicted successfully or not.
730   */
731  private boolean doEvictBlock(BlockCacheKey cacheKey, BucketEntry bucketEntry,
732    boolean evictedByEvictionProcess) {
733    if (!cacheEnabled) {
734      return false;
735    }
736    boolean existedInRamCache = removeFromRamCache(cacheKey);
737    if (bucketEntry == null) {
738      bucketEntry = backingMap.get(cacheKey);
739    }
740    final BucketEntry bucketEntryToUse = bucketEntry;
741
742    if (bucketEntryToUse == null) {
743      if (existedInRamCache && evictedByEvictionProcess) {
744        cacheStats.evicted(0, cacheKey.isPrimary());
745      }
746      return existedInRamCache;
747    } else {
748      return bucketEntryToUse.withWriteLock(offsetLock, () -> {
749        if (backingMap.remove(cacheKey, bucketEntryToUse)) {
750          LOG.debug("removed key {} from back map in the evict process", cacheKey);
751          blockEvicted(cacheKey, bucketEntryToUse, !existedInRamCache, evictedByEvictionProcess);
752          return true;
753        }
754        return false;
755      });
756    }
757  }

883   */
884  private int bucketSizesAboveThresholdCount(float minFactor) {
885    BucketAllocator.IndexStatistics[] stats = bucketAllocator.getIndexStatistics();
886    int fullCount = 0;

888      long freeGoal = (long) Math.floor(stats[i].totalCount() * (1 - minFactor));

897  /**
898   * This method will find the buckets that are minimally occupied and are not reference counted and
899   * will free them completely without any constraint on the access times of the elements, and as a
900   * process will completely free at most the number of buckets passed, sometimes it might not due
901   * to changing refCounts
902   * @param completelyFreeBucketsNeeded number of buckets to free
903   **/
904  private void freeEntireBuckets(int completelyFreeBucketsNeeded) {
905    if (completelyFreeBucketsNeeded != 0) {
906      // First we will build a set where the offsets are reference counted, usually
907      // this set is small around O(Handler Count) unless something else is wrong
908      Set<Integer> inUseBuckets = new HashSet<>();
909      backingMap.forEach((k, be) -> {
910        if (be.isRpcRef()) {
911          inUseBuckets.add(bucketAllocator.getBucketIndex(be.offset()));
912        }
913      });
914      Set<Integer> candidateBuckets =
915        bucketAllocator.getLeastFilledBuckets(inUseBuckets, completelyFreeBucketsNeeded);
916      for (Map.Entry<BlockCacheKey, BucketEntry> entry : backingMap.entrySet()) {
917        if (candidateBuckets.contains(bucketAllocator.getBucketIndex(entry.getValue().offset()))) {
918          evictBucketEntryIfNoRpcReferenced(entry.getKey(), entry.getValue());
919        }
920      }
921    }
922  }

924  /**
925   * Free the space if the used size reaches acceptableSize() or one size block couldn't be
926   * allocated. When freeing the space, we use the LRU algorithm and ensure there must be some
927   * blocks evicted
928   * @param why Why we are being called
929   */
930  void freeSpace(final String why) {
931    // Ensure only one freeSpace progress at a time
932    if (!freeSpaceLock.tryLock()) {
933      return;
934    }
935    try {
936      freeInProgress = true;
937      long bytesToFreeWithoutExtra = 0;
938      // Calculate free byte for each bucketSizeinfo
939      StringBuilder msgBuffer = LOG.isDebugEnabled() ? new StringBuilder() : null;
940      BucketAllocator.IndexStatistics[] stats = bucketAllocator.getIndexStatistics();
941      long[] bytesToFreeForBucket = new long[stats.length];
942      for (int i = 0; i < stats.length; i++) {
943        bytesToFreeForBucket[i] = 0;
944        long freeGoal = (long) Math.floor(stats[i].totalCount() * (1 - minFactor));
945        freeGoal = Math.max(freeGoal, 1);
946        if (stats[i].freeCount() < freeGoal) {
947          bytesToFreeForBucket[i] = stats[i].itemSize() * (freeGoal - stats[i].freeCount());
948          bytesToFreeWithoutExtra += bytesToFreeForBucket[i];
949          if (msgBuffer != null) {
950            msgBuffer.append("Free for bucketSize(" + stats[i].itemSize() + ")="
951              + StringUtils.byteDesc(bytesToFreeForBucket[i]) + ", ");
952          }
953        }
954      }
955      if (msgBuffer != null) {
956        msgBuffer.append("Free for total=" + StringUtils.byteDesc(bytesToFreeWithoutExtra) + ", ");
957      }
958
959      if (bytesToFreeWithoutExtra <= 0) {
960        return;
961      }
962      long currentSize = bucketAllocator.getUsedSize();
963      long totalSize = bucketAllocator.getTotalSize();
964      if (LOG.isDebugEnabled() && msgBuffer != null) {
965        LOG.debug("Free started because \"" + why + "\"; " + msgBuffer.toString()
966          + " of current used=" + StringUtils.byteDesc(currentSize) + ", actual cacheSize="
967          + StringUtils.byteDesc(realCacheSize.sum()) + ", total="
968          + StringUtils.byteDesc(totalSize));
969      }
970
971      long bytesToFreeWithExtra =
972        (long) Math.floor(bytesToFreeWithoutExtra * (1 + extraFreeFactor));
973
974      // Instantiate priority buckets
975      BucketEntryGroup bucketSingle =
976        new BucketEntryGroup(bytesToFreeWithExtra, blockSize, getPartitionSize(singleFactor));
977      BucketEntryGroup bucketMulti =
978        new BucketEntryGroup(bytesToFreeWithExtra, blockSize, getPartitionSize(multiFactor));
979      BucketEntryGroup bucketMemory =
980        new BucketEntryGroup(bytesToFreeWithExtra, blockSize, getPartitionSize(memoryFactor));
981
982      // Scan entire map putting bucket entry into appropriate bucket entry
983      // group
984      for (Map.Entry<BlockCacheKey, BucketEntry> bucketEntryWithKey : backingMap.entrySet()) {
985        switch (bucketEntryWithKey.getValue().getPriority()) {
986          case SINGLE: {
987            bucketSingle.add(bucketEntryWithKey);
988            break;
989          }
990          case MULTI: {
991            bucketMulti.add(bucketEntryWithKey);
992            break;
993          }
994          case MEMORY: {
995            bucketMemory.add(bucketEntryWithKey);
996            break;
997          }
998        }
999      }
1000
1001      PriorityQueue<BucketEntryGroup> bucketQueue =
1002        new PriorityQueue<>(3, Comparator.comparingLong(BucketEntryGroup::overflow));
1003
1004      bucketQueue.add(bucketSingle);
1005      bucketQueue.add(bucketMulti);
1006      bucketQueue.add(bucketMemory);
1007
1008      int remainingBuckets = bucketQueue.size();
1009      long bytesFreed = 0;
1010
1011      BucketEntryGroup bucketGroup;
1012      while ((bucketGroup = bucketQueue.poll()) != null) {
1013        long overflow = bucketGroup.overflow();
1014        if (overflow > 0) {
1015          long bucketBytesToFree =
1016            Math.min(overflow, (bytesToFreeWithoutExtra - bytesFreed) / remainingBuckets);
1017          bytesFreed += bucketGroup.free(bucketBytesToFree);
1018        }
1019        remainingBuckets--;
1020      }
1021
1022      // Check and free if there are buckets that still need freeing of space
1023      if (bucketSizesAboveThresholdCount(minFactor) > 0) {
1024        bucketQueue.clear();
1025        remainingBuckets = 3;
1026
1027        bucketQueue.add(bucketSingle);
1028        bucketQueue.add(bucketMulti);
1029        bucketQueue.add(bucketMemory);
1030
1031        while ((bucketGroup = bucketQueue.poll()) != null) {
1032          long bucketBytesToFree = (bytesToFreeWithExtra - bytesFreed) / remainingBuckets;
1033          bytesFreed += bucketGroup.free(bucketBytesToFree);
1034          remainingBuckets--;
1035        }
1036      }
1037
1038      // Even after the above free we might still need freeing because of the
1039      // De-fragmentation of the buckets (also called Slab Calcification problem), i.e
1040      // there might be some buckets where the occupancy is very sparse and thus are not
1041      // yielding the free for the other bucket sizes, the fix for this to evict some
1042      // of the buckets, we do this by evicting the buckets that are least fulled
1043      freeEntireBuckets(DEFAULT_FREE_ENTIRE_BLOCK_FACTOR * bucketSizesAboveThresholdCount(1.0f));
1044
1045      if (LOG.isDebugEnabled()) {
1046        long single = bucketSingle.totalSize();
1047        long multi = bucketMulti.totalSize();
1048        long memory = bucketMemory.totalSize();
1049        if (LOG.isDebugEnabled()) {
1050          LOG.debug("Bucket cache free space completed; " + "freed="
1051            + StringUtils.byteDesc(bytesFreed) + ", " + "total=" + StringUtils.byteDesc(totalSize)
1052            + ", " + "single=" + StringUtils.byteDesc(single) + ", " + "multi="
1053            + StringUtils.byteDesc(multi) + ", " + "memory=" + StringUtils.byteDesc(memory));
1054        }
1055      }
1056
1057    } catch (Throwable t) {
1058      LOG.warn("Failed freeing space", t);
1059    } finally {
1060      cacheStats.evict();
1061      freeInProgress = false;
1062      freeSpaceLock.unlock();
1063    }
1064  }

1173  /**
1174   * Flush the entries in ramCache to IOEngine and add bucket entry to backingMap. Process all that
1175   * are passed in even if failure being sure to remove from ramCache else we'll never undo the
1176   * references and we'll OOME.
1177   * @param entries Presumes list passed in here will be processed by this invocation only. No
1178   *                interference expected.
1179   */
1180  void doDrain(final List<RAMQueueEntry> entries, ByteBuffer metaBuff) throws InterruptedException {
1181    if (entries.isEmpty()) {
1182      return;
1183    }
1184    // This method is a little hard to follow. We run through the passed in entries and for each
1185    // successful add, we add a non-null BucketEntry to the below bucketEntries. Later we must
1186    // do cleanup making sure we've cleared ramCache of all entries regardless of whether we
1187    // successfully added the item to the bucketcache; if we don't do the cleanup, we'll OOME by
1188    // filling ramCache. We do the clean up by again running through the passed in entries
1189    // doing extra work when we find a non-null bucketEntries corresponding entry.
1190    final int size = entries.size();
1191    BucketEntry[] bucketEntries = new BucketEntry[size];
1192    // Index updated inside loop if success or if we can't succeed. We retry if cache is full
1193    // when we go to add an entry by going around the loop again without upping the index.
1194    int index = 0;
1195    while (cacheEnabled && index < size) {
1196      RAMQueueEntry re = null;
1197      try {
1198        re = entries.get(index);
1199        if (re == null) {
1200          LOG.warn("Couldn't get entry or changed on us; who else is messing with it?");
1201          index++;
1202          continue;
1203        }
1204        // Reset the position for reuse.
1205        // It should be guaranteed that the data in the metaBuff has been transferred to the
1206        // ioEngine safely. Otherwise, this reuse is problematic. Fortunately, the data is already
1207        // transferred with our current IOEngines. Should take care, when we have new kinds of
1208        // IOEngine in the future.
1209        metaBuff.clear();
1210        BucketEntry bucketEntry =
1211          re.writeToCache(ioEngine, bucketAllocator, realCacheSize, this::createRecycler, metaBuff);
1212        // Successfully added. Up index and add bucketEntry. Clear io exceptions.
1213        bucketEntries[index] = bucketEntry;
1214        if (ioErrorStartTime > 0) {
1215          ioErrorStartTime = -1;
1216        }
1217        index++;
1218      } catch (BucketAllocatorException fle) {
1219        long currTs = EnvironmentEdgeManager.currentTime();
1220        cacheStats.allocationFailed(); // Record the warning.
1221        if (
1222          allocFailLogPrevTs == 0 || (currTs - allocFailLogPrevTs) > ALLOCATION_FAIL_LOG_TIME_PERIOD
1223        ) {
1224          LOG.warn(getAllocationFailWarningMessage(fle, re));
1225          allocFailLogPrevTs = currTs;
1226        }
1227        // Presume can't add. Too big? Move index on. Entry will be cleared from ramCache below.
1228        bucketEntries[index] = null;
1229        index++;
1230      } catch (CacheFullException cfe) {
1231        // Cache full when we tried to add. Try freeing space and then retrying (don't up index)
1232        if (!freeInProgress) {
1233          freeSpace("Full!");
1234        } else {
1235          Thread.sleep(50);
1236        }
1237      } catch (IOException ioex) {
1238        // Hopefully transient. Retry. checkIOErrorIsTolerated disables cache if problem.
1239        LOG.error("Failed writing to bucket cache", ioex);
1240        checkIOErrorIsTolerated();
1241      }
1242    }
1243
1244    // Make sure data pages are written on media before we update maps.
1245    try {
1246      ioEngine.sync();
1247    } catch (IOException ioex) {
1248      LOG.error("Failed syncing IO engine", ioex);
1249      checkIOErrorIsTolerated();
1250      // Since we failed sync, free the blocks in bucket allocator
1251      for (int i = 0; i < entries.size(); ++i) {
1252        BucketEntry bucketEntry = bucketEntries[i];
1253        if (bucketEntry != null) {
1254          bucketAllocator.freeBlock(bucketEntry.offset(), bucketEntry.getLength());
1255          bucketEntries[i] = null;
1256        }
1257      }
1258    }
1259
1260    // Now add to backingMap if successfully added to bucket cache. Remove from ramCache if
1261    // success or error.
1262    for (int i = 0; i < size; ++i) {
1263      BlockCacheKey key = entries.get(i).getKey();
1264      // Only add if non-null entry.
1265      if (bucketEntries[i] != null) {
1266        putIntoBackingMap(key, bucketEntries[i]);
1267        if (ioEngine.isPersistent()) {
1268          setCacheInconsistent(true);
1269        }
1270      }
1271      // Always remove from ramCache even if we failed adding it to the block cache above.
1272      boolean existed = ramCache.remove(key, re -> {
1273        if (re != null) {
1274          heapSize.add(-1 * re.getData().heapSize());
1275        }
1276      });
1277      if (!existed && bucketEntries[i] != null) {
1278        // Block should have already been evicted. Remove it and free space.
1279        final BucketEntry bucketEntry = bucketEntries[i];
1280        bucketEntry.withWriteLock(offsetLock, () -> {
1281          if (backingMap.remove(key, bucketEntry)) {
1282            blockEvicted(key, bucketEntry, false, false);
1283          }
1284          return null;
1285        });
1286      }
1287    }
1288
1289    long used = bucketAllocator.getUsedSize();
1290    if (used > acceptableSize()) {
1291      freeSpace("Used=" + used + " > acceptable=" + acceptableSize());
1292    }
1293    return;
1294  }

1298   * returning.
1299   * @param receptacle Where to stash the elements taken from queue. We clear before we use it just
1300   *                   in case.
1301   * @param q          The queue to take from.

1372          + "number. Requested=" + pblen + ", Received= " + read + ", File=" + persistencePath);

1375        // In 3.0 we have enough flexibility to dump the old cache data.
1376        // TODO: In 2.x line, this might need to be filled in to support reading the old format
1377        throw new IOException(
1378          "Persistence file does not start with protobuf magic number. " + persistencePath);
1379      }
1380      parsePB(BucketCacheProtos.BucketCacheEntry.parseDelimitedFrom(in));
1381      bucketAllocator = new BucketAllocator(cacheCapacity, bucketSizes, backingMap, realCacheSize);

1470  private void parsePB(BucketCacheProtos.BucketCacheEntry proto) throws IOException {
1471    Pair<ConcurrentHashMap<BlockCacheKey, BucketEntry>, NavigableSet<BlockCacheKey>> pair =
1472      BucketProtoUtils.fromPB(proto.getDeserializersMap(), proto.getBackingMap(),
1473        this::createRecycler);
1474    backingMap = pair.getFirst();
1475    blocksByHFile = pair.getSecond();
1476    fullyCachedFiles.clear();
1477    fullyCachedFiles.putAll(BucketProtoUtils.fromPB(proto.getCachedFilesMap()));
1478    if (proto.hasChecksum()) {
1479      try {
1480        ((PersistentIOEngine) ioEngine).verifyFileIntegrity(proto.getChecksum().toByteArray(),
1481          algorithm);
1482        backingMapValidated.set(true);
1483      } catch (IOException e) {
1484        LOG.warn("Checksum for cache file failed. "
1485          + "We need to validate each cache key in the backing map. "
1486          + "This may take some time, so we'll do it in a background thread,");
1487        Runnable cacheValidator = () -> {
1488          while (bucketAllocator == null) {
1489            try {
1490              Thread.sleep(50);
1491            } catch (InterruptedException ex) {
1492              throw new RuntimeException(ex);
1493            }
1494          }
1495          long startTime = EnvironmentEdgeManager.currentTime();
1496          int totalKeysOriginally = backingMap.size();
1497          for (Map.Entry<BlockCacheKey, BucketEntry> keyEntry : backingMap.entrySet()) {
1498            try {
1499              ((FileIOEngine) ioEngine).checkCacheTime(keyEntry.getValue());
1500            } catch (IOException e1) {
1501              LOG.debug("Check for key {} failed. Evicting.", keyEntry.getKey());
1502              evictBlock(keyEntry.getKey());
1503              fullyCachedFiles.remove(keyEntry.getKey().getHfileName());
1504            }
1505          }
1506          backingMapValidated.set(true);
1507          LOG.info("Finished validating {} keys in the backing map. Recovered: {}. This took {}ms.",
1508            totalKeysOriginally, backingMap.size(),
1509            (EnvironmentEdgeManager.currentTime() - startTime));
1510        };
1511        Thread t = new Thread(cacheValidator);
1512        t.setDaemon(true);
1513        t.start();
1514      }
1515    } else {
1516      // if has not checksum, it means the persistence file is old format
1517      LOG.info("Persistent file is old format, it does not support verifying file integrity!");
1518      backingMapValidated.set(true);
1519    }
1520    updateRegionSizeMapWhileRetrievingFromFile();
1521    verifyCapacityAndClasses(proto.getCacheCapacity(), proto.getIoClass(), proto.getMapClass());
1522  }

1727    private final BlockCacheKey key;
1728    private final Cacheable data;
1729    private long accessCounter;
1730    private boolean inMemory;
1731    private boolean isCachePersistent;
1732
1733    RAMQueueEntry(BlockCacheKey bck, Cacheable data, long accessCounter, boolean inMemory,
1734      boolean isCachePersistent) {
1735      this.key = bck;
1736      this.data = data;
1737      this.accessCounter = accessCounter;
1738      this.inMemory = inMemory;
1739      this.isCachePersistent = isCachePersistent;
1740    }
1741
1742    public Cacheable getData() {
1743      return data;
1744    }

1746    public BlockCacheKey getKey() {
1747      return key;
1748    }

1750    public void access(long accessCounter) {

1753
1754    private ByteBuffAllocator getByteBuffAllocator() {
1755      if (data instanceof HFileBlock) {
1756        return ((HFileBlock) data).getByteBuffAllocator();
1757      }
1758      return ByteBuffAllocator.HEAP;
1759    }
1760
1761    public BucketEntry writeToCache(final IOEngine ioEngine, final BucketAllocator alloc,
1762      final LongAdder realCacheSize, Function<BucketEntry, Recycler> createRecycler,
1763      ByteBuffer metaBuff) throws IOException {
1764      int len = data.getSerializedLength();
1765      // This cacheable thing can't be serialized
1766      if (len == 0) {
1767        return null;
1768      }
1769      if (isCachePersistent && data instanceof HFileBlock) {
1770        len += Long.BYTES; // we need to record the cache time for consistency check in case of
1771                           // recovery
1772      }
1773      long offset = alloc.allocateBlock(len);
1774      boolean succ = false;
1775      BucketEntry bucketEntry = null;
1776      try {
1777        int diskSizeWithHeader = (data instanceof HFileBlock)
1778          ? ((HFileBlock) data).getOnDiskSizeWithHeader()
1779          : data.getSerializedLength();
1780        bucketEntry = new BucketEntry(offset, len, diskSizeWithHeader, accessCounter, inMemory,
1781          createRecycler, getByteBuffAllocator());
1782        bucketEntry.setDeserializerReference(data.getDeserializer());
1783        if (data instanceof HFileBlock) {
1784          // If an instance of HFileBlock, save on some allocations.
1785          HFileBlock block = (HFileBlock) data;
1786          ByteBuff sliceBuf = block.getBufferReadOnly();
1787          block.getMetaData(metaBuff);
1788          // adds the cache time prior to the block and metadata part
1789          if (isCachePersistent) {
1790            ByteBuffer buffer = ByteBuffer.allocate(Long.BYTES);
1791            buffer.putLong(bucketEntry.getCachedTime());
1792            buffer.rewind();
1793            ioEngine.write(buffer, offset);
1794            ioEngine.write(sliceBuf, (offset + Long.BYTES));
1795          } else {
1796            ioEngine.write(sliceBuf, offset);
1797          }
1798          ioEngine.write(metaBuff, offset + len - metaBuff.limit());
1799        } else {
1800          // Only used for testing.
1801          ByteBuffer bb = ByteBuffer.allocate(len);
1802          data.serialize(bb, true);
1803          ioEngine.write(bb, offset);
1804        }
1805        succ = true;
1806      } finally {
1807        if (!succ) {
1808          alloc.freeBlock(offset, len);
1809        }
1810      }
1811      realCacheSize.add(len);
1812      return bucketEntry;
1813    }
1814  }

1958  /**
1959   * Wrapped the delegate ConcurrentMap with maintaining its block's reference count.
1960   */
1961  static class RAMCache {
1962    /**
1963     * Defined the map as {@link ConcurrentHashMap} explicitly here, because in
1964     * {@link RAMCache#get(BlockCacheKey)} and
1965     * {@link RAMCache#putIfAbsent(BlockCacheKey, BucketCache.RAMQueueEntry)} , we need to guarantee
1966     * the atomicity of map#computeIfPresent(key, func) and map#putIfAbsent(key, func). Besides, the
1967     * func method can execute exactly once only when the key is present(or absent) and under the
1968     * lock context. Otherwise, the reference count of block will be messed up. Notice that the
1969     * {@link java.util.concurrent.ConcurrentSkipListMap} can not guarantee that.
1970     */
1971    final ConcurrentHashMap<BlockCacheKey, RAMQueueEntry> delegate = new ConcurrentHashMap<>();
1972
1973    public boolean containsKey(BlockCacheKey key) {
1974      return delegate.containsKey(key);
1975    }
1976
1977    public RAMQueueEntry get(BlockCacheKey key) {
1978      return delegate.computeIfPresent(key, (k, re) -> {
1979        // It'll be referenced by RPC, so retain atomically here. if the get and retain is not
1980        // atomic, another thread may remove and release the block, when retaining in this thread we
1981        // may retain a block with refCnt=0 which is disallowed. (see HBASE-22422)
1982        re.getData().retain();
1983        return re;
1984      });
1985    }
1986
1987    /**
1988     * Return the previous associated value, or null if absent. It has the same meaning as
1989     * {@link ConcurrentMap#putIfAbsent(Object, Object)}
1990     */
1991    public RAMQueueEntry putIfAbsent(BlockCacheKey key, RAMQueueEntry entry) {
1992      AtomicBoolean absent = new AtomicBoolean(false);
1993      RAMQueueEntry re = delegate.computeIfAbsent(key, k -> {
1994        // The RAMCache reference to this entry, so reference count should be increment.
1995        entry.getData().retain();
1996        absent.set(true);
1997        return entry;
1998      });
1999      return absent.get() ? null : re;
2000    }
2001
2002    public boolean remove(BlockCacheKey key) {
2003      return remove(key, re -> {
2004      });
2005    }
2006
2007    /**
2008     * Defined an {@link Consumer} here, because once the removed entry release its reference count,
2009     * then it's ByteBuffers may be recycled and accessing it outside this method will be thrown an
2010     * exception. the consumer will access entry to remove before release its reference count.
2011     * Notice, don't change its reference count in the {@link Consumer}
2012     */
2013    public boolean remove(BlockCacheKey key, Consumer<RAMQueueEntry> action) {
2014      RAMQueueEntry previous = delegate.remove(key);
2015      action.accept(previous);
2016      if (previous != null) {
2017        previous.getData().release();
2018      }
2019      return previous != null;
2020    }
2021
2022    public boolean isEmpty() {
2023      return delegate.isEmpty();
2024    }
2025
2026    public void clear() {
2027      Iterator<Map.Entry<BlockCacheKey, RAMQueueEntry>> it = delegate.entrySet().iterator();
2028      while (it.hasNext()) {
2029        RAMQueueEntry re = it.next().getValue();
2030        it.remove();
2031        re.getData().release();
2032      }
2033    }

2035    public boolean hasBlocksForFile(String fileName) {
2036      return delegate.keySet().stream().filter(key -> key.getHfileName().equals(fileName))
2037        .findFirst().isPresent();
2038    }
2039  }

2069  @Override
2070  public void notifyFileCachingCompleted(Path fileName, int totalBlockCount, int dataBlockCount,
2071    long size) {
2072    // block eviction may be happening in the background as prefetch runs,
2073    // so we need to count all blocks for this file in the backing map under
2074    // a read lock for the block offset
2075    final List<ReentrantReadWriteLock> locks = new ArrayList<>();
2076    LOG.debug("Notifying caching completed for file {}, with total blocks {}", fileName,
2077      dataBlockCount);
2078    try {
2079      final MutableInt count = new MutableInt();
2080      LOG.debug("iterating over {} entries in the backing map", backingMap.size());
2081      backingMap.entrySet().stream().forEach(entry -> {
2082        if (entry.getKey().getHfileName().equals(fileName.getName())) {
2083          LOG.debug("found block for file {} in the backing map. Acquiring read lock for offset {}",
2084            fileName, entry.getKey().getOffset());
2085          ReentrantReadWriteLock lock = offsetLock.getLock(entry.getKey().getOffset());
2086          lock.readLock().lock();
2087          locks.add(lock);
2088          if (backingMap.containsKey(entry.getKey())) {
2089            count.increment();
2090          }
2091        }
2092      });
2093      // We may either place only data blocks on the BucketCache or all type of blocks
2094      if (dataBlockCount == count.getValue() || totalBlockCount == count.getValue()) {
2095        LOG.debug("File {} has now been fully cached.", fileName);
2096        fileCacheCompleted(fileName, size);
2097      } else {
2098        LOG.debug(
2099          "Prefetch executor completed for {}, but only {} blocks were cached. "
2100            + "Total blocks for file: {}. Checking for blocks pending cache in cache writer queue.",
2101          fileName, count.getValue(), dataBlockCount);
2102        if (ramCache.hasBlocksForFile(fileName.getName())) {
2103          LOG.debug("There are still blocks pending caching for file {}. Will sleep 100ms "
2104            + "and try the verification again.", fileName);
2105          Thread.sleep(100);
2106          notifyFileCachingCompleted(fileName, totalBlockCount, dataBlockCount, size);
2107        } else {
2108          LOG.info(
2109            "We found only {} blocks cached from a total of {} for file {}, "
2110              + "but no blocks pending caching. Maybe cache is full?",
2111            count, dataBlockCount, fileName);
2112        }
2113      }
2114    } catch (InterruptedException e) {
2115      throw new RuntimeException(e);
2116    } finally {
2117      for (ReentrantReadWriteLock lock : locks) {
2118        lock.readLock().unlock();
2119      }
2120    }
2121  }
