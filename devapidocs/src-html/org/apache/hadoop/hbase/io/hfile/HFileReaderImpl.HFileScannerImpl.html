428    protected void readKeyValueLen() {
429      // This is a hot method. We go out of our way to make this method short so it can be
430      // inlined and is not too big to compile. We also manage position in ByteBuffer ourselves
431      // because it is faster than going via range-checked ByteBuffer methods or going through a
432      // byte buffer array a byte at a time.
433      // Get a long at a time rather than read two individual ints. In micro-benchmarking, even
434      // with the extra bit-fiddling, this is order-of-magnitude faster than getting two ints.
435      // Trying to imitate what was done - need to profile if this is better or
436      // earlier way is better by doing mark and reset?
437      // But ensure that you read long instead of two ints
438      long ll = blockBuffer.getLongAfterPosition(0);
439      // Read top half as an int of key length and bottom int as value length
440      this.currKeyLen = (int) (ll >> Integer.SIZE);
441      this.currValueLen = (int) (Bytes.MASK_FOR_LOWER_INT_IN_LONG ^ ll);
442      checkKeyValueLen();
443      this.rowLen = blockBuffer.getShortAfterPosition(Bytes.SIZEOF_LONG);
444      // Move position past the key and value lengths and then beyond the key and value
445      int p = (Bytes.SIZEOF_LONG + currKeyLen + currValueLen);
446      if (reader.getFileContext().isIncludesTags()) {
447        // Tags length is a short.
448        this.currTagsLen = blockBuffer.getShortAfterPosition(p);
449        checkTagsLen();
450        p += (Bytes.SIZEOF_SHORT + currTagsLen);
451      }
452      readMvccVersion(p);
453    }
454
455    private final void checkTagsLen() {
456      if (checkLen(this.currTagsLen)) {
457        throw new IllegalStateException(
458          "Invalid currTagsLen " + this.currTagsLen + ". Block offset: " + curBlock.getOffset()
459            + ", block length: " + this.blockBuffer.limit() + ", position: "
460            + this.blockBuffer.position() + " (without header)." + " path=" + reader.getPath());
461      }
462    }

464    /**
465     * Read mvcc. Does checks to see if we even need to read the mvcc at all.
466     */
467    protected void readMvccVersion(final int offsetFromPos) {
468      // See if we even need to decode mvcc.
469      if (!this.reader.getHFileInfo().shouldIncludeMemStoreTS()) {
470        return;
471      }
472      if (!this.reader.getHFileInfo().isDecodeMemstoreTS()) {
473        currMemstoreTS = 0;
474        currMemstoreTSLen = 1;
475        return;
476      }
477      _readMvccVersion(offsetFromPos);
478    }
479
480    /**
481     * Actually do the mvcc read. Does no checks.
482     */
483    private void _readMvccVersion(int offsetFromPos) {
484      // This is Bytes#bytesToVint inlined so can save a few instructions in this hot method; i.e.
485      // previous if one-byte vint, we'd redo the vint call to find int size.
486      // Also the method is kept small so can be inlined.
487      byte firstByte = blockBuffer.getByteAfterPosition(offsetFromPos);
488      int len = WritableUtils.decodeVIntSize(firstByte);
489      if (len == 1) {
490        this.currMemstoreTS = firstByte;
491      } else {
492        int remaining = len - 1;
493        long i = 0;
494        offsetFromPos++;
495        if (remaining >= Bytes.SIZEOF_INT) {
496          // The int read has to be converted to unsigned long so the & op
497          i = (blockBuffer.getIntAfterPosition(offsetFromPos) & 0x00000000ffffffffL);
498          remaining -= Bytes.SIZEOF_INT;
499          offsetFromPos += Bytes.SIZEOF_INT;
500        }
501        if (remaining >= Bytes.SIZEOF_SHORT) {
502          short s = blockBuffer.getShortAfterPosition(offsetFromPos);
503          i = i << 16;
504          i = i | (s & 0xFFFF);
505          remaining -= Bytes.SIZEOF_SHORT;
506          offsetFromPos += Bytes.SIZEOF_SHORT;
507        }
508        for (int idx = 0; idx < remaining; idx++) {
509          byte b = blockBuffer.getByteAfterPosition(offsetFromPos + idx);
510          i = i << 8;
511          i = i | (b & 0xFF);
512        }
513        currMemstoreTS = (WritableUtils.isNegativeVInt(firstByte) ? ~i : i);
514      }
515      this.currMemstoreTSLen = len;
516    }

518    /**
519     * Within a loaded block, seek looking for the last key that is smaller than (or equal to?) the
520     * key we are interested in. A note on the seekBefore: if you have seekBefore = true, AND the
521     * first key in the block = key, then you'll get thrown exceptions. The caller has to check for
522     * that case and load the previous block as appropriate. the key to find find the key before the
523     * given key in case of exact match.
524     * @return 0 in case of an exact key match, 1 in case of an inexact match, -2 in case of an
525     *         inexact match and furthermore, the input key less than the first key of current
526     *         block(e.g. using a faked index key)
527     */
528    protected int blockSeek(Cell key, boolean seekBefore) {
529      int klen, vlen, tlen = 0;
530      int lastKeyValueSize = -1;
531      int offsetFromPos;
532      do {
533        offsetFromPos = 0;
534        // Better to ensure that we use the BB Utils here
535        long ll = blockBuffer.getLongAfterPosition(offsetFromPos);
536        klen = (int) (ll >> Integer.SIZE);
537        vlen = (int) (Bytes.MASK_FOR_LOWER_INT_IN_LONG ^ ll);
538        if (checkKeyLen(klen) || checkLen(vlen)) {
539          throw new IllegalStateException(
540            "Invalid klen " + klen + " or vlen " + vlen + ". Block offset: " + curBlock.getOffset()
541              + ", block length: " + blockBuffer.limit() + ", position: " + blockBuffer.position()
542              + " (without header)." + " path=" + reader.getPath());
543        }
544        offsetFromPos += Bytes.SIZEOF_LONG;
545        this.rowLen = blockBuffer.getShortAfterPosition(offsetFromPos);
546        blockBuffer.asSubByteBuffer(blockBuffer.position() + offsetFromPos, klen, pair);
547        bufBackedKeyOnlyKv.setKey(pair.getFirst(), pair.getSecond(), klen, rowLen);
548        int comp =
549          PrivateCellUtil.compareKeyIgnoresMvcc(reader.getComparator(), key, bufBackedKeyOnlyKv);
550        offsetFromPos += klen + vlen;
551        if (this.reader.getFileContext().isIncludesTags()) {
552          // Read short as unsigned, high byte first
553          tlen = ((blockBuffer.getByteAfterPosition(offsetFromPos) & 0xff) << 8)
554            ^ (blockBuffer.getByteAfterPosition(offsetFromPos + 1) & 0xff);
555          if (checkLen(tlen)) {
556            throw new IllegalStateException("Invalid tlen " + tlen + ". Block offset: "
557              + curBlock.getOffset() + ", block length: " + blockBuffer.limit() + ", position: "
558              + blockBuffer.position() + " (without header)." + " path=" + reader.getPath());
559          }
560          // add the two bytes read for the tags.
561          offsetFromPos += tlen + (Bytes.SIZEOF_SHORT);
562        }
563        if (this.reader.getHFileInfo().shouldIncludeMemStoreTS()) {
564          // Directly read the mvcc based on current position
565          readMvccVersion(offsetFromPos);
566        }
567        if (comp == 0) {
568          if (seekBefore) {
569            if (lastKeyValueSize < 0) {
570              throw new IllegalStateException("blockSeek with seekBefore "
571                + "at the first key of the block: key=" + CellUtil.getCellKeyAsString(key)
572                + ", blockOffset=" + curBlock.getOffset() + ", onDiskSize="
573                + curBlock.getOnDiskSizeWithHeader() + ", path=" + reader.getPath());
574            }
575            blockBuffer.moveBack(lastKeyValueSize);
576            readKeyValueLen();
577            return 1; // non exact match.
578          }
579          currKeyLen = klen;
580          currValueLen = vlen;
581          currTagsLen = tlen;
582          return 0; // indicate exact match
583        } else if (comp < 0) {
584          if (lastKeyValueSize > 0) {
585            blockBuffer.moveBack(lastKeyValueSize);
586          }
587          readKeyValueLen();
588          if (lastKeyValueSize == -1 && blockBuffer.position() == 0) {
589            return HConstants.INDEX_KEY_MAGIC;
590          }
591          return 1;
592        }
593        // The size of this key/value tuple, including key/value length fields.
594        lastKeyValueSize = klen + vlen + currMemstoreTSLen + KEY_VALUE_LEN_SIZE;
595        // include tag length also if tags included with KV
596        if (reader.getFileContext().isIncludesTags()) {
597          lastKeyValueSize += tlen + Bytes.SIZEOF_SHORT;
598        }
599        blockBuffer.skip(lastKeyValueSize);
600      } while (blockBuffer.hasRemaining());
601
602      // Seek to the last key we successfully read. This will happen if this is
603      // the last key/value pair in the file, in which case the following call
604      // to next() has to return false.
605      blockBuffer.moveBack(lastKeyValueSize);
606      readKeyValueLen();
607      return 1; // didn't exactly find it.
608    }

621    public int reseekTo(Cell key) throws IOException {
622      int compared;
623      if (isSeeked()) {
624        compared = compareKey(reader.getComparator(), key);
625        if (compared < 1) {
626          // If the required key is less than or equal to current key, then
627          // don't do anything.
628          return compared;
629        } else {
630          // The comparison with no_next_index_key has to be checked
631          if (
632            this.nextIndexedKey != null && (this.nextIndexedKey
633                == KeyValueScanner.NO_NEXT_INDEXED_KEY
634              || PrivateCellUtil.compareKeyIgnoresMvcc(reader.getComparator(), key, nextIndexedKey)
635                  < 0)
636          ) {
637            // The reader shall continue to scan the current data block instead
638            // of querying the
639            // block index as long as it knows the target key is strictly
640            // smaller than
641            // the next indexed key or the current data block is the last data
642            // block.
643            return loadBlockAndSeekToKey(this.curBlock, nextIndexedKey, false, key, false);
644          }
645        }
646      }
647      // Don't rewind on a reseek operation, because reseek implies that we are
648      // always going forward in the file.
649      return seekTo(key, false);
650    }

652    /**
653     * An internal API function. Seek to the given key, optionally rewinding to the first key of the
654     * block before doing the seek.
655     * @param key    - a cell representing the key that we need to fetch
656     * @param rewind whether to rewind to the first key of the block before doing the seek. If this
657     *               is false, we are assuming we never go back, otherwise the result is undefined.
658     * @return -1 if the key is earlier than the first key of the file, 0 if we are at the given
659     *         key, 1 if we are past the given key -2 if the key is earlier than the first key of
660     *         the file while using a faked index key
661     */
662    public int seekTo(Cell key, boolean rewind) throws IOException {
663      HFileBlockIndex.BlockIndexReader indexReader = reader.getDataBlockIndexReader();
664      BlockWithScanInfo blockWithScanInfo = indexReader.loadDataBlockWithScanInfo(key, curBlock,
665        cacheBlocks, pread, isCompaction, getEffectiveDataBlockEncoding(), reader);
666      if (blockWithScanInfo == null || blockWithScanInfo.getHFileBlock() == null) {
667        // This happens if the key e.g. falls before the beginning of the file.
668        return -1;
669      }
670      return loadBlockAndSeekToKey(blockWithScanInfo.getHFileBlock(),
671        blockWithScanInfo.getNextIndexedKey(), rewind, key, false);
672    }

674    @Override
675    public boolean seekBefore(Cell key) throws IOException {
676      HFileBlock seekToBlock = reader.getDataBlockIndexReader().seekToDataBlock(key, curBlock,
677        cacheBlocks, pread, isCompaction, reader.getEffectiveEncodingInCache(isCompaction), reader);
678      if (seekToBlock == null) {
679        return false;
680      }
681      Cell firstKey = getFirstKeyCellInBlock(seekToBlock);
682      if (PrivateCellUtil.compareKeyIgnoresMvcc(reader.getComparator(), firstKey, key) >= 0) {
683        long previousBlockOffset = seekToBlock.getPrevBlockOffset();
684        // The key we are interested in
685        if (previousBlockOffset == -1) {
686          // we have a 'problem', the key we want is the first of the file.
687          releaseIfNotCurBlock(seekToBlock);
688          return false;
689        }
690
691        // The first key in the current block 'seekToBlock' is greater than the given
692        // seekBefore key. We will go ahead by reading the next block that satisfies the
693        // given key. Return the current block before reading the next one.
694        releaseIfNotCurBlock(seekToBlock);
695        // It is important that we compute and pass onDiskSize to the block
696        // reader so that it does not have to read the header separately to
697        // figure out the size. Currently, we do not have a way to do this
698        // correctly in the general case however.
699        // TODO: See https://issues.apache.org/jira/browse/HBASE-14576
700        int prevBlockSize = -1;
701        seekToBlock = reader.readBlock(previousBlockOffset, prevBlockSize, cacheBlocks, pread,
702          isCompaction, true, BlockType.DATA, getEffectiveDataBlockEncoding());
703        // TODO shortcut: seek forward in this block to the last key of the
704        // block.
705      }
706      loadBlockAndSeekToKey(seekToBlock, firstKey, true, key, true);
707      return true;
708    }

720    /**
721     * Scans blocks in the "scanned" section of the {@link HFile} until the next data block is
722     * found.
723     * @return the next block, or null if there are no more data blocks
724     */
725    @edu.umd.cs.findbugs.annotations.SuppressWarnings(value = "NP_NULL_ON_SOME_PATH",
726        justification = "Yeah, unnecessary null check; could do w/ clean up")
727    protected HFileBlock readNextDataBlock() throws IOException {
728      long lastDataBlockOffset = reader.getTrailer().getLastDataBlockOffset();
729      if (curBlock == null) {
730        return null;
731      }
732      HFileBlock block = this.curBlock;
733      do {
734        if (block.getOffset() >= lastDataBlockOffset) {
735          releaseIfNotCurBlock(block);
736          return null;
737        }
738        if (block.getOffset() < 0) {
739          releaseIfNotCurBlock(block);
740          throw new IOException("Invalid block offset: " + block + ", path=" + reader.getPath());
741        }
742        // We are reading the next block without block type validation, because
743        // it might turn out to be a non-data block.
744        block = reader.readBlock(block.getOffset() + block.getOnDiskSizeWithHeader(),
745          block.getNextBlockOnDiskSize(), cacheBlocks, pread, isCompaction, true, null,
746          getEffectiveDataBlockEncoding());
747        if (block != null && !block.getBlockType().isData()) {
748          // Whatever block we read we will be returning it unless
749          // it is a datablock. Just in case the blocks are non data blocks
750          block.release();
751        }
752      } while (!block.getBlockType().isData());
753      return block;
754    }

761    public Cell getCell() {
762      if (!isSeeked()) {
763        return null;
764      }
765
766      Cell ret;
767      int cellBufSize = getKVBufSize();
768      long seqId = 0L;
769      if (this.reader.getHFileInfo().shouldIncludeMemStoreTS()) {
770        seqId = currMemstoreTS;
771      }
772      if (blockBuffer.hasArray()) {
773        // TODO : reduce the varieties of KV here. Check if based on a boolean
774        // we can handle the 'no tags' case.
775        if (currTagsLen > 0) {
776          ret = new SizeCachedKeyValue(blockBuffer.array(),
777            blockBuffer.arrayOffset() + blockBuffer.position(), cellBufSize, seqId, currKeyLen,
778            rowLen);
779        } else {
780          ret = new SizeCachedNoTagsKeyValue(blockBuffer.array(),
781            blockBuffer.arrayOffset() + blockBuffer.position(), cellBufSize, seqId, currKeyLen,
782            rowLen);
783        }
784      } else {
785        ByteBuffer buf = blockBuffer.asSubByteBuffer(cellBufSize);
786        if (buf.isDirect()) {
787          ret = currTagsLen > 0
788            ? new SizeCachedByteBufferKeyValue(buf, buf.position(), cellBufSize, seqId, currKeyLen,
789              rowLen)
790            : new SizeCachedNoTagsByteBufferKeyValue(buf, buf.position(), cellBufSize, seqId,
791              currKeyLen, rowLen);
792        } else {
793          if (currTagsLen > 0) {
794            ret = new SizeCachedKeyValue(buf.array(), buf.arrayOffset() + buf.position(),
795              cellBufSize, seqId, currKeyLen, rowLen);
796          } else {
797            ret = new SizeCachedNoTagsKeyValue(buf.array(), buf.arrayOffset() + buf.position(),
798              cellBufSize, seqId, currKeyLen, rowLen);
799          }
800        }
801      }
802      return ret;
803    }

805    @Override
806    public Cell getKey() {
807      assertSeeked();
808      // Create a new object so that this getKey is cached as firstKey, lastKey
809      ObjectIntPair<ByteBuffer> keyPair = new ObjectIntPair<>();
810      blockBuffer.asSubByteBuffer(blockBuffer.position() + KEY_VALUE_LEN_SIZE, currKeyLen, keyPair);
811      ByteBuffer keyBuf = keyPair.getFirst();
812      if (keyBuf.hasArray()) {
813        return new KeyValue.KeyOnlyKeyValue(keyBuf.array(),
814          keyBuf.arrayOffset() + keyPair.getSecond(), currKeyLen);
815      } else {
816        // Better to do a copy here instead of holding on to this BB so that
817        // we could release the blocks referring to this key. This key is specifically used
818        // in HalfStoreFileReader to get the firstkey and lastkey by creating a new scanner
819        // every time. So holding onto the BB (incase of DBB) is not advised here.
820        byte[] key = new byte[currKeyLen];
821        ByteBufferUtils.copyFromBufferToArray(key, keyBuf, keyPair.getSecond(), 0, currKeyLen);
822        return new KeyValue.KeyOnlyKeyValue(key, 0, currKeyLen);
823      }
824    }

879      // Methods are small so they get inlined because they are 'hot'.
880      HFileBlock nextBlock = readNextDataBlock();
881      if (nextBlock == null) {

890      // Small method so can be inlined. It is a hot one.
891      if (blockBuffer.remaining() <= 0) {

897      return true;

1085  /**
1086   * Retrieve block from cache. Validates the retrieved block's type vs {@code expectedBlockType}
1087   * and its encoding vs. {@code expectedDataBlockEncoding}. Unpacks the block as necessary.
1088   */
1089  private HFileBlock getCachedBlock(BlockCacheKey cacheKey, boolean cacheBlock, boolean useLock,
1090    boolean updateCacheMetrics, BlockType expectedBlockType,
1091    DataBlockEncoding expectedDataBlockEncoding) throws IOException {
1092    // Check cache for block. If found return.
1093    BlockCache cache = cacheConf.getBlockCache().orElse(null);
1094    if (cache != null) {
1095      HFileBlock cachedBlock = (HFileBlock) cache.getBlock(cacheKey, cacheBlock, useLock,
1096        updateCacheMetrics, expectedBlockType);
1097      if (cachedBlock != null) {
1098        if (cacheConf.shouldCacheCompressed(cachedBlock.getBlockType().getCategory())) {
1099          HFileBlock compressedBlock = cachedBlock;
1100          cachedBlock = compressedBlock.unpack(hfileContext, fsBlockReader);
1101          // In case of compressed block after unpacking we can release the compressed block
1102          if (compressedBlock != cachedBlock) {
1103            compressedBlock.release();
1104          }
1105        }
1106        try {
1107          validateBlockType(cachedBlock, expectedBlockType);
1108        } catch (IOException e) {
1109          returnAndEvictBlock(cache, cacheKey, cachedBlock);
1110          throw e;
1111        }
1112
1113        if (expectedDataBlockEncoding == null) {
1114          return cachedBlock;
1115        }
1116        DataBlockEncoding actualDataBlockEncoding = cachedBlock.getDataBlockEncoding();
1117        // Block types other than data blocks always have
1118        // DataBlockEncoding.NONE. To avoid false negative cache misses, only
1119        // perform this check if cached block is a data block.
1120        if (
1121          cachedBlock.getBlockType().isData()
1122            && !actualDataBlockEncoding.equals(expectedDataBlockEncoding)
1123        ) {
1124          // This mismatch may happen if a Scanner, which is used for say a
1125          // compaction, tries to read an encoded block from the block cache.
1126          // The reverse might happen when an EncodedScanner tries to read
1127          // un-encoded blocks which were cached earlier.
1128          //
1129          // Because returning a data block with an implicit BlockType mismatch
1130          // will cause the requesting scanner to throw a disk read should be
1131          // forced here. This will potentially cause a significant number of
1132          // cache misses, so update so we should keep track of this as it might
1133          // justify the work on a CompoundScanner.
1134          if (
1135            !expectedDataBlockEncoding.equals(DataBlockEncoding.NONE)
1136              && !actualDataBlockEncoding.equals(DataBlockEncoding.NONE)
1137          ) {
1138            // If the block is encoded but the encoding does not match the
1139            // expected encoding it is likely the encoding was changed but the
1140            // block was not yet evicted. Evictions on file close happen async
1141            // so blocks with the old encoding still linger in cache for some
1142            // period of time. This event should be rare as it only happens on
1143            // schema definition change.
1144            LOG.info(
1145              "Evicting cached block with key {} because data block encoding mismatch; "
1146                + "expected {}, actual {}, path={}",
1147              cacheKey, actualDataBlockEncoding, expectedDataBlockEncoding, path);
1148            // This is an error scenario. so here we need to release the block.
1149            returnAndEvictBlock(cache, cacheKey, cachedBlock);
1150          }
1151          return null;
1152        }
1153        return cachedBlock;
1154      }
1155    }
1156    return null;
1157  }

1161    cache.evictBlock(cacheKey);

1164  /**
1165   * @param cacheBlock Add block to cache, if found
1166   * @return block wrapped in a ByteBuffer, with header skipped
1167   */
1168  @Override
1169  public HFileBlock getMetaBlock(String metaBlockName, boolean cacheBlock) throws IOException {
1170    if (trailer.getMetaIndexCount() == 0) {
1171      return null; // there are no meta blocks
1172    }
1173    if (metaBlockIndexReader == null) {
1174      throw new IOException(path + " meta index not loaded");
1175    }
1176
1177    byte[] mbname = Bytes.toBytes(metaBlockName);
1178    int block = metaBlockIndexReader.rootBlockContainingKey(mbname, 0, mbname.length);
1179    if (block == -1) {
1180      return null;
1181    }
1182    long blockSize = metaBlockIndexReader.getRootBlockDataSize(block);
1183
1184    // Per meta key from any given file, synchronize reads for said block. This
1185    // is OK to do for meta blocks because the meta block index is always
1186    // single-level.
1187    synchronized (metaBlockIndexReader.getRootBlockKey(block)) {
1188      // Check cache for block. If found return.
1189      long metaBlockOffset = metaBlockIndexReader.getRootBlockOffset(block);
1190      BlockCacheKey cacheKey =
1191        new BlockCacheKey(name, metaBlockOffset, this.isPrimaryReplicaReader(), BlockType.META);
1192
1193      cacheBlock &= cacheConf.shouldCacheBlockOnRead(BlockType.META.getCategory());
1194      HFileBlock cachedBlock =
1195        getCachedBlock(cacheKey, cacheBlock, false, true, BlockType.META, null);
1196      if (cachedBlock != null) {
1197        assert cachedBlock.isUnpacked() : "Packed block leak.";
1198        // Return a distinct 'shallow copy' of the block,
1199        // so pos does not get messed by the scanner
1200        return cachedBlock;
1201      }
1202      // Cache Miss, please load.
1203
1204      HFileBlock compressedBlock =
1205        fsBlockReader.readBlockData(metaBlockOffset, blockSize, true, false, true);
1206      HFileBlock uncompressedBlock = compressedBlock.unpack(hfileContext, fsBlockReader);
1207      if (compressedBlock != uncompressedBlock) {
1208        compressedBlock.release();
1209      }
1210
1211      // Cache the block
1212      if (cacheBlock) {
1213        cacheConf.getBlockCache().ifPresent(
1214          cache -> cache.cacheBlock(cacheKey, uncompressedBlock, cacheConf.isInMemory()));
1215      }
1216      return uncompressedBlock;
1217    }

1220  /**
1221   * Whether we use heap or not depends on our intent to cache the block. We want to avoid
1222   * allocating to off-heap if we intend to cache into the on-heap L1 cache. Otherwise, it's more
1223   * efficient to allocate to off-heap since we can control GC ourselves for those. So our decision
1224   * here breaks down as follows: <br>
1225   * If block cache is disabled, don't use heap. If we're not using the CombinedBlockCache, use heap
1226   * unless caching is disabled for the request. Otherwise, only use heap if caching is enabled and
1227   * the expected block type is not DATA (which goes to off-heap L2 in combined cache).
1228   * @see org.apache.hadoop.hbase.io.hfile.HFileBlock.FSReader#readBlockData(long, long, boolean,
1229   *      boolean, boolean)
1230   */
1231  private boolean shouldUseHeap(BlockType expectedBlockType, boolean cacheBlock) {
1232    if (!cacheConf.getBlockCache().isPresent()) {
1233      return false;
1234    }
1235
1236    // we only cache a block if cacheBlock is true and caching-on-read is enabled in CacheConfig
1237    // we can really only check for that if have an expectedBlockType
1238    if (expectedBlockType != null) {
1239      cacheBlock &= cacheConf.shouldCacheBlockOnRead(expectedBlockType.getCategory());
1240    }
1241
1242    if (!cacheConf.isCombinedBlockCache()) {
1243      // Block to cache in LruBlockCache must be an heap one, if caching enabled. So just allocate
1244      // block memory from heap for saving an extra off-heap to heap copying in that case.
1245      return cacheBlock;
1246    }
1247
1248    return cacheBlock && expectedBlockType != null && !expectedBlockType.isData();
1249  }

1259  @Override
1260  public HFileBlock readBlock(long dataBlockOffset, long onDiskBlockSize, final boolean cacheBlock,
1261    boolean pread, final boolean isCompaction, boolean updateCacheMetrics,
1262    BlockType expectedBlockType, DataBlockEncoding expectedDataBlockEncoding, boolean cacheOnly)
1263    throws IOException {
1264    if (dataBlockIndexReader == null) {
1265      throw new IOException(path + " block index not loaded");
1266    }
1267    long trailerOffset = trailer.getLoadOnOpenDataOffset();
1268    if (dataBlockOffset < 0 || dataBlockOffset >= trailerOffset) {
1269      throw new IOException("Requested block is out of range: " + dataBlockOffset
1270        + ", lastDataBlockOffset: " + trailer.getLastDataBlockOffset()
1271        + ", trailer.getLoadOnOpenDataOffset: " + trailerOffset + ", path=" + path);
1272    }
1273    // For any given block from any given file, synchronize reads for said
1274    // block.
1275    // Without a cache, this synchronizing is needless overhead, but really
1276    // the other choice is to duplicate work (which the cache would prevent you
1277    // from doing).
1278
1279    BlockCacheKey cacheKey =
1280      new BlockCacheKey(name, dataBlockOffset, this.isPrimaryReplicaReader(), expectedBlockType);
1281    Attributes attributes = Attributes.of(BLOCK_CACHE_KEY_KEY, cacheKey.toString());
1282
1283    boolean cacheable = cacheBlock && cacheIfCompactionsOff();
1284
1285    boolean useLock = false;
1286    IdLock.Entry lockEntry = null;
1287    final Span span = Span.current();
1288    try {
1289      while (true) {
1290        // Check cache for block. If found return.
1291        if (cacheConf.shouldReadBlockFromCache(expectedBlockType) && !cacheOnly) {
1292          if (useLock) {
1293            lockEntry = offsetLock.getLockEntry(dataBlockOffset);
1294          }
1295          // Try and get the block from the block cache. If the useLock variable is true then this
1296          // is the second time through the loop and it should not be counted as a block cache miss.
1297          HFileBlock cachedBlock = getCachedBlock(cacheKey, cacheBlock, useLock, updateCacheMetrics,
1298            expectedBlockType, expectedDataBlockEncoding);
1299          if (cachedBlock != null) {
1300            if (LOG.isTraceEnabled()) {
1301              LOG.trace("Block for file {} is coming from Cache {}",
1302                Bytes.toString(cachedBlock.getHFileContext().getTableName()), cachedBlock);
1303            }
1304            span.addEvent("block cache hit", attributes);
1305            assert cachedBlock.isUnpacked() : "Packed block leak.";
1306            if (cachedBlock.getBlockType().isData()) {
1307              if (updateCacheMetrics) {
1308                HFile.DATABLOCK_READ_COUNT.increment();
1309              }
1310              // Validate encoding type for data blocks. We include encoding
1311              // type in the cache key, and we expect it to match on a cache hit.
1312              if (cachedBlock.getDataBlockEncoding() != dataBlockEncoder.getDataBlockEncoding()) {
1313                // Remember to release the block when in exceptional path.
1314                cacheConf.getBlockCache().ifPresent(cache -> {
1315                  returnAndEvictBlock(cache, cacheKey, cachedBlock);
1316                });
1317                throw new IOException("Cached block under key " + cacheKey + " "
1318                  + "has wrong encoding: " + cachedBlock.getDataBlockEncoding() + " (expected: "
1319                  + dataBlockEncoder.getDataBlockEncoding() + "), path=" + path);
1320              }
1321            }
1322            // Cache-hit. Return!
1323            return cachedBlock;
1324          }
1325
1326          if (!useLock && cacheable && cacheConf.shouldLockOnCacheMiss(expectedBlockType)) {
1327            // check cache again with lock
1328            useLock = true;
1329            continue;
1330          }
1331          // Carry on, please load.
1332        }
1333
1334        span.addEvent("block cache miss", attributes);
1335        // Load block from filesystem.
1336        HFileBlock hfileBlock = fsBlockReader.readBlockData(dataBlockOffset, onDiskBlockSize, pread,
1337          !isCompaction, shouldUseHeap(expectedBlockType, cacheable));
1338        try {
1339          validateBlockType(hfileBlock, expectedBlockType);
1340        } catch (IOException e) {
1341          hfileBlock.release();
1342          throw e;
1343        }
1344        BlockType.BlockCategory category = hfileBlock.getBlockType().getCategory();
1345        final boolean cacheCompressed = cacheConf.shouldCacheCompressed(category);
1346        final boolean cacheOnRead = cacheConf.shouldCacheBlockOnRead(category);
1347
1348        // Don't need the unpacked block back and we're storing the block in the cache compressed
1349        if (cacheOnly && cacheCompressed && cacheOnRead) {
1350          LOG.debug("Skipping decompression of block {} in prefetch", cacheKey);
1351          // Cache the block if necessary
1352          cacheConf.getBlockCache().ifPresent(cache -> {
1353            if (cacheable && cacheConf.shouldCacheBlockOnRead(category)) {
1354              cache.cacheBlock(cacheKey, hfileBlock, cacheConf.isInMemory(), cacheOnly);
1355            }
1356          });
1357
1358          if (updateCacheMetrics && hfileBlock.getBlockType().isData()) {
1359            HFile.DATABLOCK_READ_COUNT.increment();
1360          }
1361          return hfileBlock;
1362        }
1363        HFileBlock unpacked = hfileBlock.unpack(hfileContext, fsBlockReader);
1364        // Cache the block if necessary
1365        cacheConf.getBlockCache().ifPresent(cache -> {
1366          if (cacheable && cacheConf.shouldCacheBlockOnRead(category)) {
1367            // Using the wait on cache during compaction and prefetching.
1368            cache.cacheBlock(cacheKey, cacheCompressed ? hfileBlock : unpacked,
1369              cacheConf.isInMemory(), cacheOnly);
1370          }
1371        });
1372        if (unpacked != hfileBlock) {
1373          // End of life here if hfileBlock is an independent block.
1374          hfileBlock.release();
1375        }
1376        if (updateCacheMetrics && hfileBlock.getBlockType().isData()) {
1377          HFile.DATABLOCK_READ_COUNT.increment();
1378        }
1379
1380        return unpacked;
1381      }
1382    } finally {
1383      if (lockEntry != null) {
1384        offsetLock.releaseLockEntry(lockEntry);
1385      }
1386    }
1387  }

1394  /**
1395   * Compares the actual type of a block retrieved from cache or disk with its expected type and
1396   * throws an exception in case of a mismatch. Expected block type of {@link BlockType#DATA} is
1397   * considered to match the actual block type [@link {@link BlockType#ENCODED_DATA} as well.
1398   * @param block             a block retrieved from cache or disk
1399   * @param expectedBlockType the expected block type, or null to skip the check
1400   */
1401  private void validateBlockType(HFileBlock block, BlockType expectedBlockType) throws IOException {
1402    if (expectedBlockType == null) {
1403      return;
1404    }
1405    BlockType actualBlockType = block.getBlockType();
1406    if (expectedBlockType.isData() && actualBlockType.isData()) {
1407      // We consider DATA to match ENCODED_DATA for the purpose of this
1408      // verification.
1409      return;
1410    }
1411    if (actualBlockType != expectedBlockType) {
1412      throw new IOException("Expected block type " + expectedBlockType + ", " + "but got "
1413        + actualBlockType + ": " + block + ", path=" + path);
1414    }
1415  }

1453  /**
1454   * Scanner that operates on encoded data blocks.
1455   */
1456  protected static class EncodedScanner extends HFileScannerImpl {
1457    private final HFileBlockDecodingContext decodingCtx;
1458    private final DataBlockEncoder.EncodedSeeker seeker;
1459    private final DataBlockEncoder dataBlockEncoder;
1460
1461    public EncodedScanner(HFile.Reader reader, boolean cacheBlocks, boolean pread,
1462      boolean isCompaction, HFileContext meta, Configuration conf) {
1463      super(reader, cacheBlocks, pread, isCompaction);
1464      DataBlockEncoding encoding = reader.getDataBlockEncoding();
1465      dataBlockEncoder = encoding.getEncoder();
1466      decodingCtx = dataBlockEncoder.newDataBlockDecodingContext(conf, meta);
1467      seeker = dataBlockEncoder.createSeeker(decodingCtx);
1468    }
1469
1470    @Override
1471    public boolean isSeeked() {
1472      return curBlock != null;
1473    }
1474
1475    @Override
1476    public void setNonSeekedState() {
1477      reset();
1478    }
1479
1480    /**
1481     * Updates the current block to be the given {@link HFileBlock}. Seeks to the the first
1482     * key/value pair.
1483     * @param newBlock the block to make current, and read by {@link HFileReaderImpl#readBlock},
1484     *                 it's a totally new block with new allocated {@link ByteBuff}, so if no
1485     *                 further reference to this block, we should release it carefully.
1486     */
1487    @Override
1488    protected void updateCurrentBlock(HFileBlock newBlock) throws CorruptHFileException {
1489      try {
1490        // sanity checks
1491        if (newBlock.getBlockType() != BlockType.ENCODED_DATA) {
1492          throw new IllegalStateException("EncodedScanner works only on encoded data blocks");
1493        }
1494        short dataBlockEncoderId = newBlock.getDataBlockEncodingId();
1495        if (!DataBlockEncoding.isCorrectEncoder(dataBlockEncoder, dataBlockEncoderId)) {
1496          String encoderCls = dataBlockEncoder.getClass().getName();
1497          throw new CorruptHFileException(
1498            "Encoder " + encoderCls + " doesn't support data block encoding "
1499              + DataBlockEncoding.getNameFromId(dataBlockEncoderId) + ",path=" + reader.getPath());
1500        }
1501        updateCurrBlockRef(newBlock);
1502        ByteBuff encodedBuffer = getEncodedBuffer(newBlock);
1503        seeker.setCurrentBuffer(encodedBuffer);
1504      } finally {
1505        releaseIfNotCurBlock(newBlock);
1506      }
1507      // Reset the next indexed key
1508      this.nextIndexedKey = null;
1509    }
1510
1511    private ByteBuff getEncodedBuffer(HFileBlock newBlock) {
1512      ByteBuff origBlock = newBlock.getBufferReadOnly();
1513      int pos = newBlock.headerSize() + DataBlockEncoding.ID_SIZE;
1514      origBlock.position(pos);
1515      origBlock
1516        .limit(pos + newBlock.getUncompressedSizeWithoutHeader() - DataBlockEncoding.ID_SIZE);
1517      return origBlock.slice();
1518    }
1519
1520    @Override
1521    protected boolean processFirstDataBlock() throws IOException {
1522      seeker.rewind();
1523      return true;
1524    }
1525
1526    @Override
1527    public boolean next() throws IOException {
1528      boolean isValid = seeker.next();
1529      if (!isValid) {
1530        HFileBlock newBlock = readNextDataBlock();
1531        isValid = newBlock != null;
1532        if (isValid) {
1533          updateCurrentBlock(newBlock);
1534        } else {
1535          setNonSeekedState();
1536        }
1537      }
1538      return isValid;
1539    }
1540
1541    @Override
1542    public Cell getKey() {
1543      assertValidSeek();
1544      return seeker.getKey();
1545    }
1546
1547    @Override
1548    public ByteBuffer getValue() {
1549      assertValidSeek();
1550      return seeker.getValueShallowCopy();
1551    }
1552
1553    @Override
1554    public Cell getCell() {
1555      if (this.curBlock == null) {
1556        return null;
1557      }
1558      return seeker.getCell();
1559    }
1560
1561    @Override
1562    public String getKeyString() {
1563      return CellUtil.toString(getKey(), false);
1564    }
1565
1566    @Override
1567    public String getValueString() {
1568      ByteBuffer valueBuffer = getValue();
1569      return ByteBufferUtils.toStringBinary(valueBuffer);
1570    }
1571
1572    private void assertValidSeek() {
1573      if (this.curBlock == null) {
1574        throw new NotSeekedException(reader.getPath());
1575      }
1576    }
1577
1578    @Override
1579    protected Cell getFirstKeyCellInBlock(HFileBlock curBlock) {
1580      return dataBlockEncoder.getFirstKeyCellInBlock(getEncodedBuffer(curBlock));
1581    }
1582
1583    @Override
1584    protected int loadBlockAndSeekToKey(HFileBlock seekToBlock, Cell nextIndexedKey, boolean rewind,
1585      Cell key, boolean seekBefore) throws IOException {
1586      if (this.curBlock == null || this.curBlock.getOffset() != seekToBlock.getOffset()) {
1587        updateCurrentBlock(seekToBlock);
1588      } else if (rewind) {
1589        seeker.rewind();
1590      }
1591      this.nextIndexedKey = nextIndexedKey;
1592      return seeker.seekToKeyInBlock(key, seekBefore);
1593    }
1594
1595    @Override
1596    public int compareKey(CellComparator comparator, Cell key) {
1597      return seeker.compareKey(comparator, key);
1598    }
