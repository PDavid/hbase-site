126  // On Java 8, prefer CBC ciphers since AES-NI support is lacking and GCM is slower than CBC.
127  private static final String[] DEFAULT_CIPHERS_JAVA8 =
128    ObjectArrays.concat(getCBCCiphers(), getGCMCiphers(), String.class);
129  // On Java 9 and later, prefer GCM ciphers due to improved AES-NI support.
130  // Note that this performance assumption might not hold true for architectures other than x86_64.
131  private static final String[] DEFAULT_CIPHERS_JAVA9 =
132    ObjectArrays.concat(getGCMCiphers(), getCBCCiphers(), String.class);

134  /**
135   * Enum specifying the client auth requirement of server-side TLS sockets created by this
136   * X509Util.
137   * <ul>
138   * <li>NONE - do not request a client certificate.</li>
139   * <li>WANT - request a client certificate, but allow anonymous clients to connect.</li>
140   * <li>NEED - require a client certificate, disconnect anonymous clients.</li>
141   * </ul>
142   * If the config property is not set, the default value is NEED.
143   */
144  public enum ClientAuth {
145    NONE(org.apache.hbase.thirdparty.io.netty.handler.ssl.ClientAuth.NONE),
146    WANT(org.apache.hbase.thirdparty.io.netty.handler.ssl.ClientAuth.OPTIONAL),
147    NEED(org.apache.hbase.thirdparty.io.netty.handler.ssl.ClientAuth.REQUIRE);
148
149    private final org.apache.hbase.thirdparty.io.netty.handler.ssl.ClientAuth nettyAuth;
150
151    ClientAuth(org.apache.hbase.thirdparty.io.netty.handler.ssl.ClientAuth nettyAuth) {
152      this.nettyAuth = nettyAuth;
153    }
154
155    /**
156     * Converts a property value to a ClientAuth enum. If the input string is empty or null, returns
157     * <code>ClientAuth.NEED</code>.
158     * @param prop the property string.
159     * @return the ClientAuth.
160     * @throws IllegalArgumentException if the property value is not "NONE", "WANT", "NEED", or
161     *                                  empty/null.
162     */
163    public static ClientAuth fromPropertyValue(String prop) {
164      if (prop == null || prop.length() == 0) {
165        return NEED;
166      }
167      return ClientAuth.valueOf(prop.toUpperCase());
168    }
169
170    public org.apache.hbase.thirdparty.io.netty.handler.ssl.ClientAuth toNettyClientAuth() {
171      return nettyAuth;
172    }
173  }

179  static String[] getDefaultCipherSuites() {
180    return getDefaultCipherSuitesForJavaVersion(System.getProperty("java.specification.version"));

183  static String[] getDefaultCipherSuitesForJavaVersion(String javaVersion) {
184    Objects.requireNonNull(javaVersion);
185    if (javaVersion.matches("\\d+")) {
186      // Must be Java 9 or later
187      LOG.debug("Using Java9+ optimized cipher suites for Java version {}", javaVersion);
188      return DEFAULT_CIPHERS_JAVA9;
189    } else if (javaVersion.startsWith("1.")) {
190      // Must be Java 1.8 or earlier
191      LOG.debug("Using Java8 optimized cipher suites for Java version {}", javaVersion);
192      return DEFAULT_CIPHERS_JAVA8;
193    } else {
194      LOG.debug("Could not parse java version {}, using Java8 optimized cipher suites",
195        javaVersion);
196      return DEFAULT_CIPHERS_JAVA8;
197    }
198  }

200  public static SslContext createSslContextForClient(Configuration config)
201    throws X509Exception, IOException {
202
203    SslContextBuilder sslContextBuilder = SslContextBuilder.forClient();
204
205    String keyStoreLocation = config.get(TLS_CONFIG_KEYSTORE_LOCATION, "");
206    char[] keyStorePassword = config.getPassword(TLS_CONFIG_KEYSTORE_PASSWORD);
207    String keyStoreType = config.get(TLS_CONFIG_KEYSTORE_TYPE, "");
208
209    if (keyStoreLocation.isEmpty()) {
210      LOG.warn(TLS_CONFIG_KEYSTORE_LOCATION + " not specified");
211    } else {
212      sslContextBuilder
213        .keyManager(createKeyManager(keyStoreLocation, keyStorePassword, keyStoreType));
214    }
215
216    String trustStoreLocation = config.get(TLS_CONFIG_TRUSTSTORE_LOCATION, "");
217    char[] trustStorePassword = config.getPassword(TLS_CONFIG_TRUSTSTORE_PASSWORD);
218    String trustStoreType = config.get(TLS_CONFIG_TRUSTSTORE_TYPE, "");
219
220    boolean sslCrlEnabled = config.getBoolean(TLS_CONFIG_CLR, false);
221    boolean sslOcspEnabled = config.getBoolean(TLS_CONFIG_OCSP, false);
222
223    boolean verifyServerHostname =
224      config.getBoolean(HBASE_CLIENT_NETTY_TLS_VERIFY_SERVER_HOSTNAME, true);
225    boolean allowReverseDnsLookup = config.getBoolean(TLS_CONFIG_REVERSE_DNS_LOOKUP_ENABLED, true);
226
227    if (trustStoreLocation.isEmpty()) {
228      LOG.warn(TLS_CONFIG_TRUSTSTORE_LOCATION + " not specified");
229    } else {
230      sslContextBuilder
231        .trustManager(createTrustManager(trustStoreLocation, trustStorePassword, trustStoreType,
232          sslCrlEnabled, sslOcspEnabled, verifyServerHostname, allowReverseDnsLookup));
233    }
234
235    sslContextBuilder.enableOcsp(sslOcspEnabled);
236    sslContextBuilder.protocols(getEnabledProtocols(config));
237    sslContextBuilder.ciphers(Arrays.asList(getCipherSuites(config)));
238
239    return sslContextBuilder.build();
240  }

242  public static SslContext createSslContextForServer(Configuration config)
243    throws X509Exception, IOException {
244    String keyStoreLocation = config.get(TLS_CONFIG_KEYSTORE_LOCATION, "");
245    char[] keyStorePassword = config.getPassword(TLS_CONFIG_KEYSTORE_PASSWORD);
246    String keyStoreType = config.get(TLS_CONFIG_KEYSTORE_TYPE, "");
247
248    if (keyStoreLocation.isEmpty()) {
249      throw new SSLContextException(
250        "Keystore is required for SSL server: " + TLS_CONFIG_KEYSTORE_LOCATION);
251    }
252
253    SslContextBuilder sslContextBuilder;
254    sslContextBuilder = SslContextBuilder
255      .forServer(createKeyManager(keyStoreLocation, keyStorePassword, keyStoreType));
256
257    String trustStoreLocation = config.get(TLS_CONFIG_TRUSTSTORE_LOCATION, "");
258    char[] trustStorePassword = config.getPassword(TLS_CONFIG_TRUSTSTORE_PASSWORD);
259    String trustStoreType = config.get(TLS_CONFIG_TRUSTSTORE_TYPE, "");
260
261    boolean sslCrlEnabled = config.getBoolean(TLS_CONFIG_CLR, false);
262    boolean sslOcspEnabled = config.getBoolean(TLS_CONFIG_OCSP, false);
263
264    ClientAuth clientAuth =
265      ClientAuth.fromPropertyValue(config.get(HBASE_SERVER_NETTY_TLS_CLIENT_AUTH_MODE));
266    boolean verifyClientHostname =
267      config.getBoolean(HBASE_SERVER_NETTY_TLS_VERIFY_CLIENT_HOSTNAME, true);
268    boolean allowReverseDnsLookup = config.getBoolean(TLS_CONFIG_REVERSE_DNS_LOOKUP_ENABLED, true);
269
270    if (trustStoreLocation.isEmpty()) {
271      LOG.warn(TLS_CONFIG_TRUSTSTORE_LOCATION + " not specified");
272    } else {
273      sslContextBuilder
274        .trustManager(createTrustManager(trustStoreLocation, trustStorePassword, trustStoreType,
275          sslCrlEnabled, sslOcspEnabled, verifyClientHostname, allowReverseDnsLookup));
276    }
277
278    sslContextBuilder.enableOcsp(sslOcspEnabled);
279    sslContextBuilder.protocols(getEnabledProtocols(config));
280    sslContextBuilder.ciphers(Arrays.asList(getCipherSuites(config)));
281    sslContextBuilder.clientAuth(clientAuth.toNettyClientAuth());
282
283    return sslContextBuilder.build();
284  }

286  /**
287   * Creates a key manager by loading the key store from the given file of the given type,
288   * optionally decrypting it using the given password.
289   * @param keyStoreLocation the location of the key store file.
290   * @param keyStorePassword optional password to decrypt the key store. If empty, assumes the key
291   *                         store is not encrypted.
292   * @param keyStoreType     must be JKS, PEM, PKCS12, BCFKS or null. If null, attempts to
293   *                         autodetect the key store type from the file extension (e.g. .jks /
294   *                         .pem).
295   * @return the key manager.
296   * @throws KeyManagerException if something goes wrong.
297   */
298  static X509KeyManager createKeyManager(String keyStoreLocation, char[] keyStorePassword,
299    String keyStoreType) throws KeyManagerException {
300
301    if (keyStorePassword == null) {
302      keyStorePassword = EMPTY_CHAR_ARRAY;
303    }
304
305    try {
306      KeyStoreFileType storeFileType =
307        KeyStoreFileType.fromPropertyValueOrFileName(keyStoreType, keyStoreLocation);
308      KeyStore ks = FileKeyStoreLoaderBuilderProvider.getBuilderForKeyStoreFileType(storeFileType)
309        .setKeyStorePath(keyStoreLocation).setKeyStorePassword(keyStorePassword).build()
310        .loadKeyStore();
311
312      KeyManagerFactory kmf = KeyManagerFactory.getInstance("PKIX");
313      kmf.init(ks, keyStorePassword);
314
315      for (KeyManager km : kmf.getKeyManagers()) {
316        if (km instanceof X509KeyManager) {
317          return (X509KeyManager) km;
318        }
319      }
320      throw new KeyManagerException("Couldn't find X509KeyManager");
321    } catch (IOException | GeneralSecurityException | IllegalArgumentException e) {
322      throw new KeyManagerException(e);
323    }
324  }

326  /**
327   * Creates a trust manager by loading the trust store from the given file of the given type,
328   * optionally decrypting it using the given password.
329   * @param trustStoreLocation    the location of the trust store file.
330   * @param trustStorePassword    optional password to decrypt the trust store (only applies to JKS
331   *                              trust stores). If empty, assumes the trust store is not encrypted.
332   * @param trustStoreType        must be JKS, PEM, PKCS12, BCFKS or null. If null, attempts to
333   *                              autodetect the trust store type from the file extension (e.g. .jks
334   *                              / .pem).
335   * @param crlEnabled            enable CRL (certificate revocation list) checks.
336   * @param ocspEnabled           enable OCSP (online certificate status protocol) checks.
337   * @param verifyHostName        if true, ssl peer hostname must match name in certificate
338   * @param allowReverseDnsLookup if true, allow falling back to reverse dns lookup in verifying
339   *                              hostname
340   * @return the trust manager.
341   * @throws TrustManagerException if something goes wrong.
342   */
343  static X509TrustManager createTrustManager(String trustStoreLocation, char[] trustStorePassword,
344    String trustStoreType, boolean crlEnabled, boolean ocspEnabled, boolean verifyHostName,
345    boolean allowReverseDnsLookup) throws TrustManagerException {
346
347    if (trustStorePassword == null) {
348      trustStorePassword = EMPTY_CHAR_ARRAY;
349    }
350
351    try {
352      KeyStoreFileType storeFileType =
353        KeyStoreFileType.fromPropertyValueOrFileName(trustStoreType, trustStoreLocation);
354      KeyStore ts = FileKeyStoreLoaderBuilderProvider.getBuilderForKeyStoreFileType(storeFileType)
355        .setTrustStorePath(trustStoreLocation).setTrustStorePassword(trustStorePassword).build()
356        .loadTrustStore();
357
358      PKIXBuilderParameters pbParams = new PKIXBuilderParameters(ts, new X509CertSelector());
359      if (crlEnabled || ocspEnabled) {
360        pbParams.setRevocationEnabled(true);
361        System.setProperty("com.sun.net.ssl.checkRevocation", "true");
362        if (crlEnabled) {
363          System.setProperty("com.sun.security.enableCRLDP", "true");
364        }
365        if (ocspEnabled) {
366          Security.setProperty("ocsp.enable", "true");
367        }
368      } else {
369        pbParams.setRevocationEnabled(false);
370      }
371
372      // Revocation checking is only supported with the PKIX algorithm
373      TrustManagerFactory tmf = TrustManagerFactory.getInstance("PKIX");
374      tmf.init(new CertPathTrustManagerParameters(pbParams));
375
376      for (final TrustManager tm : tmf.getTrustManagers()) {
377        if (tm instanceof X509ExtendedTrustManager) {
378          return new HBaseTrustManager((X509ExtendedTrustManager) tm, verifyHostName,
379            allowReverseDnsLookup);
380        }
381      }
382      throw new TrustManagerException("Couldn't find X509TrustManager");
383    } catch (IOException | GeneralSecurityException | IllegalArgumentException e) {
384      throw new TrustManagerException(e);
385    }
386  }

388  private static String[] getEnabledProtocols(Configuration config) {
389    String enabledProtocolsInput = config.get(TLS_ENABLED_PROTOCOLS);

405  /**
406   * Enable certificate file reloading by creating FileWatchers for keystore and truststore.
407   * AtomicReferences will be set with the new instances. resetContext - if not null - will be
408   * called when the file has been modified.
409   * @param keystoreWatcher   Reference to keystoreFileWatcher.
410   * @param trustStoreWatcher Reference to truststoreFileWatcher.
411   * @param resetContext      Callback for file changes.
412   */
413  public static void enableCertFileReloading(Configuration config,
414    AtomicReference<FileChangeWatcher> keystoreWatcher,
415    AtomicReference<FileChangeWatcher> trustStoreWatcher, Runnable resetContext)
416    throws IOException {
417    String keyStoreLocation = config.get(TLS_CONFIG_KEYSTORE_LOCATION, "");
418    keystoreWatcher.set(newFileChangeWatcher(keyStoreLocation, resetContext));
419    String trustStoreLocation = config.get(TLS_CONFIG_TRUSTSTORE_LOCATION, "");
420    trustStoreWatcher.set(newFileChangeWatcher(trustStoreLocation, resetContext));
421  }

423  private static FileChangeWatcher newFileChangeWatcher(String fileLocation, Runnable resetContext)
424    throws IOException {
425    if (fileLocation == null || fileLocation.isEmpty() || resetContext == null) {
426      return null;
427    }
428    final Path filePath = Paths.get(fileLocation).toAbsolutePath();
429    Path parentPath = filePath.getParent();
430    if (parentPath == null) {
431      throw new IOException("Key/trust store path does not have a parent: " + filePath);
432    }
433    FileChangeWatcher fileChangeWatcher = new FileChangeWatcher(parentPath, watchEvent -> {
434      handleWatchEvent(filePath, watchEvent, resetContext);
435    });
436    fileChangeWatcher.start();
437    return fileChangeWatcher;
438  }

440  /**
441   * Handler for watch events that let us know a file we may care about has changed on disk.
442   * @param filePath the path to the file we are watching for changes.
443   * @param event    the WatchEvent.
444   */
445  private static void handleWatchEvent(Path filePath, WatchEvent<?> event, Runnable resetContext) {
446    boolean shouldResetContext = false;
447    Path dirPath = filePath.getParent();
448    if (event.kind().equals(StandardWatchEventKinds.OVERFLOW)) {
449      // If we get notified about possibly missed events, reload the key store / trust store just to
450      // be sure.
451      shouldResetContext = true;
452    } else if (
453      event.kind().equals(StandardWatchEventKinds.ENTRY_MODIFY)
454        || event.kind().equals(StandardWatchEventKinds.ENTRY_CREATE)
455    ) {
456      Path eventFilePath = dirPath.resolve((Path) event.context());
457      if (filePath.equals(eventFilePath)) {
458        shouldResetContext = true;
459      }
460    }
461    // Note: we don't care about delete events
462    if (shouldResetContext) {
463      LOG.info(
464        "Attempting to reset default SSL context after receiving watch event: {} with context: {}",
465        event.kind(), event.context());
466      resetContext.run();
467    } else {
468      if (LOG.isDebugEnabled()) {
469        LOG.debug(
470          "Ignoring watch event and keeping previous default SSL context. Event kind: {} with context: {}",
471          event.kind(), event.context());
472      }
473    }
474  }
