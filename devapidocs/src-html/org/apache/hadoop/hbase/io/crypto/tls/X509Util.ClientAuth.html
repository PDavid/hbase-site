119  // On Java 8, prefer CBC ciphers since AES-NI support is lacking and GCM is slower than CBC.
120  private static final String[] DEFAULT_CIPHERS_JAVA8 =
121    ObjectArrays.concat(getCBCCiphers(), getGCMCiphers(), String.class);
122  // On Java 9 and later, prefer GCM ciphers due to improved AES-NI support.
123  // Note that this performance assumption might not hold true for architectures other than x86_64.
124  private static final String[] DEFAULT_CIPHERS_JAVA9 =
125    ObjectArrays.concat(getGCMCiphers(), getCBCCiphers(), String.class);

127  /**
128   * Enum specifying the client auth requirement of server-side TLS sockets created by this
129   * X509Util.
130   * <ul>
131   * <li>NONE - do not request a client certificate.</li>
132   * <li>WANT - request a client certificate, but allow anonymous clients to connect.</li>
133   * <li>NEED - require a client certificate, disconnect anonymous clients.</li>
134   * </ul>
135   * If the config property is not set, the default value is NEED.
136   */
137  public enum ClientAuth {
138    NONE(org.apache.hbase.thirdparty.io.netty.handler.ssl.ClientAuth.NONE),
139    WANT(org.apache.hbase.thirdparty.io.netty.handler.ssl.ClientAuth.OPTIONAL),
140    NEED(org.apache.hbase.thirdparty.io.netty.handler.ssl.ClientAuth.REQUIRE);
141
142    private final org.apache.hbase.thirdparty.io.netty.handler.ssl.ClientAuth nettyAuth;
143
144    ClientAuth(org.apache.hbase.thirdparty.io.netty.handler.ssl.ClientAuth nettyAuth) {
145      this.nettyAuth = nettyAuth;
146    }
147
148    /**
149     * Converts a property value to a ClientAuth enum. If the input string is empty or null, returns
150     * <code>ClientAuth.NEED</code>.
151     * @param prop the property string.
152     * @return the ClientAuth.
153     * @throws IllegalArgumentException if the property value is not "NONE", "WANT", "NEED", or
154     *                                  empty/null.
155     */
156    public static ClientAuth fromPropertyValue(String prop) {
157      if (prop == null || prop.length() == 0) {
158        return NEED;
159      }
160      return ClientAuth.valueOf(prop.toUpperCase());
161    }
162
163    public org.apache.hbase.thirdparty.io.netty.handler.ssl.ClientAuth toNettyClientAuth() {
164      return nettyAuth;
165    }
166  }

172  static String[] getDefaultCipherSuites() {
173    return getDefaultCipherSuitesForJavaVersion(System.getProperty("java.specification.version"));
174  }

176  static String[] getDefaultCipherSuitesForJavaVersion(String javaVersion) {
177    Objects.requireNonNull(javaVersion);
178    if (javaVersion.matches("\\d+")) {
179      // Must be Java 9 or later
180      LOG.debug("Using Java9+ optimized cipher suites for Java version {}", javaVersion);
181      return DEFAULT_CIPHERS_JAVA9;
182    } else if (javaVersion.startsWith("1.")) {
183      // Must be Java 1.8 or earlier
184      LOG.debug("Using Java8 optimized cipher suites for Java version {}", javaVersion);
185      return DEFAULT_CIPHERS_JAVA8;
186    } else {
187      LOG.debug("Could not parse java version {}, using Java8 optimized cipher suites",
188        javaVersion);
189      return DEFAULT_CIPHERS_JAVA8;
190    }
191  }

193  public static SslContext createSslContextForClient(Configuration config)
194    throws X509Exception, IOException {
195
196    SslContextBuilder sslContextBuilder = SslContextBuilder.forClient();
197
198    String keyStoreLocation = config.get(TLS_CONFIG_KEYSTORE_LOCATION, "");
199    char[] keyStorePassword = config.getPassword(TLS_CONFIG_KEYSTORE_PASSWORD);
200    String keyStoreType = config.get(TLS_CONFIG_KEYSTORE_TYPE, "");
201
202    if (keyStoreLocation.isEmpty()) {
203      LOG.warn(TLS_CONFIG_KEYSTORE_LOCATION + " not specified");
204    } else {
205      sslContextBuilder
206        .keyManager(createKeyManager(keyStoreLocation, keyStorePassword, keyStoreType));
207    }
208
209    String trustStoreLocation = config.get(TLS_CONFIG_TRUSTSTORE_LOCATION, "");
210    char[] trustStorePassword = config.getPassword(TLS_CONFIG_TRUSTSTORE_PASSWORD);
211    String trustStoreType = config.get(TLS_CONFIG_TRUSTSTORE_TYPE, "");
212
213    boolean sslCrlEnabled = config.getBoolean(TLS_CONFIG_CLR, false);
214    boolean sslOcspEnabled = config.getBoolean(TLS_CONFIG_OCSP, false);
215
216    boolean verifyServerHostname =
217      config.getBoolean(HBASE_CLIENT_NETTY_TLS_VERIFY_SERVER_HOSTNAME, true);
218    boolean allowReverseDnsLookup = config.getBoolean(TLS_CONFIG_REVERSE_DNS_LOOKUP_ENABLED, true);
219
220    if (trustStoreLocation.isEmpty()) {
221      LOG.warn(TLS_CONFIG_TRUSTSTORE_LOCATION + " not specified");
222    } else {
223      sslContextBuilder
224        .trustManager(createTrustManager(trustStoreLocation, trustStorePassword, trustStoreType,
225          sslCrlEnabled, sslOcspEnabled, verifyServerHostname, allowReverseDnsLookup));
226    }
227
228    sslContextBuilder.enableOcsp(sslOcspEnabled);
229    sslContextBuilder.protocols(getEnabledProtocols(config));
230    sslContextBuilder.ciphers(Arrays.asList(getCipherSuites(config)));
231
232    return sslContextBuilder.build();
233  }

235  public static SslContext createSslContextForServer(Configuration config)
236    throws X509Exception, IOException {
237    String keyStoreLocation = config.get(TLS_CONFIG_KEYSTORE_LOCATION, "");
238    char[] keyStorePassword = config.getPassword(TLS_CONFIG_KEYSTORE_PASSWORD);
239    String keyStoreType = config.get(TLS_CONFIG_KEYSTORE_TYPE, "");
240
241    if (keyStoreLocation.isEmpty()) {
242      throw new SSLContextException(
243        "Keystore is required for SSL server: " + TLS_CONFIG_KEYSTORE_LOCATION);
244    }
245
246    SslContextBuilder sslContextBuilder;
247
248    sslContextBuilder = SslContextBuilder
249      .forServer(createKeyManager(keyStoreLocation, keyStorePassword, keyStoreType));
250
251    String trustStoreLocation = config.get(TLS_CONFIG_TRUSTSTORE_LOCATION, "");
252    char[] trustStorePassword = config.getPassword(TLS_CONFIG_TRUSTSTORE_PASSWORD);
253    String trustStoreType = config.get(TLS_CONFIG_TRUSTSTORE_TYPE, "");
254
255    boolean sslCrlEnabled = config.getBoolean(TLS_CONFIG_CLR, false);
256    boolean sslOcspEnabled = config.getBoolean(TLS_CONFIG_OCSP, false);
257
258    ClientAuth clientAuth =
259      ClientAuth.fromPropertyValue(config.get(HBASE_SERVER_NETTY_TLS_CLIENT_AUTH_MODE));
260    boolean verifyClientHostname =
261      config.getBoolean(HBASE_SERVER_NETTY_TLS_VERIFY_CLIENT_HOSTNAME, true);
262    boolean allowReverseDnsLookup = config.getBoolean(TLS_CONFIG_REVERSE_DNS_LOOKUP_ENABLED, true);
263
264    if (trustStoreLocation.isEmpty()) {
265      LOG.warn(TLS_CONFIG_TRUSTSTORE_LOCATION + " not specified");
266    } else {
267      sslContextBuilder
268        .trustManager(createTrustManager(trustStoreLocation, trustStorePassword, trustStoreType,
269          sslCrlEnabled, sslOcspEnabled, verifyClientHostname, allowReverseDnsLookup));
270    }
271
272    sslContextBuilder.enableOcsp(sslOcspEnabled);
273    sslContextBuilder.protocols(getEnabledProtocols(config));
274    sslContextBuilder.ciphers(Arrays.asList(getCipherSuites(config)));
275    sslContextBuilder.clientAuth(clientAuth.toNettyClientAuth());
276
277    return sslContextBuilder.build();
278  }

280  /**
281   * Creates a key manager by loading the key store from the given file of the given type,
282   * optionally decrypting it using the given password.
283   * @param keyStoreLocation the location of the key store file.
284   * @param keyStorePassword optional password to decrypt the key store. If empty, assumes the key
285   *                         store is not encrypted.
286   * @param keyStoreType     must be JKS, PEM, PKCS12, BCFKS or null. If null, attempts to
287   *                         autodetect the key store type from the file extension (e.g. .jks /
288   *                         .pem).
289   * @return the key manager.
290   * @throws KeyManagerException if something goes wrong.
291   */
292  static X509KeyManager createKeyManager(String keyStoreLocation, char[] keyStorePassword,
293    String keyStoreType) throws KeyManagerException {
294
295    if (keyStorePassword == null) {
296      keyStorePassword = EMPTY_CHAR_ARRAY;
297    }
298
299    try {
300      KeyStoreFileType storeFileType =
301        KeyStoreFileType.fromPropertyValueOrFileName(keyStoreType, keyStoreLocation);
302      KeyStore ks = FileKeyStoreLoaderBuilderProvider.getBuilderForKeyStoreFileType(storeFileType)
303        .setKeyStorePath(keyStoreLocation).setKeyStorePassword(keyStorePassword).build()
304        .loadKeyStore();
305
306      KeyManagerFactory kmf = KeyManagerFactory.getInstance("PKIX");
307      kmf.init(ks, keyStorePassword);
308
309      for (KeyManager km : kmf.getKeyManagers()) {
310        if (km instanceof X509KeyManager) {
311          return (X509KeyManager) km;
312        }
313      }
314      throw new KeyManagerException("Couldn't find X509KeyManager");
315    } catch (IOException | GeneralSecurityException | IllegalArgumentException e) {
316      throw new KeyManagerException(e);
317    }
318  }

320  /**
321   * Creates a trust manager by loading the trust store from the given file of the given type,
322   * optionally decrypting it using the given password.
323   * @param trustStoreLocation    the location of the trust store file.
324   * @param trustStorePassword    optional password to decrypt the trust store (only applies to JKS
325   *                              trust stores). If empty, assumes the trust store is not encrypted.
326   * @param trustStoreType        must be JKS, PEM, PKCS12, BCFKS or null. If null, attempts to
327   *                              autodetect the trust store type from the file extension (e.g. .jks
328   *                              / .pem).
329   * @param crlEnabled            enable CRL (certificate revocation list) checks.
330   * @param ocspEnabled           enable OCSP (online certificate status protocol) checks.
331   * @param verifyHostName        if true, ssl peer hostname must match name in certificate
332   * @param allowReverseDnsLookup if true, allow falling back to reverse dns lookup in verifying
333   *                              hostname
334   * @return the trust manager.
335   * @throws TrustManagerException if something goes wrong.
336   */
337  static X509TrustManager createTrustManager(String trustStoreLocation, char[] trustStorePassword,
338    String trustStoreType, boolean crlEnabled, boolean ocspEnabled, boolean verifyHostName,
339    boolean allowReverseDnsLookup) throws TrustManagerException {
340
341    if (trustStorePassword == null) {
342      trustStorePassword = EMPTY_CHAR_ARRAY;
343    }
344
345    try {
346      KeyStoreFileType storeFileType =
347        KeyStoreFileType.fromPropertyValueOrFileName(trustStoreType, trustStoreLocation);
348      KeyStore ts = FileKeyStoreLoaderBuilderProvider.getBuilderForKeyStoreFileType(storeFileType)
349        .setTrustStorePath(trustStoreLocation).setTrustStorePassword(trustStorePassword).build()
350        .loadTrustStore();
351
352      PKIXBuilderParameters pbParams = new PKIXBuilderParameters(ts, new X509CertSelector());
353      if (crlEnabled || ocspEnabled) {
354        pbParams.setRevocationEnabled(true);
355        System.setProperty("com.sun.net.ssl.checkRevocation", "true");
356        if (crlEnabled) {
357          System.setProperty("com.sun.security.enableCRLDP", "true");
358        }
359        if (ocspEnabled) {
360          Security.setProperty("ocsp.enable", "true");
361        }
362      } else {
363        pbParams.setRevocationEnabled(false);
364      }
365
366      // Revocation checking is only supported with the PKIX algorithm
367      TrustManagerFactory tmf = TrustManagerFactory.getInstance("PKIX");
368      tmf.init(new CertPathTrustManagerParameters(pbParams));
369
370      for (final TrustManager tm : tmf.getTrustManagers()) {
371        if (tm instanceof X509ExtendedTrustManager) {
372          return new HBaseTrustManager((X509ExtendedTrustManager) tm, verifyHostName,
373            allowReverseDnsLookup);
374        }
375      }
376      throw new TrustManagerException("Couldn't find X509TrustManager");
377    } catch (IOException | GeneralSecurityException | IllegalArgumentException e) {
378      throw new TrustManagerException(e);
379    }
380  }

382  private static String[] getEnabledProtocols(Configuration config) {
383    String enabledProtocolsInput = config.get(TLS_ENABLED_PROTOCOLS);
384    if (enabledProtocolsInput == null) {
385      return new String[] { config.get(TLS_CONFIG_PROTOCOL, DEFAULT_PROTOCOL) };
386    }
387    return enabledProtocolsInput.split(",");
388  }
389
390  private static String[] getCipherSuites(Configuration config) {
391    String cipherSuitesInput = config.get(TLS_CIPHER_SUITES);
392    if (cipherSuitesInput == null) {
393      return getDefaultCipherSuites();
394    } else {
395      return cipherSuitesInput.split(",");
396    }
