145  // On Java 8, prefer CBC ciphers since AES-NI support is lacking and GCM is slower than CBC.
146  private static final String[] DEFAULT_CIPHERS_JAVA8 =
147    ObjectArrays.concat(getCBCCiphers(), getGCMCiphers(), String.class);
148  // On Java 9 and later, prefer GCM ciphers due to improved AES-NI support.
149  // Note that this performance assumption might not hold true for architectures other than x86_64.
150  private static final String[] DEFAULT_CIPHERS_JAVA9 =
151    ObjectArrays.concat(getGCMCiphers(), getCBCCiphers(), String.class);
152  private static final String[] DEFAULT_CIPHERS_JAVA11 =
153    ObjectArrays.concat(ObjectArrays.concat(getTls13Ciphers(), getGCMCiphers(), String.class),
154      getCBCCiphers(), String.class);

156  private static final String[] DEFAULT_CIPHERS_OPENSSL = getOpenSslFilteredDefaultCiphers();
157
158  /**
159   * Not all of our default ciphers are available in OpenSSL. Takes our default cipher lists and
160   * filters them to only those available in OpenSsl. Prefers TLS 1.3, then GCM, then CBC because
161   * GCM tends to be better and faster, and we don't need to worry about the java8 vs 9 performance
162   * issue if OpenSSL is handling it.
163   */
164  private static String[] getOpenSslFilteredDefaultCiphers() {
165    if (!OpenSsl.isAvailable()) {
166      return new String[0];
167    }
168
169    Set<String> openSslSuites = OpenSsl.availableJavaCipherSuites();
170    List<String> defaultSuites = new ArrayList<>();
171    Arrays.stream(getTls13Ciphers()).filter(openSslSuites::contains).forEach(defaultSuites::add);
172    Arrays.stream(getGCMCiphers()).filter(openSslSuites::contains).forEach(defaultSuites::add);
173    Arrays.stream(getCBCCiphers()).filter(openSslSuites::contains).forEach(defaultSuites::add);
174    return defaultSuites.toArray(new String[0]);
175  }

177  /**
178   * Enum specifying the client auth requirement of server-side TLS sockets created by this
179   * X509Util.
180   * <ul>
181   * <li>NONE - do not request a client certificate.</li>
182   * <li>WANT - request a client certificate, but allow anonymous clients to connect.</li>
183   * <li>NEED - require a client certificate, disconnect anonymous clients.</li>
184   * </ul>
185   * If the config property is not set, the default value is NEED.
186   */
187  public enum ClientAuth {
188    NONE(org.apache.hbase.thirdparty.io.netty.handler.ssl.ClientAuth.NONE),
189    WANT(org.apache.hbase.thirdparty.io.netty.handler.ssl.ClientAuth.OPTIONAL),
190    NEED(org.apache.hbase.thirdparty.io.netty.handler.ssl.ClientAuth.REQUIRE);
191
192    private final org.apache.hbase.thirdparty.io.netty.handler.ssl.ClientAuth nettyAuth;
193
194    ClientAuth(org.apache.hbase.thirdparty.io.netty.handler.ssl.ClientAuth nettyAuth) {
195      this.nettyAuth = nettyAuth;
196    }
197
198    /**
199     * Converts a property value to a ClientAuth enum. If the input string is empty or null, returns
200     * <code>ClientAuth.NEED</code>.
201     * @param prop the property string.
202     * @return the ClientAuth.
203     * @throws IllegalArgumentException if the property value is not "NONE", "WANT", "NEED", or
204     *                                  empty/null.
205     */
206    public static ClientAuth fromPropertyValue(String prop) {
207      if (prop == null || prop.length() == 0) {
208        return NEED;
209      }
210      return ClientAuth.valueOf(prop.toUpperCase());
211    }

222  static String[] getDefaultCipherSuites(boolean useOpenSsl) {
223    if (useOpenSsl) {
224      return DEFAULT_CIPHERS_OPENSSL;
225    }
226    return getDefaultCipherSuitesForJavaVersion(System.getProperty("java.specification.version"));
227  }
228
229  static String[] getDefaultCipherSuitesForJavaVersion(String javaVersion) {
230    Objects.requireNonNull(javaVersion);
231
232    if (javaVersion.matches("\\d+")) {
233      // Must be Java 9 or later
234      int javaVersionInt = Integer.parseInt(javaVersion);
235      if (javaVersionInt >= 11) {
236        LOG.debug(
237          "Using Java11+ optimized cipher suites for Java version {}, including TLSv1.3 support",
238          javaVersion);
239        return DEFAULT_CIPHERS_JAVA11;
240      } else {
241        LOG.debug("Using Java9+ optimized cipher suites for Java version {}", javaVersion);
242        return DEFAULT_CIPHERS_JAVA9;
243      }
244    } else if (javaVersion.startsWith("1.")) {
245      // Must be Java 1.8 or earlier
246      LOG.debug("Using Java8 optimized cipher suites for Java version {}", javaVersion);
247      return DEFAULT_CIPHERS_JAVA8;
248    } else {
249      LOG.debug("Could not parse java version {}, using Java8 optimized cipher suites",
250        javaVersion);
251      return DEFAULT_CIPHERS_JAVA8;
252    }
253  }

255  public static SslContext createSslContextForClient(Configuration config)
256    throws X509Exception, IOException {
257
258    SslContextBuilder sslContextBuilder = SslContextBuilder.forClient();
259
260    boolean useOpenSsl = configureOpenSslIfAvailable(sslContextBuilder, config);
261    String keyStoreLocation = config.get(TLS_CONFIG_KEYSTORE_LOCATION, "");
262    char[] keyStorePassword = config.getPassword(TLS_CONFIG_KEYSTORE_PASSWORD);
263    String keyStoreType = config.get(TLS_CONFIG_KEYSTORE_TYPE, "");
264
265    if (keyStoreLocation.isEmpty()) {
266      LOG.warn(TLS_CONFIG_KEYSTORE_LOCATION + " not specified");
267    } else {
268      sslContextBuilder
269        .keyManager(createKeyManager(keyStoreLocation, keyStorePassword, keyStoreType));
270    }
271
272    String trustStoreLocation = config.get(TLS_CONFIG_TRUSTSTORE_LOCATION, "");
273    char[] trustStorePassword = config.getPassword(TLS_CONFIG_TRUSTSTORE_PASSWORD);
274    String trustStoreType = config.get(TLS_CONFIG_TRUSTSTORE_TYPE, "");
275
276    boolean sslCrlEnabled = config.getBoolean(TLS_CONFIG_CLR, false);
277    boolean sslOcspEnabled = config.getBoolean(TLS_CONFIG_OCSP, false);
278
279    boolean verifyServerHostname =
280      config.getBoolean(HBASE_CLIENT_NETTY_TLS_VERIFY_SERVER_HOSTNAME, true);
281    boolean allowReverseDnsLookup = config.getBoolean(TLS_CONFIG_REVERSE_DNS_LOOKUP_ENABLED, true);
282
283    if (trustStoreLocation.isEmpty()) {
284      LOG.warn(TLS_CONFIG_TRUSTSTORE_LOCATION + " not specified");
285    } else {
286      sslContextBuilder
287        .trustManager(createTrustManager(trustStoreLocation, trustStorePassword, trustStoreType,
288          sslCrlEnabled, sslOcspEnabled, verifyServerHostname, allowReverseDnsLookup));
289    }
290
291    sslContextBuilder.enableOcsp(sslOcspEnabled);
292    sslContextBuilder.protocols(getEnabledProtocols(config));
293    sslContextBuilder.ciphers(Arrays.asList(getCipherSuites(config, useOpenSsl)));
294
295    return sslContextBuilder.build();
296  }

298  /**
299   * Adds SslProvider.OPENSSL if OpenSsl is available and enabled. In order to make it available,
300   * one must ensure that a properly shaded netty-tcnative is on the classpath. Properly shaded
301   * means relocated to be prefixed with "org.apache.hbase.thirdparty" like the rest of the netty
302   * classes.
303   */
304  private static boolean configureOpenSslIfAvailable(SslContextBuilder sslContextBuilder,
305    Configuration conf) {
306    if (OpenSsl.isAvailable() && conf.getBoolean(TLS_USE_OPENSSL, true)) {
307      LOG.debug("Using netty-tcnative to accelerate SSL handling");
308      sslContextBuilder.sslProvider(SslProvider.OPENSSL);
309      return true;
310    } else {
311      if (LOG.isDebugEnabled()) {
312        LOG.debug("Using default JDK SSL provider because netty-tcnative is not {}",
313          OpenSsl.isAvailable() ? "enabled" : "available");
314      }
315      sslContextBuilder.sslProvider(SslProvider.JDK);
316      return false;
317    }
318  }

320  public static SslContext createSslContextForServer(Configuration config)
321    throws X509Exception, IOException {
322    String keyStoreLocation = config.get(TLS_CONFIG_KEYSTORE_LOCATION, "");
323    char[] keyStorePassword = config.getPassword(TLS_CONFIG_KEYSTORE_PASSWORD);
324    String keyStoreType = config.get(TLS_CONFIG_KEYSTORE_TYPE, "");
325
326    if (keyStoreLocation.isEmpty()) {
327      throw new SSLContextException(
328        "Keystore is required for SSL server: " + TLS_CONFIG_KEYSTORE_LOCATION);
329    }
330
331    SslContextBuilder sslContextBuilder;
332    sslContextBuilder = SslContextBuilder
333      .forServer(createKeyManager(keyStoreLocation, keyStorePassword, keyStoreType));
334
335    boolean useOpenSsl = configureOpenSslIfAvailable(sslContextBuilder, config);
336    String trustStoreLocation = config.get(TLS_CONFIG_TRUSTSTORE_LOCATION, "");
337    char[] trustStorePassword = config.getPassword(TLS_CONFIG_TRUSTSTORE_PASSWORD);
338    String trustStoreType = config.get(TLS_CONFIG_TRUSTSTORE_TYPE, "");
339
340    boolean sslCrlEnabled = config.getBoolean(TLS_CONFIG_CLR, false);
341    boolean sslOcspEnabled = config.getBoolean(TLS_CONFIG_OCSP, false);
342
343    ClientAuth clientAuth =
344      ClientAuth.fromPropertyValue(config.get(HBASE_SERVER_NETTY_TLS_CLIENT_AUTH_MODE));
345    boolean verifyClientHostname =
346      config.getBoolean(HBASE_SERVER_NETTY_TLS_VERIFY_CLIENT_HOSTNAME, true);
347    boolean allowReverseDnsLookup = config.getBoolean(TLS_CONFIG_REVERSE_DNS_LOOKUP_ENABLED, true);
348
349    if (trustStoreLocation.isEmpty()) {
350      LOG.warn(TLS_CONFIG_TRUSTSTORE_LOCATION + " not specified");
351    } else {
352      sslContextBuilder
353        .trustManager(createTrustManager(trustStoreLocation, trustStorePassword, trustStoreType,
354          sslCrlEnabled, sslOcspEnabled, verifyClientHostname, allowReverseDnsLookup));
355    }
356
357    sslContextBuilder.enableOcsp(sslOcspEnabled);
358    sslContextBuilder.protocols(getEnabledProtocols(config));
359    sslContextBuilder.ciphers(Arrays.asList(getCipherSuites(config, useOpenSsl)));
360    sslContextBuilder.clientAuth(clientAuth.toNettyClientAuth());
361
362    return sslContextBuilder.build();
363  }

365  /**
366   * Creates a key manager by loading the key store from the given file of the given type,
367   * optionally decrypting it using the given password.
368   * @param keyStoreLocation the location of the key store file.
369   * @param keyStorePassword optional password to decrypt the key store. If empty, assumes the key
370   *                         store is not encrypted.
371   * @param keyStoreType     must be JKS, PEM, PKCS12, BCFKS or null. If null, attempts to
372   *                         autodetect the key store type from the file extension (e.g. .jks /
373   *                         .pem).
374   * @return the key manager.
375   * @throws KeyManagerException if something goes wrong.
376   */
377  static X509KeyManager createKeyManager(String keyStoreLocation, char[] keyStorePassword,
378    String keyStoreType) throws KeyManagerException {
379
380    if (keyStorePassword == null) {
381      keyStorePassword = EMPTY_CHAR_ARRAY;
382    }
383
384    try {
385      KeyStoreFileType storeFileType =
386        KeyStoreFileType.fromPropertyValueOrFileName(keyStoreType, keyStoreLocation);
387      KeyStore ks = FileKeyStoreLoaderBuilderProvider.getBuilderForKeyStoreFileType(storeFileType)
388        .setKeyStorePath(keyStoreLocation).setKeyStorePassword(keyStorePassword).build()
389        .loadKeyStore();
390
391      KeyManagerFactory kmf = KeyManagerFactory.getInstance("PKIX");
392      kmf.init(ks, keyStorePassword);
393
394      for (KeyManager km : kmf.getKeyManagers()) {
395        if (km instanceof X509KeyManager) {
396          return (X509KeyManager) km;
397        }
398      }
399      throw new KeyManagerException("Couldn't find X509KeyManager");
400    } catch (IOException | GeneralSecurityException | IllegalArgumentException e) {
401      throw new KeyManagerException(e);
402    }
403  }

405  /**
406   * Creates a trust manager by loading the trust store from the given file of the given type,
407   * optionally decrypting it using the given password.
408   * @param trustStoreLocation    the location of the trust store file.
409   * @param trustStorePassword    optional password to decrypt the trust store (only applies to JKS
410   *                              trust stores). If empty, assumes the trust store is not encrypted.
411   * @param trustStoreType        must be JKS, PEM, PKCS12, BCFKS or null. If null, attempts to
412   *                              autodetect the trust store type from the file extension (e.g. .jks
413   *                              / .pem).
414   * @param crlEnabled            enable CRL (certificate revocation list) checks.
415   * @param ocspEnabled           enable OCSP (online certificate status protocol) checks.
416   * @param verifyHostName        if true, ssl peer hostname must match name in certificate
417   * @param allowReverseDnsLookup if true, allow falling back to reverse dns lookup in verifying
418   *                              hostname
419   * @return the trust manager.
420   * @throws TrustManagerException if something goes wrong.
421   */
422  static X509TrustManager createTrustManager(String trustStoreLocation, char[] trustStorePassword,
423    String trustStoreType, boolean crlEnabled, boolean ocspEnabled, boolean verifyHostName,
424    boolean allowReverseDnsLookup) throws TrustManagerException {
425
426    if (trustStorePassword == null) {
427      trustStorePassword = EMPTY_CHAR_ARRAY;
428    }
429
430    try {
431      KeyStoreFileType storeFileType =
432        KeyStoreFileType.fromPropertyValueOrFileName(trustStoreType, trustStoreLocation);
433      KeyStore ts = FileKeyStoreLoaderBuilderProvider.getBuilderForKeyStoreFileType(storeFileType)
434        .setTrustStorePath(trustStoreLocation).setTrustStorePassword(trustStorePassword).build()
435        .loadTrustStore();
436
437      PKIXBuilderParameters pbParams = new PKIXBuilderParameters(ts, new X509CertSelector());
438      if (crlEnabled || ocspEnabled) {
439        pbParams.setRevocationEnabled(true);
440        System.setProperty("com.sun.net.ssl.checkRevocation", "true");
441        if (crlEnabled) {
442          System.setProperty("com.sun.security.enableCRLDP", "true");
443        }
444        if (ocspEnabled) {
445          Security.setProperty("ocsp.enable", "true");
446        }
447      } else {
448        pbParams.setRevocationEnabled(false);
449      }
450
451      // Revocation checking is only supported with the PKIX algorithm
452      TrustManagerFactory tmf = TrustManagerFactory.getInstance("PKIX");
453      tmf.init(new CertPathTrustManagerParameters(pbParams));
454
455      for (final TrustManager tm : tmf.getTrustManagers()) {
456        if (tm instanceof X509ExtendedTrustManager) {
457          return new HBaseTrustManager((X509ExtendedTrustManager) tm, verifyHostName,
458            allowReverseDnsLookup);
459        }
460      }
461      throw new TrustManagerException("Couldn't find X509TrustManager");
462    } catch (IOException | GeneralSecurityException | IllegalArgumentException e) {
463      throw new TrustManagerException(e);
464    }
465  }

484  /**
485   * Enable certificate file reloading by creating FileWatchers for keystore and truststore.
486   * AtomicReferences will be set with the new instances. resetContext - if not null - will be
487   * called when the file has been modified.
488   * @param keystoreWatcher   Reference to keystoreFileWatcher.
489   * @param trustStoreWatcher Reference to truststoreFileWatcher.
490   * @param resetContext      Callback for file changes.
491   */
492  public static void enableCertFileReloading(Configuration config,
493    AtomicReference<FileChangeWatcher> keystoreWatcher,
494    AtomicReference<FileChangeWatcher> trustStoreWatcher, Runnable resetContext)
495    throws IOException {
496    String keyStoreLocation = config.get(TLS_CONFIG_KEYSTORE_LOCATION, "");
497    keystoreWatcher.set(newFileChangeWatcher(keyStoreLocation, resetContext));
498    String trustStoreLocation = config.get(TLS_CONFIG_TRUSTSTORE_LOCATION, "");
499    // we are using the same callback for both. there's no reason to kick off two
500    // threads if keystore/truststore are both at the same location
501    if (!keyStoreLocation.equals(trustStoreLocation)) {
502      trustStoreWatcher.set(newFileChangeWatcher(trustStoreLocation, resetContext));
503    }
504  }

506  private static FileChangeWatcher newFileChangeWatcher(String fileLocation, Runnable resetContext)
507    throws IOException {
508    if (fileLocation == null || fileLocation.isEmpty() || resetContext == null) {
509      return null;
510    }
511    final Path filePath = Paths.get(fileLocation).toAbsolutePath();
512    Path parentPath = filePath.getParent();
513    if (parentPath == null) {
514      throw new IOException("Key/trust store path does not have a parent: " + filePath);
515    }
516    FileChangeWatcher fileChangeWatcher =
517      new FileChangeWatcher(parentPath, Objects.toString(filePath.getFileName()), watchEvent -> {
518        handleWatchEvent(filePath, watchEvent, resetContext);
519      });
520    fileChangeWatcher.start();
521    return fileChangeWatcher;
522  }

526   * @param filePath the path to the file we are watching for changes.
527   * @param event    the WatchEvent.
528   */
529  private static void handleWatchEvent(Path filePath, WatchEvent<?> event, Runnable resetContext) {
530    boolean shouldResetContext = false;
531    Path dirPath = filePath.getParent();
532    if (event.kind().equals(StandardWatchEventKinds.OVERFLOW)) {
533      // If we get notified about possibly missed events, reload the key store / trust store just to
534      // be sure.
535      shouldResetContext = true;
536    } else if (
537      event.kind().equals(StandardWatchEventKinds.ENTRY_MODIFY)
538        || event.kind().equals(StandardWatchEventKinds.ENTRY_CREATE)
539    ) {
540      Path eventFilePath = dirPath.resolve((Path) event.context());
541      if (filePath.equals(eventFilePath)) {
542        shouldResetContext = true;
543      }
544    }
545    // Note: we don't care about delete events
546    if (shouldResetContext) {
547      LOG.info(
548        "Attempting to reset default SSL context after receiving watch event: {} with context: {}",
549        event.kind(), event.context());
550      resetContext.run();
551    } else {
552      if (LOG.isDebugEnabled()) {
553        LOG.debug(
554          "Ignoring watch event and keeping previous default SSL context. Event kind: {} with context: {}",
555          event.kind(), event.context());
556      }
557    }
558  }
559}
