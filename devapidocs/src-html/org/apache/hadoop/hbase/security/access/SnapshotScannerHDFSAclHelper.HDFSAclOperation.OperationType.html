098  public SnapshotScannerHDFSAclHelper(Configuration configuration, Connection connection)
099      throws IOException {
100    this.conf = configuration;
101    this.pathHelper = new PathHelper(conf);

103    this.pool = Executors.newFixedThreadPool(conf.getInt(USER_SCAN_SNAPSHOT_THREAD_NUMBER, 10),

116  public void setCommonDirectoryPermission() throws IOException {
117    // Set public directory permission to 751 to make all users have access permission.
118    // And we also need the access permission of the parent of HBase root directory, but
119    // it's not set here, because the owner of HBase root directory may don't own permission
120    // to change it's parent permission to 751.
121    // The {root/.tmp} and {root/.tmp/data} directories are created to make global user HDFS
122    // ACLs can be inherited.
123    List<Path> paths = Lists.newArrayList(pathHelper.getRootDir(), pathHelper.getMobDir(),
124      pathHelper.getTmpDir(), pathHelper.getArchiveDir());
125    paths.addAll(getGlobalRootPaths());
126    for (Path path : paths) {
127      if (!fs.exists(path)) {
128        fs.mkdirs(path);
129      }
130      fs.setPermission(path, new FsPermission(
131          conf.get(COMMON_DIRECTORY_PERMISSION, COMMON_DIRECTORY_PERMISSION_DEFAULT)));
132    }
133    // create snapshot restore directory
134    Path restoreDir =
135        new Path(conf.get(SNAPSHOT_RESTORE_TMP_DIR, SNAPSHOT_RESTORE_TMP_DIR_DEFAULT));
136    if (!fs.exists(restoreDir)) {
137      fs.mkdirs(restoreDir);
138      fs.setPermission(restoreDir, new FsPermission(conf.get(SNAPSHOT_RESTORE_DIRECTORY_PERMISSION,
139        SNAPSHOT_RESTORE_DIRECTORY_PERMISSION_DEFAULT)));
140    }
141  }

144   * Set acl when grant user permission
145   * @param userPermission the user and permission
146   * @param skipNamespaces the namespace set to skip set acl because already set
147   * @param skipTables the table set to skip set acl because already set
148   * @return false if an error occurred, otherwise true
149   */
150  public boolean grantAcl(UserPermission userPermission, Set<String> skipNamespaces,
151      Set<TableName> skipTables) {
152    try {
153      long start = System.currentTimeMillis();
154      handleGrantOrRevokeAcl(userPermission, HDFSAclOperation.OperationType.MODIFY, skipNamespaces,
155        skipTables);
156      LOG.info("Set HDFS acl when grant {}, cost {} ms", userPermission,
157        System.currentTimeMillis() - start);
158      return true;
159    } catch (Exception e) {
160      LOG.error("Set HDFS acl error when grant: {}", userPermission, e);
161      return false;
162    }
163  }

187  /**
188   * Set acl when take a snapshot
189   * @param snapshot the snapshot desc
190   * @return false if an error occurred, otherwise true
191   */
192  public boolean snapshotAcl(SnapshotDescription snapshot) {
193    try {
194      long start = System.currentTimeMillis();
195      TableName tableName = snapshot.getTableName();
196      // global user permission can be inherited from default acl automatically
197      Set<String> userSet = getUsersWithTableReadAction(tableName);
198      userSet.addAll(getUsersWithNamespaceReadAction(tableName.getNamespaceAsString()));
199      Path path = pathHelper.getSnapshotDir(snapshot.getName());
200      handleHDFSAcl(new HDFSAclOperation(fs, path, userSet, HDFSAclOperation.OperationType.MODIFY,
201          true, HDFSAclOperation.AclType.DEFAULT_ADN_ACCESS)).get();
202      LOG.info("Set HDFS acl when snapshot {}, cost {} ms", snapshot.getName(),
203        System.currentTimeMillis() - start);
204      return true;
205    } catch (Exception e) {
206      LOG.error("Set HDFS acl error when snapshot {}", snapshot, e);
207      return false;
208    }
209  }

230  /**
231   * Remove table access acl from namespace dir when delete table
232   * @param tableName the table
233   * @param removeUsers the users whose access acl will be removed
234   * @return false if an error occurred, otherwise true
235   */
236  public boolean removeNamespaceAcl(TableName tableName, Set<String> removeUsers) {
237    try {
238      long start = System.currentTimeMillis();
239      List<AclEntry> aclEntries = removeUsers.stream()
240          .map(removeUser -> aclEntry(ACCESS, removeUser)).collect(Collectors.toList());
241      String namespace = tableName.getNamespaceAsString();
242      List<Path> nsPaths = Lists.newArrayList(pathHelper.getTmpNsDir(namespace),
243        pathHelper.getDataNsDir(namespace), pathHelper.getMobDataNsDir(namespace));
244      // If table has no snapshots, then remove archive ns HDFS acl, otherwise reserve the archive
245      // ns acl to make the snapshots can be scanned, in the second case, need to remove the archive
246      // ns acl when all snapshots of the deleted table are deleted (will do it in later work).
247      if (getTableSnapshotPaths(tableName).isEmpty()) {
248        nsPaths.add(pathHelper.getArchiveNsDir(namespace));
249      }
250      for (Path nsPath : nsPaths) {
251        fs.removeAclEntries(nsPath, aclEntries);
252      }
253      LOG.info("Remove HDFS acl when delete table {}, cost {} ms", tableName,
254        System.currentTimeMillis() - start);
255      return true;
256    } catch (Exception e) {
257      LOG.error("Set HDFS acl error when delete table {}", tableName, e);
258      return false;
259    }
260  }

281  private void handleGrantOrRevokeAcl(UserPermission userPermission,
282      HDFSAclOperation.OperationType operationType, Set<String> skipNamespaces,
283      Set<TableName> skipTables) throws ExecutionException, InterruptedException, IOException {
284    Set<String> users = Sets.newHashSet(userPermission.getUser());
285    switch (userPermission.getAccessScope()) {
286      case GLOBAL:
287        handleGlobalAcl(users, skipNamespaces, skipTables, operationType);
288        break;
289      case NAMESPACE:
290        NamespacePermission namespacePermission =
291            (NamespacePermission) userPermission.getPermission();
292        handleNamespaceAcl(Sets.newHashSet(namespacePermission.getNamespace()), users,
293          skipNamespaces, skipTables, operationType);
294        break;
295      case TABLE:
296        TablePermission tablePermission = (TablePermission) userPermission.getPermission();
297        handleNamespaceAccessAcl(tablePermission.getNamespace(), users, operationType);
298        handleTableAcl(Sets.newHashSet(tablePermission.getTableName()), users, skipNamespaces,
299          skipTables, operationType);
300        break;
301      default:
302        throw new IllegalArgumentException(
303            "Illegal user permission scope " + userPermission.getAccessScope());
304    }
305  }

307  private void handleGlobalAcl(Set<String> users, Set<String> skipNamespaces,
308      Set<TableName> skipTables, HDFSAclOperation.OperationType operationType)
309      throws ExecutionException, InterruptedException, IOException {
310    // handle global root directories HDFS acls
311    List<HDFSAclOperation> hdfsAclOperations = getGlobalRootPaths().stream()
312        .map(path -> new HDFSAclOperation(fs, path, users, operationType, false,
313            HDFSAclOperation.AclType.DEFAULT_ADN_ACCESS))
314        .collect(Collectors.toList());
315    handleHDFSAclParallel(hdfsAclOperations).get();
316    // handle namespace HDFS acls
317    handleNamespaceAcl(Sets.newHashSet(admin.listNamespaces()), users, skipNamespaces, skipTables,
318      operationType);
319  }
320
321  private void handleNamespaceAcl(Set<String> namespaces, Set<String> users,
322      Set<String> skipNamespaces, Set<TableName> skipTables,
323      HDFSAclOperation.OperationType operationType)
324      throws ExecutionException, InterruptedException, IOException {
325    namespaces.removeAll(skipNamespaces);
326    // handle namespace root directories HDFS acls
327    List<HDFSAclOperation> hdfsAclOperations = new ArrayList<>();
328    Set<String> skipTableNamespaces =
329        skipTables.stream().map(TableName::getNamespaceAsString).collect(Collectors.toSet());
330    for (String ns : namespaces) {
331      /**
332       * When op is REMOVE, remove the DEFAULT namespace ACL while keep the ACCESS for skipTables,
333       * otherwise remove both the DEFAULT + ACCESS ACLs. When op is MODIFY, just operate the
334       * DEFAULT + ACCESS ACLs.
335       */
336      HDFSAclOperation.OperationType op = operationType;
337      HDFSAclOperation.AclType aclType = HDFSAclOperation.AclType.DEFAULT_ADN_ACCESS;
338      if (operationType == HDFSAclOperation.OperationType.REMOVE
339          && skipTableNamespaces.contains(ns)) {
340        // remove namespace directories default HDFS acls for skip tables
341        op = HDFSAclOperation.OperationType.REMOVE;
342        aclType = HDFSAclOperation.AclType.DEFAULT;
343      }
344      for (Path path : getNamespaceRootPaths(ns)) {
345        hdfsAclOperations.add(new HDFSAclOperation(fs, path, users, op, false, aclType));
346      }
347    }
348    handleHDFSAclParallel(hdfsAclOperations).get();
349    // handle table directories HDFS acls
350    Set<TableName> tables = new HashSet<>();
351    for (String namespace : namespaces) {
352      tables.addAll(admin.listTableDescriptorsByNamespace(Bytes.toBytes(namespace)).stream()
353          .map(TableDescriptor::getTableName).collect(Collectors.toSet()));
354    }
355    handleTableAcl(tables, users, skipNamespaces, skipTables, operationType);
356  }
357
358  private void handleTableAcl(Set<TableName> tableNames, Set<String> users,
359      Set<String> skipNamespaces, Set<TableName> skipTables,
360      HDFSAclOperation.OperationType operationType)
361      throws ExecutionException, InterruptedException, IOException {
362    Set<TableName> filterTableNames = new HashSet<>();
363    for (TableName tableName : tableNames) {
364      if (!skipTables.contains(tableName)
365          && !skipNamespaces.contains(tableName.getNamespaceAsString())) {
366        filterTableNames.add(tableName);
367      }
368    }
369    List<CompletableFuture<Void>> futures = new ArrayList<>();
370    // handle table HDFS acls
371    for (TableName tableName : filterTableNames) {
372      List<HDFSAclOperation> hdfsAclOperations = getTableRootPaths(tableName, true).stream()
373          .map(path -> new HDFSAclOperation(fs, path, users, operationType, true,
374              HDFSAclOperation.AclType.DEFAULT_ADN_ACCESS))
375          .collect(Collectors.toList());
376      CompletableFuture<Void> future = handleHDFSAclSequential(hdfsAclOperations);
377      futures.add(future);
378    }
379    CompletableFuture<Void> future =
380        CompletableFuture.allOf(futures.toArray(new CompletableFuture[futures.size()]));
381    future.get();
382  }
383
384  private void handleNamespaceAccessAcl(String namespace, Set<String> users,
385      HDFSAclOperation.OperationType operationType)
386      throws ExecutionException, InterruptedException {
387    // handle namespace access HDFS acls
388    List<HDFSAclOperation> hdfsAclOperations =
389        getNamespaceRootPaths(namespace).stream().map(path -> new HDFSAclOperation(fs, path, users,
390            operationType, false, HDFSAclOperation.AclType.ACCESS)).collect(Collectors.toList());
391    CompletableFuture<Void> future = handleHDFSAclParallel(hdfsAclOperations);
392    future.get();
393  }

395  private void setTableAcl(TableName tableName, Set<String> users)
396      throws ExecutionException, InterruptedException, IOException {
397    HDFSAclOperation.OperationType operationType = HDFSAclOperation.OperationType.MODIFY;
398    handleNamespaceAccessAcl(tableName.getNamespaceAsString(), users, operationType);
399    handleTableAcl(Sets.newHashSet(tableName), users, new HashSet<>(0), new HashSet<>(0),
400      operationType);
401  }

478  private CompletableFuture<Void> handleHDFSAcl(HDFSAclOperation acl) {
479    return CompletableFuture.supplyAsync(() -> {
480      List<HDFSAclOperation> childAclOperations = new ArrayList<>();
481      try {
482        acl.handleAcl();
483        childAclOperations = acl.getChildAclOperations();
484      } catch (FileNotFoundException e) {
485        // Skip handle acl if file not found
486      } catch (IOException e) {
487        LOG.error("Set HDFS acl error for path {}", acl.path, e);
488      }
489      return childAclOperations;
490    }, pool).thenComposeAsync(this::handleHDFSAclParallel, pool);
491  }
492
493  private CompletableFuture<Void> handleHDFSAclSequential(List<HDFSAclOperation> operations) {
494    return CompletableFuture.supplyAsync(() -> {
495      try {
496        for (HDFSAclOperation hdfsAclOperation : operations) {
497          handleHDFSAcl(hdfsAclOperation).get();
498        }
499      } catch (InterruptedException | ExecutionException e) {
500        LOG.error("Set HDFS acl error", e);
501      }
502      return null;
503    }, pool);
504  }
505
506  private CompletableFuture<Void> handleHDFSAclParallel(List<HDFSAclOperation> operations) {
507    List<CompletableFuture<Void>> futures =
508        operations.stream().map(this::handleHDFSAcl).collect(Collectors.toList());
509    return CompletableFuture.allOf(futures.toArray(new CompletableFuture[futures.size()]));
510  }

512  private static AclEntry aclEntry(AclEntryScope scope, String name) {
513    return new AclEntry.Builder().setScope(scope)
514        .setType(AuthUtil.isGroupPrincipal(name) ? GROUP : USER).setName(name)
515        .setPermission(READ_EXECUTE).build();
516  }

518  /**
519   * Inner class used to describe modify or remove what type of acl entries(ACCESS, DEFAULT,
520   * ACCESS_AND_DEFAULT) for files or directories(and child files).
521   */
522  private static class HDFSAclOperation {
523    enum OperationType {
524      MODIFY, REMOVE
525    }
526
527    enum AclType {
528      ACCESS, DEFAULT, DEFAULT_ADN_ACCESS
529    }
530
531    private interface Operation {
532      void apply(FileSystem fs, Path path, List<AclEntry> aclList) throws IOException;
533    }
534
535    private FileSystem fs;
536    private Path path;
537    private Operation operation;
538    private boolean recursive;
539    private AclType aclType;
540    private List<AclEntry> defaultAndAccessAclEntries;
541    private List<AclEntry> accessAclEntries;
542    private List<AclEntry> defaultAclEntries;
543
544    HDFSAclOperation(FileSystem fs, Path path, Set<String> users, OperationType operationType,
545        boolean recursive, AclType aclType) {
546      this.fs = fs;
547      this.path = path;
548      this.defaultAndAccessAclEntries = getAclEntries(AclType.DEFAULT_ADN_ACCESS, users);
549      this.accessAclEntries = getAclEntries(AclType.ACCESS, users);
550      this.defaultAclEntries = getAclEntries(AclType.DEFAULT, users);
551      if (operationType == OperationType.MODIFY) {
552        operation = FileSystem::modifyAclEntries;
553      } else if (operationType == OperationType.REMOVE) {
554        operation = FileSystem::removeAclEntries;
555      } else {
556        throw new IllegalArgumentException("Illegal HDFS acl operation type: " + operationType);
557      }
558      this.recursive = recursive;
559      this.aclType = aclType;
560    }
561
562    HDFSAclOperation(Path path, HDFSAclOperation parent) {
563      this.fs = parent.fs;
564      this.path = path;
565      this.defaultAndAccessAclEntries = parent.defaultAndAccessAclEntries;
566      this.accessAclEntries = parent.accessAclEntries;
567      this.defaultAclEntries = parent.defaultAclEntries;
568      this.operation = parent.operation;
569      this.recursive = parent.recursive;
570      this.aclType = parent.aclType;
571    }
572
573    List<HDFSAclOperation> getChildAclOperations() throws IOException {
574      List<HDFSAclOperation> hdfsAclOperations = new ArrayList<>();
575      if (recursive && fs.isDirectory(path)) {
576        FileStatus[] fileStatuses = fs.listStatus(path);
577        for (FileStatus fileStatus : fileStatuses) {
578          hdfsAclOperations.add(new HDFSAclOperation(fileStatus.getPath(), this));
579        }
580      }
581      return hdfsAclOperations;
582    }
583
584    void handleAcl() throws IOException {
585      if (fs.exists(path)) {
586        if (fs.isDirectory(path)) {
587          switch (aclType) {
588            case ACCESS:
589              operation.apply(fs, path, accessAclEntries);
590              break;
591            case DEFAULT:
592              operation.apply(fs, path, defaultAclEntries);
593              break;
594            case DEFAULT_ADN_ACCESS:
595              operation.apply(fs, path, defaultAndAccessAclEntries);
596              break;
597            default:
598              throw new IllegalArgumentException("Illegal HDFS acl type: " + aclType);
599          }
600        } else {
601          operation.apply(fs, path, accessAclEntries);
602        }
603      }
604    }
605
606    private List<AclEntry> getAclEntries(AclType aclType, Set<String> users) {
607      List<AclEntry> aclEntries = new ArrayList<>();
608      switch (aclType) {
609        case ACCESS:
610          for (String user : users) {
611            aclEntries.add(aclEntry(ACCESS, user));
612          }
613          break;
614        case DEFAULT:
615          for (String user : users) {
616            aclEntries.add(aclEntry(DEFAULT, user));
617          }
618          break;
619        case DEFAULT_ADN_ACCESS:
620          for (String user : users) {
621            aclEntries.add(aclEntry(ACCESS, user));
622            aclEntries.add(aclEntry(DEFAULT, user));
623          }
624          break;
625        default:
626          throw new IllegalArgumentException("Illegal HDFS acl type: " + aclType);
627      }
628      return aclEntries;
629    }
