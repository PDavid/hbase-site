069/**
070 * Set HDFS ACLs to hFiles to make HBase granted users have permission to scan snapshot
071 * <p>
072 * To use this feature, please mask sure HDFS config:
073 * <ul>
074 * <li>dfs.permissions.enabled = true</li>
075 * <li>fs.permissions.umask-mode = 027 (or smaller umask than 027)</li>
076 * </ul>
077 * </p>
078 * <p>
079 * The implementation of this feature is as followings:
080 * <ul>
081 * <li>For common directories such as 'data' and 'archive', set other permission to '--x' to make
082 * everyone have the permission to access the directory.</li>
083 * <li>For namespace or table directories such as 'data/ns/table', 'archive/ns/table' and
084 * '.hbase-snapshot/snapshotName', set user 'r-x' access acl and 'r-x' default acl when following
085 * operations happen:
086 * <ul>
087 * <li>grant user with global, namespace or table permission;</li>
088 * <li>revoke user from global, namespace or table;</li>
089 * <li>snapshot table;</li>
090 * <li>truncate table;</li>
091 * </ul>
092 * </li>
093 * <li>Note: Because snapshots are at table level, so this feature just considers users with global,
094 * namespace or table permissions, ignores users with table CF or cell permissions.</li>
095 * </ul>
096 * </p>
097 */
098@CoreCoprocessor
099@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.CONFIG)
100public class SnapshotScannerHDFSAclController implements MasterCoprocessor, MasterObserver {
101  private static final Logger LOG = LoggerFactory.getLogger(SnapshotScannerHDFSAclController.class);

115  @Override
116  public void preMasterInitialization(ObserverContext<MasterCoprocessorEnvironment> c)
117      throws IOException {
118    if (c.getEnvironment().getConfiguration()
119        .getBoolean(SnapshotScannerHDFSAclHelper.USER_SCAN_SNAPSHOT_ENABLE, false)) {
120      MasterCoprocessorEnvironment mEnv = c.getEnvironment();
121      if (!(mEnv instanceof HasMasterServices)) {
122        throw new IOException("Does not implement HMasterServices");
123      }
124      MasterServices masterServices = ((HasMasterServices) mEnv).getMasterServices();
125      hdfsAclHelper = new SnapshotScannerHDFSAclHelper(masterServices.getConfiguration(),
126          masterServices.getConnection());
127      pathHelper = hdfsAclHelper.getPathHelper();
128      fs = pathHelper.getFileSystem();
129      hdfsAclHelper.setCommonDirectoryPermission();
130      initialized = true;
131      userProvider = UserProvider.instantiate(c.getEnvironment().getConfiguration());
132    } else {
133      LOG.warn("Try to initialize the coprocessor SnapshotScannerHDFSAclController but failure "
134          + "because the config " + SnapshotScannerHDFSAclHelper.USER_SCAN_SNAPSHOT_ENABLE
135          + " is false.");
136    }
137  }

139  @Override
140  public void postStartMaster(ObserverContext<MasterCoprocessorEnvironment> c) throws IOException {
141    if (checkInitialized()) {
142      try (Admin admin = c.getEnvironment().getConnection().getAdmin()) {
143        if (admin.tableExists(PermissionStorage.ACL_TABLE_NAME)) {
144          // Check if hbase acl table has 'm' CF, if not, add 'm' CF
145          TableDescriptor tableDescriptor = admin.getDescriptor(PermissionStorage.ACL_TABLE_NAME);
146          boolean containHdfsAclFamily =
147              Arrays.stream(tableDescriptor.getColumnFamilies()).anyMatch(family -> Bytes
148                  .equals(family.getName(), SnapshotScannerHDFSAclStorage.HDFS_ACL_FAMILY));
149          if (!containHdfsAclFamily) {
150            TableDescriptorBuilder builder = TableDescriptorBuilder.newBuilder(tableDescriptor)
151                .setColumnFamily(ColumnFamilyDescriptorBuilder
152                    .newBuilder(SnapshotScannerHDFSAclStorage.HDFS_ACL_FAMILY).build());
153            admin.modifyTable(builder.build());
154          }
155        } else {
156          throw new TableNotFoundException("Table " + PermissionStorage.ACL_TABLE_NAME
157              + " is not created yet. Please check if " + getClass().getName()
158              + " is configured after " + AccessController.class.getName());
159        }
160      }
161    }
162  }

171  @Override
172  public void postCompletedCreateTableAction(ObserverContext<MasterCoprocessorEnvironment> c,
173      TableDescriptor desc, RegionInfo[] regions) throws IOException {
174    if (!desc.getTableName().isSystemTable() && checkInitialized()) {
175      TableName tableName = desc.getTableName();
176      List<Path> paths = hdfsAclHelper.getTableRootPaths(tableName, false);
177      for (Path path : paths) {
178        if (!fs.exists(path)) {
179          fs.mkdirs(path);
180        }
181      }
182      // Add table owner HDFS acls
183      String owner =
184          desc.getOwnerString() == null ? getActiveUser(c).getShortName() : desc.getOwnerString();
185      hdfsAclHelper.addTableAcl(tableName, owner);
186      try (Table aclTable =
187          c.getEnvironment().getConnection().getTable(PermissionStorage.ACL_TABLE_NAME)) {
188        SnapshotScannerHDFSAclStorage.addUserTableHdfsAcl(aclTable, owner, tableName);
189      }
190    }
191  }

222  @Override
223  public void postDeleteTable(ObserverContext<MasterCoprocessorEnvironment> ctx,
224      TableName tableName) throws IOException {
225    if (!tableName.isSystemTable() && checkInitialized()) {
226      /*
227       * remove table user access HDFS acl from namespace directory if the user has no permissions
228       * of global, ns of the table or other tables of the ns, eg: Bob has 'ns1:t1' read permission,
229       * when delete 'ns1:t1', if Bob has global read permission, '@ns1' read permission or
230       * 'ns1:other_tables' read permission, then skip remove Bob access acl in ns1Dirs, otherwise,
231       * remove Bob access acl.
232       */
233      Set<String> removeUsers = new HashSet<>();
234      try (Table aclTable =
235          ctx.getEnvironment().getConnection().getTable(PermissionStorage.ACL_TABLE_NAME)) {
236        List<String> users = SnapshotScannerHDFSAclStorage.getTableUsers(aclTable, tableName);
237        SnapshotScannerHDFSAclStorage.deleteTableHdfsAcl(aclTable, tableName);
238        byte[] namespace = tableName.getNamespace();
239        for (String user : users) {
240          List<byte[]> userEntries = SnapshotScannerHDFSAclStorage.getUserEntries(aclTable, user);
241          boolean remove = true;
242          for (byte[] entry : userEntries) {
243            if (PermissionStorage.isGlobalEntry(entry)) {
244              remove = false;
245              break;
246            } else if (PermissionStorage.isNamespaceEntry(entry)
247                && Bytes.equals(PermissionStorage.fromNamespaceEntry(entry), namespace)) {
248              remove = false;
249              break;
250            } else if (Bytes.equals(TableName.valueOf(entry).getNamespace(), namespace)) {
251              remove = false;
252              break;
253            }
254          }
255          if (remove) {
256            removeUsers.add(user);
257          }
258        }
259      }
260      if (removeUsers.size() > 0) {
261        hdfsAclHelper.removeNamespaceAcl(tableName, removeUsers);
262      }
263    }
264  }

266  @Override
267  public void postDeleteNamespace(ObserverContext<MasterCoprocessorEnvironment> ctx,
268      String namespace) throws IOException {
269    if (checkInitialized()) {
270      try (Table aclTable =
271          ctx.getEnvironment().getConnection().getTable(PermissionStorage.ACL_TABLE_NAME)) {
272        SnapshotScannerHDFSAclStorage.deleteNamespaceHdfsAcl(aclTable, namespace);
273      }
274      /**
275       * Delete namespace tmp directory because it's created by this coprocessor when namespace is
276       * created to make namespace default acl can be inherited by tables. The namespace data
277       * directory is deleted by DeleteNamespaceProcedure, the namespace archive directory is
278       * deleted by HFileCleaner.
279       */
280      Path tmpNsDir = pathHelper.getTmpNsDir(namespace);
281      if (fs.exists(tmpNsDir)) {
282        if (fs.listStatus(tmpNsDir).length == 0) {
283          fs.delete(tmpNsDir, false);
284        } else {
285          LOG.error("The tmp directory {} of namespace {} is not empty after delete namespace",
286            tmpNsDir, namespace);
287        }
288      }
289    }
290  }

292  @Override
293  public void postGrant(ObserverContext<MasterCoprocessorEnvironment> c,
294      UserPermission userPermission, boolean mergeExistingPermissions) throws IOException {
295    if (!checkInitialized()) {
296      return;
297    }
298    try (Table aclTable =
299        c.getEnvironment().getConnection().getTable(PermissionStorage.ACL_TABLE_NAME)) {
300      Configuration conf = c.getEnvironment().getConfiguration();
301      String userName = userPermission.getUser();
302      switch (userPermission.getAccessScope()) {
303        case GLOBAL:
304          UserPermission perm = getUserGlobalPermission(conf, userName);
305          if (perm != null && containReadPermission(perm)) {
306            if (!isHdfsAclSet(aclTable, userName)) {
307              Pair<Set<String>, Set<TableName>> namespaceAndTable =
308                      SnapshotScannerHDFSAclStorage.getUserNamespaceAndTable(aclTable, userName);
309              Set<String> skipNamespaces = namespaceAndTable.getFirst();
310              Set<TableName> skipTables = namespaceAndTable.getSecond().stream()
311                      .filter(t -> !skipNamespaces.contains(t.getNamespaceAsString()))
312                      .collect(Collectors.toSet());
313              hdfsAclHelper.grantAcl(userPermission, skipNamespaces, skipTables);
314              SnapshotScannerHDFSAclStorage.addUserGlobalHdfsAcl(aclTable, userName);
315            }
316          } else {
317            // The merged user permission doesn't contain READ, so remove user global HDFS acls if
318            // it's set
319            removeUserGlobalHdfsAcl(aclTable, userName, userPermission);
320          }
321          break;
322        case NAMESPACE:
323          String namespace = ((NamespacePermission) userPermission.getPermission()).getNamespace();
324          UserPermission nsPerm = getUserNamespacePermission(conf, userName, namespace);
325          if (nsPerm != null && containReadPermission(nsPerm)) {
326            if (!isHdfsAclSet(aclTable, userName, namespace)) {
327              Set<TableName> skipTables = SnapshotScannerHDFSAclStorage
328                      .getUserNamespaceAndTable(aclTable, userName).getSecond();
329              hdfsAclHelper.grantAcl(userPermission, new HashSet<>(0), skipTables);
330            }
331            SnapshotScannerHDFSAclStorage.addUserNamespaceHdfsAcl(aclTable, userName, namespace);
332          } else {
333            // The merged user permission doesn't contain READ, so remove user namespace HDFS acls
334            // if it's set
335            removeUserNamespaceHdfsAcl(aclTable, userName, namespace, userPermission);
336          }
337          break;
338        case TABLE:
339          TableName tableName = ((TablePermission) userPermission.getPermission()).getTableName();
340          UserPermission tPerm = getUserTablePermission(conf, userName, tableName);
341          if (tPerm != null) {
342            TablePermission tablePermission = (TablePermission) tPerm.getPermission();
343            if (tablePermission.hasFamily() || tablePermission.hasQualifier()) {
344              break;
345            }
346          }
347          if (tPerm != null && containReadPermission(tPerm)) {
348            if (!isHdfsAclSet(aclTable, userName, tableName)) {
349              hdfsAclHelper.grantAcl(userPermission, new HashSet<>(0), new HashSet<>(0));
350            }
351            SnapshotScannerHDFSAclStorage.addUserTableHdfsAcl(aclTable, userName, tableName);
352          } else {
353            // The merged user permission doesn't contain READ, so remove user table HDFS acls if
354            // it's set
355            removeUserTableHdfsAcl(aclTable, userName, tableName, userPermission);
356          }
357          break;
358        default:
359          throw new IllegalArgumentException(
360              "Illegal user permission scope " + userPermission.getAccessScope());
361      }
362    }
363  }

365  @Override
366  public void postRevoke(ObserverContext<MasterCoprocessorEnvironment> c,
367      UserPermission userPermission) throws IOException {
368    if (checkInitialized()) {
369      try (Table aclTable =
370          c.getEnvironment().getConnection().getTable(PermissionStorage.ACL_TABLE_NAME)) {
371        String userName = userPermission.getUser();
372        Configuration conf = c.getEnvironment().getConfiguration();
373        switch (userPermission.getAccessScope()) {
374          case GLOBAL:
375            UserPermission userGlobalPerm = getUserGlobalPermission(conf, userName);
376            if (userGlobalPerm == null || !containReadPermission(userGlobalPerm)) {
377              removeUserGlobalHdfsAcl(aclTable, userName, userPermission);
378            }
379            break;
380          case NAMESPACE:
381            NamespacePermission nsPerm = (NamespacePermission) userPermission.getPermission();
382            UserPermission userNsPerm =
383                getUserNamespacePermission(conf, userName, nsPerm.getNamespace());
384            if (userNsPerm == null || !containReadPermission(userNsPerm)) {
385              removeUserNamespaceHdfsAcl(aclTable, userName, nsPerm.getNamespace(), userPermission);
386            }
387            break;
388          case TABLE:
389            TablePermission tPerm = (TablePermission) userPermission.getPermission();
390            UserPermission userTablePerm =
391                getUserTablePermission(conf, userName, tPerm.getTableName());
392            if (userTablePerm == null || !containReadPermission(userTablePerm)) {
393              removeUserTableHdfsAcl(aclTable, userName, tPerm.getTableName(), userPermission);
394            }
395            break;
396          default:
397            throw new IllegalArgumentException(
398                "Illegal user permission scope " + userPermission.getAccessScope());
399        }
400      }
401    }
402  }

404  private void removeUserGlobalHdfsAcl(Table aclTable, String userName,
405      UserPermission userPermission) throws IOException {
406    if (SnapshotScannerHDFSAclStorage.hasUserGlobalHdfsAcl(aclTable, userName)) {
407      // remove user global acls but reserve ns and table acls
408      Pair<Set<String>, Set<TableName>> namespaceAndTable =
409          SnapshotScannerHDFSAclStorage.getUserNamespaceAndTable(aclTable, userName);
410      Set<String> skipNamespaces = namespaceAndTable.getFirst();
411      Set<TableName> skipTables = namespaceAndTable.getSecond().stream()
412          .filter(t -> !skipNamespaces.contains(t.getNamespaceAsString()))
413          .collect(Collectors.toSet());
414      hdfsAclHelper.revokeAcl(userPermission, skipNamespaces, skipTables);
415      SnapshotScannerHDFSAclStorage.deleteUserGlobalHdfsAcl(aclTable, userName);
416    }
417  }
418
419  private void removeUserNamespaceHdfsAcl(Table aclTable, String userName, String namespace,
420      UserPermission userPermission) throws IOException {
421    // remove user ns acls but reserve table acls
422    if (SnapshotScannerHDFSAclStorage.hasUserNamespaceHdfsAcl(aclTable, userName, namespace)) {
423      if (!SnapshotScannerHDFSAclStorage.hasUserGlobalHdfsAcl(aclTable, userName)) {
424        Set<TableName> skipTables =
425            SnapshotScannerHDFSAclStorage.getUserNamespaceAndTable(aclTable, userName).getSecond();
426        hdfsAclHelper.revokeAcl(userPermission, new HashSet<>(), skipTables);
427      }
428      SnapshotScannerHDFSAclStorage.deleteUserNamespaceHdfsAcl(aclTable, userName, namespace);
429    }
430  }
431
432  private void removeUserTableHdfsAcl(Table aclTable, String userName, TableName tableName,
433      UserPermission userPermission) throws IOException {
434    if (SnapshotScannerHDFSAclStorage.hasUserTableHdfsAcl(aclTable, userName, tableName)) {
435      if (!SnapshotScannerHDFSAclStorage.hasUserGlobalHdfsAcl(aclTable, userName)
436          && !SnapshotScannerHDFSAclStorage.hasUserNamespaceHdfsAcl(aclTable, userName,
437            tableName.getNamespaceAsString())) {
438        // remove table acls
439        hdfsAclHelper.revokeAcl(userPermission, new HashSet<>(0), new HashSet<>(0));
440      }
441      SnapshotScannerHDFSAclStorage.deleteUserTableHdfsAcl(aclTable, userName, tableName);
442    }
443  }

445  private boolean containReadPermission(UserPermission userPermission) {
446    if (userPermission != null) {
447      return Arrays.stream(userPermission.getPermission().getActions())
448          .anyMatch(action -> action == Action.READ);
449    }
450    return false;
451  }

497  /**
498   * Check if user global/namespace/table HDFS acls is already set to hfile
499   */
500  private boolean isHdfsAclSet(Table aclTable, String userName, String namespace,
501      TableName tableName) throws IOException {
502    boolean isSet = SnapshotScannerHDFSAclStorage.hasUserGlobalHdfsAcl(aclTable, userName);
503    if (namespace != null) {
504      isSet = isSet
505          || SnapshotScannerHDFSAclStorage.hasUserNamespaceHdfsAcl(aclTable, userName, namespace);
506    }
507    if (tableName != null) {
508      isSet = isSet
509          || SnapshotScannerHDFSAclStorage.hasUserNamespaceHdfsAcl(aclTable, userName,
510            tableName.getNamespaceAsString())
511          || SnapshotScannerHDFSAclStorage.hasUserTableHdfsAcl(aclTable, userName, tableName);
512    }
513    return isSet;
514  }

524  private User getActiveUser(ObserverContext<?> ctx) throws IOException {
525    // for non-rpc handling, fallback to system user
526    Optional<User> optionalUser = ctx.getCaller();
527    if (optionalUser.isPresent()) {
528      return optionalUser.get();
529    }
530    return userProvider.getCurrent();

533  static final class SnapshotScannerHDFSAclStorage {
534    /**
535     * Add a new CF in HBase acl table to record if the HBase read permission is synchronized to
536     * related hfiles. The record has two usages: 1. check if we need to remove HDFS acls for a
537     * grant without READ permission(eg: grant user table read permission and then grant user table
538     * write permission without merging the existing permissions, in this case, need to remove HDFS
539     * acls); 2. skip some HDFS acl sync because it may be already set(eg: grant user table read
540     * permission and then grant user ns read permission; grant user table read permission and then
541     * grant user table write permission with merging the existing permissions).
542     */
543    static final byte[] HDFS_ACL_FAMILY = Bytes.toBytes("m");
544    // The value 'R' has no specific meaning, if cell value is not null, it means that the user HDFS
545    // acls is set to hfiles.
546    private static final byte[] HDFS_ACL_VALUE = Bytes.toBytes("R");
547
548    static void addUserGlobalHdfsAcl(Table aclTable, String user) throws IOException {
549      addUserEntry(aclTable, user, PermissionStorage.ACL_GLOBAL_NAME);
550    }
551
552    static void addUserNamespaceHdfsAcl(Table aclTable, String user, String namespace)
553        throws IOException {
554      addUserEntry(aclTable, user, Bytes.toBytes(PermissionStorage.toNamespaceEntry(namespace)));
555    }
556
557    static void addUserTableHdfsAcl(Table aclTable, String user, TableName tableName)
558        throws IOException {
559      addUserEntry(aclTable, user, tableName.getName());
560    }
561
562    private static void addUserEntry(Table t, String user, byte[] entry) throws IOException {
563      Put p = new Put(entry);
564      p.addColumn(HDFS_ACL_FAMILY, Bytes.toBytes(user), HDFS_ACL_VALUE);
565      t.put(p);
566    }
567
568    static void deleteUserGlobalHdfsAcl(Table aclTable, String user) throws IOException {
569      deleteUserEntry(aclTable, user, PermissionStorage.ACL_GLOBAL_NAME);
570    }
571
572    static void deleteUserNamespaceHdfsAcl(Table aclTable, String user, String namespace)
573        throws IOException {
574      deleteUserEntry(aclTable, user, Bytes.toBytes(PermissionStorage.toNamespaceEntry(namespace)));
575    }
576
577    static void deleteUserTableHdfsAcl(Table aclTable, String user, TableName tableName)
578        throws IOException {
579      deleteUserEntry(aclTable, user, tableName.getName());
580    }

607    private static List<String> getEntryUsers(Table aclTable, byte[] entry) throws IOException {
608      List<String> users = new ArrayList<>();
609      Get get = new Get(entry);
610      get.addFamily(HDFS_ACL_FAMILY);
611      Result result = aclTable.get(get);
612      List<Cell> cells = result.listCells();
613      if (cells != null) {
614        for (Cell cell : cells) {
615          if (cell != null) {
616            users.add(Bytes.toString(CellUtil.cloneQualifier(cell)));
617          }
618        }
619      }
620      return users;
621    }

623    static Pair<Set<String>, Set<TableName>> getUserNamespaceAndTable(Table aclTable,
624        String userName) throws IOException {
625      Set<String> namespaces = new HashSet<>();
626      Set<TableName> tables = new HashSet<>();
627      List<byte[]> userEntries = SnapshotScannerHDFSAclStorage.getUserEntries(aclTable, userName);
628      for (byte[] entry : userEntries) {
629        if (PermissionStorage.isNamespaceEntry(entry)) {
630          namespaces.add(Bytes.toString(PermissionStorage.fromNamespaceEntry(entry)));
631        } else if (PermissionStorage.isTableEntry(entry)) {
632          tables.add(TableName.valueOf(entry));
633        }
634      }
635      return new Pair<>(namespaces, tables);
636    }

638    static List<byte[]> getUserEntries(Table aclTable, String userName) throws IOException {
639      Scan scan = new Scan();
640      scan.addColumn(HDFS_ACL_FAMILY, Bytes.toBytes(userName));
641      ResultScanner scanner = aclTable.getScanner(scan);
642      List<byte[]> entry = new ArrayList<>();
643      for (Result result : scanner) {
644        if (result != null && result.getRow() != null) {
645          entry.add(result.getRow());
646        }
647      }
648      return entry;
649    }
