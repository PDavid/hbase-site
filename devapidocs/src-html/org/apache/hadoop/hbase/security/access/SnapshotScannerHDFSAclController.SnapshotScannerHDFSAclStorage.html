071/**
072 * Set HDFS ACLs to hFiles to make HBase granted users have permission to scan snapshot
073 * <p>
074 * To use this feature, please mask sure HDFS config:
075 * <ul>
076 * <li>dfs.namenode.acls.enabled = true</li>
077 * <li>fs.permissions.umask-mode = 027 (or smaller umask than 027)</li>
078 * </ul>
079 * </p>
080 * <p>
081 * The implementation of this feature is as followings:
082 * <ul>
083 * <li>For common directories such as 'data' and 'archive', set other permission to '--x' to make
084 * everyone have the permission to access the directory.</li>
085 * <li>For namespace or table directories such as 'data/ns/table', 'archive/ns/table' and
086 * '.hbase-snapshot/snapshotName', set user 'r-x' access acl and 'r-x' default acl when following
087 * operations happen:
088 * <ul>
089 * <li>grant user with global, namespace or table permission;</li>
090 * <li>revoke user from global, namespace or table;</li>
091 * <li>snapshot table;</li>
092 * <li>truncate table;</li>
093 * </ul>
094 * </li>
095 * <li>Note: Because snapshots are at table level, so this feature just considers users with global,
096 * namespace or table permissions, ignores users with table CF or cell permissions.</li>
097 * </ul>
098 * </p>
099 */
100@CoreCoprocessor
101@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.CONFIG)
102public class SnapshotScannerHDFSAclController implements MasterCoprocessor, MasterObserver {
103  private static final Logger LOG = LoggerFactory.getLogger(SnapshotScannerHDFSAclController.class);

118  @Override
119  public void preMasterInitialization(ObserverContext<MasterCoprocessorEnvironment> c)
120      throws IOException {
121    if (c.getEnvironment().getConfiguration()
122        .getBoolean(SnapshotScannerHDFSAclHelper.USER_SCAN_SNAPSHOT_ENABLE, false)) {
123      MasterCoprocessorEnvironment mEnv = c.getEnvironment();
124      if (!(mEnv instanceof HasMasterServices)) {
125        throw new IOException("Does not implement HMasterServices");
126      }
127      masterServices = ((HasMasterServices) mEnv).getMasterServices();
128      hdfsAclHelper = new SnapshotScannerHDFSAclHelper(masterServices.getConfiguration(),
129          masterServices.getConnection());
130      pathHelper = hdfsAclHelper.getPathHelper();
131      hdfsAclHelper.setCommonDirectoryPermission();
132      initialized = true;
133      userProvider = UserProvider.instantiate(c.getEnvironment().getConfiguration());
134    } else {
135      LOG.warn("Try to initialize the coprocessor SnapshotScannerHDFSAclController but failure "
136          + "because the config " + SnapshotScannerHDFSAclHelper.USER_SCAN_SNAPSHOT_ENABLE
137          + " is false.");
138    }
139  }

141  @Override
142  public void postStartMaster(ObserverContext<MasterCoprocessorEnvironment> c) throws IOException {
143    if (!initialized) {
144      return;
145    }
146    try (Admin admin = c.getEnvironment().getConnection().getAdmin()) {
147      if (admin.tableExists(PermissionStorage.ACL_TABLE_NAME)) {
148        // Check if acl table has 'm' CF, if not, add 'm' CF
149        TableDescriptor tableDescriptor = admin.getDescriptor(PermissionStorage.ACL_TABLE_NAME);
150        boolean containHdfsAclFamily = Arrays.stream(tableDescriptor.getColumnFamilies()).anyMatch(
151          family -> Bytes.equals(family.getName(), SnapshotScannerHDFSAclStorage.HDFS_ACL_FAMILY));
152        if (!containHdfsAclFamily) {
153          TableDescriptorBuilder builder = TableDescriptorBuilder.newBuilder(tableDescriptor)
154              .setColumnFamily(ColumnFamilyDescriptorBuilder
155                  .newBuilder(SnapshotScannerHDFSAclStorage.HDFS_ACL_FAMILY).build());
156          admin.modifyTable(builder.build());
157        }
158        aclTableInitialized = true;
159      } else {
160        throw new TableNotFoundException("Table " + PermissionStorage.ACL_TABLE_NAME
161            + " is not created yet. Please check if " + getClass().getName()
162            + " is configured after " + AccessController.class.getName());
163      }
164    }
165  }

174  @Override
175  public void postCompletedCreateTableAction(ObserverContext<MasterCoprocessorEnvironment> c,
176      TableDescriptor desc, RegionInfo[] regions) throws IOException {
177    if (needHandleTableHdfsAcl(desc, "createTable " + desc.getTableName())) {
178      TableName tableName = desc.getTableName();
179      // 1. Create table directories to make HDFS acls can be inherited
180      hdfsAclHelper.createTableDirectories(tableName);
181      // 2. Add table owner HDFS acls
182      String owner =
183          desc.getOwnerString() == null ? getActiveUser(c).getShortName() : desc.getOwnerString();
184      hdfsAclHelper.addTableAcl(tableName, Sets.newHashSet(owner), "create");
185      // 3. Record table owner permission is synced to HDFS in acl table
186      SnapshotScannerHDFSAclStorage.addUserTableHdfsAcl(c.getEnvironment().getConnection(), owner,
187        tableName);
188    }
189  }

193      NamespaceDescriptor ns) throws IOException {

212  @Override
213  public void postCompletedTruncateTableAction(ObserverContext<MasterCoprocessorEnvironment> c,
214      TableName tableName) throws IOException {
215    if (needHandleTableHdfsAcl(tableName, "truncateTable " + tableName)) {
216      // Since the table directories is recreated, so add HDFS acls again
217      Set<String> users = hdfsAclHelper.getUsersWithTableReadAction(tableName, false, false);
218      hdfsAclHelper.addTableAcl(tableName, users, "truncate");

222  @Override
223  public void postDeleteTable(ObserverContext<MasterCoprocessorEnvironment> ctx,
224      TableName tableName) throws IOException {
225    if (needHandleTableHdfsAcl(tableName, "deleteTable " + tableName)) {
226      /*
227       * Remove table user access HDFS acl from namespace directory if the user has no permissions
228       * of global, ns of the table or other tables of the ns, eg: Bob has 'ns1:t1' read permission,
229       * when delete 'ns1:t1', if Bob has global read permission, '@ns1' read permission or
230       * 'ns1:other_tables' read permission, then skip remove Bob access acl in ns1Dirs, otherwise,
231       * remove Bob access acl.
232       */
233      try (Table aclTable =
234          ctx.getEnvironment().getConnection().getTable(PermissionStorage.ACL_TABLE_NAME)) {
235        Set<String> users = SnapshotScannerHDFSAclStorage.getTableUsers(aclTable, tableName);
236        // 1. Delete table owner permission is synced to HDFS in acl table
237        SnapshotScannerHDFSAclStorage.deleteTableHdfsAcl(aclTable, tableName);
238        // 2. Remove namespace access acls
239        Set<String> removeUsers = filterUsersToRemoveNsAccessAcl(aclTable, tableName, users);
240        if (removeUsers.size() > 0) {
241          hdfsAclHelper.removeNamespaceAccessAcl(tableName, removeUsers, "delete");
242        }
243      }
244    }
245  }

247  @Override
248  public void postModifyTable(ObserverContext<MasterCoprocessorEnvironment> ctx,
249      TableName tableName, TableDescriptor oldDescriptor, TableDescriptor currentDescriptor)
250      throws IOException {
251    try (Table aclTable =
252        ctx.getEnvironment().getConnection().getTable(PermissionStorage.ACL_TABLE_NAME)) {
253      if (needHandleTableHdfsAcl(currentDescriptor, "modifyTable " + tableName)
254          && !hdfsAclHelper.isTableUserScanSnapshotEnabled(oldDescriptor)) {
255        // 1. Create table directories used for acl inherited
256        hdfsAclHelper.createTableDirectories(tableName);
257        // 2. Add table users HDFS acls
258        Set<String> tableUsers = hdfsAclHelper.getUsersWithTableReadAction(tableName, false, false);
259        Set<String> users =
260            hdfsAclHelper.getUsersWithNamespaceReadAction(tableName.getNamespaceAsString(), true);
261        users.addAll(tableUsers);
262        hdfsAclHelper.addTableAcl(tableName, users, "modify");
263        // 3. Record table user acls are synced to HDFS in acl table
264        SnapshotScannerHDFSAclStorage.addUserTableHdfsAcl(ctx.getEnvironment().getConnection(),
265          tableUsers, tableName);
266      } else if (needHandleTableHdfsAcl(oldDescriptor, "modifyTable " + tableName)
267          && !hdfsAclHelper.isTableUserScanSnapshotEnabled(currentDescriptor)) {
268        // 1. Remove empty table directories
269        List<Path> tableRootPaths = hdfsAclHelper.getTableRootPaths(tableName, false);
270        for (Path path : tableRootPaths) {
271          hdfsAclHelper.deleteEmptyDir(path);
272        }
273        // 2. Remove all table HDFS acls
274        Set<String> tableUsers = hdfsAclHelper.getUsersWithTableReadAction(tableName, false, false);
275        Set<String> users = hdfsAclHelper
276            .getUsersWithNamespaceReadAction(tableName.getNamespaceAsString(), true);
277        users.addAll(tableUsers);
278        hdfsAclHelper.removeTableAcl(tableName, users);
279        // 3. Remove namespace access HDFS acls for users who only own permission for this table
280        hdfsAclHelper.removeNamespaceAccessAcl(tableName,
281          filterUsersToRemoveNsAccessAcl(aclTable, tableName, tableUsers), "modify");
282        // 4. Record table user acl is not synced to HDFS
283        SnapshotScannerHDFSAclStorage.deleteUserTableHdfsAcl(ctx.getEnvironment().getConnection(),
284          tableUsers, tableName);
285      }
286    }
287  }

291      String namespace) throws IOException {
292    if (checkInitialized("deleteNamespace " + namespace)) {
293      // 1. Record namespace user acl is not synced to HDFS
294      SnapshotScannerHDFSAclStorage.deleteNamespaceHdfsAcl(ctx.getEnvironment().getConnection(),

296      // 2. Delete tmp namespace directory

303      hdfsAclHelper.deleteEmptyDir(pathHelper.getTmpNsDir(namespace));
304    }

307  @Override
308  public void postGrant(ObserverContext<MasterCoprocessorEnvironment> c,
309      UserPermission userPermission, boolean mergeExistingPermissions) throws IOException {
310    if (!checkInitialized(
311      "grant " + userPermission + ", merge existing permissions " + mergeExistingPermissions)) {
312      return;
313    }
314    try (Table aclTable =
315        c.getEnvironment().getConnection().getTable(PermissionStorage.ACL_TABLE_NAME)) {
316      Configuration conf = c.getEnvironment().getConfiguration();
317      String userName = userPermission.getUser();
318      switch (userPermission.getAccessScope()) {
319        case GLOBAL:
320          UserPermission perm = getUserGlobalPermission(conf, userName);
321          if (perm != null && hdfsAclHelper.containReadAction(perm)) {
322            if (!isHdfsAclSet(aclTable, userName)) {
323              // 1. Get namespaces and tables which global user acls are already synced
324              Pair<Set<String>, Set<TableName>> skipNamespaceAndTables =
325                  SnapshotScannerHDFSAclStorage.getUserNamespaceAndTable(aclTable, userName);
326              Set<String> skipNamespaces = skipNamespaceAndTables.getFirst();
327              Set<TableName> skipTables = skipNamespaceAndTables.getSecond().stream()
328                  .filter(t -> !skipNamespaces.contains(t.getNamespaceAsString()))
329                  .collect(Collectors.toSet());
330              // 2. Add HDFS acl(skip namespaces and tables directories whose acl is set)
331              hdfsAclHelper.grantAcl(userPermission, skipNamespaces, skipTables);
332              // 3. Record global acl is sync to HDFS
333              SnapshotScannerHDFSAclStorage.addUserGlobalHdfsAcl(aclTable, userName);
334            }
335          } else {
336            // The merged user permission doesn't contain READ, so remove user global HDFS acls if
337            // it's set
338            removeUserGlobalHdfsAcl(aclTable, userName, userPermission);
339          }
340          break;
341        case NAMESPACE:
342          String namespace = ((NamespacePermission) userPermission.getPermission()).getNamespace();
343          UserPermission nsPerm = getUserNamespacePermission(conf, userName, namespace);
344          if (nsPerm != null && hdfsAclHelper.containReadAction(nsPerm)) {
345            if (!isHdfsAclSet(aclTable, userName, namespace)) {
346              // 1. Get tables which namespace user acls are already synced
347              Set<TableName> skipTables = SnapshotScannerHDFSAclStorage
348                  .getUserNamespaceAndTable(aclTable, userName).getSecond();
349              // 2. Add HDFS acl(skip tables directories whose acl is set)
350              hdfsAclHelper.grantAcl(userPermission, new HashSet<>(0), skipTables);
351            }
352            // 3. Record namespace acl is synced to HDFS
353            SnapshotScannerHDFSAclStorage.addUserNamespaceHdfsAcl(aclTable, userName, namespace);
354          } else {
355            // The merged user permission doesn't contain READ, so remove user namespace HDFS acls
356            // if it's set
357            removeUserNamespaceHdfsAcl(aclTable, userName, namespace, userPermission);
358          }
359          break;
360        case TABLE:
361          TablePermission tablePerm = (TablePermission) userPermission.getPermission();
362          if (needHandleTableHdfsAcl(tablePerm)) {
363            TableName tableName = tablePerm.getTableName();
364            UserPermission tPerm = getUserTablePermission(conf, userName, tableName);
365            if (tPerm != null && hdfsAclHelper.containReadAction(tPerm)) {
366              if (!isHdfsAclSet(aclTable, userName, tableName)) {
367                // 1. Add HDFS acl
368                hdfsAclHelper.grantAcl(userPermission, new HashSet<>(0), new HashSet<>(0));
369              }
370              // 2. Record table acl is synced to HDFS
371              SnapshotScannerHDFSAclStorage.addUserTableHdfsAcl(aclTable, userName, tableName);
372            } else {
373              // The merged user permission doesn't contain READ, so remove user table HDFS acls if
374              // it's set
375              removeUserTableHdfsAcl(aclTable, userName, tableName, userPermission);
376            }
377          }
378          break;
379        default:
380          throw new IllegalArgumentException(
381              "Illegal user permission scope " + userPermission.getAccessScope());
382      }
383    }
384  }

386  @Override
387  public void postRevoke(ObserverContext<MasterCoprocessorEnvironment> c,
388      UserPermission userPermission) throws IOException {
389    if (checkInitialized("revoke " + userPermission)) {
390      try (Table aclTable =
391          c.getEnvironment().getConnection().getTable(PermissionStorage.ACL_TABLE_NAME)) {
392        String userName = userPermission.getUser();
393        Configuration conf = c.getEnvironment().getConfiguration();
394        switch (userPermission.getAccessScope()) {
395          case GLOBAL:
396            UserPermission userGlobalPerm = getUserGlobalPermission(conf, userName);
397            if (userGlobalPerm == null || !hdfsAclHelper.containReadAction(userGlobalPerm)) {
398              removeUserGlobalHdfsAcl(aclTable, userName, userPermission);
399            }
400            break;
401          case NAMESPACE:
402            NamespacePermission nsPerm = (NamespacePermission) userPermission.getPermission();
403            UserPermission userNsPerm =
404                getUserNamespacePermission(conf, userName, nsPerm.getNamespace());
405            if (userNsPerm == null || !hdfsAclHelper.containReadAction(userNsPerm)) {
406              removeUserNamespaceHdfsAcl(aclTable, userName, nsPerm.getNamespace(), userPermission);
407            }
408            break;
409          case TABLE:
410            TablePermission tPerm = (TablePermission) userPermission.getPermission();
411            if (needHandleTableHdfsAcl(tPerm)) {
412              TableName tableName = tPerm.getTableName();
413              UserPermission userTablePerm = getUserTablePermission(conf, userName, tableName);
414              if (userTablePerm == null || !hdfsAclHelper.containReadAction(userTablePerm)) {
415                removeUserTableHdfsAcl(aclTable, userName, tableName, userPermission);
416              }
417            }
418            break;
419          default:
420            throw new IllegalArgumentException(
421                "Illegal user permission scope " + userPermission.getAccessScope());
422        }
423      }
424    }
425  }

427  private void removeUserGlobalHdfsAcl(Table aclTable, String userName,
428      UserPermission userPermission) throws IOException {
429    if (SnapshotScannerHDFSAclStorage.hasUserGlobalHdfsAcl(aclTable, userName)) {
430      // 1. Get namespaces and tables which global user acls are already synced
431      Pair<Set<String>, Set<TableName>> namespaceAndTable =
432          SnapshotScannerHDFSAclStorage.getUserNamespaceAndTable(aclTable, userName);
433      Set<String> skipNamespaces = namespaceAndTable.getFirst();
434      Set<TableName> skipTables = namespaceAndTable.getSecond().stream()
435          .filter(t -> !skipNamespaces.contains(t.getNamespaceAsString()))
436          .collect(Collectors.toSet());
437      // 2. Remove user HDFS acls(skip namespaces and tables directories
438      // whose acl must be reversed)
439      hdfsAclHelper.revokeAcl(userPermission, skipNamespaces, skipTables);
440      // 3. Remove global user acl is synced to HDFS in acl table
441      SnapshotScannerHDFSAclStorage.deleteUserGlobalHdfsAcl(aclTable, userName);
442    }
443  }
444
445  private void removeUserNamespaceHdfsAcl(Table aclTable, String userName, String namespace,
446      UserPermission userPermission) throws IOException {
447    if (SnapshotScannerHDFSAclStorage.hasUserNamespaceHdfsAcl(aclTable, userName, namespace)) {
448      if (!SnapshotScannerHDFSAclStorage.hasUserGlobalHdfsAcl(aclTable, userName)) {
449        // 1. Get tables whose namespace user acls are already synced
450        Set<TableName> skipTables =
451            SnapshotScannerHDFSAclStorage.getUserNamespaceAndTable(aclTable, userName).getSecond();
452        // 2. Remove user HDFS acls(skip tables directories whose acl must be reversed)
453        hdfsAclHelper.revokeAcl(userPermission, new HashSet<>(), skipTables);
454      }
455      // 3. Remove namespace user acl is synced to HDFS in acl table
456      SnapshotScannerHDFSAclStorage.deleteUserNamespaceHdfsAcl(aclTable, userName, namespace);
457    }
458  }
459
460  private void removeUserTableHdfsAcl(Table aclTable, String userName, TableName tableName,
461      UserPermission userPermission) throws IOException {
462    if (SnapshotScannerHDFSAclStorage.hasUserTableHdfsAcl(aclTable, userName, tableName)) {
463      if (!SnapshotScannerHDFSAclStorage.hasUserGlobalHdfsAcl(aclTable, userName)
464          && !SnapshotScannerHDFSAclStorage.hasUserNamespaceHdfsAcl(aclTable, userName,
465            tableName.getNamespaceAsString())) {
466        // 1. Remove table acls
467        hdfsAclHelper.revokeAcl(userPermission, new HashSet<>(0), new HashSet<>(0));
468      }
469      // 2. Remove table user acl is synced to HDFS in acl table
470      SnapshotScannerHDFSAclStorage.deleteUserTableHdfsAcl(aclTable, userName, tableName);
471    }
472  }

474  private UserPermission getUserGlobalPermission(Configuration conf, String userName)
475      throws IOException {
476    List<UserPermission> permissions = PermissionStorage.getUserPermissions(conf,
477      PermissionStorage.ACL_GLOBAL_NAME, null, null, userName, true);
478    return permissions.size() > 0 ? permissions.get(0) : null;
479  }
480
481  private UserPermission getUserNamespacePermission(Configuration conf, String userName,
482      String namespace) throws IOException {
483    List<UserPermission> permissions =
484        PermissionStorage.getUserNamespacePermissions(conf, namespace, userName, true);
485    return permissions.size() > 0 ? permissions.get(0) : null;
486  }
487
488  private UserPermission getUserTablePermission(Configuration conf, String userName,
489      TableName tableName) throws IOException {
490    List<UserPermission> permissions = PermissionStorage
491        .getUserTablePermissions(conf, tableName, null, null, userName, true).stream()
492        .filter(userPermission -> hdfsAclHelper
493            .isNotFamilyOrQualifierPermission((TablePermission) userPermission.getPermission()))
494        .collect(Collectors.toList());
495    return permissions.size() > 0 ? permissions.get(0) : null;
496  }

512  /**
513   * Check if user global/namespace/table HDFS acls is already set
514   */
515  private boolean isHdfsAclSet(Table aclTable, String userName, String namespace,
516      TableName tableName) throws IOException {
517    boolean isSet = SnapshotScannerHDFSAclStorage.hasUserGlobalHdfsAcl(aclTable, userName);
518    if (namespace != null) {
519      isSet = isSet
520          || SnapshotScannerHDFSAclStorage.hasUserNamespaceHdfsAcl(aclTable, userName, namespace);
521    }
522    if (tableName != null) {
523      isSet = isSet
524          || SnapshotScannerHDFSAclStorage.hasUserNamespaceHdfsAcl(aclTable, userName,
525            tableName.getNamespaceAsString())
526          || SnapshotScannerHDFSAclStorage.hasUserTableHdfsAcl(aclTable, userName, tableName);
527    }
528    return isSet;
529  }

531  @VisibleForTesting
532  boolean checkInitialized(String operation) {
533    if (initialized) {
534      if (aclTableInitialized) {
535        return true;
536      } else {
537        LOG.warn("Skip set HDFS acls because acl table is not initialized when " + operation);
538      }
539    }
540    return false;
541  }

568  /**
569   * Remove table user access HDFS acl from namespace directory if the user has no permissions of
570   * global, ns of the table or other tables of the ns, eg: Bob has 'ns1:t1' read permission, when
571   * delete 'ns1:t1', if Bob has global read permission, '@ns1' read permission or
572   * 'ns1:other_tables' read permission, then skip remove Bob access acl in ns1Dirs, otherwise,
573   * remove Bob access acl.
574   * @param aclTable acl table
575   * @param tableName the name of the table
576   * @param tablesUsers table users set
577   * @return users whose access acl will be removed from the namespace of the table
578   * @throws IOException if an error occurred
579   */
580  private Set<String> filterUsersToRemoveNsAccessAcl(Table aclTable, TableName tableName,
581      Set<String> tablesUsers) throws IOException {
582    Set<String> removeUsers = new HashSet<>();
583    byte[] namespace = tableName.getNamespace();
584    for (String user : tablesUsers) {
585      List<byte[]> userEntries = SnapshotScannerHDFSAclStorage.getUserEntries(aclTable, user);
586      boolean remove = true;
587      for (byte[] entry : userEntries) {
588        if (PermissionStorage.isGlobalEntry(entry)
589            || (PermissionStorage.isNamespaceEntry(entry)
590                && Bytes.equals(PermissionStorage.fromNamespaceEntry(entry), namespace))
591            || (!Bytes.equals(tableName.getName(), entry)
592                && Bytes.equals(TableName.valueOf(entry).getNamespace(), namespace))) {
593          remove = false;
594          break;
595        }
596      }
597      if (remove) {
598        removeUsers.add(user);
599      }
600    }
601    return removeUsers;
602  }

604  static final class SnapshotScannerHDFSAclStorage {
605    /**
606     * Add a new CF in HBase acl table to record if the HBase read permission is synchronized to
607     * related hfiles. The record has two usages: 1. check if we need to remove HDFS acls for a
608     * grant without READ permission(eg: grant user table read permission and then grant user table
609     * write permission without merging the existing permissions, in this case, need to remove HDFS
610     * acls); 2. skip some HDFS acl sync because it may be already set(eg: grant user table read
611     * permission and then grant user ns read permission; grant user table read permission and then
612     * grant user table write permission with merging the existing permissions).
613     */
614    static final byte[] HDFS_ACL_FAMILY = Bytes.toBytes("m");
615    // The value 'R' has no specific meaning, if cell value is not null, it means that the user HDFS
616    // acls is set to hfiles.
617    private static final byte[] HDFS_ACL_VALUE = Bytes.toBytes("R");
618

705    private static Set<String> getEntryUsers(Table aclTable, byte[] entry) throws IOException {
706      Set<String> users = new HashSet<>();
707      Get get = new Get(entry);
708      get.addFamily(HDFS_ACL_FAMILY);
709      Result result = aclTable.get(get);
710      List<Cell> cells = result.listCells();
711      if (cells != null) {
712        for (Cell cell : cells) {
713          if (cell != null) {
714            users.add(Bytes.toString(CellUtil.cloneQualifier(cell)));
715          }
716        }
717      }
718      return users;
719    }

721    static Pair<Set<String>, Set<TableName>> getUserNamespaceAndTable(Table aclTable,
722        String userName) throws IOException {
723      Set<String> namespaces = new HashSet<>();
724      Set<TableName> tables = new HashSet<>();
725      List<byte[]> userEntries = getUserEntries(aclTable, userName);
726      for (byte[] entry : userEntries) {
727        if (PermissionStorage.isNamespaceEntry(entry)) {
728          namespaces.add(Bytes.toString(PermissionStorage.fromNamespaceEntry(entry)));
729        } else if (PermissionStorage.isTableEntry(entry)) {
730          tables.add(TableName.valueOf(entry));
731        }
732      }
733      return new Pair<>(namespaces, tables);
734    }

736    static List<byte[]> getUserEntries(Table aclTable, String userName) throws IOException {
737      Scan scan = new Scan();
738      scan.addColumn(HDFS_ACL_FAMILY, Bytes.toBytes(userName));
739      ResultScanner scanner = aclTable.getScanner(scan);
740      List<byte[]> entry = new ArrayList<>();
741      for (Result result : scanner) {
742        if (result != null && result.getRow() != null) {
743          entry.add(result.getRow());
744        }
745      }
746      return entry;
747    }
