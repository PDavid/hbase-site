104/**
105 * HBase Canary Tool for "canary monitoring" of a running HBase cluster.
106 *
107 * There are three modes:
108 * <ol>
109 * <li>region mode (Default): For each region, try to get one row per column family outputting
110 * information on failure (ERROR) or else the latency.
111 * </li>
112 *
113 * <li>regionserver mode: For each regionserver try to get one row from one table selected
114 * randomly outputting information on failure (ERROR) or else the latency.
115 * </li>
116 *
117 * <li>zookeeper mode: for each zookeeper instance, selects a znode outputting information on
118 * failure (ERROR) or else the latency.
119 * </li>
120 * </ol>
121 */
122@InterfaceAudience.Private
123public final class Canary implements Tool {

132    long getWriteFailureCount();
133    long incWriteFailureCount();
134    Map<String,String> getWriteFailures();
135    void updateWriteFailures(String regionName, String serverName);
136  }

278  /**
279   * Run a single zookeeper Task and then exit.
280   */
281  static class ZookeeperTask implements Callable<Void> {
282    private final Connection connection;
283    private final String host;
284    private String znode;
285    private final int timeout;
286    private ZookeeperStdOutSink sink;
287
288    public ZookeeperTask(Connection connection, String host, String znode, int timeout,
289        ZookeeperStdOutSink sink) {
290      this.connection = connection;
291      this.host = host;
292      this.znode = znode;
293      this.timeout = timeout;
294      this.sink = sink;
295    }
296
297    @Override public Void call() throws Exception {
298      ZooKeeper zooKeeper = null;
299      try {
300        zooKeeper = new ZooKeeper(host, timeout, EmptyWatcher.instance);
301        Stat exists = zooKeeper.exists(znode, false);
302        StopWatch stopwatch = new StopWatch();
303        stopwatch.start();
304        zooKeeper.getData(znode, false, exists);
305        stopwatch.stop();
306        sink.publishReadTiming(znode, host, stopwatch.getTime());
307      } catch (KeeperException | InterruptedException e) {
308        sink.publishReadFailure(znode, host);
309      } finally {
310        if (zooKeeper != null) {
311          zooKeeper.close();
312        }
313      }
314      return null;
315    }
316  }

318  /**
319   * Run a single Region Task and then exit. For each column family of the Region, get one row and
320   * output latency or failure.
321   */
322  static class RegionTask implements Callable<Void> {
323    public enum TaskType{
324      READ, WRITE
325    }
326    private Connection connection;
327    private RegionInfo region;
328    private RegionStdOutSink sink;
329    private TaskType taskType;
330    private boolean rawScanEnabled;
331    private ServerName serverName;
332    private LongAdder readWriteLatency;
333
334    RegionTask(Connection connection, RegionInfo region, ServerName serverName,
335        RegionStdOutSink sink, TaskType taskType, boolean rawScanEnabled, LongAdder rwLatency) {
336      this.connection = connection;
337      this.region = region;
338      this.serverName = serverName;
339      this.sink = sink;
340      this.taskType = taskType;
341      this.rawScanEnabled = rawScanEnabled;
342      this.readWriteLatency = rwLatency;
343    }
344
345    @Override
346    public Void call() {
347      switch (taskType) {
348      case READ:
349        return read();
350      case WRITE:
351        return write();
352      default:
353        return read();
354      }
355    }
356
357    public Void read() {
358      Table table = null;
359      TableDescriptor tableDesc = null;
360      try {
361        LOG.debug("Reading table descriptor for table {}", region.getTable());
362        table = connection.getTable(region.getTable());
363        tableDesc = table.getDescriptor();
364      } catch (IOException e) {
365        LOG.debug("sniffRegion {} of {} failed", region.getEncodedName(), e);
366        sink.publishReadFailure(serverName, region, e);
367        if (table != null) {
368          try {
369            table.close();
370          } catch (IOException ioe) {
371            LOG.error("Close table failed", e);
372          }
373        }
374        return null;
375      }
376
377      byte[] startKey = null;
378      Get get = null;
379      Scan scan = null;
380      ResultScanner rs = null;
381      StopWatch stopWatch = new StopWatch();
382      for (ColumnFamilyDescriptor column : tableDesc.getColumnFamilies()) {
383        stopWatch.reset();
384        startKey = region.getStartKey();
385        // Can't do a get on empty start row so do a Scan of first element if any instead.
386        if (startKey.length > 0) {
387          get = new Get(startKey);
388          get.setCacheBlocks(false);
389          get.setFilter(new FirstKeyOnlyFilter());
390          get.addFamily(column.getName());
391        } else {
392          scan = new Scan();
393          LOG.debug("rawScan {} for {}", rawScanEnabled, tableDesc.getTableName());
394          scan.setRaw(rawScanEnabled);
395          scan.setCaching(1);
396          scan.setCacheBlocks(false);
397          scan.setFilter(new FirstKeyOnlyFilter());
398          scan.addFamily(column.getName());
399          scan.setMaxResultSize(1L);
400          scan.setOneRowLimit();
401        }
402        LOG.debug("Reading from {} {} {} {}", tableDesc.getTableName(),
403            region.getRegionNameAsString(), column.getNameAsString(),
404            Bytes.toStringBinary(startKey));
405        try {
406          stopWatch.start();
407          if (startKey.length > 0) {
408            table.get(get);
409          } else {
410            rs = table.getScanner(scan);
411            rs.next();
412          }
413          stopWatch.stop();
414          this.readWriteLatency.add(stopWatch.getTime());
415          sink.publishReadTiming(serverName, region, column, stopWatch.getTime());
416        } catch (Exception e) {
417          sink.publishReadFailure(serverName, region, column, e);
418          sink.updateReadFailures(region.getRegionNameAsString(), serverName.getHostname());
419        } finally {
420          if (rs != null) {
421            rs.close();
422          }
423          scan = null;
424          get = null;
425        }
426      }
427      try {
428        table.close();
429      } catch (IOException e) {
430        LOG.error("Close table failed", e);
431      }
432      return null;
433    }
434
435    /**
436     * Check writes for the canary table
437     */
438    private Void write() {
439      Table table = null;
440      TableDescriptor tableDesc = null;
441      try {
442        table = connection.getTable(region.getTable());
443        tableDesc = table.getDescriptor();
444        byte[] rowToCheck = region.getStartKey();
445        if (rowToCheck.length == 0) {
446          rowToCheck = new byte[]{0x0};
447        }
448        int writeValueSize =
449            connection.getConfiguration().getInt(HConstants.HBASE_CANARY_WRITE_VALUE_SIZE_KEY, 10);
450        for (ColumnFamilyDescriptor column : tableDesc.getColumnFamilies()) {
451          Put put = new Put(rowToCheck);
452          byte[] value = new byte[writeValueSize];
453          Bytes.random(value);
454          put.addColumn(column.getName(), HConstants.EMPTY_BYTE_ARRAY, value);
455
456          LOG.debug("Writing to {} {} {} {}",
457            tableDesc.getTableName(), region.getRegionNameAsString(), column.getNameAsString(),
458            Bytes.toStringBinary(rowToCheck));
459          try {
460            long startTime = System.currentTimeMillis();
461            table.put(put);
462            long time = System.currentTimeMillis() - startTime;
463            this.readWriteLatency.add(time);
464            sink.publishWriteTiming(serverName, region, column, time);
465          } catch (Exception e) {
466            sink.publishWriteFailure(serverName, region, column, e);
467          }
468        }
469        table.close();
470      } catch (IOException e) {
471        sink.publishWriteFailure(serverName, region, e);
472        sink.updateWriteFailures(region.getRegionNameAsString(), serverName.getHostname() );
473      }
474      return null;
475    }
476  }

478  /**
479   * Run a single RegionServer Task and then exit.
480   * Get one row from a region on the regionserver and output latency or the failure.
481   */
482  static class RegionServerTask implements Callable<Void> {
483    private Connection connection;
484    private String serverName;
485    private RegionInfo region;
486    private RegionServerStdOutSink sink;
487    private AtomicLong successes;
488
489    RegionServerTask(Connection connection, String serverName, RegionInfo region,
490        RegionServerStdOutSink sink, AtomicLong successes) {
491      this.connection = connection;
492      this.serverName = serverName;
493      this.region = region;
494      this.sink = sink;
495      this.successes = successes;
496    }
497
498    @Override
499    public Void call() {
500      TableName tableName = null;
501      Table table = null;
502      Get get = null;
503      byte[] startKey = null;
504      Scan scan = null;
505      StopWatch stopWatch = new StopWatch();
506      // monitor one region on every region server
507      stopWatch.reset();
508      try {
509        tableName = region.getTable();
510        table = connection.getTable(tableName);
511        startKey = region.getStartKey();
512        // Can't do a get on empty start row so do a Scan of first element if any instead.
513        LOG.debug("Reading from {} {} {} {}",
514          serverName, region.getTable(), region.getRegionNameAsString(),
515          Bytes.toStringBinary(startKey));
516        if (startKey.length > 0) {
517          get = new Get(startKey);
518          get.setCacheBlocks(false);
519          get.setFilter(new FirstKeyOnlyFilter());
520          stopWatch.start();
521          table.get(get);
522          stopWatch.stop();
523        } else {
524          scan = new Scan();
525          scan.setCacheBlocks(false);
526          scan.setFilter(new FirstKeyOnlyFilter());
527          scan.setCaching(1);
528          scan.setMaxResultSize(1L);
529          scan.setOneRowLimit();
530          stopWatch.start();
531          ResultScanner s = table.getScanner(scan);
532          s.next();
533          s.close();
534          stopWatch.stop();
535        }
536        successes.incrementAndGet();
537        sink.publishReadTiming(tableName.getNameAsString(), serverName, stopWatch.getTime());
538      } catch (TableNotFoundException tnfe) {
539        LOG.error("Table may be deleted", tnfe);
540        // This is ignored because it doesn't imply that the regionserver is dead
541      } catch (TableNotEnabledException tnee) {
542        // This is considered a success since we got a response.
543        successes.incrementAndGet();
544        LOG.debug("The targeted table was disabled.  Assuming success.");
545      } catch (DoNotRetryIOException dnrioe) {
546        sink.publishReadFailure(tableName.getNameAsString(), serverName);
547        LOG.error(dnrioe.toString(), dnrioe);
548      } catch (IOException e) {
549        sink.publishReadFailure(tableName.getNameAsString(), serverName);
550        LOG.error(e.toString(), e);
551      } finally {
552        if (table != null) {
553          try {
554            table.close();
555          } catch (IOException e) {/* DO NOTHING */
556            LOG.error("Close table failed", e);
557          }
558        }
559        scan = null;
560        get = null;
561        startKey = null;
562      }
563      return null;
564    }
565  }

643  private int parseArgs(String[] args) {
644    int index = -1;
645    // Process command line args
646    for (int i = 0; i < args.length; i++) {
647      String cmd = args[i];
648
649      if (cmd.startsWith("-")) {
650        if (index >= 0) {
651          // command line args must be in the form: [opts] [table 1 [table 2 ...]]
652          System.err.println("Invalid command line options");
653          printUsageAndExit();
654        }
655
656        if (cmd.equals("-help") || cmd.equals("-h")) {
657          // user asked for help, print the help and quit.
658          printUsageAndExit();
659        } else if (cmd.equals("-daemon") && interval == 0) {
660          // user asked for daemon mode, set a default interval between checks
661          interval = DEFAULT_INTERVAL;
662        } else if (cmd.equals("-interval")) {
663          // user has specified an interval for canary breaths (-interval N)
664          i++;
665
666          if (i == args.length) {
667            System.err.println("-interval takes a numeric seconds value argument.");
668            printUsageAndExit();
669          }
670
671          try {
672            interval = Long.parseLong(args[i]) * 1000;
673          } catch (NumberFormatException e) {
674            System.err.println("-interval needs a numeric value argument.");
675            printUsageAndExit();
676          }
677        } else if (cmd.equals("-zookeeper")) {
678          this.zookeeperMode = true;
679        } else if(cmd.equals("-regionserver")) {
680          this.regionServerMode = true;
681        } else if(cmd.equals("-allRegions")) {
682          this.regionServerAllRegions = true;
683        } else if(cmd.equals("-writeSniffing")) {
684          this.writeSniffing = true;
685        } else if(cmd.equals("-treatFailureAsError") || cmd.equals("-failureAsError")) {
686          this.treatFailureAsError = true;
687        } else if (cmd.equals("-e")) {
688          this.useRegExp = true;
689        } else if (cmd.equals("-t")) {
690          i++;
691
692          if (i == args.length) {
693            System.err.println("-t takes a numeric milliseconds value argument.");
694            printUsageAndExit();
695          }
696
697          try {
698            this.timeout = Long.parseLong(args[i]);
699          } catch (NumberFormatException e) {
700            System.err.println("-t takes a numeric milliseconds value argument.");
701            printUsageAndExit();
702          }
703        } else if(cmd.equals("-writeTableTimeout")) {
704          i++;
705
706          if (i == args.length) {
707            System.err.println("-writeTableTimeout takes a numeric milliseconds value argument.");
708            printUsageAndExit();
709          }
710
711          try {
712            this.configuredWriteTableTimeout = Long.parseLong(args[i]);
713          } catch (NumberFormatException e) {
714            System.err.println("-writeTableTimeout takes a numeric milliseconds value argument.");
715            printUsageAndExit();
716          }
717        } else if (cmd.equals("-writeTable")) {
718          i++;
719
720          if (i == args.length) {
721            System.err.println("-writeTable takes a string tablename value argument.");
722            printUsageAndExit();
723          }
724          this.writeTableName = TableName.valueOf(args[i]);
725        } else if (cmd.equals("-f")) {
726          i++;
727
728          if (i == args.length) {
729            System.err
730                .println("-f needs a boolean value argument (true|false).");
731            printUsageAndExit();
732          }
733
734          this.failOnError = Boolean.parseBoolean(args[i]);
735        } else if (cmd.equals("-readTableTimeouts")) {
736          i++;
737
738          if (i == args.length) {
739            System.err.println("-readTableTimeouts needs a comma-separated list of read " +
740                "millisecond timeouts per table (without spaces).");
741            printUsageAndExit();
742          }
743          String [] tableTimeouts = args[i].split(",");
744          for (String tT: tableTimeouts) {
745            String [] nameTimeout = tT.split("=");
746            if (nameTimeout.length < 2) {
747              System.err.println("Each -readTableTimeouts argument must be of the form " +
748                  "<tableName>=<read timeout> (without spaces).");
749              printUsageAndExit();
750            }
751            long timeoutVal = 0L;
752            try {
753              timeoutVal = Long.parseLong(nameTimeout[1]);
754            } catch (NumberFormatException e) {
755              System.err.println("-readTableTimeouts read timeout for each table must be a numeric value argument.");
756              printUsageAndExit();
757            }
758            this.configuredReadTableTimeouts.put(nameTimeout[0], timeoutVal);
759          }
760        } else if (cmd.equals("-permittedZookeeperFailures")) {
761          i++;
762
763          if (i == args.length) {
764            System.err.println("-permittedZookeeperFailures needs a numeric value argument.");
765            printUsageAndExit();
766          }
767          try {
768            this.permittedFailures = Long.parseLong(args[i]);
769          } catch (NumberFormatException e) {
770            System.err.println("-permittedZookeeperFailures needs a numeric value argument.");
771            printUsageAndExit();
772          }
773        } else {
774          // no options match
775          System.err.println(cmd + " options is invalid.");
776          printUsageAndExit();
777        }
778      } else if (index < 0) {
779        // keep track of first table name specified by the user
780        index = i;
781      }
782    }
783    if (this.regionServerAllRegions && !this.regionServerMode) {
784      System.err.println("-allRegions can only be specified in regionserver mode.");
785      printUsageAndExit();
786    }
787    if (this.zookeeperMode) {
788      if (this.regionServerMode || this.regionServerAllRegions || this.writeSniffing) {
789        System.err.println("-zookeeper is exclusive and cannot be combined with "
790            + "other modes.");
791        printUsageAndExit();
792      }
793    }
794    if (this.permittedFailures != 0 && !this.zookeeperMode) {
795      System.err.println("-permittedZookeeperFailures requires -zookeeper mode.");
796      printUsageAndExit();
797    }
798    if (!this.configuredReadTableTimeouts.isEmpty() && (this.regionServerMode || this.zookeeperMode)) {
799      System.err.println("-readTableTimeouts can only be configured in region mode.");
800      printUsageAndExit();
801    }
802    return index;
803  }

805  @Override
806  public int run(String[] args) throws Exception {
807    int index = parseArgs(args);
808    ChoreService choreService = null;
809
810    // Launches chore for refreshing kerberos credentials if security is enabled.
811    // Please see http://hbase.apache.org/book.html#_running_canary_in_a_kerberos_enabled_cluster
812    // for more details.
813    final ScheduledChore authChore = AuthUtil.getAuthChore(conf);
814    if (authChore != null) {
815      choreService = new ChoreService("CANARY_TOOL");
816      choreService.scheduleChore(authChore);
817    }
818
819    // Start to prepare the stuffs
820    Monitor monitor = null;
821    Thread monitorThread = null;
822    long startTime = 0;
823    long currentTimeLength = 0;
824    // Get a connection to use in below.
825    try (Connection connection = ConnectionFactory.createConnection(this.conf)) {
826      do {
827        // Do monitor !!
828        try {
829          monitor = this.newMonitor(connection, index, args);
830          monitorThread = new Thread(monitor, "CanaryMonitor-" + System.currentTimeMillis());
831          startTime = System.currentTimeMillis();
832          monitorThread.start();
833          while (!monitor.isDone()) {
834            // wait for 1 sec
835            Thread.sleep(1000);
836            // exit if any error occurs
837            if (this.failOnError && monitor.hasError()) {
838              monitorThread.interrupt();
839              if (monitor.initialized) {
840                return monitor.errorCode;
841              } else {
842                return INIT_ERROR_EXIT_CODE;
843              }
844            }
845            currentTimeLength = System.currentTimeMillis() - startTime;
846            if (currentTimeLength > this.timeout) {
847              LOG.error("The monitor is running too long (" + currentTimeLength
848                  + ") after timeout limit:" + this.timeout
849                  + " will be killed itself !!");
850              if (monitor.initialized) {
851                return TIMEOUT_ERROR_EXIT_CODE;
852              } else {
853                return INIT_ERROR_EXIT_CODE;
854              }
855            }
856          }
857
858          if (this.failOnError && monitor.finalCheckForErrors()) {
859            monitorThread.interrupt();
860            return monitor.errorCode;
861          }
862        } finally {
863          if (monitor != null) monitor.close();
864        }
865
866        Thread.sleep(interval);
867      } while (interval > 0);
868    } // try-with-resources close
869
870    if (choreService != null) {
871      choreService.shutdown();
872    }
873    return monitor.errorCode;
874  }

974   * A Monitor super-class can be extended by users
975   */
976  public static abstract class Monitor implements Runnable, Closeable {
977    protected Connection connection;
978    protected Admin admin;
979    /**
980     * 'Target' dependent on 'mode'. Could be Tables or RegionServers or ZNodes.
981     * Passed on the command-line as arguments.
982     */
983    protected String[] targets;
984    protected boolean useRegExp;
985    protected boolean treatFailureAsError;
986    protected boolean initialized = false;
987
988    protected boolean done = false;
989    protected int errorCode = 0;
990    protected long allowedFailures = 0;
991    protected Sink sink;
992    protected ExecutorService executor;
993
994    public boolean isDone() {
995      return done;
996    }
997
998    public boolean hasError() {
999      return errorCode != 0;
1000    }
1001
1002    public boolean finalCheckForErrors() {
1003      if (errorCode != 0) {
1004        return true;
1005      }
1006      if (treatFailureAsError &&
1007          (sink.getReadFailureCount() > allowedFailures || sink.getWriteFailureCount() > allowedFailures)) {
1008        LOG.error("Too many failures detected, treating failure as error, failing the Canary.");
1009        errorCode = FAILURE_EXIT_CODE;
1010        return true;
1011      }
1012      return false;
1013    }
1014
1015    @Override
1016    public void close() throws IOException {
1017      if (this.admin != null) this.admin.close();
1018    }
1019
1020    protected Monitor(Connection connection, String[] monitorTargets, boolean useRegExp, Sink sink,
1021        ExecutorService executor, boolean treatFailureAsError, long allowedFailures) {
1022      if (null == connection) throw new IllegalArgumentException("connection shall not be null");
1023
1024      this.connection = connection;
1025      this.targets = monitorTargets;
1026      this.useRegExp = useRegExp;
1027      this.treatFailureAsError = treatFailureAsError;
1028      this.sink = sink;
1029      this.executor = executor;
1030      this.allowedFailures = allowedFailures;
1031    }
1032
1033    @Override
1034    public abstract void run();
1035
1036    protected boolean initAdmin() {
1037      if (null == this.admin) {
1038        try {
1039          this.admin = this.connection.getAdmin();
1040        } catch (Exception e) {
1041          LOG.error("Initial HBaseAdmin failed...", e);
1042          this.errorCode = INIT_ERROR_EXIT_CODE;
1043        }
1044      } else if (admin.isAborted()) {
1045        LOG.error("HBaseAdmin aborted");
1046        this.errorCode = INIT_ERROR_EXIT_CODE;
1047      }
1048      return !this.hasError();
1049    }

1053   * A monitor for region mode.
1054   */
1055  private static class RegionMonitor extends Monitor {
1056    // 10 minutes
1057    private static final int DEFAULT_WRITE_TABLE_CHECK_PERIOD = 10 * 60 * 1000;
1058    // 1 days
1059    private static final int DEFAULT_WRITE_DATA_TTL = 24 * 60 * 60;
1060
1061    private long lastCheckTime = -1;
1062    private boolean writeSniffing;
1063    private TableName writeTableName;
1064    private int writeDataTTL;
1065    private float regionsLowerLimit;
1066    private float regionsUpperLimit;
1067    private int checkPeriod;
1068    private boolean rawScanEnabled;
1069
1070    /**
1071     * This is a timeout per table. If read of each region in the table aggregated takes longer
1072     * than what is configured here, we log an ERROR rather than just an INFO.
1073     */
1074    private HashMap<String, Long> configuredReadTableTimeouts;
1075
1076    private long configuredWriteTableTimeout;
1077
1078    public RegionMonitor(Connection connection, String[] monitorTargets, boolean useRegExp,
1079        Sink sink, ExecutorService executor, boolean writeSniffing, TableName writeTableName,
1080        boolean treatFailureAsError, HashMap<String, Long> configuredReadTableTimeouts,
1081        long configuredWriteTableTimeout,
1082        long allowedFailures) {
1083      super(connection, monitorTargets, useRegExp, sink, executor, treatFailureAsError,
1084          allowedFailures);
1085      Configuration conf = connection.getConfiguration();
1086      this.writeSniffing = writeSniffing;
1087      this.writeTableName = writeTableName;
1088      this.writeDataTTL =
1089          conf.getInt(HConstants.HBASE_CANARY_WRITE_DATA_TTL_KEY, DEFAULT_WRITE_DATA_TTL);
1090      this.regionsLowerLimit =
1091          conf.getFloat(HConstants.HBASE_CANARY_WRITE_PERSERVER_REGIONS_LOWERLIMIT_KEY, 1.0f);
1092      this.regionsUpperLimit =
1093          conf.getFloat(HConstants.HBASE_CANARY_WRITE_PERSERVER_REGIONS_UPPERLIMIT_KEY, 1.5f);
1094      this.checkPeriod =
1095          conf.getInt(HConstants.HBASE_CANARY_WRITE_TABLE_CHECK_PERIOD_KEY,
1096            DEFAULT_WRITE_TABLE_CHECK_PERIOD);
1097      this.rawScanEnabled = conf.getBoolean(HConstants.HBASE_CANARY_READ_RAW_SCAN_KEY, false);
1098      this.configuredReadTableTimeouts = new HashMap<>(configuredReadTableTimeouts);
1099      this.configuredWriteTableTimeout = configuredWriteTableTimeout;

1102    private RegionStdOutSink getSink() {
1103      if (!(sink instanceof RegionStdOutSink)) {
1104        throw new RuntimeException("Can only write to Region sink");
1105      }
1106      return ((RegionStdOutSink) sink);
1107    }

1109    @Override
1110    public void run() {
1111      if (this.initAdmin()) {
1112        try {
1113          List<Future<Void>> taskFutures = new LinkedList<>();
1114          RegionStdOutSink regionSink = this.getSink();
1115          if (this.targets != null && this.targets.length > 0) {
1116            String[] tables = generateMonitorTables(this.targets);
1117            // Check to see that each table name passed in the -readTableTimeouts argument is also
1118            // passed as a monitor target.
1119            if (!new HashSet<>(Arrays.asList(tables)).
1120                containsAll(this.configuredReadTableTimeouts.keySet())) {
1121              LOG.error("-readTableTimeouts can only specify read timeouts for monitor targets " +
1122                  "passed via command line.");
1123              this.errorCode = USAGE_EXIT_CODE;
1124              return;
1125            }
1126            this.initialized = true;
1127            for (String table : tables) {
1128              LongAdder readLatency = regionSink.initializeAndGetReadLatencyForTable(table);
1129              taskFutures.addAll(Canary.sniff(admin, regionSink, table, executor, TaskType.READ,
1130                this.rawScanEnabled, readLatency));
1131            }
1132          } else {
1133            taskFutures.addAll(sniff(TaskType.READ, regionSink));
1134          }
1135
1136          if (writeSniffing) {
1137            if (EnvironmentEdgeManager.currentTime() - lastCheckTime > checkPeriod) {
1138              try {
1139                checkWriteTableDistribution();
1140              } catch (IOException e) {
1141                LOG.error("Check canary table distribution failed!", e);
1142              }
1143              lastCheckTime = EnvironmentEdgeManager.currentTime();
1144            }
1145            // sniff canary table with write operation
1146            regionSink.initializeWriteLatency();
1147            LongAdder writeTableLatency = regionSink.getWriteLatency();
1148            taskFutures.addAll(Canary.sniff(admin, regionSink, admin.getDescriptor(writeTableName),
1149              executor, TaskType.WRITE, this.rawScanEnabled, writeTableLatency));
1150          }
1151
1152          for (Future<Void> future : taskFutures) {
1153            try {
1154              future.get();
1155            } catch (ExecutionException e) {
1156              LOG.error("Sniff region failed!", e);
1157            }
1158          }
1159          Map<String, LongAdder> actualReadTableLatency = regionSink.getReadLatencyMap();
1160          for (Map.Entry<String, Long> entry : configuredReadTableTimeouts.entrySet()) {
1161            String tableName = entry.getKey();
1162            if (actualReadTableLatency.containsKey(tableName)) {
1163              Long actual = actualReadTableLatency.get(tableName).longValue();
1164              Long configured = entry.getValue();
1165              if (actual > configured) {
1166                LOG.error("Read operation for {} took {}ms (Configured read timeout {}ms.",
1167                    tableName, actual, configured);
1168              } else {
1169                LOG.info("Read operation for {} took {}ms (Configured read timeout {}ms.",
1170                    tableName, actual, configured);
1171              }
1172            } else {
1173              LOG.error("Read operation for {} failed!", tableName);
1174            }
1175          }
1176          if (this.writeSniffing) {
1177            String writeTableStringName = this.writeTableName.getNameAsString();
1178            long actualWriteLatency = regionSink.getWriteLatency().longValue();
1179            LOG.info("Write operation for {} took {}ms. Configured write timeout {}ms.",
1180                writeTableStringName, actualWriteLatency, this.configuredWriteTableTimeout);
1181            // Check that the writeTable write operation latency does not exceed the configured timeout.
1182            if (actualWriteLatency > this.configuredWriteTableTimeout) {
1183              LOG.error("Write operation for {} exceeded the configured write timeout.",
1184                  writeTableStringName);
1185            }
1186          }
1187        } catch (Exception e) {
1188          LOG.error("Run regionMonitor failed", e);
1189          this.errorCode = ERROR_EXIT_CODE;
1190        } finally {
1191          this.done = true;
1192        }
1193      }
1194      this.done = true;
1195    }

1197    /**
1198     * @return List of tables to use in test.
1199     */
1200    private String[] generateMonitorTables(String[] monitorTargets) throws IOException {
1201      String[] returnTables = null;
1202
1203      if (this.useRegExp) {
1204        Pattern pattern = null;
1205        List<TableDescriptor> tds = null;
1206        Set<String> tmpTables = new TreeSet<>();
1207        try {
1208          LOG.debug(String.format("reading list of tables"));
1209          tds = this.admin.listTableDescriptors(pattern);
1210          if (tds == null) {
1211            tds = Collections.emptyList();
1212          }
1213          for (String monitorTarget : monitorTargets) {
1214            pattern = Pattern.compile(monitorTarget);
1215            for (TableDescriptor td : tds) {
1216              if (pattern.matcher(td.getTableName().getNameAsString()).matches()) {
1217                tmpTables.add(td.getTableName().getNameAsString());
1218              }
1219            }
1220          }
1221        } catch (IOException e) {
1222          LOG.error("Communicate with admin failed", e);
1223          throw e;
1224        }
1225
1226        if (tmpTables.size() > 0) {
1227          returnTables = tmpTables.toArray(new String[tmpTables.size()]);
1228        } else {
1229          String msg = "No HTable found, tablePattern:" + Arrays.toString(monitorTargets);
1230          LOG.error(msg);
1231          this.errorCode = INIT_ERROR_EXIT_CODE;
1232          throw new TableNotFoundException(msg);
1233        }
1234      } else {
1235        returnTables = monitorTargets;
1236      }
1237
1238      return returnTables;
1239    }

1241    /*
1242     * Canary entry point to monitor all the tables.
1243     */
1244    private List<Future<Void>> sniff(TaskType taskType, RegionStdOutSink regionSink)
1245        throws Exception {
1246      LOG.debug("Reading list of tables");
1247      List<Future<Void>> taskFutures = new LinkedList<>();
1248      for (TableDescriptor td: admin.listTableDescriptors()) {
1249        if (admin.tableExists(td.getTableName()) && admin.isTableEnabled(td.getTableName()) &&
1250            (!td.getTableName().equals(writeTableName))) {
1251          LongAdder readLatency =
1252              regionSink.initializeAndGetReadLatencyForTable(td.getTableName().getNameAsString());
1253          taskFutures.addAll(Canary.sniff(admin, sink, td, executor, taskType, this.rawScanEnabled,
1254              readLatency));
1255        }
1256      }
1257      return taskFutures;
1258    }

1260    private void checkWriteTableDistribution() throws IOException {
1261      if (!admin.tableExists(writeTableName)) {
1262        int numberOfServers =
1263            admin.getClusterMetrics(EnumSet.of(Option.LIVE_SERVERS)).getLiveServerMetrics().size();
1264        if (numberOfServers == 0) {
1265          throw new IllegalStateException("No live regionservers");
1266        }
1267        createWriteTable(numberOfServers);
1268      }
1269
1270      if (!admin.isTableEnabled(writeTableName)) {
1271        admin.enableTable(writeTableName);
1272      }
1273
1274      ClusterMetrics status =
1275          admin.getClusterMetrics(EnumSet.of(Option.LIVE_SERVERS, Option.MASTER));
1276      int numberOfServers = status.getLiveServerMetrics().size();
1277      if (status.getLiveServerMetrics().containsKey(status.getMasterName())) {
1278        numberOfServers -= 1;
1279      }
1280
1281      List<Pair<RegionInfo, ServerName>> pairs =
1282          MetaTableAccessor.getTableRegionsAndLocations(connection, writeTableName);
1283      int numberOfRegions = pairs.size();
1284      if (numberOfRegions < numberOfServers * regionsLowerLimit
1285          || numberOfRegions > numberOfServers * regionsUpperLimit) {
1286        admin.disableTable(writeTableName);
1287        admin.deleteTable(writeTableName);
1288        createWriteTable(numberOfServers);
1289      }
1290      HashSet<ServerName> serverSet = new HashSet<>();
1291      for (Pair<RegionInfo, ServerName> pair : pairs) {
1292        serverSet.add(pair.getSecond());
1293      }
1294      int numberOfCoveredServers = serverSet.size();
1295      if (numberOfCoveredServers < numberOfServers) {
1296        admin.balance();
1297      }
1298    }

1300    private void createWriteTable(int numberOfServers) throws IOException {
1301      int numberOfRegions = (int)(numberOfServers * regionsLowerLimit);
1302      LOG.info("Number of live regionservers {}, pre-splitting the canary table into {} regions " +
1303        "(current lower limit of regions per server is {} and you can change it with config {}).",
1304          numberOfServers, numberOfRegions, regionsLowerLimit,
1305          HConstants.HBASE_CANARY_WRITE_PERSERVER_REGIONS_LOWERLIMIT_KEY);
1306      HTableDescriptor desc = new HTableDescriptor(writeTableName);
1307      HColumnDescriptor family = new HColumnDescriptor(CANARY_TABLE_FAMILY_NAME);
1308      family.setMaxVersions(1);
1309      family.setTimeToLive(writeDataTTL);
1310
1311      desc.addFamily(family);
1312      byte[][] splits = new RegionSplitter.HexStringSplit().split(numberOfRegions);
1313      admin.createTable(desc, splits);
1314    }
1315  }

1334  /*
1335   * Loops over regions of this table, and outputs information about the state.
1336   */
1337  private static List<Future<Void>> sniff(final Admin admin, final Sink sink,
1338      TableDescriptor tableDesc, ExecutorService executor, TaskType taskType,
1339      boolean rawScanEnabled, LongAdder rwLatency) throws Exception {
1340    LOG.debug("Reading list of regions for table {}", tableDesc.getTableName());
1341    try (Table table = admin.getConnection().getTable(tableDesc.getTableName())) {
1342      List<RegionTask> tasks = new ArrayList<>();
1343      try (RegionLocator regionLocator =
1344               admin.getConnection().getRegionLocator(tableDesc.getTableName())) {
1345        for (HRegionLocation location: regionLocator.getAllRegionLocations()) {
1346          ServerName rs = location.getServerName();
1347          RegionInfo region = location.getRegion();
1348          tasks.add(new RegionTask(admin.getConnection(), region, rs, (RegionStdOutSink)sink,
1349              taskType, rawScanEnabled, rwLatency));
1350        }
1351        return executor.invokeAll(tasks);
1352      }
1353    } catch (TableNotFoundException e) {
1354      return Collections.EMPTY_LIST;
1355    }

1358  //  monitor for zookeeper mode
1359  private static class ZookeeperMonitor extends Monitor {
1360    private List<String> hosts;
1361    private final String znode;
1362    private final int timeout;
1363
1364    protected ZookeeperMonitor(Connection connection, String[] monitorTargets, boolean useRegExp,
1365        Sink sink, ExecutorService executor, boolean treatFailureAsError, long allowedFailures)  {
1366      super(connection, monitorTargets, useRegExp,
1367          sink, executor, treatFailureAsError, allowedFailures);
1368      Configuration configuration = connection.getConfiguration();
1369      znode =
1370          configuration.get(ZOOKEEPER_ZNODE_PARENT,
1371              DEFAULT_ZOOKEEPER_ZNODE_PARENT);
1372      timeout = configuration
1373          .getInt(HConstants.ZK_SESSION_TIMEOUT, HConstants.DEFAULT_ZK_SESSION_TIMEOUT);
1374      ConnectStringParser parser =
1375          new ConnectStringParser(ZKConfig.getZKQuorumServersString(configuration));
1376      hosts = Lists.newArrayList();
1377      for (InetSocketAddress server : parser.getServerAddresses()) {
1378        hosts.add(server.toString());
1379      }
1380      if (allowedFailures > (hosts.size() - 1) / 2) {
1381        LOG.warn("Confirm allowable number of failed ZooKeeper nodes, as quorum will " +
1382                        "already be lost. Setting of {} failures is unexpected for {} ensemble size.",
1383                allowedFailures, hosts.size());
1384      }
1385    }
1386
1387    @Override public void run() {
1388      List<ZookeeperTask> tasks = Lists.newArrayList();
1389      ZookeeperStdOutSink zkSink = null;
1390      try {
1391        zkSink = this.getSink();
1392      } catch (RuntimeException e) {
1393        LOG.error("Run ZooKeeperMonitor failed!", e);
1394        this.errorCode = ERROR_EXIT_CODE;
1395      }
1396      this.initialized = true;
1397      for (final String host : hosts) {
1398        tasks.add(new ZookeeperTask(connection, host, znode, timeout, zkSink));
1399      }
1400      try {
1401        for (Future<Void> future : this.executor.invokeAll(tasks)) {
1402          try {
1403            future.get();
1404          } catch (ExecutionException e) {
1405            LOG.error("Sniff zookeeper failed!", e);
1406            this.errorCode = ERROR_EXIT_CODE;
1407          }
1408        }
1409      } catch (InterruptedException e) {
1410        this.errorCode = ERROR_EXIT_CODE;
1411        Thread.currentThread().interrupt();
1412        LOG.error("Sniff zookeeper interrupted!", e);
1413      }
1414      this.done = true;
1415    }
1416
1417    private ZookeeperStdOutSink getSink() {
1418      if (!(sink instanceof ZookeeperStdOutSink)) {
1419        throw new RuntimeException("Can only write to zookeeper sink");
1420      }
1421      return ((ZookeeperStdOutSink) sink);
1422    }

1426  /**
1427   * A monitor for regionserver mode
1428   */
1429  private static class RegionServerMonitor extends Monitor {
1430    private boolean allRegions;
1431
1432    public RegionServerMonitor(Connection connection, String[] monitorTargets, boolean useRegExp,
1433        Sink sink, ExecutorService executor, boolean allRegions,
1434        boolean treatFailureAsError, long allowedFailures) {
1435      super(connection, monitorTargets, useRegExp, sink, executor, treatFailureAsError,
1436          allowedFailures);
1437      this.allRegions = allRegions;
1438    }
1439
1440    private RegionServerStdOutSink getSink() {
1441      if (!(sink instanceof RegionServerStdOutSink)) {
1442        throw new RuntimeException("Can only write to regionserver sink");
1443      }
1444      return ((RegionServerStdOutSink) sink);
1445    }
1446
1447    @Override
1448    public void run() {
1449      if (this.initAdmin() && this.checkNoTableNames()) {
1450        RegionServerStdOutSink regionServerSink = null;
1451        try {
1452          regionServerSink = this.getSink();
1453        } catch (RuntimeException e) {
1454          LOG.error("Run RegionServerMonitor failed!", e);
1455          this.errorCode = ERROR_EXIT_CODE;
1456        }
1457        Map<String, List<RegionInfo>> rsAndRMap = this.filterRegionServerByName();
1458        this.initialized = true;
1459        this.monitorRegionServers(rsAndRMap, regionServerSink);
1460      }
1461      this.done = true;
1462    }
1463
1464    private boolean checkNoTableNames() {
1465      List<String> foundTableNames = new ArrayList<>();
1466      TableName[] tableNames = null;
1467      LOG.debug("Reading list of tables");
1468      try {
1469        tableNames = this.admin.listTableNames();
1470      } catch (IOException e) {
1471        LOG.error("Get listTableNames failed", e);
1472        this.errorCode = INIT_ERROR_EXIT_CODE;
1473        return false;
1474      }
1475
1476      if (this.targets == null || this.targets.length == 0) return true;
1477
1478      for (String target : this.targets) {
1479        for (TableName tableName : tableNames) {
1480          if (target.equals(tableName.getNameAsString())) {
1481            foundTableNames.add(target);
1482          }
1483        }
1484      }
1485
1486      if (foundTableNames.size() > 0) {
1487        System.err.println("Cannot pass a tablename when using the -regionserver " +
1488            "option, tablenames:" + foundTableNames.toString());
1489        this.errorCode = USAGE_EXIT_CODE;
1490      }
1491      return foundTableNames.isEmpty();
1492    }
1493
1494    private void monitorRegionServers(Map<String, List<RegionInfo>> rsAndRMap, RegionServerStdOutSink regionServerSink) {
1495      List<RegionServerTask> tasks = new ArrayList<>();
1496      Map<String, AtomicLong> successMap = new HashMap<>();
1497      Random rand = new Random();
1498      for (Map.Entry<String, List<RegionInfo>> entry : rsAndRMap.entrySet()) {
1499        String serverName = entry.getKey();
1500        AtomicLong successes = new AtomicLong(0);
1501        successMap.put(serverName, successes);
1502        if (entry.getValue().isEmpty()) {
1503          LOG.error("Regionserver not serving any regions - {}", serverName);
1504        } else if (this.allRegions) {
1505          for (RegionInfo region : entry.getValue()) {
1506            tasks.add(new RegionServerTask(this.connection,
1507                serverName,
1508                region,
1509                regionServerSink,
1510                successes));
1511          }
1512        } else {
1513          // random select a region if flag not set
1514          RegionInfo region = entry.getValue().get(rand.nextInt(entry.getValue().size()));
1515          tasks.add(new RegionServerTask(this.connection,
1516              serverName,
1517              region,
1518              regionServerSink,
1519              successes));
1520        }
1521      }
1522      try {
1523        for (Future<Void> future : this.executor.invokeAll(tasks)) {
1524          try {
1525            future.get();
1526          } catch (ExecutionException e) {
1527            LOG.error("Sniff regionserver failed!", e);
1528            this.errorCode = ERROR_EXIT_CODE;
1529          }
1530        }
1531        if (this.allRegions) {
1532          for (Map.Entry<String, List<RegionInfo>> entry : rsAndRMap.entrySet()) {
1533            String serverName = entry.getKey();
1534            LOG.info("Successfully read {} regions out of {} on regionserver {}",
1535                successMap.get(serverName), entry.getValue().size(), serverName);
1536          }
1537        }
1538      } catch (InterruptedException e) {
1539        this.errorCode = ERROR_EXIT_CODE;
1540        LOG.error("Sniff regionserver interrupted!", e);
1541      }
1542    }

1544    private Map<String, List<RegionInfo>> filterRegionServerByName() {
1545      Map<String, List<RegionInfo>> regionServerAndRegionsMap = this.getAllRegionServerByName();
1546      regionServerAndRegionsMap = this.doFilterRegionServerByName(regionServerAndRegionsMap);
1547      return regionServerAndRegionsMap;
1548    }
1549
1550    private Map<String, List<RegionInfo>> getAllRegionServerByName() {
1551      Map<String, List<RegionInfo>> rsAndRMap = new HashMap<>();
1552      try {
1553        LOG.debug("Reading list of tables and locations");
1554        List<TableDescriptor> tableDescs = this.admin.listTableDescriptors();
1555        List<RegionInfo> regions = null;
1556        for (TableDescriptor tableDesc: tableDescs) {
1557          try (RegionLocator regionLocator =
1558                   this.admin.getConnection().getRegionLocator(tableDesc.getTableName())) {
1559            for (HRegionLocation location : regionLocator.getAllRegionLocations()) {
1560              ServerName rs = location.getServerName();
1561              String rsName = rs.getHostname();
1562              RegionInfo r = location.getRegion();
1563              if (rsAndRMap.containsKey(rsName)) {
1564                regions = rsAndRMap.get(rsName);
1565              } else {
1566                regions = new ArrayList<>();
1567                rsAndRMap.put(rsName, regions);
1568              }
1569              regions.add(r);
1570            }
1571          }
1572        }
1573
1574        // get any live regionservers not serving any regions
1575        for (ServerName rs: this.admin.getClusterMetrics(EnumSet.of(Option.LIVE_SERVERS))
1576          .getLiveServerMetrics().keySet()) {
1577          String rsName = rs.getHostname();
1578          if (!rsAndRMap.containsKey(rsName)) {
1579            rsAndRMap.put(rsName, Collections.<RegionInfo> emptyList());
1580          }
1581        }
1582      } catch (IOException e) {
1583        LOG.error("Get HTables info failed", e);
1584        this.errorCode = INIT_ERROR_EXIT_CODE;
1585      }
1586      return rsAndRMap;
1587    }
1588
1589    private Map<String, List<RegionInfo>> doFilterRegionServerByName(
1590        Map<String, List<RegionInfo>> fullRsAndRMap) {
1591
1592      Map<String, List<RegionInfo>> filteredRsAndRMap = null;
1593
1594      if (this.targets != null && this.targets.length > 0) {
1595        filteredRsAndRMap = new HashMap<>();
1596        Pattern pattern = null;
1597        Matcher matcher = null;
1598        boolean regExpFound = false;
1599        for (String rsName : this.targets) {
1600          if (this.useRegExp) {
1601            regExpFound = false;
1602            pattern = Pattern.compile(rsName);
1603            for (Map.Entry<String, List<RegionInfo>> entry : fullRsAndRMap.entrySet()) {
1604              matcher = pattern.matcher(entry.getKey());
1605              if (matcher.matches()) {
1606                filteredRsAndRMap.put(entry.getKey(), entry.getValue());
1607                regExpFound = true;
1608              }
1609            }
1610            if (!regExpFound) {
1611              LOG.info("No RegionServerInfo found, regionServerPattern {}", rsName);
1612            }
1613          } else {
1614            if (fullRsAndRMap.containsKey(rsName)) {
1615              filteredRsAndRMap.put(rsName, fullRsAndRMap.get(rsName));
1616            } else {
1617              LOG.info("No RegionServerInfo found, regionServerName {}", rsName);
1618            }
1619          }
1620        }
1621      } else {
1622        filteredRsAndRMap = fullRsAndRMap;
1623      }
1624      return filteredRsAndRMap;
1625    }
