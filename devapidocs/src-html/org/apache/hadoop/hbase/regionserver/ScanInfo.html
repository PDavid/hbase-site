019package org.apache.hadoop.hbase.regionserver;
020
021import org.apache.commons.lang3.builder.ToStringBuilder;
022import org.apache.hadoop.conf.Configuration;
023import org.apache.hadoop.hbase.CellComparator;
024import org.apache.hadoop.hbase.client.ColumnFamilyDescriptor;
025import org.apache.hadoop.hbase.HConstants;
026import org.apache.hadoop.hbase.KeepDeletedCells;
027import org.apache.hadoop.hbase.util.Bytes;
028import org.apache.hadoop.hbase.util.ClassSize;
029import org.apache.yetus.audience.InterfaceAudience;
030
031/**
032 * Immutable information for scans over a store.
033 */
034// Has to be public for PartitionedMobCompactor to access; ditto on tests making use of a few of
035// the accessors below. Shutdown access. TODO
036@InterfaceAudience.Private
037public class ScanInfo {
038  private byte[] family;
039  private int minVersions;
040  private int maxVersions;
041  private long ttl;
042  private KeepDeletedCells keepDeletedCells;
043  private long timeToPurgeDeletes;
044  private CellComparator comparator;
045  private long tableMaxRowSize;
046  private boolean usePread;
047  private long cellsPerTimeoutCheck;
048  private boolean parallelSeekEnabled;
049  private final long preadMaxBytes;
050  private final boolean newVersionBehavior;
051
052  public static final long FIXED_OVERHEAD = ClassSize.align(ClassSize.OBJECT
053      + (2 * ClassSize.REFERENCE) + (2 * Bytes.SIZEOF_INT)
054      + (4 * Bytes.SIZEOF_LONG) + (4 * Bytes.SIZEOF_BOOLEAN));
055
056  /**
057   * @param conf
058   * @param family {@link ColumnFamilyDescriptor} describing the column family
059   * @param ttl Store's TTL (in ms)
060   * @param timeToPurgeDeletes duration in ms after which a delete marker can be purged during a
061   *          major compaction.
062   * @param comparator The store's comparator
063   */
064  public ScanInfo(Configuration conf, ColumnFamilyDescriptor family, long ttl,
065      long timeToPurgeDeletes, CellComparator comparator) {
066    this(conf, family.getName(), family.getMinVersions(), family.getMaxVersions(), ttl,
067        family.getKeepDeletedCells(), family.getBlocksize(), timeToPurgeDeletes, comparator,
068        family.isNewVersionBehavior());
069  }
070
071  private static long getCellsPerTimeoutCheck(Configuration conf) {
072    long perHeartbeat = conf.getLong(StoreScanner.HBASE_CELLS_SCANNED_PER_HEARTBEAT_CHECK,
073      StoreScanner.DEFAULT_HBASE_CELLS_SCANNED_PER_HEARTBEAT_CHECK);
074    return perHeartbeat > 0 ? perHeartbeat
075        : StoreScanner.DEFAULT_HBASE_CELLS_SCANNED_PER_HEARTBEAT_CHECK;
076  }
077
078  /**
079   * @param conf
080   * @param family Name of this store's column family
081   * @param minVersions Store's MIN_VERSIONS setting
082   * @param maxVersions Store's VERSIONS setting
083   * @param ttl Store's TTL (in ms)
084   * @param blockSize Store's block size
085   * @param timeToPurgeDeletes duration in ms after which a delete marker can
086   *        be purged during a major compaction.
087   * @param keepDeletedCells Store's keepDeletedCells setting
088   * @param comparator The store's comparator
089   */
090  public ScanInfo(Configuration conf, byte[] family, int minVersions, int maxVersions, long ttl,
091      KeepDeletedCells keepDeletedCells, long blockSize, long timeToPurgeDeletes,
092      CellComparator comparator, boolean newVersionBehavior) {
093    this(family, minVersions, maxVersions, ttl, keepDeletedCells, timeToPurgeDeletes, comparator,
094        conf.getLong(HConstants.TABLE_MAX_ROWSIZE_KEY, HConstants.TABLE_MAX_ROWSIZE_DEFAULT),
095        conf.getBoolean("hbase.storescanner.use.pread", false), getCellsPerTimeoutCheck(conf),
096        conf.getBoolean(StoreScanner.STORESCANNER_PARALLEL_SEEK_ENABLE, false),
097        conf.getLong(StoreScanner.STORESCANNER_PREAD_MAX_BYTES, 4 * blockSize), newVersionBehavior);
098  }
099
100  private ScanInfo(byte[] family, int minVersions, int maxVersions, long ttl,
101      KeepDeletedCells keepDeletedCells, long timeToPurgeDeletes, CellComparator comparator,
102      long tableMaxRowSize, boolean usePread, long cellsPerTimeoutCheck,
103      boolean parallelSeekEnabled, long preadMaxBytes, boolean newVersionBehavior) {
104    this.family = family;
105    this.minVersions = minVersions;
106    this.maxVersions = maxVersions;
107    this.ttl = ttl;
108    this.keepDeletedCells = keepDeletedCells;
109    this.timeToPurgeDeletes = timeToPurgeDeletes;
110    this.comparator = comparator;
111    this.tableMaxRowSize = tableMaxRowSize;
112    this.usePread = usePread;
113    this.cellsPerTimeoutCheck = cellsPerTimeoutCheck;
114    this.parallelSeekEnabled = parallelSeekEnabled;
115    this.preadMaxBytes = preadMaxBytes;
116    this.newVersionBehavior = newVersionBehavior;
117  }
118
119  long getTableMaxRowSize() {
120    return this.tableMaxRowSize;
121  }
122
123  boolean isUsePread() {
124    return this.usePread;
125  }
126
127  long getCellsPerTimeoutCheck() {
128    return this.cellsPerTimeoutCheck;
129  }
130
131  boolean isParallelSeekEnabled() {
132    return this.parallelSeekEnabled;
133  }
134
135  public byte[] getFamily() {
136    return family;
137  }
138
139  public int getMinVersions() {
140    return minVersions;
141  }
142
143  public int getMaxVersions() {
144    return maxVersions;
145  }
146
147  public long getTtl() {
148    return ttl;
149  }
150
151  public KeepDeletedCells getKeepDeletedCells() {
152    return keepDeletedCells;
153  }
154
155  public long getTimeToPurgeDeletes() {
156    return timeToPurgeDeletes;
157  }
158
159  public CellComparator getComparator() {
160    return comparator;
161  }
162
163  long getPreadMaxBytes() {
164    return preadMaxBytes;
165  }
166
167  public boolean isNewVersionBehavior() {
168    return newVersionBehavior;
169  }
170
171  /**
172   * Used by CP users for customizing max versions, ttl and keepDeletedCells.
173   */
174  ScanInfo customize(int maxVersions, long ttl, KeepDeletedCells keepDeletedCells) {
175    return customize(maxVersions, ttl, keepDeletedCells, minVersions, timeToPurgeDeletes);
176  }
177
178  /**
179   * Used by CP users for customizing max versions, ttl, keepDeletedCells, min versions,
180   * and time to purge deletes.
181   */
182  ScanInfo customize(int maxVersions, long ttl, KeepDeletedCells keepDeletedCells,
183    int minVersions, long timeToPurgeDeletes) {
184    return new ScanInfo(family, minVersions, maxVersions, ttl, keepDeletedCells, timeToPurgeDeletes,
185      comparator, tableMaxRowSize, usePread, cellsPerTimeoutCheck, parallelSeekEnabled,
186      preadMaxBytes, newVersionBehavior);
187  }
188
189  @Override
190  public String toString() {
191    return new ToStringBuilder(this)
192      .append("family", Bytes.toStringBinary(family))
193      .append("minVersions", minVersions)
194      .append("maxVersions", maxVersions)
195      .append("ttl", ttl)
196      .append("keepDeletedCells", keepDeletedCells)
197      .append("timeToPurgeDeletes", timeToPurgeDeletes)
198      .append("tableMaxRowSize", tableMaxRowSize)
199      .append("usePread", usePread)
200      .append("cellsPerTimeoutCheck", cellsPerTimeoutCheck)
201      .append("parallelSeekEnabled", parallelSeekEnabled)
202      .append("preadMaxBytes", preadMaxBytes)
203      .append("newVersionBehavior", newVersionBehavior).toString();
204  }
205}
