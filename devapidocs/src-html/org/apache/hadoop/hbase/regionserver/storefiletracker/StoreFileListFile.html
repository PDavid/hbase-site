021import java.io.FileNotFoundException;

048/**
049 * To fully avoid listing, here we use two files for tracking. When loading, we will try to read
050 * both the two files, if only one exists, we will trust this one, if both exist, we will compare
051 * the timestamp to see which one is newer and trust that one. And we will record in memory that
052 * which one is trusted by us, and when we need to update the store file list, we will write to the
053 * other file.
054 * <p/>
055 * So in this way, we could avoid listing when we want to load the store file list file.
056 * <p/>
057 * To prevent loading partial file, we use the first 4 bytes as file length, and also append a 4
058 * bytes crc32 checksum at the end. This is because the protobuf message parser sometimes can return
059 * without error on partial bytes if you stop at some special points, but the return message will
060 * have incorrect field value. We should try our best to prevent this happens because loading an
061 * incorrect store file list file usually leads to data loss.
062 */
063@InterfaceAudience.Private
064class StoreFileListFile {
065
066  private static final Logger LOG = LoggerFactory.getLogger(StoreFileListFile.class);
067
068  static final String TRACK_FILE_DIR = ".filelist";
069
070  private static final String TRACK_FILE_PREFIX = "f1";
071
072  private static final String TRACK_FILE_ROTATE_PREFIX = "f2";
073
074  private static final char TRACK_FILE_SEPARATOR = '.';
075
076  private static final Pattern TRACK_FILE_PATTERN = Pattern.compile("^f(1|2)\\.\\d+$");
077
078  // 16 MB, which is big enough for a tracker file
079  private static final int MAX_FILE_SIZE = 16 * 1024 * 1024;
080
081  private final StoreContext ctx;
082
083  private final Path trackFileDir;
084
085  private final Path[] trackFiles = new Path[2];
086
087  // this is used to make sure that we do not go backwards
088  private long prevTimestamp = -1;
089
090  private int nextTrackFile = -1;
091
092  StoreFileListFile(StoreContext ctx) {
093    this.ctx = ctx;
094    trackFileDir = new Path(ctx.getFamilyStoreDirectoryPath(), TRACK_FILE_DIR);
095  }
096
097  private StoreFileList load(Path path) throws IOException {
098    FileSystem fs = ctx.getRegionFileSystem().getFileSystem();
099    byte[] data;
100    int expectedChecksum;
101    try (FSDataInputStream in = fs.open(path)) {
102      int length = in.readInt();
103      if (length <= 0 || length > MAX_FILE_SIZE) {
104        throw new IOException("Invalid file length " + length
105          + ", either less than 0 or greater then max allowed size " + MAX_FILE_SIZE);
106      }
107      data = new byte[length];
108      in.readFully(data);
109      expectedChecksum = in.readInt();
110    }
111    CRC32 crc32 = new CRC32();
112    crc32.update(data);
113    int calculatedChecksum = (int) crc32.getValue();
114    if (expectedChecksum != calculatedChecksum) {
115      throw new IOException(
116        "Checksum mismatch, expected " + expectedChecksum + ", actual " + calculatedChecksum);
117    }
118    return StoreFileList.parseFrom(data);
119  }
120
121  private int select(StoreFileList[] lists) {
122    if (lists[0] == null) {
123      return 1;
124    }
125    if (lists[1] == null) {
126      return 0;
127    }
128    return lists[0].getTimestamp() >= lists[1].getTimestamp() ? 0 : 1;
129  }
130
131  // file sequence id to path
132  private NavigableMap<Long, List<Path>> listFiles() throws IOException {
133    FileSystem fs = ctx.getRegionFileSystem().getFileSystem();
134    FileStatus[] statuses;
135    try {
136      statuses = fs.listStatus(trackFileDir);
137    } catch (FileNotFoundException e) {
138      LOG.debug("Track file directory {} does not exist", trackFileDir, e);
139      return Collections.emptyNavigableMap();
140    }
141    if (statuses == null || statuses.length == 0) {
142      return Collections.emptyNavigableMap();
143    }
144    TreeMap<Long, List<Path>> map = new TreeMap<>((l1, l2) -> l2.compareTo(l1));
145    for (FileStatus status : statuses) {
146      Path file = status.getPath();
147      if (!status.isFile()) {
148        LOG.warn("Found invalid track file {}, which is not a file", file);
149        continue;
150      }
151      if (!TRACK_FILE_PATTERN.matcher(file.getName()).matches()) {
152        LOG.warn("Found invalid track file {}, skip", file);
153        continue;
154      }
155      List<String> parts = Splitter.on(TRACK_FILE_SEPARATOR).splitToList(file.getName());
156      map.computeIfAbsent(Long.parseLong(parts.get(1)), k -> new ArrayList<>()).add(file);
157    }
158    return map;
159  }
160
161  private void initializeTrackFiles(long seqId) {
162    trackFiles[0] = new Path(trackFileDir, TRACK_FILE_PREFIX + TRACK_FILE_SEPARATOR + seqId);
163    trackFiles[1] = new Path(trackFileDir, TRACK_FILE_ROTATE_PREFIX + TRACK_FILE_SEPARATOR + seqId);
164    LOG.info("Initialized track files: {}, {}", trackFiles[0], trackFiles[1]);
165  }
166
167  private void cleanUpTrackFiles(long loadedSeqId,
168    NavigableMap<Long, List<Path>> seqId2TrackFiles) {
169    LOG.info("Cleanup track file with sequence id < {}", loadedSeqId);
170    FileSystem fs = ctx.getRegionFileSystem().getFileSystem();
171    NavigableMap<Long, List<Path>> toDelete =
172      loadedSeqId >= 0 ? seqId2TrackFiles.tailMap(loadedSeqId, false) : seqId2TrackFiles;
173    toDelete.values().stream().flatMap(l -> l.stream()).forEach(file -> {
174      ForkJoinPool.commonPool().execute(() -> {
175        LOG.info("Deleting track file {}", file);
176        try {
177          fs.delete(file, false);
178        } catch (IOException e) {
179          LOG.warn("failed to delete unused track file {}", file, e);
180        }
181      });
182    });
183  }
184
185  StoreFileList load(boolean readOnly) throws IOException {
186    NavigableMap<Long, List<Path>> seqId2TrackFiles = listFiles();
187    long seqId = -1L;
188    StoreFileList[] lists = new StoreFileList[2];
189    for (Map.Entry<Long, List<Path>> entry : seqId2TrackFiles.entrySet()) {
190      List<Path> files = entry.getValue();
191      // should not have more than 2 files, if not, it means that the track files are broken, just
192      // throw exception out and fail the region open.
193      if (files.size() > 2) {
194        throw new DoNotRetryIOException("Should only have at most 2 track files for sequence id "
195          + entry.getKey() + ", but got " + files.size() + " files: " + files);
196      }
197      boolean loaded = false;
198      for (int i = 0; i < files.size(); i++) {
199        try {
200          lists[i] = load(files.get(i));
201          loaded = true;
202        } catch (EOFException e) {
203          // this is normal case, so just log at debug
204          LOG.debug("EOF loading track file {}, ignoring the exception", trackFiles[i], e);
205        }
206      }
207      if (loaded) {
208        seqId = entry.getKey();
209        break;
210      }
211    }
212    if (readOnly) {
213      return lists[select(lists)];
214    }
215
216    cleanUpTrackFiles(seqId, seqId2TrackFiles);
217
218    if (seqId < 0) {
219      initializeTrackFiles(System.currentTimeMillis());
220      nextTrackFile = 0;
221      return null;
222    }
223
224    initializeTrackFiles(Math.max(System.currentTimeMillis(), seqId + 1));
225    int winnerIndex = select(lists);
226    nextTrackFile = 1 - winnerIndex;
227    prevTimestamp = lists[winnerIndex].getTimestamp();
228    return lists[winnerIndex];
229  }
230
231  /**
232   * We will set the timestamp in this method so just pass the builder in
233   */
234  void update(StoreFileList.Builder builder) throws IOException {
235    if (nextTrackFile < 0) {
236      // we need to call load first to load the prevTimestamp and also the next file
237      // we are already in the update method, which is not read only, so pass false
238      load(false);
239    }
240    long timestamp = Math.max(prevTimestamp + 1, EnvironmentEdgeManager.currentTime());
241    byte[] actualData = builder.setTimestamp(timestamp).build().toByteArray();
242    CRC32 crc32 = new CRC32();
243    crc32.update(actualData);
244    int checksum = (int) crc32.getValue();
245    // 4 bytes length at the beginning, plus 4 bytes checksum
246    FileSystem fs = ctx.getRegionFileSystem().getFileSystem();
247    try (FSDataOutputStream out = fs.create(trackFiles[nextTrackFile], true)) {
248      out.writeInt(actualData.length);
249      out.write(actualData);
250      out.writeInt(checksum);
251    }
252    // record timestamp
253    prevTimestamp = timestamp;
254    // rotate the file
255    nextTrackFile = 1 - nextTrackFile;
256    try {
257      fs.delete(trackFiles[nextTrackFile], false);
258    } catch (IOException e) {
259      // we will create new file with overwrite = true, so not a big deal here, only for speed up
260      // loading as we do not need to read this file when loading
261      LOG.debug("Failed to delete old track file {}, ignoring the exception", e);
262    }
263  }
264}
