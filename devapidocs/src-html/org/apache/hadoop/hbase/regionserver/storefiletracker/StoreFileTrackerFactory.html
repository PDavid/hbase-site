040/**
041 * Factory method for creating store file tracker.
042 * <p/>
043 * The current implementations are:
044 * <ul>
045 * <li><em>default</em>: DefaultStoreFileTracker, see {@link DefaultStoreFileTracker}.</li>
046 * <li><em>file</em>:FileBasedStoreFileTracker, see {@link FileBasedStoreFileTracker}.</li>
047 * <li><em>migration</em>:MigrationStoreFileTracker, see {@link MigrationStoreFileTracker}.</li>
048 * </ul>
049 * @see DefaultStoreFileTracker
050 * @see FileBasedStoreFileTracker
051 * @see MigrationStoreFileTracker
052 */
053@InterfaceAudience.Private
054public final class StoreFileTrackerFactory {
055
056  private static final Logger LOG = LoggerFactory.getLogger(StoreFileTrackerFactory.class);
057
058  public static final String TRACKER_IMPL = "hbase.store.file-tracker.impl";
059
060  /**
061   * Maps between configuration names for trackers and implementation classes.
062   */
063  public enum Trackers {
064    DEFAULT(DefaultStoreFileTracker.class), FILE(FileBasedStoreFileTracker.class),
065    MIGRATION(MigrationStoreFileTracker.class);
066
067    final Class<? extends StoreFileTracker> clazz;
068
069    Trackers(Class<? extends StoreFileTracker> clazz) {
070      this.clazz = clazz;
071    }
072  }
073
074  private static final Map<Class<? extends StoreFileTracker>, Trackers> CLASS_TO_ENUM = reverse();
075
076  private static Map<Class<? extends StoreFileTracker>, Trackers> reverse() {
077    Map<Class<? extends StoreFileTracker>, Trackers> map = new HashMap<>();
078    for (Trackers tracker : Trackers.values()) {
079      map.put(tracker.clazz, tracker);
080    }
081    return Collections.unmodifiableMap(map);
082  }
083
084  private StoreFileTrackerFactory() {
085  }
086
087  public static String getStoreFileTrackerName(Configuration conf) {
088    return conf.get(TRACKER_IMPL, Trackers.DEFAULT.name());
089  }
090
091  static String getStoreFileTrackerName(Class<? extends StoreFileTracker> clazz) {
092    Trackers name = CLASS_TO_ENUM.get(clazz);
093    return name != null ? name.name() : clazz.getName();
094  }
095
096  public static Class<? extends StoreFileTracker> getTrackerClass(Configuration conf) {
097    try {
098      Trackers tracker = Trackers.valueOf(getStoreFileTrackerName(conf).toUpperCase());
099      return tracker.clazz;
100    } catch (IllegalArgumentException e) {
101      // Fall back to them specifying a class name
102      return conf.getClass(TRACKER_IMPL, Trackers.DEFAULT.clazz, StoreFileTracker.class);
103    }
104  }
105
106  public static StoreFileTracker create(Configuration conf, boolean isPrimaryReplica,
107    StoreContext ctx) {
108    Class<? extends StoreFileTracker> tracker = getTrackerClass(conf);
109    LOG.info("instantiating StoreFileTracker impl {}", tracker.getName());
110    return ReflectionUtils.newInstance(tracker, conf, isPrimaryReplica, ctx);
111  }
112
113  /**
114   * Used at master side when splitting/merging regions, as we do not have a Store, thus no
115   * StoreContext at master side.
116   */
117  public static StoreFileTracker create(Configuration conf, TableDescriptor td,
118    ColumnFamilyDescriptor cfd, HRegionFileSystem regionFs) {
119    StoreContext ctx =
120      StoreContext.getBuilder().withColumnFamilyDescriptor(cfd).withRegionFileSystem(regionFs)
121        .withFamilyStoreDirectoryPath(regionFs.getStoreDir(cfd.getNameAsString())).build();
122    return StoreFileTrackerFactory.create(mergeConfigurations(conf, td, cfd), true, ctx);
123  }
124
125  private static Configuration mergeConfigurations(Configuration global, TableDescriptor table,
126    ColumnFamilyDescriptor family) {
127    return StoreUtils.createStoreConfiguration(global, table, family);
128  }
129
130  static Class<? extends StoreFileTrackerBase>
131    getStoreFileTrackerClassForMigration(Configuration conf, String configName) {
132    String trackerName =
133      Preconditions.checkNotNull(conf.get(configName), "config %s is not set", configName);
134    try {
135      return Trackers.valueOf(trackerName.toUpperCase()).clazz
136        .asSubclass(StoreFileTrackerBase.class);
137    } catch (IllegalArgumentException e) {
138      // Fall back to them specifying a class name
139      try {
140        return Class.forName(trackerName).asSubclass(StoreFileTrackerBase.class);
141      } catch (ClassNotFoundException cnfe) {
142        throw new RuntimeException(cnfe);
143      }
144    }
145  }
146
147  /**
148   * Create store file tracker to be used as source or destination for
149   * {@link MigrationStoreFileTracker}.
150   */
151  static StoreFileTrackerBase createForMigration(Configuration conf, String configName,
152    boolean isPrimaryReplica, StoreContext ctx) {
153    Class<? extends StoreFileTrackerBase> tracker =
154      getStoreFileTrackerClassForMigration(conf, configName);
155    // prevent nest of MigrationStoreFileTracker, it will cause infinite recursion.
156    if (MigrationStoreFileTracker.class.isAssignableFrom(tracker)) {
157      throw new IllegalArgumentException("Should not specify " + configName + " as "
158        + Trackers.MIGRATION + " because it can not be nested");
159    }
160    LOG.info("instantiating StoreFileTracker impl {} as {}", tracker.getName(), configName);
161    return ReflectionUtils.newInstance(tracker, conf, isPrimaryReplica, ctx);
162  }

164  public static TableDescriptor updateWithTrackerConfigs(Configuration conf,
165      TableDescriptor descriptor) {
166    //CreateTableProcedure needs to instantiate the configured SFT impl, in order to update table
167    //descriptors with the SFT impl specific configs. By the time this happens, the table has no
168    //regions nor stores yet, so it can't create a proper StoreContext.
169    if (StringUtils.isEmpty(descriptor.getValue(TRACKER_IMPL))) {
170      StoreFileTracker tracker =
171        StoreFileTrackerFactory.create(conf, true, null);
172      TableDescriptorBuilder builder = TableDescriptorBuilder.newBuilder(descriptor);
173      return tracker.updateWithTrackerConfigs(builder).build();
174    }
175    return descriptor;
176  }

178  // should not use MigrationStoreFileTracker for new family
179  private static void checkForNewFamily(Configuration conf, TableDescriptor table,
180    ColumnFamilyDescriptor family) throws IOException {
181    Configuration mergedConf = mergeConfigurations(conf, table, family);
182    Class<? extends StoreFileTracker> tracker = getTrackerClass(mergedConf);
183    if (MigrationStoreFileTracker.class.isAssignableFrom(tracker)) {
184      throw new DoNotRetryIOException(
185        "Should not use " + Trackers.MIGRATION + " as store file tracker for new family "
186          + family.getNameAsString() + " of table " + table.getTableName());
187    }
188  }

190  /**
191   * Pre check when creating a new table.
192   * <p/>
193   * For now, only make sure that we do not use {@link Trackers#MIGRATION} for newly created tables.
194   * @throws IOException when there are check errors, the upper layer should fail the
195   *           {@code CreateTableProcedure}.
196   */
197  public static void checkForCreateTable(Configuration conf, TableDescriptor table)
198    throws IOException {
199    for (ColumnFamilyDescriptor family : table.getColumnFamilies()) {
200      checkForNewFamily(conf, table, family);
201    }
202  }

205  /**
206   * Pre check when modifying a table.
207   * <p/>
208   * The basic idea is when you want to change the store file tracker implementation, you should use
209   * {@link Trackers#MIGRATION} first and then change to the destination store file tracker
210   * implementation.
211   * <p/>
212   * There are several rules:
213   * <ul>
214   * <li>For newly added family, you should not use {@link Trackers#MIGRATION}.</li>
215   * <li>For modifying a family:
216   * <ul>
217   * <li>If old tracker is {@link Trackers#MIGRATION}, then:
218   * <ul>
219   * <li>The new tracker is also {@link Trackers#MIGRATION}, then they must have the same src and
220   * dst tracker.</li>
221   * <li>The new tracker is not {@link Trackers#MIGRATION}, then the new tracker must be the dst
222   * tracker of the old tracker.</li>
223   * </ul>
224   * </li>
225   * <li>If the old tracker is not {@link Trackers#MIGRATION}, then:
226   * <ul>
227   * <li>If the new tracker is {@link Trackers#MIGRATION}, then the old tracker must be the src
228   * tracker of the new tracker.</li>
229   * <li>If the new tracker is not {@link Trackers#MIGRATION}, then the new tracker must be the same
230   * with old tracker.</li>
231   * </ul>
232   * </li>
233   * </ul>
234   * </li>
235   * </ul>
236   * @throws IOException when there are check errors, the upper layer should fail the
237   *           {@code ModifyTableProcedure}.
238   */
239  public static void checkForModifyTable(Configuration conf, TableDescriptor oldTable,
240    TableDescriptor newTable) throws IOException {
241    for (ColumnFamilyDescriptor newFamily : newTable.getColumnFamilies()) {
242      ColumnFamilyDescriptor oldFamily = oldTable.getColumnFamily(newFamily.getName());
243      if (oldFamily == null) {
244        checkForNewFamily(conf, newTable, newFamily);
245        continue;
246      }
247      Configuration oldConf = mergeConfigurations(conf, oldTable, oldFamily);
248      Configuration newConf = mergeConfigurations(conf, newTable, newFamily);
249
250      Class<? extends StoreFileTracker> oldTracker = getTrackerClass(oldConf);
251      Class<? extends StoreFileTracker> newTracker = getTrackerClass(newConf);
252
253      if (MigrationStoreFileTracker.class.isAssignableFrom(oldTracker)) {
254        Class<? extends StoreFileTracker> oldSrcTracker =
255          MigrationStoreFileTracker.getSrcTrackerClass(oldConf);
256        Class<? extends StoreFileTracker> oldDstTracker =
257          MigrationStoreFileTracker.getDstTrackerClass(oldConf);
258        if (oldTracker.equals(newTracker)) {
259          // confirm that we have the same src tracker and dst tracker
260          Class<? extends StoreFileTracker> newSrcTracker =
261            MigrationStoreFileTracker.getSrcTrackerClass(newConf);
262          if (!oldSrcTracker.equals(newSrcTracker)) {
263            throw new DoNotRetryIOException(
264              "The src tracker has been changed from " + getStoreFileTrackerName(oldSrcTracker)
265                + " to " + getStoreFileTrackerName(newSrcTracker) + " for family "
266                + newFamily.getNameAsString() + " of table " + newTable.getTableName());
267          }
268          Class<? extends StoreFileTracker> newDstTracker =
269            MigrationStoreFileTracker.getDstTrackerClass(newConf);
270          if (!oldDstTracker.equals(newDstTracker)) {
271            throw new DoNotRetryIOException(
272              "The dst tracker has been changed from " + getStoreFileTrackerName(oldDstTracker)
273                + " to " + getStoreFileTrackerName(newDstTracker) + " for family "
274                + newFamily.getNameAsString() + " of table " + newTable.getTableName());
275          }
276        } else {
277          // we can only change to the dst tracker
278          if (!newTracker.equals(oldDstTracker)) {
279            throw new DoNotRetryIOException(
280              "Should migrate tracker to " + getStoreFileTrackerName(oldDstTracker) + " but got "
281                + getStoreFileTrackerName(newTracker) + " for family " + newFamily.getNameAsString()
282                + " of table " + newTable.getTableName());
283          }
284        }
285      } else {
286        if (!oldTracker.equals(newTracker)) {
287          // can only change to MigrationStoreFileTracker and the src tracker should be the old
288          // tracker
289          if (!MigrationStoreFileTracker.class.isAssignableFrom(newTracker)) {
290            throw new DoNotRetryIOException("Should change to " + Trackers.MIGRATION
291              + " first when migrating from " + getStoreFileTrackerName(oldTracker) + " for family "
292              + newFamily.getNameAsString() + " of table " + newTable.getTableName());
293          }
294          Class<? extends StoreFileTracker> newSrcTracker =
295            MigrationStoreFileTracker.getSrcTrackerClass(newConf);
296          if (!oldTracker.equals(newSrcTracker)) {
297            throw new DoNotRetryIOException(
298              "Should use src tracker " + getStoreFileTrackerName(oldTracker) + " first but got "
299                + getStoreFileTrackerName(newSrcTracker) + " when migrating from "
300                + getStoreFileTrackerName(oldTracker) + " for family " + newFamily.getNameAsString()
301                + " of table " + newTable.getTableName());
302          }
303          Class<? extends StoreFileTracker> newDstTracker =
304            MigrationStoreFileTracker.getDstTrackerClass(newConf);
305          // the src and dst tracker should not be the same
306          if (newSrcTracker.equals(newDstTracker)) {
307            throw new DoNotRetryIOException("The src tracker and dst tracker are both "
308              + getStoreFileTrackerName(newSrcTracker) + " for family "
309              + newFamily.getNameAsString() + " of table " + newTable.getTableName());
310          }
311        }
312      }
313    }
314  }

316  /**
317   * Makes sure restoring a snapshot does not break the current SFT setup
318   * follows StoreUtils.createStoreConfiguration
319   * @param currentTableDesc Existing Table's TableDescriptor
320   * @param snapshotTableDesc Snapshot's TableDescriptor
321   * @param baseConf Current global configuration
322   * @throws RestoreSnapshotException if restore would break the current SFT setup
323   */
324  public static void validatePreRestoreSnapshot(TableDescriptor currentTableDesc,
325    TableDescriptor snapshotTableDesc, Configuration baseConf) throws RestoreSnapshotException {
326
327    for (ColumnFamilyDescriptor cfDesc : currentTableDesc.getColumnFamilies()) {
328      ColumnFamilyDescriptor snapCFDesc = snapshotTableDesc.getColumnFamily(cfDesc.getName());
329      // if there is no counterpart in the snapshot it will be just deleted so the config does
330      // not matter
331      if (snapCFDesc != null) {
332        Configuration currentCompositeConf =
333          StoreUtils.createStoreConfiguration(baseConf, currentTableDesc, cfDesc);
334        Configuration snapCompositeConf =
335          StoreUtils.createStoreConfiguration(baseConf, snapshotTableDesc, snapCFDesc);
336        Class<? extends StoreFileTracker> currentSFT =
337          StoreFileTrackerFactory.getTrackerClass(currentCompositeConf);
338        Class<? extends StoreFileTracker> snapSFT =
339          StoreFileTrackerFactory.getTrackerClass(snapCompositeConf);
340
341        //restoration is not possible if there is an SFT mismatch
342        if (currentSFT != snapSFT) {
343          throw new RestoreSnapshotException(
344            "Restoring Snapshot is not possible because " + " the config for column family "
345              + cfDesc.getNameAsString() + " has incompatible configuration. Current SFT: "
346              + currentSFT + " SFT from snapshot: " + snapSFT);
347        }
348      }
349    }
350  }
