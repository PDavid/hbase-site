054/**
055 * This worker is spawned in every regionserver, including master. The Worker waits for log
056 * splitting tasks to be put up by the {@link org.apache.hadoop.hbase.master.SplitLogManager}
057 * running in the master and races with other workers in other serves to acquire those tasks.
058 * The coordination is done via coordination engine.
059 * <p>
060 * If a worker has successfully moved the task from state UNASSIGNED to OWNED then it owns the task.
061 * It keeps heart beating the manager by periodically moving the task from UNASSIGNED to OWNED
062 * state. On success it moves the task to TASK_DONE. On unrecoverable error it moves task state to
063 * ERR. If it cannot continue but wants the master to retry the task then it moves the task state to
064 * RESIGNED.
065 * <p>
066 * The manager can take a task away from a worker by moving the task from OWNED to UNASSIGNED. In
067 * the absence of a global lock there is a unavoidable race here - a worker might have just finished
068 * its task when it is stripped of its ownership. Here we rely on the idempotency of the log
069 * splitting task for correctness
070 */
071@InterfaceAudience.Private
072public class SplitLogWorker implements Runnable {
073
074  private static final Logger LOG = LoggerFactory.getLogger(SplitLogWorker.class);
075
076  Thread worker;
077  // thread pool which executes recovery work
078  private SplitLogWorkerCoordination coordination;
079  private RegionServerServices server;
080
081  public SplitLogWorker(Server hserver, Configuration conf, RegionServerServices server,
082      TaskExecutor splitTaskExecutor) {
083    this.server = server;
084    this.coordination = hserver.getCoordinatedStateManager().getSplitLogWorkerCoordination();
085    coordination.init(server, conf, splitTaskExecutor, this);
086  }
087
088  public SplitLogWorker(Configuration conf, RegionServerServices server,
089      LastSequenceId sequenceIdChecker, WALFactory factory) {
090    this(server, conf, server, (f, p) -> splitLog(f, p, conf, server, sequenceIdChecker, factory));
091  }

093  // returns whether we need to continue the split work
094  private static boolean processSyncReplicationWAL(String name, Configuration conf,
095      RegionServerServices server, FileSystem fs, Path walDir) throws IOException {
096    Path walFile = new Path(walDir, name);
097    String filename = walFile.getName();
098    Optional<String> optSyncPeerId =
099      SyncReplicationWALProvider.getSyncReplicationPeerIdFromWALName(filename);
100    if (!optSyncPeerId.isPresent()) {
101      return true;
102    }
103    String peerId = optSyncPeerId.get();
104    ReplicationPeerImpl peer =
105      server.getReplicationSourceService().getReplicationPeers().getPeer(peerId);
106    if (peer == null || !peer.getPeerConfig().isSyncReplication()) {
107      return true;
108    }
109    Pair<SyncReplicationState, SyncReplicationState> stateAndNewState =
110      peer.getSyncReplicationStateAndNewState();
111    if (stateAndNewState.getFirst().equals(SyncReplicationState.ACTIVE) &&
112      stateAndNewState.getSecond().equals(SyncReplicationState.NONE)) {
113      // copy the file to remote and overwrite the previous one
114      String remoteWALDir = peer.getPeerConfig().getRemoteWALDir();
115      Path remoteWALDirForPeer = ReplicationUtils.getPeerRemoteWALDir(remoteWALDir, peerId);
116      Path tmpRemoteWAL = new Path(remoteWALDirForPeer, filename + ".tmp");
117      FileSystem remoteFs = ReplicationUtils.getRemoteWALFileSystem(conf, remoteWALDir);
118      try (FSDataInputStream in = fs.open(walFile); @SuppressWarnings("deprecation")
119      FSDataOutputStream out = remoteFs.createNonRecursive(tmpRemoteWAL, true,
120        FSUtils.getDefaultBufferSize(remoteFs), remoteFs.getDefaultReplication(tmpRemoteWAL),
121        remoteFs.getDefaultBlockSize(tmpRemoteWAL), null)) {
122        IOUtils.copy(in, out);
123      }
124      Path toCommitRemoteWAL =
125        new Path(remoteWALDirForPeer, filename + ReplicationUtils.RENAME_WAL_SUFFIX);
126      // Some FileSystem implementations may not support atomic rename so we need to do it in two
127      // phases
128      FSUtils.renameFile(remoteFs, tmpRemoteWAL, toCommitRemoteWAL);
129      FSUtils.renameFile(remoteFs, toCommitRemoteWAL, new Path(remoteWALDirForPeer, filename));
130    } else if ((stateAndNewState.getFirst().equals(SyncReplicationState.ACTIVE) &&
131      stateAndNewState.getSecond().equals(SyncReplicationState.STANDBY)) ||
132      stateAndNewState.getFirst().equals(SyncReplicationState.STANDBY)) {
133      // check whether we still need to process this file
134      // actually we only write wal file which name is ended with .syncrep in A state, and after
135      // transiting to a state other than A, we will reopen all the regions so the data in the wal
136      // will be flushed so the wal file will be archived soon. But it is still possible that there
137      // is a server crash when we are transiting from A to S, to simplify the logic of the transit
138      // procedure, here we will also check the remote snapshot directory in state S, so that we do
139      // not need wait until all the wal files with .syncrep suffix to be archived before finishing
140      // the procedure.
141      String remoteWALDir = peer.getPeerConfig().getRemoteWALDir();
142      Path remoteSnapshotDirForPeer = ReplicationUtils.getPeerSnapshotWALDir(remoteWALDir, peerId);
143      FileSystem remoteFs = ReplicationUtils.getRemoteWALFileSystem(conf, remoteWALDir);
144      if (remoteFs.exists(new Path(remoteSnapshotDirForPeer, filename))) {
145        // the file has been replayed when the remote cluster was transited from S to DA, the
146        // content will be replicated back to us so give up split it.
147        LOG.warn("Giveup splitting {} since it has been replayed in the remote cluster and " +
148          "the content will be replicated back", filename);
149        return false;
150      }
151    }
152    return true;
153  }
154
155  static Status splitLog(String name, CancelableProgressable p, Configuration conf,
156      RegionServerServices server, LastSequenceId sequenceIdChecker, WALFactory factory) {
157    Path walDir;
158    FileSystem fs;
159    try {
160      walDir = FSUtils.getWALRootDir(conf);
161      fs = walDir.getFileSystem(conf);
162    } catch (IOException e) {
163      LOG.warn("could not find root dir or fs", e);
164      return Status.RESIGNED;
165    }
166    try {
167      if (!processSyncReplicationWAL(name, conf, server, fs, walDir)) {
168        return Status.DONE;
169      }
170    } catch (IOException e) {
171      LOG.warn("failed to process sync replication wal {}", name, e);
172      return Status.RESIGNED;
173    }
174    // TODO have to correctly figure out when log splitting has been
175    // interrupted or has encountered a transient error and when it has
176    // encountered a bad non-retry-able persistent error.
177    try {
178      SplitLogWorkerCoordination splitLogWorkerCoordination =
179          server.getCoordinatedStateManager() == null ? null
180              : server.getCoordinatedStateManager().getSplitLogWorkerCoordination();
181      if (!WALSplitter.splitLogFile(walDir, fs.getFileStatus(new Path(walDir, name)), fs, conf, p,
182        sequenceIdChecker, splitLogWorkerCoordination, factory, server)) {
183        return Status.PREEMPTED;
184      }
185    } catch (InterruptedIOException iioe) {
186      LOG.warn("log splitting of " + name + " interrupted, resigning", iioe);
187      return Status.RESIGNED;
188    } catch (IOException e) {
189      if (e instanceof FileNotFoundException) {
190        // A wal file may not exist anymore. Nothing can be recovered so move on
191        LOG.warn("WAL {} does not exist anymore", name, e);
192        return Status.DONE;
193      }
194      Throwable cause = e.getCause();
195      if (e instanceof RetriesExhaustedException && (cause instanceof NotServingRegionException ||
196        cause instanceof ConnectException || cause instanceof SocketTimeoutException)) {
197        LOG.warn("log replaying of " + name + " can't connect to the target regionserver, " +
198          "resigning", e);
199        return Status.RESIGNED;
200      } else if (cause instanceof InterruptedException) {
201        LOG.warn("log splitting of " + name + " interrupted, resigning", e);
202        return Status.RESIGNED;
203      }
204      LOG.warn("log splitting of " + name + " failed, returning error", e);
205      return Status.ERR;
206    }
207    return Status.DONE;
208  }
209
210  @Override
211  public void run() {
212    try {
213      LOG.info("SplitLogWorker " + server.getServerName() + " starting");
214      coordination.registerListener();
215      // wait for Coordination Engine is ready
216      boolean res = false;
217      while (!res && !coordination.isStop()) {
218        res = coordination.isReady();
219      }
220      if (!coordination.isStop()) {
221        coordination.taskLoop();
222      }
223    } catch (Throwable t) {
224      if (ExceptionUtil.isInterrupt(t)) {
225        LOG.info("SplitLogWorker interrupted. Exiting. " + (coordination.isStop() ? "" :
226            " (ERROR: exitWorker is not set, exiting anyway)"));
227      } else {
228        // only a logical error can cause here. Printing it out
229        // to make debugging easier
230        LOG.error("unexpected error ", t);
231      }
232    } finally {
233      coordination.removeListener();
234      LOG.info("SplitLogWorker " + server.getServerName() + " exiting");
235    }
236  }

238  /**
239   * If the worker is doing a task i.e. splitting a log file then stop the task.
240   * It doesn't exit the worker thread.
241   */
242  public void stopTask() {
243    LOG.info("Sending interrupt to stop the worker thread");
244    worker.interrupt(); // TODO interrupt often gets swallowed, do what else?
245  }
246
247  /**
248   * start the SplitLogWorker thread
249   */
250  public void start() {
251    worker = new Thread(null, this, "SplitLogWorker-" + server.getServerName().toShortString());
252    worker.start();
253  }
254
255  /**
256   * stop the SplitLogWorker thread
257   */
258  public void stop() {
259    coordination.stopProcessingTasks();
260    stopTask();
261  }

263  /**
264   * Objects implementing this interface actually do the task that has been
265   * acquired by a {@link SplitLogWorker}. Since there isn't a water-tight
266   * guarantee that two workers will not be executing the same task therefore it
267   * is better to have workers prepare the task and then have the
268   * {@link org.apache.hadoop.hbase.master.SplitLogManager} commit the work in
269   * SplitLogManager.TaskFinisher
270   */
271  @FunctionalInterface
272  public interface TaskExecutor {
273    enum Status {
274      DONE(),
275      ERR(),
276      RESIGNED(),
277      PREEMPTED()
278    }
279    Status exec(String name, CancelableProgressable p);
280  }
281
282  /**
283   * Returns the number of tasks processed by coordination.
284   * This method is used by tests only
285   */
286  @VisibleForTesting
287  public int getTaskReadySeq() {
288    return coordination.getTaskReadySeq();
289  }
