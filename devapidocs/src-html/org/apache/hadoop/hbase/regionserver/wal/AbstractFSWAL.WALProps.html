116/**
117 * Implementation of {@link WAL} to go against {@link FileSystem}; i.e. keep WALs in HDFS. Only one
118 * WAL is ever being written at a time. When a WAL hits a configured maximum size, it is rolled.
119 * This is done internal to the implementation.
120 * <p>
121 * As data is flushed from the MemStore to other on-disk structures (files sorted by key, hfiles), a
122 * WAL becomes obsolete. We can let go of all the log edits/entries for a given HRegion-sequence id.
123 * A bunch of work in the below is done keeping account of these region sequence ids -- what is
124 * flushed out to hfiles, and what is yet in WAL and in memory only.
125 * <p>
126 * It is only practical to delete entire files. Thus, we delete an entire on-disk file
127 * <code>F</code> when all of the edits in <code>F</code> have a log-sequence-id that's older
128 * (smaller) than the most-recent flush.
129 * <p>
130 * To read an WAL, call {@link WALFactory#createStreamReader(FileSystem, Path)} for one way read,
131 * call {@link WALFactory#createTailingReader(FileSystem, Path, Configuration, long)} for
132 * replication where we may want to tail the active WAL file.
133 * <h2>Failure Semantic</h2> If an exception on append or sync, roll the WAL because the current WAL
134 * is now a lame duck; any more appends or syncs will fail also with the same original exception. If
135 * we have made successful appends to the WAL and we then are unable to sync them, our current
136 * semantic is to return error to the client that the appends failed but also to abort the current
137 * context, usually the hosting server. We need to replay the WALs. <br>
138 * TODO: Change this semantic. A roll of WAL may be sufficient as long as we have flagged client
139 * that the append failed. <br>
140 * TODO: replication may pick up these last edits though they have been marked as failed append
141 * (Need to keep our own file lengths, not rely on HDFS).
142 */

479  // must be power of 2
480  protected final int getPreallocatedEventCount() {
481    // Preallocate objects to use on the ring buffer. The way that appends and syncs work, we will
482    // be stuck and make no progress if the buffer is filled with appends only and there is no
483    // sync. If no sync, then the handlers will be outstanding just waiting on sync completion
484    // before they return.
485    int preallocatedEventCount = this.conf.getInt(RING_BUFFER_SLOT_COUNT, 1024 * 16);
486    checkArgument(preallocatedEventCount >= 0, RING_BUFFER_SLOT_COUNT + " must > 0");
487    int floor = Integer.highestOneBit(preallocatedEventCount);
488    if (floor == preallocatedEventCount) {
489      return floor;
490    }
491    // max capacity is 1 << 30
492    if (floor >= 1 << 29) {
493      return 1 << 30;
494    }
495    return floor << 1;
496  }

515  protected AbstractFSWAL(final FileSystem fs, final Abortable abortable, final Path rootDir,
516    final String logDir, final String archiveDir, final Configuration conf,
517    final List<WALActionsListener> listeners, final boolean failIfWALExists, final String prefix,
518    final String suffix, FileSystem remoteFs, Path remoteWALDir)
519    throws FailedLogCloseException, IOException {
520    this.fs = fs;
521    this.walDir = new Path(rootDir, logDir);
522    this.walArchiveDir = new Path(rootDir, archiveDir);
523    this.conf = conf;
524    this.abortable = abortable;
525    this.remoteFs = remoteFs;
526    this.remoteWALDir = remoteWALDir;
527
528    if (!fs.exists(walDir) && !fs.mkdirs(walDir)) {
529      throw new IOException("Unable to mkdir " + walDir);
530    }
531
532    if (!fs.exists(this.walArchiveDir)) {
533      if (!fs.mkdirs(this.walArchiveDir)) {
534        throw new IOException("Unable to mkdir " + this.walArchiveDir);
535      }
536    }
537
538    // If prefix is null||empty then just name it wal
539    this.walFilePrefix = prefix == null || prefix.isEmpty()
540      ? "wal"
541      : URLEncoder.encode(prefix, StandardCharsets.UTF_8.name());
542    // we only correctly differentiate suffices when numeric ones start with '.'
543    if (suffix != null && !(suffix.isEmpty()) && !(suffix.startsWith(WAL_FILE_NAME_DELIMITER))) {
544      throw new IllegalArgumentException("WAL suffix must start with '" + WAL_FILE_NAME_DELIMITER
545        + "' but instead was '" + suffix + "'");
546    }
547    // Now that it exists, set the storage policy for the entire directory of wal files related to
548    // this FSHLog instance
549    String storagePolicy =
550      conf.get(HConstants.WAL_STORAGE_POLICY, HConstants.DEFAULT_WAL_STORAGE_POLICY);
551    CommonFSUtils.setStoragePolicy(fs, this.walDir, storagePolicy);
552    this.walFileSuffix = (suffix == null) ? "" : URLEncoder.encode(suffix, "UTF8");
553    this.prefixPathStr = new Path(walDir, walFilePrefix + WAL_FILE_NAME_DELIMITER).toString();
554
555    this.ourFiles = new PathFilter() {
556      @Override
557      public boolean accept(final Path fileName) {
558        // The path should start with dir/<prefix> and end with our suffix
559        final String fileNameString = fileName.toString();
560        if (!fileNameString.startsWith(prefixPathStr)) {
561          return false;
562        }
563        if (walFileSuffix.isEmpty()) {
564          // in the case of the null suffix, we need to ensure the filename ends with a timestamp.
565          return org.apache.commons.lang3.StringUtils
566            .isNumeric(fileNameString.substring(prefixPathStr.length()));
567        } else if (!fileNameString.endsWith(walFileSuffix)) {
568          return false;
569        }
570        return true;
571      }
572    };
573
574    if (failIfWALExists) {
575      final FileStatus[] walFiles = CommonFSUtils.listStatus(fs, walDir, ourFiles);
576      if (null != walFiles && 0 != walFiles.length) {
577        throw new IOException("Target WAL already exists within directory " + walDir);
578      }
579    }
580
581    // Register listeners. TODO: Should this exist anymore? We have CPs?
582    if (listeners != null) {
583      for (WALActionsListener i : listeners) {
584        registerWALActionsListener(i);
585      }
586    }
587    this.coprocessorHost = new WALCoprocessorHost(this, conf);
588
589    // Schedule a WAL roll when the WAL is 50% of the HDFS block size. Scheduling at 50% of block
590    // size should make it so WAL rolls before we get to the end-of-block (Block transitions cost
591    // some latency). In hbase-1 we did this differently. We scheduled a roll when we hit 95% of
592    // the block size but experience from the field has it that this was not enough time for the
593    // roll to happen before end-of-block. So the new accounting makes WALs of about the same
594    // size as those made in hbase-1 (to prevent surprise), we now have default block size as
595    // 2 times the DFS default: i.e. 2 * DFS default block size rolling at 50% full will generally
596    // make similar size logs to 1 * DFS default block size rolling at 95% full. See HBASE-19148.
597    this.blocksize = WALUtil.getWALBlockSize(this.conf, this.fs, this.walDir);
598    float multiplier = conf.getFloat(WAL_ROLL_MULTIPLIER, 0.5f);
599    this.logrollsize = (long) (this.blocksize * multiplier);
600    this.maxLogs = conf.getInt(MAX_LOGS, Math.max(32, calculateMaxLogFiles(conf, logrollsize)));
601
602    LOG.info("WAL configuration: blocksize=" + StringUtils.byteDesc(blocksize) + ", rollsize="
603      + StringUtils.byteDesc(this.logrollsize) + ", prefix=" + this.walFilePrefix + ", suffix="
604      + walFileSuffix + ", logDir=" + this.walDir + ", archiveDir=" + this.walArchiveDir
605      + ", maxLogs=" + this.maxLogs);
606    this.slowSyncNs =
607      TimeUnit.MILLISECONDS.toNanos(conf.getInt(SLOW_SYNC_TIME_MS, DEFAULT_SLOW_SYNC_TIME_MS));
608    this.rollOnSyncNs = TimeUnit.MILLISECONDS
609      .toNanos(conf.getInt(ROLL_ON_SYNC_TIME_MS, DEFAULT_ROLL_ON_SYNC_TIME_MS));
610    this.slowSyncRollThreshold =
611      conf.getInt(SLOW_SYNC_ROLL_THRESHOLD, DEFAULT_SLOW_SYNC_ROLL_THRESHOLD);
612    this.slowSyncCheckInterval =
613      conf.getInt(SLOW_SYNC_ROLL_INTERVAL_MS, DEFAULT_SLOW_SYNC_ROLL_INTERVAL_MS);
614    this.walSyncTimeoutNs =
615      TimeUnit.MILLISECONDS.toNanos(conf.getLong(WAL_SYNC_TIMEOUT_MS, DEFAULT_WAL_SYNC_TIMEOUT_MS));
616    this.syncFutureCache = new SyncFutureCache(conf);
617    this.implClassName = getClass().getSimpleName();
618    this.walTooOldNs = TimeUnit.SECONDS
619      .toNanos(conf.getInt(SURVIVED_TOO_LONG_SEC_KEY, SURVIVED_TOO_LONG_SEC_DEFAULT));
620    this.useHsync = conf.getBoolean(HRegion.WAL_HSYNC_CONF_KEY, HRegion.DEFAULT_WAL_HSYNC);
621    archiveRetries = this.conf.getInt("hbase.regionserver.walroll.archive.retries", 0);
622    this.walShutdownTimeout =
623      conf.getLong(WAL_SHUTDOWN_WAIT_TIMEOUT_MS, DEFAULT_WAL_SHUTDOWN_WAIT_TIMEOUT_MS);
624
625    int preallocatedEventCount =
626      conf.getInt("hbase.regionserver.wal.disruptor.event.count", 1024 * 16);
627    waitingConsumePayloads =
628      RingBuffer.createMultiProducer(RingBufferTruck::new, preallocatedEventCount);
629    waitingConsumePayloadsGatingSequence = new Sequence(Sequencer.INITIAL_CURSOR_VALUE);
630    waitingConsumePayloads.addGatingSequences(waitingConsumePayloadsGatingSequence);
631
632    // inrease the ringbuffer sequence so our txid is start from 1
633    waitingConsumePayloads.publish(waitingConsumePayloads.next());
634    waitingConsumePayloadsGatingSequence.set(waitingConsumePayloads.getCursor());
635
636    batchSize = conf.getLong(WAL_BATCH_SIZE, DEFAULT_WAL_BATCH_SIZE);
637  }

852  /**
853   * Archive old logs. A WAL is eligible for archiving if all its WALEdits have been flushed.
854   * <p/>
855   * Use synchronized because we may call this method in different threads, normally when replacing
856   * writer, and since now close writer may be asynchronous, we will also call this method in the
857   * closeExecutor, right after we actually close a WAL writer.
858   */
859  private synchronized void cleanOldLogs() {
860    List<Pair<Path, Long>> logsToArchive = null;
861    long now = System.nanoTime();
862    boolean mayLogTooOld = nextLogTooOldNs <= now;
863    ArrayList<byte[]> regionsBlockingWal = null;
864    // For each log file, look at its Map of regions to highest sequence id; if all sequence ids
865    // are older than what is currently in memory, the WAL can be GC'd.
866    for (Map.Entry<Path, WALProps> e : this.walFile2Props.entrySet()) {
867      if (!e.getValue().closed) {
868        LOG.debug("{} is not closed yet, will try archiving it next time", e.getKey());
869        continue;
870      }
871      Path log = e.getKey();
872      ArrayList<byte[]> regionsBlockingThisWal = null;
873      long ageNs = now - e.getValue().rollTimeNs;
874      if (ageNs > walTooOldNs) {
875        if (mayLogTooOld && regionsBlockingWal == null) {
876          regionsBlockingWal = new ArrayList<>();
877        }
878        regionsBlockingThisWal = regionsBlockingWal;
879      }
880      Map<byte[], Long> sequenceNums = e.getValue().encodedName2HighestSequenceId;
881      if (this.sequenceIdAccounting.areAllLower(sequenceNums, regionsBlockingThisWal)) {
882        if (logsToArchive == null) {
883          logsToArchive = new ArrayList<>();
884        }
885        logsToArchive.add(Pair.newPair(log, e.getValue().logSize));
886        if (LOG.isTraceEnabled()) {
887          LOG.trace("WAL file ready for archiving " + log);
888        }
889      } else if (regionsBlockingThisWal != null) {
890        StringBuilder sb = new StringBuilder(log.toString()).append(" has not been archived for ")
891          .append(TimeUnit.NANOSECONDS.toSeconds(ageNs)).append(" seconds; blocked by: ");
892        boolean isFirst = true;
893        for (byte[] region : regionsBlockingThisWal) {
894          if (!isFirst) {
895            sb.append("; ");
896          }
897          isFirst = false;
898          sb.append(Bytes.toString(region));
899        }
900        LOG.warn(sb.toString());
901        nextLogTooOldNs = now + SURVIVED_TOO_LONG_LOG_INTERVAL_NS;
902        regionsBlockingThisWal.clear();
903      }
904    }
905
906    if (logsToArchive != null) {
907      final List<Pair<Path, Long>> localLogsToArchive = logsToArchive;
908      // make it async
909      for (Pair<Path, Long> log : localLogsToArchive) {
910        logArchiveExecutor.execute(() -> {
911          archive(log);
912        });
913        this.walFile2Props.remove(log.getFirst());
914      }
915    }
916  }
917
918  protected void archive(final Pair<Path, Long> log) {
919    totalLogSize.addAndGet(-log.getSecond());
920    int retry = 1;
921    while (true) {
922      try {
923        archiveLogFile(log.getFirst());
924        // successful
925        break;
926      } catch (Throwable e) {
927        if (retry > archiveRetries) {
928          LOG.error("Failed log archiving for the log {},", log.getFirst(), e);
929          if (this.abortable != null) {
930            this.abortable.abort("Failed log archiving", e);
931            break;
932          }
933        } else {
934          LOG.error("Log archiving failed for the log {} - attempt {}", log.getFirst(), retry, e);
935        }
936        retry++;
937      }
938    }
939  }

1042  private W createCombinedWriter(W localWriter, Path localPath)
1043    throws IOException, CommonFSUtils.StreamLacksCapabilityException {
1044    // retry forever if we can not create the remote writer to prevent aborting the RS due to log
1045    // rolling error, unless the skipRemoteWal is set to true.
1046    // TODO: since for now we only have one thread doing log rolling, this may block the rolling for
1047    // other wals
1048    Path remoteWAL = new Path(remoteWALDir, localPath.getName());
1049    for (int retry = 0;; retry++) {
1050      if (skipRemoteWAL) {
1051        return localWriter;
1052      }
1053      W remoteWriter;
1054      try {
1055        remoteWriter = createWriterInstance(remoteFs, remoteWAL);
1056      } catch (IOException e) {
1057        LOG.warn("create remote writer {} failed, retry = {}", remoteWAL, retry, e);
1058        try {
1059          Thread.sleep(ConnectionUtils.getPauseTime(100, retry));
1060        } catch (InterruptedException ie) {
1061          // restore the interrupt state
1062          Thread.currentThread().interrupt();
1063          // must close local writer here otherwise no one will close it for us
1064          Closeables.close(localWriter, true);
1065          throw (IOException) new InterruptedIOException().initCause(ie);
1066        }
1067        continue;
1068      }
1069      return createCombinedWriter(localWriter, remoteWriter);
1070    }
1071  }

1073  private Map<byte[], List<byte[]>> rollWriterInternal(boolean force) throws IOException {
1074    rollWriterLock.lock();
1075    try {
1076      if (this.closed) {
1077        throw new WALClosedException("WAL has been closed");
1078      }
1079      // Return if nothing to flush.
1080      if (!force && this.writer != null && this.numEntries.get() <= 0) {
1081        return null;
1082      }
1083      Map<byte[], List<byte[]>> regionsToFlush = null;
1084      try {
1085        Path oldPath = getOldPath();
1086        Path newPath = getNewPath();
1087        // Any exception from here on is catastrophic, non-recoverable so we currently abort.
1088        W nextWriter = this.createWriterInstance(fs, newPath);
1089        if (remoteFs != null) {
1090          // create a remote wal if necessary
1091          nextWriter = createCombinedWriter(nextWriter, newPath);
1092        }
1093        tellListenersAboutPreLogRoll(oldPath, newPath);
1094        // NewPath could be equal to oldPath if replaceWriter fails.
1095        newPath = replaceWriter(oldPath, newPath, nextWriter);
1096        tellListenersAboutPostLogRoll(oldPath, newPath);
1097        if (LOG.isDebugEnabled()) {
1098          LOG.debug("Create new " + implClassName + " writer with pipeline: "
1099            + Arrays.toString(getPipeline()));
1100        }
1101        // We got a new writer, so reset the slow sync count
1102        lastTimeCheckSlowSync = EnvironmentEdgeManager.currentTime();
1103        slowSyncCount.set(0);
1104        // Can we delete any of the old log files?
1105        if (getNumRolledLogFiles() > 0) {
1106          cleanOldLogs();
1107          regionsToFlush = findRegionsToForceFlush();
1108        }
1109      } catch (CommonFSUtils.StreamLacksCapabilityException exception) {
1110        // If the underlying FileSystem can't do what we ask, treat as IO failure so
1111        // we'll abort.
1112        throw new IOException(
1113          "Underlying FileSystem can't meet stream requirements. See RS log " + "for details.",
1114          exception);
1115      }
1116      return regionsToFlush;
1117    } finally {
1118      rollWriterLock.unlock();
1119    }
1120  }

1146  @Override
1147  public void shutdown() throws IOException {
1148    if (!shutdown.compareAndSet(false, true)) {
1149      return;
1150    }
1151    closed = true;
1152    // Tell our listeners that the log is closing
1153    if (!this.listeners.isEmpty()) {
1154      for (WALActionsListener i : this.listeners) {
1155        i.logCloseRequested();
1156      }
1157    }
1158
1159    ExecutorService shutdownExecutor = Executors.newSingleThreadExecutor(
1160      new ThreadFactoryBuilder().setDaemon(true).setNameFormat("WAL-Shutdown-%d").build());
1161
1162    Future<Void> future = shutdownExecutor.submit(new Callable<Void>() {
1163      @Override
1164      public Void call() throws Exception {
1165        if (rollWriterLock.tryLock(walShutdownTimeout, TimeUnit.SECONDS)) {
1166          try {
1167            doShutdown();
1168            if (syncFutureCache != null) {
1169              syncFutureCache.clear();
1170            }
1171          } finally {
1172            rollWriterLock.unlock();
1173          }
1174        } else {
1175          throw new IOException("Waiting for rollWriterLock timeout");
1176        }
1177        return null;
1178      }
1179    });
1180    shutdownExecutor.shutdown();
1181
1182    try {
1183      future.get(walShutdownTimeout, TimeUnit.MILLISECONDS);
1184    } catch (InterruptedException e) {
1185      throw new InterruptedIOException("Interrupted when waiting for shutdown WAL");
1186    } catch (TimeoutException e) {
1187      throw new TimeoutIOException("We have waited " + walShutdownTimeout + "ms, but"
1188        + " the shutdown of WAL doesn't complete! Please check the status of underlying "
1189        + "filesystem or increase the wait time by the config \"" + WAL_SHUTDOWN_WAIT_TIMEOUT_MS
1190        + "\"", e);
1191    } catch (ExecutionException e) {
1192      if (e.getCause() instanceof IOException) {
1193        throw (IOException) e.getCause();
1194      } else {
1195        throw new IOException(e.getCause());
1196      }
1197    } finally {
1198      // in shutdown we may call cleanOldLogs so shutdown this executor in the end.
1199      // In sync replication implementation, we may shutdown a WAL without shutting down the whole
1200      // region server, if we shutdown this executor earlier we may get reject execution exception
1201      // and abort the region server
1202      logArchiveExecutor.shutdown();
1203    }
1204    // we also need to wait logArchive to finish if we want to a graceful shutdown as we may still
1205    // have some pending archiving tasks not finished yet, and in close we may archive all the
1206    // remaining WAL files, there could be race if we do not wait for the background archive task
1207    // finish
1208    try {
1209      if (!logArchiveExecutor.awaitTermination(walShutdownTimeout, TimeUnit.MILLISECONDS)) {
1210        throw new TimeoutIOException("We have waited " + walShutdownTimeout + "ms, but"
1211          + " the shutdown of WAL doesn't complete! Please check the status of underlying "
1212          + "filesystem or increase the wait time by the config \"" + WAL_SHUTDOWN_WAIT_TIMEOUT_MS
1213          + "\"");
1214      }
1215    } catch (InterruptedException e) {
1216      throw new InterruptedIOException("Interrupted when waiting for shutdown WAL");
1217    }
1218  }

1352    if (timeInNanos > this.slowSyncNs) {

1361        // action, so do so.
1362        LOG.warn("Requesting log roll because we exceeded slow sync threshold; time="

1369    }
1370    if (!listeners.isEmpty()) {
1371      for (WALActionsListener listener : listeners) {
1372        listener.postSync(timeInNanos, handlerSyncs);
1373      }
1374    }

1467  // return whether we have successfully set readyForRolling to true.
1468  private boolean trySetReadyForRolling() {
1469    // Check without holding lock first. Usually we will just return here.
1470    // waitingRoll is volatile and unacedEntries is only accessed inside event loop so it is safe to
1471    // check them outside the consumeLock.
1472    if (!waitingRoll(epochAndState) || !unackedAppends.isEmpty()) {
1473      return false;
1474    }
1475    consumeLock.lock();
1476    try {
1477      // 1. a roll is requested
1478      // 2. all out-going entries have been acked(we have confirmed above).
1479      if (waitingRoll(epochAndState)) {
1480        readyForRolling = true;
1481        readyForRollingCond.signalAll();
1482        return true;
1483      } else {
1484        return false;
1485      }
1486    } finally {
1487      consumeLock.unlock();
1488    }
1489  }

1496  private void onException(long epochWhenSync, Throwable error) {
1497    boolean shouldRequestLogRoll = true;
1498    consumeLock.lock();
1499    try {
1500      int currentEpochAndState = epochAndState;
1501      if (epoch(currentEpochAndState) != epochWhenSync || writerBroken(currentEpochAndState)) {
1502        // this is not the previous writer which means we have already rolled the writer.
1503        // or this is still the current writer, but we have already marked it as broken and request
1504        // a roll.
1505        return;
1506      }
1507      this.epochAndState = currentEpochAndState | 0b10;
1508      if (waitingRoll(currentEpochAndState)) {
1509        readyForRolling = true;
1510        readyForRollingCond.signalAll();
1511        // this means we have already in the middle of a rollWriter so just tell the roller thread
1512        // that you can continue without requesting an extra log roll.
1513        shouldRequestLogRoll = false;
1514      }
1515    } finally {
1516      consumeLock.unlock();
1517    }
1518    for (Iterator<FSWALEntry> iter = unackedAppends.descendingIterator(); iter.hasNext();) {
1519      toWriteAppends.addFirst(iter.next());
1520    }
1521    highestUnsyncedTxid = highestSyncedTxid.get();
1522    if (shouldRequestLogRoll) {
1523      // request a roll.
1524      requestLogRoll(ERROR);
1525    }
1526  }

1528  private void syncCompleted(long epochWhenSync, W writer, long processedTxid, long startTimeNs) {
1529    // Please see the last several comments on HBASE-22761, it is possible that we get a
1530    // syncCompleted which acks a previous sync request after we received a syncFailed on the same
1531    // writer. So here we will also check on the epoch and state, if the epoch has already been
1532    // changed, i.e, we have already rolled the writer, or the writer is already broken, we should
1533    // just skip here, to avoid mess up the state or accidentally release some WAL entries and
1534    // cause data corruption.
1535    // The syncCompleted call is on the critical write path so we should try our best to make it
1536    // fast. So here we do not hold consumeLock, for increasing performance. It is safe because
1537    // there are only 3 possible situations:
1538    // 1. For normal case, the only place where we change epochAndState is when rolling the writer.
1539    // Before rolling actually happen, we will only change the state to waitingRoll which is another
1540    // bit than writerBroken, and when we actually change the epoch, we can make sure that there is
1541    // no out going sync request. So we will always pass the check here and there is no problem.
1542    // 2. The writer is broken, but we have not called syncFailed yet. In this case, since
1543    // syncFailed and syncCompleted are executed in the same thread, we will just face the same
1544    // situation with #1.
1545    // 3. The writer is broken, and syncFailed has been called. Then when we arrive here, there are
1546    // only 2 possible situations:
1547    // a. we arrive before we actually roll the writer, then we will find out the writer is broken
1548    // and give up.
1549    // b. we arrive after we actually roll the writer, then we will find out the epoch is changed
1550    // and give up.
1551    // For both #a and #b, we do not need to hold the consumeLock as we will always update the
1552    // epochAndState as a whole.
1553    // So in general, for all the cases above, we do not need to hold the consumeLock.
1554    int epochAndState = this.epochAndState;
1555    if (epoch(epochAndState) != epochWhenSync || writerBroken(epochAndState)) {
1556      LOG.warn("Got a sync complete call after the writer is broken, skip");
1557      return;
1558    }
1559
1560    if (processedTxid < highestSyncedTxid.get()) {
1561      return;
1562    }
1563    highestSyncedTxid.set(processedTxid);
1564    for (Iterator<FSWALEntry> iter = unackedAppends.iterator(); iter.hasNext();) {
1565      FSWALEntry entry = iter.next();
1566      if (entry.getTxid() <= processedTxid) {
1567        entry.release();
1568        iter.remove();
1569      } else {
1570        break;
1571      }
1572    }
1573    postSync(System.nanoTime() - startTimeNs, finishSync());
1574    /**
1575     * This method is used to be compatible with the original logic of {@link FSHLog}.
1576     */
1577    checkSlowSyncCount();
1578    if (trySetReadyForRolling()) {
1579      // we have just finished a roll, then do not need to check for log rolling, the writer will be
1580      // closed soon.
1581      return;
1582    }
1583    // If we haven't already requested a roll, check if we have exceeded logrollsize
1584    if (!isLogRollRequested() && writer.getLength() > logrollsize) {
1585      if (LOG.isDebugEnabled()) {
1586        LOG.debug("Requesting log roll because of file size threshold; length=" + writer.getLength()
1587          + ", logrollsize=" + logrollsize);
1588      }
1589      requestLogRoll(SIZE);
1590    }
1591  }

1657  // try advancing the highestSyncedTxid as much as possible
1658  private int finishSync() {
1659    if (unackedAppends.isEmpty()) {
1660      // All outstanding appends have been acked.
1661      if (toWriteAppends.isEmpty()) {
1662        // Also no appends that wait to be written out, then just finished all pending syncs.
1663        long maxSyncTxid = highestSyncedTxid.get();
1664        for (SyncFuture sync : syncFutures) {
1665          maxSyncTxid = Math.max(maxSyncTxid, sync.getTxid());
1666          markFutureDoneAndOffer(sync, maxSyncTxid, null);
1667        }
1668        highestSyncedTxid.set(maxSyncTxid);
1669        int finished = syncFutures.size();
1670        syncFutures.clear();
1671        return finished;
1672      } else {
1673        // There is no append between highestProcessedAppendTxid and lowestUnprocessedAppendTxid, so
1674        // if highestSyncedTxid >= highestProcessedAppendTxid, then all syncs whose txid are between
1675        // highestProcessedAppendTxid and lowestUnprocessedAppendTxid can be finished.
1676        long lowestUnprocessedAppendTxid = toWriteAppends.peek().getTxid();
1677        assert lowestUnprocessedAppendTxid > highestProcessedAppendTxid;
1678        long doneTxid = lowestUnprocessedAppendTxid - 1;
1679        highestSyncedTxid.set(doneTxid);
1680        return finishSyncLowerThanTxid(doneTxid);
1681      }
1682    } else {
1683      // There are still unacked appends. So let's move the highestSyncedTxid to the txid of the
1684      // first unacked append minus 1.
1685      long lowestUnackedAppendTxid = unackedAppends.peek().getTxid();
1686      long doneTxid = Math.max(lowestUnackedAppendTxid - 1, highestSyncedTxid.get());
1687      highestSyncedTxid.set(doneTxid);
1688      return finishSyncLowerThanTxid(doneTxid);
1689    }
1690  }

1697  private void appendAndSync() throws IOException {
1698    final W writer = this.writer;
1699    // maybe a sync request is not queued when we issue a sync, so check here to see if we could
1700    // finish some.
1701    finishSync();
1702    long newHighestProcessedAppendTxid = -1L;
1703    // this is used to avoid calling peedLast every time on unackedAppends, appendAndAsync is single
1704    // threaded, this could save us some cycles
1705    boolean addedToUnackedAppends = false;
1706    for (Iterator<FSWALEntry> iter = toWriteAppends.iterator(); iter.hasNext();) {
1707      FSWALEntry entry = iter.next();
1708      /**
1709       * For {@link FSHog},here may throws IOException,but for {@link AsyncFSWAL}, here would not
1710       * throw any IOException.
1711       */
1712      boolean appended = appendEntry(writer, entry);
1713      newHighestProcessedAppendTxid = entry.getTxid();
1714      iter.remove();
1715      if (appended) {
1716        // This is possible, when we fail to sync, we will add the unackedAppends back to
1717        // toWriteAppends, so here we may get an entry which is already in the unackedAppends.
1718        if (
1719          addedToUnackedAppends || unackedAppends.isEmpty()
1720            || getLastTxid(unackedAppends) < entry.getTxid()
1721        ) {
1722          unackedAppends.addLast(entry);
1723          addedToUnackedAppends = true;
1724        }
1725        // See HBASE-25905, here we need to make sure that, we will always write all the entries in
1726        // unackedAppends out. As the code in the consume method will assume that, the entries in
1727        // unackedAppends have all been sent out so if there is roll request and unackedAppends is
1728        // not empty, we could just return as later there will be a syncCompleted call to clear the
1729        // unackedAppends, or a syncFailed to lead us to another state.
1730        // There could be other ways to fix, such as changing the logic in the consume method, but
1731        // it will break the assumption and then (may) lead to a big refactoring. So here let's use
1732        // this way to fix first, can optimize later.
1733        if (
1734          writer.getLength() - fileLengthAtLastSync >= batchSize
1735            && (addedToUnackedAppends || entry.getTxid() >= getLastTxid(unackedAppends))
1736        ) {
1737          break;
1738        }
1739      }
1740    }
1741    // if we have a newer transaction id, update it.
1742    // otherwise, use the previous transaction id.
1743    if (newHighestProcessedAppendTxid > 0) {
1744      highestProcessedAppendTxid = newHighestProcessedAppendTxid;
1745    } else {
1746      newHighestProcessedAppendTxid = highestProcessedAppendTxid;
1747    }
1748
1749    if (writer.getLength() - fileLengthAtLastSync >= batchSize) {
1750      // sync because buffer size limit.
1751      sync(writer);
1752      return;
1753    }
1754    if (writer.getLength() == fileLengthAtLastSync) {
1755      // we haven't written anything out, just advance the highestSyncedSequence since we may only
1756      // stamped some region sequence id.
1757      if (unackedAppends.isEmpty()) {
1758        highestSyncedTxid.set(highestProcessedAppendTxid);
1759        finishSync();
1760        trySetReadyForRolling();
1761      }
1762      return;
1763    }
1764    // reach here means that we have some unsynced data but haven't reached the batch size yet
1765    // but we will not issue a sync directly here even if there are sync requests because we may
1766    // have some new data in the ringbuffer, so let's just return here and delay the decision of
1767    // whether to issue a sync in the caller method.
1768  }

1770  private void consume() {
1771    consumeLock.lock();
1772    try {
1773      int currentEpochAndState = epochAndState;
1774      if (writerBroken(currentEpochAndState)) {
1775        return;
1776      }
1777      if (waitingRoll(currentEpochAndState)) {
1778        if (writer.getLength() > fileLengthAtLastSync) {
1779          // issue a sync
1780          sync(writer);
1781        } else {
1782          if (unackedAppends.isEmpty()) {
1783            readyForRolling = true;
1784            readyForRollingCond.signalAll();
1785          }
1786        }
1787        return;
1788      }
1789    } finally {
1790      consumeLock.unlock();
1791    }
1792    long nextCursor = waitingConsumePayloadsGatingSequence.get() + 1;
1793    for (long cursorBound = waitingConsumePayloads.getCursor(); nextCursor
1794        <= cursorBound; nextCursor++) {
1795      if (!waitingConsumePayloads.isPublished(nextCursor)) {
1796        break;
1797      }
1798      RingBufferTruck truck = waitingConsumePayloads.get(nextCursor);
1799      switch (truck.type()) {
1800        case APPEND:
1801          toWriteAppends.addLast(truck.unloadAppend());
1802          break;
1803        case SYNC:
1804          syncFutures.add(truck.unloadSync());
1805          break;
1806        default:
1807          LOG.warn("RingBufferTruck with unexpected type: " + truck.type());
1808          break;
1809      }
1810      waitingConsumePayloadsGatingSequence.set(nextCursor);
1811    }
1812
1813    /**
1814     * This method is used to be compatible with the original logic of {@link AsyncFSWAL}.
1815     */
1816    if (markerEditOnly) {
1817      drainNonMarkerEditsAndFailSyncs();
1818    }
1819    try {
1820      appendAndSync();
1821    } catch (IOException exception) {
1822      /**
1823       * For {@link FSHog},here may catch IOException,but for {@link AsyncFSWAL}, the code doesn't
1824       * go in here.
1825       */
1826      LOG.error("appendAndSync throws IOException.", exception);
1827      onAppendEntryFailed(exception);
1828      return;
1829    }
1830    if (hasConsumerTask.get()) {
1831      return;
1832    }
1833    if (toWriteAppends.isEmpty()) {
1834      if (waitingConsumePayloadsGatingSequence.get() == waitingConsumePayloads.getCursor()) {
1835        consumerScheduled.set(false);
1836        // recheck here since in append and sync we do not hold the consumeLock. Thing may
1837        // happen like
1838        // 1. we check cursor, no new entry
1839        // 2. someone publishes a new entry to ringbuffer and the consumerScheduled is true and
1840        // give up scheduling the consumer task.
1841        // 3. we set consumerScheduled to false and also give up scheduling consumer task.
1842        if (waitingConsumePayloadsGatingSequence.get() == waitingConsumePayloads.getCursor()) {
1843          // we will give up consuming so if there are some unsynced data we need to issue a sync.
1844          if (
1845            writer.getLength() > fileLengthAtLastSync && !syncFutures.isEmpty()
1846              && syncFutures.last().getTxid() > highestProcessedAppendTxidAtLastSync
1847          ) {
1848            // no new data in the ringbuffer and we have at least one sync request
1849            sync(writer);
1850          }
1851          return;
1852        } else {
1853          // maybe someone has grabbed this before us
1854          if (!consumerScheduled.compareAndSet(false, true)) {
1855            return;
1856          }
1857        }
1858      }
1859    }
1860    // reschedule if we still have something to write.
1861    consumeExecutor.execute(consumer);
1862  }

1864  private boolean shouldScheduleConsumer() {
1865    int currentEpochAndState = epochAndState;
1866    if (writerBroken(currentEpochAndState) || waitingRoll(currentEpochAndState)) {
1867      return false;
1868    }
1869    return consumerScheduled.compareAndSet(false, true);

1872  /**
1873   * Append a set of edits to the WAL.
1874   * <p/>
1875   * The WAL is not flushed/sync'd after this transaction completes BUT on return this edit must
1876   * have its region edit/sequence id assigned else it messes up our unification of mvcc and
1877   * sequenceid. On return <code>key</code> will have the region edit/sequence id filled in.
1878   * <p/>
1879   * NOTE: This append, at a time that is usually after this call returns, starts an mvcc
1880   * transaction by calling 'begin' wherein which we assign this update a sequenceid. At assignment
1881   * time, we stamp all the passed in Cells inside WALEdit with their sequenceId. You must
1882   * 'complete' the transaction this mvcc transaction by calling
1883   * MultiVersionConcurrencyControl#complete(...) or a variant otherwise mvcc will get stuck. Do it
1884   * in the finally of a try/finally block within which this append lives and any subsequent
1885   * operations like sync or update of memstore, etc. Get the WriteEntry to pass mvcc out of the
1886   * passed in WALKey <code>walKey</code> parameter. Be warned that the WriteEntry is not
1887   * immediately available on return from this method. It WILL be available subsequent to a sync of
1888   * this append; otherwise, you will just have to wait on the WriteEntry to get filled in.
1889   * @param hri        the regioninfo associated with append
1890   * @param key        Modified by this call; we add to it this edits region edit/sequence id.
1891   * @param edits      Edits to append. MAY CONTAIN NO EDITS for case where we want to get an edit
1892   *                   sequence id that is after all currently appended edits.
1893   * @param inMemstore Always true except for case where we are writing a region event meta marker
1894   *                   edit, for example, a compaction completion record into the WAL or noting a
1895   *                   Region Open event. In these cases the entry is just so we can finish an
1896   *                   unfinished compaction after a crash when the new Server reads the WAL on
1897   *                   recovery, etc. These transition event 'Markers' do not go via the memstore.
1898   *                   When memstore is false, we presume a Marker event edit.
1899   * @return Returns a 'transaction id' and <code>key</code> will have the region edit/sequence id
1900   *         in it.
1901   */
1902  protected long append(RegionInfo hri, WALKeyImpl key, WALEdit edits, boolean inMemstore)
1903    throws IOException {
1904    if (markerEditOnly && !edits.isMetaEdit()) {
1905      throw new IOException("WAL is closing, only marker edit is allowed");
1906    }
1907    long txid =
1908      stampSequenceIdAndPublishToRingBuffer(hri, key, edits, inMemstore, waitingConsumePayloads);
1909    if (shouldScheduleConsumer()) {
1910      consumeExecutor.execute(consumer);
1911    }
1912    return txid;
1913  }

1916    long txid = waitingConsumePayloads.next();
1917    SyncFuture future;
1918    try {
1919      future = getSyncFuture(txid, forceSync);
1920      RingBufferTruck truck = waitingConsumePayloads.get(txid);
1921      truck.load(future);
1922    } finally {
1923      waitingConsumePayloads.publish(txid);
1924    }
1925    if (shouldScheduleConsumer()) {
1926      consumeExecutor.execute(consumer);
1927    }
1928    blockOnSync(future);

1931  protected void doSync(long txid, boolean forceSync) throws IOException {
1932    if (highestSyncedTxid.get() >= txid) {
1933      return;
1934    }
1935    // here we do not use ring buffer sequence as txid
1936    long sequence = waitingConsumePayloads.next();
1937    SyncFuture future;
1938    try {

1940      RingBufferTruck truck = waitingConsumePayloads.get(sequence);

1942    } finally {
1943      waitingConsumePayloads.publish(sequence);
1944    }
1945    if (shouldScheduleConsumer()) {
1946      consumeExecutor.execute(consumer);
1947    }
1948    blockOnSync(future);

1951  private void drainNonMarkerEditsAndFailSyncs() {
1952    if (toWriteAppends.isEmpty()) {
1953      return;
1954    }
1955    boolean hasNonMarkerEdits = false;
1956    Iterator<FSWALEntry> iter = toWriteAppends.descendingIterator();
1957    while (iter.hasNext()) {
1958      FSWALEntry entry = iter.next();
1959      if (!entry.getEdit().isMetaEdit()) {
1960        entry.release();
1961        hasNonMarkerEdits = true;
1962        break;
1963      }
1964    }
1965    if (hasNonMarkerEdits) {
1966      for (;;) {
1967        iter.remove();
1968        if (!iter.hasNext()) {
1969          break;
1970        }
1971        iter.next().release();
1972      }
1973      for (FSWALEntry entry : unackedAppends) {
1974        entry.release();
1975      }
1976      unackedAppends.clear();
1977      // fail the sync futures which are under the txid of the first remaining edit, if none, fail
1978      // all the sync futures.
1979      long txid = toWriteAppends.isEmpty() ? Long.MAX_VALUE : toWriteAppends.peek().getTxid();
1980      IOException error = new IOException("WAL is closing, only marker edit is allowed");
1981      for (Iterator<SyncFuture> syncIter = syncFutures.iterator(); syncIter.hasNext();) {
1982        SyncFuture future = syncIter.next();
1983        if (future.getTxid() < txid) {
1984          markFutureDoneAndOffer(future, future.getTxid(), error);
1985          syncIter.remove();
1986        } else {
1987          break;
1988        }
1989      }
1990    }
1991  }

2040  protected void doReplaceWriter(Path oldPath, Path newPath, W nextWriter) throws IOException {
2041    Preconditions.checkNotNull(nextWriter);
2042    waitForSafePoint();
2043    /**
2044     * For {@link FSHLog},here would shutdown {@link FSHLog.SyncRunner}.
2045     */
2046    doCleanUpResources();
2047    // we will call rollWriter in init method, where we want to create the first writer and
2048    // obviously the previous writer is null, so here we need this null check. And why we must call
2049    // logRollAndSetupWalProps before closeWriter is that, we will call markClosedAndClean after
2050    // closing the writer asynchronously, we need to make sure the WALProps is put into
2051    // walFile2Props before we call markClosedAndClean
2052    if (writer != null) {
2053      long oldFileLen = writer.getLength();
2054      logRollAndSetupWalProps(oldPath, newPath, oldFileLen);
2055      closeWriter(writer, oldPath);
2056    } else {
2057      logRollAndSetupWalProps(oldPath, newPath, 0);
2058    }
2059    this.writer = nextWriter;
2060    /**
2061     * Here is used for {@link AsyncFSWAL} and {@link FSHLog} to set the under layer filesystem
2062     * output after writer is replaced.
2063     */
2064    onWriterReplaced(nextWriter);
2065    this.fileLengthAtLastSync = nextWriter.getLength();
2066    this.highestProcessedAppendTxidAtLastSync = 0L;
2067    consumeLock.lock();
2068    try {
2069      consumerScheduled.set(true);
2070      int currentEpoch = epochAndState >>> 2;
2071      int nextEpoch = currentEpoch == MAX_EPOCH ? 0 : currentEpoch + 1;
2072      // set a new epoch and also clear waitingRoll and writerBroken
2073      this.epochAndState = nextEpoch << 2;
2074      // Reset rollRequested status
2075      rollRequested.set(false);
2076      consumeExecutor.execute(consumer);
2077    } finally {
2078      consumeLock.unlock();
2079    }
2080  }

2111        <= cursorBound; nextCursor++) {

2161  /** Returns true if we exceeded the slow sync roll threshold over the last check interval */
2162  protected boolean doCheckSlowSync() {
2163    boolean result = false;
2164    long now = EnvironmentEdgeManager.currentTime();
2165    long elapsedTime = now - lastTimeCheckSlowSync;
2166    if (elapsedTime >= slowSyncCheckInterval) {
2167      if (slowSyncCount.get() >= slowSyncRollThreshold) {
2168        if (elapsedTime >= (2 * slowSyncCheckInterval)) {
2169          // If two or more slowSyncCheckInterval have elapsed this is a corner case
2170          // where a train of slow syncs almost triggered us but then there was a long
2171          // interval from then until the one more that pushed us over. If so, we
2172          // should do nothing and let the count reset.
2173          if (LOG.isDebugEnabled()) {
2174            LOG.debug("checkSlowSync triggered but we decided to ignore it; " + "count="
2175              + slowSyncCount.get() + ", threshold=" + slowSyncRollThreshold + ", elapsedTime="
2176              + elapsedTime + " ms, slowSyncCheckInterval=" + slowSyncCheckInterval + " ms");
2177          }
2178          // Fall through to count reset below
2179        } else {
2180          LOG.warn("Requesting log roll because we exceeded slow sync threshold; count="
2181            + slowSyncCount.get() + ", threshold=" + slowSyncRollThreshold + ", current pipeline: "
2182            + Arrays.toString(getPipeline()));
2183          result = true;
2184        }
2185      }
2186      lastTimeCheckSlowSync = now;
2187      slowSyncCount.set(0);
2188    }
2189    return result;
2190  }
