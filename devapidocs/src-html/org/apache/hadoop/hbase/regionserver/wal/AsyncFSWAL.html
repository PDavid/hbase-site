047/**
048 * An asynchronous implementation of FSWAL.
049 * <p>
050 * Here 'waitingConsumePayloads' acts as the RingBuffer in FSHLog.
051 * <p>
052 * For append, we process it as follow:
053 * <ol>
054 * <li>In the caller thread(typically, in the rpc handler thread):
055 * <ol>
056 * <li>Insert the entry into 'waitingConsumePayloads'. Use ringbuffer sequence as txid.</li>
057 * <li>Schedule the consumer task if needed. See {@link #shouldScheduleConsumer()} for more details.
058 * </li>
059 * </ol>
060 * </li>
061 * <li>In the consumer task(executed in a single threaded thread pool)
062 * <ol>
063 * <li>Poll the entry from {@link #waitingConsumePayloads} and insert it into
064 * {@link #toWriteAppends}</li>
065 * <li>Poll the entry from {@link #toWriteAppends}, append it to the AsyncWriter, and insert it into
066 * {@link #unackedAppends}</li>
067 * <li>If the buffered size reaches {@link #batchSize}, or there is a sync request, then we call
068 * sync on the AsyncWriter.</li>
069 * <li>In the callback methods:
070 * <ul>
071 * <li>If succeeded, poll the entry from {@link #unackedAppends} and drop it.</li>
072 * <li>If failed, add all the entries in {@link #unackedAppends} back to {@link #toWriteAppends} and
073 * wait for writing them again.</li>
074 * </ul>
075 * </li>
076 * </ol>
077 * </li>
078 * </ol>
079 * For sync, the processing stages are almost same. And different from FSHLog, we will open a new
080 * writer and rewrite unacked entries to the new writer and sync again if we hit a sync error.
081 * <p>
082 * Here we only describe the logic of doReplaceWriter. The main logic of rollWriter is same with
083 * FSHLog.<br>
084 * For a normal roll request(for example, we have reached the log roll size):
085 * <ol>
086 * <li>In the log roller thread, we will set {@link #waitingRoll} to true and
087 * {@link #readyForRolling} to false, and then wait on {@link #readyForRolling}(see
088 * {@link #waitForSafePoint()}).</li>
089 * <li>In the consumer thread, we will stop polling entries from {@link #waitingConsumePayloads} if
090 * {@link #waitingRoll} is true, and also stop writing the entries in {@link #toWriteAppends} out.
091 * </li>
092 * <li>If there are unflush data in the writer, sync them.</li>
093 * <li>When all out-going sync request is finished, i.e, the {@link #unackedAppends} is empty,
094 * signal the {@link #readyForRollingCond}.</li>
095 * <li>Back to the log roller thread, now we can confirm that there are no out-going entries, i.e.,
096 * we reach a safe point. So it is safe to replace old writer with new writer now.</li>
097 * <li>Set {@link #writerBroken} and {@link #waitingRoll} to false.</li>
098 * <li>Schedule the consumer task.</li>
099 * <li>Schedule a background task to close the old writer.</li>
100 * </ol>
101 * For a broken writer roll request, the only difference is that we can bypass the wait for safe
102 * point stage.
103 */
104@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.CONFIG)
105public class AsyncFSWAL extends AbstractFSWAL<AsyncWriter> {

113    "hbase.wal.async.use-shared-event-loop";

128  public AsyncFSWAL(FileSystem fs, Path rootDir, String logDir, String archiveDir,
129    Configuration conf, List<WALActionsListener> listeners, boolean failIfWALExists, String prefix,
130    String suffix, EventLoopGroup eventLoopGroup, Class<? extends Channel> channelClass)
131    throws FailedLogCloseException, IOException {
132    this(fs, null, rootDir, logDir, archiveDir, conf, listeners, failIfWALExists, prefix, suffix,
133      eventLoopGroup, channelClass, StreamSlowMonitor.create(conf, "monitorForSuffix"));
134  }
135
136  public AsyncFSWAL(FileSystem fs, Abortable abortable, Path rootDir, String logDir,
137    String archiveDir, Configuration conf, List<WALActionsListener> listeners,
138    boolean failIfWALExists, String prefix, String suffix, EventLoopGroup eventLoopGroup,
139    Class<? extends Channel> channelClass, StreamSlowMonitor monitor)
140    throws FailedLogCloseException, IOException {
141    super(fs, abortable, rootDir, logDir, archiveDir, conf, listeners, failIfWALExists, prefix,
142      suffix);
143    this.eventLoopGroup = eventLoopGroup;
144    this.channelClass = channelClass;
145    this.streamSlowMonitor = monitor;
146    if (conf.getBoolean(ASYNC_WAL_USE_SHARED_EVENT_LOOP, DEFAULT_ASYNC_WAL_USE_SHARED_EVENT_LOOP)) {
147      this.consumeExecutor = eventLoopGroup.next();
148      this.shouldShutDownConsumeExecutorWhenClose = false;
149      if (consumeExecutor instanceof SingleThreadEventExecutor) {
150        try {
151          Field field = SingleThreadEventExecutor.class.getDeclaredField("taskQueue");
152          field.setAccessible(true);
153          Queue<?> queue = (Queue<?>) field.get(consumeExecutor);
154          this.hasConsumerTask = () -> queue.peek() == consumer;
155        } catch (Exception e) {
156          LOG.warn("Can not get task queue of " + consumeExecutor
157            + ", this is not necessary, just give up", e);
158          this.hasConsumerTask = () -> false;
159        }
160      } else {
161        this.hasConsumerTask = () -> false;
162      }
163    } else {
164      this.createSingleThreadPoolConsumeExecutor("AsyncFSWAL", rootDir, prefix);
165    }
166
167    this.setWaitOnShutdownInSeconds(conf.getInt(ASYNC_WAL_WAIT_ON_SHUTDOWN_IN_SECONDS,
168      DEFAULT_ASYNC_WAL_WAIT_ON_SHUTDOWN_IN_SECONDS), ASYNC_WAL_WAIT_ON_SHUTDOWN_IN_SECONDS);
169  }
170
171  @Override
172  protected CompletableFuture<Long> doWriterSync(AsyncWriter writer, boolean shouldUseHsync,
173    long txidWhenSyn) {
174    return writer.sync(shouldUseHsync);
175  }
176
177  private void drainNonMarkerEditsAndFailSyncs() {
178    if (toWriteAppends.isEmpty()) {
179      return;
180    }
181    boolean hasNonMarkerEdits = false;
182    Iterator<FSWALEntry> iter = toWriteAppends.descendingIterator();
183    while (iter.hasNext()) {
184      FSWALEntry entry = iter.next();
185      if (!entry.getEdit().isMetaEdit()) {
186        entry.release();
187        hasNonMarkerEdits = true;
188        break;
189      }
190    }
191    if (hasNonMarkerEdits) {
192      for (;;) {
193        iter.remove();
194        if (!iter.hasNext()) {
195          break;
196        }
197        iter.next().release();
198      }
199      for (FSWALEntry entry : unackedAppends) {
200        entry.release();
201      }
202      unackedAppends.clear();
203      // fail the sync futures which are under the txid of the first remaining edit, if none, fail
204      // all the sync futures.
205      long txid = toWriteAppends.isEmpty() ? Long.MAX_VALUE : toWriteAppends.peek().getTxid();
206      IOException error = new IOException("WAL is closing, only marker edit is allowed");
207      for (Iterator<SyncFuture> syncIter = syncFutures.iterator(); syncIter.hasNext();) {
208        SyncFuture future = syncIter.next();
209        if (future.getTxid() < txid) {
210          markFutureDoneAndOffer(future, future.getTxid(), error);
211          syncIter.remove();
212        } else {
213          break;
214        }
215      }
216    }
217  }
218
219  @Override
220  protected void preAppendAndSync() {
221    if (markerEditOnly()) {
222      drainNonMarkerEditsAndFailSyncs();
223    }
224  }
225
226  // This is used by sync replication, where we are going to close the wal soon after we reopen all
227  // the regions. Will be overridden by sub classes.
228  protected boolean markerEditOnly() {
229    return false;
230  }
231
232  @Override
233  protected void precheckBeforeAppendWALEdit(RegionInfo hri, WALKeyImpl key, WALEdit edits,
234    boolean inMemstore) throws IOException {
235    if (markerEditOnly() && !edits.isMetaEdit()) {
236      throw new IOException("WAL is closing, only marker edit is allowed");
237    }
238  }
239
240  protected final AsyncWriter createAsyncWriter(FileSystem fs, Path path) throws IOException {
241    return AsyncFSWALProvider.createAsyncWriter(conf, fs, path, false, this.blocksize,
242      eventLoopGroup, channelClass, streamSlowMonitor);
243  }
244
245  @Override
246  protected AsyncWriter createWriterInstance(Path path) throws IOException {
247    return createAsyncWriter(fs, path);
248  }
249
250  @Override
251  protected void onWriterReplaced(AsyncWriter nextWriter) {
252    if (nextWriter instanceof AsyncProtobufLogWriter) {
253      this.fsOut = ((AsyncProtobufLogWriter) nextWriter).getOutput();
254    }
255  }
256
257  @Override
258  protected void doAppend(AsyncWriter writer, FSWALEntry entry) {
259    writer.append(entry);
260  }
261
262  @Override
263  DatanodeInfo[] getPipeline() {
264    AsyncFSOutput output = this.fsOut;
265    return output != null ? output.getPipeline() : new DatanodeInfo[0];
266  }
267
268  @Override
269  int getLogReplication() {
270    return getPipeline().length;
271  }
272
273  @Override
274  protected boolean doCheckLogLowReplication() {
275    // not like FSHLog, AsyncFSOutput will fail immediately if there are errors writing to DNs, so
276    // typically there is no 'low replication' state, only a 'broken' state.
277    AsyncFSOutput output = this.fsOut;
278    return output != null && output.isBroken();
279  }
