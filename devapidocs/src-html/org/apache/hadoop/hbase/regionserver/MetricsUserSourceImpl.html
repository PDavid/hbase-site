019package org.apache.hadoop.hbase.regionserver;
020
021import java.util.concurrent.atomic.AtomicBoolean;
022
023import org.apache.hadoop.metrics2.MetricHistogram;
024import org.apache.hadoop.metrics2.MetricsRecordBuilder;
025import org.apache.hadoop.metrics2.lib.DynamicMetricsRegistry;
026import org.apache.yetus.audience.InterfaceAudience;
027import org.slf4j.Logger;
028import org.slf4j.LoggerFactory;
029
030@InterfaceAudience.Private
031public class MetricsUserSourceImpl implements MetricsUserSource {
032  private static final Logger LOG = LoggerFactory.getLogger(MetricsUserSourceImpl.class);
033
034  private final String userNamePrefix;
035
036  private final String user;
037
038  private final String userGetKey;
039  private final String userScanTimeKey;
040  private final String userPutKey;
041  private final String userDeleteKey;
042  private final String userIncrementKey;
043  private final String userAppendKey;
044  private final String userReplayKey;
045
046  private MetricHistogram getHisto;
047  private MetricHistogram scanTimeHisto;
048  private MetricHistogram putHisto;
049  private MetricHistogram deleteHisto;
050  private MetricHistogram incrementHisto;
051  private MetricHistogram appendHisto;
052  private MetricHistogram replayHisto;
053
054  private final int hashCode;
055
056  private AtomicBoolean closed = new AtomicBoolean(false);
057  private final MetricsUserAggregateSourceImpl agg;
058  private final DynamicMetricsRegistry registry;
059
060  public MetricsUserSourceImpl(String user, MetricsUserAggregateSourceImpl agg) {
061    if (LOG.isDebugEnabled()) {
062      LOG.debug("Creating new MetricsUserSourceImpl for user " + user);
063    }
064
065    this.user = user;
066    this.agg = agg;
067    this.registry = agg.getMetricsRegistry();
068
069    this.userNamePrefix = "user_" + user + "_metric_";
070
071    hashCode = userNamePrefix.hashCode();
072
073    userGetKey = userNamePrefix + MetricsRegionServerSource.GET_KEY;
074    userScanTimeKey = userNamePrefix + MetricsRegionServerSource.SCAN_TIME_KEY;
075    userPutKey = userNamePrefix + MetricsRegionServerSource.PUT_KEY;
076    userDeleteKey = userNamePrefix + MetricsRegionServerSource.DELETE_KEY;
077    userIncrementKey = userNamePrefix + MetricsRegionServerSource.INCREMENT_KEY;
078    userAppendKey = userNamePrefix + MetricsRegionServerSource.APPEND_KEY;
079    userReplayKey = userNamePrefix + MetricsRegionServerSource.REPLAY_KEY;
080
081    agg.register(this);
082  }
083
084  @Override
085  public void register() {
086    synchronized (this) {
087      getHisto = registry.newTimeHistogram(userGetKey);
088      scanTimeHisto = registry.newTimeHistogram(userScanTimeKey);
089      putHisto = registry.newTimeHistogram(userPutKey);
090      deleteHisto = registry.newTimeHistogram(userDeleteKey);
091      incrementHisto = registry.newTimeHistogram(userIncrementKey);
092      appendHisto = registry.newTimeHistogram(userAppendKey);
093      replayHisto = registry.newTimeHistogram(userReplayKey);
094    }
095  }
096
097  @Override
098  public void deregister() {
099    boolean wasClosed = closed.getAndSet(true);
100
101    // Has someone else already closed this for us?
102    if (wasClosed) {
103      return;
104    }
105
106    if (LOG.isDebugEnabled()) {
107      LOG.debug("Removing user Metrics for user: " + user);
108    }
109
110    synchronized (this) {
111      registry.removeMetric(userGetKey);
112      registry.removeMetric(userScanTimeKey);
113      registry.removeMetric(userPutKey);
114      registry.removeMetric(userDeleteKey);
115      registry.removeMetric(userIncrementKey);
116      registry.removeMetric(userAppendKey);
117      registry.removeMetric(userReplayKey);
118    }
119  }
120
121  @Override
122  public String getUser() {
123    return user;
124  }
125
126  @Override
127  public int compareTo(MetricsUserSource source) {
128    if (source == null) {
129      return -1;
130    }
131    if (!(source instanceof MetricsUserSourceImpl)) {
132      return -1;
133    }
134
135    MetricsUserSourceImpl impl = (MetricsUserSourceImpl) source;
136
137    return Long.compare(hashCode, impl.hashCode);
138  }
139
140  @Override
141  public int hashCode() {
142    return hashCode;
143  }
144
145  @Override
146  public boolean equals(Object obj) {
147    return obj == this ||
148        (obj instanceof MetricsUserSourceImpl && compareTo((MetricsUserSourceImpl) obj) == 0);
149  }
150
151  void snapshot(MetricsRecordBuilder mrb, boolean ignored) {
152    // If there is a close that started be double extra sure
153    // that we're not getting any locks and not putting data
154    // into the metrics that should be removed. So early out
155    // before even getting the lock.
156    if (closed.get()) {
157      return;
158    }
159
160    // Grab the read
161    // This ensures that removes of the metrics
162    // can't happen while we are putting them back in.
163    synchronized (this) {
164
165      // It's possible that a close happened between checking
166      // the closed variable and getting the lock.
167      if (closed.get()) {
168        return;
169      }
170    }
171  }
172
173  @Override
174  public void updatePut(long t) {
175    putHisto.add(t);
176  }
177
178  @Override
179  public void updateDelete(long t) {
180    deleteHisto.add(t);
181  }
182
183  @Override
184  public void updateGet(long t) {
185    getHisto.add(t);
186  }
187
188  @Override
189  public void updateIncrement(long t) {
190    incrementHisto.add(t);
191  }
192
193  @Override
194  public void updateAppend(long t) {
195    appendHisto.add(t);
196  }
197
198  @Override
199  public void updateReplay(long t) {
200    replayHisto.add(t);
201  }
202
203  @Override
204  public void updateScanTime(long t) {
205    scanTimeHisto.add(t);
206  }
207}
