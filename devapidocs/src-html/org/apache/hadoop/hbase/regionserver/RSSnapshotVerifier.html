044import org.apache.hadoop.hbase.shaded.protobuf.generated.SnapshotProtos.SnapshotRegionManifest.StoreFile;

046/**
047 *  Used by {@link org.apache.hadoop.hbase.master.procedure.SnapshotVerifyProcedure} to verify
048 *  if the region info and store file info in RegionManifest are intact.
049 */
050@InterfaceAudience.Private
051public class RSSnapshotVerifier {
052  private static final Logger LOG = LoggerFactory.getLogger(RSSnapshotVerifier.class);
053
054  private final LoadingCache<SnapshotDescription,
055    Pair<FileSystem, Map<String, SnapshotRegionManifest>>> SNAPSHOT_MANIFEST_CACHE;
056  private final Configuration conf;
057
058  public RSSnapshotVerifier(Configuration conf) {
059    this.conf = conf;
060    long expiredTime = conf.getLong("hbase.snapshot-manifest.cache.expired.sec", 600);
061    long maxSize = conf.getLong("hbase.snapshot-manifest.cache.max.size", 10);
062    this.SNAPSHOT_MANIFEST_CACHE = CacheBuilder.newBuilder()
063      .expireAfterAccess(expiredTime, TimeUnit.SECONDS).maximumSize(maxSize)
064      .build(new SnapshotManifestCacheLoader(conf));
065  }
066
067  public void verifyRegion(SnapshotDescription snapshot, RegionInfo region) throws IOException {
068    try {
069      Pair<FileSystem, Map<String, SnapshotRegionManifest>>
070        cache = SNAPSHOT_MANIFEST_CACHE.get(snapshot);
071      Map<String, SnapshotRegionManifest> rmMap = cache.getSecond();
072      if (rmMap == null) {
073        throw new CorruptedSnapshotException(snapshot.getName() + "looks empty");
074      }
075      SnapshotRegionManifest regionManifest = rmMap.get(region.getEncodedName());
076      if (regionManifest == null) {
077        LOG.warn("No snapshot region directory found for {}", region.getRegionNameAsString());
078        return;
079      }
080      // verify region info
081      RegionInfo manifestRegionInfo = ProtobufUtil.toRegionInfo(regionManifest.getRegionInfo());
082      if (RegionInfo.COMPARATOR.compare(region, manifestRegionInfo) != 0) {
083        String msg =
084          "Manifest region info " + manifestRegionInfo + "doesn't match expected region:" + region;
085        throw new CorruptedSnapshotException(msg, ProtobufUtil.createSnapshotDesc(snapshot));
086      }
087      // verify store file
088      SnapshotReferenceUtil.visitRegionStoreFiles(regionManifest, new StoreFileVisitor() {
089        @Override public void storeFile(RegionInfo region, String familyName, StoreFile storeFile)
090            throws IOException {
091          SnapshotReferenceUtil.verifyStoreFile(conf, cache.getFirst(),
092            /* snapshotDir= */ null, // snapshotDir is never used, so it's ok to pass null here.
093                                     // maybe we can remove this parameter later.
094            snapshot, region, familyName, storeFile);
095        }
096      });
097    } catch (ExecutionException e) {
098      if (e.getCause() instanceof CorruptedSnapshotException) {
099        throw new CorruptedSnapshotException(e.getCause().getMessage(),
100          ProtobufUtil.createSnapshotDesc(snapshot));
101      } else {
102        LOG.error("Failed loading snapshot manifest for {} from filesystem",
103          snapshot.getName(), e.getCause());
104        throw new IOException(e.getCause());
105      }
106    }
107  }
108
109  // to avoid loading snapshot manifest from filesystem for each region, try to cache it here
110  private static final class SnapshotManifestCacheLoader extends
111      CacheLoader<SnapshotDescription, Pair<FileSystem, Map<String, SnapshotRegionManifest>>> {
112    private final Configuration conf;
113
114    private SnapshotManifestCacheLoader(Configuration conf) {
115      this.conf = conf;
116    }
117
118    @Override
119    public Pair<FileSystem, Map<String, SnapshotRegionManifest>>
120        load(SnapshotDescription snapshot) throws Exception {
121      Path rootDir = CommonFSUtils.getRootDir(conf);
122      Path workingDir = SnapshotDescriptionUtils.getWorkingSnapshotDir(snapshot, rootDir, conf);
123      FileSystem rootFs = CommonFSUtils.getRootDirFileSystem(conf);
124      FileSystem workingDirFS = workingDir.getFileSystem(conf);
125      SnapshotManifest manifest = SnapshotManifest.open(conf, workingDirFS, workingDir, snapshot);
126      LOG.debug("loading snapshot manifest for {} from {}", snapshot.getName(), workingDir);
127      return Pair.newPair(rootFs, manifest.getRegionManifestsMap());
128    }
129  }
