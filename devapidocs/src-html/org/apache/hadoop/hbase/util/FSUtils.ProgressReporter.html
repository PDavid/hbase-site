131  /**
132   * Compare path component of the Path URI; e.g. if hdfs://a/b/c and /a/b/c, it will compare the
133   * '/a/b/c' part. If you passed in 'hdfs://a/b/c and b/c, it would return true.  Does not consider
134   * schema; i.e. if schemas different but path or subpath matches, the two will equate.
135   * @param pathToSearch Path we will be trying to match.
136   * @param pathTail
137   * @return True if <code>pathTail</code> is tail on the path of <code>pathToSearch</code>
138   */
139  public static boolean isMatchingTail(final Path pathToSearch, final Path pathTail) {
140    Path tailPath = pathTail;
141    String tailName;
142    Path toSearch = pathToSearch;
143    String toSearchName;
144    boolean result = false;
145
146    if (pathToSearch.depth() != pathTail.depth()) {
147      return false;
148    }
149
150    do {
151      tailName = tailPath.getName();
152      if (tailName == null || tailName.isEmpty()) {
153        result = true;
154        break;
155      }
156      toSearchName = toSearch.getName();
157      if (toSearchName == null || toSearchName.isEmpty()) {
158        break;
159      }
160      // Move up a parent on each path for next go around.  Path doesn't let us go off the end.
161      tailPath = tailPath.getParent();
162      toSearch = toSearch.getParent();
163    } while(tailName.equals(toSearchName));
164    return result;
165  }

180 /**
181   * Create the specified file on the filesystem. By default, this will:
182   * <ol>
183   * <li>overwrite the file if it exists</li>
184   * <li>apply the umask in the configuration (if it is enabled)</li>
185   * <li>use the fs configured buffer size (or 4096 if not set)</li>
186   * <li>use the configured column family replication or default replication if
187   * {@link ColumnFamilyDescriptorBuilder#DEFAULT_DFS_REPLICATION}</li>
188   * <li>use the default block size</li>
189   * <li>not track progress</li>
190   * </ol>
191   * @param conf configurations
192   * @param fs {@link FileSystem} on which to write the file
193   * @param path {@link Path} to the file to write
194   * @param perm permissions
195   * @param favoredNodes favored data nodes
196   * @return output stream to the created file
197   * @throws IOException if the file cannot be created
198   */
199  public static FSDataOutputStream create(Configuration conf, FileSystem fs, Path path,
200    FsPermission perm, InetSocketAddress[] favoredNodes) throws IOException {
201    if (fs instanceof HFileSystem) {
202      FileSystem backingFs = ((HFileSystem) fs).getBackingFs();
203      if (backingFs instanceof DistributedFileSystem) {
204        // Try to use the favoredNodes version via reflection to allow backwards-
205        // compatibility.
206        short replication = Short.parseShort(conf.get(ColumnFamilyDescriptorBuilder.DFS_REPLICATION,
207          String.valueOf(ColumnFamilyDescriptorBuilder.DEFAULT_DFS_REPLICATION)));
208        try {
209          return (FSDataOutputStream) (DistributedFileSystem.class
210            .getDeclaredMethod("create", Path.class, FsPermission.class, boolean.class, int.class,
211              short.class, long.class, Progressable.class, InetSocketAddress[].class)
212            .invoke(backingFs, path, perm, true, CommonFSUtils.getDefaultBufferSize(backingFs),
213              replication > 0 ? replication : CommonFSUtils.getDefaultReplication(backingFs, path),
214              CommonFSUtils.getDefaultBlockSize(backingFs, path), null, favoredNodes));
215        } catch (InvocationTargetException ite) {
216          // Function was properly called, but threw it's own exception.
217          throw new IOException(ite.getCause());
218        } catch (NoSuchMethodException e) {
219          LOG.debug("DFS Client does not support most favored nodes create; using default create");
220          LOG.trace("Ignoring; use default create", e);
221        } catch (IllegalArgumentException | SecurityException | IllegalAccessException e) {
222          LOG.debug("Ignoring (most likely Reflection related exception) " + e);
223        }
224      }
225    }
226    return CommonFSUtils.create(fs, path, perm, true);
227  }

259   * We use reflection because {@link DistributedFileSystem#setSafeMode(
260   * HdfsConstants.SafeModeAction action, boolean isChecked)} is not in hadoop 1.1
261   *
262   * @param dfs
263   * @return whether we're in safe mode
264   * @throws IOException
265   */
266  private static boolean isInSafeMode(DistributedFileSystem dfs) throws IOException {
267    boolean inSafeMode = false;
268    try {
269      Method m = DistributedFileSystem.class.getMethod("setSafeMode", new Class<?> []{
270          org.apache.hadoop.hdfs.protocol.HdfsConstants.SafeModeAction.class, boolean.class});
271      inSafeMode = (Boolean) m.invoke(dfs,
272        org.apache.hadoop.hdfs.protocol.HdfsConstants.SafeModeAction.SAFEMODE_GET, true);
273    } catch (Exception e) {
274      if (e instanceof IOException) throw (IOException) e;
275
276      // Check whether dfs is on safemode.
277      inSafeMode = dfs.setSafeMode(
278        org.apache.hadoop.hdfs.protocol.HdfsConstants.SafeModeAction.SAFEMODE_GET);
279    }
280    return inSafeMode;
281  }

301  /**
302   * Verifies current version of file system
303   *
304   * @param fs filesystem object
305   * @param rootdir root hbase directory
306   * @return null if no version file exists, version string otherwise
307   * @throws IOException if the version file fails to open
308   * @throws DeserializationException if the version data cannot be translated into a version
309   */
310  public static String getVersion(FileSystem fs, Path rootdir)
311  throws IOException, DeserializationException {
312    final Path versionFile = new Path(rootdir, HConstants.VERSION_FILE_NAME);
313    FileStatus[] status = null;
314    try {
315      // hadoop 2.0 throws FNFE if directory does not exist.
316      // hadoop 1.0 returns null if directory does not exist.
317      status = fs.listStatus(versionFile);
318    } catch (FileNotFoundException fnfe) {
319      return null;
320    }
321    if (ArrayUtils.getLength(status) == 0) {
322      return null;
323    }
324    String version = null;
325    byte [] content = new byte [(int)status[0].getLen()];
326    FSDataInputStream s = fs.open(versionFile);
327    try {
328      IOUtils.readFully(s, content, 0, content.length);
329      if (ProtobufUtil.isPBMagicPrefix(content)) {
330        version = parseVersionFrom(content);
331      } else {
332        // Presume it pre-pb format.
333        try (DataInputStream dis = new DataInputStream(new ByteArrayInputStream(content))) {
334          version = dis.readUTF();
335        }
336      }
337    } catch (EOFException eof) {
338      LOG.warn("Version file was empty, odd, will try to set it.");
339    } finally {
340      s.close();
341    }
342    return version;
343  }

345  /**
346   * Parse the content of the ${HBASE_ROOTDIR}/hbase.version file.
347   * @param bytes The byte content of the hbase.version file
348   * @return The version found in the file as a String
349   * @throws DeserializationException if the version data cannot be translated into a version
350   */
351  static String parseVersionFrom(final byte [] bytes)
352  throws DeserializationException {
353    ProtobufUtil.expectPBMagicPrefix(bytes);
354    int pblen = ProtobufUtil.lengthOfPBMagic();
355    FSProtos.HBaseVersionFileContent.Builder builder =
356      FSProtos.HBaseVersionFileContent.newBuilder();
357    try {
358      ProtobufUtil.mergeFrom(builder, bytes, pblen, bytes.length - pblen);
359      return builder.getVersion();
360    } catch (IOException e) {
361      // Convert
362      throw new DeserializationException(e);
363    }
364  }

391  /**
392   * Verifies current version of file system
393   *
394   * @param fs file system
395   * @param rootdir root directory of HBase installation
396   * @param message if true, issues a message on System.out
397   * @param wait wait interval
398   * @param retries number of times to retry
399   *
400   * @throws IOException if the version file cannot be opened
401   * @throws DeserializationException if the contents of the version file cannot be parsed
402   */
403  public static void checkVersion(FileSystem fs, Path rootdir,
404      boolean message, int wait, int retries)
405  throws IOException, DeserializationException {
406    String version = getVersion(fs, rootdir);
407    String msg;
408    if (version == null) {
409      if (!metaRegionExists(fs, rootdir)) {
410        // rootDir is empty (no version file and no root region)
411        // just create new version file (HBASE-1195)
412        setVersion(fs, rootdir, wait, retries);
413        return;
414      } else {
415        msg = "hbase.version file is missing. Is your hbase.rootdir valid? " +
416            "You can restore hbase.version file by running 'HBCK2 filesystem -fix'. " +
417            "See https://github.com/apache/hbase-operator-tools/tree/master/hbase-hbck2";
418      }
419    } else if (version.compareTo(HConstants.FILE_SYSTEM_VERSION) == 0) {
420      return;
421    } else {
422      msg = "HBase file layout needs to be upgraded. Current filesystem version is " + version +
423          " but software requires version " + HConstants.FILE_SYSTEM_VERSION +
424          ". Consult http://hbase.apache.org/book.html for further information about " +
425          "upgrading HBase.";
426    }
427
428    // version is deprecated require migration
429    // Output on stdout so user sees it in terminal.
430    if (message) {
431      System.out.println("WARNING! " + msg);
432    }
433    throw new FileSystemVersionException(msg);
434  }

464  /**
465   * Sets version of file system
466   *
467   * @param fs filesystem object
468   * @param rootdir hbase root directory
469   * @param version version to set
470   * @param wait time to wait for retry
471   * @param retries number of times to retry before throwing an IOException
472   * @throws IOException e
473   */
474  public static void setVersion(FileSystem fs, Path rootdir, String version,
475      int wait, int retries) throws IOException {
476    Path versionFile = new Path(rootdir, HConstants.VERSION_FILE_NAME);
477    Path tempVersionFile = new Path(rootdir, HConstants.HBASE_TEMP_DIRECTORY + Path.SEPARATOR +
478      HConstants.VERSION_FILE_NAME);
479    while (true) {
480      try {
481        // Write the version to a temporary file
482        FSDataOutputStream s = fs.create(tempVersionFile);
483        try {
484          s.write(toVersionByteArray(version));
485          s.close();
486          s = null;
487          // Move the temp version file to its normal location. Returns false
488          // if the rename failed. Throw an IOE in that case.
489          if (!fs.rename(tempVersionFile, versionFile)) {
490            throw new IOException("Unable to move temp version file to " + versionFile);
491          }
492        } finally {
493          // Cleaning up the temporary if the rename failed would be trying
494          // too hard. We'll unconditionally create it again the next time
495          // through anyway, files are overwritten by default by create().
496
497          // Attempt to close the stream on the way out if it is still open.
498          try {
499            if (s != null) s.close();
500          } catch (IOException ignore) { }
501        }
502        LOG.info("Created version file at " + rootdir.toString() + " with version=" + version);
503        return;
504      } catch (IOException e) {
505        if (retries > 0) {
506          LOG.debug("Unable to create version file at " + rootdir.toString() + ", retrying", e);
507          fs.delete(versionFile, false);
508          try {
509            if (wait > 0) {
510              Thread.sleep(wait);
511            }
512          } catch (InterruptedException ie) {
513            throw (InterruptedIOException)new InterruptedIOException().initCause(ie);
514          }
515          retries--;
516        } else {
517          throw e;
518        }
519      }
520    }
521  }

553  /**
554   * Returns the value of the unique cluster ID stored for this HBase instance.
555   * @param fs the root directory FileSystem
556   * @param rootdir the path to the HBase root directory
557   * @return the unique cluster identifier
558   * @throws IOException if reading the cluster ID file fails
559   */
560  public static ClusterId getClusterId(FileSystem fs, Path rootdir)
561  throws IOException {
562    Path idPath = new Path(rootdir, HConstants.CLUSTER_ID_FILE_NAME);
563    ClusterId clusterId = null;
564    FileStatus status = fs.exists(idPath)? fs.getFileStatus(idPath):  null;
565    if (status != null) {
566      int len = Ints.checkedCast(status.getLen());
567      byte [] content = new byte[len];
568      FSDataInputStream in = fs.open(idPath);
569      try {
570        in.readFully(content);
571      } catch (EOFException eof) {
572        LOG.warn("Cluster ID file {} is empty", idPath);
573      } finally{
574        in.close();
575      }
576      try {
577        clusterId = ClusterId.parseFrom(content);
578      } catch (DeserializationException e) {
579        throw new IOException("content=" + Bytes.toString(content), e);
580      }
581      // If not pb'd, make it so.
582      if (!ProtobufUtil.isPBMagicPrefix(content)) {
583        String cid = null;
584        in = fs.open(idPath);
585        try {
586          cid = in.readUTF();
587          clusterId = new ClusterId(cid);
588        } catch (EOFException eof) {
589          LOG.warn("Cluster ID file {} is empty", idPath);
590        } finally {
591          in.close();
592        }
593        rewriteAsPb(fs, rootdir, idPath, clusterId);
594      }
595      return clusterId;
596    } else {
597      LOG.warn("Cluster ID file does not exist at {}", idPath);
598    }
599    return clusterId;
600  }

602  /**
603   * @param cid
604   * @throws IOException
605   */
606  private static void rewriteAsPb(final FileSystem fs, final Path rootdir, final Path p,
607      final ClusterId cid)
608  throws IOException {
609    // Rewrite the file as pb.  Move aside the old one first, write new
610    // then delete the moved-aside file.
611    Path movedAsideName = new Path(p + "." + System.currentTimeMillis());
612    if (!fs.rename(p, movedAsideName)) throw new IOException("Failed rename of " + p);
613    setClusterId(fs, rootdir, cid, 100);
614    if (!fs.delete(movedAsideName, false)) {
615      throw new IOException("Failed delete of " + movedAsideName);
616    }
617    LOG.debug("Rewrote the hbase.id file as pb");
618  }

620  /**
621   * Writes a new unique identifier for this cluster to the "hbase.id" file in the HBase root
622   * directory. If any operations on the ID file fails, and {@code wait} is a positive value, the
623   * method will retry to produce the ID file until the thread is forcibly interrupted.
624   *
625   * @param fs the root directory FileSystem
626   * @param rootdir the path to the HBase root directory
627   * @param clusterId the unique identifier to store
628   * @param wait how long (in milliseconds) to wait between retries
629   * @throws IOException if writing to the FileSystem fails and no wait value
630   */
631  public static void setClusterId(final FileSystem fs, final Path rootdir,
632      final ClusterId clusterId, final long wait) throws IOException {
633
634    final Path idFile = new Path(rootdir, HConstants.CLUSTER_ID_FILE_NAME);
635    final Path tempDir = new Path(rootdir, HConstants.HBASE_TEMP_DIRECTORY);
636    final Path tempIdFile = new Path(tempDir, HConstants.CLUSTER_ID_FILE_NAME);
637
638    LOG.debug("Create cluster ID file [{}] with ID: {}", idFile, clusterId);
639
640    while (true) {
641      Optional<IOException> failure = Optional.empty();
642
643      LOG.debug("Write the cluster ID file to a temporary location: {}", tempIdFile);
644      try (FSDataOutputStream s = fs.create(tempIdFile)) {
645        s.write(clusterId.toByteArray());
646      } catch (IOException ioe) {
647        failure = Optional.of(ioe);
648      }
649
650      if (!failure.isPresent()) {
651        try {
652          LOG.debug("Move the temporary cluster ID file to its target location [{}]:[{}]",
653            tempIdFile, idFile);
654
655          if (!fs.rename(tempIdFile, idFile)) {
656            failure =
657                Optional.of(new IOException("Unable to move temp cluster ID file to " + idFile));
658          }
659        } catch (IOException ioe) {
660          failure = Optional.of(ioe);
661        }
662      }
663
664      if (failure.isPresent()) {
665        final IOException cause = failure.get();
666        if (wait > 0L) {
667          LOG.warn("Unable to create cluster ID file in {}, retrying in {}ms", rootdir, wait,
668            cause);
669          try {
670            Thread.sleep(wait);
671          } catch (InterruptedException e) {
672            Thread.currentThread().interrupt();
673            throw (InterruptedIOException) new InterruptedIOException().initCause(e);
674          }
675          continue;
676        } else {
677          throw cause;
678        }
679      } else {
680        return;
681      }
682    }
683  }

791  /**
792   * Runs through the HBase rootdir and checks how many stores for each table
793   * have more than one file in them. Checks -ROOT- and hbase:meta too. The total
794   * percentage across all tables is stored under the special key "-TOTAL-".
795   *
796   * @param fs  The file system to use
797   * @param hbaseRootDir  The root directory to scan
798   * @return A map for each table and its percentage (never null)
799   * @throws IOException When scanning the directory fails
800   */
801  public static Map<String, Integer> getTableFragmentation(
802    final FileSystem fs, final Path hbaseRootDir)
803  throws IOException {
804    Map<String, Integer> frags = new HashMap<>();
805    int cfCountTotal = 0;
806    int cfFragTotal = 0;
807    PathFilter regionFilter = new RegionDirFilter(fs);
808    PathFilter familyFilter = new FamilyDirFilter(fs);
809    List<Path> tableDirs = getTableDirs(fs, hbaseRootDir);
810    for (Path d : tableDirs) {
811      int cfCount = 0;
812      int cfFrag = 0;
813      FileStatus[] regionDirs = fs.listStatus(d, regionFilter);
814      for (FileStatus regionDir : regionDirs) {
815        Path dd = regionDir.getPath();
816        // else its a region name, now look in region for families
817        FileStatus[] familyDirs = fs.listStatus(dd, familyFilter);
818        for (FileStatus familyDir : familyDirs) {
819          cfCount++;
820          cfCountTotal++;
821          Path family = familyDir.getPath();
822          // now in family make sure only one file
823          FileStatus[] familyStatus = fs.listStatus(family);
824          if (familyStatus.length > 1) {
825            cfFrag++;
826            cfFragTotal++;
827          }
828        }
829      }
830      // compute percentage per table and store in result list
831      frags.put(CommonFSUtils.getTableName(d).getNameAsString(),
832        cfCount == 0? 0: Math.round((float) cfFrag / cfCount * 100));
833    }
834    // set overall percentage for all tables
835    frags.put("-TOTAL-",
836      cfCountTotal == 0? 0: Math.round((float) cfFragTotal / cfCountTotal * 100));
837    return frags;
838  }

1275      List<FileStatus> regionDirs = FSUtils.listStatusWithStatusFilter(fs, tableDir, new RegionDirFilter(fs));
1276      if (regionDirs == null) {
1277        return finalResultMap;
1278      }
1279
1280      final List<Future<?>> futures = new ArrayList<>(regionDirs.size());
1281
1282      for (FileStatus regionDir : regionDirs) {
1283        if (null != progressReporter) {
1284          progressReporter.progress(regionDir);
1285        }
1286        final Path dd = regionDir.getPath();
1287
1288        if (!exceptions.isEmpty()) {
1289          break;
1290        }
1291
1292        Runnable getRegionStoreFileMapCall = new Runnable() {
1293          @Override
1294          public void run() {
1295            try {
1296              HashMap<String,Path> regionStoreFileMap = new HashMap<>();
1297              List<FileStatus> familyDirs = FSUtils.listStatusWithStatusFilter(fs, dd, familyFilter);
1298              if (familyDirs == null) {
1299                if (!fs.exists(dd)) {
1300                  LOG.warn("Skipping region because it no longer exists: " + dd);
1301                } else {
1302                  LOG.warn("Skipping region because it has no family dirs: " + dd);
1303                }
1304                return;
1305              }
1306              for (FileStatus familyDir : familyDirs) {
1307                if (null != progressReporter) {
1308                  progressReporter.progress(familyDir);
1309                }
1310                Path family = familyDir.getPath();
1311                if (family.getName().equals(HConstants.RECOVERED_EDITS_DIR)) {
1312                  continue;
1313                }
1314                // now in family, iterate over the StoreFiles and
1315                // put in map
1316                FileStatus[] familyStatus = fs.listStatus(family);
1317                for (FileStatus sfStatus : familyStatus) {
1318                  if (null != progressReporter) {
1319                    progressReporter.progress(sfStatus);
1320                  }
1321                  Path sf = sfStatus.getPath();
1322                  if (sfFilter == null || sfFilter.accept(sf)) {
1323                    regionStoreFileMap.put( sf.getName(), sf);
1324                  }
1325                }
1326              }
1327              finalResultMap.putAll(regionStoreFileMap);
1328            } catch (Exception e) {
1329              LOG.error("Could not get region store file map for region: " + dd, e);
1330              exceptions.add(e);
1331            }
1332          }
1333        };
1334
1335        // If executor is available, submit async tasks to exec concurrently, otherwise
1336        // just do serial sync execution
1337        if (executor != null) {
1338          Future<?> future = executor.submit(getRegionStoreFileMapCall);
1339          futures.add(future);
1340        } else {
1341          FutureTask<?> future = new FutureTask<>(getRegionStoreFileMapCall, null);
1342          future.run();
1343          futures.add(future);
1344        }
1345      }
1346
1347      // Ensure all pending tasks are complete (or that we run into an exception)
1348      for (Future<?> f : futures) {
1349        if (!exceptions.isEmpty()) {
1350          break;
1351        }
1352        try {
1353          f.get();
1354        } catch (ExecutionException e) {
1355          LOG.error("Unexpected exec exception!  Should've been caught already.  (Bug?)", e);
1356          // Shouldn't happen, we already logged/caught any exceptions in the Runnable
1357        }
1358      }
1359    } catch (IOException e) {
1360      LOG.error("Cannot execute getTableStoreFilePathMap for " + tableName, e);
1361      exceptions.add(e);
1362    } finally {
1363      if (!exceptions.isEmpty()) {
1364        // Just throw the first exception as an indication something bad happened
1365        // Don't need to propagate all the exceptions, we already logged them all anyway
1366        Throwables.propagateIfPossible(exceptions.firstElement(), IOException.class);
1367        throw new IOException(exceptions.firstElement());
1368      }
1369    }
1370
1371    return finalResultMap;
1372  }

1589  /**
1590   * This function is to scan the root path of the file system to get either the
1591   * mapping between the region name and its best locality region server or the
1592   * degree of locality of each region on each of the servers having at least
1593   * one block of that region. The output map parameters are both optional.
1594   *
1595   * @param conf
1596   *          the configuration to use
1597   * @param desiredTable
1598   *          the table you wish to scan locality for
1599   * @param threadPoolSize
1600   *          the thread pool size to use
1601   * @param regionDegreeLocalityMapping
1602   *          the map into which to put the locality degree mapping or null,
1603   *          must be a thread-safe implementation
1604   * @throws IOException
1605   *           in case of file system errors or interrupts
1606   */
1607  private static void getRegionLocalityMappingFromFS(final Configuration conf,
1608    final String desiredTable, int threadPoolSize,
1609    final Map<String, Map<String, Float>> regionDegreeLocalityMapping) throws IOException {
1610    final FileSystem fs = FileSystem.get(conf);
1611    final Path rootPath = CommonFSUtils.getRootDir(conf);
1612    final long startTime = EnvironmentEdgeManager.currentTime();
1613    final Path queryPath;
1614    // The table files are in ${hbase.rootdir}/data/<namespace>/<table>/*
1615    if (null == desiredTable) {
1616      queryPath =
1617        new Path(new Path(rootPath, HConstants.BASE_NAMESPACE_DIR).toString() + "/*/*/*/");
1618    } else {
1619      queryPath = new Path(
1620        CommonFSUtils.getTableDir(rootPath, TableName.valueOf(desiredTable)).toString() + "/*/");
1621    }
1622
1623    // reject all paths that are not appropriate
1624    PathFilter pathFilter = new PathFilter() {
1625      @Override
1626      public boolean accept(Path path) {
1627        // this is the region name; it may get some noise data
1628        if (null == path) {
1629          return false;
1630        }
1631
1632        // no parent?
1633        Path parent = path.getParent();
1634        if (null == parent) {
1635          return false;
1636        }
1637
1638        String regionName = path.getName();
1639        if (null == regionName) {
1640          return false;
1641        }
1642
1643        if (!regionName.toLowerCase(Locale.ROOT).matches("[0-9a-f]+")) {
1644          return false;
1645        }
1646        return true;
1647      }
1648    };
1649
1650    FileStatus[] statusList = fs.globStatus(queryPath, pathFilter);
1651
1652    if (LOG.isDebugEnabled()) {
1653      LOG.debug("Query Path: {} ; # list of files: {}", queryPath, Arrays.toString(statusList));
1654    }
1655
1656    if (null == statusList) {
1657      return;
1658    }
1659
1660    // lower the number of threads in case we have very few expected regions
1661    threadPoolSize = Math.min(threadPoolSize, statusList.length);
1662
1663    // run in multiple threads
1664    final ExecutorService tpe = Executors.newFixedThreadPool(threadPoolSize,
1665      Threads.newDaemonThreadFactory("FSRegionQuery"));
1666    try {
1667      // ignore all file status items that are not of interest
1668      for (FileStatus regionStatus : statusList) {
1669        if (null == regionStatus || !regionStatus.isDirectory()) {
1670          continue;
1671        }
1672
1673        final Path regionPath = regionStatus.getPath();
1674        if (null != regionPath) {
1675          tpe.execute(new FSRegionScanner(fs, regionPath, null, regionDegreeLocalityMapping));
1676        }
1677      }
1678    } finally {
1679      tpe.shutdown();
1680      final long threadWakeFrequency = (long) conf.getInt(HConstants.THREAD_WAKE_FREQUENCY,
1681        HConstants.DEFAULT_THREAD_WAKE_FREQUENCY);
1682      try {
1683        // here we wait until TPE terminates, which is either naturally or by
1684        // exceptions in the execution of the threads
1685        while (!tpe.awaitTermination(threadWakeFrequency,
1686            TimeUnit.MILLISECONDS)) {
1687          // printing out rough estimate, so as to not introduce
1688          // AtomicInteger
1689          LOG.info("Locality checking is underway: { Scanned Regions : "
1690              + ((ThreadPoolExecutor) tpe).getCompletedTaskCount() + "/"
1691              + ((ThreadPoolExecutor) tpe).getTaskCount() + " }");
1692        }
1693      } catch (InterruptedException e) {
1694        Thread.currentThread().interrupt();
1695        throw (InterruptedIOException) new InterruptedIOException().initCause(e);
1696      }
1697    }
1698
1699    long overhead = EnvironmentEdgeManager.currentTime() - startTime;
1700    LOG.info("Scan DFS for locality info takes {}ms", overhead);
1701  }

1866      return false;
1867    }
1868    if (srcServiceName.equals(desServiceName)) {
1869      return true;
1870    }
1871    if (srcServiceName.startsWith("ha-hdfs") && desServiceName.startsWith("ha-hdfs")) {
1872      Collection<String> internalNameServices =
1873        conf.getTrimmedStringCollection("dfs.internal.nameservices");
1874      if (!internalNameServices.isEmpty()) {
1875        if (internalNameServices.contains(srcServiceName.split(":")[1])) {
1876          return true;
1877        } else {
1878          return false;
1879        }
1880      }
1881    }
1882    if (srcFs instanceof DistributedFileSystem && desFs instanceof DistributedFileSystem) {
1883      // If one serviceName is an HA format while the other is a non-HA format,
1884      // maybe they refer to the same FileSystem.
1885      // For example, srcFs is "ha-hdfs://nameservices" and desFs is "hdfs://activeNamenode:port"
1886      Set<InetSocketAddress> srcAddrs = getNNAddresses((DistributedFileSystem) srcFs, conf);
1887      Set<InetSocketAddress> desAddrs = getNNAddresses((DistributedFileSystem) desFs, conf);
1888      if (Sets.intersection(srcAddrs, desAddrs).size() > 0) {
1889        return true;
1890      }
1891    }
1892
1893    return false;
1894  }

1902   * Recover the lease from HDFS, retrying multiple times.
1903   */
1904  public static void recoverFileLease(FileSystem fs, Path p, Configuration conf,
1905    CancelableProgressable reporter) throws IOException {
1906    if (fs instanceof FilterFileSystem) {
1907      fs = ((FilterFileSystem) fs).getRawFileSystem();
1908    }
1909    // lease recovery not needed for local file system case.
1910    if (!(fs instanceof DistributedFileSystem)) {
1911      return;
1912    }
1913    recoverDFSFileLease((DistributedFileSystem) fs, p, conf, reporter);

1916  /*
1917   * Run the dfs recover lease. recoverLease is asynchronous. It returns: -false when it starts the
1918   * lease recovery (i.e. lease recovery not *yet* done) - true when the lease recovery has
1919   * succeeded or the file is closed. But, we have to be careful. Each time we call recoverLease, it
1920   * starts the recover lease process over from the beginning. We could put ourselves in a situation
1921   * where we are doing nothing but starting a recovery, interrupting it to start again, and so on.
1922   * The findings over in HBASE-8354 have it that the namenode will try to recover the lease on the
1923   * file's primary node. If all is well, it should return near immediately. But, as is common, it
1924   * is the very primary node that has crashed and so the namenode will be stuck waiting on a socket
1925   * timeout before it will ask another datanode to start the recovery. It does not help if we call
1926   * recoverLease in the meantime and in particular, subsequent to the socket timeout, a
1927   * recoverLease invocation will cause us to start over from square one (possibly waiting on socket
1928   * timeout against primary node). So, in the below, we do the following: 1. Call recoverLease. 2.
1929   * If it returns true, break. 3. If it returns false, wait a few seconds and then call it again.
1930   * 4. If it returns true, break. 5. If it returns false, wait for what we think the datanode
1931   * socket timeout is (configurable) and then try again. 6. If it returns true, break. 7. If it
1932   * returns false, repeat starting at step 5. above. If HDFS-4525 is available, call it every
1933   * second and we might be able to exit early.
1934   */
1935  private static boolean recoverDFSFileLease(final DistributedFileSystem dfs, final Path p,
1936    final Configuration conf, final CancelableProgressable reporter) throws IOException {
1937    LOG.info("Recover lease on dfs file " + p);
1938    long startWaiting = EnvironmentEdgeManager.currentTime();
1939    // Default is 15 minutes. It's huge, but the idea is that if we have a major issue, HDFS
1940    // usually needs 10 minutes before marking the nodes as dead. So we're putting ourselves
1941    // beyond that limit 'to be safe'.
1942    long recoveryTimeout = conf.getInt("hbase.lease.recovery.timeout", 900000) + startWaiting;
1943    // This setting should be a little bit above what the cluster dfs heartbeat is set to.
1944    long firstPause = conf.getInt("hbase.lease.recovery.first.pause", 4000);
1945    // This should be set to how long it'll take for us to timeout against primary datanode if it
1946    // is dead. We set it to 64 seconds, 4 second than the default READ_TIMEOUT in HDFS, the
1947    // default value for DFS_CLIENT_SOCKET_TIMEOUT_KEY. If recovery is still failing after this
1948    // timeout, then further recovery will take liner backoff with this base, to avoid endless
1949    // preemptions when this value is not properly configured.
1950    long subsequentPauseBase = conf.getLong("hbase.lease.recovery.dfs.timeout", 64 * 1000);
1951
1952    Method isFileClosedMeth = null;
1953    // whether we need to look for isFileClosed method
1954    boolean findIsFileClosedMeth = true;
1955    boolean recovered = false;
1956    // We break the loop if we succeed the lease recovery, timeout, or we throw an exception.
1957    for (int nbAttempt = 0; !recovered; nbAttempt++) {
1958      recovered = recoverLease(dfs, nbAttempt, p, startWaiting);
1959      if (recovered) {
1960        break;
1961      }
1962      checkIfCancelled(reporter);
1963      if (checkIfTimedout(conf, recoveryTimeout, nbAttempt, p, startWaiting)) {
1964        break;
1965      }
1966      try {
1967        // On the first time through wait the short 'firstPause'.
1968        if (nbAttempt == 0) {
1969          Thread.sleep(firstPause);
1970        } else {
1971          // Cycle here until (subsequentPause * nbAttempt) elapses. While spinning, check
1972          // isFileClosed if available (should be in hadoop 2.0.5... not in hadoop 1 though.
1973          long localStartWaiting = EnvironmentEdgeManager.currentTime();
1974          while ((EnvironmentEdgeManager.currentTime() - localStartWaiting) < subsequentPauseBase *
1975            nbAttempt) {
1976            Thread.sleep(conf.getInt("hbase.lease.recovery.pause", 1000));
1977            if (findIsFileClosedMeth) {
1978              try {
1979                isFileClosedMeth =
1980                  dfs.getClass().getMethod("isFileClosed", new Class[] { Path.class });
1981              } catch (NoSuchMethodException nsme) {
1982                LOG.debug("isFileClosed not available");
1983              } finally {
1984                findIsFileClosedMeth = false;
1985              }
1986            }
1987            if (isFileClosedMeth != null && isFileClosed(dfs, isFileClosedMeth, p)) {
1988              recovered = true;
1989              break;
1990            }
1991            checkIfCancelled(reporter);
1992          }
1993        }
1994      } catch (InterruptedException ie) {
1995        InterruptedIOException iioe = new InterruptedIOException();
1996        iioe.initCause(ie);
1997        throw iioe;
1998      }
1999    }
2000    return recovered;
2001  }

2003  private static boolean checkIfTimedout(final Configuration conf, final long recoveryTimeout,
2004    final int nbAttempt, final Path p, final long startWaiting) {
2005    if (recoveryTimeout < EnvironmentEdgeManager.currentTime()) {
2006      LOG.warn("Cannot recoverLease after trying for " +
2007        conf.getInt("hbase.lease.recovery.timeout", 900000) +
2008        "ms (hbase.lease.recovery.timeout); continuing, but may be DATALOSS!!!; " +
2009        getLogMessageDetail(nbAttempt, p, startWaiting));
2010      return true;
2011    }
2012    return false;
2013  }

2015  /**
2016   * Try to recover the lease.
2017   * @return True if dfs#recoverLease came by true.
2018   */
2019  private static boolean recoverLease(final DistributedFileSystem dfs, final int nbAttempt,
2020    final Path p, final long startWaiting) throws FileNotFoundException {
2021    boolean recovered = false;
2022    try {
2023      recovered = dfs.recoverLease(p);
2024      LOG.info((recovered ? "Recovered lease, " : "Failed to recover lease, ") +
2025        getLogMessageDetail(nbAttempt, p, startWaiting));
2026    } catch (IOException e) {
2027      if (e instanceof LeaseExpiredException && e.getMessage().contains("File does not exist")) {
2028        // This exception comes out instead of FNFE, fix it
2029        throw new FileNotFoundException("The given WAL wasn't found at " + p);
2030      } else if (e instanceof FileNotFoundException) {
2031        throw (FileNotFoundException) e;
2032      }
2033      LOG.warn(getLogMessageDetail(nbAttempt, p, startWaiting), e);
2034    }
2035    return recovered;
2036  }
