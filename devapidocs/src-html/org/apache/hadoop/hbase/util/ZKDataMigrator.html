018package org.apache.hadoop.hbase.util;
019
020import java.io.IOException;
021import java.util.HashMap;
022import java.util.List;
023import java.util.Map;
024
025import org.apache.hadoop.hbase.TableName;
026import org.apache.hadoop.hbase.client.TableState;
027import org.apache.hadoop.hbase.exceptions.DeserializationException;
028import org.apache.hadoop.hbase.shaded.protobuf.ProtobufUtil;
029import org.apache.hadoop.hbase.shaded.protobuf.generated.ZooKeeperProtos;
030import org.apache.hadoop.hbase.zookeeper.ZKUtil;
031import org.apache.hadoop.hbase.zookeeper.ZKWatcher;
032import org.apache.hadoop.hbase.zookeeper.ZNodePaths;
033import org.apache.yetus.audience.InterfaceAudience;
034import org.apache.zookeeper.KeeperException;
035import org.slf4j.Logger;
036import org.slf4j.LoggerFactory;
037
038/**
039 * Utlity method to migrate zookeeper data across HBase versions.
040 * Used by Master mirroring table state to zk for hbase-1 clients.
041 * @deprecated Since 2.0.0. To be removed in hbase-3.0.0.
042 */
043@Deprecated
044@InterfaceAudience.Private
045public class ZKDataMigrator {
046  private static final Logger LOG = LoggerFactory.getLogger(ZKDataMigrator.class);
047
048  // Shutdown constructor.
049  private ZKDataMigrator() {}
050
051  /**
052   * Method for table states migration.
053   * Used when upgrading from pre-2.0 to 2.0
054   * Reading state from zk, applying them to internal state
055   * and delete.
056   * Used by master to clean migration from zk based states to
057   * table descriptor based states.
058   * @deprecated Since 2.0.0. To be removed in hbase-3.0.0.
059   */
060  @Deprecated
061  public static Map<TableName, TableState.State> queryForTableStates(ZKWatcher zkw)
062      throws KeeperException, InterruptedException {
063    Map<TableName, TableState.State> rv = new HashMap<>();
064    List<String> children = ZKUtil.listChildrenNoWatch(zkw, zkw.getZNodePaths().tableZNode);
065    if (children == null)
066      return rv;
067    for (String child: children) {
068      TableName tableName = TableName.valueOf(child);
069      TableState.State newState = ProtobufUtil.toTableState(getTableState(zkw, tableName));
070      rv.put(tableName, newState);
071    }
072    return rv;
073  }
074
075  /**
076   * Gets table state from ZK.
077   * @param zkw ZKWatcher instance to use
078   * @param tableName table we're checking
079   * @return Null or
080   * {@link org.apache.hadoop.hbase.shaded.protobuf.generated.ZooKeeperProtos.DeprecatedTableState.State}
081   *        found in znode.
082   * @throws KeeperException
083   * @deprecated Since 2.0.0. To be removed in hbase-3.0.0.
084   */
085  @Deprecated
086  private static ZooKeeperProtos.DeprecatedTableState.State getTableState(
087          final ZKWatcher zkw, final TableName tableName)
088      throws KeeperException, InterruptedException {
089    String znode = ZNodePaths.joinZNode(zkw.getZNodePaths().tableZNode,
090            tableName.getNameAsString());
091    try {
092      return ProtobufUtil.toTableState(ZKUtil.getData(zkw, znode));
093    } catch (IOException e) {
094      KeeperException ke = new KeeperException.DataInconsistencyException();
095      ke.initCause(e);
096      throw ke;
097    } catch (DeserializationException e) {
098      throw ZKUtil.convert(e);
099    }
100  }
101}
