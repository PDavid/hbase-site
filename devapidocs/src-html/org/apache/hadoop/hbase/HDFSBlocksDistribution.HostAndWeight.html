018package org.apache.hadoop.hbase;
019
020import java.util.ArrayList;
021import java.util.Comparator;
022import java.util.List;
023import java.util.Map;
024import java.util.NavigableSet;
025import java.util.TreeMap;
026import java.util.TreeSet;
027import org.apache.hadoop.fs.StorageType;
028import org.apache.hadoop.hbase.util.DNS;
029import org.apache.yetus.audience.InterfaceAudience;
030
031/**
032 * Data structure to describe the distribution of HDFS blocks among hosts. Adding erroneous data
033 * will be ignored silently.
034 */
035@InterfaceAudience.Private
036public class HDFSBlocksDistribution {
037  private Map<String, HostAndWeight> hostAndWeights = null;
038  private long uniqueBlocksTotalWeight = 0;
039
040  /**
041   * Stores the hostname and weight for that hostname. This is used when determining the physical
042   * locations of the blocks making up a region. To make a prioritized list of the hosts holding the
043   * most data of a region, this class is used to count the total weight for each host. The weight
044   * is currently just the size of the file.
045   */
046  public static class HostAndWeight {
047
048    private final String host;
049    private long weight;
050    private long weightForSsd;
051
052    /**
053     * Constructor
054     * @param host         the host name
055     * @param weight       the weight
056     * @param weightForSsd the weight for ssd
057     */
058    public HostAndWeight(String host, long weight, long weightForSsd) {
059      this.host = host;
060      this.weight = weight;
061      this.weightForSsd = weightForSsd;
062    }
063
064    /**
065     * add weight
066     * @param weight       the weight
067     * @param weightForSsd the weight for ssd
068     */
069    public void addWeight(long weight, long weightForSsd) {
070      this.weight += weight;
071      this.weightForSsd += weightForSsd;
072    }
073
074    /**
075     * @return the host name
076     */
077    public String getHost() {
078      return host;
079    }
080
081    /**
082     * @return the weight
083     */
084    public long getWeight() {
085      return weight;
086    }
087
088    /**
089     * @return the weight for ssd
090     */
091    public long getWeightForSsd() {
092      return weightForSsd;
093    }
094
095    /**
096     * comparator used to sort hosts based on weight
097     */
098    public static class WeightComparator implements Comparator<HostAndWeight> {
099      @Override
100      public int compare(HostAndWeight l, HostAndWeight r) {
101        if (l.getWeight() == r.getWeight()) {
102          return l.getHost().compareTo(r.getHost());
103        }
104        return l.getWeight() < r.getWeight() ? -1 : 1;
105      }
106    }
107  }
108
109  /**
110   * Constructor
111   */
112  public HDFSBlocksDistribution() {
113    this.hostAndWeights = new TreeMap<>();
114  }
115
116  /**
117   * @see java.lang.Object#toString()
118   */
119  @Override
120  public synchronized String toString() {
121    return "number of unique hosts in the distribution=" + this.hostAndWeights.size();
122  }
123
124  /**
125   * add some weight to a list of hosts, update the value of unique block weight
126   * @param hosts  the list of the host
127   * @param weight the weight
128   */
129  public void addHostsAndBlockWeight(String[] hosts, long weight) {
130    addHostsAndBlockWeight(hosts, weight, null);
131  }
132
133  /**
134   * add some weight to a list of hosts, update the value of unique block weight
135   * @param hosts  the list of the host
136   * @param weight the weight
137   */
138  public void addHostsAndBlockWeight(String[] hosts, long weight, StorageType[] storageTypes) {
139    if (hosts == null || hosts.length == 0) {
140      // erroneous data
141      return;
142    }
143
144    addUniqueWeight(weight);
145    if (storageTypes != null && storageTypes.length == hosts.length) {
146      for (int i = 0; i < hosts.length; i++) {
147        long weightForSsd = 0;
148        if (storageTypes[i] == StorageType.SSD) {
149          weightForSsd = weight;
150        }
151        addHostAndBlockWeight(hosts[i], weight, weightForSsd);
152      }
153    } else {
154      for (String hostname : hosts) {
155        addHostAndBlockWeight(hostname, weight, 0);
156      }
157    }
158  }
159
160  /**
161   * add some weight to the total unique weight
162   * @param weight the weight
163   */
164  private void addUniqueWeight(long weight) {
165    uniqueBlocksTotalWeight += weight;
166  }
167
168  /**
169   * add some weight to a specific host
170   * @param host         the host name
171   * @param weight       the weight
172   * @param weightForSsd the weight for ssd
173   */
174  private void addHostAndBlockWeight(String host, long weight, long weightForSsd) {
175    if (host == null) {
176      // erroneous data
177      return;
178    }
179
180    HostAndWeight hostAndWeight = this.hostAndWeights.get(host);
181    if (hostAndWeight == null) {
182      hostAndWeight = new HostAndWeight(host, weight, weightForSsd);
183      this.hostAndWeights.put(host, hostAndWeight);
184    } else {
185      hostAndWeight.addWeight(weight, weightForSsd);
186    }
187  }
188
189  /**
190   * @return the hosts and their weights
191   */
192  public Map<String, HostAndWeight> getHostAndWeights() {
193    return this.hostAndWeights;
194  }
195
196  /**
197   * return the weight for a specific host, that will be the total bytes of all blocks on the host
198   * @param host the host name
199   * @return the weight of the given host
200   */
201  public long getWeight(String host) {
202    long weight = 0;
203    if (host != null) {
204      HostAndWeight hostAndWeight = this.hostAndWeights.get(host);
205      if (hostAndWeight != null) {
206        weight = hostAndWeight.getWeight();
207      }
208    }
209    return weight;
210  }
211
212  /**
213   * @return the sum of all unique blocks' weight
214   */
215  public long getUniqueBlocksTotalWeight() {
216    return uniqueBlocksTotalWeight;
217  }
218
219  /**
220   * Implementations 'visit' hostAndWeight.
221   */
222  public interface Visitor {
223    long visit(final HostAndWeight hostAndWeight);
224  }
225
226  /**
227   * @param host the host name
228   * @return the locality index of the given host
229   */
230  public float getBlockLocalityIndex(String host) {
231    if (uniqueBlocksTotalWeight == 0) {
232      return 0.0f;
233    } else {
234      return (float) getBlocksLocalityWeightInternal(host, HostAndWeight::getWeight)
235        / (float) uniqueBlocksTotalWeight;
236    }
237  }
238
239  /**
240   * @param host the host name
241   * @return the locality index with ssd of the given host
242   */
243  public float getBlockLocalityIndexForSsd(String host) {
244    if (uniqueBlocksTotalWeight == 0) {
245      return 0.0f;
246    } else {
247      return (float) getBlocksLocalityWeightInternal(host, HostAndWeight::getWeightForSsd)
248        / (float) uniqueBlocksTotalWeight;
249    }
250  }
251
252  /**
253   * @param host the host name
254   * @return the blocks local weight of the given host
255   */
256  public long getBlocksLocalWeight(String host) {
257    return getBlocksLocalityWeightInternal(host, HostAndWeight::getWeight);
258  }
259
260  /**
261   * @param host the host name
262   * @return the blocks local with ssd weight of the given host
263   */
264  public long getBlocksLocalWithSsdWeight(String host) {
265    return getBlocksLocalityWeightInternal(host, HostAndWeight::getWeightForSsd);
266  }
267
268  /**
269   * @param host the host name
270   * @return the locality index of the given host
271   */
272  private long getBlocksLocalityWeightInternal(String host, Visitor visitor) {
273    long localityIndex = 0;
274    HostAndWeight hostAndWeight = this.hostAndWeights.get(host);
275    // Compatible with local mode, see HBASE-24569
276    if (hostAndWeight == null) {
277      String currentHost = "";
278      try {
279        currentHost = DNS.getDefaultHost("default", "default");
280      } catch (Exception e) {
281        // Just ignore, it's ok, avoid too many log info
282      }
283      if (host.equals(currentHost)) {
284        hostAndWeight = this.hostAndWeights.get(HConstants.LOCALHOST);
285      }
286    }
287    if (hostAndWeight != null && uniqueBlocksTotalWeight != 0) {
288      localityIndex = visitor.visit(hostAndWeight);
289    }
290    return localityIndex;
291  }
292
293  /**
294   * This will add the distribution from input to this object
295   * @param otherBlocksDistribution the other hdfs blocks distribution
296   */
297  public void add(HDFSBlocksDistribution otherBlocksDistribution) {
298    Map<String, HostAndWeight> otherHostAndWeights = otherBlocksDistribution.getHostAndWeights();
299    for (Map.Entry<String, HostAndWeight> otherHostAndWeight : otherHostAndWeights.entrySet()) {
300      addHostAndBlockWeight(otherHostAndWeight.getValue().host,
301        otherHostAndWeight.getValue().weight, otherHostAndWeight.getValue().weightForSsd);
302    }
303    addUniqueWeight(otherBlocksDistribution.getUniqueBlocksTotalWeight());
304  }
305
306  /**
307   * return the sorted list of hosts in terms of their weights
308   */
309  public List<String> getTopHosts() {
310    HostAndWeight[] hostAndWeights = getTopHostsWithWeights();
311    List<String> topHosts = new ArrayList<>(hostAndWeights.length);
312    for (HostAndWeight haw : hostAndWeights) {
313      topHosts.add(haw.getHost());
314    }
315    return topHosts;
316  }
317
318  /**
319   * return the sorted list of hosts in terms of their weights
320   */
321  public HostAndWeight[] getTopHostsWithWeights() {
322    NavigableSet<HostAndWeight> orderedHosts = new TreeSet<>(new HostAndWeight.WeightComparator());
323    orderedHosts.addAll(this.hostAndWeights.values());
324    return orderedHosts.descendingSet().toArray(new HostAndWeight[orderedHosts.size()]);
325  }
