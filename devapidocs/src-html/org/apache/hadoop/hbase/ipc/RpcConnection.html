103  protected RpcConnection(Configuration conf, HashedWheelTimer timeoutTimer, ConnectionId remoteId,
104    String clusterId, boolean isSecurityEnabled, Codec codec, CompressionCodec compressor,
105    CellBlockBuilder cellBlockBuilder, MetricsConnection metrics,
106    Map<String, byte[]> connectionAttributes) throws IOException {
107    this.timeoutTimer = timeoutTimer;
108    this.codec = codec;
109    this.compressor = compressor;
110    this.cellBlockBuilder = cellBlockBuilder;
111    this.conf = conf;
112    this.metrics = metrics;
113    this.connectionAttributes = connectionAttributes;
114    User ticket = remoteId.getTicket();
115    this.securityInfo = SecurityInfo.getInfo(remoteId.getServiceName());
116    this.useSasl = isSecurityEnabled;
117
118    // Choose the correct Token and AuthenticationProvider for this client to use
119    SaslClientAuthenticationProviders providers =
120      SaslClientAuthenticationProviders.getInstance(conf);
121    Pair<SaslClientAuthenticationProvider, Token<? extends TokenIdentifier>> pair;
122    if (useSasl && securityInfo != null) {
123      pair = providers.selectProvider(clusterId, ticket);
124      if (pair == null) {
125        if (LOG.isTraceEnabled()) {
126          LOG.trace("Found no valid authentication method from providers={} with tokens={}",
127            providers.toString(), ticket.getTokens());
128        }
129        throw new RuntimeException("Found no valid authentication method from options");
130      }
131    } else if (!useSasl) {
132      // Hack, while SIMPLE doesn't go via SASL.
133      pair = providers.getSimpleProvider();
134    } else {
135      throw new RuntimeException("Could not compute valid client authentication provider");
136    }
137
138    this.provider = pair.getFirst();
139    this.token = pair.getSecond();
140
141    LOG.debug("Using {} authentication for service={}, sasl={}",
142      provider.getSaslAuthMethod().getName(), remoteId.serviceName, useSasl);
143    reloginMaxBackoff = conf.getInt("hbase.security.relogin.maxbackoff", 5000);
144    this.remoteId = remoteId;
145  }
146
147  protected final void scheduleTimeoutTask(final Call call) {
148    if (call.timeout > 0) {
149      call.timeoutTask = timeoutTimer.newTimeout(new TimerTask() {
150
151        @Override
152        public void run(Timeout timeout) throws Exception {
153          call.setTimeout(new CallTimeoutException(call.toShortString() + ", waitTime="
154            + (EnvironmentEdgeManager.currentTime() - call.getStartTime()) + "ms, rpcTimeout="
155            + call.timeout + "ms"));
156          callTimeout(call);
157        }
158      }, call.timeout, TimeUnit.MILLISECONDS);
159    }
160  }

162  // will be overridden in tests
163  protected byte[] getConnectionHeaderPreamble() {
164    // Assemble the preamble up in a buffer first and then send it. Writing individual elements,
165    // they are getting sent across piecemeal according to wireshark and then server is messing
166    // up the reading on occasion (the passed in stream is not buffered yet).
167    int rpcHeaderLen = HConstants.RPC_HEADER.length;
168    // Preamble is six bytes -- 'HBas' + VERSION + AUTH_CODE
169    byte[] preamble = new byte[rpcHeaderLen + 2];
170    System.arraycopy(HConstants.RPC_HEADER, 0, preamble, 0, rpcHeaderLen);
171    preamble[rpcHeaderLen] = HConstants.RPC_CURRENT_VERSION;
172    synchronized (this) {
173      preamble[preamble.length - 1] = provider.getSaslAuthMethod().getCode();
174    }
175    return preamble;
176  }
177
178  protected final ConnectionHeader getConnectionHeader() {
179    final ConnectionHeader.Builder builder = ConnectionHeader.newBuilder();
180    builder.setServiceName(remoteId.getServiceName());
181    final UserInformation userInfoPB = provider.getUserInfo(remoteId.ticket);
182    if (userInfoPB != null) {
183      builder.setUserInfo(userInfoPB);
184    }
185    if (this.codec != null) {
186      builder.setCellBlockCodecClass(this.codec.getClass().getCanonicalName());
187    }
188    if (this.compressor != null) {
189      builder.setCellBlockCompressorClass(this.compressor.getClass().getCanonicalName());
190    }
191    if (connectionAttributes != null && !connectionAttributes.isEmpty()) {
192      HBaseProtos.NameBytesPair.Builder attributeBuilder = HBaseProtos.NameBytesPair.newBuilder();
193      for (Map.Entry<String, byte[]> attribute : connectionAttributes.entrySet()) {
194        attributeBuilder.setName(attribute.getKey());
195        attributeBuilder.setValue(UnsafeByteOperations.unsafeWrap(attribute.getValue()));
196        builder.addAttribute(attributeBuilder.build());
197      }
198    }
199    builder.setVersionInfo(ProtobufUtil.getVersionInfo());
200    boolean isCryptoAESEnable = conf.getBoolean(CRYPTO_AES_ENABLED_KEY, CRYPTO_AES_ENABLED_DEFAULT);
201    // if Crypto AES enable, setup Cipher transformation
202    if (isCryptoAESEnable) {
203      builder.setRpcCryptoCipherTransformation(
204        conf.get("hbase.rpc.crypto.encryption.aes.cipher.transform", "AES/CTR/NoPadding"));
205    }
206    return builder.build();
207  }

209  protected final InetSocketAddress getRemoteInetAddress(MetricsConnection metrics)
210    throws UnknownHostException {
211    if (metrics != null) {
212      metrics.incrNsLookups();
213    }
214    InetSocketAddress remoteAddr = Address.toSocketAddress(remoteId.getAddress());
215    if (remoteAddr.isUnresolved()) {
216      if (metrics != null) {
217        metrics.incrNsLookupsFailed();
218      }
219      throw new UnknownHostException(remoteId.getAddress() + " could not be resolved");
220    }
221    return remoteAddr;
222  }

257  private <T extends InputStream & DataInput> void finishCall(ResponseHeader responseHeader, T in,
258    Call call) throws IOException {
259    Message value;
260    if (call.responseDefaultType != null) {
261      Message.Builder builder = call.responseDefaultType.newBuilderForType();
262      if (!builder.mergeDelimitedFrom(in)) {
263        // The javadoc of mergeDelimitedFrom says returning false means the stream reaches EOF
264        // before reading any bytes out, so here we need to manually finish create the EOFException
265        // and finish the call
266        call.setException(new EOFException("EOF while reading response with type: "
267          + call.responseDefaultType.getClass().getName()));
268        return;
269      }
270      value = builder.build();
271    } else {
272      value = null;
273    }
274    CellScanner cellBlockScanner;
275    if (responseHeader.hasCellBlockMeta()) {
276      int size = responseHeader.getCellBlockMeta().getLength();
277      // Maybe we could read directly from the ByteBuf.
278      // The problem here is that we do not know when to release it.
279      byte[] cellBlock = new byte[size];
280      in.readFully(cellBlock);
281      cellBlockScanner = cellBlockBuilder.createCellScanner(this.codec, this.compressor, cellBlock);
282    } else {
283      cellBlockScanner = null;
284    }
285    call.setResponse(value, cellBlockScanner);
286  }

288  <T extends InputStream & DataInput> void readResponse(T in, Map<Integer, Call> id2Call,
289    Consumer<RemoteException> fatalConnectionErrorConsumer) throws IOException {
290    int totalSize = in.readInt();
291    ResponseHeader responseHeader = ResponseHeader.parseDelimitedFrom(in);
292    int id = responseHeader.getCallId();
293    if (LOG.isTraceEnabled()) {
294      LOG.trace("got response header " + TextFormat.shortDebugString(responseHeader)
295        + ", totalSize: " + totalSize + " bytes");
296    }
297    RemoteException remoteExc;
298    if (responseHeader.hasException()) {
299      ExceptionResponse exceptionResponse = responseHeader.getException();
300      remoteExc = IPCUtil.createRemoteException(exceptionResponse);
301      if (IPCUtil.isFatalConnectionException(exceptionResponse)) {
302        // Here we will cleanup all calls so do not need to fall back, just return.
303        fatalConnectionErrorConsumer.accept(remoteExc);
304        if (connectionRegistryCall != null) {
305          connectionRegistryCall.setException(remoteExc);
306          connectionRegistryCall = null;
307        }
308        return;
309      }
310    } else {
311      remoteExc = null;
312    }
313    if (id < 0) {
314      if (connectionRegistryCall != null) {
315        LOG.debug("process connection registry call");
316        finishCall(responseHeader, in, connectionRegistryCall);
317        connectionRegistryCall = null;
318        return;
319      }
320    }
321    Call call = id2Call.remove(id);
322    if (call == null) {
323      // So we got a response for which we have no corresponding 'call' here on the client-side.
324      // We probably timed out waiting, cleaned up all references, and now the server decides
325      // to return a response. There is nothing we can do w/ the response at this stage. Clean
326      // out the wire of the response so its out of the way and we can get other responses on
327      // this connection.
328      if (LOG.isDebugEnabled()) {
329        int readSoFar = IPCUtil.getTotalSizeWhenWrittenDelimited(responseHeader);
330        int whatIsLeftToRead = totalSize - readSoFar;
331        LOG.debug("Unknown callId: " + id + ", skipping over this response of " + whatIsLeftToRead
332          + " bytes");
333      }
334      return;
335    }
336    call.callStats.setResponseSizeBytes(totalSize);
337    if (remoteExc != null) {
338      call.setException(remoteExc);
339      return;
340    }
341    try {
342      finishCall(responseHeader, in, call);
343    } catch (IOException e) {
344      // As the call has been removed from id2Call map, if we hit an exception here, the
345      // exceptionCaught method can not help us finish the call, so here we need to catch the
346      // exception and finish it
347      call.setException(e);
348      // throw the exception out, the upper layer should determine whether this is a critical
349      // problem
350      throw e;
351    }
352  }
