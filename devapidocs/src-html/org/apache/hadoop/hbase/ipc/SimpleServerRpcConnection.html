082  public SimpleServerRpcConnection(SimpleRpcServer rpcServer, SocketChannel channel,
083    long lastContact) {
084    super(rpcServer);
085    this.channel = channel;
086    this.lastContact = lastContact;
087    this.data = null;
088    this.dataLengthBuffer = ByteBuffer.allocate(4);
089    this.socket = channel.socket();
090    this.addr = socket.getInetAddress();
091    if (addr == null) {
092      this.hostAddress = "*Unknown*";
093    } else {
094      this.hostAddress = addr.getHostAddress();
095    }
096    this.remotePort = socket.getPort();
097    if (rpcServer.socketSendBufferSize != 0) {
098      try {
099        socket.setSendBufferSize(rpcServer.socketSendBufferSize);
100      } catch (IOException e) {
101        SimpleRpcServer.LOG.warn(
102          "Connection: unable to set socket send buffer size to " + rpcServer.socketSendBufferSize);
103      }
104    }
105    this.responder = rpcServer.responder;
106  }

131  private int readPreamble() throws IOException {
132    if (preambleBuffer == null) {
133      preambleBuffer = ByteBuffer.allocate(6);
134    }
135    int count = this.rpcServer.channelRead(channel, preambleBuffer);
136    if (count < 0 || preambleBuffer.remaining() > 0) {
137      return count;
138    }
139    preambleBuffer.flip();
140    if (!processPreamble(preambleBuffer)) {
141      return -1;
142    }
143    preambleBuffer = null; // do not need it anymore
144    connectionPreambleRead = true;
145    return count;
146  }
147
148  private int read4Bytes() throws IOException {
149    if (this.dataLengthBuffer.remaining() > 0) {
150      return this.rpcServer.channelRead(channel, this.dataLengthBuffer);
151    } else {
152      return 0;
153    }
154  }

156  private void processUnwrappedData(byte[] inBuf) throws IOException, InterruptedException {
157    ReadableByteChannel ch = Channels.newChannel(new ByteArrayInputStream(inBuf));
158    // Read all RPCs contained in the inBuf, even partial ones
159    while (true) {
160      int count;
161      if (unwrappedDataLengthBuffer.remaining() > 0) {
162        count = this.rpcServer.channelRead(ch, unwrappedDataLengthBuffer);
163        if (count <= 0 || unwrappedDataLengthBuffer.remaining() > 0) {
164          return;
165        }
166      }
167
168      if (unwrappedData == null) {
169        unwrappedDataLengthBuffer.flip();
170        int unwrappedDataLength = unwrappedDataLengthBuffer.getInt();
171
172        if (unwrappedDataLength == RpcClient.PING_CALL_ID) {
173          if (RpcServer.LOG.isDebugEnabled()) RpcServer.LOG.debug("Received ping message");
174          unwrappedDataLengthBuffer.clear();
175          continue; // ping message
176        }
177        unwrappedData = ByteBuffer.allocate(unwrappedDataLength);
178      }
179
180      count = this.rpcServer.channelRead(ch, unwrappedData);
181      if (count <= 0 || unwrappedData.remaining() > 0) {
182        return;
183      }
184
185      if (unwrappedData.remaining() == 0) {
186        unwrappedDataLengthBuffer.clear();
187        unwrappedData.flip();
188        processOneRpc(new SingleByteBuff(unwrappedData));
189        unwrappedData = null;
190      }
191    }
192  }

194  private void saslReadAndProcess(ByteBuff saslToken) throws IOException, InterruptedException {
195    if (saslContextEstablished) {
196      RpcServer.LOG.trace("Read input token of size={} for processing by saslServer.unwrap()",
197        saslToken.limit());
198      if (!useWrap) {
199        processOneRpc(saslToken);
200      } else {
201        byte[] b = saslToken.hasArray() ? saslToken.array() : saslToken.toBytes();
202        byte[] plaintextData = saslServer.unwrap(b, 0, b.length);
203        // release the request buffer as we have already unwrapped all its content
204        callCleanupIfNeeded();
205        processUnwrappedData(plaintextData);
206      }
207    } else {
208      byte[] replyToken;
209      try {
210        try {
211          getOrCreateSaslServer();
212        } catch (Exception e) {
213          RpcServer.LOG.error("Error when trying to create instance of HBaseSaslRpcServer "
214            + "with sasl provider: " + provider, e);
215          throw e;
216        }
217        RpcServer.LOG.debug("Created SASL server with mechanism={}",
218          provider.getSaslAuthMethod().getAuthMethod());
219        RpcServer.LOG.debug(
220          "Read input token of size={} for processing by saslServer." + "evaluateResponse()",
221          saslToken.limit());
222        replyToken = saslServer
223          .evaluateResponse(saslToken.hasArray() ? saslToken.array() : saslToken.toBytes());
224      } catch (IOException e) {
225        RpcServer.LOG.debug("Failed to execute SASL handshake", e);
226        Throwable sendToClient = HBaseSaslRpcServer.unwrap(e);
227        doRawSaslReply(SaslStatus.ERROR, null, sendToClient.getClass().getName(),
228          sendToClient.getLocalizedMessage());
229        this.rpcServer.metrics.authenticationFailure();
230        String clientIP = this.toString();
231        // attempting user could be null
232        RpcServer.AUDITLOG.warn("{}{}: {}", RpcServer.AUTH_FAILED_FOR, clientIP,
233          saslServer.getAttemptingUser());
234        throw e;
235      } finally {
236        // release the request buffer as we have already unwrapped all its content
237        callCleanupIfNeeded();
238      }
239      if (replyToken != null) {
240        if (RpcServer.LOG.isDebugEnabled()) {
241          RpcServer.LOG.debug("Will send token of size " + replyToken.length + " from saslServer.");
242        }
243        doRawSaslReply(SaslStatus.SUCCESS, new BytesWritable(replyToken), null, null);
244      }
245      if (saslServer.isComplete()) {
246        String qop = saslServer.getNegotiatedQop();
247        useWrap = qop != null && !"auth".equalsIgnoreCase(qop);
248        ugi =
249          provider.getAuthorizedUgi(saslServer.getAuthorizationID(), this.rpcServer.secretManager);
250        RpcServer.LOG.debug(
251          "SASL server context established. Authenticated client: {}. Negotiated QoP is {}", ugi,
252          qop);
253        this.rpcServer.metrics.authenticationSuccess();
254        RpcServer.AUDITLOG.info(RpcServer.AUTH_SUCCESSFUL_FOR + ugi);
255        saslContextEstablished = true;
256      }
257    }
258  }

260  /**
261   * Read off the wire. If there is not enough data to read, update the connection state with what
262   * we have and returns.
263   * @return Returns -1 if failure (and caller will close connection), else zero or more.
264   */
265  public int readAndProcess() throws IOException, InterruptedException {
266    // If we have not read the connection setup preamble, look to see if that is on the wire.
267    if (!connectionPreambleRead) {
268      int count = readPreamble();
269      if (!connectionPreambleRead) {
270        return count;
271      }
272    }
273
274    // Try and read in an int. it will be length of the data to read (or -1 if a ping). We catch the
275    // integer length into the 4-byte this.dataLengthBuffer.
276    int count = read4Bytes();
277    if (count < 0 || dataLengthBuffer.remaining() > 0) {
278      return count;
279    }
280
281    // We have read a length and we have read the preamble. It is either the connection header
282    // or it is a request.
283    if (data == null) {
284      dataLengthBuffer.flip();
285      int dataLength = dataLengthBuffer.getInt();
286      if (dataLength == RpcClient.PING_CALL_ID) {
287        if (!useWrap) { // covers the !useSasl too
288          dataLengthBuffer.clear();
289          return 0; // ping message
290        }
291      }
292      if (dataLength < 0) { // A data length of zero is legal.
293        throw new DoNotRetryIOException(
294          "Unexpected data length " + dataLength + "!! from " + getHostAddress());
295      }
296
297      if (dataLength > this.rpcServer.maxRequestSize) {
298        String msg = "RPC data length of " + dataLength + " received from " + getHostAddress()
299          + " is greater than max allowed " + this.rpcServer.maxRequestSize + ". Set \""
300          + SimpleRpcServer.MAX_REQUEST_SIZE
301          + "\" on server to override this limit (not recommended)";
302        SimpleRpcServer.LOG.warn(msg);
303
304        if (connectionHeaderRead && connectionPreambleRead) {
305          incRpcCount();
306          // Construct InputStream for the non-blocking SocketChannel
307          // We need the InputStream because we want to read only the request header
308          // instead of the whole rpc.
309          ByteBuffer buf = ByteBuffer.allocate(1);
310          InputStream is = new InputStream() {
311            @Override
312            public int read() throws IOException {
313              SimpleServerRpcConnection.this.rpcServer.channelRead(channel, buf);
314              buf.flip();
315              int x = buf.get();
316              buf.flip();
317              return x;
318            }
319          };
320          CodedInputStream cis = CodedInputStream.newInstance(is);
321          int headerSize = cis.readRawVarint32();
322          Message.Builder builder = RequestHeader.newBuilder();
323          ProtobufUtil.mergeFrom(builder, cis, headerSize);
324          RequestHeader header = (RequestHeader) builder.build();
325
326          // Notify the client about the offending request
327          SimpleServerCall reqTooBig = new SimpleServerCall(header.getCallId(), this.service, null,
328            null, null, null, this, 0, this.addr, EnvironmentEdgeManager.currentTime(), 0,
329            this.rpcServer.bbAllocator, this.rpcServer.cellBlockBuilder, null, responder);
330          RequestTooBigException reqTooBigEx = new RequestTooBigException(msg);
331          this.rpcServer.metrics.exception(reqTooBigEx);
332          // Make sure the client recognizes the underlying exception
333          // Otherwise, throw a DoNotRetryIOException.
334          if (
335            VersionInfoUtil.hasMinimumVersion(connectionHeader.getVersionInfo(),
336              RequestTooBigException.MAJOR_VERSION, RequestTooBigException.MINOR_VERSION)
337          ) {
338            reqTooBig.setResponse(null, null, reqTooBigEx, msg);
339          } else {
340            reqTooBig.setResponse(null, null, new DoNotRetryIOException(msg), msg);
341          }
342          // In most cases we will write out the response directly. If not, it is still OK to just
343          // close the connection without writing out the reqTooBig response. Do not try to write
344          // out directly here, and it will cause deserialization error if the connection is slow
345          // and we have a half writing response in the queue.
346          reqTooBig.sendResponseIfReady();
347        }
348        // Close the connection
349        return -1;
350      }
351
352      // Initialize this.data with a ByteBuff.
353      // This call will allocate a ByteBuff to read request into and assign to this.data
354      // Also when we use some buffer(s) from pool, it will create a CallCleanup instance also and
355      // assign to this.callCleanup
356      initByteBuffToReadInto(dataLength);
357
358      // Increment the rpc count. This counter will be decreased when we write
359      // the response. If we want the connection to be detected as idle properly, we
360      // need to keep the inc / dec correct.
361      incRpcCount();
362    }
363
364    count = channelDataRead(channel, data);
365
366    if (count >= 0 && data.remaining() == 0) { // count==0 if dataLength == 0
367      process();
368    }
369
370    return count;
371  }
372
373  // It creates the ByteBuff and CallCleanup and assign to Connection instance.
374  private void initByteBuffToReadInto(int length) {
375    this.data = rpcServer.bbAllocator.allocate(length);
376    this.callCleanup = data::release;
377  }
378
379  protected int channelDataRead(ReadableByteChannel channel, ByteBuff buf) throws IOException {
380    int count = buf.read(channel);
381    if (count > 0) {
382      this.rpcServer.metrics.receivedBytes(count);
383    }
384    return count;
385  }

387  /**
388   * Process the data buffer and clean the connection state for the next call.
389   */
390  private void process() throws IOException, InterruptedException {
391    data.rewind();
392    try {
393      if (skipInitialSaslHandshake) {
394        skipInitialSaslHandshake = false;
395        return;
396      }
397
398      if (useSasl) {
399        saslReadAndProcess(data);
400      } else {
401        processOneRpc(data);
402      }
403    } catch (Exception e) {
404      callCleanupIfNeeded();
405      throw e;
406    } finally {
407      dataLengthBuffer.clear(); // Clean for the next call
408      data = null; // For the GC
409      this.callCleanup = null;
410    }
411  }

413  @Override
414  public synchronized void close() {
415    disposeSasl();
416    data = null;
417    callCleanupIfNeeded();
418    if (!channel.isOpen()) {
419      return;
420    }
421    try {
422      socket.shutdownOutput();
423    } catch (Exception ignored) {
424      if (SimpleRpcServer.LOG.isTraceEnabled()) {
425        SimpleRpcServer.LOG.trace("Ignored exception", ignored);
426      }
427    }
428    if (channel.isOpen()) {
429      try {
430        channel.close();
431      } catch (Exception ignored) {
432      }
433    }
434    try {
435      socket.close();
436    } catch (Exception ignored) {
437      if (SimpleRpcServer.LOG.isTraceEnabled()) {
438        SimpleRpcServer.LOG.trace("Ignored exception", ignored);
439      }
440    }
441  }
