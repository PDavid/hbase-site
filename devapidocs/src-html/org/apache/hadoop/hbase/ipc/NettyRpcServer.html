072public class NettyRpcServer extends RpcServer {
073  public static final Logger LOG = LoggerFactory.getLogger(NettyRpcServer.class);

075  /**
076   * Name of property to change the byte buf allocator for the netty channels. Default is no value,
077   * which causes us to use PooledByteBufAllocator. Valid settings here are "pooled", "unpooled",
078   * and "heap", or, the name of a class implementing ByteBufAllocator.
079   * <p>
080   * "pooled" and "unpooled" may prefer direct memory depending on netty configuration, which is
081   * controlled by platform specific code and documented system properties.
082   * <p>
083   * "heap" will prefer heap arena allocations.
084   */
085  public static final String HBASE_NETTY_ALLOCATOR_KEY = "hbase.netty.rpcserver.allocator";
086  static final String POOLED_ALLOCATOR_TYPE = "pooled";
087  static final String UNPOOLED_ALLOCATOR_TYPE = "unpooled";
088  static final String HEAP_ALLOCATOR_TYPE = "heap";
089
090  /**
091   * Low watermark for pending outbound bytes of a single netty channel. If the high watermark was
092   * exceeded, channel will have setAutoRead to true again. The server will start reading incoming
093   * bytes (requests) from the client channel.
094   */
095  public static final String CHANNEL_WRITABLE_LOW_WATERMARK_KEY =
096    "hbase.server.netty.writable.watermark.low";
097  private static final int CHANNEL_WRITABLE_LOW_WATERMARK_DEFAULT = 0;
098
099  /**
100   * High watermark for pending outbound bytes of a single netty channel. If the number of pending
101   * outbound bytes exceeds this threshold, setAutoRead will be false for the channel. The server
102   * will stop reading incoming requests from the client channel.
103   * <p>
104   * Note: any requests already in the call queue will still be processed.
105   */
106  public static final String CHANNEL_WRITABLE_HIGH_WATERMARK_KEY =
107    "hbase.server.netty.writable.watermark.high";
108  private static final int CHANNEL_WRITABLE_HIGH_WATERMARK_DEFAULT = 0;
109
110  /**
111   * Fatal watermark for pending outbound bytes of a single netty channel. If the number of pending
112   * outbound bytes exceeds this threshold, the connection will be forcibly closed so that memory
113   * can be reclaimed. The client will have to re-establish a new connection and retry any in-flight
114   * requests.
115   * <p>
116   * Note: must be higher than the high watermark, otherwise it's ignored.
117   */
118  public static final String CHANNEL_WRITABLE_FATAL_WATERMARK_KEY =
119    "hbase.server.netty.writable.watermark.fatal";
120  private static final int CHANNEL_WRITABLE_FATAL_WATERMARK_DEFAULT = 0;

125  private final Channel serverChannel;
126  final ChannelGroup allChannels = new DefaultChannelGroup(GlobalEventExecutor.INSTANCE, true);
127  private final ByteBufAllocator channelAllocator;
128  private final AtomicReference<SslContext> sslContextForServer = new AtomicReference<>();
129  private final AtomicReference<FileChangeWatcher> keyStoreWatcher = new AtomicReference<>();
130  private final AtomicReference<FileChangeWatcher> trustStoreWatcher = new AtomicReference<>();

132  private volatile int writeBufferFatalThreshold;
133  private volatile WriteBufferWaterMark writeBufferWaterMark;

135  public NettyRpcServer(Server server, String name, List<BlockingServiceAndInterface> services,
136    InetSocketAddress bindAddress, Configuration conf, RpcScheduler scheduler,
137    boolean reservoirEnabled) throws IOException {
138    super(server, name, services, bindAddress, conf, scheduler, reservoirEnabled);
139    this.bindAddress = bindAddress;
140    this.channelAllocator = getChannelAllocator(conf);
141    // Get the event loop group configuration from the server class if available.
142    NettyEventLoopGroupConfig config = null;
143    if (server instanceof HBaseServerBase) {
144      config = ((HBaseServerBase<?>) server).getEventLoopGroupConfig();
145    }
146    if (config == null) {
147      config = new NettyEventLoopGroupConfig(conf, "NettyRpcServer");
148    }
149
150    // call before creating bootstrap below so that the necessary configs can be set
151    configureNettyWatermarks(conf);
152
153    EventLoopGroup eventLoopGroup = config.group();
154    Class<? extends ServerChannel> channelClass = config.serverChannelClass();
155    ServerBootstrap bootstrap = new ServerBootstrap().group(eventLoopGroup).channel(channelClass)
156      .childOption(ChannelOption.TCP_NODELAY, tcpNoDelay)
157      .childOption(ChannelOption.SO_KEEPALIVE, tcpKeepAlive)
158      .childOption(ChannelOption.SO_REUSEADDR, true)
159      .childHandler(new ChannelInitializer<Channel>() {
160        @Override
161        protected void initChannel(Channel ch) throws Exception {
162          ch.config().setWriteBufferWaterMark(writeBufferWaterMark);
163          ch.config().setAllocator(channelAllocator);
164          ChannelPipeline pipeline = ch.pipeline();
165          FixedLengthFrameDecoder preambleDecoder = new FixedLengthFrameDecoder(6);
166          preambleDecoder.setSingleDecode(true);
167          if (conf.getBoolean(HBASE_SERVER_NETTY_TLS_ENABLED, false)) {
168            initSSL(pipeline, conf.getBoolean(HBASE_SERVER_NETTY_TLS_SUPPORTPLAINTEXT, true));
169          }
170          NettyServerRpcConnection conn = createNettyServerRpcConnection(ch);
171          pipeline.addLast(NettyRpcServerPreambleHandler.DECODER_NAME, preambleDecoder)
172            .addLast(new NettyRpcServerPreambleHandler(NettyRpcServer.this, conn))
173            // We need NettyRpcServerResponseEncoder here because NettyRpcServerPreambleHandler may
174            // send RpcResponse to client.
175            .addLast(NettyRpcServerResponseEncoder.NAME, new NettyRpcServerResponseEncoder(metrics))
176            // Add writability handler after the response encoder, so we can abort writes before
177            // they get encoded, if the fatal threshold is exceeded. We pass in suppliers here so
178            // that the handler configs can be live updated via update_config.
179            .addLast(NettyRpcServerChannelWritabilityHandler.NAME,
180              new NettyRpcServerChannelWritabilityHandler(metrics, () -> writeBufferFatalThreshold,
181                () -> isWritabilityBackpressureEnabled()));
182        }
183      });
184    try {
185      serverChannel = bootstrap.bind(this.bindAddress).sync().channel();
186      LOG.info("Bind to {}", serverChannel.localAddress());
187    } catch (InterruptedException e) {
188      throw new InterruptedIOException(e.getMessage());
189    }
190    initReconfigurable(conf);
191    this.scheduler.init(new RpcSchedulerContext(this));
192  }

200  private void configureNettyWatermarks(Configuration conf) {
201    int watermarkLow =
202      conf.getInt(CHANNEL_WRITABLE_LOW_WATERMARK_KEY, CHANNEL_WRITABLE_LOW_WATERMARK_DEFAULT);
203    int watermarkHigh =
204      conf.getInt(CHANNEL_WRITABLE_HIGH_WATERMARK_KEY, CHANNEL_WRITABLE_HIGH_WATERMARK_DEFAULT);
205    int fatalThreshold =
206      conf.getInt(CHANNEL_WRITABLE_FATAL_WATERMARK_KEY, CHANNEL_WRITABLE_FATAL_WATERMARK_DEFAULT);
207
208    WriteBufferWaterMark oldWaterMark = writeBufferWaterMark;
209    int oldFatalThreshold = writeBufferFatalThreshold;
210
211    boolean disabled = false;
212    if (watermarkHigh == 0 && watermarkLow == 0) {
213      // if both are 0, use the netty default, which we will treat as "disabled".
214      // when disabled, we won't manage autoRead in response to writability changes.
215      writeBufferWaterMark = WriteBufferWaterMark.DEFAULT;
216      disabled = true;
217    } else {
218      // netty checks pendingOutboundBytes < watermarkLow. It can never be less than 0, so set to
219      // 1 to avoid confusing behavior.
220      if (watermarkLow == 0) {
221        LOG.warn(
222          "Detected a {} value of 0, which is impossible to achieve "
223            + "due to how netty evaluates these thresholds, setting to 1",
224          CHANNEL_WRITABLE_LOW_WATERMARK_KEY);
225        watermarkLow = 1;
226      }
227
228      // netty validates the watermarks and throws an exception if high < low, fail more gracefully
229      // by disabling the watermarks and warning.
230      if (watermarkHigh <= watermarkLow) {
231        LOG.warn(
232          "Detected {} value {}, lower than {} value {}. This will fail netty validation, "
233            + "so disabling",
234          CHANNEL_WRITABLE_HIGH_WATERMARK_KEY, watermarkHigh, CHANNEL_WRITABLE_LOW_WATERMARK_KEY,
235          watermarkLow);
236        writeBufferWaterMark = WriteBufferWaterMark.DEFAULT;
237      } else {
238        writeBufferWaterMark = new WriteBufferWaterMark(watermarkLow, watermarkHigh);
239      }
240
241      // only apply this check when watermark is enabled. this way we give the operator some
242      // flexibility if they want to try enabling fatal threshold without backpressure.
243      if (fatalThreshold > 0 && fatalThreshold <= watermarkHigh) {
244        LOG.warn("Detected a {} value of {}, which is lower than the {} value of {}, ignoring.",
245          CHANNEL_WRITABLE_FATAL_WATERMARK_KEY, fatalThreshold, CHANNEL_WRITABLE_HIGH_WATERMARK_KEY,
246          watermarkHigh);
247        fatalThreshold = 0;
248      }
249    }
250
251    writeBufferFatalThreshold = fatalThreshold;
252
253    if (
254      oldWaterMark != null && (oldWaterMark.low() != writeBufferWaterMark.low()
255        || oldWaterMark.high() != writeBufferWaterMark.high()
256        || oldFatalThreshold != writeBufferFatalThreshold)
257    ) {
258      LOG.info("Updated netty outbound write buffer watermarks: low={}, high={}, fatal={}",
259        disabled ? "disabled" : writeBufferWaterMark.low(),
260        disabled ? "disabled" : writeBufferWaterMark.high(),
261        writeBufferFatalThreshold <= 0 ? "disabled" : writeBufferFatalThreshold);
262    }
263
264    // update any existing channels
265    for (Channel channel : allChannels) {
266      channel.config().setWriteBufferWaterMark(writeBufferWaterMark);
267      // if disabling watermark, set auto read to true in case channel had been exceeding
268      // previous watermark
269      if (disabled) {
270        channel.config().setAutoRead(true);
271      }
272    }
273  }
274
275  public boolean isWritabilityBackpressureEnabled() {
276    return writeBufferWaterMark != WriteBufferWaterMark.DEFAULT;
277  }

279  private ByteBufAllocator getChannelAllocator(Configuration conf) throws IOException {
280    final String value = conf.get(HBASE_NETTY_ALLOCATOR_KEY);
281    if (value != null) {
282      if (POOLED_ALLOCATOR_TYPE.equalsIgnoreCase(value)) {
283        LOG.info("Using {} for buffer allocation", PooledByteBufAllocator.class.getName());
284        return PooledByteBufAllocator.DEFAULT;
285      } else if (UNPOOLED_ALLOCATOR_TYPE.equalsIgnoreCase(value)) {
286        LOG.info("Using {} for buffer allocation", UnpooledByteBufAllocator.class.getName());
287        return UnpooledByteBufAllocator.DEFAULT;
288      } else if (HEAP_ALLOCATOR_TYPE.equalsIgnoreCase(value)) {
289        LOG.info("Using {} for buffer allocation", HeapByteBufAllocator.class.getName());
290        return HeapByteBufAllocator.DEFAULT;
291      } else {
292        // If the value is none of the recognized labels, treat it as a class name. This allows the
293        // user to supply a custom implementation, perhaps for debugging.
294        try {
295          // ReflectionUtils throws UnsupportedOperationException if there are any problems.
296          ByteBufAllocator alloc = (ByteBufAllocator) ReflectionUtils.newInstance(value);
297          LOG.info("Using {} for buffer allocation", value);
298          return alloc;
299        } catch (ClassCastException | UnsupportedOperationException e) {
300          throw new IOException(e);
301        }
302      }
303    } else {
304      LOG.info("Using {} for buffer allocation", PooledByteBufAllocator.class.getName());
305      return PooledByteBufAllocator.DEFAULT;
306    }
307  }

315  @Override
316  public synchronized void start() {
317    if (started) {
318      return;
319    }
320    authTokenSecretMgr = createSecretManager();
321    if (authTokenSecretMgr != null) {
322      // Start AuthenticationTokenSecretManager in synchronized way to avoid race conditions in
323      // LeaderElector start. See HBASE-25875
324      synchronized (authTokenSecretMgr) {
325        setSecretManager(authTokenSecretMgr);
326        authTokenSecretMgr.start();
327      }
328    }
329    this.authManager = new ServiceAuthorizationManager();
330    HBasePolicyProvider.init(conf, authManager);
331    scheduler.start();
332    started = true;
333  }
334
335  @Override
336  public synchronized void stop() {
337    if (!running) {
338      return;
339    }
340    LOG.info("Stopping server on " + this.serverChannel.localAddress());
341    FileChangeWatcher ks = keyStoreWatcher.getAndSet(null);
342    if (ks != null) {
343      ks.stop();
344    }
345    FileChangeWatcher ts = trustStoreWatcher.getAndSet(null);
346    if (ts != null) {
347      ts.stop();
348    }
349    if (authTokenSecretMgr != null) {
350      authTokenSecretMgr.stop();
351      authTokenSecretMgr = null;
352    }
353    allChannels.close().awaitUninterruptibly();
354    serverChannel.close();
355    scheduler.stop();
356    closed.countDown();
357    running = false;
358  }
359
360  @Override
361  public synchronized void join() throws InterruptedException {
362    closed.await();
363  }

379  private void initSSL(ChannelPipeline p, boolean supportPlaintext)
380    throws X509Exception, IOException {
381    SslContext nettySslContext = getSslContext();
382
383    if (supportPlaintext) {
384      p.addLast("ssl", new OptionalSslHandler(nettySslContext));
385      LOG.debug("Dual mode SSL handler added for channel: {}", p.channel());
386    } else {
387      SocketAddress remoteAddress = p.channel().remoteAddress();
388      SslHandler sslHandler;
389
390      if (remoteAddress instanceof InetSocketAddress) {
391        InetSocketAddress remoteInetAddress = (InetSocketAddress) remoteAddress;
392        String host;
393
394        if (conf.getBoolean(TLS_CONFIG_REVERSE_DNS_LOOKUP_ENABLED, true)) {
395          host = remoteInetAddress.getHostName();
396        } else {
397          host = remoteInetAddress.getHostString();
398        }
399
400        int port = remoteInetAddress.getPort();
401
402        /*
403         * our HostnameVerifier gets the host name from SSLEngine, so we have to construct the
404         * engine properly by passing the remote address
405         */
406        sslHandler = nettySslContext.newHandler(p.channel().alloc(), host, port);
407      } else {
408        sslHandler = nettySslContext.newHandler(p.channel().alloc());
409      }
410
411      p.addLast("ssl", sslHandler);
412      LOG.debug("SSL handler added for channel: {}", p.channel());
413    }
414  }
415
416  SslContext getSslContext() throws X509Exception, IOException {
417    SslContext result = sslContextForServer.get();
418    if (result == null) {
419      result = X509Util.createSslContextForServer(conf);
420      if (!sslContextForServer.compareAndSet(null, result)) {
421        // lost the race, another thread already set the value
422        result = sslContextForServer.get();
423      } else if (
424        keyStoreWatcher.get() == null && trustStoreWatcher.get() == null
425          && conf.getBoolean(X509Util.TLS_CERT_RELOAD, false)
426      ) {
427        X509Util.enableCertFileReloading(conf, keyStoreWatcher, trustStoreWatcher,
428          () -> sslContextForServer.set(null));
429      }
430    }
431    return result;
432  }

438  public Pair<Long, Long> getTotalAndMaxNettyOutboundBytes() {
439    long total = 0;
440    long max = 0;
441    for (Channel channel : allChannels) {
442      long outbound = NettyUnsafeUtils.getTotalPendingOutboundBytes(channel);
443      total += outbound;
444      max = Math.max(max, outbound);
445    }
446    return Pair.newPair(total, max);
447  }
