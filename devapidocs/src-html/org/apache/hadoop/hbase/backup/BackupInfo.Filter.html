164  public BackupInfo(String backupId, BackupType type, TableName[] tables, String targetRootDir) {
165    this();
166    this.backupId = backupId;
167    this.type = type;
168    this.backupRootDir = targetRootDir;
169    this.addTables(tables);
170    if (type == BackupType.INCREMENTAL) {
171      setHLogTargetDir(BackupUtils.getLogBackupDir(targetRootDir, backupId));
172    }
173    this.startTs = 0;
174    this.completeTs = 0;
175  }

373  public TableName getTableBySnapshot(String snapshotName) {
374    for (Entry<TableName, BackupTableInfo> entry : this.backupTableInfoMap.entrySet()) {
375      if (snapshotName.equals(entry.getValue().getSnapshotName())) {
376        return entry.getKey();
377      }
378    }
379    return null;
380  }

382  public BackupProtos.BackupInfo toProtosBackupInfo() {
383    BackupProtos.BackupInfo.Builder builder = BackupProtos.BackupInfo.newBuilder();
384    builder.setBackupId(getBackupId());
385    setBackupTableInfoMap(builder);
386    setTableSetTimestampMap(builder);
387    builder.setCompleteTs(getCompleteTs());
388    if (getFailedMsg() != null) {
389      builder.setFailedMessage(getFailedMsg());
390    }
391    if (getState() != null) {
392      builder.setBackupState(BackupProtos.BackupInfo.BackupState.valueOf(getState().name()));
393    }
394    if (getPhase() != null) {
395      builder.setBackupPhase(BackupProtos.BackupInfo.BackupPhase.valueOf(getPhase().name()));
396    }
397
398    builder.setProgress(getProgress());
399    builder.setStartTs(getStartTs());
400    builder.setBackupRootDir(getBackupRootDir());
401    builder.setBackupType(BackupProtos.BackupType.valueOf(getType().name()));
402    builder.setWorkersNumber(workers);
403    builder.setBandwidth(bandwidth);
404    return builder.build();
405  }

407  @Override
408  public int hashCode() {
409    int hash = 33 * type.hashCode() + backupId != null ? backupId.hashCode() : 0;
410    if (backupRootDir != null) {
411      hash = 33 * hash + backupRootDir.hashCode();
412    }
413    hash = 33 * hash + state.hashCode();
414    hash = 33 * hash + phase.hashCode();
415    hash = 33 * hash + (int) (startTs ^ (startTs >>> 32));
416    hash = 33 * hash + (int) (completeTs ^ (completeTs >>> 32));
417    hash = 33 * hash + (int) (totalBytesCopied ^ (totalBytesCopied >>> 32));
418    if (hlogTargetDir != null) {
419      hash = 33 * hash + hlogTargetDir.hashCode();
420    }
421    return hash;
422  }

424  @Override
425  public boolean equals(Object obj) {
426    if (obj instanceof BackupInfo) {
427      BackupInfo other = (BackupInfo) obj;
428      try {
429        return Bytes.equals(toByteArray(), other.toByteArray());
430      } catch (IOException e) {
431        LOG.error(e.toString(), e);
432        return false;
433      }
434    } else {
435      return false;
436    }
437  }

454  private void setTableSetTimestampMap(Builder builder) {
455    if (this.getTableSetTimestampMap() != null) {
456      for (Entry<TableName, Map<String, Long>> entry : this.getTableSetTimestampMap().entrySet()) {
457        builder.putTableSetTimestamp(entry.getKey().getNameAsString(),

472  public static BackupInfo fromProto(BackupProtos.BackupInfo proto) {
473    BackupInfo context = new BackupInfo();
474    context.setBackupId(proto.getBackupId());
475    context.setBackupTableInfoMap(toMap(proto.getBackupTableInfoList()));
476    context.setTableSetTimestampMap(getTableSetTimestampMap(proto.getTableSetTimestampMap()));
477    context.setCompleteTs(proto.getCompleteTs());
478    if (proto.hasFailedMessage()) {
479      context.setFailedMsg(proto.getFailedMessage());
480    }
481    if (proto.hasBackupState()) {
482      context.setState(BackupInfo.BackupState.valueOf(proto.getBackupState().name()));
483    }
484
485    context.setHLogTargetDir(BackupUtils.getLogBackupDir(proto.getBackupRootDir(),
486      proto.getBackupId()));
487
488    if (proto.hasBackupPhase()) {
489      context.setPhase(BackupPhase.valueOf(proto.getBackupPhase().name()));
490    }
491    if (proto.hasProgress()) {
492      context.setProgress(proto.getProgress());
493    }
494    context.setStartTs(proto.getStartTs());
495    context.setBackupRootDir(proto.getBackupRootDir());
496    context.setType(BackupType.valueOf(proto.getBackupType().name()));
497    context.setWorkers(proto.getWorkersNumber());
498    context.setBandwidth(proto.getBandwidth());
499    return context;
500  }

502  private static Map<TableName, BackupTableInfo> toMap(List<BackupProtos.BackupTableInfo> list) {
503    HashMap<TableName, BackupTableInfo> map = new HashMap<>();
504    for (BackupProtos.BackupTableInfo tbs : list) {
505      map.put(ProtobufUtil.toTableName(tbs.getTableName()), BackupTableInfo.convert(tbs));
506    }
507    return map;
508  }
509
510  private static Map<TableName, Map<String, Long>> getTableSetTimestampMap(
511    Map<String, BackupProtos.BackupInfo.RSTimestampMap> map) {
512    Map<TableName, Map<String, Long>> tableSetTimestampMap = new HashMap<>();
513    for (Entry<String, BackupProtos.BackupInfo.RSTimestampMap> entry : map.entrySet()) {
514      tableSetTimestampMap
515        .put(TableName.valueOf(entry.getKey()), entry.getValue().getRsTimestampMap());
516    }
517
518    return tableSetTimestampMap;
519  }

521  public String getShortDescription() {
522    StringBuilder sb = new StringBuilder();
523    sb.append("{");
524    sb.append("ID=" + backupId).append(",");
525    sb.append("Type=" + getType()).append(",");
526    sb.append("Tables=" + getTableListAsString()).append(",");
527    sb.append("State=" + getState()).append(",");
528    Date date = null;
529    Calendar cal = Calendar.getInstance();
530    cal.setTimeInMillis(getStartTs());
531    date = cal.getTime();
532    sb.append("Start time=" + date).append(",");
533    if (state == BackupState.FAILED) {
534      sb.append("Failed message=" + getFailedMsg()).append(",");
535    } else if (state == BackupState.RUNNING) {
536      sb.append("Phase=" + getPhase()).append(",");
537    } else if (state == BackupState.COMPLETE) {
538      cal = Calendar.getInstance();
539      cal.setTimeInMillis(getCompleteTs());
540      date = cal.getTime();
541      sb.append("End time=" + date).append(",");
542    }
543    sb.append("Progress=" + getProgress() + "%");
544    sb.append("}");
545
546    return sb.toString();
547  }

564  /**
565   * We use only time stamps to compare objects during sort operation
566   */
567  @Override
568  public int compareTo(BackupInfo o) {
569    Long thisTS =
570        Long.valueOf(this.getBackupId().substring(this.getBackupId().lastIndexOf("_") + 1));
571    Long otherTS = Long.valueOf(o.getBackupId().substring(o.getBackupId().lastIndexOf("_") + 1));
572    return thisTS.compareTo(otherTS);
573  }
