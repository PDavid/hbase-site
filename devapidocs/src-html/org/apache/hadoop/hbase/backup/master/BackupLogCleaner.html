051/**
052 * Implementation of a log cleaner that checks if a log is still scheduled for incremental backup
053 * before deleting it when its TTL is over.
054 */
055@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.CONFIG)
056public class BackupLogCleaner extends BaseLogCleanerDelegate {
057  private static final Logger LOG = LoggerFactory.getLogger(BackupLogCleaner.class);
058
059  private boolean stopped = false;
060  private Connection conn;
061
062  public BackupLogCleaner() {
063  }
064
065  @Override
066  public void init(Map<String, Object> params) {
067    MasterServices master = (MasterServices) MapUtils.getObject(params, HMaster.MASTER);
068    if (master != null) {
069      conn = master.getConnection();
070      if (getConf() == null) {
071        super.setConf(conn.getConfiguration());
072      }
073    }
074    if (conn == null) {
075      try {
076        conn = ConnectionFactory.createConnection(getConf());
077      } catch (IOException ioe) {
078        throw new RuntimeException("Failed to create connection", ioe);
079      }
080    }
081  }
082
083  private Map<Address, Long> getServersToOldestBackupMapping(List<BackupInfo> backups)
084    throws IOException {
085    Map<Address, Long> serverAddressToLastBackupMap = new HashMap<>();
086
087    Map<TableName, Long> tableNameBackupInfoMap = new HashMap<>();
088    for (BackupInfo backupInfo : backups) {
089      for (TableName table : backupInfo.getTables()) {
090        tableNameBackupInfoMap.putIfAbsent(table, backupInfo.getStartTs());
091        if (tableNameBackupInfoMap.get(table) <= backupInfo.getStartTs()) {
092          tableNameBackupInfoMap.put(table, backupInfo.getStartTs());
093          for (Map.Entry<String, Long> entry : backupInfo.getTableSetTimestampMap().get(table)
094            .entrySet()) {
095            serverAddressToLastBackupMap.put(Address.fromString(entry.getKey()), entry.getValue());
096          }
097        }
098      }
099    }
100
101    return serverAddressToLastBackupMap;
102  }
103
104  @Override
105  public Iterable<FileStatus> getDeletableFiles(Iterable<FileStatus> files) {
106    List<FileStatus> filteredFiles = new ArrayList<>();
107
108    // all members of this class are null if backup is disabled,
109    // so we cannot filter the files
110    if (this.getConf() == null || !BackupManager.isBackupEnabled(getConf())) {
111      LOG.debug("Backup is not enabled. Check your {} setting",
112        BackupRestoreConstants.BACKUP_ENABLE_KEY);
113      return files;
114    }
115
116    Map<Address, Long> addressToLastBackupMap;
117    try {
118      try (BackupManager backupManager = new BackupManager(conn, getConf())) {
119        addressToLastBackupMap =
120          getServersToOldestBackupMapping(backupManager.getBackupHistory(true));
121      }
122    } catch (IOException ex) {
123      LOG.error("Failed to analyse backup history with exception: {}. Retaining all logs",
124        ex.getMessage(), ex);
125      return Collections.emptyList();
126    }
127    for (FileStatus file : files) {
128      if (canDeleteFile(addressToLastBackupMap, file.getPath())) {
129        filteredFiles.add(file);
130      }
131    }
132
133    LOG.info("Total files: {}, Filtered Files: {}", IterableUtils.size(files),
134      filteredFiles.size());
135    return filteredFiles;
136  }
137
138  @Override
139  public void setConf(Configuration config) {
140    // If backup is disabled, keep all members null
141    super.setConf(config);
142    if (
143      !config.getBoolean(BackupRestoreConstants.BACKUP_ENABLE_KEY,
144        BackupRestoreConstants.BACKUP_ENABLE_DEFAULT)
145    ) {
146      LOG.warn("Backup is disabled - allowing all wals to be deleted");
147    }
148  }
149
150  @Override
151  public void stop(String why) {
152    if (!this.stopped) {
153      this.stopped = true;
154      LOG.info("Stopping BackupLogCleaner");
155    }
156  }
157
158  @Override
159  public boolean isStopped() {
160    return this.stopped;
161  }
162
163  protected static boolean canDeleteFile(Map<Address, Long> addressToLastBackupMap, Path path) {
164    if (isHMasterWAL(path)) {
165      return true;
166    }
167
168    try {
169      String hostname = BackupUtils.parseHostNameFromLogFile(path);
170      if (hostname == null) {
171        LOG.warn(
172          "Cannot parse hostname from RegionServer WAL file: {}. Ignoring cleanup of this log",
173          path);
174        return false;
175      }
176      Address walServerAddress = Address.fromString(hostname);
177      long walTimestamp = AbstractFSWALProvider.getTimestamp(path.getName());
178
179      if (
180        !addressToLastBackupMap.containsKey(walServerAddress)
181          || addressToLastBackupMap.get(walServerAddress) >= walTimestamp
182      ) {
183        return true;
184      }
185    } catch (Exception ex) {
186      LOG.warn("Error occurred while filtering file: {}. Ignoring cleanup of this log", path, ex);
187      return false;
188    }
189    return false;
190  }
191
192  private static boolean isHMasterWAL(Path path) {
193    String fn = path.getName();
194    return fn.startsWith(WALProcedureStore.LOG_PREFIX)
195      || fn.endsWith(MasterRegionFactory.ARCHIVED_WAL_SUFFIX);
196  }
197}
