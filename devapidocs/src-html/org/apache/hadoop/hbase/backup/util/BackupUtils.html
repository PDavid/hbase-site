079  /**
080   * Loop through the RS log timestamp map for the tables, for each RS, find the min timestamp value
081   * for the RS among the tables.
082   * @param rsLogTimestampMap timestamp map
083   * @return the min timestamp of each RS
084   */
085  public static HashMap<String, Long> getRSLogTimestampMins(
086      HashMap<TableName, HashMap<String, Long>> rsLogTimestampMap) {
087    if (rsLogTimestampMap == null || rsLogTimestampMap.isEmpty()) {
088      return null;
089    }
090
091    HashMap<String, Long> rsLogTimestampMins = new HashMap<>();
092    HashMap<String, HashMap<TableName, Long>> rsLogTimestampMapByRS = new HashMap<>();
093
094    for (Entry<TableName, HashMap<String, Long>> tableEntry : rsLogTimestampMap.entrySet()) {
095      TableName table = tableEntry.getKey();
096      HashMap<String, Long> rsLogTimestamp = tableEntry.getValue();
097      for (Entry<String, Long> rsEntry : rsLogTimestamp.entrySet()) {
098        String rs = rsEntry.getKey();
099        Long ts = rsEntry.getValue();
100        if (!rsLogTimestampMapByRS.containsKey(rs)) {
101          rsLogTimestampMapByRS.put(rs, new HashMap<>());
102          rsLogTimestampMapByRS.get(rs).put(table, ts);
103        } else {
104          rsLogTimestampMapByRS.get(rs).put(table, ts);
105        }
106      }
107    }
108
109    for (Entry<String, HashMap<TableName, Long>> entry : rsLogTimestampMapByRS.entrySet()) {
110      String rs = entry.getKey();
111      rsLogTimestampMins.put(rs, BackupUtils.getMinValue(entry.getValue()));
112    }
113
114    return rsLogTimestampMins;
115  }

117  /**
118   * copy out Table RegionInfo into incremental backup image need to consider move this logic into
119   * HBackupFileSystem
120   * @param conn connection
121   * @param backupInfo backup info
122   * @param conf configuration
123   * @throws IOException exception
124   */
125  public static void copyTableRegionInfo(Connection conn, BackupInfo backupInfo, Configuration conf)
126          throws IOException {
127    Path rootDir = CommonFSUtils.getRootDir(conf);
128    FileSystem fs = rootDir.getFileSystem(conf);
129
130    // for each table in the table set, copy out the table info and region
131    // info files in the correct directory structure
132    try (Admin admin = conn.getAdmin()) {
133      for (TableName table : backupInfo.getTables()) {
134        if (!admin.tableExists(table)) {
135          LOG.warn("Table " + table + " does not exists, skipping it.");
136          continue;
137        }
138        TableDescriptor orig = FSTableDescriptors.getTableDescriptorFromFs(fs, rootDir, table);
139
140        // write a copy of descriptor to the target directory
141        Path target = new Path(backupInfo.getTableBackupDir(table));
142        FileSystem targetFs = target.getFileSystem(conf);
143        FSTableDescriptors descriptors =
144          new FSTableDescriptors(targetFs, CommonFSUtils.getRootDir(conf));
145        descriptors.createTableDescriptorForTableDirectory(target, orig, false);
146        LOG.debug("Attempting to copy table info for:" + table + " target: " + target +
147          " descriptor: " + orig);
148        LOG.debug("Finished copying tableinfo.");
149        List<RegionInfo> regions = MetaTableAccessor.getTableRegions(conn, table);
150        // For each region, write the region info to disk
151        LOG.debug("Starting to write region info for table " + table);
152        for (RegionInfo regionInfo : regions) {
153          Path regionDir = FSUtils
154            .getRegionDirFromTableDir(new Path(backupInfo.getTableBackupDir(table)), regionInfo);
155          regionDir = new Path(backupInfo.getTableBackupDir(table), regionDir.getName());
156          writeRegioninfoOnFilesystem(conf, targetFs, regionDir, regionInfo);
157        }
158        LOG.debug("Finished writing region info for table " + table);
159      }
160    }
161  }

163  /**
164   * Write the .regioninfo file on-disk.
165   */
166  public static void writeRegioninfoOnFilesystem(final Configuration conf, final FileSystem fs,
167      final Path regionInfoDir, RegionInfo regionInfo) throws IOException {
168    final byte[] content = RegionInfo.toDelimitedByteArray(regionInfo);
169    Path regionInfoFile = new Path(regionInfoDir, "." + HConstants.REGIONINFO_QUALIFIER_STR);
170    // First check to get the permissions
171    FsPermission perms = CommonFSUtils.getFilePermissions(fs, conf, HConstants.DATA_FILE_UMASK_KEY);
172    // Write the RegionInfo file content
173    FSDataOutputStream out = FSUtils.create(conf, fs, regionInfoFile, perms, null);
174    try {
175      out.write(content);
176    } finally {
177      out.close();
178    }
179  }

183   * @param p path to WAL file
184   * @return hostname:port

186  public static String parseHostNameFromLogFile(Path p) {
187    try {

190      } else {
191        ServerName sname = AbstractFSWALProvider.getServerNameFromWALDirectoryName(p);

193          return sname.getAddress().toString();
194        } else {
195          LOG.error("Skip log file (can't parse): " + p);
196          return null;
197        }
198      }
199    } catch (Exception e) {
200      LOG.error("Skip log file (can't parse): " + p, e);
201      return null;
202    }
203  }

224   * Get the total length of files under the given directory recursively.
225   * @param fs The hadoop file system
226   * @param dir The target directory
227   * @return the total length of files
228   * @throws IOException exception
229   */
230  public static long getFilesLength(FileSystem fs, Path dir) throws IOException {
231    long totalLength = 0;
232    FileStatus[] files = CommonFSUtils.listStatus(fs, dir);
233    if (files != null) {
234      for (FileStatus fileStatus : files) {
235        if (fileStatus.isDirectory()) {
236          totalLength += getFilesLength(fs, fileStatus.getPath());
237        } else {
238          totalLength += fileStatus.getLen();
239        }
240      }
241    }
242    return totalLength;
243  }

245  /**
246   * Get list of all old WAL files (WALs and archive)
247   * @param c configuration
248   * @param hostTimestampMap {host,timestamp} map
249   * @return list of WAL files
250   * @throws IOException exception
251   */
252  public static List<String> getWALFilesOlderThan(final Configuration c,
253      final HashMap<String, Long> hostTimestampMap) throws IOException {
254    Path walRootDir = CommonFSUtils.getWALRootDir(c);
255    Path logDir = new Path(walRootDir, HConstants.HREGION_LOGDIR_NAME);
256    Path oldLogDir = new Path(walRootDir, HConstants.HREGION_OLDLOGDIR_NAME);
257    List<String> logFiles = new ArrayList<>();
258
259    PathFilter filter = p -> {
260      try {
261        if (AbstractFSWALProvider.isMetaFile(p)) {
262          return false;
263        }
264        String host = parseHostNameFromLogFile(p);
265        if (host == null) {
266          return false;
267        }
268        Long oldTimestamp = hostTimestampMap.get(host);
269        Long currentLogTS = BackupUtils.getCreationTime(p);
270        return currentLogTS <= oldTimestamp;
271      } catch (Exception e) {
272        LOG.warn("Can not parse" + p, e);
273        return false;
274      }
275    };
276    FileSystem walFs = CommonFSUtils.getWALFileSystem(c);
277    logFiles = BackupUtils.getFiles(walFs, logDir, logFiles, filter);
278    logFiles = BackupUtils.getFiles(walFs, oldLogDir, logFiles, filter);
279    return logFiles;
280  }

297   * @param backupStr backup
298   * @param conf configuration
299   * @return Yes if path exists
300   * @throws IOException exception
301   */
302  public static boolean checkPathExist(String backupStr, Configuration conf) throws IOException {
303    boolean isExist = false;
304    Path backupPath = new Path(backupStr);
305    FileSystem fileSys = backupPath.getFileSystem(conf);
306    String targetFsScheme = fileSys.getUri().getScheme();
307    if (LOG.isTraceEnabled()) {
308      LOG.trace("Schema of given url: " + backupStr + " is: " + targetFsScheme);
309    }
310    if (fileSys.exists(backupPath)) {
311      isExist = true;
312    }
313    return isExist;
314  }

317   * Check target path first, confirm it doesn't exist before backup
318   * @param backupRootPath backup destination path
319   * @param conf configuration
320   * @throws IOException exception
321   */
322  public static void checkTargetDir(String backupRootPath, Configuration conf) throws IOException {
323    boolean targetExists;
324    try {
325      targetExists = checkPathExist(backupRootPath, conf);
326    } catch (IOException e) {
327      String expMsg = e.getMessage();
328      String newMsg = null;
329      if (expMsg.contains("No FileSystem for scheme")) {
330        newMsg =
331            "Unsupported filesystem scheme found in the backup target url. Error Message: "
332                + expMsg;
333        LOG.error(newMsg);
334        throw new IOException(newMsg);
335      } else {
336        throw e;
337      }
338    }
339
340    if (targetExists) {
341      LOG.info("Using existing backup root dir: " + backupRootPath);
342    } else {
343      LOG.info("Backup root dir " + backupRootPath + " does not exist. Will be created.");
344    }
345  }

381   * Given the log file, parse the timestamp from the file name. The timestamp is the last number.
382   * @param p a path to the log file
383   * @return the timestamp
384   * @throws IOException exception
385   */
386  public static Long getCreationTime(Path p) throws IOException {
387    int idx = p.getName().lastIndexOf(LOGNAME_SEPARATOR);
388    if (idx < 0) {
389      throw new IOException("Cannot parse timestamp from path " + p);
390    }
391    String ts = p.getName().substring(idx + 1);
392    return Long.parseLong(ts);
393  }

395  public static List<String> getFiles(FileSystem fs, Path rootDir, List<String> files,
396      PathFilter filter) throws IOException {
397    RemoteIterator<LocatedFileStatus> it = fs.listFiles(rootDir, true);
398
399    while (it.hasNext()) {
400      LocatedFileStatus lfs = it.next();
401      if (lfs.isDirectory()) {
402        continue;
403      }
404      // apply filter
405      if (filter.accept(lfs.getPath())) {
406        files.add(lfs.getPath().toString());
407      }
408    }
409    return files;
410  }

418   * Clean up directories which are generated when DistCp copying hlogs
419   * @param backupInfo backup info
420   * @param conf configuration
421   * @throws IOException exception
422   */
423  private static void cleanupHLogDir(BackupInfo backupInfo, Configuration conf) throws IOException {
424    String logDir = backupInfo.getHLogTargetDir();
425    if (logDir == null) {
426      LOG.warn("No log directory specified for " + backupInfo.getBackupId());
427      return;
428    }
429
430    Path rootPath = new Path(logDir).getParent();
431    FileSystem fs = FileSystem.get(rootPath.toUri(), conf);
432    FileStatus[] files = listStatus(fs, rootPath, null);
433    if (files == null) {
434      return;
435    }
436    for (FileStatus file : files) {
437      LOG.debug("Delete log files: " + file.getPath().getName());
438      fs.delete(file.getPath(), true);
439    }
440  }

442  private static void cleanupTargetDir(BackupInfo backupInfo, Configuration conf) {
443    try {
444      // clean up the data at target directory
445      LOG.debug("Trying to cleanup up target dir : " + backupInfo.getBackupId());
446      String targetDir = backupInfo.getBackupRootDir();
447      if (targetDir == null) {
448        LOG.warn("No target directory specified for " + backupInfo.getBackupId());
449        return;
450      }
451
452      FileSystem outputFs = FileSystem.get(new Path(backupInfo.getBackupRootDir()).toUri(), conf);
453
454      for (TableName table : backupInfo.getTables()) {
455        Path targetDirPath =
456            new Path(getTableBackupDir(backupInfo.getBackupRootDir(), backupInfo.getBackupId(),
457              table));
458        if (outputFs.delete(targetDirPath, true)) {
459          LOG.info("Cleaning up backup data at " + targetDirPath.toString() + " done.");
460        } else {
461          LOG.info("No data has been found in " + targetDirPath.toString() + ".");
462        }
463
464        Path tableDir = targetDirPath.getParent();
465        FileStatus[] backups = listStatus(outputFs, tableDir, null);
466        if (backups == null || backups.length == 0) {
467          outputFs.delete(tableDir, true);
468          LOG.debug(tableDir.toString() + " is empty, remove it.");
469        }
470      }
471      outputFs.delete(new Path(targetDir, backupInfo.getBackupId()), true);
472    } catch (IOException e1) {
473      LOG.error("Cleaning up backup data of " + backupInfo.getBackupId() + " at "
474          + backupInfo.getBackupRootDir() + " failed due to " + e1.getMessage() + ".");
475    }
476  }

512  /**
513   * Calls fs.listStatus() and treats FileNotFoundException as non-fatal This accommodates
514   * differences between hadoop versions, where hadoop 1 does not throw a FileNotFoundException, and
515   * return an empty FileStatus[] while Hadoop 2 will throw FileNotFoundException.
516   * @param fs file system
517   * @param dir directory
518   * @param filter path filter
519   * @return null if dir is empty or doesn't exist, otherwise FileStatus array
520   */
521  public static FileStatus[] listStatus(final FileSystem fs, final Path dir,
522          final PathFilter filter) throws IOException {
523    FileStatus[] status = null;
524    try {
525      status = filter == null ? fs.listStatus(dir) : fs.listStatus(dir, filter);
526    } catch (FileNotFoundException fnfe) {
527      // if directory doesn't exist, return null
528      if (LOG.isTraceEnabled()) {
529        LOG.trace(dir + " doesn't exist");
530      }
531    }
532
533    if (status == null || status.length < 1) {
534      return null;
535    }
536
537    return status;
538  }

565  private static List<BackupInfo> getHistory(Configuration conf, Path backupRootPath)
566      throws IOException {
567    // Get all (n) history from backup root destination
568
569    FileSystem fs = FileSystem.get(backupRootPath.toUri(), conf);
570    RemoteIterator<LocatedFileStatus> it = fs.listLocatedStatus(backupRootPath);
571
572    List<BackupInfo> infos = new ArrayList<>();
573    while (it.hasNext()) {
574      LocatedFileStatus lfs = it.next();
575
576      if (!lfs.isDirectory()) {
577        continue;
578      }
579
580      String backupId = lfs.getPath().getName();
581      try {
582        BackupInfo info = loadBackupInfo(backupRootPath, backupId, fs);
583        infos.add(info);
584      } catch (IOException e) {
585        LOG.error("Can not load backup info from: " + lfs.getPath(), e);
586      }
587    }
588    // Sort
589    Collections.sort(infos, new Comparator<BackupInfo>() {
590      @Override
591      public int compare(BackupInfo o1, BackupInfo o2) {
592        long ts1 = getTimestamp(o1.getBackupId());
593        long ts2 = getTimestamp(o2.getBackupId());
594
595        if (ts1 == ts2) {
596          return 0;
597        }
598
599        return ts1 < ts2 ? 1 : -1;
600      }
601
602      private long getTimestamp(String backupId) {
603        String[] split = backupId.split("_");
604        return Long.parseLong(split[1]);
605      }
606    });
607    return infos;
608  }

610  public static List<BackupInfo> getHistory(Configuration conf, int n, Path backupRootPath,
611      BackupInfo.Filter... filters) throws IOException {
612    List<BackupInfo> infos = getHistory(conf, backupRootPath);
613    List<BackupInfo> ret = new ArrayList<>();
614    for (BackupInfo info : infos) {
615      if (ret.size() == n) {
616        break;
617      }
618      boolean passed = true;
619      for (int i = 0; i < filters.length; i++) {
620        if (!filters[i].apply(info)) {
621          passed = false;
622          break;
623        }
624      }
625      if (passed) {
626        ret.add(info);
627      }
628    }
629    return ret;
630  }

632  public static BackupInfo loadBackupInfo(Path backupRootPath, String backupId, FileSystem fs)
633      throws IOException {
634    Path backupPath = new Path(backupRootPath, backupId);
635
636    RemoteIterator<LocatedFileStatus> it = fs.listFiles(backupPath, true);
637    while (it.hasNext()) {
638      LocatedFileStatus lfs = it.next();
639      if (lfs.getPath().getName().equals(BackupManifest.MANIFEST_FILE_NAME)) {
640        // Load BackupManifest
641        BackupManifest manifest = new BackupManifest(fs, lfs.getPath().getParent());
642        BackupInfo info = manifest.toBackupInfo();
643        return info;
644      }
645    }
646    return null;
647  }

668  public static boolean validate(HashMap<TableName, BackupManifest> backupManifestMap,
669      Configuration conf) throws IOException {
670    boolean isValid = true;
671
672    for (Entry<TableName, BackupManifest> manifestEntry : backupManifestMap.entrySet()) {
673      TableName table = manifestEntry.getKey();
674      TreeSet<BackupImage> imageSet = new TreeSet<>();
675
676      ArrayList<BackupImage> depList = manifestEntry.getValue().getDependentListByTable(table);
677      if (depList != null && !depList.isEmpty()) {
678        imageSet.addAll(depList);
679      }
680
681      LOG.info("Dependent image(s) from old to new:");
682      for (BackupImage image : imageSet) {
683        String imageDir =
684            HBackupFileSystem.getTableBackupDir(image.getRootDir(), image.getBackupId(), table);
685        if (!BackupUtils.checkPathExist(imageDir, conf)) {
686          LOG.error("ERROR: backup image does not exist: " + imageDir);
687          isValid = false;
688          break;
689        }
690        LOG.info("Backup image: " + image.getBackupId() + " for '" + table + "' is available");
691      }
692    }
693    return isValid;
694  }

726  public static BulkLoadHFiles createLoader(Configuration config) {
727    // set configuration for restore:
728    // LoadIncrementalHFile needs more time
729    // <name>hbase.rpc.timeout</name> <value>600000</value>
730    // calculates
731    Configuration conf = new Configuration(config);
732    conf.setInt(HConstants.HBASE_RPC_TIMEOUT_KEY, MILLISEC_IN_HOUR);
733
734    // By default, it is 32 and loader will fail if # of files in any region exceed this
735    // limit. Bad for snapshot restore.
736    conf.setInt(BulkLoadHFiles.MAX_FILES_PER_REGION_PER_FAMILY, Integer.MAX_VALUE);
737    conf.set(BulkLoadHFiles.IGNORE_UNMATCHED_CF_CONF_KEY, "yes");
738    return BulkLoadHFiles.create(conf);
739  }

741  public static String findMostRecentBackupId(String[] backupIds) {
742    long recentTimestamp = Long.MIN_VALUE;
743    for (String backupId : backupIds) {
744      long ts = Long.parseLong(backupId.split("_")[1]);
745      if (ts > recentTimestamp) {
746        recentTimestamp = ts;
747      }
748    }
749    return BackupRestoreConstants.BACKUPID_PREFIX + recentTimestamp;
750  }
