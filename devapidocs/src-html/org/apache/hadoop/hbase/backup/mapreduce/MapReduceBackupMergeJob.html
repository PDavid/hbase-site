055/**
056 * MapReduce implementation of {@link BackupMergeJob} Must be initialized with configuration of a
057 * backup destination cluster
058 */
059@InterfaceAudience.Private
060public class MapReduceBackupMergeJob implements BackupMergeJob {
061  public static final Logger LOG = LoggerFactory.getLogger(MapReduceBackupMergeJob.class);
062
063  protected Tool player;
064  protected Configuration conf;
065
066  public MapReduceBackupMergeJob() {
067  }
068
069  @Override
070  public Configuration getConf() {
071    return conf;
072  }
073
074  @Override
075  public void setConf(Configuration conf) {
076    this.conf = conf;
077  }

079  @Override
080  public void run(String[] backupIds) throws IOException {
081    String bulkOutputConfKey;
082
083    // TODO : run player on remote cluster
084    player = new MapReduceHFileSplitterJob();
085    bulkOutputConfKey = MapReduceHFileSplitterJob.BULK_OUTPUT_CONF_KEY;
086    // Player reads all files in arbitrary directory structure and creates
087    // a Map task for each file
088    String bids = StringUtils.join(backupIds, ",");
089
090    if (LOG.isDebugEnabled()) {
091      LOG.debug("Merge backup images " + bids);
092    }
093
094    List<Pair<TableName, Path>> processedTableList = new ArrayList<>();
095    boolean finishedTables = false;
096    Connection conn = ConnectionFactory.createConnection(getConf());
097    BackupSystemTable table = new BackupSystemTable(conn);
098    FileSystem fs = FileSystem.get(getConf());
099
100    try {
101
102      // Get exclusive lock on backup system
103      table.startBackupExclusiveOperation();
104      // Start merge operation
105      table.startMergeOperation(backupIds);
106
107      // Select most recent backup id
108      String mergedBackupId = BackupUtils.findMostRecentBackupId(backupIds);
109
110      TableName[] tableNames = getTableNamesInBackupImages(backupIds);
111
112      BackupInfo bInfo = table.readBackupInfo(backupIds[0]);
113      String backupRoot = bInfo.getBackupRootDir();
114
115      for (int i = 0; i < tableNames.length; i++) {
116        LOG.info("Merge backup images for " + tableNames[i]);
117
118        // Find input directories for table
119        Path[] dirPaths = findInputDirectories(fs, backupRoot, tableNames[i], backupIds);
120        String dirs = StringUtils.join(dirPaths, ",");
121
122        Path bulkOutputPath = BackupUtils.getBulkOutputDir(
123          BackupUtils.getFileNameCompatibleString(tableNames[i]), getConf(), false);
124        // Delete content if exists
125        if (fs.exists(bulkOutputPath)) {
126          if (!fs.delete(bulkOutputPath, true)) {
127            LOG.warn("Can not delete: " + bulkOutputPath);
128          }
129        }
130        Configuration conf = getConf();
131        conf.set(bulkOutputConfKey, bulkOutputPath.toString());
132        String[] playerArgs = { dirs, tableNames[i].getNameAsString() };
133
134        player.setConf(getConf());
135        int result = player.run(playerArgs);
136        if (!succeeded(result)) {
137          throw new IOException("Can not merge backup images for " + dirs
138            + " (check Hadoop/MR and HBase logs). Player return code =" + result);
139        }
140        // Add to processed table list
141        processedTableList.add(new Pair<>(tableNames[i], bulkOutputPath));
142        LOG.debug("Merge Job finished:" + result);
143      }
144      List<TableName> tableList = toTableNameList(processedTableList);
145      table.updateProcessedTablesForMerge(tableList);
146      finishedTables = true;
147
148      // PHASE 2 (modification of a backup file system)
149      // Move existing mergedBackupId data into tmp directory
150      // we will need it later in case of a failure
151      Path tmpBackupDir =
152        HBackupFileSystem.getBackupTmpDirPathForBackupId(backupRoot, mergedBackupId);
153      Path backupDirPath = HBackupFileSystem.getBackupPath(backupRoot, mergedBackupId);
154
155      if (!fs.rename(backupDirPath, tmpBackupDir)) {
156        throw new IOException("Failed to rename " + backupDirPath + " to " + tmpBackupDir);
157      } else {
158        LOG.debug("Renamed " + backupDirPath + " to " + tmpBackupDir);
159      }
160      // Move new data into backup dest
161      for (Pair<TableName, Path> tn : processedTableList) {
162        moveData(fs, backupRoot, tn.getSecond(), tn.getFirst(), mergedBackupId);
163      }
164      // Update backup manifest
165      List<String> backupsToDelete = getBackupIdsToDelete(backupIds, mergedBackupId);
166      updateBackupManifest(tmpBackupDir.getParent().toString(), mergedBackupId, backupsToDelete);
167      // Copy meta files back from tmp to backup dir
168      copyMetaData(fs, tmpBackupDir, backupDirPath);
169      // Delete tmp dir (Rename back during repair)
170      if (!fs.delete(tmpBackupDir, true)) {
171        // WARN and ignore
172        LOG.warn("Could not delete tmp dir: " + tmpBackupDir);
173      }
174      // Delete old data
175      deleteBackupImages(backupsToDelete, conn, fs, backupRoot);
176      // Finish merge session
177      table.finishMergeOperation();
178      // Release lock
179      table.finishBackupExclusiveOperation();
180    } catch (RuntimeException e) {
181
182      throw e;
183    } catch (Exception e) {
184      LOG.error(e.toString(), e);
185      if (!finishedTables) {
186        // cleanup bulk directories and finish merge
187        // merge MUST be repeated (no need for repair)
188        cleanupBulkLoadDirs(fs, toPathList(processedTableList));
189        table.finishMergeOperation();
190        table.finishBackupExclusiveOperation();
191        throw new IOException("Backup merge operation failed, you should try it again", e);
192      } else {
193        // backup repair must be run
194        throw new IOException(
195          "Backup merge operation failed, run backup repair tool to restore system's integrity", e);
196      }
197    } finally {
198      table.close();
199      conn.close();
200    }
201  }

203  /**
204   * Copy meta data to of a backup session
205   * @param fs            file system
206   * @param tmpBackupDir  temp backup directory, where meta is locaed
207   * @param backupDirPath new path for backup
208   * @throws IOException exception
209   */
210  protected void copyMetaData(FileSystem fs, Path tmpBackupDir, Path backupDirPath)
211    throws IOException {
212    RemoteIterator<LocatedFileStatus> it = fs.listFiles(tmpBackupDir, true);
213    List<Path> toKeep = new ArrayList<Path>();
214    while (it.hasNext()) {
215      Path p = it.next().getPath();
216      if (fs.isDirectory(p)) {
217        continue;
218      }
219      // Keep meta
220      String fileName = p.toString();
221      if (
222        fileName.indexOf(FSTableDescriptors.TABLEINFO_DIR) > 0
223          || fileName.indexOf(HRegionFileSystem.REGION_INFO_FILE) > 0
224      ) {
225        toKeep.add(p);
226      }
227    }
228    // Copy meta to destination
229    for (Path p : toKeep) {
230      Path newPath = convertToDest(p, backupDirPath);
231      copyFile(fs, p, newPath);
232    }
233  }
234
235  /**
236   * Copy file in DFS from p to newPath
237   * @param fs      file system
238   * @param p       old path
239   * @param newPath new path
240   * @throws IOException exception
241   */
242  protected void copyFile(FileSystem fs, Path p, Path newPath) throws IOException {
243    try (InputStream in = fs.open(p); OutputStream out = fs.create(newPath, true)) {
244      IOUtils.copy(in, out);
245    }
246    boolean exists = fs.exists(newPath);
247    if (!exists) {
248      throw new IOException("Failed to copy meta file to: " + newPath);
249    }
250  }
251
252  /**
253   * Converts path before copying
254   * @param p             path
255   * @param backupDirPath backup root
256   * @return converted path
257   */
258  protected Path convertToDest(Path p, Path backupDirPath) {
259    String backupId = backupDirPath.getName();
260    Stack<String> stack = new Stack<String>();
261    String name = null;
262    while (true) {
263      name = p.getName();
264      if (!name.equals(backupId)) {
265        stack.push(name);
266        p = p.getParent();
267      } else {
268        break;
269      }
270    }
271    Path newPath = new Path(backupDirPath.toString());
272    while (!stack.isEmpty()) {
273      newPath = new Path(newPath, stack.pop());
274    }
275    return newPath;
276  }

278  protected List<Path> toPathList(List<Pair<TableName, Path>> processedTableList) {
279    ArrayList<Path> list = new ArrayList<>();
280    for (Pair<TableName, Path> p : processedTableList) {
281      list.add(p.getSecond());
282    }
283    return list;
284  }
285
286  protected List<TableName> toTableNameList(List<Pair<TableName, Path>> processedTableList) {
287    ArrayList<TableName> list = new ArrayList<>();
288    for (Pair<TableName, Path> p : processedTableList) {
289      list.add(p.getFirst());
290    }
291    return list;
292  }
293
294  protected void cleanupBulkLoadDirs(FileSystem fs, List<Path> pathList) throws IOException {
295    for (Path path : pathList) {
296      if (!fs.delete(path, true)) {
297        LOG.warn("Can't delete " + path);
298      }
299    }
300  }
301
302  protected void updateBackupManifest(String backupRoot, String mergedBackupId,
303    List<String> backupsToDelete) throws IllegalArgumentException, IOException {
304    BackupManifest manifest =
305      HBackupFileSystem.getManifest(conf, new Path(backupRoot), mergedBackupId);
306    manifest.getBackupImage().removeAncestors(backupsToDelete);
307    // save back
308    manifest.store(conf);
309  }
310
311  protected void deleteBackupImages(List<String> backupIds, Connection conn, FileSystem fs,
312    String backupRoot) throws IOException {
313    // Delete from backup system table
314    try (BackupSystemTable table = new BackupSystemTable(conn)) {
315      for (String backupId : backupIds) {
316        table.deleteBackupInfo(backupId);
317      }
318    }
319
320    // Delete from file system
321    for (String backupId : backupIds) {
322      Path backupDirPath = HBackupFileSystem.getBackupPath(backupRoot, backupId);
323
324      if (!fs.delete(backupDirPath, true)) {
325        LOG.warn("Could not delete " + backupDirPath);
326      }
327    }
328  }
329
330  protected List<String> getBackupIdsToDelete(String[] backupIds, String mergedBackupId) {
331    List<String> list = new ArrayList<>();
332    for (String id : backupIds) {
333      if (id.equals(mergedBackupId)) {
334        continue;
335      }
336      list.add(id);
337    }
338    return list;
339  }
340
341  protected void moveData(FileSystem fs, String backupRoot, Path bulkOutputPath,
342    TableName tableName, String mergedBackupId) throws IllegalArgumentException, IOException {
343    Path dest =
344      new Path(HBackupFileSystem.getTableBackupDir(backupRoot, mergedBackupId, tableName));
345
346    FileStatus[] fsts = fs.listStatus(bulkOutputPath);
347    for (FileStatus fst : fsts) {
348      if (fst.isDirectory()) {
349        String family = fst.getPath().getName();
350        Path newDst = new Path(dest, family);
351        if (fs.exists(newDst)) {
352          if (!fs.delete(newDst, true)) {
353            throw new IOException("failed to delete :" + newDst);
354          }
355        } else {
356          fs.mkdirs(dest);
357        }
358        boolean result = fs.rename(fst.getPath(), dest);
359        LOG.debug("MoveData from " + fst.getPath() + " to " + dest + " result=" + result);
360      }
361    }
362  }
363
364  protected TableName[] getTableNamesInBackupImages(String[] backupIds) throws IOException {
365    Set<TableName> allSet = new HashSet<>();
366
367    try (Connection conn = ConnectionFactory.createConnection(conf);
368      BackupSystemTable table = new BackupSystemTable(conn)) {
369      for (String backupId : backupIds) {
370        BackupInfo bInfo = table.readBackupInfo(backupId);
371
372        allSet.addAll(bInfo.getTableNames());
373      }
374    }
375
376    TableName[] ret = new TableName[allSet.size()];
377    return allSet.toArray(ret);
378  }
379
380  protected Path[] findInputDirectories(FileSystem fs, String backupRoot, TableName tableName,
381    String[] backupIds) throws IOException {
382    List<Path> dirs = new ArrayList<>();
383
384    for (String backupId : backupIds) {
385      Path fileBackupDirPath =
386        new Path(HBackupFileSystem.getTableBackupDir(backupRoot, backupId, tableName));
387      if (fs.exists(fileBackupDirPath)) {
388        dirs.add(fileBackupDirPath);
389      } else {
390        if (LOG.isDebugEnabled()) {
391          LOG.debug("File: " + fileBackupDirPath + " does not exist.");
392        }
393      }
394    }
395    Path[] ret = new Path[dirs.size()];
396    return dirs.toArray(ret);
397  }
