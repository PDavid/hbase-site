074  /**
075   * Backup manager constructor.
076   * @param conn connection
077   * @param conf configuration
078   * @throws IOException exception
079   */
080  public BackupManager(Connection conn, Configuration conf) throws IOException {
081    if (!conf.getBoolean(BackupRestoreConstants.BACKUP_ENABLE_KEY,
082      BackupRestoreConstants.BACKUP_ENABLE_DEFAULT)) {
083      throw new BackupException("HBase backup is not enabled. Check your "
084          + BackupRestoreConstants.BACKUP_ENABLE_KEY + " setting.");
085    }
086    this.conf = conf;
087    this.conn = conn;
088    this.systemTable = new BackupSystemTable(conn);
089  }

098  /**
099   * This method modifies the master's configuration in order to inject backup-related features
100   * (TESTs only)
101   * @param conf configuration
102   */
103  public static void decorateMasterConfiguration(Configuration conf) {
104    if (!isBackupEnabled(conf)) {
105      return;
106    }
107    // Add WAL archive cleaner plug-in
108    String plugins = conf.get(HConstants.HBASE_MASTER_LOGCLEANER_PLUGINS);
109    String cleanerClass = BackupLogCleaner.class.getCanonicalName();
110    if (!plugins.contains(cleanerClass)) {
111      conf.set(HConstants.HBASE_MASTER_LOGCLEANER_PLUGINS, plugins + "," + cleanerClass);
112    }
113
114    String classes = conf.get(ProcedureManagerHost.MASTER_PROCEDURE_CONF_KEY);
115    String masterProcedureClass = LogRollMasterProcedureManager.class.getName();
116    if (classes == null) {
117      conf.set(ProcedureManagerHost.MASTER_PROCEDURE_CONF_KEY, masterProcedureClass);
118    } else if (!classes.contains(masterProcedureClass)) {
119      conf.set(ProcedureManagerHost.MASTER_PROCEDURE_CONF_KEY,
120        classes + "," + masterProcedureClass);
121    }
122
123    plugins = conf.get(HFileCleaner.MASTER_HFILE_CLEANER_PLUGINS);
124    conf.set(HFileCleaner.MASTER_HFILE_CLEANER_PLUGINS, (plugins == null ? "" : plugins + ",") +
125      BackupHFileCleaner.class.getName());
126    if (LOG.isDebugEnabled()) {
127      LOG.debug("Added log cleaner: {}. Added master procedure manager: {}."
128        +"Added master procedure manager: {}", cleanerClass, masterProcedureClass,
129        BackupHFileCleaner.class.getName());
130    }
131  }

133  /**
134   * This method modifies the Region Server configuration in order to inject backup-related features
135   * TESTs only.
136   * @param conf configuration
137   */
138  public static void decorateRegionServerConfiguration(Configuration conf) {
139    if (!isBackupEnabled(conf)) {
140      return;
141    }
142
143    String classes = conf.get(ProcedureManagerHost.REGIONSERVER_PROCEDURE_CONF_KEY);
144    String regionProcedureClass = LogRollRegionServerProcedureManager.class.getName();
145    if (classes == null) {
146      conf.set(ProcedureManagerHost.REGIONSERVER_PROCEDURE_CONF_KEY, regionProcedureClass);
147    } else if (!classes.contains(regionProcedureClass)) {
148      conf.set(ProcedureManagerHost.REGIONSERVER_PROCEDURE_CONF_KEY,
149        classes + "," + regionProcedureClass);
150    }
151    String coproc = conf.get(CoprocessorHost.REGION_COPROCESSOR_CONF_KEY);
152    String regionObserverClass = BackupObserver.class.getName();
153    conf.set(CoprocessorHost.REGION_COPROCESSOR_CONF_KEY,
154      (coproc == null ? "" : coproc + ",") + regionObserverClass);
155    if (LOG.isDebugEnabled()) {
156      LOG.debug("Added region procedure manager: {}. Added region observer: {}",
157        regionProcedureClass, regionObserverClass);
158    }
159  }

188  /**
189   * Creates a backup info based on input backup request.
190   * @param backupId backup id
191   * @param type type
192   * @param tableList table list
193   * @param targetRootDir root dir
194   * @param workers number of parallel workers
195   * @param bandwidth bandwidth per worker in MB per sec
196   * @return BackupInfo
197   * @throws BackupException exception
198   */
199  public BackupInfo createBackupInfo(String backupId, BackupType type, List<TableName> tableList,
200      String targetRootDir, int workers, long bandwidth) throws BackupException {
201    if (targetRootDir == null) {
202      throw new BackupException("Wrong backup request parameter: target backup root directory");
203    }
204
205    if (type == BackupType.FULL && (tableList == null || tableList.isEmpty())) {
206      // If table list is null for full backup, which means backup all tables. Then fill the table
207      // list with all user tables from meta. It no table available, throw the request exception.
208      List<TableDescriptor> htds = null;
209      try (Admin admin = conn.getAdmin()) {
210        htds = admin.listTableDescriptors();
211      } catch (Exception e) {
212        throw new BackupException(e);
213      }
214
215      if (htds == null) {
216        throw new BackupException("No table exists for full backup of all tables.");
217      } else {
218        tableList = new ArrayList<>();
219        for (TableDescriptor hTableDescriptor : htds) {
220          TableName tn = hTableDescriptor.getTableName();
221          if (tn.equals(BackupSystemTable.getTableName(conf))) {
222            // skip backup system table
223            continue;
224          }
225          tableList.add(hTableDescriptor.getTableName());
226        }
227
228        LOG.info("Full backup all the tables available in the cluster: {}", tableList);
229      }
230    }
231
232    // there are one or more tables in the table list
233    backupInfo = new BackupInfo(backupId, type, tableList.toArray(new TableName[tableList.size()]),
234      targetRootDir);
235    backupInfo.setBandwidth(bandwidth);
236    backupInfo.setWorkers(workers);
237    return backupInfo;
238  }

240  /**
241   * Check if any ongoing backup. Currently, we only reply on checking status in backup system
242   * table. We need to consider to handle the case of orphan records in the future. Otherwise, all
243   * the coming request will fail.
244   * @return the ongoing backup id if on going backup exists, otherwise null
245   * @throws IOException exception
246   */
247  private String getOngoingBackupId() throws IOException {
248    ArrayList<BackupInfo> sessions = systemTable.getBackupInfos(BackupState.RUNNING);
249    if (sessions.size() == 0) {
250      return null;
251    }
252    return sessions.get(0).getBackupId();
253  }

255  /**
256   * Start the backup manager service.
257   * @throws IOException exception
258   */
259  public void initialize() throws IOException {
260    String ongoingBackupId = this.getOngoingBackupId();
261    if (ongoingBackupId != null) {
262      LOG.info("There is a ongoing backup {}"
263        + ". Can not launch new backup until no ongoing backup remains.", ongoingBackupId);
264      throw new BackupException("There is ongoing backup seesion.");
265    }
266  }

272  /**
273   * Get direct ancestors of the current backup.
274   * @param backupInfo The backup info for the current backup
275   * @return The ancestors for the current backup
276   * @throws IOException exception
277   */
278  public ArrayList<BackupImage> getAncestors(BackupInfo backupInfo) throws IOException {
279    LOG.debug("Getting the direct ancestors of the current backup {}", backupInfo.getBackupId());
280
281    ArrayList<BackupImage> ancestors = new ArrayList<>();
282
283    // full backup does not have ancestor
284    if (backupInfo.getType() == BackupType.FULL) {
285      LOG.debug("Current backup is a full backup, no direct ancestor for it.");
286      return ancestors;
287    }
288
289    // get all backup history list in descending order
290    ArrayList<BackupInfo> allHistoryList = getBackupHistory(true);
291    for (BackupInfo backup : allHistoryList) {
292
293      BackupImage.Builder builder = BackupImage.newBuilder();
294
295      BackupImage image = builder.withBackupId(backup.getBackupId()).withType(backup.getType())
296          .withRootDir(backup.getBackupRootDir()).withTableList(backup.getTableNames())
297          .withStartTime(backup.getStartTs()).withCompleteTime(backup.getCompleteTs()).build();
298
299      // Only direct ancestors for a backup are required and not entire history of backup for this
300      // table resulting in verifying all of the previous backups which is unnecessary and backup
301      // paths need not be valid beyond the lifetime of a backup.
302      //
303      // RootDir is way of grouping a single backup including one full and many incremental backups
304      if (!image.getRootDir().equals(backupInfo.getBackupRootDir())) {
305        continue;
306      }
307
308      // add the full backup image as an ancestor until the last incremental backup
309      if (backup.getType().equals(BackupType.FULL)) {
310        // check the backup image coverage, if previous image could be covered by the newer ones,
311        // then no need to add
312        if (!BackupManifest.canCoverImage(ancestors, image)) {
313          ancestors.add(image);
314        }
315      } else {
316        // found last incremental backup, if previously added full backup ancestor images can cover
317        // it, then this incremental ancestor is not the dependent of the current incremental
318        // backup, that is to say, this is the backup scope boundary of current table set.
319        // Otherwise, this incremental backup ancestor is the dependent ancestor of the ongoing
320        // incremental backup
321        if (BackupManifest.canCoverImage(ancestors, image)) {
322          LOG.debug("Met the backup boundary of the current table set:");
323          for (BackupImage image1 : ancestors) {
324            LOG.debug("  BackupID={}, BackupDir={}", image1.getBackupId(),  image1.getRootDir());
325          }
326        } else {
327          Path logBackupPath =
328              HBackupFileSystem.getBackupPath(backup.getBackupRootDir(), backup.getBackupId());
329          LOG.debug("Current backup has an incremental backup ancestor, "
330              + "touching its image manifest in {}"
331              + " to construct the dependency.", logBackupPath.toString());
332          BackupManifest lastIncrImgManifest = new BackupManifest(conf, logBackupPath);
333          BackupImage lastIncrImage = lastIncrImgManifest.getBackupImage();
334          ancestors.add(lastIncrImage);
335
336          LOG.debug(
337            "Last dependent incremental backup image: {BackupID={}" +
338                "BackupDir={}}", lastIncrImage.getBackupId(), lastIncrImage.getRootDir());
339        }
340      }
341    }
342    LOG.debug("Got {} ancestors for the current backup.", ancestors.size());
343    return ancestors;
344  }

346  /**
347   * Get the direct ancestors of this backup for one table involved.
348   * @param backupInfo backup info
349   * @param table table
350   * @return backupImages on the dependency list
351   * @throws IOException exception
352   */
353  public ArrayList<BackupImage> getAncestors(BackupInfo backupInfo, TableName table)
354      throws IOException {
355    ArrayList<BackupImage> ancestors = getAncestors(backupInfo);
356    ArrayList<BackupImage> tableAncestors = new ArrayList<>();
357    for (BackupImage image : ancestors) {
358      if (image.hasTable(table)) {
359        tableAncestors.add(image);
360        if (image.getType() == BackupType.FULL) {
361          break;
362        }
363      }
364    }
365    return tableAncestors;
366  }

381  /**
382   * Starts new backup session
383   * @throws IOException if active session already exists
384   */
385  public void startBackupSession() throws IOException {
386    long startTime = EnvironmentEdgeManager.currentTime();
387    long timeout = conf.getInt(BACKUP_EXCLUSIVE_OPERATION_TIMEOUT_SECONDS_KEY,
388      DEFAULT_BACKUP_EXCLUSIVE_OPERATION_TIMEOUT) * 1000L;
389    long lastWarningOutputTime = 0;
390    while (EnvironmentEdgeManager.currentTime() - startTime < timeout) {
391      try {
392        systemTable.startBackupExclusiveOperation();
393        return;
394      } catch (IOException e) {
395        if (e instanceof ExclusiveOperationException) {
396          // sleep, then repeat
397          try {
398            Thread.sleep(1000);
399          } catch (InterruptedException e1) {
400            // Restore the interrupted status
401            Thread.currentThread().interrupt();
402          }
403          if (lastWarningOutputTime == 0
404              || (EnvironmentEdgeManager.currentTime() - lastWarningOutputTime) > 60000) {
405            lastWarningOutputTime = EnvironmentEdgeManager.currentTime();
406            LOG.warn("Waiting to acquire backup exclusive lock for {}s",
407                +(lastWarningOutputTime - startTime) / 1000);
408          }
409        } else {
410          throw e;
411        }
412      }
413    }
414    throw new IOException(
415      "Failed to acquire backup system table exclusive lock after " + timeout / 1000 + "s");
416  }
