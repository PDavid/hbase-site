199  private void checkSystemTable() throws IOException {
200    try (Admin admin = connection.getAdmin()) {
201      verifyNamespaceExists(admin);
202      Configuration conf = connection.getConfiguration();
203      if (!admin.tableExists(tableName)) {
204        TableDescriptor backupHTD = BackupSystemTable.getSystemTableDescriptor(conf);
205        admin.createTable(backupHTD);
206      }
207      if (!admin.tableExists(bulkLoadTableName)) {
208        TableDescriptor blHTD = BackupSystemTable.getSystemTableForBulkLoadedDataDescriptor(conf);
209        admin.createTable(blHTD);
210      }
211      waitForSystemTable(admin, tableName);
212      waitForSystemTable(admin, bulkLoadTableName);
213    }
214  }
215
216  private void verifyNamespaceExists(Admin admin) throws IOException {
217    String namespaceName = tableName.getNamespaceAsString();
218    NamespaceDescriptor ns = NamespaceDescriptor.create(namespaceName).build();
219    NamespaceDescriptor[] list = admin.listNamespaceDescriptors();
220    boolean exists = false;
221    for (NamespaceDescriptor nsd : list) {
222      if (nsd.getName().equals(ns.getName())) {
223        exists = true;
224        break;
225      }
226    }
227    if (!exists) {
228      admin.createNamespace(ns);
229    }
230  }
231
232  private void waitForSystemTable(Admin admin, TableName tableName) throws IOException {
233    // Return fast if the table is available and avoid a log message
234    if (admin.tableExists(tableName) && admin.isTableAvailable(tableName)) {
235      return;
236    }
237    long TIMEOUT = 60000;
238    long startTime = EnvironmentEdgeManager.currentTime();
239    LOG.debug("Backup table {} is not present and available, waiting for it to become so",
240        tableName);
241    while (!admin.tableExists(tableName) || !admin.isTableAvailable(tableName)) {
242      try {
243        Thread.sleep(100);
244      } catch (InterruptedException e) {
245      }
246      if (EnvironmentEdgeManager.currentTime() - startTime > TIMEOUT) {
247        throw new IOException(
248          "Failed to create backup system table " + tableName + " after " + TIMEOUT + "ms");
249      }
250    }
251    LOG.debug("Backup table {} exists and available", tableName);
252  }

280    Scan scan = BackupSystemTable.createScanForBulkLoadedFiles(backupId);
281    try (Table table = connection.getTable(bulkLoadTableName);
282        ResultScanner scanner = table.getScanner(scan)) {
283      Result res = null;

285      while ((res = scanner.next()) != null) {
286        res.advance();
287        byte[] row = CellUtil.cloneRow(res.listCells().get(0));
288        for (Cell cell : res.listCells()) {
289          if (CellUtil.compareQualifiers(cell, BackupSystemTable.PATH_COL, 0,
290            BackupSystemTable.PATH_COL.length) == 0) {
291            map.put(row, Bytes.toString(CellUtil.cloneValue(cell)));
292          }
293        }
294      }
295      return map;
296    }

299  /*
300   * Used during restore
301   * @param backupId the backup Id
302   * @param sTableList List of tables
303   * @return array of Map of family to List of Paths
304   */
305  public Map<byte[], List<Path>>[] readBulkLoadedFiles(String backupId, List<TableName> sTableList)
306      throws IOException {
307    Scan scan = BackupSystemTable.createScanForBulkLoadedFiles(backupId);
308    Map<byte[], List<Path>>[] mapForSrc = new Map[sTableList == null ? 1 : sTableList.size()];
309    try (Table table = connection.getTable(bulkLoadTableName);
310        ResultScanner scanner = table.getScanner(scan)) {
311      Result res = null;
312      while ((res = scanner.next()) != null) {
313        res.advance();
314        TableName tbl = null;
315        byte[] fam = null;
316        String path = null;
317        for (Cell cell : res.listCells()) {
318          if (CellUtil.compareQualifiers(cell, BackupSystemTable.TBL_COL, 0,
319            BackupSystemTable.TBL_COL.length) == 0) {
320            tbl = TableName.valueOf(CellUtil.cloneValue(cell));
321          } else if (CellUtil.compareQualifiers(cell, BackupSystemTable.FAM_COL, 0,
322            BackupSystemTable.FAM_COL.length) == 0) {
323            fam = CellUtil.cloneValue(cell);
324          } else if (CellUtil.compareQualifiers(cell, BackupSystemTable.PATH_COL, 0,
325            BackupSystemTable.PATH_COL.length) == 0) {
326            path = Bytes.toString(CellUtil.cloneValue(cell));
327          }
328        }
329        int srcIdx = IncrementalTableBackupClient.getIndex(tbl, sTableList);
330        if (srcIdx == -1) {
331          // the table is not among the query
332          continue;
333        }
334        if (mapForSrc[srcIdx] == null) {
335          mapForSrc[srcIdx] = new TreeMap<>(Bytes.BYTES_COMPARATOR);
336        }
337        List<Path> files;
338        if (!mapForSrc[srcIdx].containsKey(fam)) {
339          files = new ArrayList<Path>();
340          mapForSrc[srcIdx].put(fam, files);
341        } else {
342          files = mapForSrc[srcIdx].get(fam);
343        }
344        files.add(new Path(path));
345        if (LOG.isDebugEnabled()) {
346          LOG.debug("found bulk loaded file : " + tbl + " " + Bytes.toString(fam) + " " + path);
347        }
348      }
349
350      return mapForSrc;
351    }
352  }

427  /*
428   * Reads the rows from backup table recording bulk loaded hfiles
429   * @param tableList list of table names
430   * @return The keys of the Map are table, region and column family. Value of the map reflects
431   * whether the hfile was recorded by preCommitStoreFile hook (true)
432   */
433  public Pair<Map<TableName, Map<String, Map<String, List<Pair<String, Boolean>>>>>, List<byte[]>>
434    readBulkloadRows(List<TableName> tableList) throws IOException {
435
436    Map<TableName, Map<String, Map<String, List<Pair<String, Boolean>>>>> map = new HashMap<>();
437    List<byte[]> rows = new ArrayList<>();
438    for (TableName tTable : tableList) {
439      Scan scan = BackupSystemTable.createScanForOrigBulkLoadedFiles(tTable);
440      Map<String, Map<String, List<Pair<String, Boolean>>>> tblMap = map.get(tTable);
441      try (Table table = connection.getTable(bulkLoadTableName);
442          ResultScanner scanner = table.getScanner(scan)) {
443        Result res = null;
444        while ((res = scanner.next()) != null) {
445          res.advance();
446          String fam = null;
447          String path = null;
448          boolean raw = false;
449          byte[] row;
450          String region = null;
451          for (Cell cell : res.listCells()) {
452            row = CellUtil.cloneRow(cell);
453            rows.add(row);
454            String rowStr = Bytes.toString(row);
455            region = BackupSystemTable.getRegionNameFromOrigBulkLoadRow(rowStr);
456            if (CellUtil.compareQualifiers(cell, BackupSystemTable.FAM_COL, 0,
457              BackupSystemTable.FAM_COL.length) == 0) {
458              fam = Bytes.toString(CellUtil.cloneValue(cell));
459            } else if (CellUtil.compareQualifiers(cell, BackupSystemTable.PATH_COL, 0,
460              BackupSystemTable.PATH_COL.length) == 0) {
461              path = Bytes.toString(CellUtil.cloneValue(cell));
462            } else if (CellUtil.compareQualifiers(cell, BackupSystemTable.STATE_COL, 0,
463              BackupSystemTable.STATE_COL.length) == 0) {
464              byte[] state = CellUtil.cloneValue(cell);
465              if (Bytes.equals(BackupSystemTable.BL_PREPARE, state)) {
466                raw = true;
467              } else {
468                raw = false;
469              }
470            }
471          }
472          if (map.get(tTable) == null) {
473            map.put(tTable, new HashMap<>());
474            tblMap = map.get(tTable);
475          }
476          if (tblMap.get(region) == null) {
477            tblMap.put(region, new HashMap<>());
478          }
479          Map<String, List<Pair<String, Boolean>>> famMap = tblMap.get(region);
480          if (famMap.get(fam) == null) {
481            famMap.put(fam, new ArrayList<>());
482          }
483          famMap.get(fam).add(new Pair<>(path, raw));
484          LOG.debug("found orig " + path + " for " + fam + " of table " + region);
485        }
486      }
487    }
488    return new Pair<>(map, rows);
489  }

522      }
523    }

588  /**
589   * Exclusive operations are: create, delete, merge
590   * @throws IOException if a table operation fails or an active backup exclusive operation is
591   *           already underway
592   */
593  public void startBackupExclusiveOperation() throws IOException {
594    LOG.debug("Start new backup exclusive operation");
595
596    try (Table table = connection.getTable(tableName)) {
597      Put put = createPutForStartBackupSession();
598      // First try to put if row does not exist
599      if (!table.checkAndMutate(ACTIVE_SESSION_ROW, SESSIONS_FAMILY).qualifier(ACTIVE_SESSION_COL)
600          .ifNotExists().thenPut(put)) {
601        // Row exists, try to put if value == ACTIVE_SESSION_NO
602        if (!table.checkAndMutate(ACTIVE_SESSION_ROW, SESSIONS_FAMILY).qualifier(ACTIVE_SESSION_COL)
603            .ifEquals(ACTIVE_SESSION_NO).thenPut(put)) {
604          throw new ExclusiveOperationException();
605        }
606      }
607    }
608  }
609
610  private Put createPutForStartBackupSession() {
611    Put put = new Put(ACTIVE_SESSION_ROW);
612    put.addColumn(SESSIONS_FAMILY, ACTIVE_SESSION_COL, ACTIVE_SESSION_YES);
613    return put;
614  }

616  public void finishBackupExclusiveOperation() throws IOException {
617    LOG.debug("Finish backup exclusive operation");
618
619    try (Table table = connection.getTable(tableName)) {
620      Put put = createPutForStopBackupSession();

1282            + StringUtils.join(toRemove, " ") + "]");
1283      } else { // disjoint.length == 0 and tables.length >0
1284        // Delete backup set
1285        LOG.info("Backup set '" + name + "' is empty. Deleting.");
1286        deleteBackupSet(name);
1287      }
1288    }
1289  }
