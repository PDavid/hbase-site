018package org.apache.hadoop.hbase.client;
019
020import static org.apache.hadoop.hbase.util.FutureUtils.addListener;
021
022import java.io.IOException;
023import java.util.List;
024import org.apache.hadoop.hbase.CellScanner;
025import org.apache.hadoop.hbase.DoNotRetryIOException;
026import org.apache.hadoop.hbase.HRegionLocation;
027import org.apache.hadoop.hbase.protobuf.ReplicationProtobufUtil;
028import org.apache.hadoop.hbase.util.Pair;
029import org.apache.hadoop.hbase.wal.WAL.Entry;
030import org.apache.yetus.audience.InterfaceAudience;
031
032import org.apache.hbase.thirdparty.io.netty.util.HashedWheelTimer;
033
034import org.apache.hadoop.hbase.shaded.protobuf.generated.AdminProtos.AdminService;
035import org.apache.hadoop.hbase.shaded.protobuf.generated.AdminProtos.ReplicateWALEntryRequest;
036
037/**
038 * For replicating edits to secondary replicas.
039 */
040@InterfaceAudience.Private
041public class AsyncRegionReplicationRetryingCaller extends AsyncRpcRetryingCaller<Void> {
042
043  private final RegionInfo replica;
044
045  private final Entry[] entries;
046
047  // whether to use replay instead of replicateToReplica, during rolling upgrading if the target
048  // region server has not been upgraded then it will not have the replicateToReplica method, so we
049  // could use replay method first, though it is not perfect.
050  private boolean useReplay;
051
052  public AsyncRegionReplicationRetryingCaller(HashedWheelTimer retryTimer,
053    AsyncClusterConnectionImpl conn, int maxAttempts, long rpcTimeoutNs, long operationTimeoutNs,
054    RegionInfo replica, List<Entry> entries) {
055    super(retryTimer, conn, ConnectionUtils.getPriority(replica.getTable()),
056      conn.connConf.getPauseNs(), conn.connConf.getPauseForCQTBENs(), maxAttempts,
057      operationTimeoutNs, rpcTimeoutNs, conn.connConf.getStartLogErrorsCnt());
058    this.replica = replica;
059    this.entries = entries.toArray(new Entry[0]);
060  }
061
062  @Override
063  protected Throwable preProcessError(Throwable error) {
064    if (error instanceof DoNotRetryIOException &&
065      error.getCause() instanceof UnsupportedOperationException) {
066      // fallback to use replay, and also return the cause to let the upper retry
067      useReplay = true;
068      return error.getCause();
069    }
070    return error;
071  }
072
073  private void onComplete(HRegionLocation loc) {
074    if (controller.failed()) {
075      onError(controller.getFailed(),
076        () -> "Call to " + loc.getServerName() + " for " + replica + " failed",
077        err -> conn.getLocator().updateCachedLocationOnError(loc, err));
078    } else {
079      future.complete(null);
080    }
081  }
082
083  private void call(HRegionLocation loc) {
084    AdminService.Interface stub;
085    try {
086      stub = conn.getAdminStub(loc.getServerName());
087    } catch (IOException e) {
088      onError(e,
089        () -> "Get async admin stub to " + loc.getServerName() + " for " + replica + " failed",
090        err -> conn.getLocator().updateCachedLocationOnError(loc, err));
091      return;
092    }
093    Pair<ReplicateWALEntryRequest, CellScanner> pair = ReplicationProtobufUtil
094      .buildReplicateWALEntryRequest(entries, replica.getEncodedNameAsBytes(), null, null, null);
095    resetCallTimeout();
096    controller.setCellScanner(pair.getSecond());
097    if (useReplay) {
098      stub.replay(controller, pair.getFirst(), r -> onComplete(loc));
099    } else {
100      stub.replicateToReplica(controller, pair.getFirst(), r -> onComplete(loc));
101    }
102  }
103
104  @Override
105  protected void doCall() {
106    long locateTimeoutNs;
107    if (operationTimeoutNs > 0) {
108      locateTimeoutNs = remainingTimeNs();
109      if (locateTimeoutNs <= 0) {
110        completeExceptionally();
111        return;
112      }
113    } else {
114      locateTimeoutNs = -1L;
115    }
116    addListener(conn.getLocator().getRegionLocation(replica.getTable(), replica.getStartKey(),
117      replica.getReplicaId(), RegionLocateType.CURRENT, locateTimeoutNs), (loc, error) -> {
118        if (error != null) {
119          onError(error, () -> "Locate " + replica + " failed", err -> {
120          });
121          return;
122        }
123        call(loc);
124      });
125  }
126}
