306  /**
307   * A helper class for sending checkAndMutate request.
308   */
309  interface CheckAndMutateBuilder {
310
311    /**
312     * @param qualifier column qualifier to check.
313     */
314    CheckAndMutateBuilder qualifier(byte[] qualifier);
315
316    /**
317     * @param timeRange timeRange to check
318     */
319    CheckAndMutateBuilder timeRange(TimeRange timeRange);
320
321    /**
322     * Check for lack of column.
323     */
324    CheckAndMutateBuilder ifNotExists();
325
326    /**
327     * Check for equality.
328     * @param value the expected value
329     */
330    default CheckAndMutateBuilder ifEquals(byte[] value) {
331      return ifMatches(CompareOperator.EQUAL, value);
332    }
333
334    /**
335     * @param compareOp comparison operator to use
336     * @param value the expected value
337     */
338    CheckAndMutateBuilder ifMatches(CompareOperator compareOp, byte[] value);
339
340    /**
341     * @param put data to put if check succeeds
342     * @return {@code true} if the new put was executed, {@code false} otherwise.
343     */
344    boolean thenPut(Put put) throws IOException;
345
346    /**
347     * @param delete data to delete if check succeeds
348     * @return {@code true} if the new delete was executed, {@code false} otherwise.
349     */
350    boolean thenDelete(Delete delete) throws IOException;
351    /**
352     * @param mutation mutations to perform if check succeeds
353     * @return true if the new mutation was executed, false otherwise.
354     */
355    boolean thenMutate(RowMutations mutation) throws IOException;
356  }

400  /**
401   * See {@link #incrementColumnValue(byte[], byte[], byte[], long, Durability)}
402   * <p>
403   * The {@link Durability} is defaulted to {@link Durability#SYNC_WAL}.
404   * @param row The row that contains the cell to increment.
405   * @param family The column family of the cell to increment.
406   * @param qualifier The column qualifier of the cell to increment.
407   * @param amount The amount to increment the cell with (or decrement, if the
408   * amount is negative).
409   * @return The new value, post increment.
410   * @throws IOException if a remote or network exception occurs.
411   */
412  default long incrementColumnValue(byte[] row, byte[] family, byte[] qualifier, long amount)
413      throws IOException {
414    Increment increment = new Increment(row).addColumn(family, qualifier, amount);
415    Cell cell = increment(increment).getColumnLatestCell(family, qualifier);
416    return Bytes.toLong(cell.getValueArray(), cell.getValueOffset(), cell.getValueLength());
417  }

419  /**
420   * Atomically increments a column value. If the column value already exists
421   * and is not a big-endian long, this could throw an exception. If the column
422   * value does not yet exist it is initialized to <code>amount</code> and
423   * written to the specified column.
424   *
425   * <p>Setting durability to {@link Durability#SKIP_WAL} means that in a fail
426   * scenario you will lose any increments that have not been flushed.
427   * @param row The row that contains the cell to increment.
428   * @param family The column family of the cell to increment.
429   * @param qualifier The column qualifier of the cell to increment.
430   * @param amount The amount to increment the cell with (or decrement, if the
431   * amount is negative).
432   * @param durability The persistence guarantee for this increment.
433   * @return The new value, post increment.
434   * @throws IOException if a remote or network exception occurs.
435   */
436  default long incrementColumnValue(byte[] row, byte[] family, byte[] qualifier,
437    long amount, Durability durability) throws IOException {
438    Increment increment = new Increment(row)
439        .addColumn(family, qualifier, amount)
440        .setDurability(durability);
441    Cell cell = increment(increment).getColumnLatestCell(family, qualifier);
442    return Bytes.toLong(cell.getValueArray(), cell.getValueOffset(), cell.getValueLength());
443  }
