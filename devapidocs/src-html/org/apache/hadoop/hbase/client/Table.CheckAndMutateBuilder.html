470  /**
471   * See {@link #incrementColumnValue(byte[], byte[], byte[], long, Durability)}
472   * <p>
473   * The {@link Durability} is defaulted to {@link Durability#SYNC_WAL}.
474   * @param row       The row that contains the cell to increment.
475   * @param family    The column family of the cell to increment.
476   * @param qualifier The column qualifier of the cell to increment.
477   * @param amount    The amount to increment the cell with (or decrement, if the amount is
478   *                  negative).
479   * @return The new value, post increment.
480   * @throws IOException if a remote or network exception occurs.
481   */
482  default long incrementColumnValue(byte[] row, byte[] family, byte[] qualifier, long amount)
483    throws IOException {
484    Increment increment = new Increment(row).addColumn(family, qualifier, amount);
485    Cell cell = increment(increment).getColumnLatestCell(family, qualifier);
486    return Bytes.toLong(cell.getValueArray(), cell.getValueOffset(), cell.getValueLength());
487  }

493   * <p>
494   * Setting durability to {@link Durability#SKIP_WAL} means that in a fail scenario you will lose

496   * @param row        The row that contains the cell to increment.

557  /**
558   * Creates an instance of the given {@link Service} subclass for each table region spanning the
559   * range from the {@code startKey} row to {@code endKey} row (inclusive), and invokes the passed
560   * {@link org.apache.hadoop.hbase.client.coprocessor.Batch.Call#call} method with each
561   * {@link Service} instance.
562   * @param service  the protocol buffer {@code Service} implementation to call
563   * @param startKey start region selection with region containing this row. If {@code null}, the
564   *                 selection will start with the first table region.
565   * @param endKey   select regions up to and including the region containing this row. If
566   *                 {@code null}, selection will continue through the last table region.
567   * @param callable this instance's
568   *                 {@link org.apache.hadoop.hbase.client.coprocessor.Batch.Call#call} method will
569   *                 be invoked once per table region, using the {@link Service} instance connected
570   *                 to that region.
571   * @param <T>      the {@link Service} subclass to connect to
572   * @param <R>      Return type for the {@code callable} parameter's
573   *                 {@link org.apache.hadoop.hbase.client.coprocessor.Batch.Call#call} method
574   * @return a map of result values keyed by region name
575   * @deprecated since 3.0.0, will removed in 4.0.0. The batch call here references the blocking
576   *             interface for of a protobuf stub, so it is not possible to do it in an asynchronous
577   *             way, even if now we are building the {@link Table} implementation based on the
578   *             {@link AsyncTable}, which is not good. Use the coprocessorService methods in
579   *             {@link AsyncTable} directly instead.
580   * @see Connection#toAsyncConnection()
581   */
582  @Deprecated
583  default <T extends Service, R> Map<byte[], R> coprocessorService(final Class<T> service,
584    byte[] startKey, byte[] endKey, final Batch.Call<T, R> callable)
585    throws ServiceException, Throwable {
586    Map<byte[], R> results =
587      Collections.synchronizedMap(new TreeMap<byte[], R>(Bytes.BYTES_COMPARATOR));
588    coprocessorService(service, startKey, endKey, callable, new Batch.Callback<R>() {
589      @Override
590      public void update(byte[] region, byte[] row, R value) {
591        if (region != null) {
592          results.put(region, value);
593        }
594      }
595    });
596    return results;
