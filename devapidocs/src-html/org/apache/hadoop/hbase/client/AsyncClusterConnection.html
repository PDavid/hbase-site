018package org.apache.hadoop.hbase.client;
019
020import java.util.List;
021import java.util.concurrent.CompletableFuture;
022import org.apache.hadoop.hbase.RegionLocations;
023import org.apache.hadoop.hbase.ServerName;
024import org.apache.hadoop.hbase.TableName;
025import org.apache.hadoop.hbase.ipc.RpcClient;
026import org.apache.hadoop.hbase.util.Pair;
027import org.apache.hadoop.hbase.wal.WAL.Entry;
028import org.apache.hadoop.hbase.zookeeper.MasterAddressTracker;
029import org.apache.hadoop.security.token.Token;
030import org.apache.yetus.audience.InterfaceAudience;
031
032import org.apache.hadoop.hbase.shaded.protobuf.generated.AdminProtos.FlushRegionResponse;
033
034/**
035 * The asynchronous connection for internal usage.
036 */
037@InterfaceAudience.Private
038public interface AsyncClusterConnection extends AsyncConnection {
039
040  /**
041   * Get the admin service for the given region server.
042   */
043  AsyncRegionServerAdmin getRegionServerAdmin(ServerName serverName);
044
045  /**
046   * Get the nonce generator for this connection.
047   */
048  NonceGenerator getNonceGenerator();
049
050  /**
051   * Get the rpc client we used to communicate with other servers.
052   */
053  RpcClient getRpcClient();
054
055  /**
056   * Flush a region and get the response.
057   */
058  CompletableFuture<FlushRegionResponse> flush(byte[] regionName, boolean writeFlushWALMarker);
059
060  /**
061   * Replicate wal edits for replica regions. The return value is the edits we skipped, as the
062   * original return value is useless.
063   */
064  CompletableFuture<Long> replay(TableName tableName, byte[] encodedRegionName, byte[] row,
065      List<Entry> entries, int replicaId, int numRetries, long operationTimeoutNs);
066
067  /**
068   * Return all the replicas for a region. Used for region replica replication.
069   */
070  CompletableFuture<RegionLocations> getRegionLocations(TableName tableName, byte[] row,
071      boolean reload);
072
073  /**
074   * Return the token for this bulk load.
075   */
076  CompletableFuture<String> prepareBulkLoad(TableName tableName);
077
078  /**
079   * Securely bulk load a list of HFiles, passing additional list of clusters ids tracking clusters
080   * where the given bulk load has already been processed (important for bulk loading replication).
081   * <p/>
082   * Defined as default here to avoid breaking callers who rely on the bulkLoad version that does
083   * not expect additional clusterIds param.
084   * @param tableName the target table
085   * @param familyPaths hdfs path for the the table family dirs containg files to be loaded.
086   * @param row row key.
087   * @param assignSeqNum seq num for the event on WAL.
088   * @param userToken user token.
089   * @param bulkToken bulk load token.
090   * @param copyFiles flag for copying the loaded hfiles.
091   * @param clusterIds list of cluster ids where the given bulk load has already been processed.
092   * @param replicate flags if the bulkload is targeted for replication.
093   */
094  CompletableFuture<Boolean> bulkLoad(TableName tableName, List<Pair<byte[], String>> familyPaths,
095    byte[] row, boolean assignSeqNum, Token<?> userToken, String bulkToken, boolean copyFiles,
096    List<String> clusterIds, boolean replicate);
097
098  /**
099   * Clean up after finishing bulk load, no matter success or not.
100   */
101  CompletableFuture<Void> cleanupBulkLoad(TableName tableName, String bulkToken);
102
103  /**
104   * Get live region servers from masters.
105   */
106  CompletableFuture<List<ServerName>> getLiveRegionServers(MasterAddressTracker masterAddrTracker,
107    int count);
108
109  /**
110   * Get the bootstrap node list of another region server.
111   */
112  CompletableFuture<List<ServerName>> getAllBootstrapNodes(ServerName regionServer);
113}
