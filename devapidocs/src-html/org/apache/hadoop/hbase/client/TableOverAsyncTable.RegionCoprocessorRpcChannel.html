114  @Override
115  public void batch(List<? extends Row> actions, Object[] results) throws IOException {
116    if (ArrayUtils.isEmpty(results)) {
117      FutureUtils.get(table.batchAll(actions));
118      return;
119    }
120    List<ThrowableWithExtraContext> errors = new ArrayList<>();
121    List<CompletableFuture<Object>> futures = table.batch(actions);
122    for (int i = 0, n = results.length; i < n; i++) {
123      try {
124        results[i] = FutureUtils.get(futures.get(i));
125      } catch (IOException e) {
126        results[i] = e;
127        errors.add(new ThrowableWithExtraContext(e, EnvironmentEdgeManager.currentTime(),
128          "Error when processing " + actions.get(i)));
129      }
130    }
131    if (!errors.isEmpty()) {
132      throw new RetriesExhaustedException(errors.size(), errors);
133    }
134  }
135
136  @Override
137  public <R> void batchCallback(List<? extends Row> actions, Object[] results, Callback<R> callback)
138      throws IOException, InterruptedException {
139    ConcurrentLinkedQueue<ThrowableWithExtraContext> errors = new ConcurrentLinkedQueue<>();
140    CountDownLatch latch = new CountDownLatch(actions.size());
141    AsyncTableRegionLocator locator = conn.getRegionLocator(getName());
142    List<CompletableFuture<R>> futures = table.<R> batch(actions);
143    for (int i = 0, n = futures.size(); i < n; i++) {
144      final int index = i;
145      FutureUtils.addListener(futures.get(i), (r, e) -> {
146        if (e != null) {
147          errors.add(new ThrowableWithExtraContext(e, EnvironmentEdgeManager.currentTime(),
148            "Error when processing " + actions.get(index)));
149          if (!ArrayUtils.isEmpty(results)) {
150            results[index] = e;
151          }
152          latch.countDown();
153        } else {
154          if (!ArrayUtils.isEmpty(results)) {
155            results[index] = r;
156          }
157          FutureUtils.addListener(locator.getRegionLocation(actions.get(index).getRow()),
158            (l, le) -> {
159              if (le != null) {
160                errors.add(new ThrowableWithExtraContext(le, EnvironmentEdgeManager.currentTime(),
161                  "Error when finding the region for row " +
162                    Bytes.toStringBinary(actions.get(index).getRow())));
163              } else {
164                callback.update(l.getRegion().getRegionName(), actions.get(index).getRow(), r);
165              }
166              latch.countDown();
167            });
168        }
169      });
170    }
171    latch.await();
172    if (!errors.isEmpty()) {
173      throw new RetriesExhaustedException(errors.size(),
174        errors.stream().collect(Collectors.toList()));
175    }
176  }

309  private static final class RegionCoprocessorRpcChannel extends RegionCoprocessorRpcChannelImpl

311
312    RegionCoprocessorRpcChannel(AsyncConnectionImpl conn, TableName tableName, RegionInfo region,
313        byte[] row, long rpcTimeoutNs, long operationTimeoutNs) {
314      super(conn, tableName, region, row, rpcTimeoutNs, operationTimeoutNs);
315    }
316
317    @Override
318    public void callMethod(MethodDescriptor method, RpcController controller, Message request,
319        Message responsePrototype, RpcCallback<Message> done) {
320      ClientCoprocessorRpcController c = new ClientCoprocessorRpcController();
321      CoprocessorBlockingRpcCallback<Message> callback = new CoprocessorBlockingRpcCallback<>();
322      super.callMethod(method, c, request, responsePrototype, callback);
323      Message ret;
324      try {
325        ret = callback.get();
326      } catch (IOException e) {
327        setCoprocessorError(controller, e);
328        return;
329      }
330      if (c.failed()) {
331        setCoprocessorError(controller, c.getFailed());
332      }
333      done.run(ret);
334    }
335
336    @Override
337    public Message callBlockingMethod(MethodDescriptor method, RpcController controller,
338        Message request, Message responsePrototype) throws ServiceException {
339      ClientCoprocessorRpcController c = new ClientCoprocessorRpcController();
340      CoprocessorBlockingRpcCallback<Message> done = new CoprocessorBlockingRpcCallback<>();
341      callMethod(method, c, request, responsePrototype, done);
342      Message ret;
343      try {
344        ret = done.get();
345      } catch (IOException e) {
346        throw new ServiceException(e);
347      }
348      if (c.failed()) {
349        setCoprocessorError(controller, c.getFailed());
350        throw new ServiceException(c.getFailed());
351      }
352      return ret;
353    }
354  }

363   * Get the corresponding start keys and regions for an arbitrary range of keys.
364   * <p>
365   * @param startKey Starting row in range, inclusive
366   * @param endKey Ending row in range
367   * @param includeEndKey true if endRow is inclusive, false if exclusive
368   * @return A pair of list of start keys and list of HRegionLocations that contain the specified
369   *         range
370   * @throws IOException if a remote or network exception occurs
371   */
372  private Pair<List<byte[]>, List<HRegionLocation>> getKeysAndRegionsInRange(final byte[] startKey,
373      final byte[] endKey, final boolean includeEndKey) throws IOException {
374    return getKeysAndRegionsInRange(startKey, endKey, includeEndKey, false);
375  }
376
377  /**
378   * Get the corresponding start keys and regions for an arbitrary range of keys.
379   * <p>
380   * @param startKey Starting row in range, inclusive
381   * @param endKey Ending row in range
382   * @param includeEndKey true if endRow is inclusive, false if exclusive
383   * @param reload true to reload information or false to use cached information
384   * @return A pair of list of start keys and list of HRegionLocations that contain the specified
385   *         range
386   * @throws IOException if a remote or network exception occurs
387   */
388  private Pair<List<byte[]>, List<HRegionLocation>> getKeysAndRegionsInRange(final byte[] startKey,
389      final byte[] endKey, final boolean includeEndKey, final boolean reload) throws IOException {
390    final boolean endKeyIsEndOfTable = Bytes.equals(endKey, HConstants.EMPTY_END_ROW);
391    if ((Bytes.compareTo(startKey, endKey) > 0) && !endKeyIsEndOfTable) {
392      throw new IllegalArgumentException(
393        "Invalid range: " + Bytes.toStringBinary(startKey) + " > " + Bytes.toStringBinary(endKey));
394    }
395    List<byte[]> keysInRange = new ArrayList<>();
396    List<HRegionLocation> regionsInRange = new ArrayList<>();
397    byte[] currentKey = startKey;
398    do {
399      HRegionLocation regionLocation =
400        FutureUtils.get(conn.getRegionLocator(getName()).getRegionLocation(currentKey, reload));
401      keysInRange.add(currentKey);
402      regionsInRange.add(regionLocation);
403      currentKey = regionLocation.getRegion().getEndKey();
404    } while (!Bytes.equals(currentKey, HConstants.EMPTY_END_ROW) &&
405      (endKeyIsEndOfTable || Bytes.compareTo(currentKey, endKey) < 0 ||
406        (includeEndKey && Bytes.compareTo(currentKey, endKey) == 0)));
407    return new Pair<>(keysInRange, regionsInRange);
408  }
409
410  private List<byte[]> getStartKeysInRange(byte[] start, byte[] end) throws IOException {
411    if (start == null) {
412      start = HConstants.EMPTY_START_ROW;
413    }
414    if (end == null) {
415      end = HConstants.EMPTY_END_ROW;
416    }
417    return getKeysAndRegionsInRange(start, end, true).getFirst();
418  }

420  @FunctionalInterface
421  private interface StubCall<R> {
422    R call(RegionCoprocessorRpcChannel channel) throws Exception;
423  }
424
425  private <R> void coprocssorService(String serviceName, byte[] startKey, byte[] endKey,
426      Callback<R> callback, StubCall<R> call) throws Throwable {
427    // get regions covered by the row range
428    ExecutorService pool = this.poolSupplier.get();
429    List<byte[]> keys = getStartKeysInRange(startKey, endKey);
430    Map<byte[], Future<R>> futures = new TreeMap<>(Bytes.BYTES_COMPARATOR);
431    try {
432      for (byte[] r : keys) {
433        RegionCoprocessorRpcChannel channel = coprocessorService(r);
434        Future<R> future = pool.submit(new Callable<R>() {
435          @Override
436          public R call() throws Exception {
437            R result = call.call(channel);
438            byte[] region = channel.getLastRegion();
439            if (callback != null) {
440              callback.update(region, r, result);
441            }
442            return result;
443          }
444        });
445        futures.put(r, future);
446      }
447    } catch (RejectedExecutionException e) {
448      // maybe the connection has been closed, let's check
449      if (conn.isClosed()) {
450        throw new DoNotRetryIOException("Connection is closed", e);
451      } else {
452        throw new HBaseIOException("Coprocessor operation is rejected", e);
453      }
454    }
455    for (Map.Entry<byte[], Future<R>> e : futures.entrySet()) {
456      try {
457        e.getValue().get();
458      } catch (ExecutionException ee) {
459        LOG.warn("Error calling coprocessor service " + serviceName + " for row " +
460          Bytes.toStringBinary(e.getKey()), ee);
461        throw ee.getCause();
462      } catch (InterruptedException ie) {
463        throw new InterruptedIOException("Interrupted calling coprocessor service " + serviceName +
464          " for row " + Bytes.toStringBinary(e.getKey())).initCause(ie);
465      }
466    }
467  }

469  @Override
470  public <T extends Service, R> void coprocessorService(Class<T> service, byte[] startKey,
471      byte[] endKey, Call<T, R> callable, Callback<R> callback) throws ServiceException, Throwable {
472    coprocssorService(service.getName(), startKey, endKey, callback, channel -> {
473      T instance = org.apache.hadoop.hbase.protobuf.ProtobufUtil.newServiceStub(service, channel);
474      return callable.call(instance);
475    });
