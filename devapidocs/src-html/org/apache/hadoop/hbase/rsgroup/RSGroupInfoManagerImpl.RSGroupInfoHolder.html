104/**
105 * This is an implementation of {@link RSGroupInfoManager} which makes use of an HBase table as the
106 * persistence store for the group information. It also makes use of zookeeper to store group
107 * information needed for bootstrapping during offline mode.
108 * <h2>Concurrency</h2> RSGroup state is kept locally in Maps. There is a rsgroup name to cached
109 * RSGroupInfo Map at {@link RSGroupInfoHolder#groupName2Group}.
110 * These Maps are persisted to the hbase:rsgroup table (and cached in zk) on each modification.
111 * <p/>
112 * Mutations on state are synchronized but reads can continue without having to wait on an instance
113 * monitor, mutations do wholesale replace of the Maps on update -- Copy-On-Write; the local Maps of
114 * state are read-only, just-in-case (see flushConfig).
115 * <p/>
116 * Reads must not block else there is a danger we'll deadlock.
117 * <p/>
118 * Clients of this class, the {@link RSGroupAdminEndpoint} for example, want to query and then act
119 * on the results of the query modifying cache in zookeeper without another thread making
120 * intermediate modifications. These clients synchronize on the 'this' instance so no other has
121 * access concurrently. Reads must be able to continue concurrently.
122 */
123@InterfaceAudience.Private

166  // There two Maps are immutable and wholesale replaced on each modification
167  // so are safe to access concurrently. See class comment.
168  private static final class RSGroupInfoHolder {
169    final ImmutableMap<String, RSGroupInfo> groupName2Group;
170    final ImmutableMap<TableName, RSGroupInfo> tableName2Group;
171
172    RSGroupInfoHolder() {
173      this(Collections.emptyMap());
174    }
175
176    RSGroupInfoHolder(Map<String, RSGroupInfo> rsGroupMap) {
177      ImmutableMap.Builder<String, RSGroupInfo> group2Name2GroupBuilder = ImmutableMap.builder();
178      ImmutableMap.Builder<TableName, RSGroupInfo> tableName2GroupBuilder = ImmutableMap.builder();
179      rsGroupMap.forEach((groupName, rsGroupInfo) -> {
180        group2Name2GroupBuilder.put(groupName, rsGroupInfo);
181        if (!groupName.equals(RSGroupInfo.DEFAULT_GROUP)) {
182          rsGroupInfo.getTables()
183            .forEach(tableName -> tableName2GroupBuilder.put(tableName, rsGroupInfo));
184        }
185      });
186      this.groupName2Group = group2Name2GroupBuilder.build();
187      this.tableName2Group = tableName2GroupBuilder.build();
188    }
189  }

201  static class RSGroupMappingScript {
202    static final String RS_GROUP_MAPPING_SCRIPT = "hbase.rsgroup.table.mapping.script";
203    static final String RS_GROUP_MAPPING_SCRIPT_TIMEOUT =
204      "hbase.rsgroup.table.mapping.script.timeout";
205    private Shell.ShellCommandExecutor rsgroupMappingScript;
206
207    RSGroupMappingScript(Configuration conf) {
208      String script = conf.get(RS_GROUP_MAPPING_SCRIPT);
209      if (script == null || script.isEmpty()) {
210        return;
211      }
212
213      rsgroupMappingScript = new Shell.ShellCommandExecutor(
214        new String[] { script, "", "" }, null, null,
215        conf.getLong(RS_GROUP_MAPPING_SCRIPT_TIMEOUT, 5000) // 5 seconds
216      );
217    }
218
219    String getRSGroup(String namespace, String tablename) {
220      if (rsgroupMappingScript == null) {
221        return null;
222      }
223      String[] exec = rsgroupMappingScript.getExecString();
224      exec[1] = namespace;
225      exec[2] = tablename;
226      try {
227        rsgroupMappingScript.execute();
228      } catch (IOException e) {
229        // This exception may happen, like process doesn't have permission to run this script.
230        LOG.error("{}, placing {} back to default rsgroup",
231          e.getMessage(),
232          TableName.valueOf(namespace, tablename));
233        return RSGroupInfo.DEFAULT_GROUP;
234      }
235      return rsgroupMappingScript.getOutput().trim();
236    }
237  }

248  private synchronized void updateDefaultServers() {
249    LOG.info("Updating default servers.");
250    Map<String, RSGroupInfo> newGroupMap = Maps.newHashMap(holder.groupName2Group);
251    RSGroupInfo oldDefaultGroupInfo = getRSGroup(RSGroupInfo.DEFAULT_GROUP);
252    assert oldDefaultGroupInfo != null;
253    RSGroupInfo newDefaultGroupInfo =
254      new RSGroupInfo(RSGroupInfo.DEFAULT_GROUP, getDefaultServers());
255    newDefaultGroupInfo.addAllTables(oldDefaultGroupInfo.getTables());
256    newGroupMap.put(RSGroupInfo.DEFAULT_GROUP, newDefaultGroupInfo);
257    // do not need to persist, as we do not persist default group.
258    resetRSGroupMap(newGroupMap);
259    LOG.info("Updated default servers, {} servers", newDefaultGroupInfo.getServers().size());
260  }

319  public synchronized Set<Address> moveServers(Set<Address> servers, String srcGroup,
320      String dstGroup) throws IOException {
321    RSGroupInfo src = getRSGroupInfo(srcGroup);
322    RSGroupInfo dst = getRSGroupInfo(dstGroup);
323    Set<Address> movedServers = new HashSet<>();
324    // If destination is 'default' rsgroup, only add servers that are online. If not online, drop
325    // it. If not 'default' group, add server to 'dst' rsgroup EVEN IF IT IS NOT online (could be a
326    // rsgroup of dead servers that are to come back later).
327    Set<Address> onlineServers =
328      dst.getName().equals(RSGroupInfo.DEFAULT_GROUP) ? getOnlineServers() : null;
329    for (Address el : servers) {
330      src.removeServer(el);
331      if (onlineServers != null) {
332        if (!onlineServers.contains(el)) {
333          if (LOG.isDebugEnabled()) {
334            LOG.debug("Dropping " + el + " during move-to-default rsgroup because not online");
335          }
336          continue;
337        }
338      }
339      dst.addServer(el);
340      movedServers.add(el);
341    }
342    Map<String, RSGroupInfo> newGroupMap = Maps.newHashMap(holder.groupName2Group);
343    newGroupMap.put(src.getName(), src);
344    newGroupMap.put(dst.getName(), dst);
345    flushConfig(newGroupMap);
346    return movedServers;
347  }

364  @Override
365  public synchronized void removeRSGroup(String groupName) throws IOException {
366    RSGroupInfo rsGroupInfo = getRSGroupInfo(groupName);
367    int serverCount = rsGroupInfo.getServers().size();
368    if (serverCount > 0) {
369      throw new ConstraintException("RSGroup " + groupName + " has " + serverCount +
370          " servers; you must remove these servers from the RSGroup before" +
371          " the RSGroup can be removed.");
372    }
373    for (TableDescriptor td : masterServices.getTableDescriptors().getAll().values()) {
374      if (td.getRegionServerGroup().map(groupName::equals).orElse(false)) {
375        throw new ConstraintException("RSGroup " + groupName + " is already referenced by " +
376            td.getTableName() + "; you must remove all the tables from the rsgroup before " +
377            "the rsgroup can be removed.");
378      }
379    }
380    for (NamespaceDescriptor ns : masterServices.getClusterSchema().getNamespaces()) {
381      String nsGroup = ns.getConfigurationValue(RSGroupInfo.NAMESPACE_DESC_PROP_GROUP);
382      if (nsGroup != null && nsGroup.equals(groupName)) {
383        throw new ConstraintException(
384            "RSGroup " + groupName + " is referenced by namespace: " + ns.getName());
385      }
386    }
387    Map<String, RSGroupInfo> rsGroupMap = holder.groupName2Group;
388    if (!rsGroupMap.containsKey(groupName) || groupName.equals(RSGroupInfo.DEFAULT_GROUP)) {
389      throw new ConstraintException(
390        "Group " + groupName + " does not exist or is a reserved " + "group");
391    }
392    Map<String, RSGroupInfo> newGroupMap = Maps.newHashMap(rsGroupMap);
393    newGroupMap.remove(groupName);
394    flushConfig(newGroupMap);
395  }

407  @Override
408  public synchronized void removeServers(Set<Address> servers) throws IOException {
409    if (servers == null || servers.isEmpty()) {
410      throw new ConstraintException("The set of servers to remove cannot be null or empty.");
411    }
412
413    // check the set of servers
414    checkForDeadOrOnlineServers(servers);
415
416    Map<String, RSGroupInfo> rsGroupInfos = new HashMap<String, RSGroupInfo>();
417    for (Address el : servers) {
418      RSGroupInfo rsGroupInfo = getRSGroupOfServer(el);
419      if (rsGroupInfo != null) {
420        RSGroupInfo newRsGroupInfo = rsGroupInfos.get(rsGroupInfo.getName());
421        if (newRsGroupInfo == null) {
422          rsGroupInfo.removeServer(el);
423          rsGroupInfos.put(rsGroupInfo.getName(), rsGroupInfo);
424        } else {
425          newRsGroupInfo.removeServer(el);
426          rsGroupInfos.put(newRsGroupInfo.getName(), newRsGroupInfo);
427        }
428      } else {
429        LOG.warn("Server " + el + " does not belong to any rsgroup.");
430      }
431    }
432
433    if (rsGroupInfos.size() > 0) {
434      Map<String, RSGroupInfo> newGroupMap = Maps.newHashMap(holder.groupName2Group);
435      newGroupMap.putAll(rsGroupInfos);
436      flushConfig(newGroupMap);
437    }
438    LOG.info("Remove decommissioned servers {} from RSGroup done", servers);
439  }

441  private List<RSGroupInfo> retrieveGroupListFromGroupTable() throws IOException {
442    List<RSGroupInfo> rsGroupInfoList = Lists.newArrayList();
443    AsyncTable<?> table = conn.getTable(RSGROUP_TABLE_NAME);
444    try (ResultScanner scanner = table.getScanner(META_FAMILY_BYTES, META_QUALIFIER_BYTES)) {
445      for (Result result;;) {
446        result = scanner.next();
447        if (result == null) {
448          break;
449        }
450        RSGroupProtos.RSGroupInfo proto = RSGroupProtos.RSGroupInfo
451            .parseFrom(result.getValue(META_FAMILY_BYTES, META_QUALIFIER_BYTES));
452        rsGroupInfoList.add(ProtobufUtil.toGroupInfo(proto));
453      }
454    }
455    return rsGroupInfoList;
456  }

458  private List<RSGroupInfo> retrieveGroupListFromZookeeper() throws IOException {
459    String groupBasePath = ZNodePaths.joinZNode(watcher.getZNodePaths().baseZNode, RS_GROUP_ZNODE);
460    List<RSGroupInfo> RSGroupInfoList = Lists.newArrayList();
461    // Overwrite any info stored by table, this takes precedence
462    try {
463      if (ZKUtil.checkExists(watcher, groupBasePath) != -1) {
464        List<String> children = ZKUtil.listChildrenAndWatchForNewChildren(watcher, groupBasePath);
465        if (children == null) {
466          return RSGroupInfoList;
467        }
468        for (String znode : children) {
469          byte[] data = ZKUtil.getData(watcher, ZNodePaths.joinZNode(groupBasePath, znode));
470          if (data != null && data.length > 0) {
471            ProtobufUtil.expectPBMagicPrefix(data);
472            ByteArrayInputStream bis =
473              new ByteArrayInputStream(data, ProtobufUtil.lengthOfPBMagic(), data.length);
474            RSGroupInfoList
475              .add(ProtobufUtil.toGroupInfo(RSGroupProtos.RSGroupInfo.parseFrom(bis)));
476          }
477        }
478        LOG.debug("Read ZK GroupInfo count:" + RSGroupInfoList.size());
479      }
480    } catch (KeeperException | DeserializationException | InterruptedException e) {
481      throw new IOException("Failed to read rsGroupZNode", e);
482    }
483    return RSGroupInfoList;
484  }

486  private void migrate(Collection<RSGroupInfo> groupList) {
487    TableDescriptors tds = masterServices.getTableDescriptors();
488    for (RSGroupInfo groupInfo : groupList) {
489      if (groupInfo.getName().equals(RSGroupInfo.DEFAULT_GROUP)) {
490        continue;
491      }
492      SortedSet<TableName> failedTables = new TreeSet<>();
493      for (TableName tableName : groupInfo.getTables()) {
494        LOG.debug("Migrating {} in group {}", tableName, groupInfo.getName());
495        TableDescriptor oldTd;
496        try {
497          oldTd = tds.get(tableName);
498        } catch (IOException e) {
499          LOG.warn("Failed to migrate {} in group {}", tableName, groupInfo.getName(), e);
500          failedTables.add(tableName);
501          continue;
502        }
503        if (oldTd == null) {
504          continue;
505        }
506        if (oldTd.getRegionServerGroup().isPresent()) {
507          // either we have already migrated it or that user has set the rs group using the new
508          // code which will set the group directly on table descriptor, skip.
509          LOG.debug("Skip migrating {} since it is already in group {}", tableName,
510            oldTd.getRegionServerGroup().get());
511          continue;
512        }
513        TableDescriptor newTd = TableDescriptorBuilder.newBuilder(oldTd)
514          .setRegionServerGroup(groupInfo.getName()).build();
515        // This is a bit tricky. Since we know that the region server group config in
516        // TableDescriptor will only be used at master side, it is fine to just update the table
517        // descriptor on file system and also the cache, without reopening all the regions. This
518        // will be much faster than the normal modifyTable. And when upgrading, we will update
519        // master first and then region server, so after all the region servers has been reopened,
520        // the new TableDescriptor will be loaded.
521        try {
522          tds.update(newTd);
523        } catch (IOException e) {
524          LOG.warn("Failed to migrate {} in group {}", tableName, groupInfo.getName(), e);
525          failedTables.add(tableName);
526          continue;
527        }
528      }
529      LOG.debug("Done migrating {}, failed tables {}", groupInfo.getName(), failedTables);
530      synchronized (RSGroupInfoManagerImpl.this) {
531        Map<String, RSGroupInfo> rsGroupMap = holder.groupName2Group;
532        RSGroupInfo currentInfo = rsGroupMap.get(groupInfo.getName());
533        if (currentInfo != null) {
534          RSGroupInfo newInfo =
535            new RSGroupInfo(currentInfo.getName(), currentInfo.getServers(), failedTables);
536          Map<String, RSGroupInfo> newGroupMap = new HashMap<>(rsGroupMap);
537          newGroupMap.put(groupInfo.getName(), newInfo);
538          try {
539            flushConfig(newGroupMap);
540          } catch (IOException e) {
541            LOG.warn("Failed to persist rs group {}", newInfo.getName(), e);
542          }
543        }
544      }
545    }
546  }

548  // Migrate the table rs group info from RSGroupInfo into the table descriptor
549  // Notice that we do not want to block the initialize so this will be done in background, and
550  // during the migrating, the rs group info maybe incomplete and cause region to be misplaced.
551  private void migrate() {
552    Thread migrateThread = new Thread(MIGRATE_THREAD_NAME) {
553
554      @Override
555      public void run() {
556        LOG.info("Start migrating table rs group config");
557        while (!masterServices.isStopped()) {
558          Collection<RSGroupInfo> groups = holder.groupName2Group.values();
559          boolean hasTables = groups.stream().anyMatch(r -> !r.getTables().isEmpty());
560          if (!hasTables) {
561            break;
562          }
563          migrate(groups);
564        }
565        LOG.info("Done migrating table rs group info");
566      }
567    };
568    migrateThread.setDaemon(true);
569    migrateThread.start();
570  }

572  /**
573   * Read rsgroup info from the source of truth, the hbase:rsgroup table. Update zk cache. Called on
574   * startup of the manager.
575   */
576  private synchronized void refresh(boolean forceOnline) throws IOException {
577    List<RSGroupInfo> groupList = new ArrayList<>();
578
579    // Overwrite anything read from zk, group table is source of truth
580    // if online read from GROUP table
581    if (forceOnline || isOnline()) {
582      LOG.debug("Refreshing in Online mode.");
583      groupList.addAll(retrieveGroupListFromGroupTable());
584    } else {
585      LOG.debug("Refreshing in Offline mode.");
586      groupList.addAll(retrieveGroupListFromZookeeper());
587    }
588
589    // This is added to the last of the list so it overwrites the 'default' rsgroup loaded
590    // from region group table or zk
591    groupList.add(new RSGroupInfo(RSGroupInfo.DEFAULT_GROUP, getDefaultServers(groupList)));
592
593    // populate the data
594    HashMap<String, RSGroupInfo> newGroupMap = Maps.newHashMap();
595    for (RSGroupInfo group : groupList) {
596      newGroupMap.put(group.getName(), group);
597    }
598    resetRSGroupMap(newGroupMap);
599    updateCacheOfRSGroups(newGroupMap.keySet());
600  }

602  private void flushConfigTable(Map<String, RSGroupInfo> groupMap) throws IOException {
603    List<Mutation> mutations = Lists.newArrayList();
604
605    // populate deletes
606    for (String groupName : prevRSGroups) {
607      if (!groupMap.containsKey(groupName)) {
608        Delete d = new Delete(Bytes.toBytes(groupName));
609        mutations.add(d);
610      }
611    }
612
613    // populate puts
614    for (RSGroupInfo gi : groupMap.values()) {
615      if (!gi.getName().equals(RSGroupInfo.DEFAULT_GROUP)) {
616        RSGroupProtos.RSGroupInfo proto = ProtobufUtil.toProtoGroupInfo(gi);
617        Put p = new Put(Bytes.toBytes(gi.getName()));
618        p.addColumn(META_FAMILY_BYTES, META_QUALIFIER_BYTES, proto.toByteArray());
619        mutations.add(p);
620      }
621    }
622
623    if (mutations.size() > 0) {
624      multiMutate(mutations);
625    }
626  }

628  private synchronized void flushConfig() throws IOException {
629    flushConfig(holder.groupName2Group);
630  }

632  private synchronized void flushConfig(Map<String, RSGroupInfo> newGroupMap) throws IOException {
633    // For offline mode persistence is still unavailable
634    // We're refreshing in-memory state but only for servers in default group
635    if (!isOnline()) {
636      if (newGroupMap == holder.groupName2Group) {
637        // When newGroupMap is this.rsGroupMap itself,
638        // do not need to check default group and other groups as followed
639        return;
640      }
641
642      Map<String, RSGroupInfo> oldGroupMap = Maps.newHashMap(holder.groupName2Group);
643      RSGroupInfo oldDefaultGroup = oldGroupMap.remove(RSGroupInfo.DEFAULT_GROUP);
644      RSGroupInfo newDefaultGroup = newGroupMap.remove(RSGroupInfo.DEFAULT_GROUP);
645      if (!oldGroupMap.equals(newGroupMap) /* compare both tables and servers in other groups */ ||
646          !oldDefaultGroup.getTables().equals(newDefaultGroup.getTables())
647      /* compare tables in default group */) {
648        throw new IOException("Only servers in default group can be updated during offline mode");
649      }
650
651      // Restore newGroupMap by putting its default group back
652      newGroupMap.put(RSGroupInfo.DEFAULT_GROUP, newDefaultGroup);
653
654      // Refresh rsGroupMap
655      // according to the inputted newGroupMap (an updated copy of rsGroupMap)
656      this.holder = new RSGroupInfoHolder(newGroupMap);
657
658      // Do not need to update tableMap
659      // because only the update on servers in default group is allowed above,
660      // or IOException will be thrown
661      return;
662    }
663
664    /* For online mode, persist to hbase:rsgroup and Zookeeper */
665    flushConfigTable(newGroupMap);
666
667    // Make changes visible after having been persisted to the source of truth
668    resetRSGroupMap(newGroupMap);
669    saveRSGroupMapToZK(newGroupMap);
670    updateCacheOfRSGroups(newGroupMap.keySet());
671  }

673  private void saveRSGroupMapToZK(Map<String, RSGroupInfo> newGroupMap) throws IOException {
674    try {
675      String groupBasePath =
676          ZNodePaths.joinZNode(watcher.getZNodePaths().baseZNode, RS_GROUP_ZNODE);
677      ZKUtil.createAndFailSilent(watcher, groupBasePath, ProtobufMagic.PB_MAGIC);
678
679      List<ZKUtil.ZKUtilOp> zkOps = new ArrayList<>(newGroupMap.size());
680      for (String groupName : prevRSGroups) {
681        if (!newGroupMap.containsKey(groupName)) {
682          String znode = ZNodePaths.joinZNode(groupBasePath, groupName);
683          zkOps.add(ZKUtil.ZKUtilOp.deleteNodeFailSilent(znode));
684        }
685      }
686
687      for (RSGroupInfo gi : newGroupMap.values()) {
688        if (!gi.getName().equals(RSGroupInfo.DEFAULT_GROUP)) {
689          String znode = ZNodePaths.joinZNode(groupBasePath, gi.getName());
690          RSGroupProtos.RSGroupInfo proto = ProtobufUtil.toProtoGroupInfo(gi);
691          LOG.debug("Updating znode: " + znode);
692          ZKUtil.createAndFailSilent(watcher, znode);
693          zkOps.add(ZKUtil.ZKUtilOp.deleteNodeFailSilent(znode));
694          zkOps.add(ZKUtil.ZKUtilOp.createAndFailSilent(znode,
695            ProtobufUtil.prependPBMagic(proto.toByteArray())));
696        }
697      }
698      LOG.debug("Writing ZK GroupInfo count: " + zkOps.size());
699
700      ZKUtil.multiOrSequential(watcher, zkOps, false);
701    } catch (KeeperException e) {
702      LOG.error("Failed to write to rsGroupZNode", e);
703      masterServices.abort("Failed to write to rsGroupZNode", e);
704      throw new IOException("Failed to write to rsGroupZNode", e);
705    }
706  }

750  private class RSGroupStartupWorker extends Thread {
751    private final Logger LOG = LoggerFactory.getLogger(RSGroupStartupWorker.class);
752    private volatile boolean online = false;
753
754    RSGroupStartupWorker() {
755      super(RSGroupStartupWorker.class.getName() + "-" + masterServices.getServerName());
756      setDaemon(true);
757    }
758
759    @Override
760    public void run() {
761      if (waitForGroupTableOnline()) {
762        LOG.info("GroupBasedLoadBalancer is now online");
763      } else {
764        LOG.warn("Quit without making region group table online");
765      }
766    }
767
768    private boolean waitForGroupTableOnline() {
769      while (isMasterRunning(masterServices)) {
770        try {
771          TableStateManager tsm = masterServices.getTableStateManager();
772          if (!tsm.isTablePresent(RSGROUP_TABLE_NAME)) {
773            createRSGroupTable();
774          }
775          // try reading from the table
776          FutureUtils.get(conn.getTable(RSGROUP_TABLE_NAME).get(new Get(ROW_KEY)));
777          LOG.info("RSGroup table={} is online, refreshing cached information", RSGROUP_TABLE_NAME);
778          RSGroupInfoManagerImpl.this.refresh(true);
779          online = true;
780          // flush any inconsistencies between ZK and HTable
781          RSGroupInfoManagerImpl.this.flushConfig();
782          // migrate after we are online.
783          migrate();
784          return true;
785        } catch (Exception e) {
786          LOG.warn("Failed to perform check", e);
787          // 100ms is short so let's just ignore the interrupt
788          Threads.sleepWithoutInterrupt(100);
789        }
790      }
791      return false;
792    }
793
794    private void createRSGroupTable() throws IOException {
795      OptionalLong optProcId = masterServices.getProcedures().stream()
796        .filter(p -> p instanceof CreateTableProcedure).map(p -> (CreateTableProcedure) p)
797        .filter(p -> p.getTableName().equals(RSGROUP_TABLE_NAME)).mapToLong(Procedure::getProcId)
798        .findFirst();
799      long procId;
800      if (optProcId.isPresent()) {
801        procId = optProcId.getAsLong();
802      } else {
803        procId = masterServices.createSystemTable(RSGROUP_TABLE_DESC);
804      }
805      // wait for region to be online
806      int tries = 600;
807      while (!(masterServices.getMasterProcedureExecutor().isFinished(procId)) &&
808        masterServices.getMasterProcedureExecutor().isRunning() && tries > 0) {
809        try {
810          Thread.sleep(100);
811        } catch (InterruptedException e) {
812          throw new IOException("Wait interrupted ", e);
813        }
814        tries--;
815      }
816      if (tries <= 0) {
817        throw new IOException("Failed to create group table in a given time.");
818      } else {
819        Procedure<?> result = masterServices.getMasterProcedureExecutor().getResult(procId);
820        if (result != null && result.isFailed()) {
821          throw new IOException("Failed to create group table. " +
822              MasterProcedureUtil.unwrapRemoteIOException(result));
823        }
824      }
825    }

836  private void multiMutate(List<Mutation> mutations) throws IOException {
837    MutateRowsRequest.Builder builder = MutateRowsRequest.newBuilder();
838    for (Mutation mutation : mutations) {
839      if (mutation instanceof Put) {
840        builder
841            .addMutationRequest(ProtobufUtil.toMutation(MutationProto.MutationType.PUT, mutation));
842      } else if (mutation instanceof Delete) {
843        builder.addMutationRequest(
844          ProtobufUtil.toMutation(MutationProto.MutationType.DELETE, mutation));
845      } else {
846        throw new DoNotRetryIOException(
847            "multiMutate doesn't support " + mutation.getClass().getName());
848      }
849    }
850    MutateRowsRequest request = builder.build();
851    AsyncTable<?> table = conn.getTable(RSGROUP_TABLE_NAME);
852    FutureUtils.get(table.<MultiRowMutationService, MutateRowsResponse> coprocessorService(
853      MultiRowMutationService::newStub,
854      (stub, controller, done) -> stub.mutateRows(controller, request, done), ROW_KEY));
855  }

870   * Check if the set of servers are belong to dead servers list or online servers list.
871   * @param servers servers to remove
872   */
873  private void checkForDeadOrOnlineServers(Set<Address> servers) throws IOException {
874    // This uglyness is because we only have Address, not ServerName.
875    Set<Address> onlineServers = new HashSet<>();
876    List<ServerName> drainingServers = masterServices.getServerManager().getDrainingServersList();
877    for (ServerName server : masterServices.getServerManager().getOnlineServers().keySet()) {
878      // Only online but not decommissioned servers are really online
879      if (!drainingServers.contains(server)) {
880        onlineServers.add(server.getAddress());
881      }
882    }
883
884    Set<Address> deadServers = new HashSet<>();
885    for(ServerName server: masterServices.getServerManager().getDeadServers().copyServerNames()) {
886      deadServers.add(server.getAddress());
887    }
888
889    for (Address address: servers) {
890      if (onlineServers.contains(address)) {
891        throw new DoNotRetryIOException(
892            "Server " + address + " is an online server, not allowed to remove.");
893      }
894      if (deadServers.contains(address)) {
895        throw new DoNotRetryIOException(
896            "Server " + address + " is on the dead servers list,"
897                + " Maybe it will come back again, not allowed to remove.");
898      }
899    }
900  }

934          state.getRegionLocation().getAddress().equals(server)) {

937    }
938    return regions;

977  private <T> void moveRegionsBetweenGroups(Set<T> regionsOwners, Set<Address> newRegionsOwners,
978    String targetGroupName, String sourceGroupName, Function<T, List<RegionInfo>> getRegionsInfo,
979    Function<RegionInfo, Boolean> validation) throws IOException {
980    // Get server names corresponding to given Addresses
981    List<ServerName> movedServerNames = new ArrayList<>(regionsOwners.size());
982    List<ServerName> srcGrpServerNames = new ArrayList<>(newRegionsOwners.size());
983    for (ServerName serverName : masterServices.getServerManager().getOnlineServers().keySet()) {
984      // In case region move failed in previous attempt, regionsOwners and newRegionsOwners
985      // can have the same servers. So for all servers below both conditions to be checked
986      if (newRegionsOwners.contains(serverName.getAddress())) {
987        srcGrpServerNames.add(serverName);
988      }
989      if (regionsOwners.contains(serverName.getAddress())) {
990        movedServerNames.add(serverName);
991      }
992    }
993    List<Pair<RegionInfo, Future<byte[]>>> assignmentFutures = new ArrayList<>();
994    int retry = 0;
995    Set<String> failedRegions = new HashSet<>();
996    IOException toThrow = null;
997    do {
998      assignmentFutures.clear();
999      failedRegions.clear();
1000      for (ServerName owner : movedServerNames) {
1001        // Get regions that are associated with this server and filter regions by group tables.
1002        for (RegionInfo region : getRegionsInfo.apply((T) owner.getAddress())) {
1003          if (!validation.apply(region)) {
1004            LOG.info("Moving region {}, which do not belong to RSGroup {}",
1005              region.getShortNameToLog(), targetGroupName);
1006            // Move region back to source RSGroup servers
1007            ServerName dest =
1008              masterServices.getLoadBalancer().randomAssignment(region, srcGrpServerNames);
1009            if (dest == null) {
1010              failedRegions.add(region.getRegionNameAsString());
1011              continue;
1012            }
1013            RegionPlan rp = new RegionPlan(region, owner, dest);
1014            try {
1015              Future<byte[]> future = masterServices.getAssignmentManager().moveAsync(rp);
1016              assignmentFutures.add(Pair.newPair(region, future));
1017            } catch (IOException ioe) {
1018              failedRegions.add(region.getRegionNameAsString());
1019              LOG.debug("Move region {} failed, will retry, current retry time is {}",
1020                region.getShortNameToLog(), retry, ioe);
1021              toThrow = ioe;
1022            }
1023          }
1024        }
1025      }
1026      waitForRegionMovement(assignmentFutures, failedRegions, sourceGroupName, retry);
1027      if (failedRegions.isEmpty()) {
1028        LOG.info("All regions from {} are moved back to {}", movedServerNames, sourceGroupName);
1029        return;
1030      } else {
1031        try {
1032          wait(1000);
1033        } catch (InterruptedException e) {
1034          LOG.warn("Sleep interrupted", e);
1035          Thread.currentThread().interrupt();
1036        }
1037        retry++;
1038      }
1039    } while (!failedRegions.isEmpty() && retry <= masterServices.getConfiguration()
1040      .getInt(FAILED_MOVE_MAX_RETRY, DEFAULT_MAX_RETRY_VALUE));
1041
1042    //has up to max retry time or there are no more regions to move
1043    if (!failedRegions.isEmpty()) {
1044      // print failed moved regions, for later process conveniently
1045      String msg = String
1046        .format("move regions for group %s failed, failed regions: %s", sourceGroupName,
1047          failedRegions);
1048      LOG.error(msg);
1049      throw new DoNotRetryIOException(
1050        msg + ", just record the last failed region's cause, more details in server log", toThrow);
1051    }
1052  }

1054  /**
1055   * Wait for all the region move to complete. Keep waiting for other region movement
1056   * completion even if some region movement fails.
1057   */
1058  private void waitForRegionMovement(List<Pair<RegionInfo, Future<byte[]>>> regionMoveFutures,
1059    Set<String> failedRegions, String sourceGroupName, int retryCount) {
1060    LOG.info("Moving {} region(s) to group {}, current retry={}", regionMoveFutures.size(),
1061      sourceGroupName, retryCount);
1062    for (Pair<RegionInfo, Future<byte[]>> pair : regionMoveFutures) {
1063      try {
1064        pair.getSecond().get();
1065        if (masterServices.getAssignmentManager().getRegionStates().
1066          getRegionState(pair.getFirst()).isFailedOpen()) {
1067          failedRegions.add(pair.getFirst().getRegionNameAsString());
1068        }
1069      } catch (InterruptedException e) {
1070        //Dont return form there lets wait for other regions to complete movement.
1071        failedRegions.add(pair.getFirst().getRegionNameAsString());
1072        LOG.warn("Sleep interrupted", e);
1073      } catch (Exception e) {
1074        failedRegions.add(pair.getFirst().getRegionNameAsString());
1075        LOG.error("Move region {} to group {} failed, will retry on next attempt",
1076          pair.getFirst().getShortNameToLog(), sourceGroupName, e);
1077      }
1078    }
1079  }

1109  /**
1110   * This is an EXPENSIVE clone. Cloning though is the safest thing to do. Can't let out original
1111   * since it can change and at least the load balancer wants to iterate this exported list. Load
1112   * balancer should iterate over this list because cloned list will ignore disabled table and split
1113   * parent region cases. This method is invoked by {@link #balanceRSGroup}
1114   * @return A clone of current assignments for this group.
1115   */
1116  Map<TableName, Map<ServerName, List<RegionInfo>>> getRSGroupAssignmentsByTable(
1117    TableStateManager tableStateManager, String groupName) throws IOException {
1118    Map<TableName, Map<ServerName, List<RegionInfo>>> result = Maps.newHashMap();
1119    Set<TableName> tablesInGroupCache = new HashSet<>();
1120    for (Map.Entry<RegionInfo, ServerName> entry : masterServices.getAssignmentManager()
1121      .getRegionStates().getRegionAssignments().entrySet()) {
1122      RegionInfo region = entry.getKey();
1123      TableName tn = region.getTable();
1124      ServerName server = entry.getValue();
1125      if (isTableInGroup(tn, groupName, tablesInGroupCache)) {
1126        if (tableStateManager
1127          .isTableState(tn, TableState.State.DISABLED, TableState.State.DISABLING)) {
1128          continue;
1129        }
1130        if (region.isSplitParent()) {
1131          continue;
1132        }
1133        result.computeIfAbsent(tn, k -> new HashMap<>())
1134          .computeIfAbsent(server, k -> new ArrayList<>()).add(region);
1135      }
1136    }
1137    RSGroupInfo rsGroupInfo = getRSGroupInfo(groupName);
1138    for (ServerName serverName : masterServices.getServerManager().getOnlineServers().keySet()) {
1139      if (rsGroupInfo.containsServer(serverName.getAddress())) {
1140        for (Map<ServerName, List<RegionInfo>> map : result.values()) {
1141          map.computeIfAbsent(serverName, k -> Collections.emptyList());
1142        }
1143      }
1144    }
1145    return result;
1146  }

1148  @Override
1149  public boolean balanceRSGroup(String groupName) throws IOException {
1150    ServerManager serverManager = masterServices.getServerManager();
1151    LoadBalancer balancer = masterServices.getLoadBalancer();
1152    getRSGroupInfo(groupName);
1153
1154    synchronized (balancer) {
1155      // If balance not true, don't run balancer.
1156      if (!masterServices.isBalancerOn()) {
1157        return false;
1158      }
1159      // Only allow one balance run at at time.
1160      Map<String, RegionState> groupRIT = rsGroupGetRegionsInTransition(groupName);
1161      if (groupRIT.size() > 0) {
1162        LOG.debug("Not running balancer because {} region(s) in transition: {}", groupRIT.size(),
1163            StringUtils.abbreviate(masterServices.getAssignmentManager().getRegionStates()
1164                    .getRegionsInTransition().toString(),
1165                256));
1166        return false;
1167      }
1168      if (serverManager.areDeadServersInProgress()) {
1169        LOG.debug("Not running balancer because processing dead regionserver(s): {}",
1170            serverManager.getDeadServers());
1171        return false;
1172      }
1173
1174      // We balance per group instead of per table
1175      Map<TableName, Map<ServerName, List<RegionInfo>>> assignmentsByTable =
1176          getRSGroupAssignmentsByTable(masterServices.getTableStateManager(), groupName);
1177      List<RegionPlan> plans = balancer.balanceCluster(assignmentsByTable);
1178      boolean balancerRan = !plans.isEmpty();
1179      if (balancerRan) {
1180        LOG.info("RSGroup balance {} starting with plan count: {}", groupName, plans.size());
1181        masterServices.executeRegionPlansWithThrottling(plans);
1182        LOG.info("RSGroup balance " + groupName + " completed");
1183      }
1184      return balancerRan;
1185    }
1186  }

1188  private void moveTablesAndWait(Set<TableName> tables, String targetGroup) throws IOException {
1189    List<Long> procIds = new ArrayList<Long>();
1190    for (TableName tableName : tables) {
1191      TableDescriptor oldTd = masterServices.getTableDescriptors().get(tableName);
1192      if (oldTd == null) {
1193        continue;
1194      }
1195      TableDescriptor newTd =
1196          TableDescriptorBuilder.newBuilder(oldTd).setRegionServerGroup(targetGroup).build();
1197      procIds.add(masterServices.modifyTable(tableName, newTd, HConstants.NO_NONCE,
1198          HConstants.NO_NONCE));
1199    }
1200    for (long procId : procIds) {
1201      Procedure<?> proc = masterServices.getMasterProcedureExecutor().getProcedure(procId);
1202      if (proc == null) {
1203        continue;
1204      }
1205      ProcedureSyncWait.waitForProcedureToCompleteIOE(masterServices.getMasterProcedureExecutor(),
1206          proc, Long.MAX_VALUE);
1207    }
1208  }

1216  public void moveServers(Set<Address> servers, String targetGroupName) throws IOException {
1217    if (servers == null) {
1218      throw new ConstraintException("The list of servers to move cannot be null.");
1219    }
1220    if (servers.isEmpty()) {
1221      // For some reason this difference between null servers and isEmpty is important distinction.
1222      // TODO. Why? Stuff breaks if I equate them.
1223      return;
1224    }
1225    if (StringUtils.isEmpty(targetGroupName)) {
1226      throw new ConstraintException("RSGroup cannot be null.");
1227    }
1228
1229    // Hold a lock on the manager instance while moving servers to prevent
1230    // another writer changing our state while we are working.
1231    synchronized (this) {
1232      // Presume first server's source group. Later ensure all servers are from this group.
1233      Address firstServer = servers.iterator().next();
1234      RSGroupInfo srcGrp = getRSGroupOfServer(firstServer);
1235      if (srcGrp == null) {
1236        // Be careful. This exception message is tested for in TestRSGroupsBase...
1237        throw new ConstraintException("Source RSGroup for server " + firstServer
1238            + " does not exist.");
1239      }
1240
1241      // Only move online servers (when moving from 'default') or servers from other
1242      // groups. This prevents bogus servers from entering groups
1243      if (RSGroupInfo.DEFAULT_GROUP.equals(srcGrp.getName())) {
1244        if (srcGrp.getServers().size() <= servers.size()) {
1245          throw new ConstraintException(KEEP_ONE_SERVER_IN_DEFAULT_ERROR_MESSAGE);
1246        }
1247        checkOnlineServersOnly(servers);
1248      }
1249      // Ensure all servers are of same rsgroup.
1250      for (Address server: servers) {
1251        String tmpGroup = getRSGroupOfServer(server).getName();
1252        if (!tmpGroup.equals(srcGrp.getName())) {
1253          throw new ConstraintException("Move server request should only come from one source " +
1254              "RSGroup. Expecting only " + srcGrp.getName() + " but contains " + tmpGroup);
1255        }
1256      }
1257      if (srcGrp.getServers().size() <= servers.size()) {
1258        // check if there are still tables reference this group
1259        for (TableDescriptor td : masterServices.getTableDescriptors().getAll().values()) {
1260          Optional<String> optGroupName = td.getRegionServerGroup();
1261          if (optGroupName.isPresent() && optGroupName.get().equals(srcGrp.getName())) {
1262            throw new ConstraintException(
1263                "Cannot leave a RSGroup " + srcGrp.getName() + " that contains tables('" +
1264                    td.getTableName() + "' at least) without servers to host them.");
1265          }
1266        }
1267      }
1268
1269      // MovedServers may be < passed in 'servers'.
1270      Set<Address> movedServers = moveServers(servers, srcGrp.getName(),
1271          targetGroupName);
1272      moveServerRegionsFromGroup(movedServers, srcGrp.getServers(), targetGroupName, srcGrp.getName());
1273      LOG.info("Move servers done: {} => {}", srcGrp.getName(), targetGroupName);
1274    }
1275  }

1282  @Override
1283  public synchronized void renameRSGroup(String oldName, String newName) throws IOException {
1284    if (oldName.equals(RSGroupInfo.DEFAULT_GROUP)) {
1285      throw new ConstraintException(RSGroupInfo.DEFAULT_GROUP + " can't be rename");
1286    }
1287    checkGroupName(newName);
1288    //getRSGroupInfo validates old RSGroup existence.
1289    RSGroupInfo oldRSG = getRSGroupInfo(oldName);
1290    Map<String, RSGroupInfo> rsGroupMap = holder.groupName2Group;
1291    if (rsGroupMap.containsKey(newName)) {
1292      throw new ConstraintException("Group already exists: " + newName);
1293    }
1294
1295    Map<String, RSGroupInfo> newGroupMap = Maps.newHashMap(rsGroupMap);
1296    newGroupMap.remove(oldRSG.getName());
1297    RSGroupInfo newRSG = new RSGroupInfo(newName, oldRSG.getServers());
1298    newGroupMap.put(newName, newRSG);
1299    flushConfig(newGroupMap);
1300    Set<TableName> updateTables =
1301      masterServices.getTableDescriptors().getAll().values()
1302                    .stream()
1303                    .filter(t -> oldName.equals(t.getRegionServerGroup().orElse(null)))
1304                    .map(TableDescriptor::getTableName)
1305                    .collect(Collectors.toSet());
1306    setRSGroup(updateTables, newName);
1307  }

1309  @Override
1310  public synchronized void updateRSGroupConfig(String groupName, Map<String, String> configuration)
1311      throws IOException {
1312    if (RSGroupInfo.DEFAULT_GROUP.equals(groupName)) {
1313      // We do not persist anything of default group, therefore, it is not supported to update
1314      // default group's configuration which lost once master down.
1315      throw new ConstraintException("configuration of " + RSGroupInfo.DEFAULT_GROUP
1316        + " can't be stored persistently");
1317    }
1318    RSGroupInfo rsGroupInfo = getRSGroupInfo(groupName);
1319    rsGroupInfo.getConfiguration().forEach((k, v) -> rsGroupInfo.removeConfiguration(k));
1320    configuration.forEach((k, v) -> rsGroupInfo.setConfiguration(k, v));
1321    flushConfig();
1322  }
