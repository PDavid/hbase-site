011package org.apache.hadoop.hbase.rsgroup;
012
013import java.io.IOException;
014import java.util.ArrayList;
015import java.util.Collection;
016import java.util.List;
017import java.util.Optional;
018import java.util.function.Predicate;
019import org.apache.hadoop.conf.Configuration;
020import org.apache.hadoop.hbase.NamespaceDescriptor;
021import org.apache.hadoop.hbase.TableName;
022import org.apache.hadoop.hbase.client.TableDescriptor;
023import org.apache.hadoop.hbase.master.ClusterSchema;
024import org.apache.hadoop.hbase.master.MasterServices;
025import org.apache.yetus.audience.InterfaceAudience;
026import org.slf4j.Logger;
027import org.slf4j.LoggerFactory;
028
029/**
030 * Helper class for RSGroup implementation
031 */
032@InterfaceAudience.Private
033public final class RSGroupUtil {
034
035  private static final Logger LOG = LoggerFactory.getLogger(RSGroupUtil.class);
036
037  public static final String RS_GROUP_ENABLED = "hbase.balancer.rsgroup.enabled";
038
039  private RSGroupUtil() {
040  }
041
042  public static boolean isRSGroupEnabled(Configuration conf) {
043    return conf.getBoolean(RS_GROUP_ENABLED, false);
044  }
045
046  public static void enableRSGroup(Configuration conf) {
047    conf.setBoolean(RS_GROUP_ENABLED, true);
048  }
049
050  public static List<TableName> listTablesInRSGroup(MasterServices master, String groupName)
051    throws IOException {
052    List<TableName> tables = new ArrayList<>();
053    boolean isDefaultGroup = RSGroupInfo.DEFAULT_GROUP.equals(groupName);
054    for (TableDescriptor td : master.getTableDescriptors().getAll().values()) {
055      // no config means in default group
056      if (RSGroupUtil.getRSGroupInfo(master, master.getRSGroupInfoManager(), td.getTableName())
057        .map(g -> g.getName().equals(groupName)).orElse(isDefaultGroup)) {
058        tables.add(td.getTableName());
059      }
060    }
061    return tables;
062  }
063
064  /**
065   * Will try to get the rsgroup from {@link TableDescriptor} first, and then try to get the rsgroup
066   * from the {@link NamespaceDescriptor}. If still not present, return empty.
067   */
068  public static Optional<RSGroupInfo> getRSGroupInfo(MasterServices master,
069      RSGroupInfoManager manager, TableName tableName) throws IOException {
070    TableDescriptor td = master.getTableDescriptors().get(tableName);
071    if (td == null) {
072      return Optional.empty();
073    }
074    // RSGroup information determined by client.
075    Optional<String> optGroupNameOfTable = td.getRegionServerGroup();
076    if (optGroupNameOfTable.isPresent()) {
077      RSGroupInfo group = manager.getRSGroup(optGroupNameOfTable.get());
078      if (group != null) {
079        return Optional.of(group);
080      }
081    }
082    // for backward compatible, where we may still have table configs in the RSGroupInfo after
083    // upgrading when migrating is still on-going.
084    RSGroupInfo groupFromOldRSGroupInfo = manager.getRSGroupForTable(tableName);
085    if (groupFromOldRSGroupInfo != null) {
086      return Optional.of(groupFromOldRSGroupInfo);
087    }
088    // RSGroup information determined by administrator.
089    String groupDeterminedByAdmin = manager.determineRSGroupInfoForTable(tableName);
090    RSGroupInfo groupInfo = null;
091    if (groupDeterminedByAdmin != null) {
092      groupInfo = manager.getRSGroup(groupDeterminedByAdmin);
093    }
094    if (groupInfo != null) {
095      return Optional.of(groupInfo);
096    }
097    // Finally, we will try to fall back to namespace as rsgroup if exists
098    ClusterSchema clusterSchema = master.getClusterSchema();
099    if (clusterSchema == null) {
100      if (TableName.isMetaTableName(tableName)) {
101        LOG.info("Can not get the namespace rs group config for meta table, since the" +
102            " meta table is not online yet, will use default group to assign meta first");
103      } else {
104        LOG.warn("ClusterSchema is null, can only use default rsgroup, should not happen?");
105      }
106      return Optional.empty();
107    }
108    NamespaceDescriptor nd = clusterSchema.getNamespace(tableName.getNamespaceAsString());
109    String groupNameOfNs = nd.getConfigurationValue(RSGroupInfo.NAMESPACE_DESC_PROP_GROUP);
110    if (groupNameOfNs == null) {
111      return Optional.empty();
112    }
113    return Optional.ofNullable(manager.getRSGroup(groupNameOfNs));
114  }
115
116  /**
117   * Fill the tables field for {@link RSGroupInfo}, for backward compatibility.
118   */
119  @SuppressWarnings("deprecation")
120  public static RSGroupInfo fillTables(RSGroupInfo rsGroupInfo, Collection<TableDescriptor> tds) {
121    RSGroupInfo newRsGroupInfo = new RSGroupInfo(rsGroupInfo);
122    Predicate<TableDescriptor> filter;
123    if (rsGroupInfo.getName().equals(RSGroupInfo.DEFAULT_GROUP)) {
124      filter = td -> {
125        Optional<String> optGroupName = td.getRegionServerGroup();
126        return !optGroupName.isPresent() || optGroupName.get().equals(RSGroupInfo.DEFAULT_GROUP);
127      };
128    } else {
129      filter = td -> {
130        Optional<String> optGroupName = td.getRegionServerGroup();
131        return optGroupName.isPresent() && optGroupName.get().equals(newRsGroupInfo.getName());
132      };
133    }
134    tds.stream().filter(filter).map(TableDescriptor::getTableName)
135        .forEach(newRsGroupInfo::addTable);
136    return newRsGroupInfo;
137  }
138}
