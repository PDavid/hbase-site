123  @VisibleForTesting
124  WALSplitter(final WALFactory factory, Configuration conf, Path walDir, FileSystem walFS,
125      Path rootDir, FileSystem rootFS, LastSequenceId idChecker,
126      SplitLogWorkerCoordination splitLogWorkerCoordination, RegionServerServices rsServices) {
127    this.conf = HBaseConfiguration.create(conf);
128    String codecClassName =
129        conf.get(WALCellCodec.WAL_CELL_CODEC_CLASS_KEY, WALCellCodec.class.getName());
130    this.conf.set(HConstants.RPC_CODEC_CONF_KEY, codecClassName);
131    this.walDir = walDir;
132    this.walFS = walFS;
133    this.rootDir = rootDir;
134    this.rootFS = rootFS;
135    this.sequenceIdChecker = idChecker;
136    this.splitLogWorkerCoordination = splitLogWorkerCoordination;
137    this.rsServices = rsServices;
138    if (rsServices != null) {
139      this.tableDescriptors = rsServices.getTableDescriptors();
140    } else {
141      this.tableDescriptors = new FSTableDescriptors(rootFS, rootDir, true, true);
142    }
143
144    this.walFactory = factory;
145    PipelineController controller = new PipelineController();
146    this.tmpDirName =
147      conf.get(HConstants.TEMPORARY_FS_DIRECTORY_KEY, HConstants.DEFAULT_TEMPORARY_HDFS_DIRECTORY);
148
149
150    // if we limit the number of writers opened for sinking recovered edits
151    boolean splitWriterCreationBounded = conf.getBoolean(SPLIT_WRITER_CREATION_BOUNDED, false);
152    boolean splitToHFile = conf.getBoolean(WAL_SPLIT_TO_HFILE, DEFAULT_WAL_SPLIT_TO_HFILE);
153    long bufferSize = this.conf.getLong(SPLIT_WAL_BUFFER_SIZE, 128 * 1024 * 1024);
154    int numWriterThreads = this.conf.getInt(SPLIT_WAL_WRITER_THREADS, 3);
155
156    if (splitToHFile) {
157      entryBuffers = new BoundedEntryBuffers(controller, bufferSize);
158      outputSink =
159          new BoundedRecoveredHFilesOutputSink(this, controller, entryBuffers, numWriterThreads);
160    } else if (splitWriterCreationBounded) {
161      entryBuffers = new BoundedEntryBuffers(controller, bufferSize);
162      outputSink =
163          new BoundedRecoveredEditsOutputSink(this, controller, entryBuffers, numWriterThreads);
164    } else {
165      entryBuffers = new EntryBuffers(controller, bufferSize);
166      outputSink = new RecoveredEditsOutputSink(this, controller, entryBuffers, numWriterThreads);
167    }
168  }

186  /**
187   * Splits a WAL file into region's recovered-edits directory.
188   * This is the main entry point for distributed log splitting from SplitLogWorker.
189   * <p>
190   * If the log file has N regions then N recovered.edits files will be produced.
191   * <p>
192   * @return false if it is interrupted by the progress-able.
193   */
194  public static boolean splitLogFile(Path walDir, FileStatus logfile, FileSystem walFS,
195      Configuration conf, CancelableProgressable reporter, LastSequenceId idChecker,
196      SplitLogWorkerCoordination splitLogWorkerCoordination, WALFactory factory,
197      RegionServerServices rsServices) throws IOException {
198    Path rootDir = CommonFSUtils.getRootDir(conf);
199    FileSystem rootFS = rootDir.getFileSystem(conf);
200    WALSplitter s = new WALSplitter(factory, conf, walDir, walFS, rootDir, rootFS, idChecker,
201        splitLogWorkerCoordination, rsServices);
202    return s.splitLogFile(logfile, reporter);
203  }

205  // A wrapper to split one log folder using the method used by distributed
206  // log splitting. Used by tools and unit tests. It should be package private.
207  // It is public only because TestWALObserver is in a different package,
208  // which uses this method to do log splitting.
209  @VisibleForTesting
210  public static List<Path> split(Path walDir, Path logDir, Path oldLogDir, FileSystem walFS,
211      Configuration conf, final WALFactory factory) throws IOException {
212    Path rootDir = CommonFSUtils.getRootDir(conf);
213    FileSystem rootFS = rootDir.getFileSystem(conf);
214    final FileStatus[] logfiles =
215        SplitLogManager.getFileList(conf, Collections.singletonList(logDir), null);
216    List<Path> splits = new ArrayList<>();
217    if (ArrayUtils.isNotEmpty(logfiles)) {
218      for (FileStatus logfile : logfiles) {
219        WALSplitter s =
220            new WALSplitter(factory, conf, walDir, walFS, rootDir, rootFS, null, null, null);
221        if (s.splitLogFile(logfile, null)) {
222          finishSplitLogFile(walDir, oldLogDir, logfile.getPath(), conf);
223          if (s.outputSink.splits != null) {
224            splits.addAll(s.outputSink.splits);
225          }
226        }
227      }
228    }
229    if (!walFS.delete(logDir, true)) {
230      throw new IOException("Unable to delete src dir: " + logDir);
231    }
232    return splits;
233  }

235  /**
236   * log splitting implementation, splits one log file.
237   * @param logfile should be an actual log file.
238   */
239  @VisibleForTesting
240  boolean splitLogFile(FileStatus logfile, CancelableProgressable reporter) throws IOException {
241    Preconditions.checkState(status == null);
242    Preconditions.checkArgument(logfile.isFile(),
243        "passed in file status is for something other than a regular file.");
244    boolean isCorrupted = false;
245    boolean skipErrors = conf.getBoolean("hbase.hlog.split.skip.errors",
246      SPLIT_SKIP_ERRORS_DEFAULT);
247    int interval = conf.getInt("hbase.splitlog.report.interval.loglines", 1024);
248    Path logPath = logfile.getPath();
249    boolean outputSinkStarted = false;
250    boolean progressFailed = false;
251    int editsCount = 0;
252    int editsSkipped = 0;
253
254    status = TaskMonitor.get().createStatus(
255          "Splitting log file " + logfile.getPath() + "into a temporary staging area.");
256    Reader logFileReader = null;
257    this.fileBeingSplit = logfile;
258    long startTS = EnvironmentEdgeManager.currentTime();
259    try {
260      long logLength = logfile.getLen();
261      LOG.info("Splitting WAL={}, size={} ({} bytes)", logPath, StringUtils.humanSize(logLength),
262          logLength);
263      status.setStatus("Opening log file");
264      if (reporter != null && !reporter.progress()) {
265        progressFailed = true;
266        return false;
267      }
268      logFileReader = getReader(logfile, skipErrors, reporter);
269      if (logFileReader == null) {
270        LOG.warn("Nothing to split in WAL={}", logPath);
271        return true;
272      }
273      long openCost = EnvironmentEdgeManager.currentTime() - startTS;
274      LOG.info("Open WAL={} cost {} ms", logPath, openCost);
275      int numOpenedFilesBeforeReporting = conf.getInt("hbase.splitlog.report.openedfiles", 3);
276      int numOpenedFilesLastCheck = 0;
277      outputSink.setReporter(reporter);
278      outputSink.startWriterThreads();
279      outputSinkStarted = true;
280      Entry entry;
281      Long lastFlushedSequenceId = -1L;
282      startTS = EnvironmentEdgeManager.currentTime();
283      while ((entry = getNextLogLine(logFileReader, logPath, skipErrors)) != null) {
284        byte[] region = entry.getKey().getEncodedRegionName();
285        String encodedRegionNameAsStr = Bytes.toString(region);
286        lastFlushedSequenceId = lastFlushedSequenceIds.get(encodedRegionNameAsStr);
287        if (lastFlushedSequenceId == null) {
288          if (sequenceIdChecker != null) {
289            RegionStoreSequenceIds ids = sequenceIdChecker.getLastSequenceId(region);
290            Map<byte[], Long> maxSeqIdInStores = new TreeMap<>(Bytes.BYTES_COMPARATOR);
291            for (StoreSequenceId storeSeqId : ids.getStoreSequenceIdList()) {
292              maxSeqIdInStores.put(storeSeqId.getFamilyName().toByteArray(),
293                storeSeqId.getSequenceId());
294            }
295            regionMaxSeqIdInStores.put(encodedRegionNameAsStr, maxSeqIdInStores);
296            lastFlushedSequenceId = ids.getLastFlushedSequenceId();
297            if (LOG.isDebugEnabled()) {
298              LOG.debug("DLS Last flushed sequenceid for " + encodedRegionNameAsStr + ": " +
299                  TextFormat.shortDebugString(ids));
300            }
301          }
302          if (lastFlushedSequenceId == null) {
303            lastFlushedSequenceId = -1L;
304          }
305          lastFlushedSequenceIds.put(encodedRegionNameAsStr, lastFlushedSequenceId);
306        }
307        if (lastFlushedSequenceId >= entry.getKey().getSequenceId()) {
308          editsSkipped++;
309          continue;
310        }
311        // Don't send Compaction/Close/Open region events to recovered edit type sinks.
312        if (entry.getEdit().isMetaEdit() && !outputSink.keepRegionEvent(entry)) {
313          editsSkipped++;
314          continue;
315        }
316        entryBuffers.appendEntry(entry);
317        editsCount++;
318        int moreWritersFromLastCheck = this.getNumOpenWriters() - numOpenedFilesLastCheck;
319        // If sufficient edits have passed, check if we should report progress.
320        if (editsCount % interval == 0
321            || moreWritersFromLastCheck > numOpenedFilesBeforeReporting) {
322          numOpenedFilesLastCheck = this.getNumOpenWriters();
323          String countsStr = (editsCount - (editsSkipped + outputSink.getTotalSkippedEdits()))
324              + " edits, skipped " + editsSkipped + " edits.";
325          status.setStatus("Split " + countsStr);
326          if (reporter != null && !reporter.progress()) {
327            progressFailed = true;
328            return false;
329          }
330        }
331      }
332    } catch (InterruptedException ie) {
333      IOException iie = new InterruptedIOException();
334      iie.initCause(ie);
335      throw iie;
336    } catch (CorruptedLogFileException e) {
337      LOG.warn("Could not parse, corrupted WAL={}", logPath, e);
338      if (splitLogWorkerCoordination != null) {
339        // Some tests pass in a csm of null.
340        splitLogWorkerCoordination.markCorrupted(walDir, logfile.getPath().getName(), walFS);
341      } else {
342        // for tests only
343        ZKSplitLog.markCorrupted(walDir, logfile.getPath().getName(), walFS);
344      }
345      isCorrupted = true;
346    } catch (IOException e) {
347      e = e instanceof RemoteException ? ((RemoteException) e).unwrapRemoteException() : e;
348      throw e;
349    } finally {
350      LOG.debug("Finishing writing output logs and closing down");
351      try {
352        if (null != logFileReader) {
353          logFileReader.close();
354        }
355      } catch (IOException exception) {
356        LOG.warn("Could not close WAL reader", exception);
357      }
358      try {
359        if (outputSinkStarted) {
360          // Set progress_failed to true as the immediate following statement will reset its value
361          // when close() throws exception, progress_failed has the right value
362          progressFailed = true;
363          progressFailed = outputSink.close() == null;
364        }
365      } finally {
366        long processCost = EnvironmentEdgeManager.currentTime() - startTS;
367        // See if length got updated post lease recovery
368        String msg = "Processed " + editsCount + " edits across " +
369            outputSink.getNumberOfRecoveredRegions() + " regions cost " + processCost +
370            " ms; edits skipped=" + editsSkipped + "; WAL=" + logPath + ", size=" +
371            StringUtils.humanSize(logfile.getLen()) + ", length=" + logfile.getLen() +
372            ", corrupted=" + isCorrupted + ", progress failed=" + progressFailed;
373        LOG.info(msg);
374        status.markComplete(msg);
375      }
376    }
377    return !progressFailed;
378  }

380  /**
381   * Create a new {@link Reader} for reading logs to split.
382   */
383  private Reader getReader(FileStatus file, boolean skipErrors, CancelableProgressable reporter)
384      throws IOException, CorruptedLogFileException {
385    Path path = file.getPath();
386    long length = file.getLen();
387    Reader in;
388
389    // Check for possibly empty file. With appends, currently Hadoop reports a
390    // zero length even if the file has been sync'd. Revisit if HDFS-376 or
391    // HDFS-878 is committed.
392    if (length <= 0) {
393      LOG.warn("File {} might be still open, length is 0", path);
394    }
395
396    try {
397      FSUtils.recoverFileLease(walFS, path, conf, reporter);
398      try {
399        in = getReader(path, reporter);
400      } catch (EOFException e) {
401        if (length <= 0) {
402          // TODO should we ignore an empty, not-last log file if skip.errors
403          // is false? Either way, the caller should decide what to do. E.g.
404          // ignore if this is the last log in sequence.
405          // TODO is this scenario still possible if the log has been
406          // recovered (i.e. closed)
407          LOG.warn("Could not open {} for reading. File is empty", path, e);
408        }
409        // EOFException being ignored
410        return null;
411      }
412    } catch (IOException e) {
413      if (e instanceof FileNotFoundException) {
414        // A wal file may not exist anymore. Nothing can be recovered so move on
415        LOG.warn("File {} does not exist anymore", path, e);
416        return null;
417      }
418      if (!skipErrors || e instanceof InterruptedIOException) {
419        throw e; // Don't mark the file corrupted if interrupted, or not skipErrors
420      }
421      throw new CorruptedLogFileException("skipErrors=true Could not open wal "
422        + path + " ignoring", e);
423    }
424    return in;
425  }
426
427  private Entry getNextLogLine(Reader in, Path path, boolean skipErrors)
428      throws CorruptedLogFileException, IOException {
429    try {
430      return in.next();
431    } catch (EOFException eof) {
432      // truncated files are expected if a RS crashes (see HBASE-2643)
433      LOG.info("EOF from wal {}. Continuing.", path);
434      return null;
435    } catch (IOException e) {
436      // If the IOE resulted from bad file format,
437      // then this problem is idempotent and retrying won't help
438      if (e.getCause() != null && (e.getCause() instanceof ParseException
439          || e.getCause() instanceof org.apache.hadoop.fs.ChecksumException)) {
440        LOG.warn("Parse exception from wal {}. Continuing", path, e);
441        return null;
442      }
443      if (!skipErrors) {
444        throw e;
445      }
446      throw new CorruptedLogFileException("skipErrors=true Ignoring exception"
447        + " while parsing wal " + path + ". Marking as corrupted", e);
448    }
449  }

478  /**
479   * Contains some methods to control WAL-entries producer / consumer interactions
480   */
481  public static class PipelineController {
482    // If an exception is thrown by one of the other threads, it will be
483    // stored here.
484    AtomicReference<Throwable> thrown = new AtomicReference<>();
485
486    // Wait/notify for when data has been produced by the writer thread,
487    // consumed by the reader thread, or an exception occurred
488    final Object dataAvailable = new Object();
489
490    void writerThreadError(Throwable t) {
491      thrown.compareAndSet(null, t);
492    }
493
494    /**
495     * Check for errors in the writer threads. If any is found, rethrow it.
496     */
497    void checkForErrors() throws IOException {
498      Throwable thrown = this.thrown.get();
499      if (thrown == null) {
500        return;
501      }
502      if (thrown instanceof IOException) {
503        throw new IOException(thrown);
504      } else {
505        throw new RuntimeException(thrown);
506      }
507    }
508  }
