057/**
058 * Base class for a WAL Provider.
059 * <p>
060 * We will put some common logic here, especially for sync replication implementation, as it must do
061 * some hacks before the normal wal creation operation.
062 * <p>
063 * All {@link WALProvider} implementations should extends this class instead of implement
064 * {@link WALProvider} directly, except {@link DisabledWALProvider}.
065 */
066@InterfaceAudience.Private
067public abstract class AbstractWALProvider implements WALProvider, PeerActionListener {
068
069  private static final Logger LOG = LoggerFactory.getLogger(AbstractWALProvider.class);
070
071  // should be package private; more visible for use in AbstractFSWAL
072  public static final String WAL_FILE_NAME_DELIMITER = ".";
073
074  protected WALFactory factory;
075  protected Configuration conf;
076  protected List<WALActionsListener> listeners = new ArrayList<>();
077  protected String providerId;
078  protected AtomicBoolean initialized = new AtomicBoolean(false);
079  // for default wal provider, logPrefix won't change
080  protected String logPrefix;
081  protected Abortable abortable;
082
083  // when switching from A to DA, we will put a Optional.empty into this map if there is no WAL for
084  // the peer yet. When getting WAL from this map the caller should know that it should not use
085  // the remote WAL any more.
086  private final ConcurrentMap<String, Optional<WAL>> peerId2WAL = new ConcurrentHashMap<>();
087
088  private final KeyLocker<String> createLock = new KeyLocker<>();
089
090  // we need to have this because when getting meta wal, there is no peer info provider yet.
091  private SyncReplicationPeerInfoProvider peerInfoProvider = new SyncReplicationPeerInfoProvider() {
092
093    @Override
094    public Optional<Pair<String, String>> getPeerIdAndRemoteWALDir(TableName table) {
095      return Optional.empty();
096    }
097
098    @Override
099    public boolean checkState(TableName table,
100      BiPredicate<SyncReplicationState, SyncReplicationState> checker) {
101      return false;
102    }
103
104  };
105
106  @Override
107  public final void init(WALFactory factory, Configuration conf, String providerId,
108    Abortable server) throws IOException {
109    if (!initialized.compareAndSet(false, true)) {
110      throw new IllegalStateException("WALProvider.init should only be called once.");
111    }
112    this.factory = factory;
113    this.conf = conf;
114    this.abortable = server;
115    doInit(factory, conf, providerId);
116  }
117
118  protected final void initWAL(WAL wal) throws IOException {
119    boolean succ = false;
120    try {
121      wal.init();
122      succ = true;
123    } finally {
124      if (!succ) {
125        safeClose(wal);
126      }
127    }
128  }
129
130  // Use a timestamp to make it identical. That means, after we transit the peer to DA/S and then
131  // back to A, the log prefix will be changed. This is used to simplify the implementation for
132  // replication source, where we do not need to consider that a terminated shipper could be added
133  // back.
134  private String getRemoteWALPrefix(String peerId) {
135    return factory.factoryId + "-" + EnvironmentEdgeManager.currentTime() + "-" + peerId;
136  }
137
138  private WAL getRemoteWAL(RegionInfo region, String peerId, String remoteWALDir)
139    throws IOException {
140    Optional<WAL> opt = peerId2WAL.get(peerId);
141    if (opt != null) {
142      return opt.orElse(null);
143    }
144    Lock lock = createLock.acquireLock(peerId);
145    try {
146      opt = peerId2WAL.get(peerId);
147      if (opt != null) {
148        return opt.orElse(null);
149      }
150      WAL wal = createRemoteWAL(region, ReplicationUtils.getRemoteWALFileSystem(conf, remoteWALDir),
151        ReplicationUtils.getPeerRemoteWALDir(remoteWALDir, peerId), getRemoteWALPrefix(peerId),
152        ReplicationUtils.SYNC_WAL_SUFFIX);
153      initWAL(wal);
154      peerId2WAL.put(peerId, Optional.of(wal));
155      return wal;
156    } finally {
157      lock.unlock();
158    }
159  }
160
161  @Override
162  public final WAL getWAL(RegionInfo region) throws IOException {
163    if (region == null) {
164      return getWAL0(null);
165    }
166    // deal with sync replication
167    Optional<Pair<String, String>> peerIdAndRemoteWALDir =
168      peerInfoProvider.getPeerIdAndRemoteWALDir(region.getTable());
169    if (peerIdAndRemoteWALDir.isPresent()) {
170      Pair<String, String> pair = peerIdAndRemoteWALDir.get();
171      WAL wal = getRemoteWAL(region, pair.getFirst(), pair.getSecond());
172      if (wal != null) {
173        return wal;
174      }
175    }
176    // fallback to normal WALProvider logic
177    return getWAL0(region);
178  }
179
180  @Override
181  public final List<WAL> getWALs() {
182    return Streams
183      .concat(peerId2WAL.values().stream().filter(Optional::isPresent).map(Optional::get),
184        getWALs0().stream())
185      .collect(Collectors.toList());
186  }
187
188  @Override
189  public PeerActionListener getPeerActionListener() {
190    return this;
191  }
192
193  @Override
194  public void peerSyncReplicationStateChange(String peerId, SyncReplicationState from,
195    SyncReplicationState to, int stage) {
196    if (from == SyncReplicationState.ACTIVE) {
197      if (stage == 0) {
198        Lock lock = createLock.acquireLock(peerId);
199        try {
200          Optional<WAL> opt = peerId2WAL.get(peerId);
201          if (opt != null) {
202            opt.ifPresent(w -> w.skipRemoteWAL(to == SyncReplicationState.STANDBY));
203          } else {
204            // add a place holder to tell the getWAL caller do not use the remote WAL any more.
205            peerId2WAL.put(peerId, Optional.empty());
206          }
207        } finally {
208          lock.unlock();
209        }
210      } else if (stage == 1) {
211        peerId2WAL.remove(peerId).ifPresent(AbstractWALProvider::safeClose);
212      }
213    }
214  }
215
216  @Override
217  public void setSyncReplicationPeerInfoProvider(SyncReplicationPeerInfoProvider provider) {
218    this.peerInfoProvider = provider;
219  }
220
221  private static void safeClose(WAL wal) {
222    if (wal != null) {
223      try {
224        wal.close();
225      } catch (IOException e) {
226        LOG.error("Close WAL failed", e);
227      }
228    }
229  }
230
231  @Override
232  public void addWALActionsListener(WALActionsListener listener) {
233    listeners.add(listener);
234  }
235
236  private void cleanup(IOExceptionConsumer<WAL> cleanupWAL, IOExceptionRunnable finalCleanup)
237    throws IOException {
238    MultipleIOException.Builder builder = new MultipleIOException.Builder();
239    for (Optional<WAL> wal : peerId2WAL.values()) {
240      if (wal.isPresent()) {
241        try {
242          cleanupWAL.accept(wal.get());
243        } catch (IOException e) {
244          LOG.error("cleanup WAL failed", e);
245          builder.add(e);
246        }
247      }
248    }
249    try {
250      finalCleanup.run();
251    } catch (IOException e) {
252      LOG.error("cleanup WAL failed", e);
253      builder.add(e);
254    }
255    if (!builder.isEmpty()) {
256      throw builder.build();
257    }
258  }
259
260  @Override
261  public final void shutdown() throws IOException {
262    cleanup(WAL::shutdown, this::shutdown0);
263  }
264
265  @Override
266  public final void close() throws IOException {
267    cleanup(WAL::close, this::close0);
268  }
269
270  private Stream<AbstractFSWAL<?>> remoteWALStream() {
271    return peerId2WAL.values().stream().filter(Optional::isPresent).map(Optional::get)
272      .filter(w -> w instanceof AbstractFSWAL).map(w -> (AbstractFSWAL<?>) w);
273  }
274
275  @Override
276  public final long getNumLogFiles() {
277    return remoteWALStream().mapToLong(AbstractFSWAL::getNumLogFiles).sum() + getNumLogFiles0();
278  }
279
280  @Override
281  public final long getLogFileSize() {
282    return remoteWALStream().mapToLong(AbstractFSWAL::getLogFileSize).sum() + getLogFileSize0();
283  }
284
285  private static final Pattern LOG_PREFIX_PATTERN = Pattern.compile(".*-\\d+-(.+)");
286
287  /**
288   * <p>
289   * Returns the peer id if the wal file name is in the special group for a sync replication peer.
290   * </p>
291   * <p>
292   * The prefix format is &lt;factoryId&gt;-&lt;ts&gt;-&lt;peerId&gt;.
293   * </p>
294   */
295  public static Optional<String> getSyncReplicationPeerIdFromWALName(String name) {
296    if (!name.endsWith(ReplicationUtils.SYNC_WAL_SUFFIX)) {
297      // fast path to return earlier if the name is not for a sync replication peer.
298      return Optional.empty();
299    }
300    String logPrefix = AbstractFSWALProvider.getWALPrefixFromWALName(name);
301    Matcher matcher = LOG_PREFIX_PATTERN.matcher(logPrefix);
302    if (matcher.matches()) {
303      return Optional.of(matcher.group(1));
304    } else {
305      return Optional.empty();
306    }
307  }
