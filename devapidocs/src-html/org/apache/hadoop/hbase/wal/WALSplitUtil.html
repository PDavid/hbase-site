080  /**
081   * Completes the work done by splitLogFile by archiving logs
082   * <p>
083   * It is invoked by SplitLogManager once it knows that one of the SplitLogWorkers have completed
084   * the splitLogFile() part. If the master crashes then this function might get called multiple
085   * times.
086   * <p>
087   * @param logfile
088   * @param conf
089   * @throws IOException
090   */
091  public static void finishSplitLogFile(String logfile, Configuration conf) throws IOException {
092    Path walDir = FSUtils.getWALRootDir(conf);
093    Path oldLogDir = new Path(walDir, HConstants.HREGION_OLDLOGDIR_NAME);
094    Path walPath;
095    if (FSUtils.isStartingWithPath(walDir, logfile)) {
096      walPath = new Path(logfile);
097    } else {
098      walPath = new Path(walDir, logfile);
099    }
100    finishSplitLogFile(walDir, oldLogDir, walPath, conf);
101  }
102
103  static void finishSplitLogFile(Path walDir, Path oldWALDir, Path walPath,
104      Configuration conf) throws IOException {
105    List<Path> processedLogs = new ArrayList<>();
106    List<Path> corruptedLogs = new ArrayList<>();
107    FileSystem walFS = walDir.getFileSystem(conf);
108    if (ZKSplitLog.isCorrupted(walDir, walPath.getName(), walFS)) {
109      corruptedLogs.add(walPath);
110    } else {
111      processedLogs.add(walPath);
112    }
113    archiveWALs(corruptedLogs, processedLogs, oldWALDir, walFS, conf);
114    Path stagingDir = ZKSplitLog.getSplitLogDir(walDir, walPath.getName());
115    walFS.delete(stagingDir, true);
116  }

118  /**
119   * Moves processed logs to a oldLogDir after successful processing Moves corrupted logs (any log
120   * that couldn't be successfully parsed to corruptDir (.corrupt) for later investigation
121   */
122  private static void archiveWALs(final List<Path> corruptedWALs, final List<Path> processedWALs,
123      final Path oldWALDir, final FileSystem walFS, final Configuration conf) throws IOException {
124    final Path corruptDir = new Path(FSUtils.getWALRootDir(conf), HConstants.CORRUPT_DIR_NAME);
125    if (conf.get("hbase.regionserver.hlog.splitlog.corrupt.dir") != null) {
126      LOG.warn("hbase.regionserver.hlog.splitlog.corrupt.dir is deprecated. Default to {}",
127        corruptDir);
128    }
129    if (!walFS.mkdirs(corruptDir)) {
130      LOG.info("Unable to mkdir {}", corruptDir);
131    }
132    walFS.mkdirs(oldWALDir);
133
134    // this method can get restarted or called multiple times for archiving
135    // the same log files.
136    for (Path corruptedWAL : corruptedWALs) {
137      Path p = new Path(corruptDir, corruptedWAL.getName());
138      if (walFS.exists(corruptedWAL)) {
139        if (!walFS.rename(corruptedWAL, p)) {
140          LOG.warn("Unable to move corrupted log {} to {}", corruptedWAL, p);
141        } else {
142          LOG.warn("Moved corrupted log {} to {}", corruptedWAL, p);
143        }
144      }
145    }
146
147    for (Path p : processedWALs) {
148      Path newPath = AbstractFSWAL.getWALArchivePath(oldWALDir, p);
149      if (walFS.exists(p)) {
150        if (!FSUtils.renameAndSetModifyTime(walFS, p, newPath)) {
151          LOG.warn("Unable to move {} to {}", p, newPath);
152        } else {
153          LOG.info("Archived processed log {} to {}", p, newPath);
154        }
155      }
156    }
157  }

159  /**
160   * Path to a file under RECOVERED_EDITS_DIR directory of the region found in <code>logEntry</code>
161   * named for the sequenceid in the passed <code>logEntry</code>: e.g.
162   * /hbase/some_table/2323432434/recovered.edits/2332. This method also ensures existence of
163   * RECOVERED_EDITS_DIR under the region creating it if necessary.
164   * @param walEntry walEntry to recover
165   * @param fileNameBeingSplit the file being split currently. Used to generate tmp file name.
166   * @param tmpDirName of the directory used to sideline old recovered edits file
167   * @param conf configuration
168   * @return Path to file into which to dump split log edits.
169   * @throws IOException
170   */
171  @SuppressWarnings("deprecation")
172  @VisibleForTesting
173  static Path getRegionSplitEditsPath(final WAL.Entry walEntry, String fileNameBeingSplit,
174      String tmpDirName, Configuration conf) throws IOException {
175    FileSystem walFS = FSUtils.getWALFileSystem(conf);
176    Path tableDir = FSUtils.getWALTableDir(conf, walEntry.getKey().getTableName());
177    String encodedRegionName = Bytes.toString(walEntry.getKey().getEncodedRegionName());
178    Path regionDir = HRegion.getRegionDir(tableDir, encodedRegionName);
179    Path dir = getRegionDirRecoveredEditsDir(regionDir);
180
181    if (walFS.exists(dir) && walFS.isFile(dir)) {
182      Path tmp = new Path(tmpDirName);
183      if (!walFS.exists(tmp)) {
184        walFS.mkdirs(tmp);
185      }
186      tmp = new Path(tmp, HConstants.RECOVERED_EDITS_DIR + "_" + encodedRegionName);
187      LOG.warn("Found existing old file: {}. It could be some "
188          + "leftover of an old installation. It should be a folder instead. "
189          + "So moving it to {}",
190        dir, tmp);
191      if (!walFS.rename(dir, tmp)) {
192        LOG.warn("Failed to sideline old file {}", dir);
193      }
194    }
195
196    if (!walFS.exists(dir) && !walFS.mkdirs(dir)) {
197      LOG.warn("mkdir failed on {}", dir);
198    }
199    // Append fileBeingSplit to prevent name conflict since we may have duplicate wal entries now.
200    // Append file name ends with RECOVERED_LOG_TMPFILE_SUFFIX to ensure
201    // region's replayRecoveredEdits will not delete it
202    String fileName = formatRecoveredEditsFileName(walEntry.getKey().getSequenceId());
203    fileName = getTmpRecoveredEditsFileName(fileName + "-" + fileNameBeingSplit);
204    return new Path(dir, fileName);
205  }
206
207  private static String getTmpRecoveredEditsFileName(String fileName) {
208    return fileName + RECOVERED_LOG_TMPFILE_SUFFIX;
209  }

235  /**
236   * Check whether there is recovered.edits in the region dir
237   * @param conf conf
238   * @param regionInfo the region to check
239   * @throws IOException IOException
240   * @return true if recovered.edits exist in the region dir
241   */
242  public static boolean hasRecoveredEdits(final Configuration conf, final RegionInfo regionInfo)
243      throws IOException {
244    // No recovered.edits for non default replica regions
245    if (regionInfo.getReplicaId() != RegionInfo.DEFAULT_REPLICA_ID) {
246      return false;
247    }
248    // Only default replica region can reach here, so we can use regioninfo
249    // directly without converting it to default replica's regioninfo.
250    Path regionDir =
251        FSUtils.getWALRegionDir(conf, regionInfo.getTable(), regionInfo.getEncodedName());
252    NavigableSet<Path> files = getSplitEditFilesSorted(FSUtils.getWALFileSystem(conf), regionDir);
253    return files != null && !files.isEmpty();
254  }

256  /**
257   * Returns sorted set of edit files made by splitter, excluding files with '.temp' suffix.
258   * @param walFS WAL FileSystem used to retrieving split edits files.
259   * @param regionDir WAL region dir to look for recovered edits files under.
260   * @return Files in passed <code>regionDir</code> as a sorted set.
261   * @throws IOException
262   */
263  public static NavigableSet<Path> getSplitEditFilesSorted(final FileSystem walFS,
264      final Path regionDir) throws IOException {
265    NavigableSet<Path> filesSorted = new TreeSet<>();
266    Path editsdir = getRegionDirRecoveredEditsDir(regionDir);
267    if (!walFS.exists(editsdir)) {
268      return filesSorted;
269    }
270    FileStatus[] files = FSUtils.listStatus(walFS, editsdir, new PathFilter() {
271      @Override
272      public boolean accept(Path p) {
273        boolean result = false;
274        try {
275          // Return files and only files that match the editfile names pattern.
276          // There can be other files in this directory other than edit files.
277          // In particular, on error, we'll move aside the bad edit file giving
278          // it a timestamp suffix. See moveAsideBadEditsFile.
279          Matcher m = EDITFILES_NAME_PATTERN.matcher(p.getName());
280          result = walFS.isFile(p) && m.matches();
281          // Skip the file whose name ends with RECOVERED_LOG_TMPFILE_SUFFIX,
282          // because it means splitwal thread is writting this file.
283          if (p.getName().endsWith(RECOVERED_LOG_TMPFILE_SUFFIX)) {
284            result = false;
285          }
286          // Skip SeqId Files
287          if (isSequenceIdFile(p)) {
288            result = false;
289          }
290        } catch (IOException e) {
291          LOG.warn("Failed isFile check on {}", p, e);
292        }
293        return result;
294      }
295    });
296    if (ArrayUtils.isNotEmpty(files)) {
297      Arrays.asList(files).forEach(status -> filesSorted.add(status.getPath()));
298    }
299    return filesSorted;
300  }

302  /**
303   * Move aside a bad edits file.
304   * @param walFS WAL FileSystem used to rename bad edits file.
305   * @param edits Edits file to move aside.
306   * @return The name of the moved aside file.
307   * @throws IOException
308   */
309  public static Path moveAsideBadEditsFile(final FileSystem walFS, final Path edits)
310      throws IOException {
311    Path moveAsideName =
312        new Path(edits.getParent(), edits.getName() + "." + System.currentTimeMillis());
313    if (!walFS.rename(edits, moveAsideName)) {
314      LOG.warn("Rename failed from {} to {}", edits, moveAsideName);
315    }
316    return moveAsideName;
317  }

320   * Is the given file a region open sequence id file.
321   */
322  @VisibleForTesting

324    return file.getName().endsWith(SEQUENCE_ID_FILE_SUFFIX)

328  private static FileStatus[] getSequenceIdFiles(FileSystem walFS, Path regionDir)
329      throws IOException {
330    // TODO: Why are we using a method in here as part of our normal region open where
331    // there is no splitting involved? Fix. St.Ack 01/20/2017.
332    Path editsDir = getRegionDirRecoveredEditsDir(regionDir);
333    try {
334      FileStatus[] files = walFS.listStatus(editsDir, WALSplitUtil::isSequenceIdFile);
335      return files != null ? files : new FileStatus[0];
336    } catch (FileNotFoundException e) {
337      return new FileStatus[0];
338    }
339  }

341  private static long getMaxSequenceId(FileStatus[] files) {
342    long maxSeqId = -1L;
343    for (FileStatus file : files) {
344      String fileName = file.getPath().getName();
345      try {
346        maxSeqId = Math.max(maxSeqId, Long
347            .parseLong(fileName.substring(0, fileName.length() - SEQUENCE_ID_FILE_SUFFIX_LENGTH)));
348      } catch (NumberFormatException ex) {
349        LOG.warn("Invalid SeqId File Name={}", fileName);
350      }
351    }
352    return maxSeqId;
353  }

362  /**
363   * Create a file with name as region's max sequence id
364   */
365  public static void writeRegionSequenceIdFile(FileSystem walFS, Path regionDir, long newMaxSeqId)
366      throws IOException {
367    FileStatus[] files = getSequenceIdFiles(walFS, regionDir);
368    long maxSeqId = getMaxSequenceId(files);
369    if (maxSeqId > newMaxSeqId) {
370      throw new IOException("The new max sequence id " + newMaxSeqId
371          + " is less than the old max sequence id " + maxSeqId);
372    }
373    // write a new seqId file
374    Path newSeqIdFile =
375        new Path(getRegionDirRecoveredEditsDir(regionDir), newMaxSeqId + SEQUENCE_ID_FILE_SUFFIX);
376    if (newMaxSeqId != maxSeqId) {
377      try {
378        if (!walFS.createNewFile(newSeqIdFile) && !walFS.exists(newSeqIdFile)) {
379          throw new IOException("Failed to create SeqId file:" + newSeqIdFile);
380        }
381        LOG.debug("Wrote file={}, newMaxSeqId={}, maxSeqId={}", newSeqIdFile, newMaxSeqId,
382          maxSeqId);
383      } catch (FileAlreadyExistsException ignored) {
384        // latest hdfs throws this exception. it's all right if newSeqIdFile already exists
385      }
386    }
387    // remove old ones
388    for (FileStatus status : files) {
389      if (!newSeqIdFile.equals(status.getPath())) {
390        walFS.delete(status.getPath(), false);
391      }
392    }
393  }

395  /** A struct used by getMutationsFromWALEntry */
396  public static class MutationReplay implements Comparable<MutationReplay> {
397    public MutationReplay(ClientProtos.MutationProto.MutationType type, Mutation mutation,
398        long nonceGroup, long nonce) {
399      this.type = type;
400      this.mutation = mutation;
401      if (this.mutation.getDurability() != Durability.SKIP_WAL) {
402        // using ASYNC_WAL for relay
403        this.mutation.setDurability(Durability.ASYNC_WAL);
404      }
405      this.nonceGroup = nonceGroup;
406      this.nonce = nonce;
407    }
408
409    private final ClientProtos.MutationProto.MutationType type;
410    public final Mutation mutation;
411    public final long nonceGroup;
412    public final long nonce;
413
414    @Override
415    public int compareTo(final MutationReplay d) {
416      return this.mutation.compareTo(d.mutation);
417    }
418
419    @Override
420    public boolean equals(Object obj) {
421      if (!(obj instanceof MutationReplay)) {
422        return false;
423      } else {
424        return this.compareTo((MutationReplay) obj) == 0;
425      }
426    }
427
428    @Override
429    public int hashCode() {
430      return this.mutation.hashCode();
431    }
432
433    public ClientProtos.MutationProto.MutationType getType() {
434      return type;
435    }
436  }

438  /**
439   * This function is used to construct mutations from a WALEntry. It also reconstructs WALKey &amp;
440   * WALEdit from the passed in WALEntry
441   * @param entry
442   * @param cells
443   * @param logEntry pair of WALKey and WALEdit instance stores WALKey and WALEdit instances
444   *          extracted from the passed in WALEntry.
445   * @return list of Pair&lt;MutationType, Mutation&gt; to be replayed
446   * @throws IOException
447   */
448  public static List<MutationReplay> getMutationsFromWALEntry(AdminProtos.WALEntry entry,
449      CellScanner cells, Pair<WALKey, WALEdit> logEntry, Durability durability) throws IOException {
450    if (entry == null) {
451      // return an empty array
452      return Collections.emptyList();
453    }
454
455    long replaySeqId =
456        (entry.getKey().hasOrigSequenceNumber()) ? entry.getKey().getOrigSequenceNumber()
457            : entry.getKey().getLogSequenceNumber();
458    int count = entry.getAssociatedCellCount();
459    List<MutationReplay> mutations = new ArrayList<>();
460    Cell previousCell = null;
461    Mutation m = null;
462    WALKeyImpl key = null;
463    WALEdit val = null;
464    if (logEntry != null) {
465      val = new WALEdit();
466    }
467
468    for (int i = 0; i < count; i++) {
469      // Throw index out of bounds if our cell count is off
470      if (!cells.advance()) {
471        throw new ArrayIndexOutOfBoundsException("Expected=" + count + ", index=" + i);
472      }
473      Cell cell = cells.current();
474      if (val != null) val.add(cell);
475
476      boolean isNewRowOrType =
477          previousCell == null || previousCell.getTypeByte() != cell.getTypeByte()
478              || !CellUtil.matchingRows(previousCell, cell);
479      if (isNewRowOrType) {
480        // Create new mutation
481        if (CellUtil.isDelete(cell)) {
482          m = new Delete(cell.getRowArray(), cell.getRowOffset(), cell.getRowLength());
483          // Deletes don't have nonces.
484          mutations.add(new MutationReplay(ClientProtos.MutationProto.MutationType.DELETE, m,
485              HConstants.NO_NONCE, HConstants.NO_NONCE));
486        } else {
487          m = new Put(cell.getRowArray(), cell.getRowOffset(), cell.getRowLength());
488          // Puts might come from increment or append, thus we need nonces.
489          long nonceGroup =
490              entry.getKey().hasNonceGroup() ? entry.getKey().getNonceGroup() : HConstants.NO_NONCE;
491          long nonce = entry.getKey().hasNonce() ? entry.getKey().getNonce() : HConstants.NO_NONCE;
492          mutations.add(
493            new MutationReplay(ClientProtos.MutationProto.MutationType.PUT, m, nonceGroup, nonce));
494        }
495      }
496      if (CellUtil.isDelete(cell)) {
497        ((Delete) m).add(cell);
498      } else {
499        ((Put) m).add(cell);
500      }
501      m.setDurability(durability);
502      previousCell = cell;
503    }
504
505    // reconstruct WALKey
506    if (logEntry != null) {
507      org.apache.hadoop.hbase.shaded.protobuf.generated.WALProtos.WALKey walKeyProto =
508          entry.getKey();
509      List<UUID> clusterIds = new ArrayList<>(walKeyProto.getClusterIdsCount());
510      for (HBaseProtos.UUID uuid : entry.getKey().getClusterIdsList()) {
511        clusterIds.add(new UUID(uuid.getMostSigBits(), uuid.getLeastSigBits()));
512      }
513      key = new WALKeyImpl(walKeyProto.getEncodedRegionName().toByteArray(),
514          TableName.valueOf(walKeyProto.getTableName().toByteArray()), replaySeqId,
515          walKeyProto.getWriteTime(), clusterIds, walKeyProto.getNonceGroup(),
516          walKeyProto.getNonce(), null);
517      logEntry.setFirst(key);
518      logEntry.setSecond(val);
519    }
520
521    return mutations;
522  }
