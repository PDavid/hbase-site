114  @VisibleForTesting
115  WALSplitter(final WALFactory factory, Configuration conf, Path walDir, FileSystem walFS,
116      LastSequenceId idChecker, SplitLogWorkerCoordination splitLogWorkerCoordination) {
117    this.conf = HBaseConfiguration.create(conf);
118    String codecClassName =
119        conf.get(WALCellCodec.WAL_CELL_CODEC_CLASS_KEY, WALCellCodec.class.getName());
120    this.conf.set(HConstants.RPC_CODEC_CONF_KEY, codecClassName);
121    this.walDir = walDir;
122    this.walFS = walFS;
123    this.sequenceIdChecker = idChecker;
124    this.splitLogWorkerCoordination = splitLogWorkerCoordination;
125
126    this.walFactory = factory;
127    PipelineController controller = new PipelineController();
128
129    this.splitWriterCreationBounded = conf.getBoolean(SPLIT_WRITER_CREATION_BOUNDED, false);
130
131    entryBuffers = new EntryBuffers(controller,
132        this.conf.getLong("hbase.regionserver.hlog.splitlog.buffersize", 128 * 1024 * 1024),
133        splitWriterCreationBounded);
134
135    int numWriterThreads = this.conf.getInt("hbase.regionserver.hlog.splitlog.writer.threads", 3);
136    if (splitWriterCreationBounded) {
137      outputSink =
138          new BoundedLogWriterCreationOutputSink(this, controller, entryBuffers, numWriterThreads);
139    } else {
140      outputSink =
141          new LogRecoveredEditsOutputSink(this, controller, entryBuffers, numWriterThreads);
142    }
143  }

158   * Splits a WAL file into region's recovered-edits directory.
159   * This is the main entry point for distributed log splitting from SplitLogWorker.
160   * <p>
161   * If the log file has N regions then N recovered.edits files will be produced.
162   * <p>
163   * @return false if it is interrupted by the progress-able.
164   */
165  public static boolean splitLogFile(Path walDir, FileStatus logfile, FileSystem walFS,
166      Configuration conf, CancelableProgressable reporter, LastSequenceId idChecker,
167      SplitLogWorkerCoordination splitLogWorkerCoordination, final WALFactory factory)
168      throws IOException {
169    WALSplitter s = new WALSplitter(factory, conf, walDir, walFS, idChecker,
170        splitLogWorkerCoordination);
171    return s.splitLogFile(logfile, reporter);
172  }

174  // A wrapper to split one log folder using the method used by distributed
175  // log splitting. Used by tools and unit tests. It should be package private.
176  // It is public only because TestWALObserver is in a different package,
177  // which uses this method to do log splitting.
178  @VisibleForTesting
179  public static List<Path> split(Path rootDir, Path logDir, Path oldLogDir,
180      FileSystem walFS, Configuration conf, final WALFactory factory) throws IOException {
181    final FileStatus[] logfiles = SplitLogManager.getFileList(conf,
182        Collections.singletonList(logDir), null);
183    List<Path> splits = new ArrayList<>();
184    if (ArrayUtils.isNotEmpty(logfiles)) {
185      for (FileStatus logfile: logfiles) {
186        WALSplitter s = new WALSplitter(factory, conf, rootDir, walFS, null, null);
187        if (s.splitLogFile(logfile, null)) {
188          finishSplitLogFile(rootDir, oldLogDir, logfile.getPath(), conf);
189          if (s.outputSink.splits != null) {
190            splits.addAll(s.outputSink.splits);
191          }
192        }
193      }
194    }
195    if (!walFS.delete(logDir, true)) {
196      throw new IOException("Unable to delete src dir: " + logDir);
197    }
198    return splits;
199  }

201  /**
202   * log splitting implementation, splits one log file.
203   * @param logfile should be an actual log file.
204   */
205  @VisibleForTesting
206  boolean splitLogFile(FileStatus logfile, CancelableProgressable reporter) throws IOException {
207    Preconditions.checkState(status == null);
208    Preconditions.checkArgument(logfile.isFile(),
209        "passed in file status is for something other than a regular file.");
210    boolean isCorrupted = false;
211    boolean skipErrors = conf.getBoolean("hbase.hlog.split.skip.errors",
212      SPLIT_SKIP_ERRORS_DEFAULT);
213    int interval = conf.getInt("hbase.splitlog.report.interval.loglines", 1024);
214    Path logPath = logfile.getPath();
215    boolean outputSinkStarted = false;
216    boolean progress_failed = false;
217    int editsCount = 0;
218    int editsSkipped = 0;
219
220    status = TaskMonitor.get().createStatus(
221          "Splitting log file " + logfile.getPath() + "into a temporary staging area.");
222    Reader logFileReader = null;
223    this.fileBeingSplit = logfile;
224    try {
225      long logLength = logfile.getLen();
226      LOG.info("Splitting WAL={}, length={}", logPath, logLength);
227      status.setStatus("Opening log file");
228      if (reporter != null && !reporter.progress()) {
229        progress_failed = true;
230        return false;
231      }
232      logFileReader = getReader(logfile, skipErrors, reporter);
233      if (logFileReader == null) {
234        LOG.warn("Nothing to split in WAL={}", logPath);
235        return true;
236      }
237      int numOpenedFilesBeforeReporting = conf.getInt("hbase.splitlog.report.openedfiles", 3);
238      int numOpenedFilesLastCheck = 0;
239      outputSink.setReporter(reporter);
240      outputSink.startWriterThreads();
241      outputSinkStarted = true;
242      Entry entry;
243      Long lastFlushedSequenceId = -1L;
244      while ((entry = getNextLogLine(logFileReader, logPath, skipErrors)) != null) {
245        byte[] region = entry.getKey().getEncodedRegionName();
246        String encodedRegionNameAsStr = Bytes.toString(region);
247        lastFlushedSequenceId = lastFlushedSequenceIds.get(encodedRegionNameAsStr);
248        if (lastFlushedSequenceId == null) {
249          if (sequenceIdChecker != null) {
250            RegionStoreSequenceIds ids = sequenceIdChecker.getLastSequenceId(region);
251            Map<byte[], Long> maxSeqIdInStores = new TreeMap<>(Bytes.BYTES_COMPARATOR);
252            for (StoreSequenceId storeSeqId : ids.getStoreSequenceIdList()) {
253              maxSeqIdInStores.put(storeSeqId.getFamilyName().toByteArray(),
254                storeSeqId.getSequenceId());
255            }
256            regionMaxSeqIdInStores.put(encodedRegionNameAsStr, maxSeqIdInStores);
257            lastFlushedSequenceId = ids.getLastFlushedSequenceId();
258            if (LOG.isDebugEnabled()) {
259              LOG.debug("DLS Last flushed sequenceid for " + encodedRegionNameAsStr + ": " +
260                  TextFormat.shortDebugString(ids));
261            }
262          }
263          if (lastFlushedSequenceId == null) {
264            lastFlushedSequenceId = -1L;
265          }
266          lastFlushedSequenceIds.put(encodedRegionNameAsStr, lastFlushedSequenceId);
267        }
268        if (lastFlushedSequenceId >= entry.getKey().getSequenceId()) {
269          editsSkipped++;
270          continue;
271        }
272        // Don't send Compaction/Close/Open region events to recovered edit type sinks.
273        if (entry.getEdit().isMetaEdit() && !outputSink.keepRegionEvent(entry)) {
274          editsSkipped++;
275          continue;
276        }
277        entryBuffers.appendEntry(entry);
278        editsCount++;
279        int moreWritersFromLastCheck = this.getNumOpenWriters() - numOpenedFilesLastCheck;
280        // If sufficient edits have passed, check if we should report progress.
281        if (editsCount % interval == 0
282            || moreWritersFromLastCheck > numOpenedFilesBeforeReporting) {
283          numOpenedFilesLastCheck = this.getNumOpenWriters();
284          String countsStr = (editsCount - (editsSkipped + outputSink.getSkippedEdits()))
285              + " edits, skipped " + editsSkipped + " edits.";
286          status.setStatus("Split " + countsStr);
287          if (reporter != null && !reporter.progress()) {
288            progress_failed = true;
289            return false;
290          }
291        }
292      }
293    } catch (InterruptedException ie) {
294      IOException iie = new InterruptedIOException();
295      iie.initCause(ie);
296      throw iie;
297    } catch (CorruptedLogFileException e) {
298      LOG.warn("Could not parse, corrupted WAL={}", logPath, e);
299      if (splitLogWorkerCoordination != null) {
300        // Some tests pass in a csm of null.
301        splitLogWorkerCoordination.markCorrupted(walDir, logfile.getPath().getName(), walFS);
302      } else {
303        // for tests only
304        ZKSplitLog.markCorrupted(walDir, logfile.getPath().getName(), walFS);
305      }
306      isCorrupted = true;
307    } catch (IOException e) {
308      e = e instanceof RemoteException ? ((RemoteException) e).unwrapRemoteException() : e;
309      throw e;
310    } finally {
311      LOG.debug("Finishing writing output logs and closing down");
312      try {
313        if (null != logFileReader) {
314          logFileReader.close();
315        }
316      } catch (IOException exception) {
317        LOG.warn("Could not close WAL reader", exception);
318      }
319      try {
320        if (outputSinkStarted) {
321          // Set progress_failed to true as the immediate following statement will reset its value
322          // when finishWritingAndClose() throws exception, progress_failed has the right value
323          progress_failed = true;
324          progress_failed = outputSink.finishWritingAndClose() == null;
325        }
326      } finally {
327        String msg =
328            "Processed " + editsCount + " edits across " + outputSink.getNumberOfRecoveredRegions()
329                + " regions; edits skipped=" + editsSkipped + "; log file=" + logPath +
330                ", length=" + logfile.getLen() + // See if length got updated post lease recovery
331                ", corrupted=" + isCorrupted + ", progress failed=" + progress_failed;
332        LOG.info(msg);
333        status.markComplete(msg);
334      }
335    }
336    return !progress_failed;
337  }

339  /**
340   * Create a new {@link Reader} for reading logs to split.
341   */
342  protected Reader getReader(FileStatus file, boolean skipErrors, CancelableProgressable reporter)
343      throws IOException, CorruptedLogFileException {
344    Path path = file.getPath();
345    long length = file.getLen();
346    Reader in;
347
348    // Check for possibly empty file. With appends, currently Hadoop reports a
349    // zero length even if the file has been sync'd. Revisit if HDFS-376 or
350    // HDFS-878 is committed.
351    if (length <= 0) {
352      LOG.warn("File {} might be still open, length is 0", path);
353    }
354
355    try {
356      FSUtils.getInstance(walFS, conf).recoverFileLease(walFS, path, conf, reporter);
357      try {
358        in = getReader(path, reporter);
359      } catch (EOFException e) {
360        if (length <= 0) {
361          // TODO should we ignore an empty, not-last log file if skip.errors
362          // is false? Either way, the caller should decide what to do. E.g.
363          // ignore if this is the last log in sequence.
364          // TODO is this scenario still possible if the log has been
365          // recovered (i.e. closed)
366          LOG.warn("Could not open {} for reading. File is empty", path, e);
367        }
368        // EOFException being ignored
369        return null;
370      }
371    } catch (IOException e) {
372      if (e instanceof FileNotFoundException) {
373        // A wal file may not exist anymore. Nothing can be recovered so move on
374        LOG.warn("File {} does not exist anymore", path, e);
375        return null;
376      }
377      if (!skipErrors || e instanceof InterruptedIOException) {
378        throw e; // Don't mark the file corrupted if interrupted, or not skipErrors
379      }
380      throw new CorruptedLogFileException("skipErrors=true Could not open wal "
381        + path + " ignoring", e);
382    }
383    return in;
384  }
385
386  static private Entry getNextLogLine(Reader in, Path path, boolean skipErrors)
387      throws CorruptedLogFileException, IOException {
388    try {
389      return in.next();
390    } catch (EOFException eof) {
391      // truncated files are expected if a RS crashes (see HBASE-2643)
392      LOG.info("EOF from wal {}. Continuing.", path);
393      return null;
394    } catch (IOException e) {
395      // If the IOE resulted from bad file format,
396      // then this problem is idempotent and retrying won't help
397      if (e.getCause() != null && (e.getCause() instanceof ParseException
398          || e.getCause() instanceof org.apache.hadoop.fs.ChecksumException)) {
399        LOG.warn("Parse exception from wal {}. Continuing", path, e);
400        return null;
401      }
402      if (!skipErrors) {
403        throw e;
404      }
405      throw new CorruptedLogFileException("skipErrors=true Ignoring exception"
406        + " while parsing wal " + path + ". Marking as corrupted", e);
407    }
408  }

437  /**
438   * Contains some methods to control WAL-entries producer / consumer interactions
439   */
440  public static class PipelineController {
441    // If an exception is thrown by one of the other threads, it will be
442    // stored here.
443    AtomicReference<Throwable> thrown = new AtomicReference<>();
444
445    // Wait/notify for when data has been produced by the writer thread,
446    // consumed by the reader thread, or an exception occurred
447    final Object dataAvailable = new Object();
448
449    void writerThreadError(Throwable t) {
450      thrown.compareAndSet(null, t);
451    }
452
453    /**
454     * Check for errors in the writer threads. If any is found, rethrow it.
455     */
456    void checkForErrors() throws IOException {
457      Throwable thrown = this.thrown.get();
458      if (thrown == null) {
459        return;
460      }
461      if (thrown instanceof IOException) {
462        throw new IOException(thrown);
463      } else {
464        throw new RuntimeException(thrown);
465      }
466    }
467  }

469  /**
470   * A buffer of some number of edits for a given region.
471   * This accumulates edits and also provides a memory optimization in order to
472   * share a single byte array instance for the table and region name.
473   * Also tracks memory usage of the accumulated edits.
474   */
475  public static class RegionEntryBuffer implements HeapSize {
476    long heapInBuffer = 0;
477    List<Entry> entryBuffer;
478    TableName tableName;
479    byte[] encodedRegionName;
480
481    RegionEntryBuffer(TableName tableName, byte[] region) {
482      this.tableName = tableName;
483      this.encodedRegionName = region;
484      this.entryBuffer = new ArrayList<>();
485    }
486
487    long appendEntry(Entry entry) {
488      internify(entry);
489      entryBuffer.add(entry);
490      long incrHeap = entry.getEdit().heapSize() +
491        ClassSize.align(2 * ClassSize.REFERENCE) + // WALKey pointers
492        0; // TODO linkedlist entry
493      heapInBuffer += incrHeap;
494      return incrHeap;
495    }
496
497    private void internify(Entry entry) {
498      WALKeyImpl k = entry.getKey();
499      k.internTableName(this.tableName);
500      k.internEncodedRegionName(this.encodedRegionName);
501    }
502
503    @Override
504    public long heapSize() {
505      return heapInBuffer;
506    }
507
508    public byte[] getEncodedRegionName() {
509      return encodedRegionName;
510    }
511
512    public List<Entry> getEntryBuffer() {
513      return entryBuffer;
514    }
515
516    public TableName getTableName() {
517      return tableName;
518    }
519  }

521  /**
522   * Class wraps the actual writer which writes data out and related statistics
523   */
524  public abstract static class SinkWriter {
525    /* Count of edits written to this path */
526    long editsWritten = 0;
527    /* Count of edits skipped to this path */
528    long editsSkipped = 0;
529    /* Number of nanos spent writing to this log */
530    long nanosSpent = 0;
531
532    void incrementEdits(int edits) {
533      editsWritten += edits;
534    }
535
536    void incrementSkippedEdits(int skipped) {
537      editsSkipped += skipped;
538    }
539
540    void incrementNanoTime(long nanos) {
541      nanosSpent += nanos;
542    }
543  }
