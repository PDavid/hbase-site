045/**
046 * Runs periodically to determine if the WAL should be rolled.
047 * <p/>
048 * NOTE: This class extends Thread rather than Chore because the sleep time can be interrupted when
049 * there is something to do, rather than the Chore sleep time which is invariant.
050 * <p/>
051 * The {@link #scheduleFlush(String, List)} is abstract here,
052 * as sometimes we hold a region without a region server but we still want to roll its WAL.
053 * <p/>
054 * TODO: change to a pool of threads
055 */
056@InterfaceAudience.Private
057public abstract class AbstractWALRoller<T extends Abortable> extends Thread
058  implements Closeable {
059  private static final Logger LOG = LoggerFactory.getLogger(AbstractWALRoller.class);
060
061  protected static final String WAL_ROLL_PERIOD_KEY = "hbase.regionserver.logroll.period";
062
063  protected final ConcurrentMap<WAL, RollController> wals = new ConcurrentHashMap<>();
064  protected final T abortable;
065  // Period to roll log.
066  private final long rollPeriod;
067  private final int threadWakeFrequency;
068  // The interval to check low replication on hlog's pipeline
069  private final long checkLowReplicationInterval;
070
071  private volatile boolean running = true;
072
073  public void addWAL(WAL wal) {
074    // check without lock first
075    if (wals.containsKey(wal)) {
076      return;
077    }
078    // this is to avoid race between addWAL and requestRollAll.
079    synchronized (this) {
080      if (wals.putIfAbsent(wal, new RollController(wal)) == null) {
081        wal.registerWALActionsListener(new WALActionsListener() {
082          @Override
083          public void logRollRequested(WALActionsListener.RollRequestReason reason) {
084            // TODO logs will contend with each other here, replace with e.g. DelayedQueue
085            synchronized (AbstractWALRoller.this) {
086              RollController controller = wals.computeIfAbsent(wal, rc -> new RollController(wal));
087              controller.requestRoll();
088              AbstractWALRoller.this.notifyAll();
089            }
090          }
091
092          @Override
093          public void postLogArchive(Path oldPath, Path newPath) throws IOException {
094            afterWALArchive(oldPath, newPath);
095          }
096        });
097      }
098    }
099  }
100
101  public void requestRollAll() {
102    synchronized (this) {
103      for (RollController controller : wals.values()) {
104        controller.requestRoll();
105      }
106      notifyAll();
107    }
108  }
109
110  protected AbstractWALRoller(String name, Configuration conf, T abortable) {
111    super(name);
112    this.abortable = abortable;
113    this.rollPeriod = conf.getLong(WAL_ROLL_PERIOD_KEY, 3600000);
114    this.threadWakeFrequency = conf.getInt(HConstants.THREAD_WAKE_FREQUENCY, 10 * 1000);
115    this.checkLowReplicationInterval =
116      conf.getLong("hbase.regionserver.hlog.check.lowreplication.interval", 30 * 1000);
117  }
118
119  /**
120   * we need to check low replication in period, see HBASE-18132
121   */
122  private void checkLowReplication(long now) {
123    try {
124      for (Entry<WAL, RollController> entry : wals.entrySet()) {
125        WAL wal = entry.getKey();
126        boolean needRollAlready = entry.getValue().needsRoll(now);
127        if (needRollAlready || !(wal instanceof AbstractFSWAL)) {
128          continue;
129        }
130        ((AbstractFSWAL<?>) wal).checkLogLowReplication(checkLowReplicationInterval);
131      }
132    } catch (Throwable e) {
133      LOG.warn("Failed checking low replication", e);
134    }
135  }
136
137  private void abort(String reason, Throwable cause) {
138    // close all WALs before calling abort on RS.
139    // This is because AsyncFSWAL replies on us for rolling a new writer to make progress, and if we
140    // failed, AsyncFSWAL may be stuck, so we need to close it to let the upper layer know that it
141    // is already broken.
142    for (WAL wal : wals.keySet()) {
143      // shutdown rather than close here since we are going to abort the RS and the wals need to be
144      // split when recovery
145      try {
146        wal.shutdown();
147      } catch (IOException e) {
148        LOG.warn("Failed to shutdown wal", e);
149      }
150    }
151    abortable.abort(reason, cause);
152  }
153
154  @Override
155  public void run() {
156    while (running) {
157      long now = EnvironmentEdgeManager.currentTime();
158      checkLowReplication(now);
159      synchronized (this) {
160        if (wals.values().stream().noneMatch(rc -> rc.needsRoll(now))) {
161          try {
162            wait(this.threadWakeFrequency);
163          } catch (InterruptedException e) {
164            // restore the interrupt state
165            Thread.currentThread().interrupt();
166          }
167          // goto the beginning to check whether again whether we should fall through to roll
168          // several WALs, and also check whether we should quit.
169          continue;
170        }
171      }
172      try {
173        for (Iterator<Entry<WAL, RollController>> iter = wals.entrySet().iterator();
174             iter.hasNext();) {
175          Entry<WAL, RollController> entry = iter.next();
176          WAL wal = entry.getKey();
177          RollController controller = entry.getValue();
178          if (controller.isRollRequested()) {
179            // WAL roll requested, fall through
180            LOG.debug("WAL {} roll requested", wal);
181          } else if (controller.needsPeriodicRoll(now)){
182            // Time for periodic roll, fall through
183            LOG.debug("WAL {} roll period {} ms elapsed", wal, this.rollPeriod);
184          } else {
185            continue;
186          }
187          try {
188            // Force the roll if the logroll.period is elapsed or if a roll was requested.
189            // The returned value is an collection of actual region and family names.
190            Map<byte[], List<byte[]>> regionsToFlush = controller.rollWal(now);
191            if (regionsToFlush != null) {
192              for (Map.Entry<byte[], List<byte[]>> r : regionsToFlush.entrySet()) {
193                scheduleFlush(Bytes.toString(r.getKey()), r.getValue());
194              }
195            }
196          } catch (WALClosedException e) {
197            LOG.warn("WAL has been closed. Skipping rolling of writer and just remove it", e);
198            iter.remove();
199          }
200        }
201      } catch (FailedLogCloseException | ConnectException e) {
202        abort("Failed log close in log roller", e);
203      } catch (IOException ex) {
204        // Abort if we get here. We probably won't recover an IOE. HBASE-1132
205        abort("IOE in log roller",
206          ex instanceof RemoteException ? ((RemoteException) ex).unwrapRemoteException() : ex);
207      } catch (Exception ex) {
208        LOG.error("Log rolling failed", ex);
209        abort("Log rolling failed", ex);
210      }
211    }
212    LOG.info("LogRoller exiting.");
213  }
214
215  protected void afterWALArchive(Path oldPath, Path newPath) {
216  }
217
218  /**
219   * @param encodedRegionName Encoded name of region to flush.
220   * @param families stores of region to flush.
221   */
222  protected abstract void scheduleFlush(String encodedRegionName, List<byte[]> families);
223
224  private boolean isWaiting() {
225    Thread.State state = getState();
226    return state == Thread.State.WAITING || state == Thread.State.TIMED_WAITING;
227  }
228
229  /**
230   * @return true if all WAL roll finished
231   */
232  public boolean walRollFinished() {
233    // TODO add a status field of roll in RollController
234    return wals.values().stream()
235        .noneMatch(rc -> rc.needsRoll(EnvironmentEdgeManager.currentTime()))
236      && isWaiting();
237  }
238
239  /**
240   * Wait until all wals have been rolled after calling {@link #requestRollAll()}.
241   */
242  public void waitUntilWalRollFinished() throws InterruptedException {
243    while (!walRollFinished()) {
244      Thread.sleep(100);
245    }
246  }
247
248  @Override
249  public void close() {
250    running = false;
251    interrupt();
252  }
253
254  /**
255   * Independently control the roll of each wal. When use multiwal,
256   * can avoid all wal roll together. see HBASE-24665 for detail
257   */
258  protected class RollController {
259    private final WAL wal;
260    private final AtomicBoolean rollRequest;
261    private long lastRollTime;
262
263    RollController(WAL wal) {
264      this.wal = wal;
265      this.rollRequest = new AtomicBoolean(false);
266      this.lastRollTime = EnvironmentEdgeManager.currentTime();
267    }
268
269    public void requestRoll() {
270      this.rollRequest.set(true);
271    }
272
273    public Map<byte[], List<byte[]>> rollWal(long now) throws IOException {
274      this.lastRollTime = now;
275      // reset the flag in front to avoid missing roll request before we return from rollWriter.
276      this.rollRequest.set(false);
277      return wal.rollWriter(true);
278    }
279
280    public boolean isRollRequested() {
281      return rollRequest.get();
282    }
283
284    public boolean needsPeriodicRoll(long now) {
285      return (now - this.lastRollTime) > rollPeriod;
286    }
287
288    public boolean needsRoll(long now) {
289      return isRollRequested() || needsPeriodicRoll(now);
290    }
291  }
