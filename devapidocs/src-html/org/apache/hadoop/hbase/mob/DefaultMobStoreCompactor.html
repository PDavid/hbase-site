156  public DefaultMobStoreCompactor(Configuration conf, HStore store) {
157    super(conf, store);
158    // The mob cells reside in the mob-enabled column family which is held by HMobStore.
159    // During the compaction, the compactor reads the cells from the mob files and
160    // probably creates new mob files. All of these operations are included in HMobStore,
161    // so we need to cast the Store to HMobStore.
162    if (!(store instanceof HMobStore)) {
163      throw new IllegalArgumentException("The store " + store + " is not a HMobStore");
164    }
165    this.mobStore = (HMobStore) store;
166    this.mobSizeThreshold = store.getColumnFamilyDescriptor().getMobThreshold();
167    this.ioOptimizedMode = conf.get(MobConstants.MOB_COMPACTION_TYPE_KEY,
168      MobConstants.DEFAULT_MOB_COMPACTION_TYPE).
169        equals(MobConstants.OPTIMIZED_MOB_COMPACTION_TYPE);
170
171  }

173  @Override
174  public List<Path> compact(CompactionRequestImpl request,
175      ThroughputController throughputController, User user) throws IOException {
176    String tableName = store.getTableName().toString();
177    String regionName = store.getRegionInfo().getRegionNameAsString();
178    String familyName = store.getColumnFamilyName();
179    LOG.info("MOB compaction: major={} isAll={} priority={} throughput controller={}" +
180      " table={} cf={} region={}",
181      request.isMajor(), request.isAllFiles(), request.getPriority(),
182      throughputController, tableName, familyName, regionName);
183    if (request.getPriority() == HStore.PRIORITY_USER) {
184      userRequest.set(Boolean.TRUE);
185    } else {
186      userRequest.set(Boolean.FALSE);
187    }
188    LOG.debug("MOB compaction table={} cf={} region={} files: {}", tableName, familyName,
189      regionName, request.getFiles());
190    // Check if I/O optimized MOB compaction
191    if (ioOptimizedMode) {
192      if (request.isMajor() && request.getPriority() == HStore.PRIORITY_USER) {
193        Path mobDir =
194            MobUtils.getMobFamilyPath(conf, store.getTableName(), store.getColumnFamilyName());
195        List<Path> mobFiles = MobUtils.getReferencedMobFiles(request.getFiles(), mobDir);
196        //reset disableIO
197        disableIO.set(Boolean.FALSE);
198        if (mobFiles.size() > 0) {
199          calculateMobLengthMap(mobFiles);
200        }
201        LOG.info("Table={} cf={} region={}. I/O optimized MOB compaction. "+
202            "Total referenced MOB files: {}", tableName, familyName, regionName, mobFiles.size());
203      }
204    }
205
206    return compact(request, scannerFactory, writerFactory, throughputController, user);
207  }

209  private void calculateMobLengthMap(List<Path> mobFiles) throws IOException {
210    FileSystem fs = store.getFileSystem();
211    HashMap<String, Long> map = mobLengthMap.get();
212    map.clear();
213    for (Path p : mobFiles) {
214      if (MobFileName.isOldMobFileName(p.getName())) {
215        disableIO.set(Boolean.TRUE);
216      }
217      FileStatus st = fs.getFileStatus(p);
218      long size = st.getLen();
219      LOG.debug("Referenced MOB file={} size={}", p, size);
220      map.put(p.getName(), fs.getFileStatus(p).getLen());
221    }
222  }

224  /**
225   * Performs compaction on a column family with the mob flag enabled. This works only when MOB
226   * compaction is explicitly requested (by User), or by Master There are two modes of a MOB
227   * compaction:<br>
228   * <p>
229   * <ul>
230   * <li>1. Full mode - when all MOB data for a region is compacted into a single MOB file.
231   * <li>2. I/O optimized mode - for use cases with no or infrequent updates/deletes of a <br>
232   * MOB data. The main idea behind i/o optimized compaction is to limit maximum size of a MOB file
233   * produced during compaction and to limit I/O write/read amplification.
234   * </ul>
235   * The basic algorithm of compaction is the following: <br>
236   * 1. If the Put cell has a mob reference tag, the cell's value is the path of the mob file.
237   * <ol>
238   * <li>If the value size of a cell is larger than the threshold, this cell is regarded as a mob,
239   * directly copy the (with mob tag) cell into the new store file.</li>
240   * <li>Otherwise, retrieve the mob cell from the mob file, and writes a copy of the cell into the
241   * new store file.</li>
242   * </ol>
243   * 2. If the Put cell doesn't have a reference tag.
244   * <ol>
245   * <li>If the value size of a cell is larger than the threshold, this cell is regarded as a mob,
246   * write this cell to a mob file, and write the path of this mob file to the store file.</li>
247   * <li>Otherwise, directly write this cell into the store file.</li>
248   * </ol>
249   * @param fd File details
250   * @param scanner Where to read from.
251   * @param writer Where to write to.
252   * @param smallestReadPoint Smallest read point.
253   * @param cleanSeqId When true, remove seqId(used to be mvcc) value which is <= smallestReadPoint
254   * @param throughputController The compaction throughput controller.
255   * @param major Is a major compaction.
256   * @param numofFilesToCompact the number of files to compact
257   * @return Whether compaction ended; false if it was interrupted for any reason.
258   */
259  @Override
260  protected boolean performCompaction(FileDetails fd, InternalScanner scanner, CellSink writer,
261      long smallestReadPoint, boolean cleanSeqId, ThroughputController throughputController,
262      boolean major, int numofFilesToCompact) throws IOException {
263    long bytesWrittenProgressForCloseCheck = 0;
264    long bytesWrittenProgressForLog = 0;
265    long bytesWrittenProgressForShippedCall = 0;
266    // Clear old mob references
267    mobRefSet.get().clear();
268    boolean isUserRequest = userRequest.get();
269    boolean compactMOBs = major && isUserRequest;
270    boolean discardMobMiss = conf.getBoolean(MobConstants.MOB_UNSAFE_DISCARD_MISS_KEY,
271      MobConstants.DEFAULT_MOB_DISCARD_MISS);
272    if (discardMobMiss) {
273      LOG.warn("{}=true. This is unsafe setting recommended only when first upgrading to a version"+
274        " with the distributed mob compaction feature on a cluster that has experienced MOB data " +
275        "corruption.", MobConstants.MOB_UNSAFE_DISCARD_MISS_KEY);
276    }
277    long maxMobFileSize = conf.getLong(MobConstants.MOB_COMPACTION_MAX_FILE_SIZE_KEY,
278      MobConstants.DEFAULT_MOB_COMPACTION_MAX_FILE_SIZE);
279    boolean ioOptimizedMode = this.ioOptimizedMode && !disableIO.get();
280    LOG.info("Compact MOB={} optimized configured={} optimized enabled={} maximum MOB file size={}"
281      + " major={} store={}", compactMOBs,
282      this.ioOptimizedMode, ioOptimizedMode, maxMobFileSize, major, getStoreInfo());
283    // Since scanner.next() can return 'false' but still be delivering data,
284    // we have to use a do/while loop.
285    List<Cell> cells = new ArrayList<>();
286    // Limit to "hbase.hstore.compaction.kv.max" (default 10) to avoid OOME
287    int closeCheckSizeLimit = HStore.getCloseCheckInterval();
288    long lastMillis = 0;
289    if (LOG.isDebugEnabled()) {
290      lastMillis = EnvironmentEdgeManager.currentTime();
291    }
292    String compactionName = ThroughputControlUtil.getNameForThrottling(store, "compaction");
293    long now = 0;
294    boolean hasMore;
295    byte[] fileName = null;
296    StoreFileWriter mobFileWriter = null;
297    /*
298     * mobCells are used only to decide if we need to commit or abort current MOB output file.
299     */
300    long mobCells = 0;
301    long cellsCountCompactedToMob = 0, cellsCountCompactedFromMob = 0;
302    long cellsSizeCompactedToMob = 0, cellsSizeCompactedFromMob = 0;
303    boolean finished = false;
304
305    ScannerContext scannerContext =
306        ScannerContext.newBuilder().setBatchLimit(compactionKVMax).build();
307    throughputController.start(compactionName);
308    KeyValueScanner kvs = (scanner instanceof KeyValueScanner) ? (KeyValueScanner) scanner : null;
309    long shippedCallSizeLimit =
310        (long) numofFilesToCompact * this.store.getColumnFamilyDescriptor().getBlocksize();
311
312    Cell mobCell = null;
313    try {
314
315      mobFileWriter = newMobWriter(fd);
316      fileName = Bytes.toBytes(mobFileWriter.getPath().getName());
317
318      do {
319        hasMore = scanner.next(cells, scannerContext);
320        now = EnvironmentEdgeManager.currentTime();
321        for (Cell c : cells) {
322          if (compactMOBs) {
323            if (MobUtils.isMobReferenceCell(c)) {
324              String fName = MobUtils.getMobFileName(c);
325              // Added to support migration
326              try {
327                mobCell = mobStore.resolve(c, true, false).getCell();
328              } catch (FileNotFoundException fnfe) {
329                if (discardMobMiss) {
330                  LOG.error("Missing MOB cell: file={} not found cell={}", fName, c);
331                  continue;
332                } else {
333                  throw fnfe;
334                }
335              }
336
337              if (discardMobMiss && mobCell.getValueLength() == 0) {
338                LOG.error("Missing MOB cell value: file={} mob cell={} cell={}", fName,
339                  mobCell, c);
340                continue;
341              } else if (mobCell.getValueLength() == 0) {
342                String errMsg = String.format("Found 0 length MOB cell in a file=%s mob cell=%s "
343                    + " cell=%s",
344                  fName, mobCell, c);
345                throw new IOException(errMsg);
346              }
347
348              if (mobCell.getValueLength() > mobSizeThreshold) {
349                // put the mob data back to the MOB store file
350                PrivateCellUtil.setSequenceId(mobCell, c.getSequenceId());
351                if (!ioOptimizedMode) {
352                  mobFileWriter.append(mobCell);
353                  mobCells++;
354                  writer.append(
355                    MobUtils.createMobRefCell(mobCell, fileName, this.mobStore.getRefCellTags()));
356                } else {
357                  // I/O optimized mode
358                  // Check if MOB cell origin file size is
359                  // greater than threshold
360                  Long size = mobLengthMap.get().get(fName);
361                  if (size == null) {
362                    // FATAL error (we should never get here though), abort compaction
363                    // This error means that meta section of store file does not contain
364                    // MOB file, which has references in at least one cell from this store file
365                    String msg = String.format(
366                      "Found an unexpected MOB file during compaction %s, aborting compaction %s",
367                      fName, getStoreInfo());
368                    throw new IOException(msg);
369                  }
370                  // Can not be null
371                  if (size < maxMobFileSize) {
372                    // If MOB cell origin file is below threshold
373                    // it is get compacted
374                    mobFileWriter.append(mobCell);
375                    // Update number of mobCells in a current mob writer
376                    mobCells++;
377                    writer.append(
378                      MobUtils.createMobRefCell(mobCell, fileName, this.mobStore.getRefCellTags()));
379                    // Update total size of the output (we do not take into account
380                    // file compression yet)
381                    long len = mobFileWriter.getPos();
382                    if (len > maxMobFileSize) {
383                      LOG.debug("Closing output MOB File, length={} file={}, store={}", len,
384                        mobFileWriter.getPath().getName(), getStoreInfo());
385                      commitOrAbortMobWriter(mobFileWriter, fd.maxSeqId, mobCells, major);
386                      mobFileWriter = newMobWriter(fd);
387                      fileName = Bytes.toBytes(mobFileWriter.getPath().getName());
388                      mobCells = 0;
389                    }
390                  } else {
391                    // We leave large MOB file as is (is not compacted),
392                    // then we update set of MOB file references
393                    // and append mob cell directly to the store's writer
394                    mobRefSet.get().add(fName);
395                    writer.append(mobCell);
396                  }
397                }
398              } else {
399                // If MOB value is less than threshold, append it directly to a store file
400                PrivateCellUtil.setSequenceId(mobCell, c.getSequenceId());
401                writer.append(mobCell);
402                cellsCountCompactedFromMob++;
403                cellsSizeCompactedFromMob += mobCell.getValueLength();
404              }
405            } else {
406              // Not a MOB reference cell
407              int size = c.getValueLength();
408              if (size > mobSizeThreshold) {
409                // This MOB cell comes from a regular store file
410                // therefore we store it into original mob output
411                mobFileWriter.append(c);
412                writer
413                    .append(MobUtils.createMobRefCell(c, fileName, this.mobStore.getRefCellTags()));
414                mobCells++;
415                cellsCountCompactedToMob++;
416                cellsSizeCompactedToMob += c.getValueLength();
417                if (ioOptimizedMode) {
418                  // Update total size of the output (we do not take into account
419                  // file compression yet)
420                  long len = mobFileWriter.getPos();
421                  if (len > maxMobFileSize) {
422                    commitOrAbortMobWriter(mobFileWriter, fd.maxSeqId, mobCells, major);
423                    mobFileWriter = newMobWriter(fd);
424                    fileName = Bytes.toBytes(mobFileWriter.getPath().getName());
425                    mobCells = 0;
426                  }
427                }
428              } else {
429                // Not a MOB cell, write it directly to a store file
430                writer.append(c);
431              }
432            }
433          } else if (c.getTypeByte() != KeyValue.Type.Put.getCode()) {
434            // Not a major compaction or major with MOB disabled
435            // If the kv type is not put, directly write the cell
436            // to the store file.
437            writer.append(c);
438          } else if (MobUtils.isMobReferenceCell(c)) {
439            // Not a major MOB compaction, Put MOB reference
440            if (MobUtils.hasValidMobRefCellValue(c)) {
441              // We do not check mobSizeThreshold during normal compaction,
442              // leaving it to a MOB compaction run
443              writer.append(c);
444              // Add MOB reference to a MOB reference set
445              mobRefSet.get().add(MobUtils.getMobFileName(c));
446            } else {
447              String errMsg = String.format("Corrupted MOB reference: %s", c.toString());
448              throw new IOException(errMsg);
449            }
450          } else if (c.getValueLength() <= mobSizeThreshold) {
451            // If the value size of a cell is not larger than the threshold, directly write it to
452            // the store file.
453            writer.append(c);
454          } else {
455            // If the value size of a cell is larger than the threshold, it's regarded as a mob,
456            // write this cell to a mob file, and write the path to the store file.
457            mobCells++;
458            // append the original keyValue in the mob file.
459            mobFileWriter.append(c);
460            Cell reference = MobUtils.createMobRefCell(c, fileName, this.mobStore.getRefCellTags());
461            // write the cell whose value is the path of a mob file to the store file.
462            writer.append(reference);
463            cellsCountCompactedToMob++;
464            cellsSizeCompactedToMob += c.getValueLength();
465            if (ioOptimizedMode) {
466              long len = mobFileWriter.getPos();
467              if (len > maxMobFileSize) {
468                commitOrAbortMobWriter(mobFileWriter, fd.maxSeqId, mobCells, major);
469                mobFileWriter = newMobWriter(fd);
470                fileName = Bytes.toBytes(mobFileWriter.getPath().getName());
471                mobCells = 0;
472              }
473            }
474          }
475
476          int len = c.getSerializedSize();
477          ++progress.currentCompactedKVs;
478          progress.totalCompactedSize += len;
479          bytesWrittenProgressForShippedCall += len;
480          if (LOG.isDebugEnabled()) {
481            bytesWrittenProgressForLog += len;
482          }
483          throughputController.control(compactionName, len);
484          // check periodically to see if a system stop is requested
485          if (closeCheckSizeLimit > 0) {
486            bytesWrittenProgressForCloseCheck += len;
487            if (bytesWrittenProgressForCloseCheck > closeCheckSizeLimit) {
488              bytesWrittenProgressForCloseCheck = 0;
489              if (!store.areWritesEnabled()) {
490                progress.cancel();
491                return false;
492              }
493            }
494          }
495          if (kvs != null && bytesWrittenProgressForShippedCall > shippedCallSizeLimit) {
496            ((ShipperListener) writer).beforeShipped();
497            kvs.shipped();
498            bytesWrittenProgressForShippedCall = 0;
499          }
500        }
501        // Log the progress of long running compactions every minute if
502        // logging at DEBUG level
503        if (LOG.isDebugEnabled()) {
504          if ((now - lastMillis) >= COMPACTION_PROGRESS_LOG_INTERVAL) {
505            String rate = String.format("%.2f",
506              (bytesWrittenProgressForLog / 1024.0) / ((now - lastMillis) / 1000.0));
507            LOG.debug("Compaction progress: {} {}, rate={} KB/sec, throughputController is {}",
508              compactionName, progress, rate, throughputController);
509            lastMillis = now;
510            bytesWrittenProgressForLog = 0;
511          }
512        }
513        cells.clear();
514      } while (hasMore);
515      finished = true;
516    } catch (InterruptedException e) {
517      progress.cancel();
518      throw new InterruptedIOException(
519          "Interrupted while control throughput of compacting " + compactionName);
520    } catch (IOException t) {
521      String msg = "Mob compaction failed for region: " +
522        store.getRegionInfo().getEncodedName();
523      throw new IOException(msg, t);
524    } finally {
525      // Clone last cell in the final because writer will append last cell when committing. If
526      // don't clone here and once the scanner get closed, then the memory of last cell will be
527      // released. (HBASE-22582)
528      ((ShipperListener) writer).beforeShipped();
529      throughputController.finish(compactionName);
530      if (!finished && mobFileWriter != null) {
531        // Remove all MOB references because compaction failed
532        mobRefSet.get().clear();
533        // Abort writer
534        LOG.debug("Aborting writer for {} because of a compaction failure, Store {}",
535          mobFileWriter.getPath(), getStoreInfo());
536        abortWriter(mobFileWriter);
537      }
538    }
539
540    // Commit last MOB writer
541    commitOrAbortMobWriter(mobFileWriter, fd.maxSeqId, mobCells, major);
542    mobStore.updateCellsCountCompactedFromMob(cellsCountCompactedFromMob);
543    mobStore.updateCellsCountCompactedToMob(cellsCountCompactedToMob);
544    mobStore.updateCellsSizeCompactedFromMob(cellsSizeCompactedFromMob);
545    mobStore.updateCellsSizeCompactedToMob(cellsSizeCompactedToMob);
546    progress.complete();
547    return true;
548  }

550  private String getStoreInfo() {
551    return String.format("[table=%s family=%s region=%s]", store.getTableName().getNameAsString(),
552      store.getColumnFamilyName(), store.getRegionInfo().getEncodedName()) ;
553  }

555  private void clearThreadLocals() {
556    Set<String> set = mobRefSet.get();
557    if (set != null) {
558      set.clear();
559    }
560    HashMap<String, Long> map = mobLengthMap.get();
561    if (map != null) {
562      map.clear();
563    }
564  }

566  private StoreFileWriter newMobWriter(FileDetails fd)
567      throws IOException {
568    try {
569      StoreFileWriter mobFileWriter = mobStore.createWriterInTmp(new Date(fd.latestPutTs),
570        fd.maxKeyCount, compactionCompression, store.getRegionInfo().getStartKey(), true);
571      LOG.debug("New MOB writer created={} store={}", mobFileWriter.getPath().getName(),
572        getStoreInfo());
573      // Add reference we get for compact MOB
574      mobRefSet.get().add(mobFileWriter.getPath().getName());
575      return mobFileWriter;
576    } catch (IOException e) {
577      // Bailing out
578      throw new IOException(String.format("Failed to create mob writer, store=%s",
579        getStoreInfo()), e);
580    }
581  }

583  private void commitOrAbortMobWriter(StoreFileWriter mobFileWriter, long maxSeqId, long mobCells,
584      boolean major) throws IOException {
585    // Commit or abort major mob writer
586    // If IOException happens during below operation, some
587    // MOB files can be committed partially, but corresponding
588    // store file won't be committed, therefore these MOB files
589    // become orphans and will be deleted during next MOB cleaning chore cycle
590
591    if (mobFileWriter != null) {
592      LOG.debug("Commit or abort size={} mobCells={} major={} file={}, store={}",
593        mobFileWriter.getPos(), mobCells, major, mobFileWriter.getPath().getName(),
594        getStoreInfo());
595      Path path =
596          MobUtils.getMobFamilyPath(conf, store.getTableName(), store.getColumnFamilyName());
597      if (mobCells > 0) {
598        // If the mob file is not empty, commit it.
599        mobFileWriter.appendMetadata(maxSeqId, major, mobCells);
600        mobFileWriter.close();
601        mobStore.commitFile(mobFileWriter.getPath(), path);
602      } else {
603        // If the mob file is empty, delete it instead of committing.
604        LOG.debug("Aborting writer for {} because there are no MOB cells, store={}",
605          mobFileWriter.getPath(), getStoreInfo());
606        // Remove MOB file from reference set
607        mobRefSet.get().remove(mobFileWriter.getPath().getName());
608        abortWriter(mobFileWriter);
609      }
610    } else {
611      LOG.debug("Mob file writer is null, skipping commit/abort, store=",
612        getStoreInfo());
613    }
614  }
615
616
617  @Override
618  protected List<Path> commitWriter(StoreFileWriter writer, FileDetails fd,
619      CompactionRequestImpl request) throws IOException {
620    List<Path> newFiles = Lists.newArrayList(writer.getPath());
621    writer.appendMetadata(fd.maxSeqId, request.isAllFiles(), request.getFiles());
622    // Append MOB references
623    Set<String> refSet = mobRefSet.get();
624    writer.appendMobMetadata(refSet);
625    writer.close();
626    clearThreadLocals();
627    return newFiles;
628  }
