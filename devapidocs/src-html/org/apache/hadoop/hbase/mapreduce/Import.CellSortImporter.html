193  public static class CellSortImporter
194      extends TableMapper<CellWritableComparable, Cell> {
195    private Map<byte[], byte[]> cfRenameMap;
196    private Filter filter;
197    private static final Logger LOG = LoggerFactory.getLogger(CellImporter.class);
198
199    /**
200     * @param row  The current table row key.
201     * @param value  The columns.
202     * @param context  The current context.
203     * @throws IOException When something is broken with the data.
204     */
205    @Override
206    public void map(ImmutableBytesWritable row, Result value,
207      Context context)
208    throws IOException {
209      try {
210        if (LOG.isTraceEnabled()) {
211          LOG.trace("Considering the row."
212              + Bytes.toString(row.get(), row.getOffset(), row.getLength()));
213        }
214        if (filter == null || !filter.filterRowKey(
215          PrivateCellUtil.createFirstOnRow(row.get(), row.getOffset(), (short) row.getLength()))) {
216          for (Cell kv : value.rawCells()) {
217            kv = filterKv(filter, kv);
218            // skip if we filtered it out
219            if (kv == null) continue;
220            Cell ret = convertKv(kv, cfRenameMap);
221            context.write(new CellWritableComparable(ret), ret);
222          }
223        }
224      } catch (InterruptedException e) {
225        LOG.error("Interrupted while emitting Cell", e);
226        Thread.currentThread().interrupt();
227      }
228    }
229
230    @Override
231    public void setup(Context context) throws IOException {
232      cfRenameMap = createCfRenameMap(context.getConfiguration());
233      filter = instantiateFilter(context.getConfiguration());
234      int reduceNum = context.getNumReduceTasks();
235      Configuration conf = context.getConfiguration();
236      TableName tableName = TableName.valueOf(context.getConfiguration().get(TABLE_NAME));
237      try (Connection conn = ConnectionFactory.createConnection(conf);
238          RegionLocator regionLocator = conn.getRegionLocator(tableName)) {
239        byte[][] startKeys = regionLocator.getStartKeys();
240        if (startKeys.length != reduceNum) {
241          throw new IOException("Region split after job initialization");
242        }
243        CellWritableComparable[] startKeyWraps =
244            new CellWritableComparable[startKeys.length - 1];
245        for (int i = 1; i < startKeys.length; ++i) {
246          startKeyWraps[i - 1] =
247              new CellWritableComparable(KeyValueUtil.createFirstOnRow(startKeys[i]));
248        }
249        CellWritableComparablePartitioner.START_KEYS = startKeyWraps;
250      }
251    }

305  public static class Importer extends TableMapper<ImmutableBytesWritable, Mutation> {
306    private Map<byte[], byte[]> cfRenameMap;
307    private List<UUID> clusterIds;
308    private Filter filter;
309    private Durability durability;
310
311    /**
312     * @param row  The current table row key.
313     * @param value  The columns.
314     * @param context  The current context.
315     * @throws IOException When something is broken with the data.
316     */
317    @Override
318    public void map(ImmutableBytesWritable row, Result value,
319      Context context)
320    throws IOException {
321      try {
322        writeResult(row, value, context);
323      } catch (InterruptedException e) {
324        LOG.error("Interrupted while writing result", e);
325        Thread.currentThread().interrupt();
326      }
327    }
328
329    private void writeResult(ImmutableBytesWritable key, Result result, Context context)
330    throws IOException, InterruptedException {
331      Put put = null;
332      Delete delete = null;
333      if (LOG.isTraceEnabled()) {
334        LOG.trace("Considering the row."
335            + Bytes.toString(key.get(), key.getOffset(), key.getLength()));
336      }
337      if (filter == null
338          || !filter.filterRowKey(PrivateCellUtil.createFirstOnRow(key.get(), key.getOffset(),
339              (short) key.getLength()))) {
340        processKV(key, result, context, put, delete);
341      }
342    }
343
344    protected void processKV(ImmutableBytesWritable key, Result result, Context context, Put put,
345        Delete delete) throws IOException, InterruptedException {
346      for (Cell kv : result.rawCells()) {
347        kv = filterKv(filter, kv);
348        // skip if we filter it out
349        if (kv == null) continue;
350
351        kv = convertKv(kv, cfRenameMap);
352        // Deletes and Puts are gathered and written when finished
353        /*
354         * If there are sequence of mutations and tombstones in an Export, and after Import the same
355         * sequence should be restored as it is. If we combine all Delete tombstones into single
356         * request then there is chance of ignoring few DeleteFamily tombstones, because if we
357         * submit multiple DeleteFamily tombstones in single Delete request then we are maintaining
358         * only newest in hbase table and ignoring other. Check - HBASE-12065
359         */
360        if (PrivateCellUtil.isDeleteFamily(kv)) {
361          Delete deleteFamily = new Delete(key.get());
362          deleteFamily.add(kv);
363          if (durability != null) {
364            deleteFamily.setDurability(durability);
365          }
366          deleteFamily.setClusterIds(clusterIds);
367          context.write(key, deleteFamily);
368        } else if (CellUtil.isDelete(kv)) {
369          if (delete == null) {
370            delete = new Delete(key.get());
371          }
372          delete.add(kv);
373        } else {
374          if (put == null) {
375            put = new Put(key.get());
376          }
377          addPutToKv(put, kv);
378        }
379      }
380      if (put != null) {
381        if (durability != null) {
382          put.setDurability(durability);
383        }
384        put.setClusterIds(clusterIds);
385        context.write(key, put);
386      }
387      if (delete != null) {
388        if (durability != null) {
389          delete.setDurability(durability);
390        }
391        delete.setClusterIds(clusterIds);
392        context.write(key, delete);
393      }
394    }
395
396    protected void addPutToKv(Put put, Cell kv) throws IOException {
397      put.add(kv);
398    }
399
400    @Override
401    public void setup(Context context) {
402      LOG.info("Setting up " + getClass() + " mapper.");
403      Configuration conf = context.getConfiguration();
404      cfRenameMap = createCfRenameMap(conf);
405      filter = instantiateFilter(conf);
406      String durabilityStr = conf.get(WAL_DURABILITY);
407      if(durabilityStr != null){
408        durability = Durability.valueOf(durabilityStr.toUpperCase(Locale.ROOT));
409        LOG.info("setting WAL durability to " + durability);
410      } else {
411        LOG.info("setting WAL durability to default.");
412      }
413      // TODO: This is kind of ugly doing setup of ZKW just to read the clusterid.
414      ZKWatcher zkw = null;
415      Exception ex = null;
416      try {
417        zkw = new ZKWatcher(conf, context.getTaskAttemptID().toString(), null);
418        clusterIds = Collections.singletonList(ZKClusterId.getUUIDForCluster(zkw));
419      } catch (ZooKeeperConnectionException e) {
420        ex = e;
421        LOG.error("Problem connecting to ZooKeper during task setup", e);
422      } catch (KeeperException e) {
423        ex = e;
424        LOG.error("Problem reading ZooKeeper data during task setup", e);
425      } catch (IOException e) {
426        ex = e;
427        LOG.error("Problem setting up task", e);
428      } finally {
429        if (zkw != null) zkw.close();
430      }
431      if (clusterIds == null) {
432        // exit early if setup fails
433        throw new RuntimeException(ex);
434      }
435    }
436  }

438  /**
439   * Create a {@link Filter} to apply to all incoming keys ({@link KeyValue KeyValues}) to
440   * optionally not include in the job output
441   * @param conf {@link Configuration} from which to load the filter
442   * @return the filter to use for the task, or <tt>null</tt> if no filter to should be used
443   * @throws IllegalArgumentException if the filter is misconfigured
444   */
445  public static Filter instantiateFilter(Configuration conf) {
446    // get the filter, if it was configured
447    Class<? extends Filter> filterClass = conf.getClass(FILTER_CLASS_CONF_KEY, null, Filter.class);
448    if (filterClass == null) {
449      LOG.debug("No configured filter class, accepting all keyvalues.");
450      return null;
451    }
452    LOG.debug("Attempting to create filter:" + filterClass);
453    String[] filterArgs = conf.getStrings(FILTER_ARGS_CONF_KEY);
454    ArrayList<byte[]> quotedArgs = toQuotedByteArrays(filterArgs);
455    try {
456      Method m = filterClass.getMethod("createFilterFromArguments", ArrayList.class);
457      return (Filter) m.invoke(null, quotedArgs);
458    } catch (IllegalAccessException e) {
459      LOG.error("Couldn't instantiate filter!", e);
460      throw new RuntimeException(e);
461    } catch (SecurityException e) {
462      LOG.error("Couldn't instantiate filter!", e);
463      throw new RuntimeException(e);
464    } catch (NoSuchMethodException e) {
465      LOG.error("Couldn't instantiate filter!", e);
466      throw new RuntimeException(e);
467    } catch (IllegalArgumentException e) {
468      LOG.error("Couldn't instantiate filter!", e);
469      throw new RuntimeException(e);
470    } catch (InvocationTargetException e) {
471      LOG.error("Couldn't instantiate filter!", e);
472      throw new RuntimeException(e);
473    }
474  }

476  private static ArrayList<byte[]> toQuotedByteArrays(String... stringArgs) {
477    ArrayList<byte[]> quotedArgs = new ArrayList<>();
478    for (String stringArg : stringArgs) {
479      // all the filters' instantiation methods expected quoted args since they are coming from
480      // the shell, so add them here, though it shouldn't really be needed :-/
481      quotedArgs.add(Bytes.toBytes("'" + stringArg + "'"));
482    }
483    return quotedArgs;
484  }

486  /**
487   * Attempt to filter out the keyvalue
488   * @param c {@link Cell} on which to apply the filter
489   * @return <tt>null</tt> if the key should not be written, otherwise returns the original
490   *         {@link Cell}
491   */
492  public static Cell filterKv(Filter filter, Cell c) throws IOException {
493    // apply the filter and skip this kv if the filter doesn't apply
494    if (filter != null) {
495      Filter.ReturnCode code = filter.filterCell(c);
496      if (LOG.isTraceEnabled()) {
497        LOG.trace("Filter returned:" + code + " for the cell:" + c);
498      }
499      // if its not an accept type, then skip this kv
500      if (!(code.equals(Filter.ReturnCode.INCLUDE) || code
501          .equals(Filter.ReturnCode.INCLUDE_AND_NEXT_COL))) {
502        return null;
503      }
504    }
505    return c;
506  }

508  // helper: create a new KeyValue based on CF rename map
509  private static Cell convertKv(Cell kv, Map<byte[], byte[]> cfRenameMap) {
510    if(cfRenameMap != null) {
511      // If there's a rename mapping for this CF, create a new KeyValue
512      byte[] newCfName = cfRenameMap.get(CellUtil.cloneFamily(kv));
513      if (newCfName != null) {
514        kv = new KeyValue(kv.getRowArray(), // row buffer
515            kv.getRowOffset(),              // row offset
516            kv.getRowLength(),              // row length
517            newCfName,                      // CF buffer
518            0,                              // CF offset
519            newCfName.length,               // CF length
520            kv.getQualifierArray(),         // qualifier buffer
521            kv.getQualifierOffset(),        // qualifier offset
522            kv.getQualifierLength(),        // qualifier length
523            kv.getTimestamp(),              // timestamp
524            KeyValue.Type.codeToType(kv.getTypeByte()), // KV Type
525            kv.getValueArray(),             // value buffer
526            kv.getValueOffset(),            // value offset
527            kv.getValueLength());           // value length
528      }
529    }
530    return kv;
531  }

533  // helper: make a map from sourceCfName to destCfName by parsing a config key
534  private static Map<byte[], byte[]> createCfRenameMap(Configuration conf) {
535    Map<byte[], byte[]> cfRenameMap = null;
536    String allMappingsPropVal = conf.get(CF_RENAME_PROP);
537    if(allMappingsPropVal != null) {
538      // The conf value format should be sourceCf1:destCf1,sourceCf2:destCf2,...
539      String[] allMappings = allMappingsPropVal.split(",");
540      for (String mapping: allMappings) {
541        if(cfRenameMap == null) {
542            cfRenameMap = new TreeMap<>(Bytes.BYTES_COMPARATOR);
543        }
544        String [] srcAndDest = mapping.split(":");
545        if(srcAndDest.length != 2) {
546            continue;
547        }
548        cfRenameMap.put(Bytes.toBytes(srcAndDest[0]), Bytes.toBytes(srcAndDest[1]));
549      }
550    }
551    return cfRenameMap;
552  }

554  /**
555   * <p>Sets a configuration property with key {@link #CF_RENAME_PROP} in conf that tells
556   * the mapper how to rename column families.
557   *
558   * <p>Alternately, instead of calling this function, you could set the configuration key
559   * {@link #CF_RENAME_PROP} yourself. The value should look like
560   * <pre>srcCf1:destCf1,srcCf2:destCf2,....</pre>. This would have the same effect on
561   * the mapper behavior.
562   *
563   * @param conf the Configuration in which the {@link #CF_RENAME_PROP} key will be
564   *  set
565   * @param renameMap a mapping from source CF names to destination CF names
566   */
567  static public void configureCfRenaming(Configuration conf,
568          Map<String, String> renameMap) {
569    StringBuilder sb = new StringBuilder();
570    for(Map.Entry<String,String> entry: renameMap.entrySet()) {
571      String sourceCf = entry.getKey();
572      String destCf = entry.getValue();
573
574      if(sourceCf.contains(":") || sourceCf.contains(",") ||
575              destCf.contains(":") || destCf.contains(",")) {
576        throw new IllegalArgumentException("Illegal character in CF names: "
577              + sourceCf + ", " + destCf);
578      }
579
580      if(sb.length() != 0) {
581        sb.append(",");
582      }
583      sb.append(sourceCf + ":" + destCf);
584    }
585    conf.set(CF_RENAME_PROP, sb.toString());
586  }

600  /**
601   * Sets up the actual job.
602   * @param conf The current configuration.
603   * @param args The command line parameters.
604   * @return The newly created job.
605   * @throws IOException When setting up the job fails.
606   */
607  public static Job createSubmittableJob(Configuration conf, String[] args)
608  throws IOException {
609    TableName tableName = TableName.valueOf(args[0]);
610    conf.set(TABLE_NAME, tableName.getNameAsString());
611    Path inputDir = new Path(args[1]);
612    Job job = Job.getInstance(conf, conf.get(JOB_NAME_CONF_KEY, NAME + "_" + tableName));
613    job.setJarByClass(Importer.class);
614    FileInputFormat.setInputPaths(job, inputDir);
615    job.setInputFormatClass(SequenceFileInputFormat.class);
616    String hfileOutPath = conf.get(BULK_OUTPUT_CONF_KEY);
617
618    // make sure we get the filter in the jars
619    try {
620      Class<? extends Filter> filter = conf.getClass(FILTER_CLASS_CONF_KEY, null, Filter.class);
621      if (filter != null) {
622        TableMapReduceUtil.addDependencyJarsForClasses(conf, filter);
623      }
624    } catch (Exception e) {
625      throw new IOException(e);
626    }
627
628    if (hfileOutPath != null && conf.getBoolean(HAS_LARGE_RESULT, false)) {
629      LOG.info("Use Large Result!!");
630      try (Connection conn = ConnectionFactory.createConnection(conf);
631          Table table = conn.getTable(tableName);
632          RegionLocator regionLocator = conn.getRegionLocator(tableName)) {
633        HFileOutputFormat2.configureIncrementalLoad(job, table.getDescriptor(), regionLocator);
634        job.setMapperClass(CellSortImporter.class);
635        job.setReducerClass(CellReducer.class);
636        Path outputDir = new Path(hfileOutPath);
637        FileOutputFormat.setOutputPath(job, outputDir);
638        job.setMapOutputKeyClass(CellWritableComparable.class);
639        job.setMapOutputValueClass(MapReduceExtendedCell.class);
640        job.getConfiguration().setClass("mapreduce.job.output.key.comparator.class",
641            CellWritableComparable.CellWritableComparator.class,
642            RawComparator.class);
643        Path partitionsPath =
644            new Path(TotalOrderPartitioner.getPartitionFile(job.getConfiguration()));
645        FileSystem fs = FileSystem.get(job.getConfiguration());
646        fs.deleteOnExit(partitionsPath);
647        job.setPartitionerClass(CellWritableComparablePartitioner.class);
648        job.setNumReduceTasks(regionLocator.getStartKeys().length);
649        TableMapReduceUtil.addDependencyJarsForClasses(job.getConfiguration(),
650            org.apache.hbase.thirdparty.com.google.common.base.Preconditions.class);
651      }
652    } else if (hfileOutPath != null) {
653      LOG.info("writing to hfiles for bulk load.");
654      job.setMapperClass(CellImporter.class);
655      try (Connection conn = ConnectionFactory.createConnection(conf);
656          Table table = conn.getTable(tableName);
657          RegionLocator regionLocator = conn.getRegionLocator(tableName)){
658        job.setReducerClass(CellSortReducer.class);
659        Path outputDir = new Path(hfileOutPath);
660        FileOutputFormat.setOutputPath(job, outputDir);
661        job.setMapOutputKeyClass(ImmutableBytesWritable.class);
662        job.setMapOutputValueClass(MapReduceExtendedCell.class);
663        HFileOutputFormat2.configureIncrementalLoad(job, table.getDescriptor(), regionLocator);
664        TableMapReduceUtil.addDependencyJarsForClasses(job.getConfiguration(),
665            org.apache.hbase.thirdparty.com.google.common.base.Preconditions.class);
666      }
667    } else {
668      LOG.info("writing directly to table from Mapper.");
669      // No reducers.  Just write straight to table.  Call initTableReducerJob
670      // because it sets up the TableOutputFormat.
671      job.setMapperClass(Importer.class);
672      TableMapReduceUtil.initTableReducerJob(tableName.getNameAsString(), null, job);
673      job.setNumReduceTasks(0);
674    }
675    return job;
676  }

715  /**
716   * If the durability is set to {@link Durability#SKIP_WAL} and the data is imported to hbase, we
717   * need to flush all the regions of the table as the data is held in memory and is also not
718   * present in the Write Ahead Log to replay in scenarios of a crash. This method flushes all the
719   * regions of the table in the scenarios of import data to hbase with {@link Durability#SKIP_WAL}
720   */
721  public static void flushRegionsIfNecessary(Configuration conf) throws IOException,
722      InterruptedException {
723    String tableName = conf.get(TABLE_NAME);
724    Admin hAdmin = null;
725    Connection connection = null;
726    String durability = conf.get(WAL_DURABILITY);
727    // Need to flush if the data is written to hbase and skip wal is enabled.
728    if (conf.get(BULK_OUTPUT_CONF_KEY) == null && durability != null
729        && Durability.SKIP_WAL.name().equalsIgnoreCase(durability)) {
730      LOG.info("Flushing all data that skipped the WAL.");
731      try {
732        connection = ConnectionFactory.createConnection(conf);
733        hAdmin = connection.getAdmin();
734        hAdmin.flush(TableName.valueOf(tableName));
735      } finally {
736        if (hAdmin != null) {
737          hAdmin.close();
738        }
739        if (connection != null) {
740          connection.close();
741        }
742      }
743    }
744  }

746  @Override
747  public int run(String[] args) throws Exception {
748    if (args.length < 2) {
749      usage("Wrong number of arguments: " + args.length);
750      return -1;
751    }
752    String inputVersionString = System.getProperty(ResultSerialization.IMPORT_FORMAT_VER);
753    if (inputVersionString != null) {
754      getConf().set(ResultSerialization.IMPORT_FORMAT_VER, inputVersionString);
755    }
756    Job job = createSubmittableJob(getConf(), args);
757    boolean isJobSuccessful = job.waitForCompletion(true);
758    if(isJobSuccessful){
759      // Flush all the regions of the table
760      flushRegionsIfNecessary(getConf());
761    }
762    long inputRecords = job.getCounters().findCounter(TaskCounter.MAP_INPUT_RECORDS).getValue();
763    long outputRecords = job.getCounters().findCounter(TaskCounter.MAP_OUTPUT_RECORDS).getValue();
764    if (outputRecords < inputRecords) {
765      System.err.println("Warning, not all records were imported (maybe filtered out).");
766      if (outputRecords == 0) {
767        System.err.println("If the data was exported from HBase 0.94 "+
768            "consider using -Dhbase.import.version=0.94.");
769      }
770    }
771
772    return (isJobSuccessful ? 0 : 1);
773  }
