056/**
057 * Change log level in runtime.
058 */
059@InterfaceAudience.Private
060public final class LogLevel {
061  private static final String USAGES = "\nUsage: General options are:\n"
062      + "\t[-getlevel <host:port> <classname>\n"
063      + "\t[-setlevel <host:port> <classname> <level> ";
064
065  public static final String PROTOCOL_HTTP = "http";
066  /**
067   * A command line implementation
068   */
069  public static void main(String[] args) throws Exception {
070    CLI cli = new CLI(new Configuration());
071    System.exit(cli.run(args));
072  }
073
074  /**
075   * Valid command line options.
076   */
077  private enum Operations {
078    GETLEVEL,
079    SETLEVEL,
080    UNKNOWN
081  }
082
083  private static void printUsage() {
084    System.err.println(USAGES);
085    System.exit(-1);
086  }
087
088  @VisibleForTesting
089  static class CLI extends Configured implements Tool {
090    private Operations operation = Operations.UNKNOWN;
091    private String hostName;
092    private String className;
093    private String level;
094
095    CLI(Configuration conf) {
096      setConf(conf);
097    }
098
099    @Override
100    public int run(String[] args) throws Exception {
101      try {
102        parseArguments(args);
103        sendLogLevelRequest();
104      } catch (HadoopIllegalArgumentException e) {
105        printUsage();
106      }
107      return 0;
108    }
109
110    /**
111     * Send HTTP request to the daemon.
112     * @throws HadoopIllegalArgumentException if arguments are invalid.
113     * @throws Exception if unable to connect
114     */
115    private void sendLogLevelRequest()
116        throws HadoopIllegalArgumentException, Exception {
117      switch (operation) {
118        case GETLEVEL:
119          doGetLevel();
120          break;
121        case SETLEVEL:
122          doSetLevel();
123          break;
124        default:
125          throw new HadoopIllegalArgumentException(
126              "Expect either -getlevel or -setlevel");
127      }
128    }
129
130    public void parseArguments(String[] args) throws
131        HadoopIllegalArgumentException {
132      if (args.length == 0) {
133        throw new HadoopIllegalArgumentException("No arguments specified");
134      }
135      int nextArgIndex = 0;
136      while (nextArgIndex < args.length) {
137        switch (args[nextArgIndex]) {
138          case "-getlevel":
139            nextArgIndex = parseGetLevelArgs(args, nextArgIndex);
140            break;
141          case "-setlevel":
142            nextArgIndex = parseSetLevelArgs(args, nextArgIndex);
143            break;
144          default:
145            throw new HadoopIllegalArgumentException(
146                "Unexpected argument " + args[nextArgIndex]);
147        }
148      }
149
150      // if operation is never specified in the arguments
151      if (operation == Operations.UNKNOWN) {
152        throw new HadoopIllegalArgumentException(
153            "Must specify either -getlevel or -setlevel");
154      }
155    }
156
157    private int parseGetLevelArgs(String[] args, int index) throws
158        HadoopIllegalArgumentException {
159      // fail if multiple operations are specified in the arguments
160      if (operation != Operations.UNKNOWN) {
161        throw new HadoopIllegalArgumentException("Redundant -getlevel command");
162      }
163      // check number of arguments is sufficient
164      if (index + 2 >= args.length) {
165        throw new HadoopIllegalArgumentException("-getlevel needs two parameters");
166      }
167      operation = Operations.GETLEVEL;
168      hostName = args[index + 1];
169      className = args[index + 2];
170      return index + 3;
171    }
172
173    private int parseSetLevelArgs(String[] args, int index) throws
174        HadoopIllegalArgumentException {
175      // fail if multiple operations are specified in the arguments
176      if (operation != Operations.UNKNOWN) {
177        throw new HadoopIllegalArgumentException("Redundant -setlevel command");
178      }
179      // check number of arguments is sufficient
180      if (index + 3 >= args.length) {
181        throw new HadoopIllegalArgumentException("-setlevel needs three parameters");
182      }
183      operation = Operations.SETLEVEL;
184      hostName = args[index + 1];
185      className = args[index + 2];
186      level = args[index + 3];
187      return index + 4;
188    }
189
190    /**
191     * Send HTTP request to get log level.
192     *
193     * @throws HadoopIllegalArgumentException if arguments are invalid.
194     * @throws Exception if unable to connect
195     */
196    private void doGetLevel() throws Exception {
197      process(PROTOCOL_HTTP + "://" + hostName + "/logLevel?log=" + className);
198    }
199
200    /**
201     * Send HTTP request to set log level.
202     *
203     * @throws HadoopIllegalArgumentException if arguments are invalid.
204     * @throws Exception if unable to connect
205     */
206    private void doSetLevel() throws Exception {
207      process(PROTOCOL_HTTP + "://" + hostName + "/logLevel?log=" + className
208          + "&level=" + level);
209    }
210
211    /**
212     * Connect to the URL. Supports HTTP and supports SPNEGO
213     * authentication. It falls back to simple authentication if it fails to
214     * initiate SPNEGO.
215     *
216     * @param url the URL address of the daemon servlet
217     * @return a connected connection
218     * @throws Exception if it can not establish a connection.
219     */
220    private URLConnection connect(URL url) throws Exception {
221      AuthenticatedURL.Token token = new AuthenticatedURL.Token();
222      AuthenticatedURL aUrl;
223      URLConnection connection;
224
225      aUrl = new AuthenticatedURL(new KerberosAuthenticator());
226      connection = aUrl.openConnection(url, token);
227      connection.connect();
228      return connection;
229    }
230
231    /**
232     * Configures the client to send HTTP request to the URL.
233     * Supports SPENGO for authentication.
234     * @param urlString URL and query string to the daemon's web UI
235     * @throws Exception if unable to connect
236     */
237    private void process(String urlString) throws Exception {
238      URL url = new URL(urlString);
239      System.out.println("Connecting to " + url);
240
241      URLConnection connection = connect(url);
242
243      // read from the servlet
244
245      try (InputStreamReader streamReader =
246            new InputStreamReader(connection.getInputStream(), Charsets.UTF_8);
247           BufferedReader bufferedReader = new BufferedReader(streamReader)) {
248        bufferedReader.lines().filter(Objects::nonNull).filter(line -> line.startsWith(MARKER))
249            .forEach(line -> System.out.println(TAG.matcher(line).replaceAll("")));
250      } catch (IOException ioe) {
251        System.err.println("" + ioe);
252      }
253    }
254  }
255
256  private static final String MARKER = "<!-- OUTPUT -->";
257  private static final Pattern TAG = Pattern.compile("<[^>]*>");
258
259  /**
260   * A servlet implementation
261   */
262  @InterfaceAudience.LimitedPrivate({"HDFS", "MapReduce"})
263  @InterfaceStability.Unstable
264  public static class Servlet extends HttpServlet {
265    private static final long serialVersionUID = 1L;
266
267    @Override
268    public void doGet(HttpServletRequest request, HttpServletResponse response)
269            throws ServletException, IOException {
270      // Do the authorization
271      if (!HttpServer.hasAdministratorAccess(getServletContext(), request,
272          response)) {
273        return;
274      }
275      response.setContentType("text/html");
276      PrintWriter out;
277      try {
278        String headerPath = "header.jsp?pageTitle=Log Level";
279        request.getRequestDispatcher(headerPath).include(request, response);
280        out = response.getWriter();
281      } catch (FileNotFoundException e) {
282        // in case file is not found fall back to old design
283        out = ServletUtil.initHTML(response, "Log Level");
284      }
285      out.println(FORMS);
286
287      String logName = ServletUtil.getParameter(request, "log");
288      String level = ServletUtil.getParameter(request, "level");
289
290      if (logName != null) {
291        out.println("<p>Results:</p>");
292        out.println(MARKER
293            + "Submitted Log Name: <b>" + logName + "</b><br />");
294
295        Logger log = LoggerFactory.getLogger(logName);
296        out.println(MARKER
297            + "Log Class: <b>" + log.getClass().getName() +"</b><br />");
298        if (level != null) {
299          out.println(MARKER + "Submitted Level: <b>" + level + "</b><br />");
300        }
301
302        if (log instanceof Log4JLogger) {
303          process(((Log4JLogger)log).getLogger(), level, out);
304        } else if (log instanceof Jdk14Logger) {
305          process(((Jdk14Logger)log).getLogger(), level, out);
306        } else if (log instanceof Log4jLoggerAdapter) {
307          process(LogManager.getLogger(logName), level, out);
308        } else {
309          out.println("Sorry, " + log.getClass() + " not supported.<br />");
310        }
311      }
312
313      try {
314        String footerPath = "footer.jsp";
315        out.println("</div>");
316        request.getRequestDispatcher(footerPath).include(request, response);
317      } catch (FileNotFoundException e) {
318        out.println(ServletUtil.HTML_TAIL);
319      }
320      out.close();
321    }

344    private static void process(org.apache.log4j.Logger log, String level, PrintWriter out) {
345      if (level != null) {
346        if (!level.equals(org.apache.log4j.Level.toLevel(level).toString())) {
347          out.println(MARKER + "<div class='text-danger'>" + "Bad level : <strong>" + level
348              + "</strong><br />" + "</div>");
349        } else {
350          log.setLevel(org.apache.log4j.Level.toLevel(level));
351          out.println(MARKER + "<div class='text-success'>" + "Setting Level to <strong>" + level
352              + "</strong> ...<br />" + "</div>");
353        }
354      }
355      out.println(MARKER
356          + "Effective level: <b>" + log.getEffectiveLevel() + "</b><br />");
357    }
358
359    private static void process(java.util.logging.Logger log, String level,
360        PrintWriter out) {
361      if (level != null) {
362        log.setLevel(java.util.logging.Level.parse(level));
363        out.println(MARKER + "Setting Level to " + level + " ...<br />");
364      }
365
366      java.util.logging.Level lev;
367
368      while ((lev = log.getLevel()) == null) {
369        log = log.getParent();
370      }
371
372      out.println(MARKER + "Effective level: <b>" + lev + "</b><br />");
373    }
374  }
