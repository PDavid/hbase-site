041/**
042 * Utilities for adding/removing constraints from a table.
043 * <p/>
044 * Since {@link TableDescriptor} is immutable now, you should use {@link TableDescriptorBuilder}.
045 * And when disabling or removing constraints, you could use
046 * {@link TableDescriptorBuilder#newBuilder(TableDescriptor)} to clone the old
047 * {@link TableDescriptor} and then pass it the below methods.
048 */
049@InterfaceAudience.Public
050public final class Constraints {
051  private static final int DEFAULT_PRIORITY = -1;
052
053  private Constraints() {
054  }
055
056  private static final Logger LOG = LoggerFactory.getLogger(Constraints.class);
057  private static final String CONSTRAINT_HTD_KEY_PREFIX = "constraint $";
058  private static final Pattern CONSTRAINT_HTD_ATTR_KEY_PATTERN =
059    Pattern.compile(CONSTRAINT_HTD_KEY_PREFIX, Pattern.LITERAL);
060
061  // configuration key for if the constraint is enabled
062  private static final String ENABLED_KEY = "_ENABLED";
063  // configuration key for the priority
064  private static final String PRIORITY_KEY = "_PRIORITY";
065
066  // smallest priority a constraiNt can have
067  private static final long MIN_PRIORITY = 0L;
068  // ensure a priority less than the smallest we could intentionally set
069  private static final long UNSET_PRIORITY = MIN_PRIORITY - 1;
070
071  private static String COUNTER_KEY = "hbase.constraint.counter";
072
073  /**
074   * Enable constraints on a table.
075   * <p/>
076   * Currently, if you attempt to add a constraint to the table, then Constraints will automatically
077   * be turned on.
078   */
079  public static TableDescriptorBuilder enable(TableDescriptorBuilder builder) throws IOException {
080    if (!builder.hasCoprocessor(ConstraintProcessor.class.getName())) {
081      builder.setCoprocessor(ConstraintProcessor.class.getName());
082    }
083    return builder;
084  }
085
086  /**
087   * Turn off processing constraints for a given table, even if constraints have been turned on or
088   * added.
089   */
090  public static TableDescriptorBuilder disable(TableDescriptorBuilder builder) throws IOException {
091    return builder.removeCoprocessor(ConstraintProcessor.class.getName());
092  }
093
094  /**
095   * Remove all {@link Constraint Constraints} that have been added to the table and turn off the
096   * constraint processing.
097   * <p/>
098   * All {@link Configuration Configurations} and their associated {@link Constraint} are removed.
099   */
100  public static TableDescriptorBuilder remove(TableDescriptorBuilder builder) throws IOException {
101    disable(builder);
102    return builder
103      .removeValue((k, v) -> CONSTRAINT_HTD_ATTR_KEY_PATTERN.split(k.toString()).length == 2);
104  }

106  /**
107   * Check to see if the Constraint is currently set.
108   * @param desc {@link TableDescriptor} to check
109   * @param clazz {@link Constraint} class to check for.
110   * @return <tt>true</tt> if the {@link Constraint} is present, even if it is disabled.
111   *         <tt>false</tt> otherwise.
112   */
113  public static boolean has(TableDescriptor desc, Class<? extends Constraint> clazz) {
114    return getKeyValueForClass(desc, clazz) != null;
115  }
116
117  /**
118   * Get the kv {@link Entry} in the descriptor for the specified class
119   * @param desc {@link TableDescriptor} to read
120   * @param clazz To search for
121   * @return The {@link Pair} of {@literal <key, value>} in the table, if that class is present.
122   *         {@code null} otherwise.
123   */
124  private static Pair<String, String> getKeyValueForClass(TableDescriptor desc,
125    Class<? extends Constraint> clazz) {
126    // get the serialized version of the constraint
127    String key = serializeConstraintClass(clazz);
128    String value = desc.getValue(key);
129
130    return value == null ? null : new Pair<>(key, value);
131  }

133  /**
134   * Get the kv {@link Entry} in the descriptor builder for the specified class
135   * @param builder {@link TableDescriptorBuilder} to read
136   * @param clazz To search for
137   * @return The {@link Pair} of {@literal <key, value>} in the table, if that class is present.
138   *         {@code null} otherwise.
139   */
140  private static Pair<String, String> getKeyValueForClass(TableDescriptorBuilder builder,
141    Class<? extends Constraint> clazz) {
142    // get the serialized version of the constraint
143    String key = serializeConstraintClass(clazz);
144    String value = builder.getValue(key);
145
146    return value == null ? null : new Pair<>(key, value);
147  }

149  /**
150   * Add configuration-less constraints to the table.
151   * <p/>
152   * This will overwrite any configuration associated with the previous constraint of the same
153   * class.
154   * <p/>
155   * Each constraint, when added to the table, will have a specific priority, dictating the order in
156   * which the {@link Constraint} will be run. A {@link Constraint} earlier in the list will be run
157   * before those later in the list. The same logic applies between two Constraints over time
158   * (earlier added is run first on the regionserver).
159   * @param builder {@link TableDescriptorBuilder} to add a {@link Constraint}
160   * @param constraints {@link Constraint Constraints} to add. All constraints are considered
161   *          automatically enabled on add
162   * @throws IOException If constraint could not be serialized/added to table
163   */
164  @SafeVarargs
165  public static TableDescriptorBuilder add(TableDescriptorBuilder builder,
166    Class<? extends Constraint>... constraints) throws IOException {
167    // make sure constraints are enabled
168    enable(builder);
169    long priority = getNextPriority(builder);
170
171    // store each constraint
172    for (Class<? extends Constraint> clazz : constraints) {
173      addConstraint(builder, clazz, null, priority++);
174    }
175    return updateLatestPriority(builder, priority);
176  }
177
178  /**
179   * Add constraints and their associated configurations to the table.
180   * <p>
181   * Adding the same constraint class twice will overwrite the first constraint's configuration
182   * <p>
183   * Each constraint, when added to the table, will have a specific priority, dictating the order in
184   * which the {@link Constraint} will be run. A {@link Constraint} earlier in the list will be run
185   * before those later in the list. The same logic applies between two Constraints over time
186   * (earlier added is run first on the regionserver).
187   * @param builder {@link TableDescriptorBuilder} to add a {@link Constraint}
188   * @param constraints {@link Pair} of a {@link Constraint} and its associated
189   *          {@link Configuration}. The Constraint will be configured on load with the specified
190   *          configuration.All constraints are considered automatically enabled on add
191   * @throws IOException if any constraint could not be deserialized. Assumes if 1 constraint is not
192   *           loaded properly, something has gone terribly wrong and that all constraints need to
193   *           be enforced.
194   */
195  @SafeVarargs
196  public static TableDescriptorBuilder add(TableDescriptorBuilder builder,
197    Pair<Class<? extends Constraint>, Configuration>... constraints) throws IOException {
198    enable(builder);
199    long priority = getNextPriority(builder);
200    for (Pair<Class<? extends Constraint>, Configuration> pair : constraints) {
201      addConstraint(builder, pair.getFirst(), pair.getSecond(), priority++);
202    }
203    return updateLatestPriority(builder, priority);
204  }
205
206  /**
207   * Add a {@link Constraint} to the table with the given configuration
208   * <p/>
209   * Each constraint, when added to the table, will have a specific priority, dictating the order in
210   * which the {@link Constraint} will be run. A {@link Constraint} added will run on the
211   * regionserver before those added to the {@link TableDescriptorBuilder} later.
212   * @param builder {@link TableDescriptorBuilder} to add a {@link Constraint}
213   * @param constraint to be added
214   * @param conf configuration associated with the constraint
215   * @throws IOException if any constraint could not be deserialized. Assumes if 1 constraint is not
216   *           loaded properly, something has gone terribly wrong and that all constraints need to
217   *           be enforced.
218   */
219  public static TableDescriptorBuilder add(TableDescriptorBuilder builder,
220    Class<? extends Constraint> constraint, Configuration conf) throws IOException {
221    enable(builder);
222    long priority = getNextPriority(builder);
223    addConstraint(builder, constraint, conf, priority++);
224
225    return updateLatestPriority(builder, priority);
226  }
227
228  /**
229   * Write the raw constraint and configuration to the descriptor.
230   * <p/>
231   * This method takes care of creating a new configuration based on the passed in configuration and
232   * then updating that with enabled and priority of the constraint.
233   * <p/>
234   * When a constraint is added, it is automatically enabled.
235   */
236  private static TableDescriptorBuilder addConstraint(TableDescriptorBuilder builder,
237    Class<? extends Constraint> clazz, Configuration conf, long priority) throws IOException {
238    return writeConstraint(builder, serializeConstraintClass(clazz),
239      configure(conf, true, priority));
240  }
241
242  /**
243   * Setup the configuration for a constraint as to whether it is enabled and its priority
244   * @param conf on which to base the new configuration
245   * @param enabled <tt>true</tt> if it should be run
246   * @param priority relative to other constraints
247   * @return a new configuration, storable in the {@link TableDescriptor}
248   */
249  private static Configuration configure(Configuration conf, boolean enabled, long priority) {
250    // create the configuration to actually be stored
251    // clone if possible, but otherwise just create an empty configuration
252    Configuration toWrite = conf == null ? new Configuration() : new Configuration(conf);
253
254    // update internal properties
255    toWrite.setBooleanIfUnset(ENABLED_KEY, enabled);
256
257    // set if unset long
258    if (toWrite.getLong(PRIORITY_KEY, UNSET_PRIORITY) == UNSET_PRIORITY) {
259      toWrite.setLong(PRIORITY_KEY, priority);
260    }
261
262    return toWrite;
263  }
264
265  /**
266   * Just write the class to a String representation of the class as a key for the
267   * {@link TableDescriptor}
268   * @param clazz Constraint class to convert to a {@link TableDescriptor} key
269   * @return key to store in the {@link TableDescriptor}
270   */
271  private static String serializeConstraintClass(Class<? extends Constraint> clazz) {
272    String constraintClazz = clazz.getName();
273    return CONSTRAINT_HTD_KEY_PREFIX + constraintClazz;
274  }
275
276  /**
277   * Write the given key and associated configuration to the {@link TableDescriptorBuilder}.
278   */
279  private static TableDescriptorBuilder writeConstraint(TableDescriptorBuilder builder, String key,
280    Configuration conf) throws IOException {
281    // store the key and conf in the descriptor
282    return builder.setValue(key, serializeConfiguration(conf));
283  }
284
285  /**
286   * Write the configuration to a String
287   * @param conf to write
288   * @return String representation of that configuration
289   */
290  private static String serializeConfiguration(Configuration conf) throws IOException {
291    // write the configuration out to the data stream
292    ByteArrayOutputStream bos = new ByteArrayOutputStream();
293    DataOutputStream dos = new DataOutputStream(bos);
294    conf.writeXml(dos);
295    dos.flush();
296    byte[] data = bos.toByteArray();
297    return Bytes.toString(data);
298  }
299
300  /**
301   * Read the {@link Configuration} stored in the byte stream.
302   * @param bytes to read from
303   * @return A valid configuration
304   */
305  private static Configuration readConfiguration(byte[] bytes) throws IOException {
306    ByteArrayInputStream is = new ByteArrayInputStream(bytes);
307    Configuration conf = new Configuration(false);
308    conf.addResource(is);
309    return conf;
310  }
311
312  /**
313   * Read in the configuration from the String encoded configuration
314   * @param bytes to read from
315   * @return A valid configuration
316   * @throws IOException if the configuration could not be read
317   */
318  private static Configuration readConfiguration(String bytes) throws IOException {
319    return readConfiguration(Bytes.toBytes(bytes));
320  }
321
322  private static long getNextPriority(TableDescriptorBuilder builder) {
323    String value = builder.getValue(COUNTER_KEY);
324
325    long priority;
326    // get the current priority
327    if (value == null) {
328      priority = MIN_PRIORITY;
329    } else {
330      priority = Long.parseLong(value) + 1;
331    }
332
333    return priority;
334  }
335
336  private static TableDescriptorBuilder updateLatestPriority(TableDescriptorBuilder builder,
337    long priority) {
338    // update the max priority
339    return builder.setValue(COUNTER_KEY, Long.toString(priority));
340  }

342  /**
343   * Update the configuration for the {@link Constraint}; does not change the order in which the
344   * constraint is run.
345   * @param builder {@link TableDescriptorBuilder} to update
346   * @param clazz {@link Constraint} to update
347   * @param configuration to update the {@link Constraint} with.
348   * @throws IOException if the Constraint was not stored correctly
349   * @throws IllegalArgumentException if the Constraint was not present on this table.
350   */
351  public static TableDescriptorBuilder setConfiguration(TableDescriptorBuilder builder,
352    Class<? extends Constraint> clazz, Configuration configuration)
353    throws IOException, IllegalArgumentException {
354    // get the entry for this class
355    Pair<String, String> e = getKeyValueForClass(builder, clazz);
356
357    if (e == null) {
358      throw new IllegalArgumentException(
359        "Constraint: " + clazz.getName() + " is not associated with this table.");
360    }
361
362    // clone over the configuration elements
363    Configuration conf = new Configuration(configuration);
364
365    // read in the previous info about the constraint
366    Configuration internal = readConfiguration(e.getSecond());
367
368    // update the fields based on the previous settings
369    conf.setIfUnset(ENABLED_KEY, internal.get(ENABLED_KEY));
370    conf.setIfUnset(PRIORITY_KEY, internal.get(PRIORITY_KEY));
371
372    // update the current value
373    return writeConstraint(builder, e.getFirst(), conf);
374  }

378   * @param builder {@link TableDescriptorBuilder} to modify

384    return builder.removeValue(key);

399  /**
400   * Disable the given {@link Constraint}. Retains all the information (e.g. Configuration) for the
401   * {@link Constraint}, but it just doesn't load the {@link Constraint} on the table.
402   * @param builder {@link TableDescriptorBuilder} to modify
403   * @param clazz {@link Constraint} to disable.
404   * @throws IOException if the constraint cannot be found
405   */
406  public static void disableConstraint(TableDescriptorBuilder builder,
407    Class<? extends Constraint> clazz) throws IOException {
408    changeConstraintEnabled(builder, clazz, false);
409  }

411  /**
412   * Change the whether the constraint (if it is already present) is enabled or disabled.
413   */
414  private static TableDescriptorBuilder changeConstraintEnabled(TableDescriptorBuilder builder,
415    Class<? extends Constraint> clazz, boolean enabled) throws IOException {
416    // get the original constraint
417    Pair<String, String> entry = getKeyValueForClass(builder, clazz);
418    if (entry == null) {
419      throw new IllegalArgumentException("Constraint: " + clazz.getName() +
420        " is not associated with this table. You can't enable it!");
421    }
422
423    // create a new configuration from that conf
424    Configuration conf = readConfiguration(entry.getSecond());
425
426    // set that it is enabled
427    conf.setBoolean(ENABLED_KEY, enabled);
428
429    // write it back out
430    return writeConstraint(builder, entry.getFirst(), conf);
431  }

433  /**
434   * Check to see if the given constraint is enabled.
435   * @param desc {@link TableDescriptor} to check.
436   * @param clazz {@link Constraint} to check for
437   * @return <tt>true</tt> if the {@link Constraint} is present and enabled. <tt>false</tt>
438   *         otherwise.
439   * @throws IOException If the constraint has improperly stored in the table
440   */
441  public static boolean enabled(TableDescriptor desc, Class<? extends Constraint> clazz)
442    throws IOException {
443    // get the kv
444    Pair<String, String> entry = getKeyValueForClass(desc, clazz);
445    // its not enabled so just return false. In fact, its not even present!
446    if (entry == null) {
447      return false;
448    }
449
450    // get the info about the constraint
451    Configuration conf = readConfiguration(entry.getSecond());
452
453    return conf.getBoolean(ENABLED_KEY, false);
454  }

456  /**
457   * Get the constraints stored in the table descriptor
458   * @param desc To read from
459   * @param classloader To use when loading classes. If a special classloader is used on a region,
460   *          for instance, then that should be the classloader used to load the constraints. This
461   *          could also apply to unit-testing situation, where want to ensure that class is
462   *          reloaded or not.
463   * @return List of configured {@link Constraint Constraints}
464   * @throws IOException if any part of reading/arguments fails
465   */
466  static List<? extends Constraint> getConstraints(TableDescriptor desc, ClassLoader classloader)
467    throws IOException {
468    List<Constraint> constraints = new ArrayList<>();
469    // loop through all the key, values looking for constraints
470    for (Map.Entry<Bytes, Bytes> e : desc.getValues().entrySet()) {
471      // read out the constraint
472      String key = Bytes.toString(e.getKey().get()).trim();
473      String[] className = CONSTRAINT_HTD_ATTR_KEY_PATTERN.split(key);
474      if (className.length == 2) {
475        key = className[1];
476        if (LOG.isDebugEnabled()) {
477          LOG.debug("Loading constraint:" + key);
478        }
479
480        // read in the rest of the constraint
481        Configuration conf;
482        try {
483          conf = readConfiguration(e.getValue().get());
484        } catch (IOException e1) {
485          // long that we don't have a valid configuration stored, and move on.
486          LOG.warn("Corrupted configuration found for key:" + key + ",  skipping it.");
487          continue;
488        }
489        // if it is not enabled, skip it
490        if (!conf.getBoolean(ENABLED_KEY, false)) {
491          LOG.debug("Constraint: {} is DISABLED - skipping it", key);
492          // go to the next constraint
493          continue;
494        }
495
496        try {
497          // add the constraint, now that we expect it to be valid.
498          Class<? extends Constraint> clazz =
499            classloader.loadClass(key).asSubclass(Constraint.class);
500          Constraint constraint = clazz.getDeclaredConstructor().newInstance();
501          constraint.setConf(conf);
502          constraints.add(constraint);
503        } catch (InvocationTargetException | NoSuchMethodException | ClassNotFoundException
504          | InstantiationException | IllegalAccessException e1) {
505          throw new IOException(e1);
506        }
507      }
508    }
509    // sort them, based on the priorities
510    Collections.sort(constraints, constraintComparator);
511    return constraints;
512  }
513
514  private static final Comparator<Constraint> constraintComparator = new Comparator<Constraint>() {
515    @Override
516    public int compare(Constraint c1, Constraint c2) {
517      // compare the priorities of the constraints stored in their configuration
518      return Long.compare(c1.getConf().getLong(PRIORITY_KEY, DEFAULT_PRIORITY),
519        c2.getConf().getLong(PRIORITY_KEY, DEFAULT_PRIORITY));
520    }
521  };
