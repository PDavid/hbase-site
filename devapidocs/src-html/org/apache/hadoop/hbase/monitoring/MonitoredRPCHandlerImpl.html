019package org.apache.hadoop.hbase.monitoring;
020
021import java.util.ArrayList;
022import java.util.HashMap;
023import java.util.Map;
024
025import org.apache.hadoop.hbase.client.Operation;
026import org.apache.hadoop.hbase.util.Bytes;
027import org.apache.hadoop.hbase.util.EnvironmentEdgeManager;
028
029import org.apache.yetus.audience.InterfaceAudience;
030import org.apache.hbase.thirdparty.com.google.protobuf.Message;
031
032/**
033 * A MonitoredTask implementation designed for use with RPC Handlers 
034 * handling frequent, short duration tasks. String concatenations and object 
035 * allocations are avoided in methods that will be hit by every RPC call.
036 */
037@InterfaceAudience.Private
038public class MonitoredRPCHandlerImpl extends MonitoredTaskImpl 
039  implements MonitoredRPCHandler {
040  private String clientAddress;
041  private int remotePort;
042  private long rpcQueueTime;
043  private long rpcStartTime;
044  private String methodName = "";
045  private Object [] params = {};
046  private Message packet;
047  private boolean snapshot = false;
048  private Map<String, Object> callInfoMap = new HashMap<>();
049
050  public MonitoredRPCHandlerImpl() {
051    super();
052    // in this implementation, WAITING indicates that the handler is not 
053    // actively servicing an RPC call.
054    setState(State.WAITING);
055  }
056
057  @Override
058  public synchronized MonitoredRPCHandlerImpl clone() {
059    MonitoredRPCHandlerImpl clone = (MonitoredRPCHandlerImpl) super.clone();
060    clone.callInfoMap = generateCallInfoMap();
061    clone.snapshot = true;
062    return clone;
063  }
064
065  /**
066   * Gets the status of this handler; if it is currently servicing an RPC,
067   * this status will include the RPC information.
068   * @return a String describing the current status.
069   */
070  @Override
071  public String getStatus() {
072    if (getState() != State.RUNNING) {
073      return super.getStatus();
074    }
075    return super.getStatus() + " from " + getClient() + ": " + getRPC();
076  }
077
078  /**
079   * Accesses the queue time for the currently running RPC on the 
080   * monitored Handler.
081   * @return the queue timestamp or -1 if there is no RPC currently running.
082   */
083  @Override
084  public long getRPCQueueTime() {
085    if (getState() != State.RUNNING) {
086      return -1;
087    }
088    return rpcQueueTime;
089  }
090
091  /**
092   * Accesses the start time for the currently running RPC on the 
093   * monitored Handler.
094   * @return the start timestamp or -1 if there is no RPC currently running.
095   */
096  @Override
097  public long getRPCStartTime() {
098    if (getState() != State.RUNNING) {
099      return -1;
100    }
101    return rpcStartTime;
102  }
103
104  /**
105   * Produces a string representation of the method currently being serviced
106   * by this Handler.
107   * @return a string representing the method call without parameters
108   */
109  @Override
110  public synchronized String getRPC() {
111    return getRPC(false);
112  }
113
114  /**
115   * Produces a string representation of the method currently being serviced
116   * by this Handler.
117   * @param withParams toggle inclusion of parameters in the RPC String
118   * @return A human-readable string representation of the method call.
119   */
120  @Override
121  public synchronized String getRPC(boolean withParams) {
122    if (getState() != State.RUNNING) {
123      // no RPC is currently running
124      return "";
125    }
126    StringBuilder buffer = new StringBuilder(256);
127    buffer.append(methodName);
128    if (withParams) {
129      buffer.append("(");
130      for (int i = 0; i < params.length; i++) {
131        if (i != 0)
132          buffer.append(", ");
133        buffer.append(params[i]);
134      }
135      buffer.append(")");
136    }
137    return buffer.toString();
138  }
139
140  /**
141   * Produces a string representation of the method currently being serviced
142   * by this Handler.
143   * @return A human-readable string representation of the method call.
144   */
145  @Override
146  public long getRPCPacketLength() {
147    if (getState() != State.RUNNING || packet == null) {
148      // no RPC is currently running, or we don't have an RPC's packet info
149      return -1L;
150    }
151    return packet.getSerializedSize();
152  }
153
154  /**
155   * If an RPC call is currently running, produces a String representation of 
156   * the connection from which it was received.
157   * @return A human-readable string representation of the address and port 
158   *  of the client.
159   */
160  @Override
161  public String getClient() {
162    return clientAddress + ":" + remotePort;
163  }
164
165  /**
166   * Indicates to the client whether this task is monitoring a currently active 
167   * RPC call.
168   * @return true if the monitored handler is currently servicing an RPC call.
169   */
170  @Override
171  public boolean isRPCRunning() {
172    return getState() == State.RUNNING;
173  }
174
175  /**
176   * Indicates to the client whether this task is monitoring a currently active 
177   * RPC call to a database command. (as defined by 
178   * o.a.h.h.client.Operation)
179   * @return true if the monitored handler is currently servicing an RPC call
180   * to a database command.
181   */
182  @Override
183  public synchronized boolean isOperationRunning() {
184    if(!isRPCRunning()) {
185      return false;
186    }
187    for(Object param : params) {
188      if (param instanceof Operation) {
189        return true;
190      }
191    }
192    return false;
193  }
194
195  /**
196   * Tells this instance that it is monitoring a new RPC call.
197   * @param methodName The name of the method that will be called by the RPC.
198   * @param params The parameters that will be passed to the indicated method.
199   */
200  @Override
201  public synchronized void setRPC(String methodName, Object [] params, 
202      long queueTime) {
203    this.methodName = methodName;
204    this.params = params;
205    long now = EnvironmentEdgeManager.currentTime();
206    this.rpcStartTime = now;
207    setWarnTime(now);
208    this.rpcQueueTime = queueTime;
209    this.state = State.RUNNING;
210  }
211
212  /**
213   * Gives this instance a reference to the protobuf received by the RPC, so 
214   * that it can later compute its size if asked for it.
215   * @param param The protobuf received by the RPC for this call
216   */
217  @Override
218  public void setRPCPacket(Message param) {
219    this.packet = param;
220  }
221
222  /**
223   * Registers current handler client details.
224   * @param clientAddress the address of the current client
225   * @param remotePort the port from which the client connected
226   */
227  @Override
228  public void setConnection(String clientAddress, int remotePort) {
229    this.clientAddress = clientAddress;
230    this.remotePort = remotePort;
231  }
232
233  @Override
234  public synchronized void markComplete(String status) {
235    super.markComplete(status);
236    this.params = null;
237    this.packet = null;
238  }
239
240  @Override
241  public synchronized Map<String, Object> toMap() {
242    return this.snapshot ? this.callInfoMap : generateCallInfoMap();
243  }
244
245  private Map<String, Object> generateCallInfoMap() {
246    // only include RPC info if the Handler is actively servicing an RPC call
247    Map<String, Object> map = super.toMap();
248    if (getState() != State.RUNNING) {
249      return map;
250    }
251    Map<String, Object> rpcJSON = new HashMap<>();
252    ArrayList paramList = new ArrayList();
253    map.put("rpcCall", rpcJSON);
254    rpcJSON.put("queuetimems", getRPCQueueTime());
255    rpcJSON.put("starttimems", getRPCStartTime());
256    rpcJSON.put("clientaddress", clientAddress);
257    rpcJSON.put("remoteport", remotePort);
258    rpcJSON.put("packetlength", getRPCPacketLength());
259    rpcJSON.put("method", methodName);
260    rpcJSON.put("params", paramList);
261    for(Object param : params) {
262      if(param instanceof byte []) {
263        paramList.add(Bytes.toStringBinary((byte []) param));
264      } else if (param instanceof Operation) {
265        paramList.add(((Operation) param).toMap());
266      } else {
267        paramList.add(param.toString());
268      }
269    }
270    return map;
271  }
272
273  @Override
274  public String toString() {
275    if (getState() != State.RUNNING) {
276      return super.toString();
277    }
278    return super.toString()
279        + ", queuetimems=" + getRPCQueueTime()
280        + ", starttimems=" + getRPCStartTime()
281        + ", clientaddress=" + clientAddress
282        + ", remoteport=" + remotePort
283        + ", packetlength=" + getRPCPacketLength()
284        + ", rpcMethod=" + getRPC();
285  }
