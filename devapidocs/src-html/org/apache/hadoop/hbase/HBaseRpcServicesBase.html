118  protected HBaseRpcServicesBase(S server, String processName) throws IOException {
119    this.server = server;
120    Configuration conf = server.getConfiguration();
121    final RpcSchedulerFactory rpcSchedulerFactory;
122    try {
123      rpcSchedulerFactory = getRpcSchedulerFactoryClass(conf).asSubclass(RpcSchedulerFactory.class)
124        .getDeclaredConstructor().newInstance();
125    } catch (NoSuchMethodException | InvocationTargetException | InstantiationException
126      | IllegalAccessException e) {
127      throw new IllegalArgumentException(e);
128    }
129    String hostname = DNS.getHostname(conf, getDNSServerType());
130    int port = conf.getInt(getPortConfigName(), getDefaultPort());
131    // Creation of a HSA will force a resolve.
132    final InetSocketAddress initialIsa = new InetSocketAddress(hostname, port);
133    final InetSocketAddress bindAddress = new InetSocketAddress(getHostname(conf, hostname), port);
134    if (initialIsa.getAddress() == null) {
135      throw new IllegalArgumentException("Failed resolve of " + initialIsa);
136    }
137    priority = createPriority();
138    // Using Address means we don't get the IP too. Shorten it more even to just the host name
139    // w/o the domain.
140    final String name = processName + "/" +
141      Address.fromParts(initialIsa.getHostName(), initialIsa.getPort()).toStringWithoutDomain();
142    server.setName(name);
143    // Set how many times to retry talking to another server over Connection.
144    ConnectionUtils.setServerSideHConnectionRetriesConfig(conf, name, LOG);
145    boolean reservoirEnabled =
146      conf.getBoolean(ByteBuffAllocator.ALLOCATOR_POOL_ENABLED_KEY, defaultReservoirEnabled());
147    try {
148      // use final bindAddress for this server.
149      rpcServer = RpcServerFactory.createRpcServer(server, name, getServices(), bindAddress, conf,
150        rpcSchedulerFactory.create(conf, this, server), reservoirEnabled);
151    } catch (BindException be) {
152      throw new IOException(be.getMessage() + ". To switch ports use the '" + getPortConfigName() +
153        "' configuration property.", be.getCause() != null ? be.getCause() : be);
154    }
155    final InetSocketAddress address = rpcServer.getListenerAddress();
156    if (address == null) {
157      throw new IOException("Listener channel is closed");
158    }
159    // Set our address, however we need the final port that was given to rpcServer
160    isa = new InetSocketAddress(initialIsa.getHostName(), address.getPort());
161    rpcServer.setErrorHandler(this);
162  }

180  protected final void internalStart(ZKWatcher zkWatcher) {
181    if (AccessChecker.isAuthorizationSupported(getConfiguration())) {
182      accessChecker = new AccessChecker(getConfiguration());
183    } else {
184      accessChecker = new NoopAccessChecker(getConfiguration());
185    }
186    zkPermissionWatcher =
187      new ZKPermissionWatcher(zkWatcher, accessChecker.getAuthManager(), getConfiguration());
188    try {
189      zkPermissionWatcher.start();
190    } catch (KeeperException e) {
191      LOG.error("ZooKeeper permission watcher initialization failed", e);
192    }
193    rpcServer.start();
194  }
195
196  protected final void requirePermission(String request, Permission.Action perm)
197    throws IOException {
198    if (accessChecker != null) {
199      accessChecker.requirePermission(RpcServer.getRequestUser().orElse(null), request, null, perm);
200    }
201  }

277  @Override
278  public GetMastersResponse getMasters(RpcController controller, GetMastersRequest request)
279    throws ServiceException {
280    GetMastersResponse.Builder builder = GetMastersResponse.newBuilder();
281    server.getActiveMaster()
282      .ifPresent(activeMaster -> builder.addMasterServers(GetMastersResponseEntry.newBuilder()
283        .setServerName(ProtobufUtil.toServerName(activeMaster)).setIsActive(true)));
284    server.getBackupMasters()
285      .forEach(backupMaster -> builder.addMasterServers(GetMastersResponseEntry.newBuilder()
286        .setServerName(ProtobufUtil.toServerName(backupMaster)).setIsActive(false)));
287    return builder.build();
288  }

299  @Override
300  public final GetBootstrapNodesResponse getBootstrapNodes(RpcController controller,
301    GetBootstrapNodesRequest request) throws ServiceException {
302    List<ServerName> bootstrapNodes = new ArrayList<>(server.getRegionServers());
303    Collections.shuffle(bootstrapNodes, ThreadLocalRandom.current());
304    int maxNodeCount = server.getConfiguration().getInt(CLIENT_BOOTSTRAP_NODE_LIMIT,
305      DEFAULT_CLIENT_BOOTSTRAP_NODE_LIMIT);
306    GetBootstrapNodesResponse.Builder builder = GetBootstrapNodesResponse.newBuilder();
307    bootstrapNodes.stream().limit(maxNodeCount).map(ProtobufUtil::toServerName)
308      .forEach(builder::addServerName);
309    return builder.build();
310  }

324  @Override
325  @QosPriority(priority = HConstants.ADMIN_QOS)
326  public ClearSlowLogResponses clearSlowLogsResponses(final RpcController controller,
327    final ClearSlowLogResponseRequest request) throws ServiceException {
328    try {
329      requirePermission("clearSlowLogsResponses", Permission.Action.ADMIN);
330    } catch (IOException e) {
331      throw new ServiceException(e);
332    }
333    final NamedQueueRecorder namedQueueRecorder = this.server.getNamedQueueRecorder();
334    boolean slowLogsCleaned = Optional.ofNullable(namedQueueRecorder)
335      .map(
336        queueRecorder -> queueRecorder.clearNamedQueue(NamedQueuePayload.NamedQueueEvent.SLOW_LOG))
337      .orElse(false);
338    ClearSlowLogResponses clearSlowLogResponses =
339      ClearSlowLogResponses.newBuilder().setIsCleaned(slowLogsCleaned).build();
340    return clearSlowLogResponses;
341  }
342
343  private List<SlowLogPayload> getSlowLogPayloads(SlowLogResponseRequest request,
344    NamedQueueRecorder namedQueueRecorder) {
345    if (namedQueueRecorder == null) {
346      return Collections.emptyList();
347    }
348    List<SlowLogPayload> slowLogPayloads;
349    NamedQueueGetRequest namedQueueGetRequest = new NamedQueueGetRequest();
350    namedQueueGetRequest.setNamedQueueEvent(RpcLogDetails.SLOW_LOG_EVENT);
351    namedQueueGetRequest.setSlowLogResponseRequest(request);
352    NamedQueueGetResponse namedQueueGetResponse =
353      namedQueueRecorder.getNamedQueueRecords(namedQueueGetRequest);
354    slowLogPayloads = namedQueueGetResponse != null ? namedQueueGetResponse.getSlowLogPayloads() :
355      Collections.emptyList();
356    return slowLogPayloads;
357  }

359  @Override
360  @QosPriority(priority = HConstants.ADMIN_QOS)
361  public HBaseProtos.LogEntry getLogEntries(RpcController controller,
362    HBaseProtos.LogRequest request) throws ServiceException {
363    try {
364      final String logClassName = request.getLogClassName();
365      Class<?> logClass = Class.forName(logClassName).asSubclass(Message.class);
366      Method method = logClass.getMethod("parseFrom", ByteString.class);
367      if (logClassName.contains("SlowLogResponseRequest")) {
368        SlowLogResponseRequest slowLogResponseRequest =
369          (SlowLogResponseRequest) method.invoke(null, request.getLogMessage());
370        final NamedQueueRecorder namedQueueRecorder = this.server.getNamedQueueRecorder();
371        final List<SlowLogPayload> slowLogPayloads =
372          getSlowLogPayloads(slowLogResponseRequest, namedQueueRecorder);
373        SlowLogResponses slowLogResponses =
374          SlowLogResponses.newBuilder().addAllSlowLogPayloads(slowLogPayloads).build();
375        return HBaseProtos.LogEntry.newBuilder()
376          .setLogClassName(slowLogResponses.getClass().getName())
377          .setLogMessage(slowLogResponses.toByteString()).build();
378      }
379    } catch (ClassNotFoundException | NoSuchMethodException | IllegalAccessException
380      | InvocationTargetException e) {
381      LOG.error("Error while retrieving log entries.", e);
382      throw new ServiceException(e);
383    }
384    throw new ServiceException("Invalid request params");
385  }
