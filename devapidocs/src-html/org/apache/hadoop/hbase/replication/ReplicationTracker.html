019package org.apache.hadoop.hbase.replication;
020
021import java.io.IOException;
022import java.util.Set;
023import org.apache.hadoop.hbase.ServerName;
024import org.apache.yetus.audience.InterfaceAudience;
025
026/**
027 * This is the interface for a Replication Tracker.
028 * <p/>
029 * A replication tracker provides the facility to subscribe and track events that reflect a change
030 * in replication state. These events are used by the ReplicationSourceManager to coordinate
031 * replication tasks such as addition/deletion of queues and queue failover. These events are
032 * defined in the ReplicationListener interface. If a class would like to listen to replication
033 * events it must implement the ReplicationListener interface and register itself with a Replication
034 * Tracker.
035 */
036@InterfaceAudience.Private
037public interface ReplicationTracker {
038
039  /**
040   * Register a replication listener to receive replication events.
041   * @param listener the listener to register
042   */
043  void registerListener(ReplicationListener listener);
044
045  /**
046   * Remove a replication listener
047   * @param listener the listener to remove
048   */
049  void removeListener(ReplicationListener listener);
050
051  /**
052   * In this method, you need to load the newest list of region server list and return it, and all
053   * later changes to the region server list must be passed to the listeners.
054   * <p/>
055   * This is very important for us to not miss a region server crash.
056   * <p/>
057   * Notice that this method can only be called once.
058   * @return Set of region servers.
059   */
060  Set<ServerName> loadLiveRegionServersAndInitializeListeners() throws IOException;
061}
