063/**
064 * A {@link org.apache.hadoop.hbase.replication.ReplicationEndpoint} endpoint which receives the WAL
065 * edits from the WAL, and sends the edits to replicas of regions.
066 */
067@InterfaceAudience.Private
068public class RegionReplicaReplicationEndpoint extends HBaseReplicationEndpoint {
069
070  private static final Logger LOG = LoggerFactory.getLogger(RegionReplicaReplicationEndpoint.class);
071
072  // Can be configured differently than hbase.client.retries.number
073  private static String CLIENT_RETRIES_NUMBER =
074    "hbase.region.replica.replication.client.retries.number";

076  private Configuration conf;
077  private AsyncClusterConnection connection;
078  private TableDescriptors tableDescriptors;
079
080  private int numRetries;
081
082  private long operationTimeoutNs;

084  private LoadingCache<TableName, Optional<TableDescriptor>> tableDescriptorCache;
085
086  private Cache<TableName, TableName> disabledTableCache;
087
088  private final RetryCounterFactory retryCounterFactory =
089    new RetryCounterFactory(Integer.MAX_VALUE, 1000, 60000);

091  @Override
092  public void init(Context context) throws IOException {
093    super.init(context);
094    this.conf = context.getConfiguration();
095    this.tableDescriptors = context.getTableDescriptors();
096    int memstoreReplicationEnabledCacheExpiryMs = conf
097      .getInt("hbase.region.replica.replication.cache.memstoreReplicationEnabled.expiryMs", 5000);
098    // A cache for the table "memstore replication enabled" flag.
099    // It has a default expiry of 5 sec. This means that if the table is altered
100    // with a different flag value, we might miss to replicate for that amount of
101    // time. But this cache avoid the slow lookup and parsing of the TableDescriptor.
102    tableDescriptorCache = CacheBuilder.newBuilder()
103      .expireAfterWrite(memstoreReplicationEnabledCacheExpiryMs, TimeUnit.MILLISECONDS)
104      .initialCapacity(10).maximumSize(1000)
105      .build(new CacheLoader<TableName, Optional<TableDescriptor>>() {
106
107        @Override
108        public Optional<TableDescriptor> load(TableName tableName) throws Exception {
109          // check if the table requires memstore replication
110          // some unit-test drop the table, so we should do a bypass check and always replicate.
111          return Optional.ofNullable(tableDescriptors.get(tableName));
112        }
113      });
114    int nonExistentTableCacheExpiryMs =
115      conf.getInt("hbase.region.replica.replication.cache.disabledAndDroppedTables.expiryMs", 5000);
116    // A cache for non existing tables that have a default expiry of 5 sec. This means that if the
117    // table is created again with the same name, we might miss to replicate for that amount of
118    // time. But this cache prevents overloading meta requests for every edit from a deleted file.
119    disabledTableCache = CacheBuilder.newBuilder()
120      .expireAfterWrite(nonExistentTableCacheExpiryMs, TimeUnit.MILLISECONDS).initialCapacity(10)
121      .maximumSize(1000).build();
122    // HRS multiplies client retries by 10 globally for meta operations, but we do not want this.
123    // We are resetting it here because we want default number of retries (35) rather than 10 times
124    // that which makes very long retries for disabled tables etc.
125    int defaultNumRetries = conf.getInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER,
126      HConstants.DEFAULT_HBASE_CLIENT_RETRIES_NUMBER);
127    if (defaultNumRetries > 10) {
128      int mult = conf.getInt(HConstants.HBASE_CLIENT_SERVERSIDE_RETRIES_MULTIPLIER,
129        HConstants.DEFAULT_HBASE_CLIENT_SERVERSIDE_RETRIES_MULTIPLIER);
130      defaultNumRetries = defaultNumRetries / mult; // reset if HRS has multiplied this already
131    }
132    this.numRetries = conf.getInt(CLIENT_RETRIES_NUMBER, defaultNumRetries);
133    // use the regular RPC timeout for replica replication RPC's
134    this.operationTimeoutNs =
135      TimeUnit.MILLISECONDS.toNanos(conf.getInt(HConstants.HBASE_CLIENT_OPERATION_TIMEOUT,
136        HConstants.DEFAULT_HBASE_CLIENT_OPERATION_TIMEOUT));
137    this.connection = context.getServer().getAsyncClusterConnection();
138  }
139
140  /**
141   * returns true if the specified entry must be replicated. We should always replicate meta
142   * operations (e.g. flush) and use the user HTD flag to decide whether or not replicate the
143   * memstore.
144   */
145  private boolean requiresReplication(Optional<TableDescriptor> tableDesc,
146      Entry entry) {
147    // empty edit does not need to be replicated
148    if (entry.getEdit().isEmpty() || !tableDesc.isPresent()) {
149      return false;
150    }
151    // meta edits (e.g. flush) must be always replicated
152    return entry.getEdit().isMetaEdit() || tableDesc.get().hasRegionMemStoreReplication();
153  }

155  private void getRegionLocations(CompletableFuture<RegionLocations> future,
156      TableDescriptor tableDesc, byte[] encodedRegionName, byte[] row, boolean reload) {
157    FutureUtils.addListener(connection.getRegionLocations(tableDesc.getTableName(), row, reload),
158      (locs, e) -> {
159        if (e != null) {
160          future.completeExceptionally(e);
161          return;
162        }
163        // if we are not loading from cache, just return
164        if (reload) {
165          future.complete(locs);
166          return;
167        }
168        // check if the number of region replicas is correct, and also the primary region name
169        // matches.
170        if (locs.size() == tableDesc.getRegionReplication() &&
171          locs.getDefaultRegionLocation() != null &&
172          Bytes.equals(locs.getDefaultRegionLocation().getRegion().getEncodedNameAsBytes(),
173            encodedRegionName)) {
174          future.complete(locs);
175        } else {
176          // reload again as the information in cache maybe stale
177          getRegionLocations(future, tableDesc, encodedRegionName, row, true);
178        }
179      });
180  }

182  private void replicate(CompletableFuture<Long> future, RegionLocations locs,
183      TableDescriptor tableDesc, byte[] encodedRegionName, byte[] row, List<Entry> entries) {
184    if (locs.size() == 1) {
185      LOG.info("Only one location for {}.{}, refresh the location cache only for meta now",
186        tableDesc.getTableName(), Bytes.toString(encodedRegionName));
187
188      // This could happen to meta table. In case of meta table comes with no replica and
189      // later it is changed to multiple replicas. The cached location for meta may only has
190      // the primary region. In this case, it needs to clean up and refresh the cached meta
191      // locations.
192      if (tableDesc.isMetaTable()) {
193        connection.getRegionLocator(tableDesc.getTableName()).clearRegionLocationCache();
194      }
195      future.complete(Long.valueOf(entries.size()));
196      return;
197    }
198    RegionInfo defaultReplica = locs.getDefaultRegionLocation().getRegion();
199    if (!Bytes.equals(defaultReplica.getEncodedNameAsBytes(), encodedRegionName)) {
200      // the region name is not equal, this usually means the region has been split or merged, so
201      // give up replicating as the new region(s) should already have all the data of the parent
202      // region(s).
203      if (LOG.isTraceEnabled()) {
204        LOG.trace(
205          "Skipping {} entries in table {} because located region {} is different than" +
206            " the original region {} from WALEdit",
207          tableDesc.getTableName(), defaultReplica.getEncodedName(),
208          Bytes.toStringBinary(encodedRegionName));
209      }
210      future.complete(Long.valueOf(entries.size()));
211      return;
212    }
213    AtomicReference<Throwable> error = new AtomicReference<>();
214    AtomicInteger remainingTasks = new AtomicInteger(locs.size() - 1);
215    AtomicLong skippedEdits = new AtomicLong(0);
216
217    for (int i = 1, n = locs.size(); i < n; i++) {
218      // Do not use the elements other than the default replica as they may be null. We will fail
219      // earlier if the location for default replica is null.
220      final RegionInfo replica = RegionReplicaUtil.getRegionInfoForReplica(defaultReplica, i);
221      FutureUtils
222        .addListener(connection.replay(tableDesc.getTableName(), replica.getEncodedNameAsBytes(),
223          row, entries, replica.getReplicaId(), numRetries, operationTimeoutNs), (r, e) -> {
224            if (e != null) {
225              LOG.warn("Failed to replicate to {}", replica, e);
226              error.compareAndSet(null, e);
227            } else {
228              AtomicUtils.updateMax(skippedEdits, r.longValue());
229            }
230            if (remainingTasks.decrementAndGet() == 0) {
231              if (error.get() != null) {
232                future.completeExceptionally(error.get());
233              } else {
234                future.complete(skippedEdits.get());
235              }
236            }
237          });
238    }
239  }
240
241  private void logSkipped(TableName tableName, List<Entry> entries, String reason) {
242    if (LOG.isTraceEnabled()) {
243      LOG.trace("Skipping {} entries because table {} is {}", entries.size(), tableName, reason);
244      for (Entry entry : entries) {
245        LOG.trace("Skipping : {}", entry);
246      }
247    }
248  }
249
250  private CompletableFuture<Long> replicate(TableDescriptor tableDesc, byte[] encodedRegionName,
251      List<Entry> entries) {
252    if (disabledTableCache.getIfPresent(tableDesc.getTableName()) != null) {
253      logSkipped(tableDesc.getTableName(), entries, "cached as a disabled table");
254      return CompletableFuture.completedFuture(Long.valueOf(entries.size()));
255    }
256    byte[] row = CellUtil.cloneRow(entries.get(0).getEdit().getCells().get(0));
257    CompletableFuture<RegionLocations> locateFuture = new CompletableFuture<>();
258    getRegionLocations(locateFuture, tableDesc, encodedRegionName, row, false);
259    CompletableFuture<Long> future = new CompletableFuture<>();
260    FutureUtils.addListener(locateFuture, (locs, error) -> {
261      if (error != null) {
262        future.completeExceptionally(error);
263      } else if (locs.getDefaultRegionLocation() == null) {
264        future.completeExceptionally(
265          new HBaseIOException("No location found for default replica of table=" +
266            tableDesc.getTableName() + " row='" + Bytes.toStringBinary(row) + "'"));
267      } else {
268        replicate(future, locs, tableDesc, encodedRegionName, row, entries);
269      }
270    });
271    return future;
272  }

274  @Override
275  public boolean replicate(ReplicateContext replicateContext) {
276    Map<byte[], Pair<TableDescriptor, List<Entry>>> encodedRegionName2Entries =
277      new TreeMap<>(Bytes.BYTES_COMPARATOR);
278    long skippedEdits = 0;
279    RetryCounter retryCounter = retryCounterFactory.create();
280    outer: while (isRunning()) {
281      encodedRegionName2Entries.clear();
282      skippedEdits = 0;
283      for (Entry entry : replicateContext.getEntries()) {
284        Optional<TableDescriptor> tableDesc;
285        try {
286          tableDesc = tableDescriptorCache.get(entry.getKey().getTableName());
287        } catch (ExecutionException e) {
288          LOG.warn("Failed to load table descriptor for {}, attempts={}",
289            entry.getKey().getTableName(), retryCounter.getAttemptTimes(), e.getCause());
290          if (!retryCounter.shouldRetry()) {
291            return false;
292          }
293          try {
294            retryCounter.sleepUntilNextRetry();
295          } catch (InterruptedException e1) {
296            // restore the interrupted state
297            Thread.currentThread().interrupt();
298            return false;
299          }
300          continue outer;
301        }
302        if (!requiresReplication(tableDesc, entry)) {
303          skippedEdits++;
304          continue;
305        }
306        byte[] encodedRegionName = entry.getKey().getEncodedRegionName();
307        encodedRegionName2Entries
308          .computeIfAbsent(encodedRegionName, k -> Pair.newPair(tableDesc.get(), new ArrayList<>()))
309          .getSecond().add(entry);
310      }
311      break;
312    }
313    // send the request to regions
314    retryCounter = retryCounterFactory.create();
315    while (isRunning()) {
316      List<Pair<CompletableFuture<Long>, byte[]>> futureAndEncodedRegionNameList =
317        new ArrayList<Pair<CompletableFuture<Long>, byte[]>>();
318      for (Map.Entry<byte[], Pair<TableDescriptor, List<Entry>>> entry : encodedRegionName2Entries
319        .entrySet()) {
320        CompletableFuture<Long> future =
321          replicate(entry.getValue().getFirst(), entry.getKey(), entry.getValue().getSecond());
322        futureAndEncodedRegionNameList.add(Pair.newPair(future, entry.getKey()));
323      }
324      for (Pair<CompletableFuture<Long>, byte[]> pair : futureAndEncodedRegionNameList) {
325        byte[] encodedRegionName = pair.getSecond();
326        try {
327          skippedEdits += pair.getFirst().get().longValue();
328          encodedRegionName2Entries.remove(encodedRegionName);
329        } catch (InterruptedException e) {
330          // restore the interrupted state
331          Thread.currentThread().interrupt();
332          return false;
333        } catch (ExecutionException e) {
334          Pair<TableDescriptor, List<Entry>> tableAndEntries =
335            encodedRegionName2Entries.get(encodedRegionName);
336          TableName tableName = tableAndEntries.getFirst().getTableName();
337          List<Entry> entries = tableAndEntries.getSecond();
338          Throwable cause = e.getCause();
339          // The table can be disabled or dropped at this time. For disabled tables, we have no
340          // cheap mechanism to detect this case because meta does not contain this information.
341          // ClusterConnection.isTableDisabled() is a zk call which we cannot do for every replay
342          // RPC. So instead we start the replay RPC with retries and check whether the table is
343          // dropped or disabled which might cause SocketTimeoutException, or
344          // RetriesExhaustedException or similar if we get IOE.
345          if (cause instanceof TableNotFoundException) {
346            // add to cache that the table does not exist
347            tableDescriptorCache.put(tableName, Optional.empty());
348            logSkipped(tableName, entries, "dropped");
349            skippedEdits += entries.size();
350            encodedRegionName2Entries.remove(encodedRegionName);
351            continue;
352          }
353          boolean disabled = false;
354          try {
355            disabled = connection.getAdmin().isTableDisabled(tableName).get();
356          } catch (InterruptedException e1) {
357            // restore the interrupted state
358            Thread.currentThread().interrupt();
359            return false;
360          } catch (ExecutionException e1) {
361            LOG.warn("Failed to test whether {} is disabled, assume it is not disabled", tableName,
362              e1.getCause());
363          }
364          if (disabled) {
365            disabledTableCache.put(tableName, tableName);
366            logSkipped(tableName, entries, "disabled");
367            skippedEdits += entries.size();
368            encodedRegionName2Entries.remove(encodedRegionName);
369            continue;
370          }
371          LOG.warn("Failed to replicate {} entries for region {} of table {}", entries.size(),
372            Bytes.toStringBinary(encodedRegionName), tableName);
373        }
374      }
375      // we have done
376      if (encodedRegionName2Entries.isEmpty()) {
377        ctx.getMetrics().incrLogEditsFiltered(skippedEdits);
378        return true;
379      } else {
380        LOG.warn("Failed to replicate all entries, retry={}", retryCounter.getAttemptTimes());
381        if (!retryCounter.shouldRetry()) {
382          return false;
383        }
384        try {
385          retryCounter.sleepUntilNextRetry();
386        } catch (InterruptedException e) {
387          // restore the interrupted state
388          Thread.currentThread().interrupt();
389          return false;
390        }
391      }
392    }
393
394    return false;
395  }
