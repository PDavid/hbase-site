076/**
077 * This class is responsible to manage all the replication sources. There are two classes of
078 * sources:
079 * <ul>
080 * <li>Normal sources are persistent and one per peer cluster</li>
081 * <li>Old sources are recovered from a failed region server and our only goal is to finish
082 * replicating the WAL queue it had</li>
083 * </ul>
084 * <p>
085 * When a region server dies, this class uses a watcher to get notified and it tries to grab a lock
086 * in order to transfer all the queues in a local old source.
087 * <p>
088 * Synchronization specification:
089 * <ul>
090 * <li>No need synchronized on {@link #sources}. {@link #sources} is a ConcurrentHashMap and there
091 * is a Lock for peer id in {@link PeerProcedureHandlerImpl}. So there is no race for peer
092 * operations.</li>
093 * <li>Need synchronized on {@link #walsById}. There are four methods which modify it,
094 * {@link #addPeer(String)}, {@link #removePeer(String)},
095 * {@link #cleanOldLogs(String, boolean, ReplicationSourceInterface)} and {@link #preLogRoll(Path)}.
096 * {@link #walsById} is a ConcurrentHashMap and there is a Lock for peer id in
097 * {@link PeerProcedureHandlerImpl}. So there is no race between {@link #addPeer(String)} and
098 * {@link #removePeer(String)}. {@link #cleanOldLogs(String, boolean, ReplicationSourceInterface)}
099 * is called by {@link ReplicationSourceInterface}. So no race with {@link #addPeer(String)}.
100 * {@link #removePeer(String)} will terminate the {@link ReplicationSourceInterface} firstly, then
101 * remove the wals from {@link #walsById}. So no race with {@link #removePeer(String)}. The only
102 * case need synchronized is {@link #cleanOldLogs(String, boolean, ReplicationSourceInterface)} and
103 * {@link #preLogRoll(Path)}.</li>
104 * <li>No need synchronized on {@link #walsByIdRecoveredQueues}. There are three methods which
105 * modify it, {@link #removePeer(String)} ,
106 * {@link #cleanOldLogs(String, boolean, ReplicationSourceInterface)} and
107 * {@link ReplicationSourceManager.NodeFailoverWorker#run()}.
108 * {@link #cleanOldLogs(String, boolean, ReplicationSourceInterface)} is called by
109 * {@link ReplicationSourceInterface}. {@link #removePeer(String)} will terminate the
110 * {@link ReplicationSourceInterface} firstly, then remove the wals from
111 * {@link #walsByIdRecoveredQueues}. And {@link ReplicationSourceManager.NodeFailoverWorker#run()}
112 * will add the wals to {@link #walsByIdRecoveredQueues} firstly, then start up a
113 * {@link ReplicationSourceInterface}. So there is no race here. For
114 * {@link ReplicationSourceManager.NodeFailoverWorker#run()} and {@link #removePeer(String)}, there
115 * is already synchronized on {@link #oldsources}. So no need synchronized on
116 * {@link #walsByIdRecoveredQueues}.</li>
117 * <li>Need synchronized on {@link #latestPaths} to avoid the new open source miss new log.</li>
118 * <li>Need synchronized on {@link #oldsources} to avoid adding recovered source for the
119 * to-be-removed peer.</li>
120 * </ul>
121 */

173  /**
174   * Creates a replication manager and sets the watch on all the other registered region servers
175   * @param queueStorage the interface for manipulating replication queues
176   * @param replicationPeers
177   * @param replicationTracker
178   * @param conf the configuration to use
179   * @param server the server for this region server
180   * @param fs the file system to use
181   * @param logDir the directory that contains all wal directories of live RSs
182   * @param oldLogDir the directory where old logs are archived
183   * @param clusterId
184   */
185  public ReplicationSourceManager(ReplicationQueueStorage queueStorage,
186      ReplicationPeers replicationPeers, ReplicationTracker replicationTracker, Configuration conf,
187      Server server, FileSystem fs, Path logDir, Path oldLogDir, UUID clusterId,
188      WALFileLengthProvider walFileLengthProvider,
189      SyncReplicationPeerMappingManager syncReplicationPeerMappingManager) throws IOException {
190    this.sources = new ConcurrentHashMap<>();
191    this.queueStorage = queueStorage;
192    this.replicationPeers = replicationPeers;
193    this.replicationTracker = replicationTracker;
194    this.server = server;
195    this.walsById = new ConcurrentHashMap<>();
196    this.walsByIdRecoveredQueues = new ConcurrentHashMap<>();
197    this.oldsources = new ArrayList<>();
198    this.conf = conf;
199    this.fs = fs;
200    this.logDir = logDir;
201    this.oldLogDir = oldLogDir;
202    // 30 seconds
203    this.sleepBeforeFailover = conf.getLong("replication.sleep.before.failover", 30000);
204    this.clusterId = clusterId;
205    this.walFileLengthProvider = walFileLengthProvider;
206    this.syncReplicationPeerMappingManager = syncReplicationPeerMappingManager;
207    this.replicationTracker.registerListener(this);
208    // It's preferable to failover 1 RS at a time, but with good zk servers
209    // more could be processed at the same time.
210    int nbWorkers = conf.getInt("replication.executor.workers", 1);
211    // use a short 100ms sleep since this could be done inline with a RS startup
212    // even if we fail, other region servers can take care of it
213    this.executor = new ThreadPoolExecutor(nbWorkers, nbWorkers, 100,
214        TimeUnit.MILLISECONDS, new LinkedBlockingQueue<>());
215    ThreadFactoryBuilder tfb = new ThreadFactoryBuilder();
216    tfb.setNameFormat("ReplicationExecutor-%d");
217    tfb.setDaemon(true);
218    this.executor.setThreadFactory(tfb.build());
219    this.latestPaths = new HashMap<>();
220    this.replicationForBulkLoadDataEnabled = conf.getBoolean(
221      HConstants.REPLICATION_BULKLOAD_ENABLE_KEY, HConstants.REPLICATION_BULKLOAD_ENABLE_DEFAULT);
222    this.sleepForRetries = this.conf.getLong("replication.source.sync.sleepforretries", 1000);
223    this.maxRetriesMultiplier =
224      this.conf.getInt("replication.source.sync.maxretriesmultiplier", 60);
225  }

227  /**
228   * Adds a normal source per registered peer cluster and tries to process all old region server wal
229   * queues
230   * <p>
231   * The returned future is for adoptAbandonedQueues task.
232   */
233  Future<?> init() throws IOException {
234    for (String id : this.replicationPeers.getAllPeerIds()) {
235      addSource(id);
236      if (replicationForBulkLoadDataEnabled) {
237        // Check if peer exists in hfile-refs queue, if not add it. This can happen in the case
238        // when a peer was added before replication for bulk loaded data was enabled.
239        throwIOExceptionWhenFail(() -> this.queueStorage.addPeerToHFileRefs(id));
240      }
241    }
242    return this.executor.submit(this::adoptAbandonedQueues);
243  }
244
245  private void adoptAbandonedQueues() {
246    List<ServerName> currentReplicators = null;
247    try {
248      currentReplicators = queueStorage.getListOfReplicators();
249    } catch (ReplicationException e) {
250      server.abort("Failed to get all replicators", e);
251      return;
252    }
253    if (currentReplicators == null || currentReplicators.isEmpty()) {
254      return;
255    }
256    List<ServerName> otherRegionServers = replicationTracker.getListOfRegionServers().stream()
257        .map(ServerName::valueOf).collect(Collectors.toList());
258    LOG.info(
259      "Current list of replicators: " + currentReplicators + " other RSs: " + otherRegionServers);
260
261    // Look if there's anything to process after a restart
262    for (ServerName rs : currentReplicators) {
263      if (!otherRegionServers.contains(rs)) {
264        transferQueues(rs);
265      }
266    }
267  }

270   * <ol>
271   * <li>Add peer to replicationPeers</li>
272   * <li>Add the normal source and related replication queue</li>
273   * <li>Add HFile Refs</li>
274   * </ol>
275   * @param peerId the id of replication peer
276   */
277  public void addPeer(String peerId) throws IOException {
278    boolean added = false;
279    try {
280      added = this.replicationPeers.addPeer(peerId);
281    } catch (ReplicationException e) {
282      throw new IOException(e);
283    }
284    if (added) {
285      addSource(peerId);
286      if (replicationForBulkLoadDataEnabled) {
287        throwIOExceptionWhenFail(() -> this.queueStorage.addPeerToHFileRefs(peerId));
288      }
289    }

292  /**
293   * <ol>
294   * <li>Remove peer for replicationPeers</li>
295   * <li>Remove all the recovered sources for the specified id and related replication queues</li>
296   * <li>Remove the normal source and related replication queue</li>
297   * <li>Remove HFile Refs</li>
298   * </ol>
299   * @param peerId the id of the replication peer
300   */
301  public void removePeer(String peerId) {
302    ReplicationPeer peer = replicationPeers.removePeer(peerId);
303    String terminateMessage = "Replication stream was removed by a user";
304    List<ReplicationSourceInterface> oldSourcesToDelete = new ArrayList<>();
305    // synchronized on oldsources to avoid adding recovered source for the to-be-removed peer
306    // see NodeFailoverWorker.run
307    synchronized (this.oldsources) {
308      // First close all the recovered sources for this peer
309      for (ReplicationSourceInterface src : oldsources) {
310        if (peerId.equals(src.getPeerId())) {
311          oldSourcesToDelete.add(src);
312        }
313      }
314      for (ReplicationSourceInterface src : oldSourcesToDelete) {
315        src.terminate(terminateMessage);
316        removeRecoveredSource(src);
317      }
318    }
319    LOG.info(
320      "Number of deleted recovered sources for " + peerId + ": " + oldSourcesToDelete.size());
321    // Now close the normal source for this peer
322    ReplicationSourceInterface srcToRemove = this.sources.get(peerId);
323    if (srcToRemove != null) {
324      srcToRemove.terminate(terminateMessage);
325      removeSource(srcToRemove);
326    } else {
327      // This only happened in unit test TestReplicationSourceManager#testPeerRemovalCleanup
328      // Delete queue from storage and memory and queue id is same with peer id for normal
329      // source
330      deleteQueue(peerId);
331      this.walsById.remove(peerId);
332    }
333    ReplicationPeerConfig peerConfig = peer.getPeerConfig();
334    if (peerConfig.isSyncReplication()) {
335      syncReplicationPeerMappingManager.remove(peerId, peerConfig);
336    }
337    // Remove HFile Refs
338    abortWhenFail(() -> this.queueStorage.removePeerFromHFileRefs(peerId));
339  }

357  /**
358   * Add a normal source for the given peer on this region server. Meanwhile, add new replication
359   * queue to storage. For the newly added peer, we only need to enqueue the latest log of each wal
360   * group and do replication
361   * @param peerId the id of the replication peer
362   * @return the source that was created
363   */
364  @VisibleForTesting
365  ReplicationSourceInterface addSource(String peerId) throws IOException {
366    ReplicationPeer peer = replicationPeers.getPeer(peerId);
367    ReplicationSourceInterface src = createSource(peerId, peer);
368    // synchronized on latestPaths to avoid missing the new log
369    synchronized (this.latestPaths) {
370      this.sources.put(peerId, src);
371      Map<String, NavigableSet<String>> walsByGroup = new HashMap<>();
372      this.walsById.put(peerId, walsByGroup);
373      // Add the latest wal to that source's queue
374      if (!latestPaths.isEmpty()) {
375        for (Map.Entry<String, Path> walPrefixAndPath : latestPaths.entrySet()) {
376          Path walPath = walPrefixAndPath.getValue();
377          NavigableSet<String> wals = new TreeSet<>();
378          wals.add(walPath.getName());
379          walsByGroup.put(walPrefixAndPath.getKey(), wals);
380          // Abort RS and throw exception to make add peer failed
381          abortAndThrowIOExceptionWhenFail(
382            () -> this.queueStorage.addWAL(server.getServerName(), peerId, walPath.getName()));
383          src.enqueueLog(walPath);
384          LOG.trace("Enqueued {} to source {} during source creation.", walPath, src.getQueueId());
385        }
386      }
387    }
388    ReplicationPeerConfig peerConfig = peer.getPeerConfig();
389    if (peerConfig.isSyncReplication()) {
390      syncReplicationPeerMappingManager.add(peer.getId(), peerConfig);
391    }
392    src.startup();
393    return src;
394  }

396  /**
397   * <p>
398   * This is used when we transit a sync replication peer to {@link SyncReplicationState#STANDBY}.
399   * </p>
400   * <p>
401   * When transiting to {@link SyncReplicationState#STANDBY}, we can remove all the pending wal
402   * files for a replication peer as we do not need to replicate them any more. And this is
403   * necessary, otherwise when we transit back to {@link SyncReplicationState#DOWNGRADE_ACTIVE}
404   * later, the stale data will be replicated again and cause inconsistency.
405   * </p>
406   * <p>
407   * See HBASE-20426 for more details.
408   * </p>
409   * @param peerId the id of the sync replication peer
410   */
411  public void drainSources(String peerId) throws IOException, ReplicationException {
412    String terminateMessage = "Sync replication peer " + peerId +
413      " is transiting to STANDBY. Will close the previous replication source and open a new one";
414    ReplicationPeer peer = replicationPeers.getPeer(peerId);
415    assert peer.getPeerConfig().isSyncReplication();
416    ReplicationSourceInterface src = createSource(peerId, peer);
417    // synchronized here to avoid race with preLogRoll where we add new log to source and also
418    // walsById.
419    ReplicationSourceInterface toRemove;
420    Map<String, NavigableSet<String>> wals = new HashMap<>();
421    synchronized (latestPaths) {
422      toRemove = sources.put(peerId, src);
423      if (toRemove != null) {
424        LOG.info("Terminate replication source for " + toRemove.getPeerId());
425        toRemove.terminate(terminateMessage);
426        toRemove.getSourceMetrics().clear();
427      }
428      // Here we make a copy of all the remaining wal files and then delete them from the
429      // replication queue storage after releasing the lock. It is not safe to just remove the old
430      // map from walsById since later we may fail to delete them from the replication queue
431      // storage, and when we retry next time, we can not know the wal files that need to be deleted
432      // from the replication queue storage.
433      walsById.get(peerId).forEach((k, v) -> wals.put(k, new TreeSet<>(v)));
434    }
435    LOG.info("Startup replication source for " + src.getPeerId());
436    src.startup();
437    for (NavigableSet<String> walsByGroup : wals.values()) {
438      for (String wal : walsByGroup) {
439        queueStorage.removeWAL(server.getServerName(), peerId, wal);
440      }
441    }
442    synchronized (walsById) {
443      Map<String, NavigableSet<String>> oldWals = walsById.get(peerId);
444      wals.forEach((k, v) -> {
445        NavigableSet<String> walsByGroup = oldWals.get(k);
446        if (walsByGroup != null) {
447          walsByGroup.removeAll(v);
448        }
449      });
450    }
451    // synchronized on oldsources to avoid race with NodeFailoverWorker. Since NodeFailoverWorker is
452    // a background task, we will delete the file from replication queue storage under the lock to
453    // simplify the logic.
454    synchronized (this.oldsources) {
455      for (Iterator<ReplicationSourceInterface> iter = oldsources.iterator(); iter.hasNext();) {
456        ReplicationSourceInterface oldSource = iter.next();
457        if (oldSource.getPeerId().equals(peerId)) {
458          String queueId = oldSource.getQueueId();
459          oldSource.terminate(terminateMessage);
460          oldSource.getSourceMetrics().clear();
461          queueStorage.removeQueue(server.getServerName(), queueId);
462          walsByIdRecoveredQueues.remove(queueId);
463          iter.remove();
464        }
465      }
466    }
467  }

469  /**
470   * Close the previous replication sources of this peer id and open new sources to trigger the new
471   * replication state changes or new replication config changes. Here we don't need to change
472   * replication queue storage and only to enqueue all logs to the new replication source
473   * @param peerId the id of the replication peer
474   */
475  public void refreshSources(String peerId) throws IOException {
476    String terminateMessage = "Peer " + peerId +
477      " state or config changed. Will close the previous replication source and open a new one";
478    ReplicationPeer peer = replicationPeers.getPeer(peerId);
479    ReplicationSourceInterface src = createSource(peerId, peer);
480    // synchronized on latestPaths to avoid missing the new log
481    synchronized (this.latestPaths) {
482      ReplicationSourceInterface toRemove = this.sources.put(peerId, src);
483      if (toRemove != null) {
484        LOG.info("Terminate replication source for " + toRemove.getPeerId());
485        // Do not clear metrics
486        toRemove.terminate(terminateMessage, null, false);
487      }
488      for (NavigableSet<String> walsByGroup : walsById.get(peerId).values()) {
489        walsByGroup.forEach(wal -> src.enqueueLog(new Path(this.logDir, wal)));
490      }
491    }
492    LOG.info("Startup replication source for " + src.getPeerId());
493    src.startup();
494
495    List<ReplicationSourceInterface> toStartup = new ArrayList<>();
496    // synchronized on oldsources to avoid race with NodeFailoverWorker
497    synchronized (this.oldsources) {
498      List<String> previousQueueIds = new ArrayList<>();
499      for (ReplicationSourceInterface oldSource : this.oldsources) {
500        if (oldSource.getPeerId().equals(peerId)) {
501          previousQueueIds.add(oldSource.getQueueId());
502          oldSource.terminate(terminateMessage);
503          this.oldsources.remove(oldSource);
504        }
505      }
506      for (String queueId : previousQueueIds) {
507        ReplicationSourceInterface replicationSource = createSource(queueId, peer);
508        this.oldsources.add(replicationSource);
509        for (SortedSet<String> walsByGroup : walsByIdRecoveredQueues.get(queueId).values()) {
510          walsByGroup.forEach(wal -> src.enqueueLog(new Path(wal)));
511        }
512        toStartup.add(replicationSource);
513      }
514    }
515    for (ReplicationSourceInterface replicationSource : toStartup) {
516      replicationSource.startup();
517    }
518  }

520  /**
521   * Clear the metrics and related replication queue of the specified old source
522   * @param src source to clear
523   */
524  private boolean removeRecoveredSource(ReplicationSourceInterface src) {
525    if (!this.oldsources.remove(src)) {
526      return false;
527    }
528    LOG.info("Done with the recovered queue {}", src.getQueueId());
529    // Delete queue from storage and memory
530    deleteQueue(src.getQueueId());
531    this.walsByIdRecoveredQueues.remove(src.getQueueId());
532    return true;
533  }
534
535  void finishRecoveredSource(ReplicationSourceInterface src) {
536    synchronized (oldsources) {
537      if (!removeRecoveredSource(src)) {
538        return;
539      }
540    }
541    LOG.info("Finished recovering queue {} with the following stats: {}", src.getQueueId(),
542      src.getStats());
543  }

545  /**
546   * Clear the metrics and related replication queue of the specified old source
547   * @param src source to clear
548   */
549  void removeSource(ReplicationSourceInterface src) {
550    LOG.info("Done with the queue " + src.getQueueId());
551    this.sources.remove(src.getPeerId());
552    // Delete queue from storage and memory
553    deleteQueue(src.getQueueId());
554    this.walsById.remove(src.getQueueId());
555
556  }

571  /**
572   * Refresh replication source will terminate the old source first, then the source thread will be
573   * interrupted. Need to handle it instead of abort the region server.
574   */
575  private void interruptOrAbortWhenFail(ReplicationQueueOperation op) {
576    try {
577      op.exec();
578    } catch (ReplicationException e) {
579      if (e.getCause() != null && e.getCause() instanceof KeeperException.SystemErrorException
580          && e.getCause().getCause() != null && e.getCause()
581          .getCause() instanceof InterruptedException) {
582        throw new RuntimeException(
583            "Thread is interrupted, the replication source may be terminated");
584      }
585      server.abort("Failed to operate on replication queue", e);
586    }
587  }

590    try {
591      op.exec();
592    } catch (ReplicationException e) {
593      server.abort("Failed to operate on replication queue", e);
594    }
595  }

628  /**
629   * Cleans a log file and all older logs from replication queue. Called when we are sure that a log
630   * file is closed and has no more entries.
631   * @param log Path to the log
632   * @param inclusive whether we should also remove the given log file
633   * @param source the replication source
634   */
635  @VisibleForTesting
636  void cleanOldLogs(String log, boolean inclusive, ReplicationSourceInterface source) {
637    String logPrefix = AbstractFSWALProvider.getWALPrefixFromWALName(log);
638    if (source.isRecovered()) {
639      NavigableSet<String> wals = walsByIdRecoveredQueues.get(source.getQueueId()).get(logPrefix);
640      if (wals != null) {
641        NavigableSet<String> walsToRemove = wals.headSet(log, inclusive);
642        if (walsToRemove.isEmpty()) {
643          return;
644        }
645        cleanOldLogs(walsToRemove, source);
646        walsToRemove.clear();
647      }
648    } else {
649      NavigableSet<String> wals;
650      NavigableSet<String> walsToRemove;
651      // synchronized on walsById to avoid race with preLogRoll
652      synchronized (this.walsById) {
653        wals = walsById.get(source.getQueueId()).get(logPrefix);
654        if (wals == null) {
655          return;
656        }
657        walsToRemove = wals.headSet(log, inclusive);
658        if (walsToRemove.isEmpty()) {
659          return;
660        }
661        walsToRemove = new TreeSet<>(walsToRemove);
662      }
663      // cleanOldLogs may spend some time, especially for sync replication where we may want to
664      // remove remote wals as the remote cluster may have already been down, so we do it outside
665      // the lock to avoid block preLogRoll
666      cleanOldLogs(walsToRemove, source);
667      // now let's remove the files in the set
668      synchronized (this.walsById) {
669        wals.removeAll(walsToRemove);
670      }
671    }
672  }
673
674  private void removeRemoteWALs(String peerId, String remoteWALDir, Collection<String> wals)
675      throws IOException {
676    Path remoteWALDirForPeer = ReplicationUtils.getPeerRemoteWALDir(remoteWALDir, peerId);
677    FileSystem fs = ReplicationUtils.getRemoteWALFileSystem(conf, remoteWALDir);
678    for (String wal : wals) {
679      Path walFile = new Path(remoteWALDirForPeer, wal);
680      try {
681        if (!fs.delete(walFile, false) && fs.exists(walFile)) {
682          throw new IOException("Can not delete " + walFile);
683        }
684      } catch (FileNotFoundException e) {
685        // Just ignore since this means the file has already been deleted.
686        // The javadoc of the FileSystem.delete methods does not specify the behavior of deleting an
687        // inexistent file, so here we deal with both, i.e, check the return value of the
688        // FileSystem.delete, and also catch FNFE.
689        LOG.debug("The remote wal {} has already been deleted?", walFile, e);
690      }
691    }
692  }
693
694  private void cleanOldLogs(NavigableSet<String> wals, ReplicationSourceInterface source) {
695    LOG.debug("Removing {} logs in the list: {}", wals.size(), wals);
696    // The intention here is that, we want to delete the remote wal files ASAP as it may effect the
697    // failover time if you want to transit the remote cluster from S to A. And the infinite retry
698    // is not a problem, as if we can not contact with the remote HDFS cluster, then usually we can
699    // not contact with the HBase cluster either, so the replication will be blocked either.
700    if (source.isSyncReplication()) {
701      String peerId = source.getPeerId();
702      String remoteWALDir = source.getPeer().getPeerConfig().getRemoteWALDir();
703      // Filter out the wals need to be removed from the remote directory. Its name should be the
704      // special format, and also, the peer id in its name should match the peer id for the
705      // replication source.
706      List<String> remoteWals = wals.stream().filter(w -> SyncReplicationWALProvider
707        .getSyncReplicationPeerIdFromWALName(w).map(peerId::equals).orElse(false))
708        .collect(Collectors.toList());
709      LOG.debug("Removing {} logs from remote dir {} in the list: {}", remoteWals.size(),
710        remoteWALDir, remoteWals);
711      if (!remoteWals.isEmpty()) {
712        for (int sleepMultiplier = 0;;) {
713          try {
714            removeRemoteWALs(peerId, remoteWALDir, remoteWals);
715            break;
716          } catch (IOException e) {
717            LOG.warn("Failed to delete remote wals from remote dir {} for peer {}", remoteWALDir,
718              peerId);
719          }
720          if (!source.isSourceActive()) {
721            // skip the following operations
722            return;
723          }
724          if (ReplicationUtils.sleepForRetries("Failed to delete remote wals", sleepForRetries,
725            sleepMultiplier, maxRetriesMultiplier)) {
726            sleepMultiplier++;
727          }
728        }
729      }
730    }
731    String queueId = source.getQueueId();
732    for (String wal : wals) {
733      interruptOrAbortWhenFail(
734        () -> this.queueStorage.removeWAL(server.getServerName(), queueId, wal));
735    }
736  }

738  // public because of we call it in TestReplicationEmptyWALRecovery
739  @VisibleForTesting
740  public void preLogRoll(Path newLog) throws IOException {
741    String logName = newLog.getName();
742    String logPrefix = AbstractFSWALProvider.getWALPrefixFromWALName(logName);
743    // synchronized on latestPaths to avoid the new open source miss the new log
744    synchronized (this.latestPaths) {
745      // Add log to queue storage
746      for (ReplicationSourceInterface source : this.sources.values()) {
747        // If record log to queue storage failed, abort RS and throw exception to make log roll
748        // failed
749        abortAndThrowIOExceptionWhenFail(
750          () -> this.queueStorage.addWAL(server.getServerName(), source.getQueueId(), logName));
751      }
752
753      // synchronized on walsById to avoid race with cleanOldLogs
754      synchronized (this.walsById) {
755        // Update walsById map
756        for (Map.Entry<String, Map<String, NavigableSet<String>>> entry : this.walsById
757          .entrySet()) {
758          String peerId = entry.getKey();
759          Map<String, NavigableSet<String>> walsByPrefix = entry.getValue();
760          boolean existingPrefix = false;
761          for (Map.Entry<String, NavigableSet<String>> walsEntry : walsByPrefix.entrySet()) {
762            SortedSet<String> wals = walsEntry.getValue();
763            if (this.sources.isEmpty()) {
764              // If there's no slaves, don't need to keep the old wals since
765              // we only consider the last one when a new slave comes in
766              wals.clear();
767            }
768            if (logPrefix.equals(walsEntry.getKey())) {
769              wals.add(logName);
770              existingPrefix = true;
771            }
772          }
773          if (!existingPrefix) {
774            // The new log belongs to a new group, add it into this peer
775            LOG.debug("Start tracking logs for wal group {} for peer {}", logPrefix, peerId);
776            NavigableSet<String> wals = new TreeSet<>();
777            wals.add(logName);
778            walsByPrefix.put(logPrefix, wals);
779          }
780        }
781      }
782
783      // Add to latestPaths
784      latestPaths.put(logPrefix, newLog);
785    }
786  }

804  /**
805   * Transfer all the queues of the specified to this region server. First it tries to grab a lock
806   * and if it works it will move the old queues and finally will delete the old queues.
807   * <p>
808   * It creates one old source for any type of source of the old rs.
809   */
810  private void transferQueues(ServerName deadRS) {
811    if (server.getServerName().equals(deadRS)) {
812      // it's just us, give up
813      return;
814    }
815    NodeFailoverWorker transfer = new NodeFailoverWorker(deadRS);
816    try {
817      this.executor.execute(transfer);
818    } catch (RejectedExecutionException ex) {
819      CompatibilitySingletonFactory.getInstance(MetricsReplicationSourceFactory.class)
820          .getGlobalSource().incrFailedRecoveryQueue();
821      LOG.info("Cancelling the transfer of " + deadRS + " because of " + ex.getMessage());
822    }
823  }

825  /**
826   * Class responsible to setup new ReplicationSources to take care of the queues from dead region
827   * servers.
828   */
829  class NodeFailoverWorker extends Thread {
830
831    private final ServerName deadRS;
832    // After claim the queues from dead region server, the NodeFailoverWorker will skip to start
833    // the RecoveredReplicationSource if the peer has been removed. but there's possible that
834    // remove a peer with peerId = 2 and add a peer with peerId = 2 again during the
835    // NodeFailoverWorker. So we need a deep copied <peerId, peer> map to decide whether we
836    // should start the RecoveredReplicationSource. If the latest peer is not the old peer when
837    // NodeFailoverWorker begin, we should skip to start the RecoveredReplicationSource, Otherwise
838    // the rs will abort (See HBASE-20475).
839    private final Map<String, ReplicationPeerImpl> peersSnapshot;
840
841    @VisibleForTesting
842    public NodeFailoverWorker(ServerName deadRS) {
843      super("Failover-for-" + deadRS);
844      this.deadRS = deadRS;
845      peersSnapshot = new HashMap<>(replicationPeers.getPeerCache());
846    }
847
848    private boolean isOldPeer(String peerId, ReplicationPeerImpl newPeerRef) {
849      ReplicationPeerImpl oldPeerRef = peersSnapshot.get(peerId);
850      return oldPeerRef != null && oldPeerRef == newPeerRef;
851    }
852
853    @Override
854    public void run() {
855      // Wait a bit before transferring the queues, we may be shutting down.
856      // This sleep may not be enough in some cases.
857      try {
858        Thread.sleep(sleepBeforeFailover +
859          (long) (ThreadLocalRandom.current().nextFloat() * sleepBeforeFailover));
860      } catch (InterruptedException e) {
861        LOG.warn("Interrupted while waiting before transferring a queue.");
862        Thread.currentThread().interrupt();
863      }
864      // We try to lock that rs' queue directory
865      if (server.isStopped()) {
866        LOG.info("Not transferring queue since we are shutting down");
867        return;
868      }
869      Map<String, Set<String>> newQueues = new HashMap<>();
870      try {
871        List<String> queues = queueStorage.getAllQueues(deadRS);
872        while (!queues.isEmpty()) {
873          Pair<String, SortedSet<String>> peer = queueStorage.claimQueue(deadRS,
874            queues.get(ThreadLocalRandom.current().nextInt(queues.size())), server.getServerName());
875          long sleep = sleepBeforeFailover / 2;
876          if (!peer.getSecond().isEmpty()) {
877            newQueues.put(peer.getFirst(), peer.getSecond());
878            sleep = sleepBeforeFailover;
879          }
880          try {
881            Thread.sleep(sleep);
882          } catch (InterruptedException e) {
883            LOG.warn("Interrupted while waiting before transferring a queue.");
884            Thread.currentThread().interrupt();
885          }
886          queues = queueStorage.getAllQueues(deadRS);
887        }
888        if (queues.isEmpty()) {
889          queueStorage.removeReplicatorIfQueueIsEmpty(deadRS);
890        }
891      } catch (ReplicationException e) {
892        LOG.error(String.format("ReplicationException: cannot claim dead region (%s)'s " +
893            "replication queue. Znode : (%s)" +
894            " Possible solution: check if znode size exceeds jute.maxBuffer value. " +
895            " If so, increase it for both client and server side." + e),  deadRS,
896            queueStorage.getRsNode(deadRS));
897        server.abort("Failed to claim queue from dead regionserver.", e);
898        return;
899      }
900      // Copying over the failed queue is completed.
901      if (newQueues.isEmpty()) {
902        // We either didn't get the lock or the failed region server didn't have any outstanding
903        // WALs to replicate, so we are done.
904        return;
905      }
906
907      for (Map.Entry<String, Set<String>> entry : newQueues.entrySet()) {
908        String queueId = entry.getKey();
909        Set<String> walsSet = entry.getValue();
910        try {
911          // there is not an actual peer defined corresponding to peerId for the failover.
912          ReplicationQueueInfo replicationQueueInfo = new ReplicationQueueInfo(queueId);
913          String actualPeerId = replicationQueueInfo.getPeerId();
914
915          ReplicationPeerImpl peer = replicationPeers.getPeer(actualPeerId);
916          if (peer == null || !isOldPeer(actualPeerId, peer)) {
917            LOG.warn("Skipping failover for peer {} of node {}, peer is null", actualPeerId,
918              deadRS);
919            abortWhenFail(() -> queueStorage.removeQueue(server.getServerName(), queueId));
920            continue;
921          }
922          if (server instanceof ReplicationSyncUp.DummyServer
923              && peer.getPeerState().equals(PeerState.DISABLED)) {
924            LOG.warn("Peer {} is disabled. ReplicationSyncUp tool will skip "
925                + "replicating data to this peer.",
926              actualPeerId);
927            continue;
928          }
929
930          ReplicationSourceInterface src = createSource(queueId, peer);
931          // synchronized on oldsources to avoid adding recovered source for the to-be-removed peer
932          synchronized (oldsources) {
933            peer = replicationPeers.getPeer(src.getPeerId());
934            if (peer == null || !isOldPeer(src.getPeerId(), peer)) {
935              src.terminate("Recovered queue doesn't belong to any current peer");
936              deleteQueue(queueId);
937              continue;
938            }
939            // Do not setup recovered queue if a sync replication peer is in STANDBY state, or is
940            // transiting to STANDBY state. The only exception is we are in STANDBY state and
941            // transiting to DA, under this state we will replay the remote WAL and they need to be
942            // replicated back.
943            if (peer.getPeerConfig().isSyncReplication()) {
944              Pair<SyncReplicationState, SyncReplicationState> stateAndNewState =
945                peer.getSyncReplicationStateAndNewState();
946              if ((stateAndNewState.getFirst().equals(SyncReplicationState.STANDBY) &&
947                stateAndNewState.getSecond().equals(SyncReplicationState.NONE)) ||
948                stateAndNewState.getSecond().equals(SyncReplicationState.STANDBY)) {
949                src.terminate("Sync replication peer is in STANDBY state");
950                deleteQueue(queueId);
951                continue;
952              }
953            }
954            // track sources in walsByIdRecoveredQueues
955            Map<String, NavigableSet<String>> walsByGroup = new HashMap<>();
956            walsByIdRecoveredQueues.put(queueId, walsByGroup);
957            for (String wal : walsSet) {
958              String walPrefix = AbstractFSWALProvider.getWALPrefixFromWALName(wal);
959              NavigableSet<String> wals = walsByGroup.get(walPrefix);
960              if (wals == null) {
961                wals = new TreeSet<>();
962                walsByGroup.put(walPrefix, wals);
963              }
964              wals.add(wal);
965            }
966            oldsources.add(src);
967            LOG.trace("Added source for recovered queue: " + src.getQueueId());
968            for (String wal : walsSet) {
969              LOG.trace("Enqueueing log from recovered queue for source: " + src.getQueueId());
970              src.enqueueLog(new Path(oldLogDir, wal));
971            }
972            src.startup();
973          }
974        } catch (IOException e) {
975          // TODO manage it
976          LOG.error("Failed creating a source", e);
977        }
978      }
979    }
980  }
