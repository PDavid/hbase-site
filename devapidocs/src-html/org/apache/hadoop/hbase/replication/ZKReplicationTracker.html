018package org.apache.hadoop.hbase.replication;
019
020import java.io.IOException;
021import java.util.HashSet;
022import java.util.List;
023import java.util.Set;
024import java.util.stream.Collectors;
025import org.apache.hadoop.hbase.Abortable;
026import org.apache.hadoop.hbase.ServerName;
027import org.apache.hadoop.hbase.Stoppable;
028import org.apache.hadoop.hbase.zookeeper.ZKListener;
029import org.apache.hadoop.hbase.zookeeper.ZKUtil;
030import org.apache.hadoop.hbase.zookeeper.ZKWatcher;
031import org.apache.yetus.audience.InterfaceAudience;
032import org.apache.zookeeper.KeeperException;
033
034import org.apache.hbase.thirdparty.com.google.common.base.Splitter;
035
036/**
037 * This class is a ZooKeeper implementation of the ReplicationTracker interface. This class is
038 * responsible for handling replication events that are defined in the ReplicationListener
039 * interface.
040 */
041@InterfaceAudience.Private
042class ZKReplicationTracker extends ReplicationTrackerBase {
043
044  // Zookeeper
045  private final ZKWatcher zookeeper;
046  // Server to abort.
047  private final Abortable abortable;
048  // All about stopping
049  private final Stoppable stopper;
050  // List of all the other region servers in this cluster
051  private final Set<ServerName> regionServers = new HashSet<>();
052
053  ZKReplicationTracker(ReplicationTrackerParams params) {
054    this.zookeeper = params.zookeeper();
055    this.abortable = params.abortable();
056    this.stopper = params.stopptable();
057    this.zookeeper.registerListener(new OtherRegionServerWatcher(this.zookeeper));
058  }
059
060  /**
061   * Watcher used to be notified of the other region server's death in the local cluster. It
062   * initiates the process to transfer the queues if it is able to grab the lock.
063   */
064  public class OtherRegionServerWatcher extends ZKListener {
065
066    /**
067     * Construct a ZooKeeper event listener.
068     */
069    public OtherRegionServerWatcher(ZKWatcher watcher) {
070      super(watcher);
071    }
072
073    /**
074     * Called when a new node has been created.
075     * @param path full path of the new node
076     */
077    @Override
078    public void nodeCreated(String path) {
079      if (stopper.isStopped()) {
080        return;
081      }
082      refreshListIfRightPath(path);
083    }
084
085    /**
086     * Called when a node has been deleted
087     * @param path full path of the deleted node
088     */
089    @Override
090    public void nodeDeleted(String path) {
091      if (stopper.isStopped()) {
092        return;
093      }
094      if (!refreshListIfRightPath(path)) {
095        return;
096      }
097      notifyListeners(ServerName.valueOf(getZNodeName(path)));
098    }
099
100    /**
101     * Called when an existing node has a child node added or removed.
102     * @param path full path of the node whose children have changed
103     */
104    @Override
105    public void nodeChildrenChanged(String path) {
106      if (stopper.isStopped()) {
107        return;
108      }
109      refreshListIfRightPath(path);
110    }
111
112    private boolean refreshListIfRightPath(String path) {
113      if (!path.startsWith(this.watcher.getZNodePaths().rsZNode)) {
114        return false;
115      }
116      return refreshRegionServerList();
117    }
118  }
119
120  /**
121   * Extracts the znode name of a peer cluster from a ZK path
122   * @param fullPath Path to extract the id from
123   * @return the id or an empty string if path is invalid
124   */
125  private String getZNodeName(String fullPath) {
126    List<String> parts = Splitter.on('/').splitToList(fullPath);
127    return parts.size() > 0 ? parts.get(parts.size() - 1) : "";
128  }
129
130  /**
131   * Reads the list of region servers from ZK and atomically clears our local view of it and
132   * replaces it with the updated list.
133   * @return true if the local list of the other region servers was updated with the ZK data (even
134   *         if it was empty), false if the data was missing in ZK
135   */
136  private boolean refreshRegionServerList() {
137    Set<ServerName> newRsList = getRegisteredRegionServers();
138    if (newRsList == null) {
139      return false;
140    } else {
141      synchronized (regionServers) {
142        regionServers.clear();
143        regionServers.addAll(newRsList);
144      }
145    }
146    return true;
147  }
148
149  /**
150   * Get a list of all the other region servers in this cluster and set a watch
151   * @return a list of server nanes
152   */
153  private Set<ServerName> getRegisteredRegionServers() {
154    List<String> result = null;
155    try {
156      result =
157        ZKUtil.listChildrenAndWatchThem(this.zookeeper, this.zookeeper.getZNodePaths().rsZNode);
158    } catch (KeeperException e) {
159      this.abortable.abort("Get list of registered region servers", e);
160    }
161    return result == null ? null :
162      result.stream().map(ServerName::parseServerName).collect(Collectors.toSet());
163  }
164
165  @Override
166  protected Set<ServerName> internalLoadLiveRegionServersAndInitializeListeners()
167    throws IOException {
168    if (!refreshRegionServerList()) {
169      throw new IOException("failed to refresh region server list");
170    }
171    synchronized (regionServers) {
172      return new HashSet<>(regionServers);
173    }
174  }
175}
