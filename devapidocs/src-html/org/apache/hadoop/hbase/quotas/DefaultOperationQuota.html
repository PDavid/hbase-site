018package org.apache.hadoop.hbase.quotas;
019
020import java.util.Arrays;
021import java.util.List;
022import org.apache.hadoop.conf.Configuration;
023import org.apache.hadoop.hbase.client.Mutation;
024import org.apache.hadoop.hbase.client.Result;
025import org.apache.hadoop.hbase.ipc.RpcCall;
026import org.apache.hadoop.hbase.ipc.RpcServer;
027import org.apache.yetus.audience.InterfaceAudience;
028import org.apache.yetus.audience.InterfaceStability;
029
030@InterfaceAudience.Private
031@InterfaceStability.Evolving
032public class DefaultOperationQuota implements OperationQuota {
033
034  protected final List<QuotaLimiter> limiters;
035  private final long writeCapacityUnit;
036  private final long readCapacityUnit;
037
038  // the available read/write quota size in bytes
039  protected long readAvailable = 0;
040  // estimated quota
041  protected long writeConsumed = 0;
042  protected long readConsumed = 0;
043  protected long writeCapacityUnitConsumed = 0;
044  protected long readCapacityUnitConsumed = 0;
045  // real consumed quota
046  private final long[] operationSize;
047  // difference between estimated quota and real consumed quota used in close method
048  // to adjust quota amount. Also used by ExceedOperationQuota which is a subclass
049  // of DefaultOperationQuota
050  protected long writeDiff = 0;
051  protected long readDiff = 0;
052  protected long writeCapacityUnitDiff = 0;
053  protected long readCapacityUnitDiff = 0;
054  private boolean useResultSizeBytes;
055  private long blockSizeBytes;
056
057  public DefaultOperationQuota(final Configuration conf, final int blockSizeBytes,
058    final QuotaLimiter... limiters) {
059    this(conf, Arrays.asList(limiters));
060    this.useResultSizeBytes =
061      conf.getBoolean(OperationQuota.USE_RESULT_SIZE_BYTES, USE_RESULT_SIZE_BYTES_DEFAULT);
062    this.blockSizeBytes = blockSizeBytes;
063  }
064
065  /**
066   * NOTE: The order matters. It should be something like [user, table, namespace, global]
067   */
068  public DefaultOperationQuota(final Configuration conf, final List<QuotaLimiter> limiters) {
069    this.writeCapacityUnit =
070      conf.getLong(QuotaUtil.WRITE_CAPACITY_UNIT_CONF_KEY, QuotaUtil.DEFAULT_WRITE_CAPACITY_UNIT);
071    this.readCapacityUnit =
072      conf.getLong(QuotaUtil.READ_CAPACITY_UNIT_CONF_KEY, QuotaUtil.DEFAULT_READ_CAPACITY_UNIT);
073    this.limiters = limiters;
074    int size = OperationType.values().length;
075    operationSize = new long[size];
076
077    for (int i = 0; i < size; ++i) {
078      operationSize[i] = 0;
079    }
080  }
081
082  @Override
083  public void checkQuota(int numWrites, int numReads, int numScans) throws RpcThrottlingException {
084    updateEstimateConsumeQuota(numWrites, numReads, numScans);
085
086    readAvailable = Long.MAX_VALUE;
087    for (final QuotaLimiter limiter : limiters) {
088      if (limiter.isBypass()) continue;
089
090      limiter.checkQuota(numWrites, writeConsumed, numReads + numScans, readConsumed,
091        writeCapacityUnitConsumed, readCapacityUnitConsumed);
092      readAvailable = Math.min(readAvailable, limiter.getReadAvailable());
093    }
094
095    for (final QuotaLimiter limiter : limiters) {
096      limiter.grabQuota(numWrites, writeConsumed, numReads + numScans, readConsumed,
097        writeCapacityUnitConsumed, readCapacityUnitConsumed);
098    }
099  }
100
101  @Override
102  public void close() {
103    // Adjust the quota consumed for the specified operation
104    writeDiff = operationSize[OperationType.MUTATE.ordinal()] - writeConsumed;
105
106    long resultSize =
107      operationSize[OperationType.GET.ordinal()] + operationSize[OperationType.SCAN.ordinal()];
108    if (useResultSizeBytes) {
109      readDiff = resultSize - readConsumed;
110    } else {
111      long blockBytesScanned =
112        RpcServer.getCurrentCall().map(RpcCall::getBlockBytesScanned).orElse(0L);
113      readDiff = Math.max(blockBytesScanned, resultSize) - readConsumed;
114    }
115
116    writeCapacityUnitDiff =
117      calculateWriteCapacityUnitDiff(operationSize[OperationType.MUTATE.ordinal()], writeConsumed);
118    readCapacityUnitDiff = calculateReadCapacityUnitDiff(
119      operationSize[OperationType.GET.ordinal()] + operationSize[OperationType.SCAN.ordinal()],
120      readConsumed);
121
122    for (final QuotaLimiter limiter : limiters) {
123      if (writeDiff != 0) {
124        limiter.consumeWrite(writeDiff, writeCapacityUnitDiff);
125      }
126      if (readDiff != 0) {
127        limiter.consumeRead(readDiff, readCapacityUnitDiff);
128      }
129    }
130  }
131
132  @Override
133  public long getReadAvailable() {
134    return readAvailable;
135  }
136
137  @Override
138  public long getReadConsumed() {
139    return readConsumed;
140  }
141
142  @Override
143  public void addGetResult(final Result result) {
144    operationSize[OperationType.GET.ordinal()] += QuotaUtil.calculateResultSize(result);
145  }
146
147  @Override
148  public void addScanResult(final List<Result> results) {
149    operationSize[OperationType.SCAN.ordinal()] += QuotaUtil.calculateResultSize(results);
150  }
151
152  @Override
153  public void addMutation(final Mutation mutation) {
154    operationSize[OperationType.MUTATE.ordinal()] += QuotaUtil.calculateMutationSize(mutation);
155  }
156
157  /**
158   * Update estimate quota(read/write size/capacityUnits) which will be consumed
159   * @param numWrites the number of write requests
160   * @param numReads  the number of read requests
161   * @param numScans  the number of scan requests
162   */
163  protected void updateEstimateConsumeQuota(int numWrites, int numReads, int numScans) {
164    writeConsumed = estimateConsume(OperationType.MUTATE, numWrites, 100);
165
166    if (useResultSizeBytes) {
167      readConsumed = estimateConsume(OperationType.GET, numReads, 100);
168      readConsumed += estimateConsume(OperationType.SCAN, numScans, 1000);
169    } else {
170      // assume 1 block required for reads. this is probably a low estimate, which is okay
171      readConsumed = numReads > 0 ? blockSizeBytes : 0;
172      readConsumed += numScans > 0 ? blockSizeBytes : 0;
173    }
174
175    writeCapacityUnitConsumed = calculateWriteCapacityUnit(writeConsumed);
176    readCapacityUnitConsumed = calculateReadCapacityUnit(readConsumed);
177  }
178
179  private long estimateConsume(final OperationType type, int numReqs, long avgSize) {
180    if (numReqs > 0) {
181      return avgSize * numReqs;
182    }
183    return 0;
184  }
185
186  private long calculateWriteCapacityUnit(final long size) {
187    return (long) Math.ceil(size * 1.0 / this.writeCapacityUnit);
188  }
189
190  private long calculateReadCapacityUnit(final long size) {
191    return (long) Math.ceil(size * 1.0 / this.readCapacityUnit);
192  }
193
194  private long calculateWriteCapacityUnitDiff(final long actualSize, final long estimateSize) {
195    return calculateWriteCapacityUnit(actualSize) - calculateWriteCapacityUnit(estimateSize);
196  }
197
198  private long calculateReadCapacityUnitDiff(final long actualSize, final long estimateSize) {
199    return calculateReadCapacityUnit(actualSize) - calculateReadCapacityUnit(estimateSize);
200  }
201}
