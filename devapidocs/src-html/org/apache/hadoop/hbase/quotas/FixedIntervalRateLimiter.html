018package org.apache.hadoop.hbase.quotas;
019
020import org.apache.hadoop.hbase.util.EnvironmentEdgeManager;
021import org.apache.yetus.audience.InterfaceAudience;
022import org.apache.yetus.audience.InterfaceStability;
023
024/**
025 * With this limiter resources will be refilled only after a fixed interval of time.
026 */
027@InterfaceAudience.Private
028@InterfaceStability.Evolving
029public class FixedIntervalRateLimiter extends RateLimiter {
030  private long nextRefillTime = -1L;
031
032  @Override
033  public long refill(long limit) {
034    final long now = EnvironmentEdgeManager.currentTime();
035    if (now < nextRefillTime) {
036      return 0;
037    }
038    nextRefillTime = now + super.getTimeUnitInMillis();
039    return limit;
040  }
041
042  @Override
043  public long getWaitInterval(long limit, long available, long amount) {
044    if (nextRefillTime == -1) {
045      return 0;
046    }
047    final long now = EnvironmentEdgeManager.currentTime();
048    final long refillTime = nextRefillTime;
049    long diff = amount - available;
050    // We will add limit at next interval. If diff is less than that limit, the wait interval
051    // is just time between now and then.
052    long nextRefillInterval = refillTime - now;
053    if (diff <= limit) {
054      return nextRefillInterval;
055    }
056
057    // Otherwise, we need to figure out how many refills are needed.
058    // There will be one at nextRefillInterval, and then some number of extra refills.
059    // Division will round down if not even, so we can just add that to our next interval
060    long extraRefillsNecessary = diff / limit;
061    // If it's even, subtract one since that will be covered by nextRefillInterval
062    if (diff % limit == 0) {
063      extraRefillsNecessary--;
064    }
065    return nextRefillInterval + (extraRefillsNecessary * super.getTimeUnitInMillis());
066  }
067
068  // This method is for strictly testing purpose only
069  @Override
070  public void setNextRefillTime(long nextRefillTime) {
071    this.nextRefillTime = nextRefillTime;
072  }
073
074  @Override
075  public long getNextRefillTime() {
076    return this.nextRefillTime;
077  }
078}
